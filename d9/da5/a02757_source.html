183 WmipGetFreeBuffer(
184  IN PWMI_LOGGER_CONTEXT LoggerContext
185  )
186 //
187 // This routine works at any IRQL
188 //
189 {
190  PWMI_BUFFER_HEADER Buffer;
191  PSLIST_ENTRY Entry;
192  if (LoggerContext->SwitchingInProgress == 0) {
193  //
194  // Not in the middle of switching.
195  //
196 
197  Entry = InterlockedPopEntrySList(&LoggerContext->FreeList);
198 
199  if (Entry != NULL) {
200  Buffer = CONTAINING_RECORD (Entry,
201  WMI_BUFFER_HEADER,
202  SlistEntry);
203 
204  //
205  // Reset the buffer.
206  // For circular persist mode, we want to write the buffers as
207  // RunDown buffers so that post processing would work properly.
208  //
209 
210  if (LoggerContext->RequestFlag & REQUEST_FLAG_CIRCULAR_PERSIST) {
211  WmipResetBufferHeader( LoggerContext, Buffer, WMI_BUFFER_TYPE_RUNDOWN);
212  }
213  else {
214  WmipResetBufferHeader( LoggerContext, Buffer, WMI_BUFFER_TYPE_GENERIC);
215  }
216 
217  //
218  // Maintain some Wmi logger context buffer counts
219  //
220  InterlockedDecrement((PLONG) &LoggerContext->BuffersAvailable);
221  InterlockedIncrement((PLONG) &LoggerContext->BuffersInUse);
222 
223  TraceDebug((2, "WmipGetFreeBuffer: %2d, %p, Free: %d, InUse: %d, Dirty: %d, Total: %d\n",
224  LoggerContext->LoggerId,
225  Buffer,
226  LoggerContext->BuffersAvailable,
227  LoggerContext->BuffersInUse,
228  LoggerContext->BuffersDirty,
229  LoggerContext->NumberOfBuffers));
230 
231  return Buffer;
232  } else {
233  if (LoggerContext->LoggerMode & EVENT_TRACE_BUFFERING_MODE) {
234  //
235  // If we are in BUFFERING Mode, put all buffers from
236  // Flushlist into FreeList.
237  //
238 
239  if (InterlockedIncrement((PLONG) &LoggerContext->SwitchingInProgress) == 1) {
240  while (Entry = InterlockedPopEntrySList(&LoggerContext->FlushList)) {
241  Buffer = CONTAINING_RECORD (Entry,
242  WMI_BUFFER_HEADER,
243  SlistEntry);
244 
245  WmipPushFreeBuffer (LoggerContext, Buffer);
246  }
247  }
248  InterlockedDecrement((PLONG) &LoggerContext->SwitchingInProgress);
249  }
250  return NULL;
251  }
252  } else {
253  return NULL;
254  }

258 ULONG
259 WmipAllocateFreeBuffers(
260  IN PWMI_LOGGER_CONTEXT LoggerContext,
261  IN ULONG NumberOfBuffers
262  )
263 
264 /*++
265 
266 Routine Description:
267 
268  This routine allocate addition buffers into the free buffer list.
269  Logger can allocate more buffer to handle bursty logging behavior.
270  This routine can be called by multiple places and counters must be
271  manipulated using interlocked operations.
272 
273 Arguments:
274 
275  LoggerContext - Logger Context
276  NumberOfBuffers - Number of buffers to be allocated.
277 
278 Return Value:
279 
280  The total number of buffers actually allocated. When it is fewer than the requested number:
281  If this is called when trace is turned on, we fail to turn on trace.
282  If this is called by walker thread to get more buffer, it is OK.
283 
284 Environment:
285 
286  Kernel mode.
287 
288 --*/
289 {
290  ULONG i;
291  PWMI_BUFFER_HEADER Buffer;
292  ULONG TotalBuffers;
293 
294  for (i=0; i<NumberOfBuffers; i++) {
295  //
296  // Multiple threads can ask for more buffers, make sure
297  // we do not go over the maximum.
298  //
299  TotalBuffers = InterlockedIncrement(&LoggerContext->NumberOfBuffers);
300  if (TotalBuffers <= LoggerContext->MaximumBuffers) {
301 
302  Buffer = (PWMI_BUFFER_HEADER)
303  ExAllocatePoolWithTag(LoggerContext->PoolType,
304  LoggerContext->BufferSize,
305  TRACEPOOLTAG);
306  if (Buffer != NULL) {
307 
308  TraceDebug((3,
309  "WmipAllocateFreeBuffers: Allocated buffer size %d type %d\n",
310  LoggerContext->BufferSize, LoggerContext->PoolType));
311  InterlockedIncrement(&LoggerContext->BuffersAvailable);
312  //
313  // Initialize newly created buffer
314  //
315  RtlZeroMemory(Buffer, sizeof(WMI_BUFFER_HEADER));
316  Buffer->CurrentOffset = sizeof(WMI_BUFFER_HEADER);
317  KeQuerySystemTime(&Buffer->TimeStamp);
318  Buffer->State.Free = 1;
319 
320  //
321  // Insert it into the free List
322  //
323  InterlockedPushEntrySList(&LoggerContext->FreeList,
324  (PSLIST_ENTRY) &Buffer->SlistEntry);
325 
326  InterlockedPushEntrySList(&LoggerContext->GlobalList,
327  (PSLIST_ENTRY) &Buffer->GlobalEntry);
328  } else {
329  //
330  // Allocation failed, decrement the NumberOfBuffers
331  // we increment earlier.
332  //
333  InterlockedDecrement(&LoggerContext->NumberOfBuffers);
334  break;
335  }
336  } else {
337  //
338  // Maximum is reached, decrement the NumberOfBuffers
339  // we increment earlier.
340  //
341  InterlockedDecrement(&LoggerContext->NumberOfBuffers);
342  break;
343  }
344  }
345 
346  TraceDebug((2, "WmipAllocateFreeBuffers %3d (%3d): Free: %d, InUse: %d, Dirty: %d, Total: %d\n",
347  NumberOfBuffers,
348  i,
349  LoggerContext->BuffersAvailable,
350  LoggerContext->BuffersInUse,
351  LoggerContext->BuffersDirty,
352  LoggerContext->NumberOfBuffers));
353 
354  return i;
355 }

531 WmipReserveTraceBuffer(
532  IN PWMI_LOGGER_CONTEXT LoggerContext,
533  IN ULONG RequiredSize,
534  OUT PWMI_BUFFER_HEADER *BufferResource,
535  OUT PLARGE_INTEGER TimeStamp
536  )
537 /*++
538 
539 Routine Description:
540  This function is the main logging function that reserves spaces for any events.
541  The algorithm is as follows:
542 
543  Every time a space is needed, we InterlockedExchangeAdd CurrentOffset.
544  A local variable Offset is used to track the initial value when
545  InterlockedExchangeAdd is taken. If there is enough space for this
546  event (i.e., (Offset + RequiredSize) <= BufferSize), then we have successfully
547  reserved the space.
548 
549  If there is not enough space left on this buffer, we will call WmipSwitchBuffer
550  for a new buffer. In this case, CurrentOffset should be larger than the buffersize.
551  Since other thread can still be trying to reserve space using their buffer, we
552  saved the offset on SavedOffset the the logger thread knows the real offset to be
553  written to disk.
554 
555  Note that, since the CurrentOffset if growing monotonically, only one thread is
556  advancing the CurrentOffset from below BufferSize to beyond BufferSize.
557  It is this thread's responsibility to set the SavedOffset properly.
558 
559 Arguments:
560 
561  LoggerContext - Logger context from current logging session.
562 
563  RequiredSize - The space needed for logging the data.
564 
565  Buffer - Pointer to a buffer header
566 
567  TimeStamp - TimeStamp of the event
568 
569 Return Value:
570 
571  The status of running the buffer manager
572 
573 Environment:
574 
575  Kernel mode. This routine should work at any IRQL.
576 
577 --*/
578 {
579  PVOID ReservedSpace;
580  PWMI_BUFFER_HEADER Buffer = NULL;
581  ULONG Offset;
582 
583  //
584  // Mark it volatile to work around compiler bug.
585  //
586  volatile ULONG Processor;
587  NTSTATUS Status;
588 
589  if (!WmipIsValidLogger(LoggerContext)) {
590  return NULL;
591  }
592  if (!LoggerContext->CollectionOn) {
593  return NULL;
594  }
595 
596  *BufferResource = NULL;
597 
598  RequiredSize = (ULONG) ALIGN_TO_POWER2(RequiredSize, WmiTraceAlignment);
599 
600  if (RequiredSize > LoggerContext->BufferSize - sizeof (WMI_BUFFER_HEADER)) {
601  goto LostEvent;
602  }
603 
604  //
605  // Get processor number again here due to possible context switch
606  //
607  Processor = KeGetCurrentProcessorNumber();
608 
609  //
610  // Get the processor specific buffer pool
611  //
612  Buffer = LoggerContext->ProcessorBuffers[Processor];
613 
614  if (Buffer == NULL) {
615  //
616  // Nothing in per process list, ask to get a new buffer
617  //
618  Status = WmipSwitchBuffer(LoggerContext,
619  &Buffer,
620  &LoggerContext->ProcessorBuffers[Processor],
621  Processor);
622 
623  if (!NT_SUCCESS(Status)) {
624  //
625  // Nothing available
626  //
627  goto LostEvent;
628  }
629 
630  ASSERT(Buffer != NULL);
631  }
632 
633 TryFindSpace:
634 
635  //
636  // Increment refcount to buffer first to prevent it from going away
637  //
638  InterlockedIncrement(&Buffer->ReferenceCount);
639 
640  //
641  // Check if there is enough space in this buffer.
642  //
643  Offset = (ULONG) InterlockedExchangeAdd(
644  (PLONG) &Buffer->CurrentOffset, RequiredSize);
645 
646  //
647  // If many threads concurrently hit this line, the Offset+RequiredSize
648  // might arithmetically overflow. Therefore we need to add an extra
649  // check to make sure the Offset is in range.
650  //
651 
652  if ( (Offset+RequiredSize <= LoggerContext->BufferSize) &&
653  (Offset < LoggerContext->BufferSize) ) {
654 
655  //
656  // Successfully reserved the space
657  // Get the timestamp of the event
658  //
659  if (TimeStamp) {
660  TimeStamp->QuadPart = (*LoggerContext->GetCpuClock)();
661  }
662 
663  //
664  // Set up the space pointer
665  //
666  ReservedSpace = (PVOID) (Offset + (char*)Buffer);
667 
668  if (LoggerContext->SequencePtr) {
669  *((PULONG) ReservedSpace) =
670  (ULONG)InterlockedIncrement(LoggerContext->SequencePtr);
671  }
672  goto FoundSpace;
673  } else {
674  //
675  // There is not enough space left to log this event,
676  // Ask for buffer switch. The WmipSwitchBuffer()
677  // will push this current buffer into Dirty list.
678  //
679  // Before asking for buffer switch,
680  // check if I am the one that got overboard.
681  // If yes, put the correct offset back.
682  //
683  if (Offset <= LoggerContext->BufferSize) {
684  Buffer->SavedOffset = Offset;
685  }
686 
687  //
688  // Also, dereference the buffer, so it can be freed.
689  //
690  InterlockedDecrement((PLONG) &Buffer->ReferenceCount);
691 
692  //
693  // Nothing in per process list, ask to get a new buffer
694  //
695  Status = WmipSwitchBuffer(LoggerContext,
696  &Buffer,
697  &LoggerContext->ProcessorBuffers[Processor],
698  Processor);
699 
700  if (!NT_SUCCESS(Status)) {
701  //
702  // Nothing available
703  //
704  goto LostEvent;
705  }
706 
707  ASSERT (Buffer != NULL);
708  goto TryFindSpace;
709  }
710 
711 LostEvent:
712  //
713  // Will get here it we are throwing away the event
714  //
715  ASSERT(Buffer == NULL);
716  LoggerContext->EventsLost++; // best attempt to be accurate
717  ReservedSpace = NULL;
718  if (LoggerContext->SequencePtr) {
719  InterlockedIncrement(LoggerContext->SequencePtr);
720  }
721 
722 FoundSpace:
723  //
724  // notify the logger after critical section
725  //
726  *BufferResource = Buffer;
727 
728  return ReservedSpace;
729 }

833  LoggerContext->LogFileName = NewFileName;

868  STATUS_SEVERITY_INFORMATIONAL);

871  }
872  }

883 WmipRequestLogFile(
884  IN PWMI_LOGGER_CONTEXT LoggerContext
885  )
886 /*++
887 
888 Routine Description:
889 
890  This routine switches logfiles for an active logger. This routine must
891  be called only by the Logger Thread. It will close the previous logfile,
892  if any, properly by terminating it with FLUSH_MARKER and Finalizing the
893  LogHeader.
894 
895  Two different cases to consider here are:
896  1. The newfile was created in user mode with headers and rundown data
897  2. The newfile is created in kernel (need to add LogFileHeader)
898 
899  The callers of this function are
900  1. UpdateLogger: Sets the REQUEST_FLAG_NEW_FILE after presenting
901  a user mode created logfile in NewLogFile.
902  2. NT Kernel Logger session: Switches from DELAY_OPEN mode with
903  a user mode created logfile
904  3. FILE_MODE_NEWFILE: When current logfile reaches the FileLimit
905  FlushBuffer requests a newfile.
906  4. GlobalLogger: Started in DELAY_OPEN && ADD_HEADER mode needs
907  to create the logfile when the FileSystem is ready.
908 
909  In all cases, when the switch is made the old logfile needs to be
910  properly closed after Finalizing its header.
911 
912 
913 
914 Arguments:
915 
916  LoggerContext - Logger Context
917 
918 Return Value:
919 
920  Status
921 
922 Environment:
923 
924  Kernel mode.
925 --*/
926 
927 {
928  NTSTATUS Status = STATUS_SUCCESS;
929 
930  PAGED_CODE();
931 
932  if (!WmipFileSystemReady) {
933  //
934  // FileSystem is not ready yet, so return for now.
935  //
936  return Status;
937  }
938 
939  //
940  // In order for us to act on this request we need something to create
941  // a file with, such as FileName, Pattern etc.,
942  //
943 
944  if ((LoggerContext->LogFileName.Buffer == NULL ) &&
945  (LoggerContext->LogFilePattern.Buffer == NULL) &&

952  // With the REQUEST_FLAG_NEW_FILE set, flush all active buffers.
953  //
954 
955  if (LoggerContext->LogFileHandle != NULL ) {
956  Status = WmipFlushActiveBuffers(LoggerContext, TRUE);
957  }
958 
959  if (NT_SUCCESS(Status)) {
960  if ( (LoggerContext->LoggerMode & EVENT_TRACE_FILE_MODE_NEWFILE) ||
961  ( (LoggerContext->LoggerMode & EVENT_TRACE_DELAY_OPEN_FILE_MODE) &&
962  (LoggerContext->LoggerMode & EVENT_TRACE_ADD_HEADER_MODE))) {
963

965  }
966  else {

968  // UpdateTrace case
969  //

973  EVENT_TRACE_FILE_MODE_APPEND);
974  if (NT_SUCCESS(Status)) {
975  LoggerContext->LoggerMode &= ~EVENT_TRACE_DELAY_OPEN_FILE_MODE;

978  // This is to release the Update thread from the wait
979  //

981  }
982  }
983 
984  if (! NT_SUCCESS(Status)) {
985  LoggerContext->LoggerStatus = Status;
986  }
987  return Status;

996 VOID
997 WmipLogger(
998  IN PWMI_LOGGER_CONTEXT LoggerContext
999  )
1000 
1001 /*++
1002 
1003 Routine Description:
1004  This function is the logger itself. It is started as a system thread.
1005  It will not return until someone has stopped data collection or it
1006  is not successful is flushing out a buffer (e.g. disk is full).
1007 
1008 Arguments:
1009 
1010  None.
1011 
1012 Return Value:
1013 
1014  The status of running the buffer manager
1015 
1016 --*/
1017 
1018 {
1019  NTSTATUS Status, InitialStatus;
1020  ULONG ErrorCount;
1021  ULONG FlushTimeOut;
1022  ULONG64 LastFlushTime=0;
1023 
1024  PAGED_CODE();
1025 
1026  //
1027  // Set the logger status to PENDING before touching logger context.
1028  // If the starting threads times out and sees STATUS_PENDING, it will wait
1029  // until we notify.
1030  //
1031  InitialStatus = InterlockedExchange(&LoggerContext->LoggerStatus,
1032  STATUS_PENDING);
1033 
1034  if (STATUS_CANCELLED == InitialStatus) {
1035  //
1036  // The logger thread was too late to notify the starting thread.
1037  //
1038  // Clean up the LoggerContext, and go away here.
1039  //
1040  WmipFreeLoggerContext(LoggerContext);
1041  PsTerminateSystemThread(InitialStatus);
1042  return;
1043  }
1044 
1045  LoggerContext->LoggerThread = PsGetCurrentThread();
1046 
1047  if ((LoggerContext->LoggerMode & EVENT_TRACE_DELAY_OPEN_FILE_MODE)
1048  || (LoggerContext->LogFileName.Length == 0)) {
1049 
1050  // If EVENT_TRACE_DELAY_OPEN_FILE_MODE is specified, WMI does not
1051  // need to create logfile now.
1052  //
1053  // If there is no LogFileName specified, WMI does not need to create
1054  // logfile either. WmipStartLogger() already checks all possible
1055  // combination of LoggerMode and LogFileName, so we don't need to
1056  // perform the same check again.
1057  //
1058  Status = STATUS_SUCCESS;
1059  } else {
1060  Status = WmipCreateLogFile(LoggerContext,
1061  FALSE,
1062  LoggerContext->LoggerMode & EVENT_TRACE_FILE_MODE_APPEND);
1063  }
1064 
1065 
1066  //
1067  // Set the logger status to let the starting thread know what happened.
1068  //
1069  InitialStatus = InterlockedExchange(&LoggerContext->LoggerStatus,
1070  Status);
1071  ASSERT(InitialStatus != STATUS_CANCELLED);
1072 
1073  if (!NT_SUCCESS(Status)){
1074  //
1075  // Logger thread failed to open the file. We will notify the starting
1076  // thread, but we will clean up the logger context here.
1077  //
1078  if (LoggerContext->LogFileHandle != NULL) {
1079  Status = ZwClose(LoggerContext->LogFileHandle);
1080  LoggerContext->LogFileHandle = NULL;
1081  }
1082  KeSetEvent(&LoggerContext->LoggerEvent, 0, FALSE);
1083  WmipFreeLoggerContext(LoggerContext);
1084  PsTerminateSystemThread(Status);
1085  return;
1086  }
1087  else { // All succeeded.
1088  //
1089  // It is safe to go.
1090  // This is the only place where CollectionOn will be turn on!!!
1091  //
1092  LoggerContext->CollectionOn = TRUE;
1093  KeSetEvent(&LoggerContext->LoggerEvent, 0, FALSE);
1094  }
1095 
1096  ErrorCount = 0;
1097  // by now, the caller has been notified that the logger is running
1098 
1099  //
1100  // Loop and wait for buffers to be filled until someone turns off CollectionOn
1101  //
1102  KeSetBasePriorityThread(KeGetCurrentThread(), LOW_REALTIME_PRIORITY-1);
1103 
1104  while (LoggerContext->CollectionOn) {
1105 
1106  if (LoggerContext->LoggerMode & EVENT_TRACE_BUFFERING_MODE) {
1107  //
1108  // Wait forever until signaled by when logging is terminated.
1109  //
1110  Status = KeWaitForSingleObject(
1111  &LoggerContext->LoggerSemaphore,
1112  Executive,
1113  KernelMode,
1114  FALSE,
1115  NULL);
1116  LoggerContext->LoggerStatus = STATUS_SUCCESS;
1117  } else {
1118  ULONG FlushAll = 0;
1119  ULONG FlushFlag;
1120 
1121  FlushTimeOut = LoggerContext->FlushTimer;
1122  //
1123  // Wake up every second to see if there are any buffers in
1124  // flush list.
1125  //
1126  Status = KeWaitForSingleObject(
1127  &LoggerContext->LoggerSemaphore,
1128  Executive,
1129  KernelMode,
1130  FALSE,
1131  &WmiOneSecond);
1132 
1133  //
1134  // Check if number of buffers need to be adjusted.
1135  //
1136  WmipAdjustFreeBuffers(LoggerContext);
1137 
1138  LoggerContext->LoggerStatus = STATUS_SUCCESS;
1139 
1140  if ((LoggerContext->RequestFlag & REQUEST_FLAG_NEW_FILE) ||
1141  ((LoggerContext->LoggerMode & EVENT_TRACE_DELAY_OPEN_FILE_MODE)
1142  && (LoggerContext->LoggerMode & EVENT_TRACE_ADD_HEADER_MODE)) ) {
1143  Status = WmipRequestLogFile( LoggerContext);
1144  }
1145 
1146  //
1147  // Check to see if we need to FlushAll
1148  //
1149  if (FlushTimeOut) {
1150  ULONG64 Now;
1151  KeQuerySystemTime((PLARGE_INTEGER) &Now);
1152  if ( ((Now - LastFlushTime) / 10000000) >= FlushTimeOut) {
1153  FlushAll = 1;
1154  LastFlushTime = Now;
1155  }
1156  else {
1157  FlushAll = 0;
1158  }
1159  }
1160 
1161  FlushFlag = (LoggerContext->RequestFlag & REQUEST_FLAG_FLUSH_BUFFERS);
1162  if ( FlushFlag )
1163  FlushAll = TRUE;
1164 
1165  Status = WmipFlushActiveBuffers(LoggerContext, FlushAll);
1166  //
1167  // Should check the status, and if failed to write a log file
1168  // header, should clean up. As the log file is bad anyway.
1169  //
1170  if ( FlushFlag ) {
1171  LoggerContext->RequestFlag &= ~REQUEST_FLAG_FLUSH_BUFFERS;
1172  //
1173  // If this was a flush for persistent events, this request
1174  // flag must be reset here.
1175  //
1176  if (LoggerContext->RequestFlag &
1177  REQUEST_FLAG_CIRCULAR_TRANSITION) {
1178  if (LoggerContext->LogFileHandle != NULL) {
1179  WmipFinalizeHeader(LoggerContext->LogFileHandle,
1180  LoggerContext);
1181  }
1182 
1183  LoggerContext->RequestFlag &= ~REQUEST_FLAG_CIRCULAR_TRANSITION;
1184  }
1185 
1186  LoggerContext->LoggerStatus = Status;
1187  KeSetEvent(&LoggerContext->FlushEvent, 0, FALSE);
1188 
1189  }
1190  if (!NT_SUCCESS(Status)) {
1191  LoggerContext->LoggerStatus = Status;
1192  WmipStopLoggerInstance(LoggerContext);
1193  }
1194  }
1195  } // while loop
1196 
1197  if (Status == STATUS_TIMEOUT) {
1198  Status = STATUS_SUCCESS;
1199  }
1200 //
1201 // if a normal collection end, flush out all the buffers before stopping
1202 //
1203 
1204  TraceDebug((2, "WmipLogger: Flush all buffers before stopping...\n"));
1205 //
1206 // First, move the per processor buffer out to FlushList
1207 //
1208  // This is to force buffers to be written
1209  // in FlushBuffer without snapping back to this routine to create a file.
1210  LoggerContext->RequestFlag |= REQUEST_FLAG_NEW_FILE;
1211 
1212  while ((LoggerContext->NumberOfBuffers > 0) &&
1213  (LoggerContext->NumberOfBuffers > LoggerContext->BuffersAvailable)) {
1214  Status = KeWaitForSingleObject(
1215  &LoggerContext->LoggerSemaphore,
1216  Executive,
1217  KernelMode,
1218  FALSE,
1219  &WmiOneSecond);
1220  WmipFlushActiveBuffers(LoggerContext, 1);
1221  TraceDebug((2, "WmipLogger: Stop %d %d %d %d %d\n",
1222  LoggerContext->LoggerId,
1223  LoggerContext->BuffersAvailable,
1224  LoggerContext->BuffersInUse,
1225  LoggerContext->BuffersDirty,
1226  LoggerContext->NumberOfBuffers));
1227  }
1228 
1229  //
1230  // Note that LoggerContext->LogFileObject needs to remain set
1231  // for QueryLogger to work after close
1232  //
1233  if (LoggerContext->LogFileHandle != NULL) {
1234  ZwClose(LoggerContext->LogFileHandle);
1235  TraceDebug((1, "WmipLogger: Close logfile with status=%X\n", Status));
1236  }
1237  LoggerContext->LogFileHandle = NULL;
1238  KeSetEvent(&LoggerContext->FlushEvent, 0, FALSE);
1239  KeSetEvent(&LoggerContext->LoggerEvent, 0, FALSE);
1240 #if DBG
1241  if (!NT_SUCCESS(Status)) {
1242  TraceDebug((1, "WmipLogger: Aborting %d %X\n",
1243  LoggerContext->LoggerId, LoggerContext->LoggerStatus));
1244  }
1245 #endif
1246 
1247  WmipFreeLoggerContext(LoggerContext);
1248  PsTerminateSystemThread(Status);
1249 }

1285 NTSTATUS
1286 WmipFlushBuffer(
1287  IN PWMI_LOGGER_CONTEXT LoggerContext,
1288  IN PWMI_BUFFER_HEADER Buffer,
1289  IN USHORT BufferFlag
1290  )
1291 /*++
1292 
1293 Routine Description:
1294  This function is responsible for flushing a filled buffer out to
1295  disk, or to a real time consumer.
1296 
1297 Arguments:
1298 
1299  LoggerContext - Context of the logger
1300 
1301  Buffer -
1302 
1303  BufferFlag -
1304 
1305 Return Value:
1306 
1307  The status of flushing the buffer
1308 
1309 --*/
1310 {
1311  IO_STATUS_BLOCK IoStatus;
1312  NTSTATUS Status;
1313  ULONG BufferSize;
1314  ULONG BufferPersistenceData = LoggerContext->RequestFlag
1315  & ( REQUEST_FLAG_CIRCULAR_PERSIST
1316  | REQUEST_FLAG_CIRCULAR_TRANSITION);
1317 
1318  ASSERT(LoggerContext != NULL);
1319  ASSERT(Buffer != NULL);
1320 
1321  if (LoggerContext == NULL || Buffer == NULL) {
1322  return STATUS_SEVERITY_ERROR;
1323  }
1324 
1325  //
1326  // Grab the buffer to be flushed
1327  //
1328  BufferSize = LoggerContext->BufferSize;
1329 
1330  //
1331  // Put end of record marker in buffer if available space
1332  //
1333 
1334  TraceDebug((2, "WmipFlushBuffer: %p, Flushed %X %8x %8x %5d\n",
1335  Buffer,
1336  Buffer->ClientContext, Buffer->SavedOffset,
1337  Buffer->CurrentOffset, LoggerContext->BuffersWritten));
1338 
1339  Status = WmipPrepareHeader(LoggerContext, Buffer, BufferFlag);
1340 
1341  if (Status == STATUS_SUCCESS) {
1342 
1343  //
1344  // Buffering mode is mutually exclusive with REAL_TIME_MODE
1345  //
1346  if (!(LoggerContext->LoggerMode & EVENT_TRACE_BUFFERING_MODE)) {
1347  if (LoggerContext->LoggerMode & EVENT_TRACE_REAL_TIME_MODE) {
1348 
1349  if (LoggerContext->UsePerfClock == EVENT_TRACE_CLOCK_PERFCOUNTER) {
1350  Buffer->Wnode.Flags |= WNODE_FLAG_USE_TIMESTAMP;
1351  }
1352 
1353  // need to see if we can send anymore
1354  // check for queue length
1355  if (! NT_SUCCESS(WmipProcessEvent((PWNODE_HEADER)Buffer,
1356  FALSE,
1357  FALSE))) {
1358  LoggerContext->RealTimeBuffersLost++;
1359  }
1360  }
1361  }
1362 
1363  if (LoggerContext->LogFileHandle != NULL) {
1364 
1365  if (LoggerContext->MaximumFileSize > 0) { // if quota given
1366  ULONG64 FileSize = (ULONG64)(LoggerContext->LastFlushedBuffer) * BufferSize;
1367  ULONG64 FileLimit = (ULONG64)(LoggerContext->MaximumFileSize) * BYTES_PER_MB;
1368  if (LoggerContext->LoggerMode & EVENT_TRACE_USE_KBYTES_FOR_SIZE) {
1369  FileLimit = LoggerContext->MaximumFileSize * 1024;
1370  }
1371 
1372  if ( FileSize >= FileLimit ) {
1373 
1374  ULONG LoggerMode = LoggerContext->LoggerMode & 0X000000FF;
1375  //
1376  // Files from user mode always have the APPEND flag.
1377  // We mask it out here to simplify the testing below.
1378  //
1379  LoggerMode &= ~EVENT_TRACE_FILE_MODE_APPEND;
1380  //
1381  // PREALLOCATE flag has to go, too.
1382  //
1383  LoggerMode &= ~EVENT_TRACE_FILE_MODE_PREALLOCATE;
1384 
1385  if (LoggerMode == EVENT_TRACE_FILE_MODE_SEQUENTIAL) {
1386  // do not write to logfile anymore
1387 
1388  Status = STATUS_LOG_FILE_FULL; // control needs to stop logging
1389  // need to fire up a Wmi Event to control console
1390  WmipSendNotification(LoggerContext,
1391  Status, STATUS_SEVERITY_ERROR);
1392  }
1393  else if (LoggerMode == EVENT_TRACE_FILE_MODE_CIRCULAR ||
1394  LoggerMode == EVENT_TRACE_FILE_MODE_CIRCULAR_PERSIST) {
1395  if (BufferPersistenceData > 0) {
1396  // treat circular logfile as sequential logfile if
1397  // logger still processes Persistence events (events
1398  // that cannot be overwritten in circular manner).
1399  //
1400  Status = STATUS_LOG_FILE_FULL;
1401  WmipSendNotification(LoggerContext,
1402  Status, STATUS_SEVERITY_ERROR);
1403  }
1404  else {
1405  // reposition file
1406 
1407  LoggerContext->ByteOffset
1408  = LoggerContext->FirstBufferOffset;
1409  LoggerContext->LastFlushedBuffer = (ULONG)
1410  (LoggerContext->FirstBufferOffset.QuadPart
1411  / LoggerContext->BufferSize);
1412  }
1413  }
1414  else if (LoggerMode & EVENT_TRACE_FILE_MODE_NEWFILE) {
1415 
1416  //
1417  // We will set the RequestFlag to initiate a file switch.
1418  // If that flag is already set, then we continue to flush
1419  // past the FileLimit.
1420  //
1421  // There should be no race condition with an UpdateTrace
1422  // setting the RequestFlag since we do not allow change of
1423  // filename for NEWFILE mode.
1424  //
1425 
1426  if ( (LoggerContext->RequestFlag & REQUEST_FLAG_NEW_FILE) !=
1427  REQUEST_FLAG_NEW_FILE)
1428  {
1429  LoggerContext->RequestFlag |= REQUEST_FLAG_NEW_FILE;
1430  }
1431  }
1432  }
1433  }
1434 
1435  if (NT_SUCCESS(Status)) {
1436  Status = ZwWriteFile(
1437  LoggerContext->LogFileHandle,
1438  NULL,
1439  NULL,
1440  NULL,
1441  &IoStatus,
1442  Buffer,
1443  BufferSize,
1444  &LoggerContext->ByteOffset,
1445  NULL);
1446 
1447  if (NT_SUCCESS(Status)) {
1448  LoggerContext->ByteOffset.QuadPart += BufferSize;
1449 
1450  //
1451  // update FirstBufferOffset so that persistence event will
1452  // not be overwritten in circular logfile. We need to check that
1453  // buffer type is RUNDOWN because if persistent --> non persistent
1454  // transition happens at high IRQL we will not be able to flush logger
1455  // and might end up with non RUNDOWN buffers in the rundown stream
1456  //
1457  if ((BufferPersistenceData > 0) &&
1458  (Buffer->BufferType == WMI_BUFFER_TYPE_RUNDOWN)) {
1459 
1460  LoggerContext->FirstBufferOffset.QuadPart += BufferSize;
1461  }
1462  }
1463  else if (Status == STATUS_LOG_FILE_FULL ||
1464  Status == STATUS_DISK_FULL) {
1465  // need to fire up a Wmi Event to control console
1466  WmipSendNotification(LoggerContext,
1467  STATUS_LOG_FILE_FULL, STATUS_SEVERITY_ERROR);
1468  }
1469  else {
1470  TraceDebug((2, "WmipFlushBuffer: Unknown WriteFile Failure with status=%X\n", Status));
1471  }
1472  }
1473  }
1474 
1475  // Now do callbacks. This happens whether a file exists or not.
1476  if (WmipGlobalBufferCallback) {
1477  (WmipGlobalBufferCallback) (Buffer, WmipGlobalCallbackContext);
1478  }
1479  if (LoggerContext->BufferCallback) {
1480  (LoggerContext->BufferCallback) (Buffer, LoggerContext->CallbackContext);
1481  }
1482  }
1483 
1484  if (NT_SUCCESS(Status)) {
1485  LoggerContext->BuffersWritten++;
1486  LoggerContext->LastFlushedBuffer++;
1487  }
1488  else {
1489 #if DBG
1490  if (Status == STATUS_NO_DATA_DETECTED) {
1491  TraceDebug((2, "WmipFlushBuffer: Empty buffer detected\n"));
1492  }
1493  else if (Status == STATUS_SEVERITY_WARNING) {
1494  TraceDebug((2, "WmipFlushBuffer: Buffer could be corrupted\n"));
1495  }
1496  else {
1497  TraceDebug((2,
1498  "WmipFlushBuffer: Unable to write buffer: status=%X\n",
1499  Status));
1500  }
1501 #endif
1502  if ((Status != STATUS_NO_DATA_DETECTED) &&
1503  (Status != STATUS_SEVERITY_WARNING))
1504  LoggerContext->LogBuffersLost++;
1505  }
1506 
1507  return Status;
1508 }
