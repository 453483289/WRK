89 LpcInitSystem (
90  VOID
91 )
92 
93 /*++
94 
95 Routine Description:
96 
97  This function performs the system initialization for the LPC package.
98  LPC stands for Local Inter-Process Communication.
99 
100 Arguments:
101 
102  None.
103 
104 Return Value:
105 
106  TRUE if successful and FALSE if an error occurred.
107 
108  The following errors can occur:
109 
110  - insufficient memory
111 
112 --*/
113 
114 {
115  OBJECT_TYPE_INITIALIZER ObjectTypeInitializer;
116  UNICODE_STRING PortTypeName;
117  ULONG ZoneElementSize;
118 
119  //
120  // Initialize our global lpc lock
121  //
122 
123  LpcpInitializeLpcpLock();
124 
125  //
126  // Create the object type for the port object
127  //
128 
129  RtlInitUnicodeString( &PortTypeName, L"Port" );
130 
131  RtlZeroMemory( &ObjectTypeInitializer, sizeof( ObjectTypeInitializer ));
132 
133  ObjectTypeInitializer.Length = sizeof( ObjectTypeInitializer );
134  ObjectTypeInitializer.GenericMapping = LpcpPortMapping;
135  ObjectTypeInitializer.MaintainTypeList = FALSE;
136  ObjectTypeInitializer.PoolType = PagedPool;
137  ObjectTypeInitializer.DefaultPagedPoolCharge = FIELD_OFFSET( LPCP_PORT_OBJECT, WaitEvent );
138  ObjectTypeInitializer.DefaultNonPagedPoolCharge = sizeof( LPCP_NONPAGED_PORT_QUEUE );
139  ObjectTypeInitializer.InvalidAttributes = OBJ_VALID_ATTRIBUTES ^ PORT_VALID_OBJECT_ATTRIBUTES;
140  ObjectTypeInitializer.ValidAccessMask = PORT_ALL_ACCESS;
141  ObjectTypeInitializer.CloseProcedure = LpcpClosePort;
142  ObjectTypeInitializer.DeleteProcedure = LpcpDeletePort;
143  ObjectTypeInitializer.UseDefaultObject = TRUE ;
144 
145  ObCreateObjectType( &PortTypeName,
146  &ObjectTypeInitializer,
147  (PSECURITY_DESCRIPTOR)NULL,
148  &LpcPortObjectType );
149 
150  //
151  // Create the object type for the waitable port object
152  //
153 
154  RtlInitUnicodeString( &PortTypeName, L"WaitablePort" );
155  ObjectTypeInitializer.PoolType = NonPagedPool;
156  ObjectTypeInitializer.DefaultNonPagedPoolCharge += sizeof( LPCP_PORT_OBJECT );
157  ObjectTypeInitializer.DefaultPagedPoolCharge = 0;
158  ObjectTypeInitializer.UseDefaultObject = FALSE;
159 
160  ObCreateObjectType( &PortTypeName,
161  &ObjectTypeInitializer,
162  (PSECURITY_DESCRIPTOR)NULL,
163  &LpcWaitablePortObjectType );
164 
165  //
166  // Initialize the lpc port zone. Each element can contain a max
167  // message, plus an LPCP message structure, plus an LPCP connection
168  // message
169  //
170 
171  ZoneElementSize = PORT_MAXIMUM_MESSAGE_LENGTH +
172  sizeof( LPCP_MESSAGE ) +
173  sizeof( LPCP_CONNECTION_MESSAGE );
174 
175  //
176  // Round up the size to the next 16 byte alignment
177  //
178 
179  ZoneElementSize = (ZoneElementSize + LPCP_ZONE_ALIGNMENT - 1) &
180  LPCP_ZONE_ALIGNMENT_MASK;
181 
182  //
183  // Initialize the zone
184  //
185 
186  LpcpInitializePortZone( ZoneElementSize );
187 
188  LpcpInitilizeLogging();
189 
190  return( TRUE );
191 }
