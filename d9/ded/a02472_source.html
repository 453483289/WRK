33 // Test for alertable condition.
34 //
35 // If alertable is TRUE and the thread is alerted for a processor
36 // mode that is equal to the wait mode, then return immediately
37 // with a wait completion status of ALERTED.
38 //
39 // Else if alertable is TRUE, the wait mode is user, and the user APC

78 /*++
79 
80 Routine Description:
81 
82  If the current thread is not a time critical or real time thread, then
83  adjust its quantum in accordance with the adjustment that would have
84  occurred if the thread had actually waited.
85 
86  N.B. This routine is entered at SYNCH_LEVEL and exits at the wait
87  IRQL of the subject thread after having exited the scheduler.
88 
89 Arguments:
90 
91  Thread - Supplies a pointer to the current thread.
92 
93 Return Value:
94 
95  None.
96 
97 --*/
98 
99 {
100 
101  PKPRCB Prcb;
102  PKTHREAD NewThread;
103 
104  //
105  // Acquire the thread lock and the PRCB lock.
106  //
107  // If the thread is not a real time or time critical thread, then adjust
108  // the thread quantum.
109  //
110 
111  Prcb = KeGetCurrentPrcb();
112  KiAcquireThreadLock(Thread);
113  KiAcquirePrcbLock(Prcb);
114  if ((Thread->Priority < LOW_REALTIME_PRIORITY) &&
115  (Thread->BasePriority < TIME_CRITICAL_PRIORITY_BOUND)) {
116 
117  Thread->Quantum -= WAIT_QUANTUM_DECREMENT;
118  if (Thread->Quantum <= 0) {
119 
120  //
121  // Quantum end has occurred. Adjust the thread priority.
122  //
123 
124  Thread->Quantum = Thread->QuantumReset;
125 
126  //
127  // Compute the new thread priority and attempt to reschedule the
128  // current processor as if a quantum end had occurred.
129  //
130  // N.B. The new priority will never be greater than the previous
131  // priority.
132  //
133 
134  Thread->Priority = KiComputeNewPriority(Thread, 1);
135  if (Prcb->NextThread == NULL) {
136  if ((NewThread = KiSelectReadyThread(Thread->Priority, Prcb)) != NULL) {
137  NewThread->State = Standby;
138  Prcb->NextThread = NewThread;
139  }
140 
141  } else {
142  Thread->Preempted = FALSE;
143  }
144  }
145  }
146 
147  //
148  // Release the thread lock, release the PRCB lock, exit the scheduler,
149  // and return.
150  //
151 
152  KiReleasePrcbLock(Prcb);
153  KiReleaseThreadLock(Thread);
154  KiExitDispatcher(Thread->WaitIrql);
155  return;
156 }

241 /*++
242 
243 Routine Description:
244 
245  This function delays the execution of the current thread for the specified
246  interval of time.
247 
248 Arguments:
249 
250  WaitMode - Supplies the processor mode in which the delay is to occur.
251 
252  Alertable - Supplies a boolean value that specifies whether the delay
253  is alertable.
254 
255  Interval - Supplies a pointer to the absolute or relative time over which
256  the delay is to occur.
257 
258 Return Value:
259 
260  The wait completion status. A value of STATUS_SUCCESS is returned if
261  the delay occurred. A value of STATUS_ALERTED is returned if the wait
262  was aborted to deliver an alert to the current thread. A value of
263  STATUS_USER_APC is returned if the wait was aborted to deliver a user
264  APC to the current thread.
265 
266 --*/
267 
268 {
269 
270  LARGE_INTEGER DueTime;
271  ULONG Hand;
272  LARGE_INTEGER NewTime;
273  PLARGE_INTEGER OriginalTime;
274  PKPRCB Prcb;
275  PRKQUEUE Queue;
276  LOGICAL StackSwappable;
277  PKTHREAD Thread;
278  PRKTIMER Timer;
279  PKWAIT_BLOCK WaitBlock;
280  NTSTATUS WaitStatus;
281 
282  //
283  // If the specfied wait interval is zero and the wait mode is not kernel
284  // mode, then attempt to yield execution.
285  //
286 
287  Hand = 0;
288  Thread = KeGetCurrentThread();
289  if ((Interval->QuadPart == 0) &&
290  (WaitMode != KernelMode)) {
291 
292  if ((Alertable == FALSE) &&
293  (Thread->ApcState.UserApcPending == FALSE)) {
294 
295  return NtYieldExecution();
296  }
297  }
298 
299  //
300  // Set constant variables.
301  //
302 
303  OriginalTime = Interval;
304  Timer = &Thread->Timer;
305  WaitBlock = &Thread->WaitBlock[TIMER_WAIT_BLOCK];
306 
307  //
308  // If the dispatcher database is already held, then initialize the thread
309  // local variables. Otherwise, raise IRQL to DPC level, initialize the
310  // thread local variables, and lock the dispatcher database.
311  //
312 
313  if (ReadForWriteAccess(&Thread->WaitNext) == FALSE) {
314  goto WaitStart;
315  }
316 
317  Thread->WaitNext = FALSE;
318  InitializeDelayExecution();
319 
320  //
321  // Start of delay loop.
322  //
323  // Note this loop is repeated if a kernel APC is delivered in the middle
324  // of the delay or a kernel APC is pending on the first attempt through
325  // the loop.
326  //
327 
328  do {
329 
330  //
331  // Test to determine if a kernel APC is pending.
332  //
333  // If a kernel APC is pending, the special APC disable count is zero,
334  // and the previous IRQL was less than APC_LEVEL, then a kernel APC
335  // was queued by another processor just after IRQL was raised to
336  // DISPATCH_LEVEL, but before the dispatcher database was locked.
337  //
338  // N.B. This can only happen in a multiprocessor system.
339  //
340 
341  Thread->Preempted = FALSE;
342  if (Thread->ApcState.KernelApcPending &&
343  (Thread->SpecialApcDisable == 0) &&
344  (Thread->WaitIrql < APC_LEVEL)) {
345 
346  //
347  // Unlock the dispatcher database and lower IRQL to its previous
348  // value. An APC interrupt will immediately occur which will result
349  // in the delivery of the kernel APC if possible.
350  //
351 
352  KiUnlockDispatcherDatabase(Thread->WaitIrql);
353 
354  } else {
355 
356  //
357  // Test for alert pending.
358  //
359 
360  TestForAlertPending(Alertable);
361 
362  //
363  // Check if the timer has already expired.
364  //
365  // N.B. The constant fields of the timer wait block are
366  // initialized when the thread is initialized. The
367  // constant fields include the wait object, wait key,
368  // wait type, and the wait list entry link pointers.
369  //
370 
371  Prcb = KeGetCurrentPrcb();
372  if (KiCheckDueTime(Timer) == FALSE) {
373  goto NoWait;
374  }
375 
376  //
377  // If the current thread is processing a queue entry, then attempt
378  // to activate another thread that is blocked on the queue object.
379  //
380 
381  Queue = Thread->Queue;
382  if (Queue != NULL) {
383  KiActivateWaiterQueue(Queue);
384  }
385 
386  //
387  // Set the thread wait parameters, set the thread dispatcher
388  // state to Waiting, and insert the thread in the wait list if
389  // the kernel stack of the current thread is swappable.
390  //
391 
392  Thread->State = Waiting;
393  if (StackSwappable != FALSE) {
394  InsertTailList(&Prcb->WaitListHead, &Thread->WaitListEntry);
395  }
396 
397  //
398  // Set swap busy for the current thread, unlock the dispatcher
399  // database, and switch to a new thread.
400  //
401  // Control is returned at the original IRQL.
402  //
403 
404  ASSERT(Thread->WaitIrql <= DISPATCH_LEVEL);
405 
406  KiSetContextSwapBusy(Thread);
407  KiInsertOrSignalTimer(Timer, Hand);
408  WaitStatus = (NTSTATUS)KiSwapThread(Thread, Prcb);
409 
410  //
411  // If the thread was not awakened to deliver a kernel mode APC,
412  // then return the wait status.
413  //
414 
415  if (WaitStatus != STATUS_KERNEL_APC) {
416  if (WaitStatus == STATUS_TIMEOUT) {
417  WaitStatus = STATUS_SUCCESS;
418  }
419 
420  return WaitStatus;
421  }
422 
423  //
424  // Reduce the time remaining before the time delay expires.
425  //
426 
427  Interval = KiComputeWaitInterval(OriginalTime,
428  &DueTime,
429  &NewTime);
430  }
431 
432  //
433  // Raise IRQL to SYNCH level, initialize the thread local variables,
434  // and lock the dispatcher database.
435  //
436 
437 WaitStart:
438 
439  Thread->WaitIrql = KeRaiseIrqlToSynchLevel();
440  InitializeDelayExecution();
441  KiLockDispatcherDatabaseAtSynchLevel();
442 
443  } while (TRUE);
444 
445  //
446  // The thread is alerted or a user APC should be delivered. Unlock the
447  // dispatcher database, lower IRQL to its previous value, and return the
448  // wait status.
449  //
450 
451  KiUnlockDispatcherDatabase(Thread->WaitIrql);
452  return WaitStatus;
453 
454  //
455  // The wait has been satisfied without actually waiting.
456  //
457  // If the wait time is zero, then unlock the dispatcher database and
458  // yield execution. Otherwise, unlock the dispatcher database, remain
459  // at SYNCH level, adjust the thread quantum, exit the dispatcher, and
460  // and return the wait completion status.
461  //
462 
463 NoWait:
464 
465  if (Interval->QuadPart == 0) {
466  KiUnlockDispatcherDatabase(Thread->WaitIrql);
467  return NtYieldExecution();
468 
469  } else {
470  KiUnlockDispatcherDatabaseFromSynchLevel();
471  KiAdjustQuantumThread(Thread);
472  return STATUS_SUCCESS;
473  }
474 }

559 /*++
560 
561 Routine Description:
562 
563  This function waits until the specified objects attain a state of
564  Signaled. The wait can be specified to wait until all of the objects
565  attain a state of Signaled or until one of the objects attains a state
566  of Signaled. An optional timeout can also be specified. If a timeout
567  is not specified, then the wait will not be satisfied until the objects
568  attain a state of Signaled. If a timeout is specified, and the objects
569  have not attained a state of Signaled when the timeout expires, then
570  the wait is automatically satisfied. If an explicit timeout value of
571  zero is specified, then no wait will occur if the wait cannot be satisfied
572  immediately. The wait can also be specified as alertable.
573 
574 Arguments:
575 
576  Count - Supplies a count of the number of objects that are to be waited
577  on.
578 
579  Object[] - Supplies an array of pointers to dispatcher objects.
580 
581  WaitType - Supplies the type of wait to perform (WaitAll, WaitAny).
582 
583  WaitReason - Supplies the reason for the wait.
584 
585  WaitMode - Supplies the processor mode in which the wait is to occur.
586 
587  Alertable - Supplies a boolean value that specifies whether the wait is
588  alertable.
589 
590  Timeout - Supplies a pointer to an optional absolute of relative time over
591  which the wait is to occur.
592 
593  WaitBlockArray - Supplies an optional pointer to an array of wait blocks
594  that are to used to describe the wait operation.
595 
596 Return Value:
597 
598  The wait completion status. A value of STATUS_TIMEOUT is returned if a
599  timeout occurred. The index of the object (zero based) in the object
600  pointer array is returned if an object satisfied the wait. A value of
601  STATUS_ALERTED is returned if the wait was aborted to deliver an alert
602  to the current thread. A value of STATUS_USER_APC is returned if the
603  wait was aborted to deliver a user APC to the current thread.
604 
605 --*/
606 
607 {
608 
609  PKPRCB CurrentPrcb;
610  LARGE_INTEGER DueTime;
611  ULONG Hand;
612  ULONG_PTR Index;
613  LARGE_INTEGER NewTime;
614  PKMUTANT Objectx;
615  PLARGE_INTEGER OriginalTime;
616  PRKQUEUE Queue;
617  LOGICAL StackSwappable;
618  PRKTHREAD Thread;
619  PRKTIMER Timer;
620  PRKWAIT_BLOCK WaitBlock;
621  NTSTATUS WaitStatus;
622  PKWAIT_BLOCK WaitTimer;

624  //
625  // Set constant variables.
626  //
627 
628  Hand = 0;
629  Thread = KeGetCurrentThread();
630  OriginalTime = Timeout;
631  Timer = &Thread->Timer;
632  WaitTimer = &Thread->WaitBlock[TIMER_WAIT_BLOCK];
633 
634  //
635  // If a wait block array has been specified, then the maximum number of
636  // objects that can be waited on is specified by MAXIMUM_WAIT_OBJECTS.
637  // Otherwise the builtin wait blocks in the thread object are used and
638  // the maximum number of objects that can be waited on is specified by
639  // THREAD_WAIT_OBJECTS. If the specified number of objects is not within
640  // limits, then bugcheck.
641  //
642 
643  if (ARGUMENT_PRESENT(WaitBlockArray)) {
644  if (Count > MAXIMUM_WAIT_OBJECTS) {
645  KeBugCheck(MAXIMUM_WAIT_OBJECTS_EXCEEDED);
646  }
647 
648  } else {
649  if (Count > THREAD_WAIT_OBJECTS) {
650  KeBugCheck(MAXIMUM_WAIT_OBJECTS_EXCEEDED);
651  }
652 
653  WaitBlockArray = &Thread->WaitBlock[0];
654  }
655 
656  ASSERT(Count != 0);
657 
658  //
659  // If the dispatcher database is already held, then initialize the thread
660  // local variables. Otherwise, raise IRQL to DPC level, initialize the
661  // thread local variables, and lock the dispatcher database.
662  //
663 
664  if (ReadForWriteAccess(&Thread->WaitNext) == FALSE) {
665  goto WaitStart;
666  }
667 
668  Thread->WaitNext = FALSE;
669  InitializeWaitMultiple();
670 
671  //
672  // Start of wait loop.
673  //
674  // Note this loop is repeated if a kernel APC is delivered in the middle
675  // of the wait or a kernel APC is pending on the first attempt through
676  // the loop.
677  //
678 
679  do {
680 
681  //
682  // Test to determine if a kernel APC is pending.
683  //
684  // If a kernel APC is pending, the special APC disable count is zero,
685  // and the previous IRQL was less than APC_LEVEL, then a kernel APC
686  // was queued by another processor just after IRQL was raised to
687  // DISPATCH_LEVEL, but before the dispatcher database was locked.
688  //
689  // N.B. This can only happen in a multiprocessor system.
690  //
691 
692  Thread->Preempted = FALSE;
693  if (Thread->ApcState.KernelApcPending &&
694  (Thread->SpecialApcDisable == 0) &&
695  (Thread->WaitIrql < APC_LEVEL)) {
696 
697  //
698  // Unlock the dispatcher database and lower IRQL to its previous
699  // value. An APC interrupt will immediately occur which will result
700  // in the delivery of the kernel APC if possible.
701  //
702 
703  KiUnlockDispatcherDatabase(Thread->WaitIrql);
704 
705  } else {
706 
707  //
708  // Construct wait blocks and check to determine if the wait is
709  // already satisfied. If the wait is satisfied, then perform
710  // wait completion and return. Else put current thread in a wait
711  // state if an explicit timeout value of zero is not specified.
712  //
713 
714  Index = 0;
715  if (WaitType == WaitAny) {
716  do {
717 
718  //
719  // Test if wait can be satisfied immediately.
720  //
721 
722  Objectx = (PKMUTANT)Object[Index];
723 
724  ASSERT(Objectx->Header.Type != QueueObject);
725 
726  //
727  // If the object is a mutant object and the mutant object
728  // has been recursively acquired MINLONG times, then raise
729  // an exception. Otherwise if the signal state of the mutant
730  // object is greater than zero, or the current thread is
731  // the owner of the mutant object, then satisfy the wait.
732  //
733 
734  if (Objectx->Header.Type == MutantObject) {
735  if ((Objectx->Header.SignalState > 0) ||
736  (Thread == Objectx->OwnerThread)) {
737  if (Objectx->Header.SignalState != MINLONG) {
738  KiWaitSatisfyMutant(Objectx, Thread);
739  WaitStatus = (NTSTATUS)(Index | Thread->WaitStatus);
740  goto NoWait;
741 
742  } else {
743  KiUnlockDispatcherDatabase(Thread->WaitIrql);
744  ExRaiseStatus(STATUS_MUTANT_LIMIT_EXCEEDED);
745  }
746  }
747 
748  //
749  // If the signal state is greater than zero, then satisfy
750  // the wait.
751  //
752 
753  } else if (Objectx->Header.SignalState > 0) {
754  KiWaitSatisfyOther(Objectx);
755  WaitStatus = (NTSTATUS)(Index);
756  goto NoWait;
757  }
758 
759  Index += 1;
760 
761  } while(Index < Count);
762 
763  } else {
764  do {
765 
766  //
767  // Test if wait can be satisfied.
768  //
769 
770  Objectx = (PKMUTANT)Object[Index];
771 
772  ASSERT(Objectx->Header.Type != QueueObject);
773 
774  //
775  // If the object is a mutant object and the mutant object
776  // has been recursively acquired MINLONG times, then raise
777  // an exception. Otherwise if the signal state of the mutant
778  // object is less than or equal to zero and the current
779  // thread is not the owner of the mutant object, then the
780  // wait cannot be satisfied.
781  //
782 
783  if (Objectx->Header.Type == MutantObject) {
784  if ((Thread == Objectx->OwnerThread) &&
785  (Objectx->Header.SignalState == MINLONG)) {
786  KiUnlockDispatcherDatabase(Thread->WaitIrql);
787  ExRaiseStatus(STATUS_MUTANT_LIMIT_EXCEEDED);
788 
789  } else if ((Objectx->Header.SignalState <= 0) &&
790  (Thread != Objectx->OwnerThread)) {
791  break;
792  }
793 
794  //
