142 //
143 // NOTES:
144 // KeyPath is relative to currentcontrolset. So, if the variable
145 // of interest is
146 // "\registry\machine\system\currentcontrolset\control\fruit\apple:x"
147 // the entry is
148 // { L"fruit\\apple",
149 // L"x",
150 // &Xbuffer,
151 // sizeof(ULONG),
152 // &Xtype
153 // }
154 //
155 // *BufferLength is available space on input
156 // on output:
157 // -1 = no such key or value
158 // 0 = key and value exist, but have 0 length data
159 // > input = buffer too small, filled to available space,
160 // value is actual size of data in registry
161 // <= input = number of bytes copied out
162 //
163 typedef struct _CM_SYSTEM_CONTROL_VECTOR {
164  PWSTR KeyPath; // path name relative to
165  // current control set
166  PWSTR ValueName; // name of value entry
167  PVOID Buffer; // data goes here
168  PULONG BufferLength; // IN: space allocated
169  // OUT: space used, -1 for no such
170  // key or value, 0 for key/value
171  // found but has 0 length data
172  // if NULL pointer, assume 4 bytes
173  // (reg DWORD) available and do not
174  // report actual size
175  PULONG Type; // return type of found data, may
176  // be NULL
177 } CM_SYSTEM_CONTROL_VECTOR, *PCM_SYSTEM_CONTROL_VECTOR;
