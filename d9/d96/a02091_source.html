46 HvpBuildMapAndCopy(
47  PHHIVE Hive,
48  PVOID Image
49  )
50 /*++
51 
52 Routine Description:
53 
54  Creates the map for the Stable storage of the hive, and inits
55  the map for the volatile storage.
56 
57  Following fields in hive must already be filled in:
58 
59  Allocate, Free
60 
61  Will initialize Storage structure of HHIVE.
62 
63  This function is called for the HINIT_MEMORY case. The hive is guaranteed
64  to be in paged-pool. More than that, the hive image is contiguous.
65  It'll then copy from that image to the new paged-pool allocations.
66 
67 Arguments:
68 
69  Hive - Pointer to hive control structure to build map for.
70 
71  Image - pointer to flat memory image of original hive.
72 
73 Return Value:
74 
75  TRUE - it worked
76  FALSE - either hive is corrupt or no memory for map
77 
78 --*/
79 {
80  PHBASE_BLOCK BaseBlock;
81  ULONG Length;
82  ULONG MapSlots;
83  ULONG Tables;
84  PHMAP_TABLE t = NULL;
85  PHMAP_DIRECTORY d = NULL;
86  PHBIN Bin;
87  PHBIN CurrentBin;
88  ULONG Offset;
89  ULONG_PTR Address;
90  PHMAP_ENTRY Me;
91  NTSTATUS Status;
92  PULONG Vector;
93 
94 
95  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_HIVE,"HvpBuildMap:\n"));
96  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_HIVE,"\tHive=%p",Hive));
97 
98 
99  //
100  // Compute size of data region to be mapped
101  //
102  BaseBlock = Hive->BaseBlock;
103  Length = BaseBlock->Length;
104  if ((Length % HBLOCK_SIZE) != 0 ) {
105  Status = STATUS_REGISTRY_CORRUPT;
106  goto ErrorExit1;
107  }
108  MapSlots = Length / HBLOCK_SIZE;
109  if( MapSlots > 0 ) {
110  Tables = (MapSlots-1) / HTABLE_SLOTS;
111  } else {
112  Tables = 0;
113  }
114 
115  Hive->Storage[Stable].Length = Length;
116 
117  //
118  // allocate dirty vector if one is not already present (from HvpRecoverData)
119  //
120 
121  if (Hive->DirtyVector.Buffer == NULL) {
122  Vector = (PULONG)((Hive->Allocate)(ROUND_UP(Length/HSECTOR_SIZE/8,sizeof(ULONG)), TRUE,CM_FIND_LEAK_TAG22));
123  if (Vector == NULL) {
124  Status = STATUS_NO_MEMORY;
125  goto ErrorExit1;
126  }
127  RtlZeroMemory(Vector, Length / HSECTOR_SIZE / 8);
128  RtlInitializeBitMap(&Hive->DirtyVector, Vector, Length / HSECTOR_SIZE);
129  Hive->DirtyAlloc = ROUND_UP(Length/HSECTOR_SIZE/8,sizeof(ULONG));
130  }
131 
132  //
133  // allocate and build structure for map
134  //
135  if (Tables == 0) {
136 
137  //
138  // Just 1 table, no need for directory
139  //
140  t = (Hive->Allocate)(sizeof(HMAP_TABLE), FALSE,CM_FIND_LEAK_TAG23);
141  if (t == NULL) {
142  Status = STATUS_INSUFFICIENT_RESOURCES;
143  goto ErrorExit1;
144  }
145  RtlZeroMemory(t, sizeof(HMAP_TABLE));
146  Hive->Storage[Stable].Map =
147  (PHMAP_DIRECTORY)&(Hive->Storage[Stable].SmallDir);
148  Hive->Storage[Stable].SmallDir = t;
149 
150  } else {
151 
152  //
153  // Need directory and multiple tables
154  //
155  d = (PHMAP_DIRECTORY)(Hive->Allocate)(sizeof(HMAP_DIRECTORY), FALSE,CM_FIND_LEAK_TAG24);
156  if (d == NULL) {
157  Status = STATUS_INSUFFICIENT_RESOURCES;
158  goto ErrorExit1;
159  }
160  RtlZeroMemory(d, sizeof(HMAP_DIRECTORY));
161 
162  //
163  // Allocate tables and fill in dir
164  //
165  if (HvpAllocateMap(Hive, d, 0, Tables) == FALSE) {
166  Status = STATUS_INSUFFICIENT_RESOURCES;
167  goto ErrorExit2;
168  }
169  Hive->Storage[Stable].Map = d;
170  Hive->Storage[Stable].SmallDir = 0;
171  }
172 
173  //
174  // Now we have to allocate the memory for the HBINs and fill in
175  // the map appropriately. We are careful never to allocate less
176  // than a page to avoid fragmenting pool. As long as the page
177  // size is a multiple of HBLOCK_SIZE (a fairly good assumption as
178  // long as HBLOCK_SIZE is 4k) this strategy will prevent pool
179  // fragmentation.
180  //
181  // If we come across an HBIN that is entirely composed of a freed
182  // HCELL, then we do not allocate memory, but mark its HBLOCKs in
183  // the map as not present. HvAllocateCell will allocate memory for
184  // the bin when it is needed.
185  //
186  Offset = 0;
187  Bin = (PHBIN)Image;
188 
189  while (Bin < (PHBIN)((PUCHAR)(Image) + Length)) {
190 
191  if ( (Bin->Size > (Length-Offset)) ||
192  (Bin->Signature != HBIN_SIGNATURE) ||
193  (Bin->FileOffset != Offset)
194  )
195  {
196  //
197  // Bin is bogus
198  //
199  Status = STATUS_REGISTRY_CORRUPT;
200  goto ErrorExit2;
201  }
202 
203  CurrentBin = (PHBIN)(Hive->Allocate)(Bin->Size, FALSE,CM_FIND_LEAK_TAG25);
204  if (CurrentBin==NULL) {
205  Status = STATUS_INSUFFICIENT_RESOURCES;
206  goto ErrorExit2;
207  }
208  RtlCopyMemory(CurrentBin,
209  (PUCHAR)Image+Offset,
210  Bin->Size);
211 
212  //
213  // create map entries for each block/page in bin
214  //
215  Address = (ULONG_PTR)CurrentBin;
216  do {
217  Me = HvpGetCellMap(Hive, Offset);
218  VALIDATE_CELL_MAP(__LINE__,Me,Hive,Offset);
219  Me->BlockAddress = Address;
220  Me->BinAddress = (ULONG_PTR)CurrentBin;
221 
222  if (Address == (ULONG_PTR)CurrentBin) {
223  Me->BinAddress |= HMAP_NEWALLOC;
224  Me->MemAlloc = CurrentBin->Size;
225  } else {
226  Me->MemAlloc = 0;
227  }
228 
229  Me->BinAddress |= HMAP_INPAGEDPOOL;
230  // we don't need to set this - just for debug purposes
231  ASSERT( (Me->CmView = NULL) == NULL );
232 
233  Address += HBLOCK_SIZE;
234  Offset += HBLOCK_SIZE;
235  } while ( Address < ((ULONG_PTR)CurrentBin + CurrentBin->Size ));
236 
237  if (Hive->ReadOnly == FALSE) {
238 
239  //
240  // add free cells in the bin to the appropriate free lists
241  //
242  if ( ! HvpEnlistFreeCells(Hive,
243  CurrentBin,
244  CurrentBin->FileOffset
245  )) {
246  Status = STATUS_REGISTRY_CORRUPT;
247  goto ErrorExit2;
248  }
249  }
250 
251  Bin = (PHBIN)((ULONG_PTR)Bin + Bin->Size);
252  }
253 
254  return STATUS_SUCCESS;
255 
256 
257 ErrorExit2:
258  if (d != NULL) {
259 
260  //
261  // directory was built and allocated, so clean it up
262  //
263 
264  HvpFreeMap(Hive, d, 0, Tables);
265  (Hive->Free)(d, sizeof(HMAP_DIRECTORY));
266  }
267 
268 ErrorExit1:
269  return Status;
270 }

410 HvpEnlistBinInMap(
411  PHHIVE Hive,
412  ULONG Length,
413  PHBIN Bin,
414  ULONG Offset,
415  PVOID CmView OPTIONAL
416  )
417 /*++
418 
419 Routine Description:
420 
421  Creates map entries and enlist free cells for the specified bin
422 
423 Arguments:
424 
425  Hive - Pointer to hive control structure containing the target map
426 
427  Length - the Length of the hive image
428 
429  Bin - the bin to be enlisted
430 
431  Offset - the offset within the hive file
432 
433  CmView - pointer to the mapped view of the bin. If NULL, the bin resides in paged pool
434 
435 Return Value:
436 
437  STATUS_SUCCESS - it worked
438  STATUS_REGISTRY_CORRUPT - the bin is inconsistent
439  STATUS_REGISTRY_RECOVERED - if we have fixed the bin on-the-fly (self heal feature).
440 
441 --*/
442 {
443  NTSTATUS Status = STATUS_SUCCESS;
444  ULONG BinOffset;
445  ULONG_PTR Address;
446  PHMAP_ENTRY Me;
447 
448 #if DBG
449  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_HIVE,"HvpEnlistBinInMap:\n"));
450  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_HIVE,"\tHive=%p\t Offset=%08lx",Hive,Offset));
451  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"HvpEnlistBinInMap: BinAddress = 0x%p\t Size = 0x%lx\n", Bin, Bin->Size));
452 #endif
453 
454  //
455  // create map entries for each block/page in bin
456  //
457  BinOffset = Offset;
458  for (Address = (ULONG_PTR)Bin;
459  Address < ((ULONG_PTR)Bin + Bin->Size);
460  Address += HBLOCK_SIZE
461  )
462  {
463  Me = HvpGetCellMap(Hive, Offset);
464  VALIDATE_CELL_MAP(__LINE__,Me,Hive,Offset);
465  Me->BlockAddress = Address;
466  Me->BinAddress = (ULONG_PTR)Bin;
467  if (Offset == BinOffset) {
468  Me->BinAddress |= HMAP_NEWALLOC;
469  Me->MemAlloc = Bin->Size;
470  } else {
471  Me->MemAlloc = 0;
472  }
473 
474  //
475  // take care here
476  //
477  if( CmView == NULL ) {
478  Me->BinAddress |= HMAP_INPAGEDPOOL;
479  // we don't need to set this - just for debug purposes
480  ASSERT( (Me->CmView = NULL) == NULL );
481  } else {
482  Me->BinAddress |= HMAP_INVIEW;
483  }
484 
485  Offset += HBLOCK_SIZE;
486  }
487 
488  if (Hive->ReadOnly == FALSE) {
489 
490  //
491  // add free cells in the bin to the appropriate free lists
492  //
493  if ( ! HvpEnlistFreeCells(Hive, Bin, BinOffset)) {
494  HvCheckHiveDebug.Hive = Hive;
495  HvCheckHiveDebug.Status = 0xA002;
496  HvCheckHiveDebug.Space = Length;
497  HvCheckHiveDebug.MapPoint = BinOffset;
498  HvCheckHiveDebug.BinPoint = Bin;
499  if( CmDoSelfHeal() ) {
500  Status = STATUS_REGISTRY_RECOVERED;
501  } else {
502  Status = STATUS_REGISTRY_CORRUPT;
503  goto ErrorExit;
504  }
505  }
506 
507  }
508 
509  //
510  // logical consistency check
511  //
512  ASSERT(Offset == (BinOffset + Bin->Size));
513 
514 ErrorExit:
515  return Status;
516 }

519 HvpBuildMap(
520  PHHIVE Hive,
521  PVOID Image
522  )
523 /*++
524 
525 Routine Description:
526 
527  Creates the map for the Stable storage of the hive, and inits
528  the map for the volatile storage.
529 
530  Following fields in hive must already be filled in:
531 
532  Allocate, Free
533 
534  Will initialize Storage structure of HHIVE.
535 
536 Arguments:
537 
538  Hive - Pointer to hive control structure to build map for.
539 
540  Image - pointer to in memory image of the hive
541 
542 Return Value:
543 
544  TRUE - it worked
545  FALSE - either hive is corrupt or no memory for map
546 
547 --*/
548 {
549  PHBIN Bin;
550  ULONG Offset;
551  NTSTATUS Status;
552  ULONG Length;
553 
554 
555 #if DBG
556  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_HIVE,"HvpBuildMap:\n"));
557  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_HIVE,"\tHive=%p",Hive));
558 #endif
559 
560  //
561  // Init the map
562  //
563  Status = HvpInitMap(Hive);
564 
565  if( !NT_SUCCESS(Status) ) {
566  //
567  // just return failure; HvpInitMap took care of cleanup
568  //
569  return Status;
570  }
571 
572  //
573  // Fill in the map
574  //
575  Offset = 0;
576  Bin = (PHBIN)Image;
577  Length = Hive->Storage[Stable].Length;
578 
579  while (Bin < (PHBIN)((PUCHAR)(Image) + Length)) {
580 
581  //
582  // Check the validity of the bin header
583  //
584  if ( (Bin->Size > Length) ||
585  (Bin->Size < HBLOCK_SIZE) ||
586  (Bin->Signature != HBIN_SIGNATURE) ||
587  (Bin->FileOffset != Offset)) {
588  //
589  // Bin is bogus
590  //
591  HvCheckHiveDebug.Hive = Hive;
592  HvCheckHiveDebug.Status = 0xA001;
593  HvCheckHiveDebug.Space = Length;
594  HvCheckHiveDebug.MapPoint = Offset;
595  HvCheckHiveDebug.BinPoint = Bin;
596  //
597  // for the loader.
598  //
599  if( CmDoSelfHeal() ) {
600  //
601  // put the correct signature, fileoffset and binsize in place;
602  // HvEnlistBinInMap will take care of the cells consistency.
603  //
604  Bin->Signature = HBIN_SIGNATURE;
605  Bin->FileOffset = Offset;
606  if ( ((Offset + Bin->Size) > Length) ||
607  (Bin->Size < HBLOCK_SIZE) ||
608  (Bin->Size % HBLOCK_SIZE) ) {
609  Bin->Size = HBLOCK_SIZE;
610  }
611  //
612  // signal back to the caller that we have altered the hive.
613  //
614  CmMarkSelfHeal(Hive);
615  } else {
616  Status = STATUS_REGISTRY_CORRUPT;
617  goto ErrorExit;
618  }
619  }
620 
621  //
622  // enlist this bin
623  //
624  Status = HvpEnlistBinInMap(Hive, Length, Bin, Offset, NULL);
625  //
626  // for the loader.
627  //
628  if( CmDoSelfHeal() && (Status == STATUS_REGISTRY_RECOVERED) ) {
629  CmMarkSelfHeal(Hive);
630  Status = STATUS_SUCCESS;
631  }
632 
633  if( !NT_SUCCESS(Status) ) {
634  goto ErrorExit;
635  }
636 
637  //
638  // the next bin
639  //
640  Offset += Bin->Size;
641 
642  Bin = (PHBIN)((ULONG_PTR)Bin + Bin->Size);
643  }
644 
645  return STATUS_SUCCESS;
646 
647 
648 ErrorExit:
649  //
650  // Clean up the directory table
651  //
652  HvpCleanMap( Hive );
653 
654  return Status;
655 }

657 BOOLEAN
658 HvpEnlistFreeCells(
659  PHHIVE Hive,
660  PHBIN Bin,
661  ULONG BinOffset
662  )
663 /*++
664 
665 Routine Description:
666 
667  Scan through the cells in the bin, locating the free ones.
668  Enlist them in the hive's free list set.
669 
670  N.B. Bin MUST already be mapped when this is called.
671 
672 Arguments:
673 
674  Hive - pointer to hive control structure map is being built for
675 
676  Bin - pointer to bin to enlist cells from
677 
678  BinOffset - offset of Bin in image
679 
680 Return Value:
681 
682  FALSE - registry is corrupt
683 
684  TRUE - it worked
685 
686 --*/
687 {
688  PHCELL p;
689  ULONG celloffset;
690  ULONG size;
691  HCELL_INDEX cellindex;
692  BOOLEAN Result = TRUE;
693 
694  //
695  // Scan all the cells in the bin, total free and allocated, check
696  // for impossible pointers.
697  //
698  celloffset = sizeof(HBIN);
699  p = (PHCELL)((PUCHAR)Bin + sizeof(HBIN));
700 
701  while (p < (PHCELL)((PUCHAR)Bin + Bin->Size)) {
702 
703  //
704  // if free cell, check it out, add it to free list for hive
705  //
706  if (p->Size >= 0) {
707 
708  size = (ULONG)p->Size;
709 
710  if ( (size > Bin->Size) ||
711  ( (PHCELL)(size + (PUCHAR)p) >
712  (PHCELL)((PUCHAR)Bin + Bin->Size) ) ||
713  ((size % HCELL_PAD(Hive)) != 0) ||
714  (size == 0) )
715  {
716  Result = FALSE;
717  if( CmDoSelfHeal() ) {
718  //
719  // self heal mode; enlist the remaining of the bin as free
720  // also zero it out so any references into the tampered area will be
721  // detected and fixed by the logical check later on
722  //
723  p->Size = (LONG)((PUCHAR)((PUCHAR)Bin + Bin->Size) - (PUCHAR)p);
724  RtlZeroMemory((PUCHAR)p + sizeof(ULONG),p->Size - sizeof(ULONG));
725  size = (ULONG)p->Size;
726  CmMarkSelfHeal(Hive);
727  } else {
728  goto Exit;
729  }
730  }
731 
732 
733  //
734  // cell is free, and is not obviously corrupt, add to free list
735  //
736  celloffset = (ULONG)((PUCHAR)p - (PUCHAR)Bin);
737  cellindex = BinOffset + celloffset;
738 
739  //
740  // Enlist this free cell, but do not coalesce with the next free cell
741  // as we haven't gotten that far yet.
742  //
743  HvpEnlistFreeCell(Hive, cellindex, size, Stable, FALSE);
744 
745  } else {
746 
747  size = (ULONG)(p->Size * -1);
748 
749  if ( (size > Bin->Size) ||
750  ( (PHCELL)(size + (PUCHAR)p) >
751  (PHCELL)((PUCHAR)Bin + Bin->Size) ) ||
752  ((size % HCELL_PAD(Hive)) != 0) ||
753  (size == 0) )
754  {
755  Result = FALSE;
756  if( CmDoSelfHeal() ) {
757  //
758  // Self heal mode; we have no other way than to enlist this cell as a free cell
759  //
760  p->Size = (LONG)((PUCHAR)((PUCHAR)Bin + Bin->Size) - (PUCHAR)p);
761  RtlZeroMemory((PUCHAR)p + sizeof(ULONG),p->Size - sizeof(ULONG));
762  size = (ULONG)p->Size;
763 
764  celloffset = (ULONG)((PUCHAR)p - (PUCHAR)Bin);
765  cellindex = BinOffset + celloffset;
766 
767  HvpEnlistFreeCell(Hive, cellindex, size, Stable, FALSE);
768  CmMarkSelfHeal(Hive);
769  } else {
770  goto Exit;
771  }
772  }
773 
774  }
775 
776  ASSERT( ((LONG)size) >= 0);
777  p = (PHCELL)((PUCHAR)p + size);
778  }
779 
780 Exit:
781  return Result;
782 }

788 /*++
789
