179 PREGENTRY WmipAllocRegEntry(
180  PDEVICE_OBJECT DeviceObject,
181  ULONG Flags
182  )
183 /*++
184 
185 Routine Description:
186 
187  Allocate a REGENTRY structure. If successful the RegEntry returns with
188  a ref count of 1.
189 
190  NOTE: This routine assumes that the registration critical section is held
191 
192 Arguments:
193 
194  DeviceObject is the value to fill in the DeviceObject field of the
195  RegEntry.
196 
197 Return Value:
198 
199  pointer to a REGENTRY or NULL if no memory is available
200 
201 --*/
202 {
203  PREGENTRY RegEntry;
204 
205  PAGED_CODE();
206 
207  RegEntry = ExAllocateFromNPagedLookasideList(&WmipRegLookaside);
208 
209  if (RegEntry != NULL)
210  {
211  //
212  // Initialize the RegEntry. Note that the regentry will start out with
213  // a ref count of 1
214  KeInitializeEvent(&RegEntry->Event,
215  SynchronizationEvent,
216  FALSE);
217 
218 
219  RegEntry->Flags = Flags;
220  RegEntry->DeviceObject = DeviceObject;
221  RegEntry->RefCount = 1;
222  RegEntry->IrpCount = 0;
223  RegEntry->PDO = NULL;
224  RegEntry->DataSource = NULL;
225 
226  RegEntry->ProviderId = WmipAllocProviderId(DeviceObject);
227 
228  //
229  // Now place the RegEntry on the in use list
230  InterlockedIncrement(&WmipInUseRegEntryCount);
231 
232  ExInterlockedInsertTailList(&WmipInUseRegEntryHead,
233  &RegEntry->InUseEntryList,
234  &WmipRegistrationSpinLock);
235  }
236  return(RegEntry);
237 }

241  )
242 /*++
243 
244 Routine Description:
245 
246  Remove a reference on a REGENTRY. If the last reference is removed
247  then mark the RegEntry as available and put it on the free list;
248 
249 Arguments:
250 
251  RegEntry is pointer to entry to free
252 
253 Return Value:
254 
255  On checked builds returns TRUE if last ref count was removed from REGENTRY
256  and it was placed back on free list.
257 
258 --*/

260  BOOLEAN Freed;
261  ULONG ProviderId;
262 
263  PAGED_CODE();

265  WmipEnterSMCritSection();

267  if (Freed)
268  {
269  //
270  // We should only ever free this after the driver has released it
271  WmipAssert(RegEntry->Flags & REGENTRY_FLAG_RUNDOWN);

282

286  "WMI: RegEntry %p removed from list\n", \
287  RegEntry, __FILE__, __LINE__)); \
288  ProviderId = RegEntry->ProviderId;
289  ExInterlockedRemoveHeadList(RegEntry->InUseEntryList.Blink,
290  &WmipRegistrationSpinLock);
291  InterlockedDecrement(&WmipInUseRegEntryCount);
292  WmipLeaveSMCritSection();
293 
294  WmipRemoveDS(RegEntry);
295 
296  ExFreeToNPagedLookasideList(&WmipRegLookaside,
297  RegEntry);
298  } else {
299  WmipLeaveSMCritSection();
300  }

345 NTSTATUS WmipRegisterDevice(
346  PDEVICE_OBJECT DeviceObject,
347  ULONG RegistrationFlag
348  )
349 /*++
350 
351 Routine Description:
352 
353  Remember information about a new device being registered and
354  then go and get the registration information.
355 
356 Arguments:
357 
358  DeviceObject is a pointer to the device object being registered
359  or the callback entry point
360 
361  RegistrationFlag is either WMIREG_FLAG_CALLBACK if DeviceObject is
362  a callback pointer, or WMIREG_FLAG_TRACE_PROVIDER is DeviceObject
363  can also generate event traces.
364 
365 Return Value:
366 
367  NT status code
368 
369 --*/
370 {
371  PREGENTRY RegEntry;
372  NTSTATUS Status;
373  ULONG Flags;
374  ULONG IsCallback = RegistrationFlag & WMIREG_FLAG_CALLBACK;
375  BOOLEAN UpdateDeviceStackSize = FALSE;
376 
377  PAGED_CODE();
378 
379  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
380  DPFLTR_REGISTRATION_LEVEL,
381  "WMI: Registering device %p flags %x\n",
382  DeviceObject,
383  RegistrationFlag
384  ));
385 
386  WmipEnterSMCritSection();
387  RegEntry = WmipFindRegEntryByDevice(DeviceObject, FALSE);
388  if (RegEntry == NULL)
389  {
390  if (! IsCallback)
391  {
392  //
393  // Data providers that register with a device object and not a
394  // callback have their device object referenced so that it will
395  // stick around while WMI needs it. This reference is removed
396  // when the device unregisters with WMI and all WMI irps are
397  // completed.
398  Status = ObReferenceObjectByPointer(DeviceObject,
399  0,
400  NULL, /* *IoDeviceObjectType */
401  KernelMode);
402  if (NT_SUCCESS(Status))
403  {
404  UpdateDeviceStackSize = TRUE;
405  }
406  } else {
407  //
408  // No reference counting is done for callbacks. It is the data
409  // provider's responsibility to synchronize any unloading and
410  // deregistration issues.
411  Status = STATUS_SUCCESS;
412  }
413 
414  if (NT_SUCCESS(Status))
415  {
416  //
417  // Allocate, initialize and place on active list
418  Flags = REGENTRY_FLAG_NEWREGINFO | REGENTRY_FLAG_INUSE |
419  (IsCallback ? REGENTRY_FLAG_CALLBACK : 0);
420 
421  if (RegistrationFlag & WMIREG_FLAG_TRACE_PROVIDER) {
422  Flags |= REGENTRY_FLAG_TRACED;
423  Flags |= (RegistrationFlag & WMIREG_FLAG_TRACE_NOTIFY_MASK);
424  }
425 
426  Flags |= REGENTRY_FLAG_REG_IN_PROGRESS;
427  RegEntry = WmipAllocRegEntry(DeviceObject, Flags);
428 
429  if (RegEntry != NULL)
430  {
431  // We need to take an extra ref count before
432  // releasing the critical section. One class of drivers
433  // (kmixer) will register and unregister multiple times
434  // in different threads and this can lead to a race where
435  // the regentry is removed from the list twice
436  //
437  WmipReferenceRegEntry(RegEntry);
438  WmipLeaveSMCritSection();
439 
440  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
441  DPFLTR_REGISTRATION_LEVEL,
442  "WMI: Register allocated REGENTRY %p for %p\n",
443  RegEntry,
444  DeviceObject
445  ));
446  //
447  // Go and get registration information from the driver
448  //
449  if (IsCallback)
450  {
451  //
452  // We can perform registration callback now since
453  // we do not need to worry about deadlocks
454  //
455  Status = WmipRegisterDS(RegEntry);
456  if (NT_SUCCESS(Status))
457  {
458  //
459  // Mark regentry as fully registered so now we can start
460  // accepting unregister calls
461  //
462  RegEntry->Flags &= ~REGENTRY_FLAG_REG_IN_PROGRESS;
463  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
464  DPFLTR_REGISTRATION_LEVEL,
465  "WMI: WmipRegisterDS(%p) succeeded for callback %p\n",
466  RegEntry, RegEntry->DeviceObject));
467  } else {
468  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
469  DPFLTR_REGISTRATION_LEVEL,
470  "WMI: WmipRegisterDS(%p) failed %x for device %p\n",
471  RegEntry, Status, RegEntry->DeviceObject));
472 
473  //
474  // Remove ref so regentry goes away
475  //
476  WmipUnreferenceRegEntry(RegEntry);
477  }
478 
479  } else {
480  //
481  // We need to send the registration irp from within
482  // a work item and not in the context of this
483  // routine. This is because some drivers will not
484  // process irps while in the StartDevice/AddDevice
485  // context, so we'd get deadlock
486  //
487  Status = WmipQueueRegWork(RegisterSingleDriver, RegEntry);
488  if (! NT_SUCCESS(Status))
489  {
490  //
491  // If failed then remove regentry from list
492  //
493  RegEntry->Flags |= (REGENTRY_FLAG_RUNDOWN |
494  REGENTRY_FLAG_NOT_ACCEPTING_IRPS);
495  WmipUnreferenceRegEntry(RegEntry);
496  }
497  }
498 
499  //
500  // Remove extra regentry ref count taken above
501  //
502  WmipUnreferenceRegEntry(RegEntry);
503 
504  } else {
505  WmipLeaveSMCritSection();
506  Status = STATUS_INSUFFICIENT_RESOURCES;
507  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
508  DPFLTR_REGISTRATION_LEVEL,
509  "WMI: Register could not alloc REGENTRY for %p\n",
510  DeviceObject
511  ));
512  }
513  } else {
514  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
515  DPFLTR_REGISTRATION_LEVEL,
516  "WMI: Register could not ObRef %p status %x\n",
517  DeviceObject,
518  Status
519  ));
520  WmipLeaveSMCritSection();
521  }
522  } else {
523  //
524  // A device object may only register once
525  WmipLeaveSMCritSection();
526  Status = STATUS_OBJECT_NAME_EXISTS;
527  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
528  DPFLTR_REGISTRATION_LEVEL,
529  "WMI: Device Object %x attempting to register twice\n",
530  DeviceObject));
531  WmipUnreferenceRegEntry(RegEntry);
532  }
533 
534  if (UpdateDeviceStackSize)
535  {
536  //
537  // Since WMI will be forwarding irps to this device the WMI irp
538  // stack size must be at least one larger than that of the device
539  WmipUpdateDeviceStackSize(
540  (CCHAR)(DeviceObject->StackSize+1));
541  }
542 
543  return(Status);
544 }

547  PDEVICE_OBJECT DeviceObject
548  )
549 /*++
550 
551 Routine Description:
552 
553  Remove registration entry for a device
554 
555 Arguments:
556 
557  DeviceObject is a pointer to the device object being deregistered
558 
559 
560 Return Value:
561 
562  NT status code
563 
564 --*/
565 {
566  NTSTATUS Status;
567  PREGENTRY RegEntry;
568  ULONG Flags;
569 
570  PAGED_CODE();
571 
572  WmipEnterSMCritSection();
573  RegEntry = WmipFindRegEntryByDevice(DeviceObject, FALSE);
574  if (RegEntry != NULL)
575  {
576 
577  //
578  // Mark the regentry as invalid so that no more irps are sent to the
579  // device and the event will set when the last irp completes.
580  Flags = InterlockedExchange(&RegEntry->Flags,
581  (REGENTRY_FLAG_RUNDOWN |
582  REGENTRY_FLAG_NOT_ACCEPTING_IRPS) );
583 
584  //
585  // Once the regentry is marked as RUNDOWN then it will not be found
586  // later so it is safe to release the lock.
587  WmipLeaveSMCritSection();
588  WmipUnreferenceRegEntry(RegEntry);
589 
590  //
591  // Now if there are any outstanding irps for the device then we need
592  // to wait here until they complete.
593  WmipWaitForIrpCompletion(RegEntry);
594  if (! (Flags & REGENTRY_FLAG_CALLBACK))
595  {
596  ObDereferenceObject(DeviceObject);
597  }
598 
599  //
600  // Release last reference to REGENTRY after KMREGINFO is set
601  WmipUnreferenceRegEntry(RegEntry);
602 
603  Status = STATUS_SUCCESS;
604  } else {
605  WmipLeaveSMCritSection();
606  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
607  DPFLTR_REGISTRATION_LEVEL,
608  "WMI: WmipDeregisterDevice called with invalid Device Object %x\n",
609  DeviceObject));
610  Status = STATUS_INVALID_PARAMETER;
611  }
612 
613 
614  return(Status);

896  data provider and convert any PDO instance name references to a
897  static instance name reference.
898 
899 Arguments:
900 
901  Irp points at the registration query irp
902 
903  MaxBufferSize is the maximum size that will fit into buffer
904 
905  RegEntry is registration structure for device being registered

914  PUCHAR FreeSpacePtr;

927  BOOLEAN AllowPnPIdMap = TRUE;

929  PUCHAR FreeSpacePadPtr;
930  ULONG PadSpace, FreeSpaceOffset;

937  FreeSpacePtr = NULL;

940  MaxInstanceNames = 0;

951  // any default mapping done here.

953  &WmipDataProviderPnpidGuid)) ||
954  (IsEqualGUID(&WmiRegGuid->Guid,
955  &WmipDataProviderPnPIdInstanceNamesGuid)))
956  {

961  // used for PnPID mapping then make sure to deref it
962  //

964  {

967  }
968  }
969

974  // device instance name

976  {
977  //
978  // Determine where any free space is in output buffer by
979  // figuring out where the last WmiRegInfo ends
980  WmiRegInfo2 = (PWMIREGINFO)WmiRegInfoBase;
981  while (WmiRegInfo2->NextWmiRegInfo != 0)
982  {
983  WmiRegInfo2 = (PWMIREGINFO)((PUCHAR)WmiRegInfo2 +
984  WmiRegInfo2->NextWmiRegInfo);
985  }

987  ((WmiRegInfo2->BufferSize + 1) & 0xfffffffe);
988  FreeSpaceLeft = MaxBufferSize - (ULONG)(FreeSpacePtr - WmiRegInfoBase);
989 
990  }
991

993  // Keep track of the max number of instances for the PDO name
994  MaxInstanceNames = MaxInstanceNames < WmiRegGuid->InstanceCount ?
995  WmiRegGuid->InstanceCount :
996  MaxInstanceNames;

1004  WmiRegGuid->BaseNameOffset = LastBaseNameOffset;
1005  } else {
1006

1008  if (NT_SUCCESS(Status))
1009  {

1011  ((PnPIdPDO == NULL) || (PnPIdPDO == PDO)))
1012  {
1013  if (PnPIdPDO == NULL)
1014  {

1016  ObReferenceObject(PnPIdPDO);
1017  }
1018  } else {
1019  //
1020  // If the PDO value changes then we don't
1021  // do any instance name stuff. In this case
1022  // make sure we remove any ref on the PDO
1023  //
1024  AllowPnPIdMap = FALSE;
1025 
1026  if (PnPIdPDO != NULL)
1027  {
1028  ObDereferenceObject(PnPIdPDO);
1029  PnPIdPDO = NULL;
1030  }
1031  }

1038  (InstancePathLength > FreeSpaceLeft))
1039  {
1040  WmiRegInfoTooSmall = TRUE;
1041  } else {

1043

1045  LastPDO = PDO;
1046 
1047  WmiRegGuid->BaseNameOffset = LastBaseNameOffset;

1050  FreeSpacePtr += sizeof(USHORT);

1052  InstancePath.Buffer,
1053  InstancePath.Length);
1054  FreeSpacePtr += InstancePath.Length;

1056  FreeSpacePtr += sizeof(WCHAR);
1057  FreeSpaceLeft -= InstancePathLength;
1058  }
1059  }
1060 
1061  if (NT_SUCCESS(Status))
1062  {
1063  RtlFreeUnicodeString(&InstancePath);
1064  }
1065  }

1068  {
1069  ObDereferenceObject(PDO);
1070  }
1071  }
1072  }
1073  LastPDO = NULL;
1074  NextWmiRegInfo = WmiRegInfo->NextWmiRegInfo;
1075  WmiRegInfo = (PWMIREGINFO)((PUCHAR)WmiRegInfo + NextWmiRegInfo);
1076 
1077  } while (NextWmiRegInfo != 0);
1078 
1079  //
1080  // If we can do automatic support for device information guid so add
1081  // registration for this guid to the registration information
1082  if (AllowPnPIdMap && (PnPIdPDO != NULL))
1083  {
1084  Status = WmipPDOToDeviceInstanceName(PDO, &InstancePath);
1085  if (NT_SUCCESS(Status))
1086  {
1087  //
1088  // Pad so that new WmiRegInfo starts on 8 byte boundary and
1089  // adjust free buffer size
1090  FreeSpacePadPtr = (PUCHAR)(((ULONG_PTR)FreeSpacePtr+7) & ~7);
1091  PadSpace = (ULONG)(FreeSpacePadPtr - FreeSpacePtr);
1092  FreeSpaceLeft -= PadSpace;
1093  FreeSpacePtr = FreeSpacePadPtr;
1094 
1095  //
1096  // Figure out how much space we will need to include extra guid
1097  InstancePathLength = InstancePath.Length +
1098  sizeof(USHORT) + sizeof(WCHAR);
1099 
1100  ExtraRoom = 2 * (InstancePathLength + sizeof(WMIREGGUID)) +
1101  sizeof(WMIREGINFO);
1102 
1103  SizeNeeded += ExtraRoom + PadSpace;
1104 
1105  if ((WmiRegInfoTooSmall) ||
1106  (ExtraRoom > FreeSpaceLeft))
1107  {
1108  WmiRegInfoTooSmall = TRUE;
1109  } else {
1110  if (RegEntry->PDO == NULL)
1111  {
1112  //
1113  // If we haven't already established a PDO for this
1114  // data provider then remember PDO and count of
1115  // instance names for this device
1116  // so we can get device properties
1117  //
1118  ObReferenceObject(PnPIdPDO);
1119  RegEntry->PDO = PnPIdPDO;
1120  RegEntry->MaxInstanceNames = MaxInstanceNames;
1121 
1122  WmiRegInfo->NextWmiRegInfo = (ULONG)(FreeSpacePtr -
1123  (PUCHAR)WmiRegInfo);
1124 
1125  WmiRegInfo = (PWMIREGINFO)FreeSpacePtr;
1126  FreeSpaceOffset = sizeof(WMIREGINFO) + 2*sizeof(WMIREGGUID);
1127  FreeSpacePtr += FreeSpaceOffset;
1128 
1129  RtlZeroMemory(WmiRegInfo, FreeSpaceOffset);
1130  WmiRegInfo->BufferSize = ExtraRoom;
1131  WmiRegInfo->GuidCount = 2;
1132 
1133  WmiRegGuid = &WmiRegInfo->WmiRegGuid[0];
1134  WmiRegGuid->Flags = WMIREG_FLAG_INSTANCE_BASENAME |
1135  WMIREG_FLAG_INSTANCE_PDO;
1136  WmiRegGuid->InstanceCount = MaxInstanceNames;
1137  WmiRegGuid->Guid = WmipDataProviderPnpidGuid;
1138  WmiRegGuid->BaseNameOffset = FreeSpaceOffset;
1139 
1140  (*(PUSHORT)FreeSpacePtr) = InstancePath.Length + sizeof(WCHAR);
1141  FreeSpacePtr += sizeof(USHORT);
1142  RtlCopyMemory(FreeSpacePtr,
1143  InstancePath.Buffer,
1144  InstancePath.Length);
1145  FreeSpacePtr += InstancePath.Length;
1146  *((PWCHAR)FreeSpacePtr) = L'_';
1147  FreeSpacePtr += sizeof(WCHAR);
1148  FreeSpaceOffset += sizeof(USHORT) +
1149  InstancePath.Length + sizeof(WCHAR);
1150 
1151 
1152  WmiRegGuid = &WmiRegInfo->WmiRegGuid[1];
1153  WmiRegGuid->Flags = WMIREG_FLAG_INSTANCE_LIST;
1154  WmiRegGuid->InstanceCount = 1;
1155  WmiRegGuid->Guid = WmipDataProviderPnPIdInstanceNamesGuid;
1156  WmiRegGuid->BaseNameOffset = FreeSpaceOffset;
1157 
1158  (*(PUSHORT)FreeSpacePtr) = InstancePath.Length;
1159  FreeSpacePtr += sizeof(USHORT);
1160  RtlCopyMemory(FreeSpacePtr,
1161  InstancePath.Buffer,
1162  InstancePath.Length);
1163  FreeSpacePtr += InstancePath.Length;
1164  }
1165 
1166  }
1167 
1168  RtlFreeUnicodeString(&InstancePath);
1169  }
1170 
1171  ObDereferenceObject(PnPIdPDO);
1172  } else {
1173  WmipAssert(PnPIdPDO == NULL);
1174  }
1175 
1176  if (WmiRegInfoTooSmall)
1177  {
1178  *((PULONG)Buffer) = SizeNeeded;
1179  Irp->IoStatus.Information = sizeof(ULONG);
1180  } else {
1181  WmiRegInfo = (PWMIREGINFO)WmiRegInfoBase;
1182  WmiRegInfo->BufferSize = SizeNeeded;
1183  Irp->IoStatus.Information = SizeNeeded;
1184  }
1185 }

1240  sizes and offsets are not out of bounds of the buffer. It will register
1241  the guids for each one. Note that if at least one of the WMIREGINFOs does
1242  register successfully then STATUS_SUCCESS is returned, but all
1243  WMIREGINFOs following the bad one are not registered.
1244 
1245 Arguments:
1246 
1247  RegEntry is the RegEntry for the device or user mode object
1248 
1249  WmiRegInfo is the registration information to register
1250 
1251  BufferSize is the size of WmiRegInfo in bytes
1252 
1253  RequestObject is the request object associated with the UM provider.
1254  If this is NULL then the registration is for a driver
1255

1259 
1260  STATUS_SUCCESS or an error code
1261 
1262 ---*/

1265  NTSTATUS Status, FinalStatus;
1266  PWCHAR RegPath, ResourceName;
1267  ULONG GuidBufferSize;
1268 
1269  PAGED_CODE();
1270

1272

1274  //
1275  // First we validate that the WMIREGINFO looks correct
1276  //

1278  {
1279  //
1280  // BufferSize specified in WmiRegInfo is beyond bounds of buffer
1281  //
1282  Status = STATUS_INVALID_PARAMETER;
1283  break;
1284  }
1285

1287  // Validate registry path string
1288  //
1289  Status = WmipValidateWmiRegInfoString(WmiRegInfo,

1295  break;
1296  }
1297

1300  //

1302  BufferSize,

1304  &ResourceName);

1307  break;
1308  }

1312  // buffer. Note that WmipAddDataSource verifies that the instance
1313  // names within each guid is within bounds.
1314  //

1316  WmiRegInfo->GuidCount * sizeof(WMIREGGUID);

1320  break;

1335  BufferSize);

1350  ResourceName,
1351  RequestObject,
1352  IsUserMode);
1353  }

1356  {
1357  //
1358  // if at least one of the registrations was added
1359  // successfully then the final status is success
1360  //

1362 
1363  } else {
1364  WmipDebugPrintEx((DPFLTR_WMICORE_ID,
