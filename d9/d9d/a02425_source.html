81  OUT PKGDTENTRY GdtEntry,
82  IN ULONG Base,
83  IN ULONG Limit,
84  IN USHORT Type
85  )

87 /*++
88 
89 Routine Description:
90 
91  This function initializes a GDT entry for abios specific code. Base,
92  Limit, and Type (code, data) are set according to parameters. All other
93  fields of the entry are set to match standard system values.
94 
95  N.B. The BIG and GRANULARITY are always set to 0.
96 
97 Arguments:
98 
99  GdtEntry - GDT descriptor to be filled in.
100 
101  Base - Linear address of the first byte mapped by the selector.
102 
103  Limit - Size of the selector in BYTE.
104 
105  Type - Code or Data. All code selectors are marked readable,
106  all data selectors are marked writeable.
107 
108 Return Value:
109 
110  Pointer to the GDT entry.
111 
112 --*/
113 
114 {
115  GdtEntry->LimitLow = (USHORT)(Limit & 0xffff);
116  GdtEntry->BaseLow = (USHORT)(Base & 0xffff);
117  GdtEntry->HighWord.Bytes.BaseMid = (UCHAR)((Base & 0xff0000) >> 16);
118  GdtEntry->HighWord.Bits.Type = Type;
119  GdtEntry->HighWord.Bits.Dpl = 0;
120  GdtEntry->HighWord.Bits.Pres = 1;
121  GdtEntry->HighWord.Bits.LimitHi = (Limit & 0xf0000) >> 16;
122  GdtEntry->HighWord.Bits.Sys = 0;
123  GdtEntry->HighWord.Bits.Reserved_0 = 0;
124  GdtEntry->HighWord.Bits.Default_Big = 0;
125  GdtEntry->HighWord.Bits.Granularity = 0;
126  GdtEntry->HighWord.Bytes.BaseHi = (UCHAR)((Base & 0xff000000) >> 24);
127 }

166 NTSTATUS
167 KeI386GetLid(
168  IN USHORT DeviceId,
169  IN USHORT RelativeLid,
170  IN BOOLEAN SharedLid,
171  IN PDRIVER_OBJECT DriverObject,
172  OUT PUSHORT LogicalId
173  )
174 
175 /*++
176 
177 Routine Description:
178 
179  This function searches Device Blocks and Common Data Area for the
180  Logical Id matching the specified Device Id.
181 
182  N.B. To speed the search, this routine ASSUMES that
183  the LIDs with the same Device ID always appear consecutively in the
184  Common Data Area. IBM ABIOS doc does not explicitly specify this.
185  But from the way ABIOS initializes Device Block and Function Transfer
186  Table, I think the assumption is true.
187 
188 Arguments:
189 
190  DeviceId - Desired Device Id.
191 
192  RelativeLid - Specifies the Nth logical Id for this device Id. A value
193  of 0 indicates the first available Lid.
194 
195  SharedLid - A boolean value indicates if it is a shared or exclusively
196  owned logical Id.
197 
198  DriverObject - Supplies a 32-bit flat pointer of the requesting device
199  driver's driver object. The DriverObject is used to establish
200  the ownership of the desired LID.
201 
202  LogicalId - A pointer to a variable which will receive the Lid.
203 
204 Return Value:
205 
206  STATUS_SUCCESS - If the requested LID is available.
207 
208  STATUS_ABIOS_NOT_PRESENT - If there is no ABIOS support in the system.
209 
210  STATUS_ABIOS_LID_NOT_EXIST - If the specified LID does not exist.
211 
212  STATUS_ABIOS_LID_ALREADY_OWNED - If the caller requests an exclusively
213  owned LID.
214 
215 --*/
216 
217 {
218  PKDB_FTT_SECTION CdaPointer;
219  PKDEVICE_BLOCK DeviceBlock;
220  USHORT Lid, RelativeLidCount = 1;
221  ULONG Owner;
222  USHORT Increment;
223  KIRQL OldIrql;
224  NTSTATUS Status;
225 
226  if (!KiAbiosPresent) {
227  return STATUS_ABIOS_NOT_PRESENT;
228  }
229 
230  if (SharedLid) {
231  Owner = LID_NO_SPECIFIC_OWNER;
232  Increment = 1;
233  } else {
234  Owner = (ULONG)DriverObject;
235  Increment = 0;
236  }
237 
238  //
239  // If the Logical Id Table hasn't been created yet, create it now.
240  //
241  if (KiLogicalIdTable==NULL) {
242  KiLogicalIdTable = ExAllocatePoolWithTag(NonPagedPool,
243  NUMBER_LID_TABLE_ENTRIES *
244  sizeof(KLID_TABLE_ENTRY),
245  ' eK');
246  if (KiLogicalIdTable == NULL) {
247  return(STATUS_NO_MEMORY);
248  }
249  RtlZeroMemory(KiLogicalIdTable, NUMBER_LID_TABLE_ENTRIES*sizeof(KLID_TABLE_ENTRY));
250  }
251 
252  //
253  // For each Lid defined in Common Data Area, we check if it has non
254  // empty device block and function transfer table. If yes, we proceed
255  // to check the device id. Otherwise, we skip the Lid.
256  //
257 
258  CdaPointer = (PKDB_FTT_SECTION)KiCommonDataArea + 2;
259  Status = STATUS_ABIOS_LID_NOT_EXIST;
260 
261  ExAcquireSpinLock(&KiAbiosLidTableLock, &OldIrql);
262 
263  for (Lid = 2; Lid < KiCommonDataArea->NumberLids; Lid++) {
264  if (CdaPointer->DeviceBlock.Selector != 0 &&
265  CdaPointer->FunctionTransferTable.Selector != 0) {
266 
267  DeviceBlock = (PKDEVICE_BLOCK)(KiI386SelectorBase(
268  CdaPointer->DeviceBlock.Selector)
269  + (CdaPointer->DeviceBlock.Offset));
270  if (DeviceBlock->DeviceId == DeviceId) {
271  if (RelativeLid == RelativeLidCount || RelativeLid == 0) {
272  if (KiLogicalIdTable[Lid].Owner == 0L) {
273  KiLogicalIdTable[Lid].Owner = Owner;
274  KiLogicalIdTable[Lid].OwnerCount += Increment;
275  *LogicalId = Lid;
276  Status = STATUS_SUCCESS;
277  } else if (KiLogicalIdTable[Lid].Owner == LID_NO_SPECIFIC_OWNER) {
278  if (SharedLid) {
279  *LogicalId = Lid;
280  KiLogicalIdTable[Lid].OwnerCount += Increment;
281  Status = STATUS_SUCCESS;
282  } else {
283  Status = STATUS_ABIOS_LID_ALREADY_OWNED;
284  }
285  } else if (KiLogicalIdTable[Lid].Owner == (ULONG)DriverObject) {
286  *LogicalId = Lid;
287  Status = STATUS_SUCCESS;
288  } else if (RelativeLid != 0) {
289  Status = STATUS_ABIOS_LID_ALREADY_OWNED;
290  }
291  break;
292  } else {
293  RelativeLidCount++;
294  }
295  }
296  }
297  CdaPointer++;
298  }
299 
300  ExReleaseSpinLock(&KiAbiosLidTableLock, OldIrql);
301  return Status;
302 }

364 KeI386AbiosCall(

366  IN PDRIVER_OBJECT DriverObject,
367  IN PUCHAR RequestBlock,
368  IN USHORT EntryPoint
369  )
370 
371 /*++
372 
373 Routine Description:
374 
375  This function calls an ABIOS service routine on behave of device driver
376  using Operating System Transfer Convention.
377 
378 Arguments:
379 
380  LogicalId - Logical Id for the call.
381 
382  DriverObject - Supplies a 32-bit flat pointer of the requesting device
383  driver's driver object. The DriverObject is used to verify
384  the ownership of the desired LID.
385 
386  RequestBlock - A 16:16 (selector:offset) pointer to the request block.
387 
388  EntryPoint - Specifies which ABIOS entry point:
389 
390  0 - Start Routine
391  1 - Interrupt Routine
392  2 - Timeout Routine
393 
394 Return Value:
395 
396  STATUS_SUCCESS - If no error.
397 
398  STATUS_ABIOS_NOT_PRESENT - If there is no ABIOS support in the system.
399 
400  STATUS_ABIOS_INVALID_COMMAND - if the specified entry point is not supported.
401 
402  STATUS_ABIOS_INVALID_LID - If the Lid specified is invalid.
403 
404  STATUS_ABIOS_NOT_LID_OWNER - If the caller does not own this Lid.
405 
406  (Note that the request specific ABIOS returned code is in RequestBlock.)
407 
408 --*/
409 
410 {
411 
412  KABIOS_POINTER FuncTransferTable;
413  KABIOS_POINTER DeviceBlock;
414  KABIOS_POINTER AbiosFunction;
415  PKFUNCTION_TRANSFER_TABLE FttPointer;
416 
417  if (!KiAbiosPresent) {
418  return STATUS_ABIOS_NOT_PRESENT;
419  }
420 
421  if (LogicalId >= KiCommonDataArea->NumberLids) {
422  return STATUS_ABIOS_INVALID_LID;
423  } else if (KiLogicalIdTable[LogicalId].Owner != (ULONG)DriverObject &&
424  KiLogicalIdTable[LogicalId].Owner != LID_NO_SPECIFIC_OWNER) {
425  return STATUS_ABIOS_NOT_LID_OWNER;
426  } else if (EntryPoint > 2) {
427  return STATUS_ABIOS_INVALID_COMMAND;
428  }
429 
430  FuncTransferTable = ((PKDB_FTT_SECTION)KiCommonDataArea + LogicalId)->
431  FunctionTransferTable;
432  DeviceBlock = ((PKDB_FTT_SECTION)KiCommonDataArea + LogicalId)->DeviceBlock;
433  FttPointer = (PKFUNCTION_TRANSFER_TABLE)(KiI386SelectorBase(FuncTransferTable.Selector) +
434  (ULONG)FuncTransferTable.Offset);
435  AbiosFunction = FttPointer->CommonRoutine[EntryPoint];
436  KiI386CallAbios(AbiosFunction,
437  DeviceBlock,
438  FuncTransferTable,
439  *(PKABIOS_POINTER)&RequestBlock
440  );
441 
442  return STATUS_SUCCESS;
443 }

446 KeI386AllocateGdtSelectors(
447  OUT PUSHORT SelectorArray,
448  IN USHORT NumberOfSelectors
449  )
450 
451 /*++
452 
453 Routine Description:
454 
455  This function allocates a set of GDT selectors for a device driver to use.
456  Usually this allocation is performed at device driver initialization time
457  to reserve the selectors for later use.
458 
459 Arguments:
460 
461  SelectorArray - Supplies a pointer to an array of USHORT to be filled
462  in with the GDT selectors allocated.
463 
464  NumberOfSelectors - Specifies the number of selectors to be allocated.
465 
466 Return Value:
467 
468  STATUS_SUCCESS - If the requested selectors are allocated.
469 
470  STATUS_ABIOS_SELECTOR_NOT_AVAILABLE - if system can not allocate the number
471  of selectors requested.
472 
473 --*/
474 
475 {
476  PKFREE_GDT_ENTRY GdtEntry;
477  KIRQL OldIrql;
478 
479  if (KiNumberFreeSelectors >= NumberOfSelectors) {
480  ExAcquireSpinLock(&KiAbiosGdtLock, &OldIrql);
481 
482  //
483  // The Free Gdt link list is maintained on Processor 0's GDT ONLY.
484  // Because the 'selector' is an offset to the beginning of GDT and
485  // it should be the same across all the processors.
486  //
487 
488  KiNumberFreeSelectors = KiNumberFreeSelectors - NumberOfSelectors;
489  GdtEntry = KiFreeGdtListHead;
490  while (NumberOfSelectors != 0) {
491  *SelectorArray++ = (USHORT)((ULONG)GdtEntry - KiAbiosGdt[0]);
492  GdtEntry = GdtEntry->Flink;
493  NumberOfSelectors--;
494  }
495  KiFreeGdtListHead = GdtEntry;
496  ExReleaseSpinLock(&KiAbiosGdtLock, OldIrql);
497  return STATUS_SUCCESS;
498  } else {
499  return STATUS_ABIOS_SELECTOR_NOT_AVAILABLE;
500  }
501 }

554 NTSTATUS
555 KeI386FlatToGdtSelector(
556  IN ULONG SelectorBase,
557  IN USHORT Length,
558  IN USHORT Selector
559  )
560 
561 /*++
562 
563 Routine Description:
564 
565  This function converts a 32-bit flat address to a GDT selector-offset
566  pair. The segment set up is always 16-bit ring 0 data segment.
567 
568 Arguments:
569 
570  SelectorBase - Supplies 32 bit flat address to be set as the base address
571  of the desired selector.
572 
573  Length - Supplies the Length of the segment. The Length is a 16 bit value
574  and zero means 64KB.
575 
576  Selector - Supplies the selector to be set up.
577 
578 Return Value:
579 
580  STATUS_SUCCESS - If the requested LID is released.
581 
582  STATUS_ABIOS_NOT_PRESENT - If there is no ABIOS support in the system.
583 
584  STATUS_ABIOS_INVALID_SELECTOR - If the selector supplied is invalid.
585 
586 
587 --*/
588 
589 {
590  PKGDTENTRY GdtEntry, GdtEntry1;
591  KIRQL OldIrql;
592  ULONG i;
593 
594  if (!KiAbiosPresent) {
595  return STATUS_ABIOS_NOT_PRESENT;
596  }
597  if (Selector < RESERVED_GDT_ENTRIES * sizeof(KGDTENTRY)) {
598  return STATUS_ABIOS_INVALID_SELECTOR;
599  } else {
600  ExAcquireSpinLock(&KiAbiosGdtLock, &OldIrql);
601  GdtEntry = (PKGDTENTRY)(KiAbiosGdt[0] + Selector);
602  GdtEntry->LimitLow = (USHORT)(Length - 1);
603  GdtEntry->BaseLow = LOWWORD(SelectorBase);
604  GdtEntry->HighWord.Bytes.BaseMid = LOWBYTE(HIGHWORD(SelectorBase));
605  GdtEntry->HighWord.Bytes.BaseHi = HIGHBYTE(HIGHWORD(SelectorBase));
606  GdtEntry->HighWord.Bits.Pres = 1;
607  GdtEntry->HighWord.Bits.Type = TYPE_DATA;
608  GdtEntry->HighWord.Bits.Dpl = DPL_SYSTEM;
609  for (i = 1; i < (ULONG)KeNumberProcessors; i++) {
610  GdtEntry1 = (PKGDTENTRY)(KiAbiosGdt[i] + Selector);
611  *GdtEntry1 = *GdtEntry;
612  }
613  ExReleaseSpinLock(&KiAbiosGdtLock, OldIrql);
614  return STATUS_SUCCESS;
615  }

655 VOID
656 KiInitializeAbios (
657  IN UCHAR Processor
658  )
659 
660 /*++
661 
662 Routine Description:
663 
664  This function initializes gdt free list and sets up selector for
665  KiI386AbiosCall (16-bit code).
666 
667 Arguments:
668 
669  Processor - the processor who performs the initialization.
670 
671 Return Value:
672 
673  None.
674 
675 --*/
676 
677 {
678 
679  ULONG GdtLength;
680  PKGDTENTRY AliasGdtSelectorEntry;
681  PKFREE_GDT_ENTRY EndOfGdt;
682 
683  //
684  // First check if abios is recognized by osloader.
685  //
686 
687  KiCommonDataArea = KeLoaderBlock->u.I386.CommonDataArea;
688 
689  //
690  // NOTE For now we want to disable ABIOS support on MP.
691  //
692 
693  if (KiCommonDataArea == NULL || Processor != 0) {
694  KiAbiosPresent = FALSE;
695  } else {
696  KiAbiosPresent = TRUE;
697  }
698 
699  //
700  // Initialize the spinlocks for accessing GDTs and Lid Table.
701  //
702 
703  KeInitializeSpinLock( &KiAbiosGdtLock );
704  KeInitializeSpinLock( &KiAbiosLidTableLock );
705 
706  //
707  // Determine the starting and ending addresses of GDT.
708  //
709 
710  KiAbiosGdt[Processor] = KiAbiosGetGdt();
711 
712  AliasGdtSelectorEntry = (PKGDTENTRY)(KiAbiosGetGdt() + KGDT_GDT_ALIAS);
713  GdtLength = 1 + (ULONG)(AliasGdtSelectorEntry->LimitLow) +
714  (ULONG)(AliasGdtSelectorEntry->HighWord.Bits.LimitHi << 16);
715  EndOfGdt = (PKFREE_GDT_ENTRY)(KiAbiosGetGdt() + GdtLength);
716 
717  //
718  // Prepare selector for 16 bit stack segment
719  //
720 
721  KiStack16GdtEntry = KiAbiosGetGdt() + KGDT_STACK16;
722 
723  KiInitializeAbiosGdtEntry(
724  (PKGDTENTRY)KiStack16GdtEntry,
725  0L,
726  0xffff,
727  TYPE_DATA
728  );
729 
730  //
731  // Establish the addressability of Common Data Area selector.
732  //
733 
734  KiInitializeAbiosGdtEntry(
735  (PKGDTENTRY)(KiAbiosGetGdt() + KGDT_CDA16),
736  (ULONG)KiCommonDataArea,
737  0xffff,
738  TYPE_DATA
739  );
740 
741  //
742  // Set up 16-bit code selector for KiI386AbiosCall
743  //
744 
745  KiInitializeAbiosGdtEntry(
746  (PKGDTENTRY)(KiAbiosGetGdt() + KGDT_CODE16),
747  (ULONG)&KiI386CallAbios,
748  (ULONG)&KiEndOfCode16 - (ULONG)&KiI386CallAbios - 1,
749  0x18 // TYPE_CODE
750  );
751 
752  //
753  // Link all the unused GDT entries to our GDT free list.
754  //
755 
756  if (Processor == 0) {
757  Ki386InitializeGdtFreeList(EndOfGdt);
758  }
759 }
