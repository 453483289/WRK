57  //

72  //

79  //

92  if (Pending != FALSE) {

178 /*++
179 
180 Routine Description:
181 
182  This function tests if a wait can be satisfied when an object attains
183  a state of signaled. If a wait can be satisfied, then the subject thread
184  is unwaited with a completion status that is the WaitKey of the wait
185  block from the object wait list. As many waits as possible are satisfied.
186 
187 Arguments:
188 
189  Object - Supplies a pointer to a dispatcher object.
190 
191 Return Value:
192 
193  None.
194 
195 --*/
196 
197 {
198 
199  PKEVENT Event;
200  PLIST_ENTRY ListHead;
201  PRKTHREAD Thread;
202  PRKWAIT_BLOCK WaitBlock;
203  PLIST_ENTRY WaitEntry;
204  NTSTATUS WaitStatus;
205 
206  //
207  // As long as the signal state of the specified object is Signaled and
208  // there are waiters in the object wait list, then try to satisfy a wait.
209  //
210 
211  Event = (PKEVENT)Object;
212  ListHead = &Event->Header.WaitListHead;
213  WaitEntry = ListHead->Flink;
214  while ((Event->Header.SignalState > 0) &&
215  (WaitEntry != ListHead)) {
216 
217  WaitBlock = CONTAINING_RECORD(WaitEntry, KWAIT_BLOCK, WaitListEntry);
218  Thread = WaitBlock->Thread;
219  WaitStatus = STATUS_KERNEL_APC;
220 
221  //
222  // N.B. The below code only satisfies the wait for wait any types.
223  // Wait all types are satisfied in the wait code itself. This
224  // is done with a eye to the future when the dispatcher lock is
225  // split into a lock per waitable object type and a scheduling
226  // state lock. For now, a kernel APC is simulated for wait all
227  // types.
228  //
229 
230  if (WaitBlock->WaitType == WaitAny) {
231  WaitStatus = (NTSTATUS)WaitBlock->WaitKey;
232  KiWaitSatisfyAny((PKMUTANT)Event, Thread);
233  }
234 
235  KiUnwaitThread(Thread, WaitStatus, Increment);
236  WaitEntry = ListHead->Flink;
237  }
238 
239  return;
240 }
