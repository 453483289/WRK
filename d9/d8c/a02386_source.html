29 ; NTSTATUS
30 ; NtContinue (
31 ; IN PCONTEXT ContextRecord,
32 ; IN BOOLEAN TestAlert
33 ; )
34 ;
35 ; Routine Description:
36 ;
37 ; This routine is called as a system service to continue execution after
38 ; an exception has occurred. Its function is to transfer information from
39 ; the specified context record into the trap frame that was built when the
40 ; system service was executed, and then exit the system as if an exception
41 ; had occurred.
42 ;
43 ; Arguments:
44 ;
45 ; ContextRecord (rcx) - Supplies a pointer to a context record.
46 ;
47 ; TestAlert (dl) - Supplies a boolean value that specifies whether alert
48 ; should be tested for the previous processor mode.
49 ;
50 ; Implicit Arguments:
51 ;
52 ; rbp - Supplies the address of a trap frame.
53 ;
54 ; Return Value:
55 ;
56 ; Normally there is no return from this routine. However, if the specified
57 ; context record is misaligned or is not accessible, then the appropriate
58 ; status code is returned.
59 ;
60 ;--
61 
62  NESTED_ENTRY NtContinue, _TEXT$00
63 
64  GENERATE_EXCEPTION_FRAME ; generate exception frame
65 
66 ;
67 ; Copy context to kernel frames or bypass for user APC.
68 ;
69 ; N.B. The context record and test alert arguments are in the same
70 ; registers.
71 ;
72 
73  mov r8, rsp ; set exception frame address
74  lea r9, (-128)[rbp] ; set trap frame address
75  call KiContinueEx ; transfer context to kernel frames
76 
77 ;
78 ; If the return status is less than or equal to zero, then return via the
79 ; system service dispatcher. Otherwise, return via exception exit.
80 ;
81 
82  test eax, eax ; test return status value
83  jle short KiCO10 ; if le, return via service dispatcher
84 
85 ;
86 ; Return via exception exit.
87 ;
88 ; If the legacy stated is switched, then restore the legacy floating state.
89 ;
90 ; N.B. If the legacy state is restored, then xmm6-xmm15 are also restored
91 ; with potentially incorrect values. Fortunately, exception exit will
92 ; restore these registers with their proper value before returning to
93 ; the user.
94 ;
95 ; N.B. The below code uses an unusual sequence to transfer control. This
96 ; instruction sequence is required to avoid detection as an epilogue.
97 ;
98 
99  mov rcx, gs:[PcCurrentThread] ; get current thread address
100  test byte ptr TrSegCs[rbp], MODE_MASK ; test if previous mode kernel
101  jnz short KiCO05 ; if nz, previous mode user
102  mov rdx, TrTrapFrame[rbp] ; restore previous trap frame address
103  mov ThTrapFrame[rcx], rdx ;
104  mov dl, TrPreviousMode[rbp] ; restore thread previous mode
105  mov ThPreviousMode[rcx], dl ;
106 KiCO05: lea rcx, KiExceptionExit ; get address of exception exit
107  jmp rcx ; finish in common code
108 
109 ;
110 ; Return via the system service dispatcher.
111 ;
112 ; N.B. The nonvolatile state in the exception frame is not restored since
113 ; the state is not used in this function.
114 ;
115 
116 KiCO10: add rsp, (KEXCEPTION_FRAME_LENGTH - (1 * 8)) ; deallocate stack frame
117  ret ; return
118 
119  NESTED_END NtContinue, _TEXT$00

122 ;++
123 ;
124 ; NTSTATUS
125 ; NtRaiseException (
126 ; IN PEXCEPTION_RECORD ExceptionRecord,
127 ; IN PCONTEXT ContextRecord,
128 ; IN BOOLEAN FirstChance
129 ; )
130 ;
131 ; Routine Description:
132 ;
133 ; This routine is called as a system service to raise an exception. Its
134 ; function is to transfer information from the specified context record
135 ; into the trap frame that was built when the system service was executed.
136 ; The exception may be raised as a first or second chance exception.
137 ;
138 ; Arguments:
139 ;
140 ; ExceptionRecord (rcx) - Supplies a pointer to an exception record.
141 ;
142 ; ContextRecord (rdx) - Supplies a pointer to a context record.
143 ;
144 ; FirstChance (r8b) - Supplies a boolean value that specifies whether
145 ; this is the first (TRUE) or second chance (FALSE) for dispatching
146 ; the exception.
147 ;
148 ; Implicit Arguments:
149 ;
150 ; rbp - Supplies a pointer to a trap frame.
151 ;
152 ; Return Value:
153 ;
154 ; Normally there is no return from this routine. However, if the specified
155 ; context record or exception record is misaligned or is not accessible,
156 ; then the appropriate status code is returned.
157 ;
158 ;--
159 
160  NESTED_ENTRY NtRaiseException, _TEXT$00
161 
162  GENERATE_EXCEPTION_FRAME ; generate exception frame
163 
164 ;
165 ; Record the address of the caller in a place that won't be overwritten
166 ; by the following context to k-frames.
167 ;
168 
169  mov rax, TrRip[rbp] ; copy fault address to preserved entry
170  mov TrFaultAddress[rbp], rax ;
171 
172 ;
173 ; Call the raise exception kernel routine which will marshall the arguments
174 ; and then call the exception dispatcher.
175 ;
176 
177  mov ExP5[rsp], r8b ; set first chance parameter
178  mov r8, rsp ; set exception frame address
179  lea r9, (-128)[rbp] ; set trap frame address
180  call KiRaiseException ; call raise exception routine
181 
182 ;
183 ; If the kernel raise exception routine returns success, then exit via the
184 ; exception exit code. Otherwise, return to the system service dispatcher.
185 ;
186 
187  test eax, eax ; test if service failed
188  jnz short KiRE20 ; if nz, service failed
189 
190 ;
191 ; Exit via the exception exit code which will restore the machine state.
192 ;
193 ; N.B. The below code uses an unusual sequence to transfer control. This
194 ; instruction sequence is required to avoid detection as an epilogue.
195 ;
196 
197  test byte ptr TrSegCs[rbp], MODE_MASK ; test if previous mode kernel
198  jnz short KiRE10 ; if nz, previous mode user
199  mov rbx, gs:[PcCurrentThread] ; get current thread address
200  mov rdx, TrTrapFrame[rbp] ; restore previous trap frame address
201  mov ThTrapFrame[rbx], rdx ;
202  mov dl, TrPreviousMode[rbp] ; restore thread previous mode
203  mov ThPreviousMode[rbx], dl ;
204 KiRE10: lea rcx, KiExceptionExit ; get address of exception exit
205  jmp rcx ; finish in common code
206 
207 ;
208 ; The context or exception record is misaligned or not accessible, or the
209 ; exception was not handled.
210 ;
211 
212 KiRE20: RESTORE_EXCEPTION_STATE ; restore exception state/deallocate
213 
214  ret ; return
215 
216  NESTED_END NtRaiseException, _TEXT$00

221 ; KiDivide6432 (

223 ; IN ULONG Divisor

225 ;
226 ; Routine Description:
227 ;
228 ; This routine performs a divide operation with a dividend of exactly
229 ; 64 bits, a divisor of exactly 32 bits, and a quotient of exactly 32
230 ; bits.
231 ;
232 ; Arguments:
233 ;
234 ; Dividend - Supplies a 64-bit unsigned dividend.
235 ;
236 ; Divisor - Supplies a 32-bit unsigned divisor.
237 ;
238 ; Return Value:
239 ;
240 ; Returns the 32-bit unsigned quotient.
241 ;
242 ;--
243 
244  LEAF_ENTRY KiDivide6432, INIT
245 
246  mov eax, ecx ; set low 32-bits of dividend
247  mov r8, rdx ; save 32-bit divisor
248  shr rcx, 32 ; set high 32-bits of dividend
249  mov edx, ecx ;
250  div r8d ; perform 64/32 to 32 divide
251  ret ;
252 
253  LEAF_END KiDivide6432, INIT

255  end
