61 /*++
62 
63 Routine Description:
64 
65  This routine is the initialization routine for WMI and is called by IO
66  within IoInitSystem on NT. This routine assumes that the
67  IO system is initialized enough to call IoCreateDriver. The rest of the
68  initialization occurs in the DriverEntry routine.
69 
70 Arguments:
71 
72  Pass specifies the pass of initialization needed
73 
74 Return Value:
75 
76  TRUE if initialization was successful
77 
78 --*/
79 {
80 //
81 // We name the driver this so that any eventlogs fired will have the
82 // source name WMIxWDM and thus get the eventlog messages from
83 // ntiologc.mc
84 //
85 #define WMIDRIVERNAME L"\\Driver\\WMIxWDM"
86 
87  UNICODE_STRING DriverName;
88  NTSTATUS Status;
89 
90 #if !DBG
91  UNREFERENCED_PARAMETER (LoaderBlockPtr);
92 #endif
93 
94  if (Phase == 0)
95  {
96  WmipAssert(WmipServiceDeviceObject == NULL);
97  WmipAssert(LoaderBlockPtr != NULL);
98 
99  RtlInitUnicodeString(&DriverName, WMIDRIVERNAME);
100  Status = IoCreateDriver(&DriverName, WmipDriverEntry);
101 
102  } else {
103  WmipAssert(LoaderBlockPtr == NULL);
104 
105  WmipInitializeRegistration(Phase);
106  WmipRegisterFirmwareProviders();
107 
108  Status = STATUS_SUCCESS;
109  }
110 
111 #if defined(_X86_) || defined(_AMD64_)
112  //
113  // Give MCA a chance to init during phase 0 and 1
114  //

116 #endif
117 
118  return(NT_SUCCESS(Status) ? TRUE : FALSE);

121 NTSTATUS IoWMIRegistrationControl(
122  __in PDEVICE_OBJECT DeviceObject,
123  __in ULONG Action
124 )
125 /*++
126 
127 Routine Description:
128 
129  This routine informs WMI of the existence and disappearance of a device
130  object that would support WMI.
131 
132 Arguments:
133 
134  DeviceObject - Pointer to device object or callback address
135 
136  Action - Registration action code
137 
138  WMIREG_ACTION_REGISTER - If set action is to inform WMI that the
139  device object supports and is ready to receive WMI IRPS.
140 
141  WMIREG_ACTION_DEREGISTER - If set action is to inform WMI that the
142  device object no longer supports and is not ready to receive WMI
143  IRPS.
144 
145  WMIREG_ACTION_REREGISTER - If set action is to requery the device
146  object for the guids that it supports. This has the effect of
147  deregistering followed by registering.
148 
149  WMIREG_ACTION_UPDATE_GUIDS - If set action is to query for information
150  that is used to update already registered guids.
151 
152  WMIREG_ACTION_BLOCK_IRPS - If set action is to block any further irps
153  from being sent to the device. The irps are failed by WMI.
154 
155  If the WMIREG_FLAG_CALLBACK is set then DeviceObject actually specifies a callback
156  address and not a DeviceObject
157 
158 Return Value:
159 
160  Returns status code
161 
162 --*/
163 {
164  NTSTATUS Status;
165  ULONG RegistrationFlag = 0;
166  ULONG IsTraceProvider = FALSE;
167  ULONG TraceClass = 0;
168  PREGENTRY RegEntry;
169 
170  PAGED_CODE();
171 
172  if (WmipIsWmiNotSetupProperly())
173  {
174  return(STATUS_UNSUCCESSFUL);
175  }
176 
177  if (Action & WMIREG_FLAG_CALLBACK)
178  {
179  RegistrationFlag |= WMIREG_FLAG_CALLBACK;
180  Action &= ~WMIREG_FLAG_CALLBACK;
181  }
182 
183  if (Action & WMIREG_FLAG_TRACE_PROVIDER)
184  {
185  TraceClass = Action & WMIREG_FLAG_TRACE_NOTIFY_MASK;
186 
187  Action &= ~WMIREG_FLAG_TRACE_PROVIDER & ~WMIREG_FLAG_TRACE_NOTIFY_MASK;
188  IsTraceProvider = TRUE;
189  RegistrationFlag |= WMIREG_FLAG_TRACE_PROVIDER | TraceClass;
190  }
191 
192  switch(Action)
193  {
194  case WMIREG_ACTION_REGISTER:
195  {
196  Status = WmipRegisterDevice(
197  DeviceObject,
198  RegistrationFlag);
199 
200  if (IsTraceProvider)
201  {
202  WmipSetTraceNotify(DeviceObject, TraceClass, TRUE);
203  }
204  break;
205  }
206 
207  case WMIREG_ACTION_DEREGISTER:
208  {
209  Status = WmipDeregisterDevice(DeviceObject);
210  break;
211  }
212 
213  case WMIREG_ACTION_REREGISTER:
214  {
215  Status = WmipDeregisterDevice(DeviceObject);
216  if (NT_SUCCESS(Status))
217  {
218  Status = WmipRegisterDevice(
219  DeviceObject,
220  RegistrationFlag);
221  }
222  break;
223  }
224 
225  case WMIREG_ACTION_UPDATE_GUIDS:
226  {
227  Status = WmipUpdateRegistration(DeviceObject);
228  break;
229  }
230 
231  case WMIREG_ACTION_BLOCK_IRPS:
232  {
233  RegEntry = WmipFindRegEntryByDevice(DeviceObject, FALSE);
234  if (RegEntry != NULL)
235  {
236  //
237  // Mark the regentry as invalid so that no more irps
238  // are sent to the device and the event will set when
239  // the last irp completes.
240  WmipEnterSMCritSection();
241  RegEntry->Flags |= REGENTRY_FLAG_NOT_ACCEPTING_IRPS;
242  WmipLeaveSMCritSection();
243  WmipUnreferenceRegEntry(RegEntry);
244  Status = STATUS_SUCCESS;
245  } else {
246  Status = STATUS_INVALID_PARAMETER;
247  }
248 
249  break;
250  }
251  default:
252  {
253  Status = STATUS_INVALID_PARAMETER;
254  break;
255  }
256  }
257  return(Status);
258 }

261 NTSTATUS IoWMIAllocateInstanceIds(
262  __in GUID *Guid,
263  __in ULONG InstanceCount,
264  __out ULONG *FirstInstanceId
265  )
266 /*++
267 
268  Routine Description:
269 
270  This routine allocates a range of instance ids that are unique to the
271  guid. This routine is to be called only at PASSIVE_LEVEL.
272 
273  Arguments:
274 
275  Guid - Pointer to guid for which instance ids are needed.
276  InstanceCount - Count of instance ids to allocate.
277  *FirstInstanceId - Returns first instance id in the range.
278 
279  Return Value:
280 
281  Returns a status code
282 
283 --*/
284 {
285  PINSTIDCHUNK InstIdChunk, LastInstIdChunk = NULL;
286  PINSTID InstId;
287  ULONG i;
288 
289  PAGED_CODE();
290 
291  if (WmipIsWmiNotSetupProperly())
292  {
293  return(STATUS_UNSUCCESSFUL);
294  }
295 
296  WmipEnterSMCritSection();
297 
298  //
299  // See if the guid is already in the list
300  InstIdChunk = WmipInstIdChunkHead;
301 
302  while (InstIdChunk != NULL)
303  {
304  for (i = 0; i < INSTIDSPERCHUNK; i++)
305  {
306  InstId = &InstIdChunk->InstId[i];
307  if (InstId->BaseId == ~0)
308  {
309  //
310  // Since InstIds are always filled sequentially and are
311  // never freed, if we hit a free one then we know that
312  // our guid is not in the list and we need to fill in a
313  // new entry.
314  goto FillInstId;
315  }
316 
317  if (IsEqualGUID(Guid, &InstId->Guid))
318  {
319  //
320  // We found an entry for our guid so use its information
321  *FirstInstanceId = InstId->BaseId;
322  InstId->BaseId += InstanceCount;
323  WmipLeaveSMCritSection();
324  return(STATUS_SUCCESS);
325  }
326  }
327  LastInstIdChunk = InstIdChunk;
328  InstIdChunk = InstIdChunk->Next;
329  }
330 
331  //
332  // We need to allocate a brand new chunk to accommodate the entry
333  InstIdChunk = ExAllocatePoolWithTag(PagedPool,
334  sizeof(INSTIDCHUNK),
335  WMIIIPOOLTAG);
336  if (InstIdChunk != NULL)
337  {
338  RtlFillMemory(InstIdChunk, sizeof(INSTIDCHUNK), 0xff);
339  InstIdChunk->Next = NULL;
340  if (LastInstIdChunk == NULL)
341  {
342  WmipInstIdChunkHead = InstIdChunk;
343  } else {
344  LastInstIdChunk->Next = InstIdChunk;
345  }
346 
347  InstId = &InstIdChunk->InstId[0];
348  } else {
349  WmipLeaveSMCritSection();
350  return(STATUS_INSUFFICIENT_RESOURCES);
351  }
352 
353 FillInstId:
354  RtlCopyMemory(&InstId->Guid, Guid, sizeof(GUID));
355  InstId->BaseId = InstanceCount;
356  WmipLeaveSMCritSection();
357  *FirstInstanceId = 0;
358 
359  return(STATUS_SUCCESS);
360 }

363  __in_opt PDEVICE_OBJECT PhysicalDeviceObject,
364  __in_opt PUNICODE_STRING SymbolicLinkName,

367  )
368 /*++
369 
370  Routine Description:
371 
372  This routine is used by a device driver to suggest a base name with which
373  to build WMI instance names for the device. A driver is not bound to
374  follow the instance name returned.
375 
376  Arguments:
377 
378  PhysicalDeviceObject - PDO of device for which a suggested instance name
379  is being requested
380 
381  SymbolicLinkName - Symbolic link name returned from
382  IoRegisterDeviceInterface.
383 
384  CombineNames - If TRUE then the suggested names arising from the
385  PhysicalDeviceObject and the SymbolicLinkName are combined to create
386  the resultant suggested name.
387 
388  SuggestedInstanceName - Supplies a pointer to an empty (i.e., Buffer
389  field set to NULL) UNICODE_STRING structure which, upon success, will
390  be set to a newly-allocated string buffer containing the suggested
391  instance name. The caller is responsible for freeing
392  SuggestedInstanceName->Buffer when it is no longer needed.
393 
394 
395  Note: If CombineNames is TRUE then both PhysicalDeviceObject and
396  SymbolicLinkName must be specified. Otherwise only one of them
397  must be specified.
398 
399  Return Value:
400 
401  Returns a status code
402 
403 --*/
404 {
405  NTSTATUS Status = STATUS_INVALID_PARAMETER_MIX;
406  ULONG DeviceDescSizeRequired;
407  ULONG DeviceDescSize;
408  PWCHAR DeviceDescBuffer;
409  HANDLE DeviceInstanceKey;
410  PKEY_VALUE_FULL_INFORMATION InfoBuffer;
411  PWCHAR SymLinkDescBuffer;

413  ULONG ResultDescSize;
414  PWCHAR ResultDescBuffer;

416 
417  PAGED_CODE();
418

420  {
421  return(STATUS_UNSUCCESSFUL);
422  }
423 
424  DeviceDescBuffer = NULL;
425  DeviceDescSizeRequired = 0;
426  DeviceDescSize = 0;

522 
523  }

528  }

531  {
532  ExFreePool(DeviceDescBuffer);
533  DeviceDescBuffer = NULL;
534  }

572  } else {

574  SuggestedInstanceName->Length = (USHORT)0;
575  SuggestedInstanceName->MaximumLength = 0;
576  }
577  }
578 
579  return(Status);

582 NTSTATUS IoWMIWriteEvent(
583  __inout PVOID WnodeEventItem
584  )
585 /*++
586 
587 Routine Description:
588 
589  This routine will queue the passed WNODE_EVENT_ITEM for delivery to the
590  WMI user mode agent. Once the event is delivered the WNODE_EVENT_ITEM
591  buffer will be returned to the pool.
592 
593  This routine may be called at DPC level
594 
595 Arguments:
596 
597  WnodeEventItem - Pointer to WNODE_EVENT_ITEM that has event information.
598 
599 Return Value:
600 
601  Returns STATUS_SUCCESS or an error code
602 
603 --*/
604 {
605  NTSTATUS Status;
606  PWNODE_HEADER WnodeHeader = (PWNODE_HEADER)WnodeEventItem;
607  PULONG TraceMarker = (PULONG) WnodeHeader;
608  KIRQL OldIrql;
609  PREGENTRY RegEntry;
610  PEVENTWORKCONTEXT EventContext;
611  ULONG ProviderId;
612 
613  if (WmipIsWmiNotSetupProperly())
614  {
615  return(STATUS_UNSUCCESSFUL);
616  }
617 
618  //
619  // Special mode with high order bit set
620  //
621  if ((*TraceMarker & 0xC0000000) == TRACE_HEADER_FLAG)
622  {
623  return WmiTraceFastEvent(WnodeHeader);
624  }
625 
626  if ( (WnodeHeader->Flags & WNODE_FLAG_TRACED_GUID) ||
627  (WnodeHeader->Flags & WNODE_FLAG_LOG_WNODE) )
628  {
629  ULONG LoggerId = WmiGetLoggerId(WnodeHeader->HistoricalContext);
630  ULONG IsTrace = WnodeHeader->Flags & WNODE_FLAG_TRACED_GUID;
631  ULONG SavedSize = WnodeHeader->BufferSize;
632  PULONG TraceMarker = (PULONG) WnodeHeader;
633 
634  if (SavedSize < sizeof(WNODE_HEADER))
635  return STATUS_BUFFER_TOO_SMALL;
636 
637  //
638  // If trace header, turn higher bit on and support
639  // only full header
640  //
641  if (IsTrace)
642  {
643  if (SavedSize > 0XFFFF) // restrict to USHORT max size
644  return STATUS_BUFFER_OVERFLOW;
645 
646  *TraceMarker |= TRACE_HEADER_FLAG | TRACE_HEADER_EVENT_TRACE |
647  (TRACE_HEADER_TYPE_FULL_HEADER << 16);
648  }
649  else
650  {
651  if (SavedSize & TRACE_HEADER_FLAG)
652  return STATUS_BUFFER_OVERFLOW;
653  }
654 
655  Status = STATUS_INVALID_HANDLE;
656  if (LoggerId > 0 && LoggerId < MAXLOGGERS)
657  {
658  if (WmipLoggerContext[LoggerId] != NULL)
659  {
660  //
661  // NOTE: The rule here is that IoWMIWriteEvent is always
662  // called in kernel mode, and the buffer needs not be probed!
663  //
664  Status = WmiTraceEvent(WnodeHeader, KernelMode);
665  }
666  }
667  // NOTE: If it is a trace, we will not go any further
668  // Otherwise, if it is a regular WMI event, it will still
669  // be processed by WMI.
670 
671  if (IsTrace)
672  {
673  WnodeHeader->BufferSize = SavedSize;
674  return Status;
675  }
676  }
677 
678  //
679  // Memory for event buffers is limited so the size of any event is also
680  // limited.
681 #if DBG
682  if (WnodeHeader->BufferSize > LARGEKMWNODEEVENTSIZE)
683  {
684  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_EVENT_INFO_LEVEL,
685  "WMI: Large event %p fired by %x via WMI\n",
686  WnodeEventItem,
687  ((PWNODE_HEADER)WnodeEventItem)->ProviderId));
688  }
689 #endif
690 
691  if (WnodeHeader->BufferSize <= WmipMaxKmWnodeEventSize)
692  {
693 
694  EventContext = ExAllocatePoolWithTag(NonPagedPool,
695  sizeof(EVENTWORKCONTEXT),
696  WMINWPOOLTAG);
697  if (EventContext != NULL)
698  {
699  //
700  // Try to take a refcount on the regentry associated with the
701  // provider id in the event. If we are successful then we set a
702  // flag in the wnode header saying so. When processing the
703  // event in the work item we check the flag and if it is set
704  // we'll go looking for the regentry on the active and zombie
705  // lists and then use it. At that time it will give up the ref
706  // count taken here so that if the regentry really is a zombie
707  // then it will go away peacefully.
708  //
709 
710  ProviderId = WnodeHeader->ProviderId;
711  KeAcquireSpinLock(&WmipRegistrationSpinLock,
712  &OldIrql);
713 
714  RegEntry = WmipDoFindRegEntryByProviderId(ProviderId,
715  REGENTRY_FLAG_RUNDOWN);
716  if (RegEntry != NULL)
717  {
718  WmipReferenceRegEntry(RegEntry);
719  }
720 
721  KeReleaseSpinLock(&WmipRegistrationSpinLock,
722  OldIrql);
723 
724  WnodeHeader->ClientContext = WnodeHeader->Version;
725 
726  EventContext->RegEntry = RegEntry;
727  EventContext->Wnode = WnodeHeader;
728 
729  ExInterlockedInsertTailList(
730  &WmipNPEvent,
731  &EventContext->ListEntry,
732  &WmipNPNotificationSpinlock);
733  //
734  // If the queue was empty then there was no work item outstanding
735  // to move from non paged to paged memory. So fire up a work item
736  // to do so.
737  if (InterlockedIncrement(&WmipEventWorkItems) == 1)
738  {
739  ExQueueWorkItem( &WmipEventWorkQueueItem, DelayedWorkQueue );
740  }
741  Status = STATUS_SUCCESS;
742  } else {
743  Status = STATUS_INSUFFICIENT_RESOURCES;
744  }
745  } else {
746  Status = STATUS_BUFFER_OVERFLOW;
747  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_EVENT_INFO_LEVEL,
748  "WMI: IoWMIWriteEvent detected an event %p fired by %x that exceeds the maximum event size\n",
749  WnodeEventItem,
750  ((PWNODE_HEADER)WnodeEventItem)->ProviderId));
751  }
752 
753  return(Status);
754 }
