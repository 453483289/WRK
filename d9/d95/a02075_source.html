114 //
115 // Routine to actually call to do a tree copy (or sync)
116 //
117 
118 BOOLEAN
119 CmpCopySyncTree(
120  PHHIVE SourceHive,
121  HCELL_INDEX SourceCell,
122  PHHIVE TargetHive,
123  HCELL_INDEX TargetCell,
124  BOOLEAN CopyVolatile,
125  CMP_COPY_TYPE CopyType
126  )
127 /*++
128 
129 Routine Description:
130 
131  This routine can perform two distinct (yet similar) tasks:
132  a tree copy or a tree synchronization (sync). Which task
133  is performed is determined by the TreeSync parameter.
134 
135  For both operations:
136  --------------------
137 
138  The source root key and target root key must exist in advance.
139  These root nodes and their value entries will NOT be copied/synced.
140 
141  NOTE: Volatile keys are only copied/synced if the CopyVolatile
142  parameter is set to true.
143 
144 
145  For a tree copy:
146  ----------------
147 
148  A tree is copied from source to destination. The subkeys
149  of the source root key and the full trees under those
150  subkeys will be copied to a new tree at target root key.
151 
152  NOTE: If this call fails part way through, it will NOT undo
153  any successfully completed key copies, thus a partial
154  tree copy CAN occur.
155 
156  For a tree sync:
157  ----------------
158 
159  The target tree is synchronized with the source tree. It is
160  assumed that for a certain period of the time the target tree
161  has remained unmodified while modifications may have been made
162  to the source tree. During a sync, any such modifications
163  to the source tree are made to the target tree. Thus, at the
164  end of a successful sync, the target tree is identical to the
165  source tree.
166 
167  Since only things that have changed in the source tree
168  are modified in the target tree, a sync operation is far
169  more efficient than the delete/copy operations necessary
170  to accomplish the same results.
171 
172  NOTE: It is assumed that no open handles are held
173  on any target tree keys. Registry in-memory data
174  structures may be corrupted if this is not true.
175 
176 Arguments:
177 
178  SourceHive - pointer to hive control structure for source
179 
180  SourceCell - index of cell at root of tree to copy/sync
181 
182  TargetHive - pointer to hive control structure for target
183 
184  TargetCell - pointer to cell at root of target tree
185 
186  CopyVolatile - indicates whether volatile keys should be
187  copied/synced.
188 
189  CopyType - indicates the type of the copy operation:
190  Copy - A copy is requested
191  Sync - A sync is requested
192  Merge - A merge is requested i.e.:
193  1. the target nodes that are not present on the source tree are not
194  deleted.
195  2. the target nodes that are present in the source tree gets overrided
196  no matter what the LastWriteTime value is.
197 Return Value:
198 
199  BOOLEAN - Result code from call, among the following:
200  TRUE - it worked
201  FALSE - the tree copy/sync was not completed (though more than 0
202  keys may have been copied/synced)
203 
204 --*/
205 {
206  BOOLEAN result;
207  PCMP_COPY_STACK_ENTRY CmpCopyStack;
208  PRELEASE_CELL_ROUTINE TargetReleaseCellRoutine;
209 
210  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_SAVRES,"CmpCopyTree:\n"));
211 
212  CmpCopyStack = ExAllocatePool(
213  PagedPool,
214  sizeof(CMP_COPY_STACK_ENTRY)*CMP_INITIAL_STACK_SIZE
215  );
216  if (CmpCopyStack == NULL) {
217  return FALSE;
218  }
219  CmpCopyStack[0].SourceCell = SourceCell;
220  CmpCopyStack[0].TargetCell = TargetCell;
221 
222  //
223  // the target is something temporary (in memory and local var)
224  // So, we just set the release routines to NULL and restore after
225  // the copy is complete; this saves some pain
226  //
227  TargetReleaseCellRoutine = TargetHive->ReleaseCellRoutine;
228  TargetHive->ReleaseCellRoutine = NULL;
229 
230  result = CmpCopySyncTree2(
231  CmpCopyStack,
232  CMP_INITIAL_STACK_SIZE,
233  0,
234  SourceHive,
235  TargetHive,
236  CopyVolatile,
237  CopyType
238  );
239 
240  TargetHive->ReleaseCellRoutine = TargetReleaseCellRoutine;
241 
242  ExFreePool(CmpCopyStack);
243  return result;
244 }

261 /*++
262 
263 Routine Description:
264 
265  This is a helper routine for CmpCopySyncTree. It accomplishes
266  the functionality described by that routine in a "virtually"
267  recursive manner which frees this routine from the limitations
268  of the Kernel stack.
269 
270  This routine should not be called directly. Use CmpCopySyncTree!.
271 
272 Arguments:
273 
274  (All of these are "virtual globals")
275 
276  CmpCopyStack - "global" pointer to stack for frames
277 
278  CmpCopyStackSize - allocateed size of stack
279 
280  CmpCopyStackTop - current top
281 
282  CmpSourceHive, CmpTargetHive - source and target hives
283 
284  CopyVolatile, CopyType - same as CmpCopySyncTree.
285 
286 
287 Return Value:
288 
289  BOOLEAN - Result code from call, among the following:
290  TRUE - it worked
291  FALSE - the tree copy/sync was not completed (though more than 0
292  keys may have been copied/synced)
293 
294 --*/
295 {
296  PCMP_COPY_STACK_ENTRY Frame;
297  HCELL_INDEX SourceChild;
298  HCELL_INDEX NewSubKey;
299 
300  BOOLEAN Ret = FALSE, SyncNeedsTreeCopy = FALSE;
301  UNICODE_STRING KeyName;
302  PCM_KEY_NODE SourceChildCell, TargetChildCell;
303  PCM_KEY_NODE SourceCell, TargetCell, TempNode;
304  ULONG SyncTreeCopyStackStart = CmpCopyStackTop - 1;
305  WCHAR *NameBuffer = NULL;
306 
307  // A merge is a particular case of a sync !!!
308  BOOLEAN TreeSync = (CopyType == Sync || CopyType == Merge)?TRUE:FALSE;
309 
310  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_SAVRES,"CmpCopyTree2:\n"));
311 
312  if (TreeSync) {
313 
314  //
315  // The sync operation involves some work with key names,
316  // so we must allocate a buffer used for key name decompression.
317  //
318 
319  NameBuffer = ExAllocatePool(PagedPool, REG_MAX_KEY_NAME_LENGTH);
320  if(!NameBuffer) return FALSE;
321 
322  }
323 
324  //
325  // outer loop, apply to entire tree, emulate recursion here
326  // jump to here is a virtual call
327  //
328  Outer: while (TRUE) {
329 
330  Frame = &(CmpCopyStack[CmpCopyStackTop]);
331 
332  Frame->i = 0;
333 
334  //
335  // inner loop, applies to one key
336  // jump to here is a virtual return
337  //
338  Inner: while (TRUE) {
339 
340  SourceCell = (PCM_KEY_NODE)HvGetCell(CmpSourceHive, Frame->SourceCell);
341  if( SourceCell == NULL ) {
342  //
343  // we couldn't map the bin containing this cell
344  //
345  goto CopyEnd;
346  }
347 
348  SourceChild = CmpFindSubKeyByNumber(CmpSourceHive,
349  SourceCell,
350  Frame->i);
351  (Frame->i)++;
352 
353  if ((SourceChild == HCELL_NIL) || (!CopyVolatile &&
354  (HvGetCellType(SourceChild) == Volatile))) {
355 
356  //
357  // we've stepped through all the children (or we are only
358  // interested in stable children and have just stepped through
359  // the stable children and into the volatile ones)
360  //
361 
362  if(TreeSync && (CopyType != Merge))
363  {
364  //
365  // If we are here during a sync, that means most of sync operations
366  // applied to the current SourceCell have been completed.
367  // That is, we have:
368  // 1) Synchronized SourceCell's values with its counterpart in the
369  // target tree.
370  // 2) Synchronized any new SourceCell subkeys (subkeys present
371  // in SourceCell but not its counterpart) by creating
372  // and copying them to the proper place in the target tree.
373  //
374  // What this means is that SourceCell's counterpart in the target tree
375  // (TargetCell) now has at least as many subkeys as SourceCell.
376  //
377  // This implies that if TargetCell now has more subkeys that SourceCell
378  // than some subkeys of TargetCell are not present in the source tree
379  // (probably because those keys were deleted from the source tree
380  // during the period between the previous sync and now).
381  //
382  // If such keys exist, then they must be delete them from TargetCell
383  // in order to complete the sync. We do this below.
384  //
385 
386  TargetCell = (PCM_KEY_NODE)HvGetCell(CmpTargetHive, Frame->TargetCell);
387  if( TargetCell == NULL ) {
388  //
389  // we couldn't map the bin containing this cell
390  //
391  HvReleaseCell(CmpSourceHive, Frame->SourceCell);
392  goto CopyEnd;
393  }
394 
395  //
396  // Does TargetCell have more subkeys than SourceCell?
397  //
398 
399  if((TargetCell->SubKeyCounts[Stable] +
400  TargetCell->SubKeyCounts[Volatile]) >
401 
402  (SourceCell->SubKeyCounts[Stable] +
403 
404  // We only count the volatile keys if we are actually
405  // syncing them. Note, however, that we always use
406  // the volatile counts in TargetCell since we may
407  // be syncing to a volatile tree where all keys are volatile.
408 
409  (CopyVolatile ? SourceCell->SubKeyCounts[Volatile] : 0)))
410 
411  {
412  //
413  // Delete what should be deleted from TargetCell
414  //
415 
416  CmpSyncSubKeysAfterDelete(CmpSourceHive,
417  SourceCell,
418  CmpTargetHive,
419  TargetCell,
420  NameBuffer);
421  }
422  //
423  // release target cell as we don't need it anymore
424  //
425  HvReleaseCell(CmpTargetHive, Frame->TargetCell);
426  }
427  //
428  // release the source cell
429  //
430  HvReleaseCell(CmpSourceHive, Frame->SourceCell);
431  break;
432  } else {
433  //
434  // release the source cell
435  //
436  HvReleaseCell(CmpSourceHive, Frame->SourceCell);
437  }
438 
439  if (TreeSync) {
440 
441  //
442  // For a sync, we want to check if the current child (subkey)
443  // of SourceCell is also a child of TargetCell - i.e. if
444  // the subkey in question has a counterpart in the target tree.
445  //
446  // There is no guarantee that the counterpart's index number
447  // will be the same so we must perform this check using
448  // the subkey name.
449  //
450 
451  //
452  // Get the name of the current child
453  //
454 
455  SourceChildCell = (PCM_KEY_NODE)HvGetCell(CmpSourceHive,
456  SourceChild);
457 
458  if( SourceChildCell == NULL ) {
459  //
460  // we couldn't map the bin containing this cell
461  //
462  goto CopyEnd;
463  }
464  CmpInitializeKeyNameString(SourceChildCell,
465  &KeyName,
466  NameBuffer);
467 
468  //
469  // Try to find the current child's counterpart in
470  // in the target tree using the child's name.
471  //
472 
473  TempNode = (PCM_KEY_NODE)HvGetCell(CmpTargetHive,Frame->TargetCell);
474  if( TempNode == NULL ) {
475  //
476  // we couldn't map the bin containing this cell
477  //
478  HvReleaseCell(CmpSourceHive,SourceChild);
479  goto CopyEnd;
480  }
481 
482  NewSubKey = CmpFindSubKeyByName(CmpTargetHive,
483  TempNode,
484  &KeyName);
485 
486  // release the temporary node
487  HvReleaseCell(CmpTargetHive,Frame->TargetCell);
488 
489  if (NewSubKey != HCELL_NIL) {
490 
491  //
492  // Found it, the current child (subkey) has a counterpart
493  // in the target tree. Thus, we just need to check if
494  // the counterpart's values are out of date and should
495  // be updated.
496  //
497 
498  TargetChildCell = (PCM_KEY_NODE)HvGetCell(CmpTargetHive,
499  NewSubKey);
500  if( TargetChildCell == NULL ) {
501  //
502  // we couldn't map the bin containing this cell
503  //
504  HvReleaseCell(CmpSourceHive,SourceChild);
505  goto CopyEnd;
506  }
507 
508  //
509  // Check if the current subkey has been modified
510  // more recently than its target tree counterpart.
511  // When we are doing a tree merge, always override the target.
512  //
513 
514  if ( (CopyType == Merge) ||
515  ((TargetChildCell->LastWriteTime.QuadPart) <
516  (SourceChildCell->LastWriteTime.QuadPart))) {
517 
518  //
519  // The counterpart is out of date. Its values
520  // must be synchronized with the current subkey.
521  //
522 
523  //
524  // Sync up the key's values, sd, & class
525  //
526 
527  if(CopyType == Merge) {
528  if(!CmpMergeKeyValues(CmpSourceHive, SourceChild, SourceChildCell,
529  CmpTargetHive, NewSubKey, TargetChildCell)) {
530  HvReleaseCell(CmpSourceHive,SourceChild);
531  HvReleaseCell(CmpTargetHive,NewSubKey);
532  goto CopyEnd;
533  }
534  } else {
535  if(!CmpSyncKeyValues(CmpSourceHive, SourceChild, SourceChildCell,
536  CmpTargetHive, NewSubKey, TargetChildCell)) {
537  HvReleaseCell(CmpSourceHive,SourceChild);
538  HvReleaseCell(CmpTargetHive,NewSubKey);
539  goto CopyEnd;
540  }
541  }
542 
543  //
544  // Sync the timestamps so that we don't do this again.
545  //
546 
547  TargetChildCell->LastWriteTime.QuadPart =
548  SourceChildCell->LastWriteTime.QuadPart;
549 
550  }
551 
552  //
553  // If we are here, then the current subkey's target
554  // tree counterpart has been synchronized (or did not need
555  // to be). Transfer control to the code that will apply
556  // this function "recursively" to the current subkey in order
557  // to continue the sync.
558  //
559 
560  HvReleaseCell(CmpSourceHive,SourceChild);
561  HvReleaseCell(CmpTargetHive,NewSubKey);
562  goto NewKeyCreated;
563 
564  }
565 
566  //
567  // If we are here, it means that the current child (subkey)
568  // does not have a counterpart in the target tree. This means
569  // we have encountered a new subkey in the source tree and must
570  // create it in the target tree.
571  //
572  // The standard copy code below will create this subkey. However,
573  // we must also make sure that the tree under this subkey is properly
574  // copied from source to target. The most efficient way of doing
575  // this is to temporarily forget that we are in a sync operation
576  // and merely perform a copy until the desired result is achieved.
577  //
578 
579  //
580  // Indicate that we will just copy and not sync for a while
581  //
582

584  //
585  // release this cell as we don't need it anymore
586  //
587  HvReleaseCell(CmpSourceHive,SourceChild);
588  }

591  CmpSourceHive,
592  SourceChild,
593  CmpTargetHive,
594  Frame->TargetCell,
595  TRUE
596  );
597

600 
601  goto CopyEnd;
602  }
603

605  CmpTargetHive,
606  Frame->TargetCell,
607  NewSubKey
608  )
609  ) {
610 
611  goto CopyEnd;
612  }
613

615  // Sync up the MaxNameLen field in the parent
616  //

618  PCM_KEY_NODE TargetNodeParent, TargetNodeSon;
619  USHORT NameLength;
620  TargetNodeParent = (PCM_KEY_NODE)HvGetCell(CmpTargetHive,Frame->TargetCell);
621  if( TargetNodeParent == NULL ) {
622  goto CopyEnd;
623  }
624  //
625  // make sure it's dirty as we are going to alter it.
626  //
627  if (! HvMarkCellDirty(CmpTargetHive,Frame->TargetCell,FALSE)) {
628  HvReleaseCell(CmpTargetHive,Frame->TargetCell);
629  goto CopyEnd;
630  }

633  if( TargetNodeSon == NULL ) {
634  HvReleaseCell(CmpTargetHive,Frame->TargetCell);
635  goto CopyEnd;
636  }
637

639  if( TargetNodeSon->Flags & KEY_COMP_NAME ) {
640  NameLength *= sizeof(WCHAR);
641  }
642  if (TargetNodeParent->MaxNameLen < NameLength) {
643  TargetNodeParent->MaxNameLen = NameLength;
644  }
645 
646  HvReleaseCell(CmpTargetHive,NewSubKey);
647  HvReleaseCell(CmpTargetHive,Frame->TargetCell);
648  }
649

651  // Check if the sync operation determined that this
652  // subtree should be copied
653  //
654 
655  if(TreeSync && SyncNeedsTreeCopy) {
656 
657  //
658  // We have just created a new key in the target tree
659  // with the above code. However, since this is a sync,
660  // the parent of that new key has not been created by our
661  // code and thus may not have been modified at all before
662  // the creation of the new key. But this parent now
663  // has a new child, and must therefore be marked as dirty.
664  //
665 
666  if (! CmpMarkKeyParentDirty(CmpTargetHive, NewSubKey)) {
667 
668  goto CopyEnd;
669  }
670

672  // Record the stack level where we start the copy
673  // (and temporarily abandon the sync)
674  // so that we can return to the sync operation when this
675  // stack level is reached again (i.e. when the tree
676  // under the current subkey is fully copied)
677  //
678 
679  SyncTreeCopyStackStart = CmpCopyStackTop;
680 
681  //
682  // Pretend that this is not a sync in order
683  // to simply start copying
684  //
685 
686  TreeSync = FALSE;
687  }
688

690 
691  //
692  // We succeeded in copying/syncing the subkey, apply
693  // ourselves to it
694  //
695  CmpCopyStackTop++;
696 
697  if (CmpCopyStackTop >= CmpCopyStackSize) {
698 
699  //
700  // if we're here, it means that the tree
701  // we're trying to copy is more than 1024
702  // COMPONENTS deep (from 2048 to 256k bytes)
703  // we could grow the stack, but this is pretty
704  // severe, so return FALSE and fail the copy
705  //
706 
707  goto CopyEnd;
708  }
709

711  SourceChild;
712 
713  CmpCopyStack[CmpCopyStackTop].TargetCell =
714  NewSubKey;
715 
716  goto Outer;
717 
718 
719  } // Inner: while
720

722  Ret = TRUE;
723  goto CopyEnd;
724  }
725 
726  CmpCopyStackTop--;
727  Frame = &(CmpCopyStack[CmpCopyStackTop]);

731  // This is a good time to check if we need to resume a temporarily
732  // suspended sync operation.
733  //
734

740  //
741

743  SyncNeedsTreeCopy = FALSE;
744  }
745 
746

748 
749  } // Outer: while

752 
753  if (NameBuffer) ExFreePool(NameBuffer);
754  return Ret;
