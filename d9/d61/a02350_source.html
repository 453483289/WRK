52 // IopInitializeIrp(
53 // IN OUT PIRP Irp,
54 // IN USHORT PacketSize,
55 // IN CCHAR StackSize
56 // )
57 //
58 // Routine Description:
59 //
60 // Initializes an IRP.
61 //
62 // Arguments:
63 //
64 // Irp - a pointer to the IRP to initialize.
65 //
66 // PacketSize - length, in bytes, of the IRP.
67 //
68 // StackSize - Number of stack locations in the IRP.
69 //
70 // Return Value:
71 //
72 // None.
73 //
74 //--
75 
76 #define IopInitializeIrp( Irp, PacketSize, StackSize ) { \
77  RtlZeroMemory( (Irp), (PacketSize) ); \
78  (Irp)->Type = (CSHORT) IO_TYPE_IRP; \
79  (Irp)->Size = (USHORT) ((PacketSize)); \
80  (Irp)->StackCount = (CCHAR) ((StackSize)); \
81  (Irp)->CurrentLocation = (CCHAR) ((StackSize) + 1); \
82  (Irp)->ApcEnvironment = KeGetCurrentApcEnvironment(); \
83  InitializeListHead (&(Irp)->ThreadListEntry); \
84  (Irp)->Tail.Overlay.CurrentStackLocation = \
85  ((PIO_STACK_LOCATION) ((UCHAR *) (Irp) + \
86  sizeof( IRP ) + \
87  ( (StackSize) * sizeof( IO_STACK_LOCATION )))); }

172 IopQueueThreadIrp(
173  IN PIRP Irp
174  )
175 /*++
176 
177 Routine Description:
178 
179  This routine queues the specified I/O Request Packet (IRP) to the thread
180  whose TCB address is stored in the packet.
181 
182 Arguments:
183 
184  Irp - Supplies the IRP to be queued for the specified thread.
185 
186 Return Value:
187 
188  None.
189 
190 --*/
191 {
192  PETHREAD Thread;
193  PLIST_ENTRY Head, Entry;
194 
195  Thread = Irp->Tail.Overlay.Thread;
196  Head = &Thread->IrpList;
197  Entry = &Irp->ThreadListEntry;
198 
199  KeEnterGuardedRegionThread (&Thread->Tcb);
200 
201  InsertHeadList( Head,
202  Entry );
203 
204  KeLeaveGuardedRegionThread (&Thread->Tcb);
205 
206 }
