44 NtSetSecurityObject (
45  __in HANDLE Handle,
46  __in SECURITY_INFORMATION SecurityInformation,
47  __in PSECURITY_DESCRIPTOR SecurityDescriptor
48  )
49 
50 /*++
51 
52 Routine Description:
53 
54  This routine is used to invoke an object's security routine. It
55  is used to set the object's security state.
56 
57 Arguments:
58 
59  Handle - Supplies the handle for the object being modified
60 
61  SecurityInformation - Indicates the type of information we are
62  interested in setting. e.g., owner, group, dacl, or sacl.
63 
64  SecurityDescriptor - Supplies the security descriptor for the
65  object being modified.
66 
67 Return Value:
68 
69  An appropriate NTSTATUS value
70 
71 --*/
72 
73 {
74  NTSTATUS Status;
75  PVOID Object;
76  ACCESS_MASK DesiredAccess;
77  OBJECT_HANDLE_INFORMATION HandleInformation;
78  KPROCESSOR_MODE RequestorMode;
79  SECURITY_DESCRIPTOR_RELATIVE *CapturedDescriptor;
80 
81  PAGED_CODE();
82 
83  //
84  // Make sure the passed security descriptor is really there.
85  // SeCaptureSecurityDescriptor doesn't mind being passed a NULL
86  // SecurityDescriptor, and will just return NULL back.
87  //
88 
89  if (!ARGUMENT_PRESENT( SecurityDescriptor )) {
90 
91  return( STATUS_ACCESS_VIOLATION );
92  }
93 
94  //
95  // Establish the accesses needed to the object based upon the
96  // security information being modified.
97  //
98 
99  SeSetSecurityAccessMask( SecurityInformation, &DesiredAccess );
100 
101  Status = ObReferenceObjectByHandle( Handle,
102  DesiredAccess,
103  NULL,
104  RequestorMode = KeGetPreviousMode(),
105  &Object,
106  &HandleInformation );
107 
108  if (NT_SUCCESS( Status )) {
109 
110  //
111  // Probe and capture the input security descriptor, and return
112  // right away if it is ill-formed.
113  //
114  // Because the security descriptor is always captured the returned
115  // security descriptor is in self-relative format.
116  //
117 
118  Status = SeCaptureSecurityDescriptor( SecurityDescriptor,
119  RequestorMode,
120  PagedPool,
121  TRUE,
122  (PSECURITY_DESCRIPTOR *)&CapturedDescriptor );
123 
124  if (NT_SUCCESS( Status )) {
125 
126  //
127  // Now check for a valid combination of what the user wants to set
128  // and what was supplied in the input security descriptor. If the
129  // caller wants to set the owner then the owner field of the
130  // security descriptor better not be null, likewise for the group
131  // setting. If anything is missing we'll return and error.
132  //
133 
134  ASSERT(CapturedDescriptor->Control & SE_SELF_RELATIVE);
135 
136  if (((SecurityInformation & OWNER_SECURITY_INFORMATION) &&
137  (CapturedDescriptor->Owner == 0))
138 
139  ||
140 
141  ((SecurityInformation & GROUP_SECURITY_INFORMATION) &&
142  (CapturedDescriptor->Group == 0))) {
143 
144  SeReleaseSecurityDescriptor( (PSECURITY_DESCRIPTOR)CapturedDescriptor,
145  RequestorMode,
146  TRUE );
147 
148  ObDereferenceObject( Object );
149 
150  return( STATUS_INVALID_SECURITY_DESCR );
151  }
152 
153  Status = ObSetSecurityObjectByPointer( Object,
154  SecurityInformation,
155  CapturedDescriptor );
156 
157  SeReleaseSecurityDescriptor( (PSECURITY_DESCRIPTOR)CapturedDescriptor,
158  RequestorMode,
159  TRUE );
160  }
161 
162  ObDereferenceObject( Object );
163 
164  }
165 
166  return( Status );
167 }

250 NTSTATUS
251 NtQuerySecurityObject (
252  __in HANDLE Handle,
253  __in SECURITY_INFORMATION SecurityInformation,
254  __out_bcount_opt(Length) PSECURITY_DESCRIPTOR SecurityDescriptor,
255  __in ULONG Length,
256  __out PULONG LengthNeeded
257  )
258 
259 /*++
260 
261 Routine Description:
262 
263  This routine is used to query the security descriptor for an
264  object.
265 
266 Arguments:
267 
268  Handle - Supplies the handle for the object being investigated
269 
270  SecurityInformation - Indicates the type of information we are
271  interested in getting. e.g., owner, group, dacl, or sacl.
272 
273  SecurityDescriptor - Supplies a pointer to where the information
274  should be returned
275 
276  Length - Supplies the size, in bytes, of the output buffer
277 
278  LengthNeeded - Receives the length, in bytes, needed to store
279  the output security descriptor
280 
281 Return Value:
282 
283  An appropriate NTSTATUS value
284 
285 --*/
286 
287 {
288  NTSTATUS Status;
289  PVOID Object;
290  ACCESS_MASK DesiredAccess;
291  OBJECT_HANDLE_INFORMATION HandleInformation;
292  KPROCESSOR_MODE RequestorMode;
293  POBJECT_HEADER ObjectHeader;
294  POBJECT_TYPE ObjectType;
295 
296  PAGED_CODE();
297 
298  //
299  // Probe output parameters
300  //
301 
302  RequestorMode = KeGetPreviousMode();
303 
304  if (RequestorMode != KernelMode) {
305 
306  try {
307 
308  ProbeForWriteUlong( LengthNeeded );
309 
310  ProbeForWrite( SecurityDescriptor, Length, sizeof(ULONG) );
311 
312  } except(EXCEPTION_EXECUTE_HANDLER) {
313 
314  return GetExceptionCode();
315  }
316  }
317 
318  //
319  // Establish the accesses needed to the object based upon the
320  // security information being queried
321  //
322 
323  SeQuerySecurityAccessMask( SecurityInformation, &DesiredAccess );
324 
325  Status = ObReferenceObjectByHandle( Handle,
326  DesiredAccess,
327  NULL,
328  RequestorMode,
329  &Object,
330  &HandleInformation );
331 
332  if (!NT_SUCCESS( Status )) {
333 
334  return( Status );
335  }
336 
337  //
338  // Map the object body to an object header and the corresponding
339  // object type
340  //
341 
342  ObjectHeader = OBJECT_TO_OBJECT_HEADER( Object );
343  ObjectType = ObjectHeader->Type;
344 
345  //
346  // Invoke the object type's security callback routine to query
347  // the object. This routine is assumed to have a try-except around
348  // the setting of the output security descriptor
349  //
350 
351  Status = (ObjectType->TypeInfo.SecurityProcedure)( Object,
352  QuerySecurityDescriptor,
353  &SecurityInformation,
354  SecurityDescriptor,
355  &Length,
356  &ObjectHeader->SecurityDescriptor,
357  ObjectType->TypeInfo.PoolType,
358  &ObjectType->TypeInfo.GenericMapping );
359 
360  //
361  // Indicate the length needed for the security descriptor. This
362  // will be set even if the callback failed so the caller will know
363  // the number of bytes necessary
364  //
365 
366  try {
367 
368  *LengthNeeded = Length;
369 
370  } except(EXCEPTION_EXECUTE_HANDLER) {
371 
372  ObDereferenceObject( Object );
373 
374  return(GetExceptionCode());
375  }
376 
377  //
378  // And return to our caller
379  //
380 
381  ObDereferenceObject( Object );
382 
383  return( Status );
384 }

387 BOOLEAN
388 ObCheckObjectAccess (
389  __in PVOID Object,
390  __inout PACCESS_STATE AccessState,
391  __in BOOLEAN TypeMutexLocked,
392  __in KPROCESSOR_MODE AccessMode,
393  __out PNTSTATUS AccessStatus
394  )
395 
396 /*++
397 
398 Routine Description:
399 
400  This routine performs access validation on the passed object. The
401  remaining desired access mask is extracted from the AccessState
402  parameter and passes to the appropriate security routine to perform the
403  access check.
404 
405  If the access attempt is successful, SeAccessCheck returns a mask
406  containing the granted accesses. The bits in this mask are turned
407  on in the PreviouslyGrantedAccess field of the AccessState, and
408  are turned off in the RemainingDesiredAccess field.
409 
410 Arguments:
411 
412  Object - The object being examined.
413 
414  AccessState - The ACCESS_STATE structure containing accumulated
415  information about the current attempt to gain access to the object.
416 
417  TypeMutexLocked - Indicates whether the type mutex for this object's
418  type is locked. The type mutex is used to protect the object's
419  security descriptor from being modified while it is being accessed.
420 
421  AccessMode - The previous processor mode.
422 
423  AccessStatus - Pointer to a variable to return the status code of the
424  access attempt. In the case of failure this status code must be
425  propagated back to the user.
426 
427 
428 Return Value:
429 
430  BOOLEAN - TRUE if access is allowed and FALSE otherwise
431 
432 --*/
433 
434 {
435  ACCESS_MASK GrantedAccess = 0;
436  BOOLEAN AccessAllowed;
437  BOOLEAN MemoryAllocated;
438  NTSTATUS Status;
439  PSECURITY_DESCRIPTOR SecurityDescriptor = NULL;
440  POBJECT_HEADER ObjectHeader;
441  POBJECT_TYPE ObjectType;
442  PPRIVILEGE_SET Privileges = NULL;
443 
444  PAGED_CODE();
445 
446  UNREFERENCED_PARAMETER (TypeMutexLocked);
447 
448  //
449  // Map the object body to an object header and the
450  // corresponding object type
451  //
452 
453  ObjectHeader = OBJECT_TO_OBJECT_HEADER( Object );
454  ObjectType = ObjectHeader->Type;
455 
456  //
457  // Obtain the object's security descriptor
458  //
459 
460  Status = ObGetObjectSecurity( Object,
461  &SecurityDescriptor,
462  &MemoryAllocated );
463 
464  //
465  // If we failed in getting the security descriptor then
466  // put the object type lock back where it was and return
467  // the error back to our caller
468  //
469 
470  if (!NT_SUCCESS( Status )) {
471 
472  *AccessStatus = Status;
473 
474  return( FALSE );
475 
476  } else {
477 
478  //
479  // Otherwise we've been successful at getting the
480  // object's security descriptor, but now make sure
481  // it is not null.
482 
483  if (SecurityDescriptor == NULL) {
484 
485  *AccessStatus = Status;
486 
487  return(TRUE);
488  }
489  }
490 
491  //
492  // We have a non-null security descriptor so now
493  // lock the caller's tokens until after auditing has been
494  // performed.
495  //
496 
497  SeLockSubjectContext( &AccessState->SubjectSecurityContext );
498 
499  //
500  // Do the access check, and if we have some privileges then
501  // put those in the access state too.
502  //
503 
504  AccessAllowed = SeAccessCheck( SecurityDescriptor,
505  &AccessState->SubjectSecurityContext,
506  TRUE, // Tokens are locked
507  AccessState->RemainingDesiredAccess,
508  AccessState->PreviouslyGrantedAccess,
509  &Privileges,
510  &ObjectType->TypeInfo.GenericMapping,
511  AccessMode,
512  &GrantedAccess,
513  AccessStatus );
514 
515  if (Privileges != NULL) {
516 
517  Status = SeAppendPrivileges( AccessState,
518  Privileges );
519 
520  SeFreePrivileges( Privileges );
521  }
522 
523  //
524  // If we were granted access then set that fact into
525  // what we've been granted and remove it from what remains
526  // to be granted.
527  //
528 
529  if (AccessAllowed) {
530 
531  AccessState->PreviouslyGrantedAccess |= GrantedAccess;
532  AccessState->RemainingDesiredAccess &= ~(GrantedAccess | MAXIMUM_ALLOWED);
533  }
534 
535  //
536  // Audit the attempt to open the object, audit
537  // the creation of its handle later.
538  //
539 
540  if ( SecurityDescriptor != NULL ) {
541 
542  SeOpenObjectAuditAlarm( &ObjectType->Name,
543  Object,
544  NULL, // AbsoluteObjectName
545  SecurityDescriptor,
546  AccessState,
547  FALSE, // ObjectCreated (FALSE, only open here)
548  AccessAllowed,
549  AccessMode,
550  &AccessState->GenerateOnClose );
551  }
552 
553  SeUnlockSubjectContext( &AccessState->SubjectSecurityContext );
554 
555  //
556  // Free the security descriptor before returning to
557  // our caller
558  //
559 
560  ObReleaseObjectSecurity( SecurityDescriptor,
561  MemoryAllocated );
562 
563  return( AccessAllowed );
564 }

568 ObpCheckObjectReference (
569  IN PVOID Object,
570  IN OUT PACCESS_STATE AccessState,
571  IN BOOLEAN TypeMutexLocked,
572  IN KPROCESSOR_MODE AccessMode,
573  OUT PNTSTATUS AccessStatus
574  )
575 
576 /*++
577 
578 Routine Description:
579 
580  The routine performs access validation on the passed object. The
581  remaining desired access mask is extracted from the AccessState
582  parameter and passes to the appropriate security routine to
583  perform the access check.
584 
585  If the access attempt is successful, SeAccessCheck returns a mask
586  containing the granted accesses. The bits in this mask are turned
587  on in the PreviouslyGrantedAccess field of the AccessState, and
588  are turned off in the RemainingDesiredAccess field.
589 
590  This routine differs from ObpCheckObjectAccess in that it calls
591  a different audit routine.
592 
593 Arguments:
594 
595  Object - The object being examined.
596 
597  AccessState - The ACCESS_STATE structure containing accumulated
598  information about the current attempt to gain access to the object.
599 
600  TypeMutexLocked - Indicates whether the type mutex for this object's
601  type is locked. The type mutex is used to protect the object's
602  security descriptor from being modified while it is being accessed.
603 
604  AccessMode - The previous processor mode.
605 
606  AccessStatus - Pointer to a variable to return the status code of the
607  access attempt. In the case of failure this status code must be
608  propagated back to the user.
609 
610 
611 Return Value:
612 
613  BOOLEAN - TRUE if access is allowed and FALSE otherwise
614 
615 --*/
616 
617 {
618  BOOLEAN AccessAllowed;
619  ACCESS_MASK GrantedAccess = 0;
620  BOOLEAN MemoryAllocated;
621  PSECURITY_DESCRIPTOR SecurityDescriptor;
622  NTSTATUS Status;
623  POBJECT_HEADER ObjectHeader;
624  POBJECT_TYPE ObjectType;
625  PPRIVILEGE_SET Privileges = NULL;
626 
627  PAGED_CODE();
628 
629  UNREFERENCED_PARAMETER (TypeMutexLocked);
630 
631  //
632  // Map the object body to an object header and the
633  // corresponding object type
634  //
635 
636  ObjectHeader = OBJECT_TO_OBJECT_HEADER( Object );
637  ObjectType = ObjectHeader->Type;
638 
639  //
640  // Obtain the object's security descriptor
641  //
642 
643  Status = ObGetObjectSecurity( Object,
644  &SecurityDescriptor,
645  &MemoryAllocated );
646 
647  //
648  // If we failed in getting the security descriptor then
649  // put the object type lock back where it was and return
650  // the error back to our caller
651  //
652 
653  if (!NT_SUCCESS( Status )) {
654 
655  *AccessStatus = Status;
656 
657  return( FALSE );
658  }
659 
660  //
661  // Lock the caller's tokens until after auditing has been
662  // performed.
663  //
664 
665  SeLockSubjectContext( &AccessState->SubjectSecurityContext );
666 
667  //
668  // Do the access check, and if we have some privileges then
669  // put those in the access state too.
670  //
671 
672  AccessAllowed = SeAccessCheck( SecurityDescriptor,
673  &AccessState->SubjectSecurityContext,
674  TRUE, // Tokens are locked
675  AccessState->RemainingDesiredAccess,
676  AccessState->PreviouslyGrantedAccess,
677  &Privileges,
678  &ObjectType->TypeInfo.GenericMapping,
679  AccessMode,
680  &GrantedAccess,
681  AccessStatus );
682 
683  if (AccessAllowed) {
684 
685  AccessState->PreviouslyGrantedAccess |= GrantedAccess;
686  AccessState->RemainingDesiredAccess &= ~GrantedAccess;
687  }
688 
689  //
690  // If we have a security descriptor then call the security routine
691  // to audit this reference and then unlock the caller's token
692  //
693 
694  if ( SecurityDescriptor != NULL ) {
695 
696  SeObjectReferenceAuditAlarm( &AccessState->OperationID,
697  Object,
698  SecurityDescriptor,
699  &AccessState->SubjectSecurityContext,
700  AccessState->RemainingDesiredAccess | AccessState->PreviouslyGrantedAccess,
701  ((PAUX_ACCESS_DATA)(AccessState->AuxData))->PrivilegesUsed,
702  AccessAllowed,
703  AccessMode );
704  }
705 
706  SeUnlockSubjectContext( &AccessState->SubjectSecurityContext );
707 
708  //
709  // Finally free the security descriptor
710  // and return to our caller
711  //
712 
713  ObReleaseObjectSecurity( SecurityDescriptor,
714  MemoryAllocated );
715 
716  return( AccessAllowed );
717 }

720 BOOLEAN
721 ObpCheckTraverseAccess (
722  IN PVOID DirectoryObject,
723  IN ACCESS_MASK TraverseAccess,
724  IN PACCESS_STATE AccessState,
725  IN BOOLEAN TypeMutexLocked,
726  IN KPROCESSOR_MODE PreviousMode,
727  OUT PNTSTATUS AccessStatus
728  )
729 
730 /*++
731 
732 Routine Description:
733 
734  This routine checks for traverse access to the given directory object.
735 
736  Note that the contents of the AccessState structure are not
737  modified, since it is assumed that this access check is incidental
738  to another access operation.
739 
740 Arguments:
741 
742  DirectoryObject - The object body of the object being examined.
743 
744  TraverseAccess - The desired access to the object, most likely DIRECTORY
745  TRAVERSE access.
746 
747  AccessState - Checks for traverse access will typically be incidental
748  to some other access attempt. Information on the current state of
749  that access attempt is required so that the constituent access
750  attempts may be associated with each other in the audit log.
751 
752  TypeMutexLocked - Indicates whether the type mutex for this object's
753  type is locked. The type mutex is used to protect the object's
754  security descriptor from being modified while it is being accessed.
755 
756  PreviousMode - The previous processor mode.
757 
758  AccessStatus - Pointer to a variable to return the status code of the
759  access attempt. In the case of failure this status code must be
760  propagated back to the user.
761 
762 Return Value:
763 
764  BOOLEAN - TRUE if access is allowed and FALSE otherwise. AccessStatus
765  contains the status code to be passed back to the caller. It is not
766  correct to simply pass back STATUS_ACCESS_DENIED, since this will have
767  to change with the advent of mandatory access control.
768 
769 --*/
770 
771 {
772  BOOLEAN AccessAllowed;
773  ACCESS_MASK GrantedAccess = 0;
774  PSECURITY_DESCRIPTOR SecurityDescriptor;
775  BOOLEAN MemoryAllocated;
776  NTSTATUS Status;
777  POBJECT_HEADER ObjectHeader;
778  POBJECT_TYPE ObjectType;
779  PPRIVILEGE_SET Privileges = NULL;
780 
781  PAGED_CODE();
782 
783  UNREFERENCED_PARAMETER (TypeMutexLocked);
784 
785  //
786  // Map the object body to an object header and corresponding
787  // object type
788  //
789 
790  ObjectHeader = OBJECT_TO_OBJECT_HEADER( DirectoryObject );
791  ObjectType = ObjectHeader->Type;
792 
793  //
794  // Obtain the object's security descriptor and make it was
795  // successful
796  //
797 
798  Status = ObGetObjectSecurity( DirectoryObject,
799  &SecurityDescriptor,
800  &MemoryAllocated );
801 
802  if (!NT_SUCCESS( Status )) {
803 
804  *AccessStatus = Status;
805 
806  return( FALSE );
807  }
808 
809  if (!SeFastTraverseCheck( SecurityDescriptor,
810  AccessState,
811  DIRECTORY_TRAVERSE,
812  PreviousMode )) {
813 
814  //
815  // SeFastTraverseCheck could be modified to tell us that
816  // no one has any access to this directory. However,
817  // we're going to have to fail this entire call if
818  // that is the case, so we really don't need to worry
819  // all that much about making it blindingly fast.
820  //
821  // The world does not have traverse access and we have
822  // the client's access state so lock down the client's
823  // token and then do the access check, appending privileges
824  // if present. The access check will give the answer
825  // we return back to our caller
826  //
827 
828  SeLockSubjectContext( &AccessState->SubjectSecurityContext );
829 
830  AccessAllowed = SeAccessCheck( SecurityDescriptor,
831  &AccessState->SubjectSecurityContext,
832  TRUE, // Tokens are locked
833  TraverseAccess,
834  0,
835  &Privileges,
836  &ObjectType->TypeInfo.GenericMapping,
837  PreviousMode,
838  &GrantedAccess,
839  AccessStatus );
840 
841  if (Privileges != NULL) {
842 
843  Status = SeAppendPrivileges( AccessState,
844  Privileges );
845 
846  SeFreePrivileges( Privileges );
847  }
848 
849  //
850  // If the client's token is locked then now we can unlock it
851  //
852 
853  SeUnlockSubjectContext( &AccessState->SubjectSecurityContext );
854 
855  } else {
856 
857  //
858  // At this point the world has traverse access
859  //
860 
861  AccessAllowed = TRUE;
862  }
863 
864  //
865  // Finally free the security descriptor
866  // and then return to our caller
867  //
868 
869  ObReleaseObjectSecurity( SecurityDescriptor,
870  MemoryAllocated );
871 
872  return( AccessAllowed );
873 }
