222 VOID
223 KeSetSystemTime (
224  IN PLARGE_INTEGER NewTime,
225  OUT PLARGE_INTEGER OldTime,
226  IN BOOLEAN AdjustInterruptTime,
227  IN PLARGE_INTEGER HalTimeToSet OPTIONAL
228  )
229 
230 /*++
231 
232 Routine Description:
233 
234  This function sets the system time to the specified value and updates
235  timer queue entries to reflect the difference between the old system
236  time and the new system time.
237 
238 Arguments:
239 
240  NewTime - Supplies a pointer to a variable that specifies the new system
241  time.
242 
243  OldTime - Supplies a pointer to a variable that will receive the previous
244  system time.
245 
246  AdjustInterruptTime - If TRUE the amount of time being adjusted is
247  also applied to InterruptTime and TickCount.
248 
249  HalTimeToSet - Supplies an optional time that if specified is to be used
250  to set the time in the realtime clock.
251 
252 Return Value:
253 
254  None.
255 
256 --*/
257 
258 {
259 
260  LIST_ENTRY AbsoluteListHead;
261  LIST_ENTRY ExpiredListHead;
262  ULONG Hand;
263  ULONG Index;
264  PLIST_ENTRY ListHead;
265  PKSPIN_LOCK_QUEUE LockQueue;
266  PLIST_ENTRY NextEntry;
267  KIRQL OldIrql1;
268  KIRQL OldIrql2;
269  LARGE_INTEGER TimeDelta;
270  TIME_FIELDS TimeFields;
271  PKTIMER Timer;
272 
273  ASSERT((NewTime->HighPart & 0xf0000000) == 0);
274 
275  ASSERT(KeGetCurrentIrql() < DISPATCH_LEVEL);
276 
277  //
278  // If a realtime clock value is specified, then convert the time value
279  // to time fields.
280  //
281 
282  if (ARGUMENT_PRESENT(HalTimeToSet)) {
283  RtlTimeToTimeFields(HalTimeToSet, &TimeFields);
284  }
285 
286  //
287  // Set affinity to the processor that keeps the system time, raise IRQL
288  // to dispatcher level and lock the dispatcher database, then raise IRQL
289  // to HIGH_LEVEL to synchronize with the clock interrupt routine.
290  //
291 
292  KeSetSystemAffinityThread((KAFFINITY)1);
293  KiLockDispatcherDatabase(&OldIrql1);
294  KeRaiseIrql(HIGH_LEVEL, &OldIrql2);
295 
296  //
297  // Save the previous system time, set the new system time, and set
298  // the realtime clock, if a time value is specified.
299  //
300 
301  KiQuerySystemTime(OldTime);
302 
303 #if defined(_AMD64_)
304 
305  SharedUserData->SystemTime.High2Time = NewTime->HighPart;
306  *((volatile ULONG64 *)&SharedUserData->SystemTime) = NewTime->QuadPart;
307 
308 #else
309 
310  SharedUserData->SystemTime.High2Time = NewTime->HighPart;
311  SharedUserData->SystemTime.LowPart = NewTime->LowPart;
312  SharedUserData->SystemTime.High1Time = NewTime->HighPart;
313 
314 #endif
315 
316  if (ARGUMENT_PRESENT(HalTimeToSet)) {
317  ExCmosClockIsSane = HalSetRealTimeClock(&TimeFields);
318  }
319 
320  //
321  // Compute the difference between the previous system time and the new
322  // system time.
323  //
324 
325  TimeDelta.QuadPart = NewTime->QuadPart - OldTime->QuadPart;
326 
327  //
328  // Update the boot time to reflect the delta. This keeps time based
329  // on boot time constant
330  //
331 
332  KeBootTime.QuadPart = KeBootTime.QuadPart + TimeDelta.QuadPart;
333 
334  //
335  // Track the overall bias applied to the boot time.
336  //
337 
338  KeBootTimeBias = KeBootTimeBias + TimeDelta.QuadPart;
339 
340  //
341  // Lower IRQL to dispatch level and if needed adjust the physical
342  // system interrupt time.
343  //
344 
345  KeLowerIrql(OldIrql2);
346  if (AdjustInterruptTime != FALSE) {
347  AdjustInterruptTime = KeAdjustInterruptTime(TimeDelta.QuadPart);
348  }
349 
350  //
351  // If the physical interrupt time of the system was not adjusted, then
352  // recompute any absolute timers in the system for the new system time.
353  //
354 
355  if (AdjustInterruptTime == FALSE) {
356 
357  //
358  // Acquire the timer table lock, remove all absolute timers from the
359  // timer queue so their due time can be recomputed, and release the
360  // timer table lock.
361  //
362 
363  InitializeListHead(&AbsoluteListHead);
364  for (Index = 0; Index < TIMER_TABLE_SIZE; Index += 1) {
365  ListHead = &KiTimerTableListHead[Index].Entry;
366  LockQueue = KiAcquireTimerTableLock(Index);
367  NextEntry = ListHead->Flink;
368  while (NextEntry != ListHead) {
369  Timer = CONTAINING_RECORD(NextEntry, KTIMER, TimerListEntry);
370  NextEntry = NextEntry->Flink;
371  if (Timer->Header.Absolute != FALSE) {
372  KiRemoveEntryTimer(Timer);
373  InsertTailList(&AbsoluteListHead, &Timer->TimerListEntry);
374  }
375  }
376 
377  KiReleaseTimerTableLock(LockQueue);
378  }
379 
380  //
381  // Recompute the due time and reinsert all absolute timers in the timer
382  // tree. If a timer has already expired, then insert the timer in the
383  // expired timer list.
384  //
385 
386  InitializeListHead(&ExpiredListHead);
387  while (AbsoluteListHead.Flink != &AbsoluteListHead) {
388  Timer = CONTAINING_RECORD(AbsoluteListHead.Flink, KTIMER, TimerListEntry);
389  RemoveEntryList(&Timer->TimerListEntry);
390  Timer->DueTime.QuadPart -= TimeDelta.QuadPart;
391  Hand = KiComputeTimerTableIndex(Timer->DueTime.QuadPart);
392  Timer->Header.Hand = (UCHAR)Hand;
393  LockQueue = KiAcquireTimerTableLock(Hand);
394  if (KiInsertTimerTable(Timer, Hand) == TRUE) {
395  KiRemoveEntryTimer(Timer);
396  InsertTailList(&ExpiredListHead, &Timer->TimerListEntry);
397  }
398 
399  KiReleaseTimerTableLock(LockQueue);
400  }
401 
402  //
403  // If any of the attempts to reinsert a timer failed, then timers have
404  // already expired and must be processed.
405  //
406  // N.B. The following function returns with the dispatcher database
407  // unlocked.
408  //
409 
410  KiTimerListExpire(&ExpiredListHead, OldIrql1);
411 
412  } else {
413  KiUnlockDispatcherDatabase(OldIrql1);
414  }
415 
416  //
417  // Set affinity back to its original value.
418  //
419 
420  KeRevertToUserAffinityThread();
421  return;
422 }

424 BOOLEAN
425 KeAdjustInterruptTime (
426  IN LONGLONG TimeDelta
427  )
428 
429 /*++
430 
431 Routine Description:
432 
433  This function moves the physical interrupt time of the system forward by
434  the specified time delta after a system wake has occurred.
435 
436 Arguments:
437 
438  TimeDelta - Supplies the time delta to be added to the interrupt time, tick
439  count and the performance counter in 100ns units.
440 
441 Return Value:
442 
443  None.
444 
445 --*/
446 
447 {
448 
449  ADJUST_INTERRUPT_TIME_CONTEXT Adjust;
450 
451  //
452  // Time can only be moved forward.
453  //
454 
455  if (TimeDelta < 0) {
456  return FALSE;
457 
458  } else {
459  Adjust.KiNumber = KeNumberProcessors;
460  Adjust.HalNumber = KeNumberProcessors;
461  Adjust.Adjustment = (ULONGLONG) TimeDelta;
462  Adjust.Barrier = 1;
463  KeIpiGenericCall((PKIPI_BROADCAST_WORKER)KiCalibrateTimeAdjustment,
464  (ULONG_PTR)(&Adjust));
465 
466  return TRUE;
467  }
468 }

470 VOID
471 KiCalibrateTimeAdjustment (
472  PADJUST_INTERRUPT_TIME_CONTEXT Adjust
473  )
474 
475 /*++
476 
477 Routine Description:
478 
479  This function calibrates the adjustment of time on all processors.
480 
481 Arguments:
482 
483  Adjust - Supplies the operation context.
484 
485 Return Value:
486 
487  None.
488 
489 --*/
490 
491 {
492 
493  ULONG cl;
494  ULONG divisor;
495  BOOLEAN Enable;
496  LARGE_INTEGER InterruptTime;
497  ULARGE_INTEGER li;
498  PERFINFO_PO_CALIBRATED_PERFCOUNTER LogEntry;
499  LARGE_INTEGER NewTickCount;
500  ULONG NewTickOffset;
501  LARGE_INTEGER PerfCount;
502  LARGE_INTEGER PerfFreq;
503  LARGE_INTEGER SetTime;
504 
505  //
506  // As each processor arrives, decrement the remaining processor count. If
507  // this is the last processor to arrive, then compute the time change, and
508  // signal all processor when to apply the performance counter change.
509  //
510 
511  if (InterlockedDecrement((PLONG)&Adjust->KiNumber)) {
512  Enable = KeDisableInterrupts();
513 
514  //
515  // It is possible to deadlock if one or more of the other processors
516  // receives and processes a freeze request while this processor has
517  // interrupts disabled. Poll for a freeze request until all processors
518  // are known to be in this code.
519  //
520 
521  do {
522  KiPollFreezeExecution();
523  } while (Adjust->KiNumber != (ULONG)-1);
524 
525  //
526  // Wait to perform the time set.
527  //
528 
529  while (Adjust->Barrier);
530 
531  } else {
532 
533  //
534  // Set timer expiration dpc to scan the timer queues once for any
535  // expired timers.
536  //
537 
538  KeRemoveQueueDpc(&KiTimerExpireDpc);
539  KeInsertQueueDpc(&KiTimerExpireDpc,
540  ULongToPtr(KiQueryLowTickCount() - TIMER_TABLE_SIZE),
541  NULL);
542 
543  //
544  // Disable interrupts and indicate that this processor is now
545  // in final portion of this code.
546  //
547 
548  Enable = KeDisableInterrupts();
549  InterlockedDecrement((PLONG) &Adjust->KiNumber);
550 
551  //
552  // Adjust Interrupt Time.
553  //
554 
555  InterruptTime.QuadPart = KeQueryInterruptTime() + Adjust->Adjustment;
556  SetTime.QuadPart = Adjust->Adjustment;
557 
558  //
559  // Get the current times
560  //
561 
562  PerfCount = KeQueryPerformanceCounter(&PerfFreq);
563 
564  //
565  // Compute performance counter for current time.
566  //
567  // Multiply SetTime * PerfCount and obtain 96-bit result in cl,
568  // li.LowPart, li.HighPart. Then divide the 96-bit result by
569  // 10,000,000 to get new performance counter value.
570  //
571 
572  li.QuadPart = RtlEnlargedUnsignedMultiply((ULONG)SetTime.LowPart,
573  (ULONG)PerfFreq.LowPart).QuadPart;
574 
575  cl = li.LowPart;
576  li.QuadPart =
577  li.HighPart + RtlEnlargedUnsignedMultiply((ULONG)SetTime.LowPart,
578  (ULONG)PerfFreq.HighPart).QuadPart;
579 
580  li.QuadPart =
581  li.QuadPart + RtlEnlargedUnsignedMultiply((ULONG)SetTime.HighPart,
582  (ULONG)PerfFreq.LowPart).QuadPart;
583 
584  li.HighPart = li.HighPart + SetTime.HighPart * PerfFreq.HighPart;
585  divisor = 10000000;
586  Adjust->NewCount.HighPart = RtlEnlargedUnsignedDivide(li,
587  divisor,
588  &li.HighPart);
589 
590  li.LowPart = cl;
591  Adjust->NewCount.LowPart = RtlEnlargedUnsignedDivide(li,
592  divisor,
593  NULL);
594 
595  Adjust->NewCount.QuadPart += PerfCount.QuadPart;
596 
597  //
598  // Compute tick count and tick offset for current interrupt time.
599  //
600 
601  NewTickCount = RtlExtendedLargeIntegerDivide(InterruptTime,
602  KeMaximumIncrement,
603  &NewTickOffset);
604 
605  //
606  // Apply changes to interrupt time, tick count, tick offset, and the
607  // performance counter.
608  //
609 
610  KiTickOffset = KeMaximumIncrement - NewTickOffset;
611  KeInterruptTimeBias += Adjust->Adjustment;
612  SharedUserData->TickCount.High2Time = NewTickCount.HighPart;
613 
614 #if defined(_WIN64)
615 
616  SharedUserData->TickCountQuad = NewTickCount.QuadPart;
617 
618 #else
619 
620  SharedUserData->TickCount.LowPart = NewTickCount.LowPart;
621  SharedUserData->TickCount.High1Time = NewTickCount.HighPart;
622 
623 #endif
624 
625  //
626  // N.B. There is no global tick count variable on AMD64.
627  //
628 
629 #if defined(_X86_)
630 
631  KeTickCount.High2Time = NewTickCount.HighPart;
632  KeTickCount.LowPart = NewTickCount.LowPart;
633  KeTickCount.High1Time = NewTickCount.HighPart;
634 
635 #endif
636 
637 #if defined(_AMD64_)
638 
639  SharedUserData->InterruptTime.High2Time = InterruptTime.HighPart;
640  *((volatile ULONG64 *)&SharedUserData->InterruptTime) = InterruptTime.QuadPart;
641 
642 #else
643 
644  SharedUserData->InterruptTime.High2Time = InterruptTime.HighPart;
645  SharedUserData->InterruptTime.LowPart = InterruptTime.LowPart;
646  SharedUserData->InterruptTime.High1Time = InterruptTime.HighPart;
647 
648 #endif
649 
650  //
651  // Apply the performance counter change.
652  //
653 
654  Adjust->Barrier = 0;
655  }
656 
657  KeGetCurrentPrcb()->TickOffset = KiTickOffset;
658 
659 #if defined(_AMD64_)
660 
661  KeGetCurrentPrcb()->MasterOffset = KiTickOffset;
662 
663 #endif
664 
665  HalCalibratePerformanceCounter((LONG volatile *)&Adjust->HalNumber,
666  (ULONGLONG)Adjust->NewCount.QuadPart);
667 
668  //
669  // Log an event that the performance counter has been calibrated
670  // properly and indicate the new performance counter value.
671  //
672 
673  if (PERFINFO_IS_GROUP_ON(PERF_POWER)) {
674  LogEntry.PerformanceCounter = KeQueryPerformanceCounter(NULL);
675  PerfInfoLogBytes(PERFINFO_LOG_TYPE_PO_CALIBRATED_PERFCOUNTER,
676  &LogEntry,
677  sizeof(LogEntry));
678  }
679 
680  KeEnableInterrupts(Enable);
681  return;
682 }

728 BOOLEAN
729 KeAddSystemServiceTable (
730  IN PULONG_PTR Base,
731  IN PULONG Count OPTIONAL,
732  IN ULONG Limit,
733  IN PUCHAR Number,
734  IN ULONG Index
735  )
736 
737 /*++
738 
739 Routine Description:
740 
741  This function adds the specified system service table to the system.
742 
743 Arguments:
744 
745  Base - Supplies the address of the system service table dispatch table.
746 
747  Count - Supplies an optional pointer to a table of per system service
748  counters.
749 
750  Limit - Supplies the limit of the service table. Services greater
751  than or equal to this limit will fail.
752 
753  Arguments - Supplies the address of the argument count table.
754 
755  Index - Supplies index of the service table.
756 
757 Return Value:
758 
759  TRUE - The operation was successful.
760 
761  FALSE - the operation failed. A service table is already bound to
762  the specified location, or the specified index is larger than
763  the maximum allowed index.
764 
765 --*/
766 
767 {
768 
769  PAGED_CODE();
770 
771  //
772  // If a system service table is already defined for the specified
773  // index, then return FALSE. Otherwise, establish the new system
774  // service table.
775  //
776 
777  if ((Index > NUMBER_SERVICE_TABLES - 1) ||
778  (KeServiceDescriptorTable[Index].Base != NULL) ||
779  (KeServiceDescriptorTableShadow[Index].Base != NULL)) {
780 
781  return FALSE;
782 
783  } else {
784 
785  //
786  // If the service table index is equal to the Win32 table, then
787  // only update the shadow system service table. Otherwise, both
788  // the shadow and static system service tables are updated.
789  //
790 
791  KeServiceDescriptorTableShadow[Index].Base = Base;
792  KeServiceDescriptorTableShadow[Index].Count = Count;
793  KeServiceDescriptorTableShadow[Index].Limit = Limit;
794  KeServiceDescriptorTableShadow[Index].Number = Number;
795  if (Index != WIN32K_SERVICE_INDEX) {
796  KeServiceDescriptorTable[Index].Base = Base;
797  KeServiceDescriptorTable[Index].Count = Count;
798  KeServiceDescriptorTable[Index].Limit = Limit;
799  KeServiceDescriptorTable[Index].Number = Number;
800  }
801 
802  return TRUE;
803  }
804 }

892 Routine Description:
