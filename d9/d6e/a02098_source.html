113 /*++
114 
115 Routine Description:
116 
117  This function opens a handle to the file associated with the processes
118  section. The file is opened such that it can be duplicated all the way to
119  the UI where the UI can either map the file or read the file to get
120  the debug info.
121 
122 Arguments:
123 
124  SectionObject - Supplies the section whose associated file is to be opened.
125 
126 Return Value:
127 
128  NULL - The file could not be opened.
129 
130  NON-NULL - Returns a handle to the file associated with the specified
131  section.
132 
133 --*/
134 
135 {
136  NTSTATUS Status;
137  OBJECT_ATTRIBUTES Obja;
138  IO_STATUS_BLOCK IoStatusBlock;
139  HANDLE Handle;
140  POBJECT_NAME_INFORMATION FileNameInfo;
141 
142  PAGED_CODE();
143 
144  Status = MmGetFileNameForSection(SectionObject, &FileNameInfo);
145  if ( !NT_SUCCESS(Status) ) {
146  return NULL;
147  }
148 
149  InitializeObjectAttributes(
150  &Obja,
151  &FileNameInfo->Name,
152  OBJ_CASE_INSENSITIVE | OBJ_FORCE_ACCESS_CHECK | OBJ_KERNEL_HANDLE,

154  NULL
155  );

470 DbgkExitThread(
471  NTSTATUS ExitStatus
472  )
473 
474 /*++
475 
476 Routine Description:
477 
478  This function is called when a new thread terminates. At this
479  point, the thread will no longer execute in user-mode. No other
480  exit processing has occured.
481 
482  If a message is sent, then while the thread is awaiting a reply,
483  all other threads in the process are suspended.
484 
485 Arguments:
486 
487  ExitStatus - Supplies the ExitStatus of the exiting thread.
488 
489 Return Value:
490 
491  None.
492 
493 --*/
494 
495 {
496  PVOID Port;
497  DBGKM_APIMSG m;
498  PDBGKM_EXIT_THREAD args;
499  PEPROCESS Process;
500  BOOLEAN Frozen;
501 
502  PAGED_CODE();
503 
504  Process = PsGetCurrentProcess();
505  if (PsGetCurrentThread()->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_HIDEFROMDBG) {
506  Port = NULL;
507  } else {
508  Port = Process->DebugPort;
509  }
510 
511  if ( !Port ) {
512  return;
513  }
514 
515  if (PsGetCurrentThread()->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_DEADTHREAD) {
516  return;
517  }
518 
519  args = &m.u.ExitThread;
520  args->ExitStatus = ExitStatus;
521 
522  DBGKM_FORMAT_API_MSG(m,DbgKmExitThreadApi,sizeof(*args));
523 
524  Frozen = DbgkpSuspendProcess();
525 
526  DbgkpSendApiMessage(&m,FALSE);
527 
528  if (Frozen) {
529  DbgkpResumeProcess();
530  }
531 }
