74 /*++
75 
76 Routine Description:
77 
78  Registers a new callback.
79 
80 Arguments:
81 
82 
83 
84 Return Value:
85 
86 
87 --*/
88 {
89  PEX_CALLBACK_ROUTINE_BLOCK RoutineBlock;
90  ULONG i;
91  PCM_CALLBACK_CONTEXT_BLOCK CmCallbackContext;
92 
93  PAGED_CODE();
94 
95  CmCallbackContext = (PCM_CALLBACK_CONTEXT_BLOCK)ExAllocatePoolWithTag (PagedPool,
96  sizeof (CM_CALLBACK_CONTEXT_BLOCK),
97  'bcMC');
98  if( CmCallbackContext == NULL ) {
99  return STATUS_INSUFFICIENT_RESOURCES;
100  }
101 
102  RoutineBlock = ExAllocateCallBack (Function,CmCallbackContext);
103  if( RoutineBlock == NULL ) {
104  ExFreePool(CmCallbackContext);
105  return STATUS_INSUFFICIENT_RESOURCES;
106  }
107 
108  //
109  // init the context
110  //
111  KeQuerySystemTime(&(CmCallbackContext->Cookie));
112  *Cookie = CmCallbackContext->Cookie;
113  InitializeListHead(&(CmCallbackContext->ThreadListHead));
114  ExInitializePushLock(&(CmCallbackContext->ThreadListLock));
115  CmCallbackContext->CallerContext = Context;
116 
117  //
118  // find a spot where we could add this callback
119  //
120  for( i=0;i<CM_MAX_CALLBACKS;i++) {
121  if( ExCompareExchangeCallBack (&CmpCallBackVector[i],RoutineBlock,NULL) ) {
122  InterlockedExchangeAdd ((PLONG) &CmpCallBackCount, 1);
123  return STATUS_SUCCESS;
124  }
125  }
126 
127  //
128  // no more callbacks
129  //
130  ExFreePool(CmCallbackContext);
131  ExFreeCallBack(RoutineBlock);
132  return STATUS_INSUFFICIENT_RESOURCES;
133 }

138 /*++
139 
140 Routine Description:
141 
142  Unregisters a callback.
143 
144 Arguments:
145 
146 Return Value:
147 
148 
149 --*/
150 {
151  ULONG i;
152  PCM_CALLBACK_CONTEXT_BLOCK CmCallbackContext;
153  PEX_CALLBACK_ROUTINE_BLOCK RoutineBlock;
154 
155  PAGED_CODE();
156 
157  //
158  // Search for this cookie
159  //
160  for( i=0;i<CM_MAX_CALLBACKS;i++) {
161  RoutineBlock = ExReferenceCallBackBlock(&(CmpCallBackVector[i]) );
162  if( RoutineBlock ) {
163  CmCallbackContext = (PCM_CALLBACK_CONTEXT_BLOCK)ExGetCallBackBlockContext(RoutineBlock);
164  if( CmCallbackContext && (CmCallbackContext->Cookie.QuadPart == Cookie.QuadPart) ) {
165  //
166  // found it
167  //
168  if( ExCompareExchangeCallBack (&CmpCallBackVector[i],NULL,RoutineBlock) ) {
169  InterlockedExchangeAdd ((PLONG) &CmpCallBackCount, -1);
170 
171  ExDereferenceCallBackBlock (&(CmpCallBackVector[i]),RoutineBlock);
172  //
173  // wait for others to release their reference, then tear down the structure
174  //
175  ExWaitForCallBacks (RoutineBlock);
176 
177  ExFreePool(CmCallbackContext);
178  ExFreeCallBack(RoutineBlock);
179  return STATUS_SUCCESS;
180  }
181 
182  } else {
183  ExDereferenceCallBackBlock (&(CmpCallBackVector[i]),RoutineBlock);
184  }
185  }
186 
187  }
188 
189  return STATUS_INVALID_PARAMETER;
190 }

195 NTSTATUS
196 CmpCallCallBacks (
197  IN REG_NOTIFY_CLASS Type,
198  IN PVOID Argument,
199  IN BOOLEAN Wind,
200  IN REG_NOTIFY_CLASS PostType,
201  IN PVOID Object
202  )
203 /*++
204 
205 Routine Description:
206 
207  This function calls the callback thats inside a callback structure
208 
209 Arguments:
210 
211  Type - Nt call selector
212 
213  Argument - Caller provided argument to pass on (one of the REG_*_INFORMATION )
214 
215  Wind - tells if this is a pre or a post callback
216 
217  PostType - matching post notify class
218 
219  Object - to be used in case we fail part through.
220 
221 Return Value:
222 
223  NTSTATUS - STATUS_SUCCESS or error status returned by the first callback
224 
225 --*/
226 {
227  NTSTATUS Status = STATUS_SUCCESS;
228  LONG i;
229  LONG Direction;
230  PEX_CALLBACK_ROUTINE_BLOCK RoutineBlock;
231  PCM_CALLBACK_CONTEXT_BLOCK CmCallbackContext;
232  BOOLEAN InternalUnWind = FALSE;
233 
234  PAGED_CODE();
235 
236  if( Wind == TRUE ) {
237  Direction = 1;
238  i = 0;
239  } else {
240  Direction = -1;
241  i = CM_MAX_CALLBACKS - 1;
242  }
243  for(;(i >= 0) && (i < CM_MAX_CALLBACKS);i += Direction) {
244  RoutineBlock = ExReferenceCallBackBlock(&(CmpCallBackVector[i]) );
245  if( RoutineBlock != NULL ) {
246  //
247  // we have a safe reference on this block.
248  //
249  //
250  // record thread on a stack struct, so we don't need to allocate pool for it. We unlink
251  // it from our lists prior to this function exit, so we are on the safe side.
252  //
253  CM_ACTIVE_NOTIFY_THREAD ActiveThreadInfo;
254 
255  //
256  // get context info
257  //
258  CmCallbackContext = (PCM_CALLBACK_CONTEXT_BLOCK)ExGetCallBackBlockContext(RoutineBlock);
259  ASSERT( CmCallbackContext != NULL );
260 
261  ActiveThreadInfo.Thread = PsGetCurrentThread();
262 #if DBG
263  InitializeListHead(&(ActiveThreadInfo.ThreadList));
264 #endif //DBG
265 
266  if( CmpCheckRecursionAndRecordThreadInfo(CmCallbackContext,&ActiveThreadInfo) ) {
267  if( InternalUnWind == TRUE ) {
268  //
269  // this is an internal unwind due to fail part through. need to call posts manually
270  //
271  REG_POST_OPERATION_INFORMATION PostInfo;
272  PostInfo.Object = Object;
273  PostInfo.Status = Status;
274  ExGetCallBackBlockRoutine(RoutineBlock)(CmCallbackContext->CallerContext,(PVOID)(ULONG_PTR)PostType,&PostInfo);
275  // ignore return status.
276  } else {
277  //
278  // regular callback call
279  //
280  Status = ExGetCallBackBlockRoutine(RoutineBlock)(CmCallbackContext->CallerContext,(PVOID)(ULONG_PTR)Type,Argument);
281  if( Wind == FALSE ) {
282  //
283  // always ignore post return calls so they all get a chance to run
284  //
285  Status = STATUS_SUCCESS;
286  }
287  }
288  //
289  // now that we're down, remove ourselves from the thread list
290  //
291  CmpLockContext(CmCallbackContext);
292  RemoveEntryList(&(ActiveThreadInfo.ThreadList));
293  CmpUnlockContext(CmCallbackContext);
294  } else {
295  ASSERT( IsListEmpty(&(ActiveThreadInfo.ThreadList)) );
296  }
297 
298  ExDereferenceCallBackBlock (&(CmpCallBackVector[i]),RoutineBlock);
299 
300  if( !NT_SUCCESS(Status) ) {
301  if( Wind == TRUE ) {
302  //
303  // switch direction and start calling the posts
304  //
305  Wind = FALSE;
306  Direction *= -1;
307  InternalUnWind = TRUE;
308  } else if( InternalUnWind == FALSE ) {
309  return Status;
310  }
311  //
312  // else fall through so the unwind completes
313  //
314  }
315  }
316  }
317 
318  return Status;
319 }

353 /*++
354 
355 Routine Description:
356 
357  Checks if current thread is already inside the callback (recursion avoidance)
358 
359 Arguments:
360 
361 
362 Return Value:
363 
364 
365 --*/
366 {
367  PLIST_ENTRY AnchorAddr;
368  PCM_ACTIVE_NOTIFY_THREAD CurrentThreadInfo;
369 
370  PAGED_CODE();
371 
372  CmpLockContext(CallbackBlock);
373  //
374  // walk the ActiveThreadList and see if we are already active
375  //
376  AnchorAddr = &(CallbackBlock->ThreadListHead);
377  CurrentThreadInfo = (PCM_ACTIVE_NOTIFY_THREAD)(CallbackBlock->ThreadListHead.Flink);
378 
379  while ( CurrentThreadInfo != (PCM_ACTIVE_NOTIFY_THREAD)AnchorAddr ) {
380  CurrentThreadInfo = CONTAINING_RECORD(
381  CurrentThreadInfo,
382  CM_ACTIVE_NOTIFY_THREAD,
383  ThreadList
384  );
385  if( CurrentThreadInfo->Thread == ActiveThreadInfo->Thread ) {
386  //
387  // already there!
388  //
389  CmpUnlockContext(CallbackBlock);
390  return FALSE;
391  }
392  //
393  // skip to the next element
394  //
395  CurrentThreadInfo = (PCM_ACTIVE_NOTIFY_THREAD)(CurrentThreadInfo->ThreadList.Flink);
396  }
397 
398  //
399  // add this thread
400  //
401  InsertTailList(&(CallbackBlock->ThreadListHead), &(ActiveThreadInfo->ThreadList));
402  CmpUnlockContext(CallbackBlock);
403  return TRUE;
