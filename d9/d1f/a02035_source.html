47 CcScheduleLazyWriteScan (
48  IN BOOLEAN FastScan
49  )
50 
51 /*++
52 
53 Routine Description:
54 
55  This routine may be called to schedule the next lazy writer scan,
56  during which lazy write and lazy close activity is posted to other
57  worker threads. Callers should acquire the lazy writer spin lock
58  to see if the scan is currently active, and then call this routine
59  still holding the spin lock if not. One special call is used at
60  the end of the lazy write scan to propagate lazy write active once
61  we go active. This call is "the" scan thread, and it can therefore
62  safely schedule the next scan without taking out the spin lock.
63 
64 Arguments:
65 
66  FastScan - if set, make the scan happen immediately
67 
68 Return Value:
69 
70  None.
71 
72 --*/
73 
74 {
75  //
76  // It is important to set the active flag TRUE first for the propagate
77  // case, because it is conceivable that once the timer is set, another
78  // thread could actually run and make the scan go idle before we then
79  // jam the flag TRUE.
80  //
81  // When going from idle to active, we delay a little longer to let the
82  // app finish saving its file.
83  //
84 
85  if (FastScan) {
86 
87  LazyWriter.ScanActive = TRUE;
88  KeSetTimer( &LazyWriter.ScanTimer, CcNoDelay, &LazyWriter.ScanDpc );
89 
90  } else if (LazyWriter.ScanActive) {
91 
92  KeSetTimer( &LazyWriter.ScanTimer, CcIdleDelay, &LazyWriter.ScanDpc );
93 
94  } else {
95 
96  LazyWriter.ScanActive = TRUE;
97  KeSetTimer( &LazyWriter.ScanTimer, CcFirstDelay, &LazyWriter.ScanDpc );
98  }
99 }

224 VOID
225 CcLazyWriteScan (
226  )
227 
228 /*++
229 
230 Routine Description:
231 
232  This routine implements the Lazy Writer scan for dirty data to flush
233  or any other work to do (lazy close). This routine is scheduled by
234  calling CcScheduleLazyWriteScan.
235 
236 Arguments:
237 
238  None.
239 
240 Return Value:
241 
242  None.
243 
244 --*/
245 
246 {
247  ULONG PagesToWrite, ForegroundRate, EstimatedDirtyNextInterval;
248  PSHARED_CACHE_MAP SharedCacheMap, FirstVisited, NextSharedCacheMap;
249  KIRQL OldIrql;
250  ULONG LoopsWithLockHeld = 0;
251  BOOLEAN AlreadyMoved = FALSE;
252  BOOLEAN MoveBehindCursor = FALSE;
253 
254  LIST_ENTRY PostTickWorkQueue;
255 
256  //
257  // Top of Lazy Writer scan.
258  //
259 
260  try {
261 
262  //
263  // If there is no work to do, then we will go inactive, and return.
264  //
265 
266  CcAcquireMasterLock( &OldIrql );
267 
268  if ((CcTotalDirtyPages == 0) && !LazyWriter.OtherWork) {
269 
270  //
271  // Sleep if there are no deferred writes. It is important to check
272  // proactively because writes may be blocked for reasons external
273  // to the cache manager. The lazy writer must keep poking since it
274  // may have no bytes to write itself.
275  //
276 
277 #if DBG
278  //
279  // In DBG builds, make sure that the CcDirtySharedCacheMapList
280  // is really empty (except for the cursor) if we are going to sleep
281  // because we think there is no more work to do.
282  //
283 
284  {
285  PLIST_ENTRY CurrentEntry = CcDirtySharedCacheMapList.SharedCacheMapLinks.Flink;
286  PSHARED_CACHE_MAP CurrentScm;
287  ULONG Count = 0;
288 
289  while( CurrentEntry != &CcDirtySharedCacheMapList.SharedCacheMapLinks ) {
290 
291  CurrentScm = CONTAINING_RECORD( CurrentEntry,
292  SHARED_CACHE_MAP,
293  SharedCacheMapLinks );
294 
295  if (FlagOn(CurrentScm->Flags, WAITING_FOR_TEARDOWN) &&
296  !FlagOn(CurrentScm->Flags, WRITE_QUEUED)) {
297  Count++;
298  }
299  CurrentEntry = CurrentEntry->Flink;
300  }
301 
302  if (!IsListEmpty( &CcDeferredWrites )) {
303 
304  Count = 0;
305  }
306 
307  ASSERTMSG( "CcLazyWriteScan stopped scan while SCM with the flag WAITING_FOR_TEARDOWN are still in the dirty list!\n",
308  Count == 0 );
309  }
310 #endif
311 
312  if (IsListEmpty(&CcDeferredWrites)) {
313 
314  LazyWriter.ScanActive = FALSE;
315  CcReleaseMasterLock( OldIrql );
316 
317  } else {
318 
319  CcReleaseMasterLock( OldIrql );
320 
321  //
322  // Check for writes and schedule the next scan.
323  //
324 
325  CcPostDeferredWrites();
326  CcScheduleLazyWriteScan( FALSE );
327  }
328 
329  return;
330  }
331 
332  //
333  // Pull out the post tick workitems for this pass. It is important that
334  // we are doing this at the top since more could be queued as we rummage
335  // for work to do. Post tick workitems are guaranteed to occur after all
336  // work generated in a complete scan.
337  //
338 
339  InitializeListHead( &PostTickWorkQueue );
340  while (!IsListEmpty( &CcPostTickWorkQueue )) {
341 
342  PLIST_ENTRY Entry = RemoveHeadList( &CcPostTickWorkQueue );
343  InsertTailList( &PostTickWorkQueue, Entry );
344  }
345 
346  //
347  // Calculate the next sweep time stamp, then update all relevant fields for
348  // the next time around. Also we can clear the OtherWork flag.
349  //
350 
351  LazyWriter.OtherWork = FALSE;
352 
353  //
354  // Assume we will write our usual fraction of dirty pages. Do not do the
355  // divide if there is not enough dirty pages, or else we will never write
356  // the last few pages.
357  //
358 
359  PagesToWrite = CcTotalDirtyPages;
360  if (PagesToWrite > LAZY_WRITER_MAX_AGE_TARGET) {
361  PagesToWrite /= LAZY_WRITER_MAX_AGE_TARGET;
362  }
363 
364  //
365  // Estimate the rate of dirty pages being produced in the foreground.
366  // This is the total number of dirty pages now plus the number of dirty
367  // pages we scheduled to write last time, minus the number of dirty
368  // pages we have now. Throw out any cases which would not produce a
369  // positive rate.
370  //
371 
372  ForegroundRate = 0;
373 
374  if ((CcTotalDirtyPages + CcPagesWrittenLastTime) > CcDirtyPagesLastScan) {
375  ForegroundRate = (CcTotalDirtyPages + CcPagesWrittenLastTime) -
376  CcDirtyPagesLastScan;
377  }
378 
379  //
380  // If we estimate that we will exceed our dirty page target by the end
381  // of this interval, then we must write more. Try to arrive on target.
382  //
383 
384  EstimatedDirtyNextInterval = CcTotalDirtyPages - PagesToWrite + ForegroundRate;
385 
386  if (EstimatedDirtyNextInterval > CcDirtyPageTarget) {
387 
388  PagesToWrite += EstimatedDirtyNextInterval - CcDirtyPageTarget;
389  }
390 
391  //
392  // Now save away the number of dirty pages and the number of pages we
393  // just calculated to write.
394  //
395 
396  CcDirtyPagesLastScan = CcTotalDirtyPages;
397  CcPagesYetToWrite = CcPagesWrittenLastTime = PagesToWrite;
398 
399  //
400  // Loop to flush enough Shared Cache Maps to write the number of pages
401  // we just calculated.
402  //
403 
404  SharedCacheMap = CONTAINING_RECORD( CcLazyWriterCursor.SharedCacheMapLinks.Flink,
405  SHARED_CACHE_MAP,
406  SharedCacheMapLinks );
407 
408  DebugTrace( 0, me, "Start of Lazy Writer Scan\n", 0 );
409 
410  //
411  // Normally we would just like to visit every Cache Map once on each scan,
412  // so the scan will terminate normally when we return to FirstVisited. But
413  // in the off chance that FirstVisited gets deleted, we are guaranteed to stop
414  // when we get back to our own listhead.
415  //
416 
417  FirstVisited = NULL;
418  while ((SharedCacheMap != FirstVisited) &&
419  (&SharedCacheMap->SharedCacheMapLinks != &CcLazyWriterCursor.SharedCacheMapLinks)) {
420 
421  if (FirstVisited == NULL) {
422  FirstVisited = SharedCacheMap;
423  }
424 
425  //
426  // Skip the SharedCacheMap if a write behind request is
427  // already queued, write behind has been disabled, or
428  // if there is no work to do (either dirty data to be written
429  // or a delete is required).
430  //
431  // Note that for streams where modified writing is disabled, we
432  // need to take out Bcbs exclusive, which serializes with foreground
433  // activity. Therefore we use a special counter in the SharedCacheMap
434  // to only service these once every n intervals.
435  //
436  // Skip temporary files unless we currently could not write as many
437  // bytes as we might charge some hapless thread for throttling, unless
438  // it has been closed. We assume that the "tick" of the lazy writer,
439  // delayed temporarily by the passcount check, will permit the common
440  // open/write/close/delete action on temporary files to sneak in and
441  // truncate the file before we really write the data, if the file was
442  // not opened delete-on-close to begin with.
443  //
444  // Since we will write closed files with dirty pages as part of the
445  // regular pass (even temporary ones), only do lazy close on files
446  // with no dirty pages.
447  //
448 
449  if (!FlagOn(SharedCacheMap->Flags, WRITE_QUEUED | IS_CURSOR)
450 
451  &&
452 
453  (((SharedCacheMap->DirtyPages != 0)
454  &&
455  (FlagOn(SharedCacheMap->Flags, WAITING_FOR_TEARDOWN)
456  ||
457  ((PagesToWrite != 0)
458  &&
459  (((++SharedCacheMap->LazyWritePassCount & 0xF) == 0) ||
460  !FlagOn(SharedCacheMap->Flags, MODIFIED_WRITE_DISABLED) ||
461  (CcCapturedSystemSize == MmSmallSystem) ||
462  (SharedCacheMap->DirtyPages >= (4 * (MAX_WRITE_BEHIND / PAGE_SIZE))))
463  &&
464  (!FlagOn(SharedCacheMap->FileObject->Flags, FO_TEMPORARY_FILE) ||
465  (SharedCacheMap->OpenCount == 0) ||
466  !CcCanIWrite(SharedCacheMap->FileObject, WRITE_CHARGE_THRESHOLD, FALSE, MAXUCHAR)))))
467 
468  ||
469 
470  ((SharedCacheMap->OpenCount == 0) &&
471  (SharedCacheMap->DirtyPages == 0) ||
472  (SharedCacheMap->FileSize.QuadPart == 0)))) {
473 
474  PWORK_QUEUE_ENTRY WorkQueueEntry;
475 
476  //
477  // If this is a metadata stream with at least 4 times
478  // the maximum write behind I/O size, then let's tell
479  // this guy to write 1/8 of his dirty data on this pass
480  // so it doesn't build up.
481  //
482  // Else assume we can write everything (PagesToWrite only affects
483  // metadata streams - otherwise writing is controlled by the Mbcb -
484  // this throttle is engaged in CcWriteBehind).
485  //
486 
487  SharedCacheMap->PagesToWrite = SharedCacheMap->DirtyPages;
488 
489  if (FlagOn(SharedCacheMap->Flags, MODIFIED_WRITE_DISABLED) &&
490  (SharedCacheMap->PagesToWrite >= (4 * (MAX_WRITE_BEHIND / PAGE_SIZE))) &&
491  (CcCapturedSystemSize != MmSmallSystem)) {
492 
493  SharedCacheMap->PagesToWrite /= 8;
494  }
495 
496  //
497  // If still searching for pages to write, adjust our targets.
498  //
499 
500  if (!AlreadyMoved) {
501 
502  //
503  // See if he exhausts the number of pages to write. (We
504  // keep going in case there are any closes to do.)
505  //
506 
507  if (SharedCacheMap->PagesToWrite >= PagesToWrite) {
508 
509  //
510  // Here is where we should move the cursor to. Figure
511  // out if we should resume on this stream or the next one.
512  //
513 
514  //
515  // For Metadata streams, set up to resume on the next stream on the
516  // next scan. Also force a push forward every n intervals if all of
517  // the pages came from this stream, so we don't get preoccupied with
518  // one stream at the expense of others (which may be waiting for a
519  // lazy close). Normally we would like to avoid seek overhead and
520  // take the common case of a large sequential series of writes.
521  //
522  // This is similar to hotspot detection.
523  //
524  // Note, to ensure that we iterate through the entire
525  // CcDirtySharedCacheMap list, we cannot move this shared
526  // cache map behind the cursor now. We will just
527  // remember that we want to move this to the end of the
528  // list and do the actual move when we are ready to read
529  // the next entry.
530  //
531 
532  if (FlagOn(SharedCacheMap->Flags, MODIFIED_WRITE_DISABLED) ||
533  ((FirstVisited == SharedCacheMap) &&
534  ((SharedCacheMap->LazyWritePassCount & 0xF) == 0))) {
535 
536  MoveBehindCursor = TRUE;
537 
538  //
539  // For other streams, set up to resume on the same stream on the
540  // next scan.
541  //
542 
543  } else {
544 
545  RemoveEntryList( &CcLazyWriterCursor.SharedCacheMapLinks );
546  InsertTailList( &SharedCacheMap->SharedCacheMapLinks, &CcLazyWriterCursor.SharedCacheMapLinks );
547  }
548 
549  PagesToWrite = 0;
550  AlreadyMoved = TRUE;
551 
552  } else {
553 
554  PagesToWrite -= SharedCacheMap->PagesToWrite;
555  }
556  }
557 
558  //
559  // Otherwise show we are actively writing, and keep it in the dirty
560  // list.
561  //
562 
563  SetFlag(SharedCacheMap->Flags, WRITE_QUEUED);
564  SharedCacheMap->DirtyPages += 1;
565 
566  CcReleaseMasterLock( OldIrql );
567 
568  //
569  // Queue the request to do the work to a worker thread.
570  //
571 
572  WorkQueueEntry = CcAllocateWorkQueueEntry();
573 
574  //
575  // If we failed to allocate a WorkQueueEntry, things must
576  // be in pretty bad shape. However, all we have to do is
577  // break out of our current loop, and try to go back and
578  // delay a while. Even if the current guy should have gone
579  // away when we clear WRITE_QUEUED, we will find him again
580  // in the LW scan.
581  //
582 
583  if (WorkQueueEntry == NULL) {
584 
585  CcAcquireMasterLock( &OldIrql );
586  ClearFlag(SharedCacheMap->Flags, WRITE_QUEUED);
587  SharedCacheMap->DirtyPages -= 1;
588  break;
589  }
590 
591  WorkQueueEntry->Function = (UCHAR)WriteBehind;
592  WorkQueueEntry->Parameters.Write.SharedCacheMap = SharedCacheMap;
593 
594  //
595  // Post it to the regular work queue.
596  //
597 
598  CcAcquireMasterLock( &OldIrql );
599  SharedCacheMap->DirtyPages -= 1;
600 
601  if (FlagOn( SharedCacheMap->Flags, WAITING_FOR_TEARDOWN )) {
602 
603  //
604  // If we are waiting for this shared cache map to be torn
605  // down, put it at the head of the express work queue so
606  // that it gets processed right away.
607  //
608 
609  CcPostWorkQueue( WorkQueueEntry, &CcExpressWorkQueue );
610 
611  } else {
612 
613  //
614  // We aren't anxiously awaiting for this shared cached map
615  // to go away, so just process this work item via the
616  // regular work queue.
617  //
618 
619  CcPostWorkQueue( WorkQueueEntry, &CcRegularWorkQueue );
620  }
621 
622  LoopsWithLockHeld = 0;
623 
624  //
625  // Make sure we occasionally drop the lock. Set WRITE_QUEUED
626  // to keep the guy from going away.
627  //
628 
629  } else if ((++LoopsWithLockHeld >= 20) &&
630  !FlagOn(SharedCacheMap->Flags, WRITE_QUEUED | IS_CURSOR)) {
631 
632  SetFlag(SharedCacheMap->Flags, WRITE_QUEUED);
633  SharedCacheMap->DirtyPages += 1;
634  CcReleaseMasterLock( OldIrql );
635  LoopsWithLockHeld = 0;
636  CcAcquireMasterLock( &OldIrql );
637  ClearFlag(SharedCacheMap->Flags, WRITE_QUEUED);
638  SharedCacheMap->DirtyPages -= 1;
639  }
640 
641  //
642  // Now loop back.
643  //
644  // If we want to put this shared cache map at the end of the
645  // dirty list, we will do it AFTER we determine the next shared
646  // cache map to go to. This ensures that we loop through the entire
647  // list during this scan tick.
648  //
649 
650  NextSharedCacheMap =
651  CONTAINING_RECORD( SharedCacheMap->SharedCacheMapLinks.Flink,
652  SHARED_CACHE_MAP,
653  SharedCacheMapLinks );
654 
655  if (MoveBehindCursor) {
656 
657  RemoveEntryList( &CcLazyWriterCursor.SharedCacheMapLinks );
658  InsertHeadList( &SharedCacheMap->SharedCacheMapLinks, &CcLazyWriterCursor.SharedCacheMapLinks );
659  MoveBehindCursor = FALSE;
660  }
661 
662  SharedCacheMap = NextSharedCacheMap;
663  }
664 
665  DebugTrace( 0, me, "End of Lazy Writer Scan\n", 0 );
666 
667  //
668  // Queue up our post tick workitems for this pass.
669  //
670 
671  while (!IsListEmpty( &PostTickWorkQueue )) {
672 
673  PLIST_ENTRY Entry = RemoveHeadList( &PostTickWorkQueue );
674  CcPostWorkQueue( CONTAINING_RECORD( Entry, WORK_QUEUE_ENTRY, WorkQueueLinks ),
675  &CcRegularWorkQueue );
676  }
677 
678  //
679  // Now we can release the global list and loop back, per chance to sleep.
680  //
681 
682  CcReleaseMasterLock( OldIrql );
683 
684  //
685  // Once again we need to give the deferred writes a poke. We can have all dirty
686  // pages on disable_write_behind files but also have an external condition that
687  // caused the cached IO to be deferred. If so, this serves as our only chance to
688  // issue it when the condition clears.
689  //
690 
691  if (!IsListEmpty(&CcDeferredWrites)) {
692 
693  CcPostDeferredWrites();
694  }
695 
696  //
697  // Now go ahead and schedule the next scan.
698  //
699 
700  CcScheduleLazyWriteScan( FALSE );
701 
702  //
703  // Basically, the Lazy Writer thread should never get an exception,
704  // so we put a try-except around it that bugchecks one way or the other.
705  // Better we bugcheck here than worry about what happens if we let one
706  // get by.
707  //
708 
709  } except( CcExceptionFilter( GetExceptionCode() )) {
710 
711  CcBugCheck( GetExceptionCode(), 0, 0 );
712  }
713 }

764 CcPostWorkQueue (
765  IN PWORK_QUEUE_ENTRY WorkQueueEntry,
766  IN PLIST_ENTRY WorkQueue
767  )
768 
769 /*++
770 
771 Routine Description:
772 
773  This routine queues a WorkQueueEntry, which has been allocated and
774  initialized by the caller, to the WorkQueue for FIFO processing by
775  the work threads.
776 
777 Arguments:
778 
779  WorkQueueEntry - supplies a pointer to the entry to queue
780 
781 Return Value:
782 
783  None
784 
785 --*/
786 
787 {
788  KIRQL OldIrql;
789  PLIST_ENTRY WorkerThreadEntry = NULL;
790 
791  ASSERT(FIELD_OFFSET(WORK_QUEUE_ITEM, List) == 0);
792 
793  DebugTrace(+1, me, "CcPostWorkQueue:\n", 0 );
794  DebugTrace( 0, me, " WorkQueueEntry = %08lx\n", WorkQueueEntry );
795 
796  //
797  // Queue the entry to the respective work queue.
798  //
799 
800  CcAcquireWorkQueueLock( &OldIrql );
801  InsertTailList( WorkQueue, &WorkQueueEntry->WorkQueueLinks );
802 
803  //
804  // Now, if we aren't throttled and have any more idle threads we can
805  // use, activate one.
806  //
807 
808  if (!CcQueueThrottle && !IsListEmpty(&CcIdleWorkerThreadList)) {
809  WorkerThreadEntry = RemoveHeadList( &CcIdleWorkerThreadList );

879  //

888  //

898  //

905  //
906

1003  //

1012

1022  //

1027  }
1028  }

1032  //
1033 
1034  if (IoStatus.Information != CC_REQUEUE) {
1035 
1036  CcFreeWorkQueueEntry( WorkQueueEntry );
1037  }
1038  }
1039

1042  //
1043 
1044  InsertTailList( &CcIdleWorkerThreadList,
1045  &((PWORK_QUEUE_ITEM)ExWorkQueueItem)->List );
