57 /*++
58 
59 Routine Description:
60 
61  This routine implements the FileSystem control operations
62 
63 Arguments:
64 
65  Vcb - Supplies the volume being queried.
66 
67  Irp - Supplies the Irp being processed.
68 
69  IrpSp - Supplies parameters describing the FileSystem control operation.
70 
71 Return Value:
72 
73  NTSTATUS - The status for the IRP
74 
75 --*/
76 
77 {
78  NTSTATUS Status;
79 
80  PAGED_CODE();
81 
82  //
83  // We know this is a file system control so we'll case on the
84  // minor function, and call an internal worker routine.
85  //
86 
87  switch (IrpSp->MinorFunction) {
88 
89  case IRP_MN_USER_FS_REQUEST:
90 
91  Status = RawUserFsCtrl( IrpSp, Vcb );
92  break;
93 
94  case IRP_MN_MOUNT_VOLUME:
95 
96  Status = RawMountVolume( IrpSp );
97  break;
98 
99  case IRP_MN_VERIFY_VOLUME:
100 
101  Status = RawVerifyVolume( IrpSp, Vcb );
102  break;
103 
104  default:
105 
106  Status = STATUS_INVALID_DEVICE_REQUEST;
107  break;
108  }
109 
110  RawCompleteRequest( Irp, Status );
111 
112  return Status;
113 }

121 RawMountVolume (
122  IN PIO_STACK_LOCATION IrpSp
123  )
124 
125 /*++
126 
127 Routine Description:
128 
129  This routine performs the mount volume operation.
130 
131 Arguments:
132 
133  IrpSp - Supplies the IrpSp parameters to process
134 
135 Return Value:
136 
137  NTSTATUS - The return status for the operation
138 
139 --*/
140 
141 {
142  NTSTATUS Status;
143  PDEVICE_OBJECT DeviceObjectWeTalkTo;
144 
145  PVOLUME_DEVICE_OBJECT VolumeDeviceObject;
146 
147  PAGED_CODE();
148 
149  //
150  // Save some references to make our life a little easier
151  //
152 
153  DeviceObjectWeTalkTo = IrpSp->Parameters.MountVolume.DeviceObject;
154 
155  //
156  // A mount operation has been requested. Create a
157  // new device object to represent this volume.
158  //
159 
160  Status = IoCreateDevice( IrpSp->DeviceObject->DriverObject,
161  sizeof(VOLUME_DEVICE_OBJECT) - sizeof(DEVICE_OBJECT),
162  NULL,
163  FILE_DEVICE_DISK_FILE_SYSTEM,
164  0,
165  FALSE,
166  (PDEVICE_OBJECT *)&VolumeDeviceObject );
167 
168  if ( !NT_SUCCESS( Status ) ) {
169 
170  return Status;
171  }
172 
173  //
174  // Our alignment requirement is the larger of the processor alignment requirement
175  // already in the volume device object and that in the DeviceObjectWeTalkTo
176  //
177 
178  if (DeviceObjectWeTalkTo->AlignmentRequirement > VolumeDeviceObject->DeviceObject.AlignmentRequirement) {
179 
180  VolumeDeviceObject->DeviceObject.AlignmentRequirement = DeviceObjectWeTalkTo->AlignmentRequirement;
181  }
182 
183  //
184  // Set sector size to the same value as the DeviceObjectWeTalkTo.
185  //
186 
187  VolumeDeviceObject->DeviceObject.SectorSize = DeviceObjectWeTalkTo->SectorSize;
188 
189  VolumeDeviceObject->DeviceObject.Flags |= DO_DIRECT_IO;
190 
191  //
192  // Initialize the Vcb for this volume
193  //
194 
195  Status = RawInitializeVcb( &VolumeDeviceObject->Vcb,
196  IrpSp->Parameters.MountVolume.DeviceObject,
197  IrpSp->Parameters.MountVolume.Vpb );
198 
199 
200  if ( !NT_SUCCESS( Status ) ) {
201 
202  //
203  // Unlike the other points of teardown we do not need to deref the target device
204  // a iosubsys will automatically do that for a failed mount
205  //
206 
207  IoDeleteDevice( (PDEVICE_OBJECT)VolumeDeviceObject );
208  return Status;
209  }
210 
211  //
212  // Finally, make it look as if the volume has been
213  // mounted. This includes storing the
214  // address of this file system's device object (the one
215  // that was created to handle this volume) in the VPB so
216  // all requests are directed to this file system from
217  // now until the volume is initialized with a real file
218  // structure.
219  //
220 
221  VolumeDeviceObject->Vcb.Vpb->DeviceObject = (PDEVICE_OBJECT)VolumeDeviceObject;
222  VolumeDeviceObject->Vcb.Vpb->SerialNumber = 0xFFFFFFFF;
223  VolumeDeviceObject->Vcb.Vpb->VolumeLabelLength = 0;
224 
225  VolumeDeviceObject->DeviceObject.Flags &= ~DO_DEVICE_INITIALIZING;
226  VolumeDeviceObject->DeviceObject.StackSize = (UCHAR) (DeviceObjectWeTalkTo->StackSize + 1);
227 
228  {
229  PFILE_OBJECT VolumeFileObject = NULL;
230 
231  //
232  // We need a file object to do the notification.
233  //
234 
235  try {
236  VolumeFileObject = IoCreateStreamFileObjectLite( NULL, &VolumeDeviceObject->DeviceObject );
237  } except (EXCEPTION_EXECUTE_HANDLER) {
238  Status = GetExceptionCode();
239  }
240 
241  if (!NT_SUCCESS(Status)) {
242  IoDeleteDevice( (PDEVICE_OBJECT)VolumeDeviceObject );
243  return Status;
244  }
245 
246  //
247  // We need to bump the count up 2 now so that the close we do in a few lines
248  // doesn't make the Vcb go away now.
249  //
250 
251  VolumeDeviceObject->Vcb.OpenCount += 2;
252  FsRtlNotifyVolumeEvent( VolumeFileObject, FSRTL_VOLUME_MOUNT );
253  ObDereferenceObject( VolumeFileObject );
254 
255  //
256  // Okay, the close is over, now we can safely decrement the open count again
257  // (back to 0) so the Vcb can go away when we're really done with it.
258  //
259 
260  VolumeDeviceObject->Vcb.OpenCount -= 2;
261  }
262 
263  return Status;
264 }

271 NTSTATUS
272 RawVerifyVolume (
273  IN PIO_STACK_LOCATION IrpSp,
274  IN PVCB Vcb
275  )
276 
277 /*++
278 
279 Routine Description:
280 
281  This routine verifies a volume.
282 
283 Arguments:
284 
285  IrpSp - Supplies the IrpSp parameters to process
286 
287 Return Value:
288 
289  NTSTATUS - The return status for the operation
290 
291 --*/
292 
293 {
294  NTSTATUS Status;
295  BOOLEAN DeleteVolume = FALSE;
296  KIRQL Irql;
297  PVPB vpb;
298  BOOLEAN Mounted;
299 
300  //
301  // If the volume is somehow stale, dismount. We must synchronize
302  // our inspection of the close count so we don't rip the volume up
303  // while racing with a close, for instance. The VPB refcount drops
304  // *before* the close comes into the filesystem.
305  //
306 
307  //
308  // By this time its possible that the volume has been dismounted by
309  // RawClose. So check if its mounted. If so, take a reference on the VPB
310  // The reference on the VPB will prevent close from deleting the device.
311  //
312 
313  IoAcquireVpbSpinLock(&Irql);
314 
315  Mounted = FALSE;
316  vpb = IrpSp->Parameters.VerifyVolume.Vpb;
317  if (vpb->Flags & VPB_MOUNTED) {
318  vpb->ReferenceCount++;
319  Mounted = TRUE;
320  }
321 
322  IoReleaseVpbSpinLock(Irql);
323 
324  if (!Mounted) {
325  return STATUS_WRONG_VOLUME;
326  }
327 
328  Status = KeWaitForSingleObject( &Vcb->Mutex,
329  Executive,
330  KernelMode,
331  FALSE,
332  (PLARGE_INTEGER) NULL );
333  ASSERT( NT_SUCCESS( Status ) );
334 
335  //
336  // Since we ignore all verify errors from the disk driver itself,
337  // this request must have originated from a file system, thus
338  // since we weren't the originators, we're going to say this isn't
339  // our volume, and if the open count is zero, dismount the volume.
340  //
341 
342  IoAcquireVpbSpinLock(&Irql);
343  vpb->ReferenceCount--;
344  IoReleaseVpbSpinLock(Irql);
345 
346  Vcb->Vpb->RealDevice->Flags &= ~DO_VERIFY_VOLUME;
347 
348  if (Vcb->OpenCount == 0) {
349 
350  DeleteVolume = RawCheckForDismount( Vcb, FALSE );
351  }
352 
353  if (!DeleteVolume) {
354  (VOID)KeReleaseMutex( &Vcb->Mutex, FALSE );
355  }
356 
357  return STATUS_WRONG_VOLUME;
358 }

366 NTSTATUS
367 RawUserFsCtrl (
368  IN PIO_STACK_LOCATION IrpSp,
369  IN PVCB Vcb
370  )
371 
372 /*++
373 
374 Routine Description:
375 
376  This is the common routine for implementing the user's requests made
377  through NtFsControlFile.
378 
379 Arguments:
380 
381  IrpSp - Supplies the IrpSp parameters to process
382 
383  Vcb - Supplies the volume we are working on.
384 
385 Return Value:
386 
387  NTSTATUS - The return status for the operation
388 
389 --*/
390 
391 {
392  NTSTATUS Status;
393  ULONG FsControlCode;
394  PFILE_OBJECT FileObject;
395 
396  PAGED_CODE();
397 
398  FsControlCode = IrpSp->Parameters.FileSystemControl.FsControlCode;
399  FileObject = IrpSp->FileObject;
400 
401  //
402  // Do pre-notification before entering the volume mutex so that we
403  // can be reentered by good threads cleaning up their resources.
404  //
405 
406  switch (FsControlCode) {
407  case FSCTL_LOCK_VOLUME:
408 
409  FsRtlNotifyVolumeEvent( FileObject, FSRTL_VOLUME_LOCK );
410  break;
411 
412  case FSCTL_DISMOUNT_VOLUME:
413 
414  FsRtlNotifyVolumeEvent( FileObject, FSRTL_VOLUME_DISMOUNT );
415  break;
416 
417  default:
418  break;
419  }
420 
421  Status = KeWaitForSingleObject( &Vcb->Mutex,
422  Executive,
423  KernelMode,
424  FALSE,
425  (PLARGE_INTEGER) NULL );
426  ASSERT( NT_SUCCESS( Status ) );
427 
428  switch ( FsControlCode ) {
429 
430  case FSCTL_REQUEST_OPLOCK_LEVEL_1:
431  case FSCTL_REQUEST_OPLOCK_LEVEL_2:
432  case FSCTL_OPLOCK_BREAK_ACKNOWLEDGE:
433  case FSCTL_OPLOCK_BREAK_NOTIFY:
434 
435  Status = STATUS_NOT_IMPLEMENTED;
436  break;
437 
438  case FSCTL_LOCK_VOLUME:
439 
440  if ( !FlagOn(Vcb->VcbState, VCB_STATE_FLAG_LOCKED) &&
441  (Vcb->OpenCount == 1) ) {
442 
443  Vcb->VcbState |= VCB_STATE_FLAG_LOCKED;
444 
445  Status = STATUS_SUCCESS;
446 
447  } else {
448 
449  Status = STATUS_ACCESS_DENIED;
450  }
451 
452  break;
453 
454  case FSCTL_UNLOCK_VOLUME:
455 
456  if ( !FlagOn(Vcb->VcbState, VCB_STATE_FLAG_LOCKED) ) {
457 
458  Status = STATUS_NOT_LOCKED;
459 
460  } else {
461 
462  Vcb->VcbState &= ~VCB_STATE_FLAG_LOCKED;
463 
464  Status = STATUS_SUCCESS;
465  }
466 
467  break;
468 
469  case FSCTL_DISMOUNT_VOLUME:
470 
471  //
472  // Right now the logic in cleanup.c assumes that there can
473  // only be one handle on the volume if locked. The code
474  // there needs to be fixed if forced dismounts are allowed.
475  //
476 
477  if (FlagOn(Vcb->VcbState, VCB_STATE_FLAG_LOCKED)) {
478 
479  Vcb->VcbState |= VCB_STATE_FLAG_DISMOUNTED;
480  Status = STATUS_SUCCESS;
481 
482  } else {
483 
484  Status = STATUS_ACCESS_DENIED;
485  }
486 
487  break;
488 
489  default:
490 
491  Status = STATUS_INVALID_PARAMETER;
492  break;
493  }
494 
495  (VOID)KeReleaseMutex( &Vcb->Mutex, FALSE );
496 
497  //
498  // Now perform post-notification as required.
499  //
500 
501  if (NT_SUCCESS( Status )) {
502 
503  switch ( FsControlCode ) {
504  case FSCTL_UNLOCK_VOLUME:
505 
506  FsRtlNotifyVolumeEvent( FileObject, FSRTL_VOLUME_UNLOCK );
507  break;
508 
509  default:
510  break;
511  }
512 
513  } else {
514 
515  switch ( FsControlCode ) {
516  case FSCTL_LOCK_VOLUME:
517 
518  FsRtlNotifyVolumeEvent( FileObject, FSRTL_VOLUME_LOCK_FAILED );
519  break;
520 
521  case FSCTL_DISMOUNT_VOLUME:
522 
523  FsRtlNotifyVolumeEvent( FileObject, FSRTL_VOLUME_DISMOUNT_FAILED );
524  break;
525 
526  default:
527  break;
528  }
529  }
530 
531  return Status;
532 }
