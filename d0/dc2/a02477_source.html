30 LpcpClosePort (
31  IN PEPROCESS Process OPTIONAL,
32  IN PVOID Object,

38 /*++
39 
40 Routine Description:
41 
42  This routine is the callback used for closing a port object.
43 
44 Arguments:
45 
46  Process - Supplies an optional pointer to the process whose port is being
47  closed
48 
49  Object - Supplies a pointer to the port object being closed
50 
51  GrantedAccess - Supplies the access granted to the handle closing port
52  object
53 
54  ProcessHandleCount - Supplies the number of process handles remaining to
55  the object
56 
57  SystemHandleCount - Supplies the number of system handles remaining to
58  the object
59 
60 Return Value:
61 
62  None.
63 
64 --*/
65 
66 {
67  //
68  // Translate the object to what it really is, an LPCP port object
69  //
70 
71  PLPCP_PORT_OBJECT Port = Object;
72 
73  UNREFERENCED_PARAMETER (Process);
74  UNREFERENCED_PARAMETER (GrantedAccess);
75  UNREFERENCED_PARAMETER (ProcessHandleCount);
76 
77  //
78  // We only have work to do if the object is a server communication port
79  //
80 
81  if ( (Port->Flags & PORT_TYPE) == SERVER_CONNECTION_PORT ) {
82 
83  //
84  // If this is a server communication port without any system handles
85  // then we can completely destroy the communication queue for the
86  // port
87  //
88 
89  if ( SystemHandleCount == 0 ) {
90 
91  LpcpDestroyPortQueue( Port, TRUE );
92 
93  //
94  // If there is only one system handle left then we'll reset the
95  // communication queue for the port
96  //
97 
98  } else if ( SystemHandleCount == 1 ) {
99 
100  LpcpDestroyPortQueue( Port, FALSE );
101  }
102 
103  //
104  // Otherwise we do nothing
105  //
106  }
107 
108  return;
109 }

113 LpcpDeletePort (
114  IN PVOID Object
115  )
116 
117 /*++
118 
119 Routine Description:
120 
121  This routine is the callback used for deleting a port object.
122 
123 Arguments:
124 
125  Object - Supplies a pointer to the port object being deleted
126 
127 Return Value:
128 
129  None.
130 
131 --*/
132 
133 {
134  PETHREAD CurrentThread;
135  PLPCP_PORT_OBJECT Port = Object;
136  PLPCP_PORT_OBJECT ConnectionPort;
137  LPC_CLIENT_DIED_MSG ClientPortClosedDatagram;
138  PLPCP_MESSAGE Msg;
139  PLIST_ENTRY Head, Next;
140  HANDLE CurrentProcessId;
141  NTSTATUS Status;
142  LARGE_INTEGER RetryInterval = {(ULONG)(-10 * 1000 * 100), -1}; // 100 milliseconds
143 
144  PAGED_CODE();
145 
146  CurrentThread = PsGetCurrentThread ();
147 
148  //
149  // If the port is a server communication port then make sure that if
150  // there is a dangling client thread that we get rid of it. This
151  // handles the case of someone calling NtAcceptConnectPort and not
152  // calling NtCompleteConnectPort
153  //
154 
155  if ((Port->Flags & PORT_TYPE) == SERVER_COMMUNICATION_PORT) {
156 
157  PETHREAD ClientThread;
158 
159  LpcpAcquireLpcpLockByThread(CurrentThread);
160 
161  if ((ClientThread = Port->ClientThread) != NULL) {
162 
163  Port->ClientThread = NULL;
164 
165  LpcpReleaseLpcpLock();
166 
167  ObDereferenceObject( ClientThread );
168 
169  } else {
170 
171  LpcpReleaseLpcpLock();
172  }
173  }
174 
175  //
176  // Send an LPC_PORT_CLOSED datagram to whoever is connected
177  // to this port so they know they are no longer connected.
178  //
179 
180  if ((Port->Flags & PORT_TYPE) == CLIENT_COMMUNICATION_PORT) {
181 
182  ClientPortClosedDatagram.PortMsg.u1.s1.TotalLength = sizeof( ClientPortClosedDatagram );
183  ClientPortClosedDatagram.PortMsg.u1.s1.DataLength = sizeof( ClientPortClosedDatagram.CreateTime );
184 
185  ClientPortClosedDatagram.PortMsg.u2.s2.Type = LPC_PORT_CLOSED;
186  ClientPortClosedDatagram.PortMsg.u2.s2.DataInfoOffset = 0;
187 
188  ClientPortClosedDatagram.CreateTime = PsGetCurrentProcess()->CreateTime;
189 
190  Status = LpcRequestPort( Port, (PPORT_MESSAGE)&ClientPortClosedDatagram );
191 
192  while (Status == STATUS_NO_MEMORY) {
193 
194  KeDelayExecutionThread(KernelMode, FALSE, &RetryInterval);
195 
196  Status = LpcRequestPort( Port, (PPORT_MESSAGE)&ClientPortClosedDatagram );
197  }
198  }
199 
200  //
201  // If connected, disconnect the port, and then scan the message queue
202  // for this port and dereference any messages in the queue.
203  //
204 
205  LpcpDestroyPortQueue( Port, TRUE );
206 
207  //
208  // If we had mapped sections into the server or client communication ports,
209  // we need to unmap them in the context of that process.
210  //
211 
212  if ( (Port->ClientSectionBase != NULL) ||
213  (Port->ServerSectionBase != NULL) ) {
214 
215  //
216  // If the client has a port memory view, then unmap it
217  //
218 
219  if (Port->ClientSectionBase != NULL) {
220 
221  MmUnmapViewOfSection( Port->MappingProcess,
222  Port->ClientSectionBase );
223 
224  }
225 
226  //
227  // If the server has a port memory view, then unmap it
228  //
229 
230  if (Port->ServerSectionBase != NULL) {
231 
232  MmUnmapViewOfSection( Port->MappingProcess,
233  Port->ServerSectionBase );
234 
235  }
236 
237  //
238  // Removing the reference added while mapping the section
239  //
240 
241  ObDereferenceObject( Port->MappingProcess );
242 
243  Port->MappingProcess = NULL;
244  }
245 
246  //
247  // Dereference the pointer to the connection port if it is not
248  // this port.
249  //
250 
251  LpcpAcquireLpcpLockByThread(CurrentThread);
252 
253  ConnectionPort = Port->ConnectionPort;
254 
255  if (ConnectionPort) {
256 
257  CurrentProcessId = CurrentThread->Cid.UniqueProcess;
258 
259  Head = &ConnectionPort->LpcDataInfoChainHead;
260  Next = Head->Flink;
261 
262  while (Next != Head) {
263 
264  Msg = CONTAINING_RECORD( Next, LPCP_MESSAGE, Entry );
265  Next = Next->Flink;
266 
267  if (Port == ConnectionPort) {
268 
269  //
270  // If the Connection port is going away free all queued messages
271  //
272 
273  RemoveEntryList( &Msg->Entry );
274  InitializeListHead( &Msg->Entry );
275 
276  LpcpFreeToPortZone( Msg, LPCP_MUTEX_OWNED );
277 
278  //
279  // In LpcpFreeToPortZone the LPC lock is released and reacquired.
280  // Another thread might free the LPC message captured above
281  // in Next. We need to restart the search at the list head.
282  //
283 
284  Next = Head->Flink;
285 
286  } else if ((Msg->Request.ClientId.UniqueProcess == CurrentProcessId)
287  &&
288  ((Msg->SenderPort == Port)
289  ||
290  (Msg->SenderPort == Port->ConnectedPort)
291  ||
292  (Msg->SenderPort == ConnectionPort))) {
293 
294  //
295  // Test whether the message come from the same port and process
296  //
297 
298  LpcpTrace(( "%s Freeing DataInfo Message %lx (%u.%u) Port: %lx\n",
299  PsGetCurrentProcess()->ImageFileName,
300  Msg,
301  Msg->Request.MessageId,
302  Msg->Request.CallbackId,
303  ConnectionPort ));
304 
305  RemoveEntryList( &Msg->Entry );
306  InitializeListHead( &Msg->Entry );
307 
308  LpcpFreeToPortZone( Msg, LPCP_MUTEX_OWNED );
309 
310  //
311  // In LpcpFreeToPortZone the LPC lock is released and reacquired.
312  // Another thread might free the LPC message captured above
313  // in Next. We need to restart the search at the list head.
314  //
315 
316  Next = Head->Flink;
317  }
318  }
319 
320  LpcpReleaseLpcpLock();
321 
322  if (ConnectionPort != Port) {
323 
324  ObDereferenceObject( ConnectionPort );
325  }
326 
327  } else {
328 
329  LpcpReleaseLpcpLock();
330  }
331 
332  if (((Port->Flags & PORT_TYPE) == SERVER_CONNECTION_PORT) &&
333  (ConnectionPort->ServerProcess != NULL)) {
334 
335  ObDereferenceObject( ConnectionPort->ServerProcess );
336 
337  ConnectionPort->ServerProcess = NULL;
338  }
339 
340  //
341  // Free any static client security context
342  //
343 
344  LpcpFreePortClientSecurity( Port );
345 
346  //
347  // And return to our caller
348  //
349 
350  return;
351 }

354 VOID
355 LpcExitThread (
356  PETHREAD Thread
357  )
358 
359 /*++
360 
361 Routine Description:
362 
363  This routine is called whenever a thread is exiting and need to cleanup the
364  lpc port for the thread.
365 
366 Arguments:
367 
368  Thread - Supplies the thread being terminated
369 
370 Return Value:
371 
372  None.
373 
374 --*/
375 
376 {
377  PLPCP_MESSAGE Msg;
378 
379  //
380  // Acquire the mutex that protects the LpcReplyMessage field of
381  // the thread. Zero the field so nobody else tries to process it
382  // when we release the lock.
383  //
384 
385  ASSERT (Thread == PsGetCurrentThread());
386 
387  LpcpAcquireLpcpLockByThread(Thread);
388 
389  if (!IsListEmpty( &Thread->LpcReplyChain )) {
390 
391  RemoveEntryList( &Thread->LpcReplyChain );
392  }
393 
394  //
395  // Indicate that this thread is exiting
396  //
397 
398  Thread->LpcExitThreadCalled = TRUE;
399  Thread->LpcReplyMessageId = 0;
400 
401  //
402  // If we need to reply to a message then if the thread that we need to reply
403  // to is still around we want to dereference the thread and free the message
404  //
405 
406  Msg = LpcpGetThreadMessage(Thread);
407 
408  if (Msg != NULL) {
409 
410  Thread->LpcReplyMessage = NULL;
411 
412  if (Msg->RepliedToThread != NULL) {
413 
414  ObDereferenceObject( Msg->RepliedToThread );
415 
416  Msg->RepliedToThread = NULL;
417  }
418 
419  LpcpTrace(( "Cleanup Msg %lx (%d) for Thread %lx allocated\n", Msg, IsListEmpty( &Msg->Entry ), Thread ));
420 
421  LpcpFreeToPortZone( Msg, LPCP_MUTEX_OWNED | LPCP_MUTEX_RELEASE_ON_RETURN );
422  }
423  else {
424 
425  //
426  // Free the global lpc mutex.
427  //
428 
429  LpcpReleaseLpcpLock();
430  }
431 
432  //
433  // And return to our caller
434  //
435 
436  return;
437 }
