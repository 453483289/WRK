32 /*++
33 
34 Routine Description:
35 
36  This procedure maps the specified physical page into hyper space
37  and returns the virtual address which maps the page.
38 
39  ************************************
40  * *
41  * Returns with a spin lock held!!! *
42  * *
43  ************************************
44 
45 Arguments:
46 
47  Process - Supplies the current process.
48 
49  PageFrameIndex - Supplies the physical page number to map.
50 
51  OldIrql - Supplies a pointer in which to return the entry IRQL.
52 
53 Return Value:
54 
55  Returns the address where the requested page was mapped.
56 
57  RETURNS WITH THE HYPERSPACE SPIN LOCK HELD!!!!
58 
59  The routine MiUnmapHyperSpaceMap MUST be called to release the lock!!!!
60 
61 Environment:
62 
63  Kernel mode.
64 
65 --*/
66 
67 {
68  MMPTE TempPte;
69  PMMPFN Pfn1;
70  PMMPTE PointerPte;
71  PFN_NUMBER offset;
72 
73  ASSERT (PageFrameIndex != 0);
74  ASSERT (MI_IS_PFN (PageFrameIndex));
75 
76  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
77 
78  TempPte = ValidPtePte;
79 
80  if (Pfn1->u3.e1.CacheAttribute == MiWriteCombined) {
81  MI_SET_PTE_WRITE_COMBINE (TempPte);
82  }
83  else if (Pfn1->u3.e1.CacheAttribute == MiNonCached) {
84  MI_DISABLE_CACHING (TempPte);
85  }
86 
87  TempPte.u.Hard.PageFrameNumber = PageFrameIndex;
88 
89  PointerPte = MmFirstReservedMappingPte;
90 
91 #if defined(NT_UP)
92  UNREFERENCED_PARAMETER (Process);
93 #endif
94 
95  LOCK_HYPERSPACE (Process, OldIrql);
96 
97  //
98  // Get offset to first free PTE.
99  //
100 
101  offset = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
102 
103  if (offset == 0) {
104 
105  //
106  // All the reserved PTEs have been used, make them all invalid.
107  //
108 
109 #if DBG
110  {
111  PMMPTE LastPte;
112 
113  LastPte = PointerPte + NUMBER_OF_MAPPING_PTES;
114 
115  do {
116  ASSERT (LastPte->u.Long == 0);
117  LastPte -= 1;
118  } while (LastPte > PointerPte);
119  }
120 #endif
121 
122  //
123  // Use the page frame number field of the first PTE as an
124  // offset into the available mapping PTEs.
125  //
126 
127  offset = NUMBER_OF_MAPPING_PTES;
128 
129  //
130  // Flush entire TB only on processors executing this process.
131  //
132 
133  MI_FLUSH_PROCESS_TB (FALSE);
134  }
135 
136  //
137  // Change offset for next time through.
138  //
139 
140  PointerPte->u.Hard.PageFrameNumber = offset - 1;
141 
142  //
143  // Point to free entry and make it valid.
144  //
145 
146  PointerPte += offset;
147  ASSERT (PointerPte->u.Hard.Valid == 0);
148 
149  MI_WRITE_VALID_PTE (PointerPte, TempPte);
150 
151  //
152  // Return the VA that maps the page.
153  //
154 
155  return MiGetVirtualAddressMappedByPte (PointerPte);
156 }

293  IN PMMPFN Pfn1,

297 /*++
298 
299 Routine Description:
300 
301  This procedure maps the specified physical pages for the zero page thread
302  and returns the virtual address which maps them.
303 
304  This is ONLY to be used by THE zeroing page thread.
305 
306 Arguments:
307 
308  Pfn1 - Supplies the pointer to the physical page numbers to map.
309 
310  NumberOfPages - Supplies the number of pages to map.
311 
312 Return Value:
313 
314  Returns the virtual address where the specified physical pages were
315  mapped.
316 
317 Environment:
318 
319  PASSIVE_LEVEL.
320 
321 --*/
322 
323 {
324  PFN_NUMBER offset;
325  MMPTE TempPte;
326  MMPTE DefaultCachedPte;
327  PMMPTE PointerPte;
328  PFN_NUMBER PageFrameIndex;
329 
330  ASSERT (KeGetCurrentIrql() == PASSIVE_LEVEL);
331 
332  ASSERT (NumberOfPages != 0);
333  ASSERT (NumberOfPages <= NUMBER_OF_ZEROING_PTES);
334 
335  PointerPte = MiFirstReservedZeroingPte;
336 
337  //
338  // Get offset to first free PTE.
339  //
340 
341  offset = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
342 
343  if (NumberOfPages > offset) {
344 
345  //
346  // Not enough unused PTEs left, make them all invalid.
347  //
348 
349 #if DBG
350  {
351  PMMPTE LastPte;
352 
353  LastPte = PointerPte + NUMBER_OF_ZEROING_PTES;
354 
355  do {
356  ASSERT (LastPte->u.Long == 0);
357  LastPte -= 1;
358  } while (LastPte > PointerPte);
359  }
360 #endif
361 
362  //
363  // Use the page frame number field of the first PTE as an
364  // offset into the available zeroing PTEs.
365  //
366 
367  offset = NUMBER_OF_ZEROING_PTES;
368  PointerPte->u.Hard.PageFrameNumber = offset;
369 
370  //
371  // Flush entire TB on all processors as this thread may have migrated.
372  //
373 
374  MI_FLUSH_ENTIRE_TB (0x25);
375  }
376 
377  //
378  // Change offset for next time through.
379  //
380 
381  PointerPte->u.Hard.PageFrameNumber = offset - NumberOfPages;
382 
383  //
384  // Point to free entries and make them valid. Note that the frames
385  // are mapped in reverse order but our caller doesn't care anyway.
386  //
387 
388  PointerPte += (offset + 1);
389 
390  DefaultCachedPte = ValidPtePte;
391 
392  ASSERT (Pfn1 != (PMMPFN) MM_EMPTY_LIST);
393 
394  do {
395 
396  PageFrameIndex = MI_PFN_ELEMENT_TO_INDEX (Pfn1);
397 
398  TempPte = DefaultCachedPte;
399 
400  if (Pfn1->u3.e1.CacheAttribute == MiWriteCombined) {
401  MI_SET_PTE_WRITE_COMBINE (TempPte);
402  }
403  else if (Pfn1->u3.e1.CacheAttribute == MiNonCached) {
404  MI_DISABLE_CACHING (TempPte);
405  }
406 
407  TempPte.u.Hard.PageFrameNumber = PageFrameIndex;
408 
409  PointerPte -= 1;
410 
411  ASSERT (PointerPte->u.Hard.Valid == 0);
412 
413  MI_WRITE_VALID_PTE (PointerPte, TempPte);
414 
415  Pfn1 = (PMMPFN) Pfn1->u1.Flink;
416 
417  } while (Pfn1 != (PMMPFN) MM_EMPTY_LIST);
418 
419  //
420  // Return the VA that maps the page.
421  //
422 
423  return MiGetVirtualAddressMappedByPte (PointerPte);
424 }
