112  OBJECT_ATTRIBUTES ObjectAttributes;
113  HANDLE Thread;
114  ULONG i;
115  NTSTATUS Status = STATUS_SUCCESS;
116 
117  //
118  // Create worker threads to handle normal and paging overflow reads.
119  //
120 
121  InitializeObjectAttributes(&ObjectAttributes, NULL, 0, NULL, NULL);
122 
123  for (i=0; i < 2; i++) {
124 
125  //
126  // Initialize the FsRtl stack overflow work Queue objects.
127  //
128 
129  KeInitializeQueue(&FsRtlWorkerQueues[i], 0);
130 
131  Status = PsCreateSystemThread( &Thread,
132  THREAD_ALL_ACCESS,
133  &ObjectAttributes,
134  0L,
135  NULL,
136  FsRtlWorkerThread,
137  ULongToPtr( i ));
138 
139  if (!NT_SUCCESS( Status )) {
140 
141  return Status;
142  }
143 
144  ZwClose( Thread );
145  }
146 
147  //
148  // Initialize the serial workitem so we can guarantee to post items
149  // for paging files to the worker threads.
150  //
151 
152  KeInitializeEvent( &StackOverflowFallbackSerialEvent, SynchronizationEvent, TRUE );
153 
154  //
155  // Pass a NOP routine through FsRtlPostStackOverfow() in order to flush out
156  // any prefetchw instructions that may lie in that path. Otherwise we'll
157  // try to patch the prefetchw instruction when we can ill afford the stack
158  // space.
159  //
160 
161 #if defined(_AMD64_)
162 
163  FsRtlPostStackOverflow( NULL, NULL, FsRtlpNopStackOverflowRoutine );

250 FsRtlpPostStackOverflow (
251  IN PVOID Context,
252  IN PKEVENT Event,
253  IN PFSRTL_STACK_OVERFLOW_ROUTINE StackOverflowRoutine,
254  IN BOOLEAN PagingFile
255  )
256 
257 /*++
258 
259 Routine Description:
260 
261  This routines posts a stack overflow item to the stack overflow
262  thread and returns.
263 
264 Arguments:
265 
266  Context - Supplies the context to pass to the stack overflow
267  call back routine. If the low order bit is set, then
268  this overflow was a read to a paging file.
269 
270  Event - Supplies a pointer to an event to pass to the stack
271  overflow call back routine.
272 
273  StackOverflowRoutine - Supplies the call back to use when
274  processing the request in the overflow thread.
275 
276  PagingFile - Indicates if the read is destined to a paging file.
277 
278 Return Value:
279 
280  None.
281 
282 --*/
283 
284 {
285  PSTACK_OVERFLOW_ITEM StackOverflowItem;
286 
287  //
288  // Allocate a stack overflow work item it will later be deallocated by
289  // the stack overflow thread. Conserve stack by raising here.
290  //
291 
292  StackOverflowItem = ExAllocatePoolWithTag( NonPagedPool,
293  sizeof(STACK_OVERFLOW_ITEM),
294  MODULE_POOL_TAG );
295 
296  //
297  // If this fails, go to the fallback item for the paging file overflows.
298  // We can't have a single fallback item for non-pagingfile IO since this
299  // could lead to deadlocks if it waits on a thread that itself needs
300  // the fallback item.
301  //
302 
303  if (StackOverflowItem == NULL) {
304 
305  if (!PagingFile) {
306 
307  ExRaiseStatus( STATUS_INSUFFICIENT_RESOURCES );
308  }
309 
310  KeWaitForSingleObject( &StackOverflowFallbackSerialEvent,
311  Executive,
312  KernelMode,
313  FALSE,
314  NULL );
315 
316  StackOverflowItem = &StackOverflowFallback;
317  }
318 
319  //
320  // Fill in the fields in the new item
321  //
322 
323  StackOverflowItem->Context = Context;
324  StackOverflowItem->Event = Event;
325  StackOverflowItem->StackOverflowRoutine = StackOverflowRoutine;
326 
327  ExInitializeWorkItem( &StackOverflowItem->Item,
328  &FsRtlStackOverflowRead,
329  StackOverflowItem );
330 
331  //
332  // Safely add it to the overflow queue
333  //
334 
335  KeInsertQueue( &FsRtlWorkerQueues[PagingFile],
336  &StackOverflowItem->Item.List );
337 
338  //
339  // And return to our caller
340  //
341 
342  return;
343 }

356 
357 Routine Description:
358 
359  This routine processes all of the stack overflow request posted by
360  the various file systems
361 
362 Arguments:
363

370 {
371  PSTACK_OVERFLOW_ITEM StackOverflowItem;
372

375  // TopLevelIrp field appropriately so that recursive reads
376  // from this point will not think they are top level.
377  //
378 
379  PsGetCurrentThread()->TopLevelIrp = FSRTL_FSP_TOP_LEVEL_IRP;

382  // Get a pointer to the stack overflow item and then call
383  // the callback routine to do the work
384  //
385

387 
388  (StackOverflowItem->StackOverflowRoutine)(StackOverflowItem->Context,

393  //

395  if (StackOverflowItem == &StackOverflowFallback) {
396

398 
399  } else {
400 
401  ExFreePool( StackOverflowItem );
402  }
403 
404  PsGetCurrentThread()->TopLevelIrp = (ULONG_PTR)NULL;

407 VOID
408 FsRtlWorkerThread(
409  IN PVOID StartContext
410  )
411 
412 {
413  PLIST_ENTRY Entry;
414  PWORK_QUEUE_ITEM WorkItem;
415  ULONG PagingFile = (ULONG)(ULONG_PTR)StartContext;
416 
417  //
418  // Set our priority to low realtime, or +1 for PagingFile.
419  //
420 
421  (VOID)KeSetPriorityThread( &PsGetCurrentThread()->Tcb,
422  LOW_REALTIME_PRIORITY + PagingFile );
423 
424  //
425  // Loop forever waiting for a work queue item, calling the processing
426  // routine, and then waiting for another work queue item.
427  //
428 
429  do {
430 
431  //
432  // Wait until something is put in the queue.
433  //
434  // By specifying a wait mode of KernelMode, the thread's kernel stack is
435  // NOT swappable
436  //
437 
438  Entry = KeRemoveQueue(&FsRtlWorkerQueues[PagingFile], KernelMode, NULL);
439  WorkItem = CONTAINING_RECORD(Entry, WORK_QUEUE_ITEM, List);
440 
441  //
442  // Execute the specified routine.
443  //
444 
445  (WorkItem->WorkerRoutine)(WorkItem->Parameter);
446  if (KeGetCurrentIrql() != 0) {
447  KeBugCheckEx(
448  IRQL_NOT_LESS_OR_EQUAL,
449  (ULONG_PTR)WorkItem->WorkerRoutine,
450  (ULONG_PTR)KeGetCurrentIrql(),
451  (ULONG_PTR)WorkItem->WorkerRoutine,
452  (ULONG_PTR)WorkItem
453  );
454  }
455 
456  } while(TRUE);
457 }
