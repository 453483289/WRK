420 NTSTATUS
421 ExpRaiseHardError (
422  IN NTSTATUS ErrorStatus,
423  IN ULONG NumberOfParameters,
424  IN ULONG UnicodeStringParameterMask,
425  IN PULONG_PTR Parameters,
426  IN ULONG ValidResponseOptions,
427  OUT PULONG Response
428  )
429 {
430  PTEB Teb;
431  PETHREAD Thread;
432  PEPROCESS Process;
433  ULONG_PTR MessageBuffer[PORT_TOTAL_MAXIMUM_MESSAGE_LENGTH/sizeof(ULONG_PTR)];
434  PHARDERROR_MSG m;
435  NTSTATUS Status;
436  HANDLE ErrorPort;
437  KPROCESSOR_MODE PreviousMode;
438  BOOLEAN DoingShutdown;
439 
440  PAGED_CODE();
441 
442  m = (PHARDERROR_MSG)&MessageBuffer[0];
443  PreviousMode = KeGetPreviousMode();
444 
445  DoingShutdown = FALSE;
446 
447  if (ValidResponseOptions == OptionShutdownSystem) {
448 
449  //
450  // Check to see if the caller has the privilege to make this call.
451  //
452 
453  if (!SeSinglePrivilegeCheck (SeShutdownPrivilege, PreviousMode)) {
454  return STATUS_PRIVILEGE_NOT_HELD;
455  }
456 
457  ExReadyForErrors = FALSE;
458  HardErrorState = SHUTDOWN;
459  DoingShutdown = TRUE;
460  }
461 
462  Thread = PsGetCurrentThread();
463  Process = PsGetCurrentProcess();
464 
465  //
466  // If the default handler is not installed, then
467  // call the fatal hard error handler if the error
468  // status is error
469  //
470  // Let GDI override this since it does not want to crash the machine
471  // when a bad driver was loaded via MmLoadSystemImage.
472  //
473 
474  if ((Thread->CrossThreadFlags & PS_CROSS_THREAD_FLAGS_HARD_ERRORS_DISABLED) == 0) {
475 
476  if (NT_ERROR(ErrorStatus) && (HardErrorState == STARTING || DoingShutdown)) {
477 
478  ExpSystemErrorHandler (
479  ErrorStatus,
480  NumberOfParameters,
481  UnicodeStringParameterMask,
482  Parameters,
483  (BOOLEAN)((PreviousMode != KernelMode) ? TRUE : FALSE));
484  }
485  }
486 
487  //
488  // If the process has an error port, then if it wants default
489  // handling, use its port. If it disabled default handling, then
490  // return the error to the caller. If the process does not
491  // have a port, then use the registered default handler.
492  //
493 
494  ErrorPort = NULL;
495 
496  if (Process->ExceptionPort) {
497  if (Process->DefaultHardErrorProcessing & 1) {
498  ErrorPort = Process->ExceptionPort;
499  } else {
500 
501  //
502  // If error processing is disabled, check the error override
503  // status.
504  //
505 
506  if (ErrorStatus & HARDERROR_OVERRIDE_ERRORMODE) {
507  ErrorPort = Process->ExceptionPort;
508  }
509  }
510  } else {
511  if (Process->DefaultHardErrorProcessing & 1) {
512  ErrorPort = ExpDefaultErrorPort;
513  } else {
514 
515  //
516  // If error processing is disabled, check the error override
517  // status.
518  //
519 
520  if (ErrorStatus & HARDERROR_OVERRIDE_ERRORMODE) {
521  ErrorPort = ExpDefaultErrorPort;
522  }
523  }
524  }
525 
526  if ((Thread->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_HARD_ERRORS_DISABLED) != 0) {
527  ErrorPort = NULL;
528  }
529 
530  if ((ErrorPort != NULL) && (!IS_SYSTEM_THREAD(Thread))) {
531  Teb = (PTEB)PsGetCurrentThread()->Tcb.Teb;
532  try {
533  if (Teb->HardErrorMode & RTL_ERRORMODE_FAILCRITICALERRORS) {
534  ErrorPort = NULL;
535  }
536  } except (EXCEPTION_EXECUTE_HANDLER) {
537  ;
538  }
539  }
540 
541  if (ErrorPort == NULL) {
542  *Response = (ULONG)ResponseReturnToCaller;
543  return STATUS_SUCCESS;
544  }
545 
546  if (Process == ExpDefaultErrorPortProcess) {
547  if (NT_ERROR(ErrorStatus)) {
548  ExpSystemErrorHandler (ErrorStatus,
549  NumberOfParameters,
550  UnicodeStringParameterMask,
551  Parameters,
552  (BOOLEAN)((PreviousMode != KernelMode) ? TRUE : FALSE));
553  }
554  *Response = (ULONG)ResponseReturnToCaller;
555  Status = STATUS_SUCCESS;
556  return Status;
557  }
558 
559  m->h.u1.Length = HARDERROR_API_MSG_LENGTH;
560  m->h.u2.ZeroInit = LPC_ERROR_EVENT;
561  m->Status = ErrorStatus & ~HARDERROR_OVERRIDE_ERRORMODE;
562  m->ValidResponseOptions = ValidResponseOptions;
563  m->UnicodeStringParameterMask = UnicodeStringParameterMask;
564  m->NumberOfParameters = NumberOfParameters;
565 
566  if (Parameters != NULL) {
567  try {
568  RtlCopyMemory (&m->Parameters,
569  Parameters,
570  sizeof(ULONG_PTR)*NumberOfParameters);
571  } except (EXCEPTION_EXECUTE_HANDLER) {
572  }
573  }
574 
575  KeQuerySystemTime(&m->ErrorTime);
576 
577  Status = LpcRequestWaitReplyPortEx (ErrorPort,
578  (PPORT_MESSAGE) m,
579  (PPORT_MESSAGE) m);
580 
581  if (NT_SUCCESS(Status)) {
582  switch (m->Response) {
583  case ResponseReturnToCaller :
584  case ResponseNotHandled :
585  case ResponseAbort :
586  case ResponseCancel :
587  case ResponseIgnore :
588  case ResponseNo :
589  case ResponseOk :
590  case ResponseRetry :
591  case ResponseYes :
592  case ResponseTryAgain :
593  case ResponseContinue :
594  break;
595  default:
596  m->Response = (ULONG)ResponseReturnToCaller;
597  break;
598  }
599  *Response = m->Response;
600  }
601 
602  return Status;
603 }

606 NtRaiseHardError (
607  __in NTSTATUS ErrorStatus,
608  __in ULONG NumberOfParameters,
609  __in ULONG UnicodeStringParameterMask,
610  __in_ecount(NumberOfParameters) PULONG_PTR Parameters,
611  __in ULONG ValidResponseOptions,
612  __out PULONG Response
613  )
614 {
615  NTSTATUS Status;
616  ULONG_PTR CapturedParameters[MAXIMUM_HARDERROR_PARAMETERS];
617  KPROCESSOR_MODE PreviousMode;
618  ULONG LocalResponse;
619  UNICODE_STRING CapturedString;
620  ULONG Counter;
621 
622  PAGED_CODE();
623 
624  if (NumberOfParameters > MAXIMUM_HARDERROR_PARAMETERS) {
625  return STATUS_INVALID_PARAMETER_2;
626  }
627 
628  if (ARGUMENT_PRESENT(Parameters) && NumberOfParameters == 0) {
629  return STATUS_INVALID_PARAMETER_2;
630  }
631 
632  PreviousMode = KeGetPreviousMode();
633  if (PreviousMode != KernelMode) {
634  switch (ValidResponseOptions) {
635  case OptionAbortRetryIgnore :
636  case OptionOk :
637  case OptionOkCancel :
638  case OptionRetryCancel :
639  case OptionYesNo :
640  case OptionYesNoCancel :
641  case OptionShutdownSystem :
642  case OptionOkNoWait :
643  case OptionCancelTryContinue:
644  break;
645  default :
646  return STATUS_INVALID_PARAMETER_4;
647  }
648 
649  try {
650  ProbeForWriteUlong(Response);
651 
652  if (ARGUMENT_PRESENT(Parameters)) {
653  ProbeForRead (Parameters,
654  sizeof(ULONG_PTR)*NumberOfParameters,
655  sizeof(ULONG_PTR));
656 
657  RtlCopyMemory (CapturedParameters,
658  Parameters,
659  sizeof(ULONG_PTR)*NumberOfParameters);
660 
661  //
662  // Probe all strings.
663  //
664 
665  if (UnicodeStringParameterMask) {
666 
667  for (Counter = 0;Counter < NumberOfParameters; Counter += 1) {
668 
669  //
670  // if there is a string in this position,
671  // then probe and capture the string
672  //
673 
674  if (UnicodeStringParameterMask & (1<<Counter)) {
675 
676  ProbeForReadSmallStructure ((PVOID)CapturedParameters[Counter],
677  sizeof(UNICODE_STRING),
678  sizeof(ULONG_PTR));
679 
680  RtlCopyMemory (&CapturedString,
681  (PVOID)CapturedParameters[Counter],
682  sizeof(UNICODE_STRING));
683 
684  //
685  // Now probe the string
686  //
687 
688  ProbeForRead (CapturedString.Buffer,
689  CapturedString.MaximumLength,
690  sizeof(UCHAR));
691  }
692  }
693  }
694  }
695  }
696  except(EXCEPTION_EXECUTE_HANDLER) {
697  return GetExceptionCode();
698  }
699 
700  if ((ErrorStatus == STATUS_SYSTEM_IMAGE_BAD_SIGNATURE) &&
701  (KdDebuggerEnabled)) {
702 
703  if ((NumberOfParameters != 0) && (ARGUMENT_PRESENT(Parameters))) {
704  DbgPrint("****************************************************************\n");
705  DbgPrint("* The system detected a bad signature on file %wZ\n",(PUNICODE_STRING)CapturedParameters[0]);
706  DbgPrint("****************************************************************\n");
707  }
708  return STATUS_SUCCESS;
709  }
710 
711  //
712  // Call ExpRaiseHardError. All parameters are probed and everything
713  // should be user-mode.
714  // ExRaiseHardError will squirt all strings into user-mode
715  // without any probing
716  //
717 
718  Status = ExpRaiseHardError (ErrorStatus,
719  NumberOfParameters,
720  UnicodeStringParameterMask,
721  CapturedParameters,
722  ValidResponseOptions,
723  &LocalResponse);
724 
725  try {
726  *Response = LocalResponse;
727  }
728  except (EXCEPTION_EXECUTE_HANDLER) {
729  NOTHING;
730  }
731  }
732  else {
733  Status = ExRaiseHardError (ErrorStatus,
734  NumberOfParameters,
735  UnicodeStringParameterMask,
736  Parameters,
737  ValidResponseOptions,
738  &LocalResponse);
739 
740  *Response = LocalResponse;
741  }
742 
743  return Status;
744 }

747 ExRaiseHardError (
748  __in NTSTATUS ErrorStatus,
749  __in ULONG NumberOfParameters,
750  __in ULONG UnicodeStringParameterMask,
751  __in_ecount(NumberOfParameters) PULONG_PTR Parameters,
752  __in ULONG ValidResponseOptions,
753  __out PULONG Response
754  )
755 {
756  NTSTATUS Status;
757  PULONG_PTR ParameterBlock;
758  PULONG_PTR UserModeParameterBase;
759  PUNICODE_STRING UserModeStringsBase;
760  PUCHAR UserModeStringDataBase;
761  UNICODE_STRING CapturedStrings[MAXIMUM_HARDERROR_PARAMETERS];
762  ULONG LocalResponse;
763  ULONG Counter;
764  SIZE_T UserModeSize;
765 
766  PAGED_CODE();
767 
768  //
769  // If we are in the process of shutting down the system, do not allow
770  // hard errors.
771  //
772 
773  if (ExpTooLateForErrors) {
774 
775  *Response = ResponseNotHandled;
776 
777  return STATUS_SUCCESS;
778  }
779 
780  ParameterBlock = NULL;
781 
782  //
783  // If the parameters contain strings, we need to capture
784  // the strings and the string descriptors and push them into
785  // user-mode.
786  //
787 
788  if (ARGUMENT_PRESENT(Parameters)) {
789  if (UnicodeStringParameterMask) {
790 
791  //
792  // We have strings - push them into usermode.
793  //
794 
795  UserModeSize = (sizeof(ULONG_PTR)+sizeof(UNICODE_STRING))*MAXIMUM_HARDERROR_PARAMETERS;
796  UserModeSize += sizeof(UNICODE_STRING);
797 
798  for (Counter = 0; Counter < NumberOfParameters; Counter += 1) {
799 
800  //
801  // If there is a string in this position,
802  // then probe and capture the string.
803  //
804 
805  if (UnicodeStringParameterMask & 1<<Counter) {
806 
807  RtlCopyMemory (&CapturedStrings[Counter],
808  (PVOID)Parameters[Counter],
809  sizeof(UNICODE_STRING));
810 
811  UserModeSize += CapturedStrings[Counter].MaximumLength;
812  }
813  }
814 
815  //
816  // Now we have the user-mode size all figured out.
817  // Allocate some memory and point to it with the
818  // parameter block. Then go through and copy all
819  // of the parameters, string descriptors, and
820  // string data into the memory.
821  //
822 
823  Status = ZwAllocateVirtualMemory (NtCurrentProcess(),
