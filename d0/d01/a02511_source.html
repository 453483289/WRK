78 /*++
79 
80 Routine Description:
81 
82  This routine allocates the top level page directory pointer structure.
83  This structure will contain 4 PDPTEs.
84 
85 Arguments:
86 
87  Va - Supplies a place to put the virtual address this page can be accessed
88  at.
89 
90 Return Value:
91 
92  Returns a virtual and physical address suitable for use as a top
93  level page directory pointer page. The page returned must be below
94  physical 4GB as required by the processor.
95 
96  Returns 0 if no page was allocated.
97 
98 Environment:
99 
100  Kernel mode. No locks may be held.
101 
102 --*/
103 
104 {
105  LOGICAL FlushedOnce;
106  PPAE_ENTRY Pae2;
107  PPAE_ENTRY Pae3;
108  PPAE_ENTRY Pae3Base;
109  PPAE_ENTRY Pae;
110  PPAE_ENTRY PaeBase;
111  PFN_NUMBER PageFrameIndex;
112  PSLIST_ENTRY SingleListEntry;
113  ULONG j;
114  ULONG Entries;
115  KLOCK_QUEUE_HANDLE LockHandle;
116 #if DBG
117  PMMPFN Pfn1;
118 #endif
119 
120  FlushedOnce = FALSE;
121 
122  ASSERT (KeGetCurrentIrql() <= APC_LEVEL);
123 
124  do {
125 
126  //
127  // Pop an entry from the freelist.
128  //
129 
130  SingleListEntry = InterlockedPopEntrySList (&MiPaeEntrySList);
131 
132  if (SingleListEntry != NULL) {
133  Pae = CONTAINING_RECORD (SingleListEntry,
134  PAE_ENTRY,
135  NextPae);
136 
137  PaeBase = (PPAE_ENTRY)PAGE_ALIGN(Pae);
138 
139  *Va = Pae;
140 
141  PageFrameIndex = PaeBase->PaeEntry.PageFrameNumber;
142  ASSERT (PageFrameIndex <= MM_HIGHEST_PAE_PAGE);
143 
144  return (PageFrameIndex << PAGE_SHIFT) + BYTE_OFFSET (Pae);
145  }
146 
147  KeAcquireInStackQueuedSpinLock (&MiPaeLock, &LockHandle);
148 
149  if (MiFreePaeEntries != 0) {
150 
151  ASSERT (IsListEmpty (&MiFirstFreePae.PaeEntry.ListHead) == 0);
152 
153  Pae = (PPAE_ENTRY) RemoveHeadList (&MiFirstFreePae.PaeEntry.ListHead);
154 
155  PaeBase = (PPAE_ENTRY)PAGE_ALIGN(Pae);
156  PaeBase->PaeEntry.EntriesInUse += 1;
157 #if DBG
158  RtlZeroMemory ((PVOID)Pae, sizeof(PAE_ENTRY));
159 
160  Pfn1 = MI_PFN_ELEMENT (PaeBase->PaeEntry.PageFrameNumber);
161  ASSERT (Pfn1->u2.ShareCount == 1);
162  ASSERT (Pfn1->u3.e2.ReferenceCount == 1);
163  ASSERT (Pfn1->u3.e1.PageLocation == ActiveAndValid);
164  ASSERT (Pfn1->u3.e1.CacheAttribute == MiCached);
165 #endif
166 
167  MiFreePaeEntries -= 1;
168 
169  //
170  // Since we're holding the spinlock, dequeue a chain of entries
171  // for the SLIST.
172  //
173 
174  Entries = MiFreePaeEntries;
175 
176  if (Entries != 0) {
177  if (Entries > MINIMUM_PAE_SLIST_THRESHOLD) {
178  Entries = MINIMUM_PAE_SLIST_THRESHOLD;
179  }
180 
181  ASSERT (IsListEmpty (&MiFirstFreePae.PaeEntry.ListHead) == 0);
182 
183  Pae2 = (PPAE_ENTRY) RemoveHeadList (&MiFirstFreePae.PaeEntry.ListHead);
184  Pae2->NextPae.Next = NULL;
185  Pae3 = Pae2;
186  Pae3Base = (PPAE_ENTRY)PAGE_ALIGN(Pae3);
187  Pae3Base->PaeEntry.EntriesInUse += 1;
188 
189  for (j = 1; j < Entries; j += 1) {
190  ASSERT (IsListEmpty (&MiFirstFreePae.PaeEntry.ListHead) == 0);
191 
192  Pae3->NextPae.Next = (PSLIST_ENTRY) RemoveHeadList (&MiFirstFreePae.PaeEntry.ListHead);
193 
194  Pae3 = (PPAE_ENTRY) Pae3->NextPae.Next;
195  Pae3Base = (PPAE_ENTRY)PAGE_ALIGN(Pae3);
196  Pae3Base->PaeEntry.EntriesInUse += 1;
197  }
198 
199  MiFreePaeEntries -= Entries;
200 
201  KeReleaseInStackQueuedSpinLock (&LockHandle);
202 
203  Pae3->NextPae.Next = NULL;
204 
205  InterlockedPushListSList (&MiPaeEntrySList,
206  (PSLIST_ENTRY) Pae2,
207  (PSLIST_ENTRY) Pae3,
208  Entries);
209  }
210  else {
211  KeReleaseInStackQueuedSpinLock (&LockHandle);
212  }
213 
214  ASSERT (KeGetCurrentIrql() <= APC_LEVEL);
215  *Va = Pae;
216 
217  PageFrameIndex = PaeBase->PaeEntry.PageFrameNumber;
218  ASSERT (PageFrameIndex <= MM_HIGHEST_PAE_PAGE);
219 
220  return (PageFrameIndex << PAGE_SHIFT) + BYTE_OFFSET (Pae);
221  }
222 
223  KeReleaseInStackQueuedSpinLock (&LockHandle);
224 
225  if (FlushedOnce == TRUE) {
226  break;
227  }
228 
229  //
230  // No free pages in the cachelist, replenish the list now.
231  //
232 
233  if (MiPaeAllocatePages () == 0) {
234 
235  InterlockedIncrement (&MiDelayPageFaults);
236 
237  //
238  // Attempt to move pages to the standby list.
239  //
240 
241  MmEmptyAllWorkingSets ();
242  MiFlushAllPages();
243 
244  KeDelayExecutionThread (KernelMode,
245  FALSE,
246  (PLARGE_INTEGER)&MmHalfSecond);
247 
248  InterlockedDecrement (&MiDelayPageFaults);
249 
250  FlushedOnce = TRUE;
251 
252  //
253  // Since all the working sets have been trimmed, check whether
254  // another thread has replenished our list. If not, then attempt
255  // to do so since the working set pain has already been absorbed.
256  //
257 
258  if (MiFreePaeEntries < MINIMUM_PAE_THRESHOLD) {
259  MiPaeAllocatePages ();
260  }
261  }
262 
263  } while (TRUE);
264 
265  ASSERT (KeGetCurrentIrql() <= APC_LEVEL);
266 
267  return 0;
268 }

275 /*++
276 
277 Routine Description:
278 
279  This routine releases the top level page directory pointer page.
280 
281 Arguments:
282 
283  PageFrameIndex - Supplies the top level page directory pointer page.
284 
285 Return Value:
286 
287  None.
288 
289 Environment:
290 
291  Kernel mode. No locks may be held.
292 
293 --*/
294 
295 {
296  ULONG i;
297  PLIST_ENTRY NextEntry;
298  PPAE_ENTRY PaeBase;
299  KLOCK_QUEUE_HANDLE LockHandle;
300 
301 #if DBG
302  PMMPTE PointerPte;
303  PFN_NUMBER PageFrameIndex;
304  PMMPFN Pfn1;
305 
306  PointerPte = MiGetPteAddress (Pae);
307  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
308 
309  //
310  // This page must be in the first 4GB of RAM.
311  //
312 
313  ASSERT (PageFrameIndex <= MM_HIGHEST_PAE_PAGE);
314 
315  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
316 
317  ASSERT (Pfn1->u2.ShareCount == 1);
318  ASSERT (Pfn1->u3.e2.ReferenceCount == 1);
319  ASSERT (Pfn1->u3.e1.PageLocation == ActiveAndValid);
320  ASSERT (Pfn1->u3.e1.CacheAttribute == MiCached);
321 #endif
322 
323  if (ExQueryDepthSList (&MiPaeEntrySList) < MINIMUM_PAE_SLIST_THRESHOLD) {
324  InterlockedPushEntrySList (&MiPaeEntrySList, &Pae->NextPae);
325  return;
326  }
327 
328  PaeBase = (PPAE_ENTRY)PAGE_ALIGN(Pae);
329 
330  KeAcquireInStackQueuedSpinLock (&MiPaeLock, &LockHandle);
331 
332  PaeBase->PaeEntry.EntriesInUse -= 1;
333 
334  if ((PaeBase->PaeEntry.EntriesInUse == 0) &&
335  (MiFreePaeEntries > EXCESS_PAE_THRESHOLD)) {
336 
337  //
338  // Free the entire page.
339  //
340 
341  i = 1;
342  NextEntry = MiFirstFreePae.PaeEntry.ListHead.Flink;
343  while (NextEntry != &MiFirstFreePae.PaeEntry.ListHead) {
344 
345  Pae = CONTAINING_RECORD (NextEntry,
346  PAE_ENTRY,
347  PaeEntry.ListHead);
348 
349  if (PAGE_ALIGN(Pae) == PaeBase) {
350  RemoveEntryList (NextEntry);
351  i += 1;
352  }
353  NextEntry = Pae->PaeEntry.ListHead.Flink;
354  }
355  ASSERT (i == PAES_PER_PAGE - 1);
356  MiFreePaeEntries -= (PAES_PER_PAGE - 1);
357  KeReleaseInStackQueuedSpinLock (&LockHandle);
358 
359  MiPaeFreePages (PaeBase);
360  }
361  else {
362 
363  InsertTailList (&MiFirstFreePae.PaeEntry.ListHead,
364  &Pae->PaeEntry.ListHead);
365  MiFreePaeEntries += 1;
366  KeReleaseInStackQueuedSpinLock (&LockHandle);
367  }
368 
369  return;
370 }

373 MiPaeAllocatePages (
374  VOID
375  )
376 
377 /*++
378 
379 Routine Description:
380 
381  This routine replenishes the PAE top level mapping list.
382 
383 Arguments:
384 
385  None.
386 
387 Return Value:
388 
389  The number of pages allocated.
390 
391 Environment:
392 
393  Kernel mode, IRQL of APC_LEVEL or below.
394 
395 --*/
396 {
397  PMDL MemoryDescriptorList;
398  LONG AllocatedPaePages;
399  ULONG i;
400  ULONG j;
401  PPFN_NUMBER SlidePage;
402  PPFN_NUMBER Page;
403  PFN_NUMBER PageFrameIndex;

406  PVOID BaseAddress;
407  PPAE_ENTRY Pae;

409  MMPTE TempPte;

411  PHYSICAL_ADDRESS LowAddress;
412  PHYSICAL_ADDRESS SkipBytes;
413  KLOCK_QUEUE_HANDLE LockHandle;
414 
415 #if defined (_MI_MORE_THAN_4GB_)
416  if (MiNoLowMemory != 0) {
417  BaseAddress = MiAllocateLowMemory (PAGE_SIZE,
418  0,
419  MiNoLowMemory - 1,
420  0,
421  (PVOID)0x123,
422  MmCached,
423  'DeaP');

426  }

429

432  Pae->PaeEntry.PageFrameNumber = PageFrameIndex;
433  Pae += 1;
434 
435  KeAcquireInStackQueuedSpinLock (&MiPaeLock, &LockHandle);
436 
437  for (i = 1; i < PAES_PER_PAGE; i += 1) {
438  InsertTailList (&MiFirstFreePae.PaeEntry.ListHead,
439  &Pae->PaeEntry.ListHead);
440  Pae += 1;
441  }
442  MiFreePaeEntries += (PAES_PER_PAGE - 1);
443

445

448  }
449 #endif
450 
451  NumberOfPages = REPLENISH_PAE_SIZE / PAES_PER_PAGE;
452  AllocatedPaePages = 0;
453 
454  HighAddress.QuadPart = (ULONGLONG)_4gb - 1;
455  LowAddress.QuadPart = 0;
456  SkipBytes.QuadPart = 0;
457 
458  //
459  // This is a potentially expensive call so pick up a chunk of pages
460  // at once to amortize the cost.
461  //
462 
463  MemoryDescriptorList = MiAllocatePagesForMdl (LowAddress,
464  HighAddress,
465  SkipBytes,
466  NumberOfPages << PAGE_SHIFT,
467  MiCached,
468  0);
469 
470  if (MemoryDescriptorList == NULL) {
471  return 0;
472  }
473 
474  ActualPages = MemoryDescriptorList->ByteCount >> PAGE_SHIFT;
475 
476  TempPte = ValidKernelPte;
477  Page = (PPFN_NUMBER)(MemoryDescriptorList + 1);
478 
479  //
480  // Map each page individually as they may need to be freed individually
481  // later.
482  //
483 
484  for (i = 0; i < ActualPages; i += 1) {
485  PageFrameIndex = *Page;
486 
487  PointerPte = MiReserveSystemPtes (1, SystemPteSpace);
488 
489  if (PointerPte == NULL) {
490 
491  //
492  // Free any remaining pages in the MDL as they are not mapped.
493  // Slide the MDL pages forward so the mapped ones are kept.
494  //
495 
496  MmInitializeMdl (MemoryDescriptorList,
497  0,
498  (ActualPages - i) << PAGE_SHIFT);
499 
500  SlidePage = (PPFN_NUMBER)(MemoryDescriptorList + 1);
501 
502  while (i < ActualPages) {
503  i += 1;
504  *SlidePage = *Page;
505  SlidePage += 1;
506  Page += 1;
507  }
508 
509  MmFreePagesFromMdl (MemoryDescriptorList);
510 
511  break;
512  }
513 
514  TempPte.u.Hard.PageFrameNumber = PageFrameIndex;
515  MI_WRITE_VALID_PTE (PointerPte, TempPte);
516 
517  BaseAddress = MiGetVirtualAddressMappedByPte (PointerPte);
518 
519  Pae = (PPAE_ENTRY) BaseAddress;
520 
521  Pae->PaeEntry.EntriesInUse = 0;
522  Pae->PaeEntry.PageFrameNumber = PageFrameIndex;
523  Pae += 1;
524 
525  //
526  // Put the first chunk into the SLIST if it's still low, and just
527  // enqueue all the other entries normally.
528  //
529 
530  if ((i == 0) &&
531  (ExQueryDepthSList (&MiPaeEntrySList) < MINIMUM_PAE_SLIST_THRESHOLD)) {
532 
533  (Pae - 1)->PaeEntry.EntriesInUse = PAES_PER_PAGE - 1;
534 
535  for (j = 1; j < PAES_PER_PAGE - 1; j += 1) {
536  Pae->NextPae.Next = (PSLIST_ENTRY) (Pae + 1);
537  Pae += 1;
538  }
539 
540  Pae->NextPae.Next = NULL;
541 
542  InterlockedPushListSList (&MiPaeEntrySList,
543  (PSLIST_ENTRY)((PPAE_ENTRY) BaseAddress + 1),
544  (PSLIST_ENTRY) Pae,
545  PAES_PER_PAGE - 1);
546  }
547  else {
548 
549  KeAcquireInStackQueuedSpinLock (&MiPaeLock, &LockHandle);
550 
551  for (j = 1; j < PAES_PER_PAGE; j += 1) {
552  InsertTailList (&MiFirstFreePae.PaeEntry.ListHead,
553  &Pae->PaeEntry.ListHead);
554  Pae += 1;
555  }
556 
557  MiFreePaeEntries += (PAES_PER_PAGE - 1);
558 
559  KeReleaseInStackQueuedSpinLock (&LockHandle);
560  }
561 
562  AllocatedPaePages += 1;
563 
564  Page += 1;
565  }
566 
567  ExFreePool (MemoryDescriptorList);
568 
569  InterlockedExchangeAdd (&MmAllocatedPaePages, AllocatedPaePages);
570 
571  return AllocatedPaePages;
572 }
