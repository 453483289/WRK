585 
586 /*++
587 
588 Routine Description:
589 
590  This function locates the system dll and creates a section for the
591  DLL and maps it into the system process.
592 
593 Arguments:
594 
595  None.

598 
599  TRUE - Initialization was successful.
600

603 --*/

605 {
606 
607  HANDLE File;
608  HANDLE Section;
609  NTSTATUS st;
610  OBJECT_ATTRIBUTES ObjectAttributes;
611  IO_STATUS_BLOCK IoStatus;
612  PVOID NtDllSection;
613 
614  //
615  // First see if we need to load this DLL at all.
616  //
617  if (ExVerifySuite (EmbeddedNT) && (PsEmbeddedNTMask&PS_EMBEDDED_NO_USERMODE)) {
618  return STATUS_SUCCESS;
619  }
620 
621  if (!ReplaceExisting) {
622 
623  ExInitializePushLock(&PspSystemDll.DllLock);
624  }
625 
626  //
627  // Initialize the system DLL
628  //
629 
630  InitializeObjectAttributes (&ObjectAttributes,
631  (PUNICODE_STRING) &PsNtDllPathName,
632  OBJ_CASE_INSENSITIVE|OBJ_KERNEL_HANDLE,
633  NULL,
634  NULL);
635 
636  st = ZwOpenFile (&File,
637  SYNCHRONIZE | FILE_EXECUTE,
638  &ObjectAttributes,
639  &IoStatus,
640  FILE_SHARE_READ,
641  0);
642 
643  if (!NT_SUCCESS (st)) {
644

651  }
652

654  }
655

657

759 NTSTATUS
760 PsMapSystemDll (
761  IN PEPROCESS Process,
762  OUT PVOID *DllBase OPTIONAL,
763  IN LOGICAL UseLargePages
764  )
765 
766 /*++
767 
768 Routine Description:
769 
770  This function maps the system DLL into the specified process.
771 
772 Arguments:
773 
774  Process - Supplies the address of the process to map the DLL into.
775 
776  DllBase - Receives a pointer to the base address of the system DLL.
777 
778  UseLargePages - Attempt the map using large pages. If this fails,
779  the map is automatically performed using small pages.
780 
781 --*/
782 
783 {
784  NTSTATUS st;
785  PVOID ViewBase;
786  LARGE_INTEGER SectionOffset;
787  SIZE_T ViewSize;
788  PVOID CapturedSection;
789 
790  PAGED_CODE();
791 
792  ViewBase = NULL;
793  SectionOffset.LowPart = 0;
794  SectionOffset.HighPart = 0;
795  ViewSize = 0;
796 
797  KeEnterCriticalRegion();
798  ExAcquirePushLockShared(&PspSystemDll.DllLock);
799 
800  CapturedSection = PspSystemDll.Section;
801  ObReferenceObject(CapturedSection);
802 
803  ExReleasePushLockShared(&PspSystemDll.DllLock);
804  KeLeaveCriticalRegion();
805 
806  //
807  // Map the system dll into the user part of the address space
808  //
809 
810  st = MmMapViewOfSection(
811  CapturedSection,
812  Process,
813  &ViewBase,
814  0L,
815  0L,
816  &SectionOffset,
817  &ViewSize,
818  ViewShare,
819  (UseLargePages ? MEM_LARGE_PAGES : 0L),
820  PAGE_READWRITE
821  );
822 
823  ObDereferenceObject(CapturedSection);
824 
825  if (st != STATUS_SUCCESS) {
826 #if DBG
827  DbgPrint("PS: Unable to map system dll at based address.\n");
828 #endif
829  st = STATUS_CONFLICTING_ADDRESSES;
830  }
831 
832  if (ARGUMENT_PRESENT (DllBase)) {
833  *DllBase = ViewBase;
834  }
835 
836  return st;
837 }

937 PsChangeQuantumTable (
938  BOOLEAN ModifyActiveProcesses,
939  ULONG PrioritySeparation
940  )
941 {
942 
943  PEPROCESS Process;
944  PETHREAD CurrentThread;
945  PLIST_ENTRY NextProcess;
946  ULONG QuantumIndex;
947  SCHAR QuantumReset;
948  SCHAR const* QuantumTableBase;
949  PEJOB Job;
950 
951  //
952  // extract priority separation value
953  //
954  switch (PrioritySeparation & PROCESS_PRIORITY_SEPARATION_MASK) {
955  case 3:
956  PsPrioritySeparation = PROCESS_PRIORITY_SEPARATION_MAX;
957  break;
958 
959  default:
960  PsPrioritySeparation = PrioritySeparation & PROCESS_PRIORITY_SEPARATION_MASK;
961  break;
962  }
963 
964  //
965  // determine if we are using fixed or variable quantums
966  //
967 
968  switch (PrioritySeparation & PROCESS_QUANTUM_VARIABLE_MASK) {
969  case PROCESS_QUANTUM_VARIABLE_VALUE:
970  QuantumTableBase = PspVariableQuantums;
971  break;
972 
973  case PROCESS_QUANTUM_FIXED_VALUE:
974  QuantumTableBase = PspFixedQuantums;
975  break;
976 
977  case PROCESS_QUANTUM_VARIABLE_DEF:
978  default:
979  if (MmIsThisAnNtAsSystem ()) {
980  QuantumTableBase = PspFixedQuantums;
981 
982  } else {
983  QuantumTableBase = PspVariableQuantums;
984  }
985 
986  break;
987  }
988 
989  //
990  // determine if we are using long or short
991  //
992  switch (PrioritySeparation & PROCESS_QUANTUM_LONG_MASK) {
993  case PROCESS_QUANTUM_LONG_VALUE:
994  QuantumTableBase = QuantumTableBase + 3;
995  break;
996 
997  case PROCESS_QUANTUM_SHORT_VALUE:
998  break;
999 
1000  case PROCESS_QUANTUM_LONG_DEF:
1001  default:
1002  if (MmIsThisAnNtAsSystem ()) {
1003  QuantumTableBase = QuantumTableBase + 3;
1004  }
1005 
1006  break;
1007  }
1008 
1009  //
1010  // Job Scheduling classes are ONLY meaningful if long fixed quantums
1011  // are selected. In practice, this means stock NTS configurations
1012  //
1013 
1014  if (QuantumTableBase == &PspFixedQuantums[3]) {
1015  PspUseJobSchedulingClasses = TRUE;
1016 
1017  } else {
1018  PspUseJobSchedulingClasses = FALSE;
1019  }
1020 
1021  RtlCopyMemory (PspForegroundQuantum, QuantumTableBase, sizeof(PspForegroundQuantum));
1022  if (ModifyActiveProcesses) {
1023  CurrentThread = PsGetCurrentThread ();
1024  PspLockProcessList (CurrentThread);
1025  NextProcess = PsActiveProcessHead.Flink;
1026  while (NextProcess != &PsActiveProcessHead) {
1027  Process = CONTAINING_RECORD(NextProcess,
1028  EPROCESS,
1029  ActiveProcessLinks);
1030 
1031  if (Process->Vm.Flags.MemoryPriority == MEMORY_PRIORITY_BACKGROUND) {
1032  QuantumIndex = 0;
1033  } else {
1034  QuantumIndex = PsPrioritySeparation;
1035  }
1036 
1037  if (Process->PriorityClass != PROCESS_PRIORITY_CLASS_IDLE) {
1038 
1039  //
1040  // If the process is contained within a JOB, AND we are
1041  // running Fixed, Long Quantums, use the quantum associated
1042  // with the Job's scheduling class
1043  //
1044 
1045  Job = Process->Job;
1046  if (Job != NULL && PspUseJobSchedulingClasses) {
1047  QuantumReset = PspJobSchedulingClasses[Job->SchedulingClass];
1048 
1049  } else {
1050  QuantumReset = PspForegroundQuantum[QuantumIndex];
1051  }
1052 
1053  } else {
1054  QuantumReset = THREAD_QUANTUM;
1055  }
1056 
1057  KeSetQuantumProcess(&Process->Pcb, QuantumReset);
1058  NextProcess = NextProcess->Flink;
1059  }
1060 
1061  PspUnlockProcessList (CurrentThread);
1062  }
1063 
1064  return;
1065 }
