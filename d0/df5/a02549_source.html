13 Abstract:
14 
15  This module houses routines that do fast referencing of object manager
16  objects. This is just a thin layer around the fast ref package in EX.
17  The EX routines are all inline so their description is here.
18 
19  The basic principle of these routines is to allow fast referencing of
20  objects held in pointers protected by locks. This is done by assuming
21  the pointer to an object is aligned on a 8 byte boundary and using the
22  bottom 3 bits of the pointer as a fast referencing mechanism. The
23  assumption of this algorithm is that the pointer changes far less
24  frequently than it is referenced.
25 
26  Given the following bit definition of a
27  pointer:
28 
29  +-----------+---+
30  | p | n |
31  +-----------+---+
32 
33  p << 3 : Object pointer. Bottom three bits are zero. p may be null in
34  which case n must be zero
35  n : Total number of pre-references unused
36 
37  For a non-null p the total number of references on the target object
38  associated with this structure is >= 1 + 7 - n. There is an associated
39  reference for the pointer itself and one for each of the possible
40  extra references.
41 
42  Fast references proceed to perform one of the following transformation:
43 
44  +-----------+---+ +-----------+-----+
45  | p | n | => | p | n-1 | n > 0, p != NULL
46  +-----------+---+ +-----------+-----+
47 
48  +-----------+---+ +-----------+---+
49  | NULL | 0 | => | NULL | 0 | NULL pointers are never fast refed
50  +-----------+---+ +-----------+---+ and never have cached references
51 
52  Slow references do the following transformation:
53 
54  +-----------+---+ +-----------+-----+
55  | p | 0 | => | p | 7 | An addition 8 references are
56  +-----------+---+ +-----------+-----+ added to the object
57 
58  The second transformation is either done under a lock or done by the
59  thread that does the transition with n = 1 => n = 0.
60 
61  The reference obtained by this fast algorithm may be released by
62  dereferencing the target object directly or by attempting to return the
63  reference to the pointer. Returning the reference to the pointer has
64  the following transformations:
65 
66  +-----------+---+ +-----------+-----+
67  | p | n | => | p | n+1 | n < 7, p != NULL
68  +-----------+---+ +-----------+-----+
69 
70  +-----------+---+ +-----------+-----+
71  | p | 7 | => | p | 0 | Dereference the object directly
72  +-----------+---+ +-----------+-----+
73 
74  +-----------+---+ +-----------+-----+ Dereference the object directly
75  | q | n | => | q | n | as the pointer p has been
76  +-----------+---+ +-----------+-----+ replaced by q. q May be NULL
77 
78 --*/

130 /*++
131 
132 Routine Description:
133 
134  This routine attempts a fast reference of an object in a fast ref
135  structure.
136 
137 Arguments:
138 
139  FastRef - Rundown block to be used for the reference
140 
141 Return Value:
142 
143  PVOID - Object that was referenced or NULL if we failed
144 
145 --*/
146 {
147  EX_FAST_REF OldRef;
148  PVOID Object;
149  ULONG RefsToAdd, Unused;
150  //
151  // Attempt the fast reference
152  //
153  OldRef = ExFastReference (FastRef);
154 
155  Object = ExFastRefGetObject (OldRef);
156  //
157  // We fail if there wasn't an object or if it has no cached references
158  // left. Both of these cases had the cached reference count zero.
159  //
160  Unused = ExFastRefGetUnusedReferences (OldRef);
161 
162  if (Unused <= 1) {
163  if (Unused == 0) {
164  return NULL;
165  }
166  //
167  // If we took the counter to zero then attempt to make life easier for
168  // the next referencer by resetting the counter to its max. Since we now
169  // have a reference to the object we can do this.
170  //
171  RefsToAdd = ExFastRefGetAdditionalReferenceCount ();
172  ObReferenceObjectEx (Object, RefsToAdd);
173 
174  //
175  // Try to add the added references to the cache. If we fail then just
176  // release them.
177  //
178  if (!ExFastRefAddAdditionalReferenceCounts (FastRef, Object, RefsToAdd)) {
179  ObDereferenceObjectEx (Object, RefsToAdd);
180  }
181  }
182  return Object;
183 }

287  //

293  //

296  if (RefsToReturn > 0) {

299  }
