174 NTSTATUS
175 RtlpSysVolCheckOwnerAndSecurity(
176  IN HANDLE Handle,
177  IN PACL StandardAcl
178  )
179 
180 {
181  NTSTATUS status;
182  ULONG sdLength, sdLength2;
183  PSECURITY_DESCRIPTOR sd, sd2;
184  PSID sid;
185  BOOLEAN ownerDefaulted, daclPresent, daclDefaulted;
186  PACL acl;
187  ULONG i;
188  PACCESS_ALLOWED_ACE ace;
189  PSID systemSid;
190  UCHAR sidBuffer[2*sizeof(SID)];
191  PSID adminSid;
192  UCHAR sidBuffer2[2*sizeof(SID)];
193 
194  status = NtQuerySecurityObject(Handle, OWNER_SECURITY_INFORMATION |
195  DACL_SECURITY_INFORMATION, NULL, 0,
196  &sdLength);
197  if (status != STATUS_BUFFER_TOO_SMALL) {
198  // The file system does not support security.
199  return STATUS_SUCCESS;
200  }
201 
202  sd = RtlpSysVolAllocate(sdLength);
203  if (!sd) {
204  return STATUS_INSUFFICIENT_RESOURCES;
205  }
206 
207  status = NtQuerySecurityObject(Handle, OWNER_SECURITY_INFORMATION |
208  DACL_SECURITY_INFORMATION, sd, sdLength,
209  &sdLength);
210  if (!NT_SUCCESS(status)) {
211  RtlpSysVolFree(sd);
212  return status;
213  }
214 
215  status = RtlGetDaclSecurityDescriptor(sd, &daclPresent, &acl,
216  &daclDefaulted);
217  if (!NT_SUCCESS(status)) {
218  RtlpSysVolFree(sd);
219  return status;
220  }
221 
222  status = RtlGetOwnerSecurityDescriptor(sd, &sid, &ownerDefaulted);
223  if (!NT_SUCCESS(status)) {
224  RtlpSysVolFree(sd);
225  return status;
226  }
227 
228  //
229  // Setup well know SIDs
230  //
231 
232  systemSid = (PSID) sidBuffer;
233  adminSid = (PSID) sidBuffer2;
234 
235  RtlpSysVolSystemSid(systemSid);
236  RtlpSysVolAdminSid(adminSid);
237 
238 
239  if (!sid) {
240  goto ResetSecurity;
241  }
242 
243  if (!RtlEqualSid(sid, adminSid)) {
244  goto ResetSecurity;
245  }
246 
247  if (!daclPresent || (daclPresent && !acl)) {
248  goto ResetSecurity;
249  }
250 
251  for (i = 0; ; i++) {
252  status = RtlGetAce(acl, i, &ace);
253  if (!NT_SUCCESS(status)) {
254  ace = NULL;
255  }
256  if (!ace) {
257  break;
258  }
259 
260  if (ace->Header.AceType != ACCESS_ALLOWED_ACE_TYPE) {
261  continue;
262  }
263 
264  sid = (PSID) &ace->SidStart;
265  if (!RtlEqualSid(sid, systemSid)) {
266  continue;
267  }
268 
269  break;
270  }
271 
272  if (!ace) {
273  goto ResetSecurity;
274  }
275 
276  if (!(ace->Header.AceFlags&OBJECT_INHERIT_ACE) ||
277  !(ace->Header.AceFlags&CONTAINER_INHERIT_ACE)) {
278 
279  ace->Header.AceFlags |= OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE;
280 
281  status = NtSetSecurityObject(Handle, DACL_SECURITY_INFORMATION, sd);
282 
283  } else {
284  status = STATUS_SUCCESS;
285  }
286 
287  RtlpSysVolFree(sd);
288 
289  return status;
290 
291 ResetSecurity:
292 
293  sdLength2 = sdLength;
294  status = RtlSelfRelativeToAbsoluteSD2(sd, &sdLength2);
295  if (status == STATUS_BUFFER_TOO_SMALL) {
296  sd2 = RtlpSysVolAllocate(sdLength2);
297  if (!sd2) {
298  RtlpSysVolFree(sd);
299  return STATUS_INSUFFICIENT_RESOURCES;
300  }
301 
302  RtlCopyMemory(sd2, sd, sdLength);
303  RtlpSysVolFree(sd);
304  sd = sd2;
305  sdLength = sdLength2;
306 
307  status = RtlSelfRelativeToAbsoluteSD2(sd, &sdLength);
308  if (!NT_SUCCESS(status)) {
309  RtlpSysVolFree(sd);
310  return status;
311  }
312  }
313 
314  status = RtlSetOwnerSecurityDescriptor(sd, adminSid, FALSE);
315  if (!NT_SUCCESS(status)) {
316  RtlpSysVolFree(sd);
317  return status;
318  }
319 
320  status = RtlSetDaclSecurityDescriptor(sd, TRUE, StandardAcl, FALSE);
321  if (!NT_SUCCESS(status)) {
322  RtlpSysVolFree(sd);
323  return status;
324  }
325 
326  sdLength2 = 0;
327  status = RtlMakeSelfRelativeSD(sd, NULL, &sdLength2);
328  if (status != STATUS_BUFFER_TOO_SMALL) {
329  RtlpSysVolFree(sd);
330  return status;
331  }
332 
333  sd2 = RtlpSysVolAllocate(sdLength2);
334  if (!sd2) {
335  RtlpSysVolFree(sd);
336  return STATUS_INSUFFICIENT_RESOURCES;
337  }
338 
339  status = RtlMakeSelfRelativeSD(sd, sd2, &sdLength2);
340  RtlpSysVolFree(sd);
341  if (!NT_SUCCESS(status)) {
342  RtlpSysVolFree(sd2);
343  return status;
344  }
345 
346  sd = sd2;
347  sdLength = sdLength2;
348 
349  status = NtSetSecurityObject(Handle, OWNER_SECURITY_INFORMATION |
350  DACL_SECURITY_INFORMATION, sd);
351 
352  RtlpSysVolFree(sd);
353 
354  return status;
355 }

358 RtlpSysVolTakeOwnership(
359  IN PUNICODE_STRING DirectoryName
360  )
361 
362 /*++
363 
364 Routine Description:
365 
366  This routine is called when the open for the directory failed. This
367  routine will attempt to set the owner of the file to the caller's
368  ownership so that another attempt to open the file can be attempted.
369 
370 Arguments:
371 
372  DirectoryName - Supplies the directory name.
373 
374 Return Value:
375 
376  None.
377 
378 --*/
379 
380 {
381  NTSTATUS status;
382  HANDLE tokenHandle, fileHandle;
383  TOKEN_PRIVILEGES tokenPrivileges;
384  OBJECT_ATTRIBUTES oa;
385  IO_STATUS_BLOCK ioStatus;
386  SECURITY_DESCRIPTOR sd;
387  PSID adminSid;
388  UCHAR sidBuffer[2*sizeof(SID)];
389 
390  status = NtOpenProcessToken(NtCurrentProcess(), TOKEN_ADJUST_PRIVILEGES |
391  TOKEN_QUERY, &tokenHandle);
392  if (!NT_SUCCESS(status)) {
393  return;
394  }
395 
396  tokenPrivileges.PrivilegeCount = 1;
397  tokenPrivileges.Privileges[0].Luid =
398  RtlConvertLongToLuid(SE_TAKE_OWNERSHIP_PRIVILEGE);
399  tokenPrivileges.Privileges[0].Attributes = SE_PRIVILEGE_ENABLED;
400 
401  status = NtAdjustPrivilegesToken(tokenHandle, FALSE, &tokenPrivileges,
402  sizeof(tokenPrivileges), NULL, NULL);
403  if (!NT_SUCCESS(status)) {
404  NtClose(tokenHandle);
405  return;
406  }
407 
408  InitializeObjectAttributes(&oa, DirectoryName, OBJ_CASE_INSENSITIVE, NULL,
409  NULL);
410  status = NtOpenFile(&fileHandle, WRITE_OWNER | SYNCHRONIZE, &oa, &ioStatus,
411  FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
412  FILE_SYNCHRONOUS_IO_NONALERT | FILE_DIRECTORY_FILE);
413  if (!NT_SUCCESS(status)) {
414  NtClose(tokenHandle);
415  return;
416  }
417 
418  RtlCreateSecurityDescriptor(&sd, SECURITY_DESCRIPTOR_REVISION);
419  adminSid = (PSID) sidBuffer;
420  RtlpSysVolAdminSid(adminSid);
421 
422  status = RtlSetOwnerSecurityDescriptor(&sd, adminSid, FALSE);
423  if (!NT_SUCCESS(status)) {
424  NtClose(fileHandle);
425  NtClose(tokenHandle);
426  return;
427  }
428 
429  status = NtSetSecurityObject(fileHandle, OWNER_SECURITY_INFORMATION, &sd);
430  if (!NT_SUCCESS(status)) {
431  NtClose(fileHandle);
432  NtClose(tokenHandle);
433  return;
434  }
435 
436  NtClose(fileHandle);
437  NtClose(tokenHandle);
438 }

440 NTSTATUS
441 RtlCreateSystemVolumeInformationFolder(
442  IN PUNICODE_STRING VolumeRootPath
443  )
444 
445 /*++
446 
447 Routine Description:
448 
449  This routine verifies the existence of the "System Volume Information"
450  folder on the given volume. If the folder is not present, then the
451  folder is created with one ACE indicating full access for SYSTEM. The ACE
452  will have the inheritance bits set. The folder will be created with
453  the HIDDEN and SYSTEM attributes set.
454 
455  If the folder is already present, the ACE that indicates full control
456  for SYSTEM will be checked and if necessary modified to have the
457  inheritance bits set.
458 
459 Arguments:
460 
461  VolumeRootPath - Supplies a path to the root of an NTFS volume.
462 
463 Return Value:
464 
465  NTSTATUS
466 
467 --*/
468 
469 {
470  UNICODE_STRING sysVolName;
471  UNICODE_STRING dirName;
472  BOOLEAN needBackslash;
473  NTSTATUS status;
474  PSECURITY_DESCRIPTOR securityDescriptor;
475  PACL acl;
476  OBJECT_ATTRIBUTES oa;
477  HANDLE h;
478  IO_STATUS_BLOCK ioStatus;
479 
480  RtlInitUnicodeString(&sysVolName, RTL_SYSTEM_VOLUME_INFORMATION_FOLDER);
481 
482  dirName.Length = VolumeRootPath->Length + sysVolName.Length;
483 
484  //
485  // Check for wrapping.
486  //
487 
488  if ( dirName.Length < VolumeRootPath->Length
489  || dirName.Length < sysVolName.Length ) {
490  return STATUS_INVALID_PARAMETER;
491  }
492 
493  if (VolumeRootPath->Buffer[VolumeRootPath->Length/sizeof(WCHAR) - 1] !=
494  '\\') {
495 
496  dirName.Length += sizeof(WCHAR);
497  needBackslash = TRUE;
498  } else {
499  needBackslash = FALSE;
500  }
501  dirName.MaximumLength = dirName.Length + sizeof(WCHAR);
502  dirName.Buffer = RtlpSysVolAllocate(dirName.MaximumLength);
503  if (!dirName.Buffer) {
504  return STATUS_INSUFFICIENT_RESOURCES;
505  }
506 
507  RtlCopyMemory(dirName.Buffer, VolumeRootPath->Buffer,
508  VolumeRootPath->Length);
509  dirName.Length = VolumeRootPath->Length;
510  if (needBackslash) {
511  dirName.Buffer[VolumeRootPath->Length/sizeof(WCHAR)] = '\\';
512  dirName.Length += sizeof(WCHAR);
513  }
514  RtlCopyMemory((PCHAR) dirName.Buffer + dirName.Length,
515  sysVolName.Buffer, sysVolName.Length);
516  dirName.Length += sysVolName.Length;
517  dirName.Buffer[dirName.Length/sizeof(WCHAR)] = 0;
518 
519  status = RtlpSysVolCreateSecurityDescriptor(&securityDescriptor, &acl);
520  if (!NT_SUCCESS(status)) {
521  RtlpSysVolFree(dirName.Buffer);
522  return status;
523  }
524 
525  InitializeObjectAttributes(&oa, &dirName, OBJ_CASE_INSENSITIVE, NULL,
526  securityDescriptor);
527 
528  status = NtCreateFile(&h, READ_CONTROL | WRITE_DAC | WRITE_OWNER |
529  SYNCHRONIZE, &oa, &ioStatus, NULL,
530  FILE_ATTRIBUTE_HIDDEN | FILE_ATTRIBUTE_SYSTEM,
531  FILE_SHARE_READ | FILE_SHARE_WRITE |
532  FILE_SHARE_DELETE, FILE_OPEN_IF,
533  FILE_SYNCHRONOUS_IO_NONALERT | FILE_DIRECTORY_FILE,
534  NULL, 0);
535  if (!NT_SUCCESS(status)) {
536  RtlpSysVolTakeOwnership(&dirName);
537  status = NtCreateFile(&h, READ_CONTROL | WRITE_DAC | WRITE_OWNER |
538  SYNCHRONIZE, &oa, &ioStatus, NULL,
539  FILE_ATTRIBUTE_HIDDEN | FILE_ATTRIBUTE_SYSTEM,
540  FILE_SHARE_READ | FILE_SHARE_WRITE |
541  FILE_SHARE_DELETE, FILE_OPEN_IF,
542  FILE_SYNCHRONOUS_IO_NONALERT |
543  FILE_DIRECTORY_FILE, NULL, 0);
544  }
545 
546  RtlpSysVolFree(dirName.Buffer);
547 
548  if (!NT_SUCCESS(status)) {
549  RtlpSysVolFree(acl);
550  RtlpSysVolFree(securityDescriptor);
551  return status;
552  }
553 
554  RtlpSysVolFree(securityDescriptor);
555 
556  status = RtlpSysVolCheckOwnerAndSecurity(h, acl);
557 
558  NtClose(h);
559  RtlpSysVolFree(acl);
560 
561  return status;
562 }
