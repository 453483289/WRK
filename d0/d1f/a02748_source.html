107 WmipIsISFlagsSet(
108  PBGUIDENTRY GuidEntry,
109  ULONG Flags
110  )
111 /*++
112 
113 Routine Description:
114 
115  This routine determines whether any of the instance sets associated
116  with the GuidEntry has ALL of the flags set
117 
118 Arguments:
119 
120  GuidEntry Pointer to the Guid Entry structure.
121 
122  Flags has flags required
123 
124 Return Value:
125 
126 
127 --*/
128 {
129  PLIST_ENTRY InstanceSetList;
130  PBINSTANCESET InstanceSet;
131 
132  PAGED_CODE();
133 
134  if (GuidEntry != NULL)
135  {
136  WmipEnterSMCritSection();
137  InstanceSetList = GuidEntry->ISHead.Flink;
138  while (InstanceSetList != &GuidEntry->ISHead)
139  {
140  InstanceSet = CONTAINING_RECORD(InstanceSetList,
141  INSTANCESET,
142  GuidISList);
143  if ( (InstanceSet->Flags & Flags) == Flags )
144  {
145  WmipLeaveSMCritSection();
146  return (TRUE);
147  }
148  InstanceSetList = InstanceSetList->Flink;
149  }
150  WmipLeaveSMCritSection();
151  }
152  return (FALSE);
153 }

155 NTSTATUS WmipDeliverWnodeToDS(
156  CHAR ActionCode,
157  PBDATASOURCE DataSource,
158  PWNODE_HEADER Wnode,
159  ULONG BufferSize
160  )
161 {
162  NTSTATUS Status;
163  IO_STATUS_BLOCK Iosb;
164  PWMIGUIDOBJECT GuidObject;
165 
166  PAGED_CODE();
167 
168  if (DataSource->Flags & DS_KERNEL_MODE)
169  {
170  //
171  // If KM provider then send an irp
172  //
173  Status = WmipSendWmiIrp(ActionCode,
174  DataSource->ProviderId,
175  &Wnode->Guid,
176  BufferSize,
177  Wnode,
178  &Iosb);
179  } else if (DataSource->Flags & DS_USER_MODE) {
180  //
181  // If UM provider then send a MB message
182  //
183  GuidObject = DataSource->RequestObject;
184  if (GuidObject != NULL)
185  {
186  Wnode->Flags |= WNODE_FLAG_INTERNAL;
187  Wnode->ProviderId = ActionCode;
188  Wnode->CountLost = GuidObject->Cookie;
189  WmipEnterSMCritSection();
190  Status = WmipWriteWnodeToObject(GuidObject,
191  Wnode,
192  TRUE);
193  WmipLeaveSMCritSection();
194  } else {
195  Status = STATUS_SUCCESS;
196  }
197  } else {
198  ASSERT(FALSE);
199  Status = STATUS_UNSUCCESSFUL;
200  }
201 
202  return(Status);
203 }

205 NTSTATUS WmipSendEnableDisableRequest(
206  UCHAR ActionCode,
207  PBGUIDENTRY GuidEntry,
208  BOOLEAN IsEvent,
209  BOOLEAN IsTraceLog,
210  ULONG64 LoggerContext
211  )
212 /*++
213 
214 Routine Description:
215 
216  This routine will deliver an event or collection WNODE to all data
217  providers of a guid. This routine assumes that it is called with the
218  SM critical section held. The routine does not hold the critical
219  section for the duration of the call.
220 
221 Arguments:
222 
223  ActionCode is WMI_ENABLE_EVENTS, WMI_DISABLE_EVENTS,
224  WMI_ENABLE_COLLECTION or WMI_DISABLE_COLLECTION
225 
226  GuidEntry is the guid entry for the guid that is being enabled/disable
227  or collected/stop collected
228 
229  IsEvent is TRUE then ActionCode is to enable or disable events.
230  If FALSE then ActionCode is to enable or disbale collecton
231 
232  IsTraceLog is TRUE then enable is only sent to those guids registered as
233  being a tracelog guid
234 
235  LoggerContext is a logger context handle that should be placed in the
236  HistoricalContext field of the WNODE_HEADER if IsTraceLog is TRUE.
237 
238 Return Value:
239 
240  ERROR_SUCCESS or an error code
241 
242 --*/
243 {
244 #if DBG
245 #define AVGISPERGUID 1
246 #else
247 #define AVGISPERGUID 64
248 #endif
249 
250  PLIST_ENTRY InstanceSetList;
251  PBINSTANCESET InstanceSet;
252  PBDATASOURCE DataSourceArray[AVGISPERGUID];
253  PBDATASOURCE *DataSourceList;
254  ULONG BufferSize;
255  ULONG Status = 0;
256  PWNODE_HEADER pWnode;
257  ULONG i;
258  PBDATASOURCE DataSource;
259  ULONG DSCount;
260  BOOLEAN IsEnable;
261  ULONG IsFlags, IsUpdate;
262 
263  WMITRACE_NOTIFY_HEADER TraceNotifyHeader;
264 
265  PAGED_CODE();
266 
267  if (GuidEntry->Flags & GE_FLAG_INTERNAL)
268  {
269  //
270  // Guids that have been unregistered and Internally defined guids
271  // have no data source to send requests to, so just leave happily
272  return(STATUS_SUCCESS);
273  }
274 
275 
276  IsEnable = ((ActionCode == IRP_MN_ENABLE_EVENTS) ||
277  (ActionCode == IRP_MN_ENABLE_COLLECTION));
278  IsFlags = IsEvent ? IS_ENABLE_EVENT : IS_ENABLE_COLLECTION;
279 
280  //
281  // Determine whether this is an update call and reset the bit
282  //
283  IsUpdate = (GuidEntry->Flags & GE_NOTIFICATION_TRACE_UPDATE);
284 
285 
286  //
287  // First we make a list of all of the DataSources that need to be called
288  // while we have the critical section and take a reference on them so
289  // they don't go away after we release them. Note that the DataSource
290  // structure will stay, but the actual data provider may in fact go away.
291  // In this case sending the request will fail.
292  DSCount = 0;
293 
294  if (GuidEntry->ISCount > AVGISPERGUID)
295  {
296  DataSourceList = WmipAlloc(GuidEntry->ISCount * sizeof(PBDATASOURCE));
297  if (DataSourceList == NULL)
298  {
299  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: alloc failed for DataSource array in WmipSendEnableDisableRequest\n"));
300 
301  return(STATUS_INSUFFICIENT_RESOURCES);
302  }
303  } else {
304  DataSourceList = &DataSourceArray[0];
305  }
306 #if DBG
307  memset(DataSourceList, 0, GuidEntry->ISCount * sizeof(PBDATASOURCE));
308 #endif
309 
310  InstanceSetList = GuidEntry->ISHead.Flink;
311  while ((InstanceSetList != &GuidEntry->ISHead) &&
312  (DSCount < GuidEntry->ISCount))
313  {
314  WmipAssert(DSCount < GuidEntry->ISCount);
315  InstanceSet = CONTAINING_RECORD(InstanceSetList,
316  INSTANCESET,
317  GuidISList);
318 
319 
320  //
321  // We send requests to those data providers that are not inprocs when
322  // it is an event being enabled or it is collection being enabled
323  // and they are defined to be expensive (collection needs to be
324  // enabled)
325  if (
326  ( (IsTraceLog && (InstanceSet->Flags & IS_TRACED)) ||
327  ( ! IsTraceLog && (! (InstanceSet->Flags & IS_TRACED)) &&
328  (IsEvent || (InstanceSet->Flags & IS_EXPENSIVE))
329  )
330  )
331  )
332  {
333 
334  if ( (! IsEnable && (InstanceSet->Flags & IsFlags)) ||
335  ((IsEnable && ! (InstanceSet->Flags & IsFlags)) ||
336  (IsUpdate && IsTraceLog))
337  )
338  {
339  DataSourceList[DSCount] = InstanceSet->DataSource;
340  WmipReferenceDS(DataSourceList[DSCount]);
341  DSCount++;
342  }
343 
344  if (IsEnable)
345  {
346  InstanceSet->Flags |= IsFlags;
347  } else {
348  InstanceSet->Flags &= ~IsFlags;
349  }
350  }
351 
352  InstanceSetList = InstanceSetList->Flink;
353  }
354 
355 
356  if (IsUpdate)
357  {
358  GuidEntry->Flags &= ~GE_NOTIFICATION_TRACE_UPDATE;
359  }
360 
361 
362  WmipLeaveSMCritSection();
363 
364  //
365  // Now without the critical section we send the request to all of the
366  // data providers. Any new data providers who register after we made our
367  // list will be enabled by the registration code.
368  if (DSCount > 0)
369  {
370  pWnode = &TraceNotifyHeader.Wnode;
371  RtlZeroMemory(pWnode, sizeof(TraceNotifyHeader));
372  RtlCopyMemory(&pWnode->Guid, &GuidEntry->Guid, sizeof(GUID));
373  BufferSize = sizeof(WNODE_HEADER);
374 
375  if (IsTraceLog)
376  {
377  BufferSize = sizeof(TraceNotifyHeader);
378  TraceNotifyHeader.LoggerContext = LoggerContext;
379  pWnode->Flags |= WNODE_FLAG_TRACED_GUID;
380  //
381  // If this GUID is already enabled then this must
382  // an update call. So mark it so.
383  //
384  if ( IsEnable && IsUpdate ) {
385  pWnode->ClientContext = IsUpdate;
386  }
387 
388  }
389  pWnode->BufferSize = BufferSize;
390 
391  for (i = 0; i < DSCount; i++)
392  {
393  DataSource = DataSourceList[i];
394  WmipAssert(DataSource != NULL);
395  if (IsTraceLog) {
396  if (DataSource->Flags & DS_KERNEL_MODE) {
397  pWnode->HistoricalContext = LoggerContext;
398  }
399  else if (DataSource->Flags & DS_USER_MODE) {
400  pWnode->HistoricalContext = 0;
401  }
402  else {
403  ASSERT(FALSE);
404  }
405  }
406 
407  Status |= WmipDeliverWnodeToDS(ActionCode,
408  DataSource,
409  pWnode,
410  BufferSize);
411 
412 
413  WmipUnreferenceDS(DataSource);
414  }
415  }
416 
417  if( ! IsTraceLog )
418  {
419 
420  Status = STATUS_SUCCESS;
421  }
422 
423  if (DataSourceList != DataSourceArray)
424  {
425  WmipFree(DataSourceList);
426  }
427 
428  WmipEnterSMCritSection();
429 
430  return(Status);
431 }

456 void WmipWaitForCollectionEnabled(
457  PBGUIDENTRY GuidEntry
458  )
459 {
460  PAGED_CODE();
461 
462  WmipAssert((GuidEntry->Flags & GE_FLAG_COLLECTION_IN_PROGRESS) ==
463  GE_FLAG_COLLECTION_IN_PROGRESS);
464 
465  //
466  // Collection Enable/Disable is in progress so
467  // we cannot return just yet. Right now there could be a
468  // disable request being processed and if we didn't wait, we
469  // might get back to this caller before that disable request
470  // got around to realizing that it needs to send and enable
471  // request (needed by this thread's caller). So we'd have a
472  // situation where a thread though that collection was enabled
473  // but in reality it wasn't yet enabled.
474  if ((GuidEntry->Flags & GE_FLAG_WAIT_ENABLED) == 0)
475  {
476  KeInitializeEvent(GuidEntry->CollectInProgress,
477  NotificationEvent,
478  FALSE);
479  GuidEntry->Flags |= GE_FLAG_WAIT_ENABLED;
480  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p for %p %x created event\n",
481  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
482  GuidEntry,
483  GuidEntry->Flags));
484  }
485 
486  WmipLeaveSMCritSection();
487  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p waiting for %p %x on event\n",
488  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
489  GuidEntry,
490  GuidEntry->Flags));
491  KeWaitForSingleObject(GuidEntry->CollectInProgress,
492  Executive,
493  KernelMode,
494  FALSE,
495  NULL);
496  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p done %p %x waiting on event\n",
497  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
498  GuidEntry,
499  GuidEntry->Flags));
500  WmipEnterSMCritSection();
501 
502 }

504 ULONG WmipSendEnableRequest(
505  PBGUIDENTRY GuidEntry,
506  BOOLEAN IsEvent,
507  BOOLEAN IsTraceLog,
508  ULONG64 LoggerContext
509  )
510 /*++
511 Routine Description:
512 
513  This routine will send an enable collection or notification request to
514  all of the data providers that have registered the guid being enabled.
515  This routine will manage any race conditions that might occur when
516  multiple threads are enabling and disabling the notification
517  simultaneously.
518 
519  This routine is called while the SM critical section is being held and
520  will increment the appropriate reference count. if the ref count
521  transitions from 0 to 1 then the enable request will need to be forwarded
522  to the data providers otherwise the routine is all done and returns.
523  Before sending the enable request the routine checks to see if any
524  enable or disable requests are currently in progress and if not then sets
525  the in progress flag, releases the critical section and sends the enable
526  request. If there was a request in progress then the routine does not
527  send a request, but just returns. When the other thread that was sending
528  the request returns from processing the request it will recheck the
529  refcount and notice that it is greater than 0 and then send the enable
530  request.
531 
532 
533 Arguments:
534 
535  GuidEntry is the guid entry that describes the guid being enabled. For
536  a notification it may be NULL.
537 
538  NotificationContext is the notification context to use if enabling events
539 
540  IsEvent is TRUE if notifications are being enables else FALSE if
541  collection is being enabled
542 
543  IsTraceLog is TRUE if enable is for a trace log guid
544 
545  LoggerContext is a context value to forward in the enable request
546 
547 Return Value:
548 
549  ERROR_SUCCESS or an error code
550 --*/
551 {
552  ULONG InProgressFlag;
553  ULONG RefCount;
554  ULONG Status;
555 
556  PAGED_CODE();
557 
558  if (IsEvent)
559  {
560  InProgressFlag = GE_FLAG_NOTIFICATION_IN_PROGRESS;
561  RefCount = GuidEntry->EventRefCount++;
562  } else {
563  InProgressFlag = GE_FLAG_COLLECTION_IN_PROGRESS;
564  RefCount = GuidEntry->CollectRefCount++;
565  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p enable collect for %p %x\n",
566  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
567  GuidEntry, GuidEntry->Flags ));
568  }
569 
570  //
571  // If the guid is transitioning from a refcount of 0 to 1 and there
572  // is not currently a request in progress, then we need to set the
573  // request in progress flag, release the critical section and
574  // send an enable request. If there is a request in progress we can't
575  // do another request. Whenever the current request finishes it
576  // will notice the ref count change and send the enable request on
577  // our behalf.
578  if ((RefCount == 0) &&
579  ! (GuidEntry->Flags & InProgressFlag))
580  {
581  //
582  // Take an extra ref count so that even if this gets disabled
583  // while the enable request is in progress the GuidEntry
584  // will stay valid.
585  WmipReferenceGE(GuidEntry);
586  GuidEntry->Flags |= InProgressFlag;
587  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p NE %p flags -> %x at %d\n",
588  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
589  GuidEntry,
590  GuidEntry->Flags,
591  __LINE__));
592 
593 EnableNotification:
594  Status = WmipSendEnableDisableRequest((UCHAR)(IsEvent ?
595  IRP_MN_ENABLE_EVENTS :
596  IRP_MN_ENABLE_COLLECTION),
597  GuidEntry,
598  IsEvent,
599  IsTraceLog,
600  LoggerContext);
601 
602  RefCount = IsEvent ? GuidEntry->EventRefCount :
603  GuidEntry->CollectRefCount;
604 
605  if (RefCount == 0)
606  {
607  // This is the bogus situation we were worried about. While
608  // the enable request was being processed the notification
609  // was disabled. So leave the in progress flag set and
610  // send the disable.
611 
612  Status = WmipSendEnableDisableRequest((UCHAR)(IsEvent ?
613  IRP_MN_DISABLE_EVENTS :
614  IRP_MN_DISABLE_COLLECTION),
615  GuidEntry,
616  IsEvent,
617  IsTraceLog,
618  LoggerContext);
619 
620  RefCount = IsEvent ? GuidEntry->EventRefCount :
621  GuidEntry->CollectRefCount;
622 
623  if (RefCount > 0)
624  {
625  //
626  // We have hit a pathological case. One thread called to
627  // enable and while the enable request was being processed
628  // another thread called to disable, but was postponed
629  // since the enable was in progress. So once the enable
630  // completed we realized that the ref count reached 0 and
631  // so we need to disable and sent the disable request.
632  // But while the disable request was being processed
633  // an enable request came in so now we need to enable
634  // the notification.
635  goto EnableNotification;
636  }
637  }
638  GuidEntry->Flags &= ~InProgressFlag;
639  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p NE %p flags -> %x at %d\n",
640  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
641  GuidEntry,
642  GuidEntry->Flags,
643  __LINE__));
644 
645  //
646  // If there are any other threads that were waiting until all of
647  // the enable/disable work completed, we close the event handle
648  // to release them from their wait.
649  //
650  if (! IsEvent)
651  {
652  WmipReleaseCollectionEnabled(GuidEntry);
653  }
654 
655  //
656  // Get rid of extra ref count we took above. Note that the
657  // GuidEntry could be going away here if there was a
658  // disable while the enable was in progress.
659  WmipUnreferenceGE(GuidEntry);
660 
661  } else if (IsTraceLog && (GuidEntry->Flags & GE_NOTIFICATION_TRACE_UPDATE) ) {
662  //
663  // If it's a tracelog and we have a trace Update enable call, ignore the
664  // refcount and send it through.
665  //
666 
667  WmipReferenceGE(GuidEntry);
668 
669  Status = WmipSendEnableDisableRequest((UCHAR)(IsEvent ?
670  IRP_MN_ENABLE_EVENTS :
671  IRP_MN_ENABLE_COLLECTION),
672  GuidEntry,
673  IsEvent,
674  IsTraceLog,
675  LoggerContext);
676  GuidEntry->EventRefCount--;
677 
678  WmipUnreferenceGE(GuidEntry);
679 
680  } else {
681  if ((! IsEvent) && (GuidEntry->Flags & InProgressFlag))
682  {
683  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p going to wait for %p %x at %d\n",
684  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
685  GuidEntry,
686  GuidEntry->Flags,
687  __LINE__));
688  WmipWaitForCollectionEnabled(GuidEntry);
689  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p done to wait for %p %x at %d\n",
690  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
691  GuidEntry,
692  GuidEntry->Flags,
693  __LINE__));
694 
695  }
696 
697  Status = STATUS_SUCCESS;
698  }
699 
700  if (! IsEvent)
701  {
702  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p enable collect done for %p %x\n",
703  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
704  GuidEntry,
705  GuidEntry->Flags));
706  }
707 
708  return(Status);
709 }

711 ULONG WmipDoDisableRequest(
712  PBGUIDENTRY GuidEntry,
713  BOOLEAN IsEvent,
714  BOOLEAN IsTraceLog,
715  ULONG64 LoggerContext,
716  ULONG InProgressFlag
717  )
718 {
719  ULONG RefCount;
720  ULONG Status;
721 
722  PAGED_CODE();
723 
724 DisableNotification:
725  Status = WmipSendEnableDisableRequest((UCHAR)(IsEvent ?
726  IRP_MN_DISABLE_EVENTS :
727  IRP_MN_DISABLE_COLLECTION),
728  GuidEntry,
729  IsEvent,
730  IsTraceLog,
731  LoggerContext);
732 
733  RefCount = IsEvent ? GuidEntry->EventRefCount :
734  GuidEntry->CollectRefCount;
735 
736  if (RefCount > 0)
737  {
738  //
739  // While we were processing the disable request an
740  // enable request arrived. Since the in progress
741  // flag was set the enable request was not sent
742  // so now we need to do that.
743 
744  Status = WmipSendEnableDisableRequest((UCHAR)(IsEvent ?
745  IRP_MN_ENABLE_EVENTS :
746  IRP_MN_ENABLE_COLLECTION),
747  GuidEntry,
748  IsEvent,
749  IsTraceLog,
750  LoggerContext);
751 
752  RefCount = IsEvent ? GuidEntry->EventRefCount:
753  GuidEntry->CollectRefCount;
754 
755  if (RefCount == 0)
756  {
757  //
758  // While processing the enable request above the
759  // notification was disabled and since a request
760  // was in progress the disable request was not
761  // forwarded. Now it is time to forward the
762  // request.
763  goto DisableNotification;
764  }
765  }
766  GuidEntry->Flags &= ~InProgressFlag;

790  ULONG64 LoggerContext
791  )

794 
795  This routine will send an disable collection or notification request to
796  all of the data providers that have registered the guid being disabled.
797  This routine will manage any race conditions that might occur when
798  multiple threads are enabling and disabling the notification
799  simultaneously.
800

802  will increment the appropriate reference count. if the ref count
803  transitions from 1 to 0 then the disable request will need to be forwarded
804  to the data providers otherwise the routine is all done and returns.
805  Before sending the disable request the routine checks to see if any
806  enable or disable requests are currently in progress and if not then sets
807  the in progress flag, releases the critical section and sends the disable
808  request. If there was a request in progress then the routine does not
809  send a request, but just returns. When the other thread that was sending
810  the request returns from processing the request it will recheck the
811  refcount and notice that it is 0 and then send the disable
812  request.
813 
814 
815 Arguments:
816 
817  GuidEntry is the Notification entry that describes the guid
818  being enabled.
819 
820  GuidEntry is the guid entry that describes the guid being enabled. For
821  a notification it may be NULL.
822 
823  NotificationContext is the notification context to use if enabling events
824

826  collection is being enabled
827 
828  IsTraceLog is TRUE if enable is for a trace log guid
829 
830  LoggerContext is a context value to forward in the enable request
831 
832 Return Value:
833 
834  ERROR_SUCCESS or an error code
835 --*/

837  ULONG InProgressFlag;
838  ULONG RefCount;
839  ULONG Status;
840 
841  PAGED_CODE();
842 
843  if (IsEvent)
844  {
845  InProgressFlag = GE_FLAG_NOTIFICATION_IN_PROGRESS;
846  RefCount = GuidEntry->EventRefCount;
847  if (RefCount == 0)
848  {
849  //
850  // A bad data consumer is disabling his event more
851  // than once. Just ignore it
852  return(STATUS_SUCCESS);
853  }
854 
855  RefCount = --GuidEntry->EventRefCount;
856  } else {
857  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p Disabling for %p %x\n",

859  GuidEntry,
860  GuidEntry->Flags));

862  RefCount = --GuidEntry->CollectRefCount;
863  WmipAssert(RefCount != 0xffffffff);
864  }
865 
866  //
867  // If we have transitioned to a refcount of zero and there is
868  // not a request in progress then forward the disable request.
869  if ((RefCount == 0) &&
870  ! (GuidEntry->Flags & InProgressFlag))
871  {
872 
873  //
874  // Take an extra ref count so that even if this gets
875  // disabled while the disable request is in progress the
876  // GuidEntry will stay valid.
877  GuidEntry->Flags |= InProgressFlag;
878  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p NE %p flags -> %x at %d\n",
879  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
880  GuidEntry,
881  GuidEntry->Flags,
882  __LINE__));
883

885  IsEvent,
886  IsTraceLog,
887  LoggerContext,
888  InProgressFlag);
889 
890  } else {
891  Status = STATUS_SUCCESS;
892  }
893 
894  if (! IsEvent)
895  {
896  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: %p.%p Disable complete for %p %x\n",
897  PsGetCurrentProcessId(), PsGetCurrentThreadId(),
898  GuidEntry,
899  GuidEntry->Flags));
