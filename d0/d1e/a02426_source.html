57 /*++
58 
59 Routine Description:
60 
61  This function invokes specified ROM BIOS code by executing
62  "INT BiosCommand." Before executing the BIOS code, this function
63  will setup VDM context, change stack pointer ...etc. If for some reason
64  the operation fails, a status code will be returned. Otherwise, this
65  function always returns success regardless of the result of the BIOS
66  call.
67 
68  N.B. This implementation relies on the fact that the direct
69  I/O access operations between apps are serialized by win user.
70 
71 Arguments:
72 
73  BiosCommand - Supplies which ROM BIOS function to invoke.
74 
75  BiosArguments - Supplies a pointer to the context which will be used
76  to invoke ROM BIOS.
77 
78 Return Value:
79 
80  NTSTATUS code to specify the failure.
81 
82 --*/
83 
84 {
85 
86  PVDM_TIB VdmTib;
87  PUCHAR BaseAddress = (PUCHAR)V86_CODE_ADDRESS;
88  PTEB UserInt10Teb = (PTEB)INT_10_TEB;
89  PKTSS Tss;
90  PKPROCESS Process;
91  PKTHREAD Thread;
92  USHORT OldIopmOffset, OldIoMapBase;
93  PVDM_PROCESS_OBJECTS VdmObjects;
94  ULONG ContextLength;
95  UCHAR ThreadDebugActiveMask;
96 
97  //
98  // Map in ROM BIOS area to perform the int 10 code
99  //
100 
101  try {
102 
103  RtlZeroMemory(UserInt10Teb, sizeof(TEB));
104 
105  //
106  // Write "Int BiosCommand; bop" to reserved user space (0x1000).
107  // Later control will transfer to the user space to execute
108  // these two instructions.
109  //
110 
111  *BaseAddress++ = INT_OPCODE;
112  *BaseAddress++ = (UCHAR)BiosCommand;
113  *(PULONG)BaseAddress = V86_BOP_OPCODE;
114 
115  //
116  // Set up Vdm(v86) context to execute the int BiosCommand
117  // instruction by copying user supplied context to VdmContext
118  // and updating the control registers to predefined values.
119  //
120 
121  //
122  // We want to use a constant number for the int10.
123  //
124  // Create a fake TEB so we can switch the thread to it while we
125  // do an int10
126  //
127 
128  UserInt10Teb->Vdm = (PVOID)VDM_TIB_ADDRESS;
129  VdmTib = (PVDM_TIB)VDM_TIB_ADDRESS;
130  RtlZeroMemory(VdmTib, sizeof(VDM_TIB));
131  VdmTib->Size = sizeof(VDM_TIB);
132 
133  *FIXED_NTVDMSTATE_LINEAR_PC_AT = 0;
134 
135  //
136  // extended registers are never going to matter to
137  // an Int10 call, so only copy the old part of the
138  // context record.
139  //
140  ContextLength = FIELD_OFFSET(CONTEXT, ExtendedRegisters);
141 
142  RtlCopyMemory(&(VdmTib->VdmContext), BiosArguments, ContextLength);
143  VdmTib->VdmContext.SegCs = (ULONG)BaseAddress >> 4;
144  VdmTib->VdmContext.SegSs = (ULONG)BaseAddress >> 4;
145  VdmTib->VdmContext.Eip = 0;
146  VdmTib->VdmContext.Esp = 2 * PAGE_SIZE - sizeof(ULONG);
147  VdmTib->VdmContext.EFlags |= EFLAGS_V86_MASK | EFLAGS_INTERRUPT_MASK;
148  VdmTib->VdmContext.ContextFlags = CONTEXT_FULL;
149 
150  } except (EXCEPTION_EXECUTE_HANDLER) {
151 
152  return GetExceptionCode();
153  }
154 
155  //
156  // The vdm kernel code finds the Tib by looking at a pointer cached in
157  // kernel memory, which was probed at Vdm creation time. Since the
158  // creation semantics for this vdm are peculiar, we do something similar
159  // here.
160  //
161 
162  //
163  // We never get here on a process that is a real vdm. If we do,
164  // bad things will happen (pool leak, failure to execute dos and
165  // windows apps).
166  //
167 
168  ASSERT(PsGetCurrentProcess()->VdmObjects == NULL);
169 
170  VdmObjects = ExAllocatePoolWithTag (NonPagedPool,
171  sizeof(VDM_PROCESS_OBJECTS),
172  ' eK'
173  );
174 
175  //
176  // Since we are doing this on behalf of CSR not a user process, we aren't
177  // charging quota.
178  //
179 
180  if (VdmObjects == NULL) {
181  return STATUS_NO_MEMORY;
182  }
183 
184  //
185  // We are only initializing the VdmTib pointer, because that's the only
186  // part of the VdmObjects we use for ROM calls. We aren't set up
187  // to simulate interrupts, or any of the other stuff that would be done
188  // in a conventional vdm
189  //
190 
191  RtlZeroMemory( VdmObjects, sizeof(VDM_PROCESS_OBJECTS));
192 
193  VdmObjects->VdmTib = VdmTib;
194 
195  PsGetCurrentProcess()->VdmObjects = VdmObjects;
196  PS_SET_BITS(&PsGetCurrentProcess()->Flags, PS_PROCESS_FLAGS_VDM_ALLOWED);
197 
198  //
199  // Since we are going to v86 mode and accessing some I/O ports, we
200  // need to make sure the IopmOffset is set correctly across context
201  // swap and the I/O bit map has all the bits cleared.
202  // N.B. This implementation assumes that there is only one full
203  // screen DOS app and the io access between full screen DOS
204  // app and the server code is serialized by win user. That
205  // means even we change the IOPM, the full screen dos app won't
206  // be able to run on this IOPM.
207  // * In another words, IF THERE IS
208  // * MORE THAN ONE FULL SCREEN DOS APPS, THIS CODE IS BROKEN.*
209  //
210  // NOTE This code works on the assumption that winuser serializes
211  // direct I/O access operations.
212  //
213 
214  //
215  // Call the bios from the processor which booted the machine.
216  //
217 
218  Thread = KeGetCurrentThread();
219  KeSetSystemAffinityThread(1);
220  Tss = KeGetPcr()->TSS;
221 
222  //
223  // Save away the original IOPM bit map and clear all the IOPM bits
224  // to allow v86 int 10 code to access ALL the io ports.
225  //
226 
227  //
228  // Make sure there are at least 2 IOPM maps.
229  //
230 
231  ASSERT(KeGetPcr()->GDT[KGDT_TSS / 8].LimitLow >= (0x2000 + IOPM_OFFSET - 1));
232  RtlCopyMemory (Ki386IopmSaveArea,
233  (PVOID)&Tss->IoMaps[0].IoMap,
234  PAGE_SIZE * 2
235  );
236  RtlZeroMemory ((PVOID)&Tss->IoMaps[0].IoMap, PAGE_SIZE * 2);
237 
238  Process = Thread->ApcState.Process;
239  OldIopmOffset = Process->IopmOffset;
240  OldIoMapBase = Tss->IoMapBase;
241  Process->IopmOffset = (USHORT)(IOPM_OFFSET); // Set Process IoPmOffset before
242  Tss->IoMapBase = (USHORT)(IOPM_OFFSET); // updating Tss IoMapBase
243 
244  //
245  // The context setup for the BIOS will not have valid debug registers
246  // in it, don't try to load them.
247  //
248 
249  ThreadDebugActiveMask = (UCHAR) Thread->Header.DebugActive;
250  Thread->Header.DebugActive = (BOOLEAN) 0;
251 
252  //
253  // Call ASM routine to switch stack to exit to v86 mode to
254  // run Int BiosCommand.
255  //
256 
257  Ki386SetupAndExitToV86Code(UserInt10Teb);
258 
259  //
260  // After we return from v86 mode, the control comes here.
261  //
262  // Restore Thread's DebugActive flag.
263  //
264 
265  Thread->Header.DebugActive = (BOOLEAN) ThreadDebugActiveMask;
266 
267  //
268  // Restore old IOPM
269  //
270 
271  RtlCopyMemory ((PVOID)&Tss->IoMaps[0].IoMap,
272  Ki386IopmSaveArea,
273  PAGE_SIZE * 2
274  );
275 
276  Process->IopmOffset = OldIopmOffset;
277  Tss->IoMapBase = OldIoMapBase;
278 
279  //
280  // Restore old affinity for current thread.
281  //
282 
283  KeRevertToUserAffinityThread();
284 
285  //
286  // Copy 16 bit vdm context back to caller.
287  //
288  // Extended register state is not going to matter,
289  // so copy only the old part of the context record.
290  //
291 
292  ContextLength = FIELD_OFFSET(CONTEXT, ExtendedRegisters);
293  RtlCopyMemory(BiosArguments, &(VdmTib->VdmContext), ContextLength);
294  BiosArguments->ContextFlags = CONTEXT_FULL;
295 
296  //
297  // Free the pool used for the VdmTib pointer
298  //
299 
300  PsGetCurrentProcess()->VdmObjects = NULL;
301  PS_CLEAR_BITS(&PsGetCurrentProcess()->Flags, PS_PROCESS_FLAGS_VDM_ALLOWED);
302 
303  ExFreePool(VdmObjects);
304 
305  return STATUS_SUCCESS;
306 }
