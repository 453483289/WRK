39 VOID
40 CmpDeleteKeyObject(
41  IN PVOID Object
42  )
43 /*++
44 
45 Routine Description:
46 
47  This routine interfaces to the NT Object Manager. It is invoked when
48  the last reference to a particular Key object (or Key Root object)
49  is destroyed.
50 
51  If the Key object going away holds the last reference to
52  the extension it is associated with, that extension is destroyed.
53 
54 Arguments:
55 
56  Object - supplies a pointer to a KeyRoot or Key, thus -> KEY_BODY.
57 
58 Return Value:
59 
60  NONE.
61 
62 --*/
63 {
64  PCM_KEY_CONTROL_BLOCK KeyControlBlock;
65  PCM_KEY_BODY KeyBody;
66  PCMHIVE CmHive = NULL;
67  BOOLEAN DoUnloadCheck = FALSE;
68 
69  CM_PAGED_CODE();
70 
71  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FLOW,"CmpDeleteKeyObject: Object = %p\n", Object));
72 
73  //
74  // HandleClose callback
75  //
76  if ( CmAreCallbacksRegistered() ) {
77  REG_KEY_HANDLE_CLOSE_INFORMATION KeyHandleCloseInfo;
78 
79  KeyHandleCloseInfo.Object = Object;
80 
81  CmpCallCallBacks(RegNtPreKeyHandleClose,&KeyHandleCloseInfo,TRUE,RegNtPostKeyHandleClose,Object);
82  }
83 
84  KeyBody = (PCM_KEY_BODY)Object;
85 
86  BEGIN_LOCK_CHECKPOINT;
87 
88  CmpLockRegistry();
89 
90  if (KeyBody->Type==KEY_BODY_TYPE) {
91  KeyControlBlock = KeyBody->KeyControlBlock;
92 
93  //
94  // the keybody should be initialized; when kcb is null, something went wrong
95  // between the creation and the dereferenciation of the object
96  //
97  if( KeyControlBlock != NULL ) {
98  //
99  // Clean up any outstanding notifies attached to the KeyBody
100  //
101  CmpFlushNotify(KeyBody,FALSE);
102 
103  //
104  // Remove our reference to the KeyControlBlock, clean it up, perform any
105  // pend-till-final-close operations.
106  //
107  // NOTE: Delete notification is seen at the parent of the deleted key,
108  // not the deleted key itself. If any notify was outstanding on
109  // this key, it was cleared away above us. Only parent/ancestor
110  // keys will see the report.
111  //
112  //
113  // The dereference will free the KeyControlBlock. If the key was deleted, it
114  // has already been removed from the hash table, and relevant notifications
115  // posted then as well. All we are doing is freeing the tombstone.
116  //
117  // If it was not deleted, we're both cutting the kcb out of
118  // the kcb list/tree, AND freeing its storage.
119  //
120 
121 
122  //
123  // Replace this with the definition so we avoid dropping and reacquiring the lock
124  DelistKeyBodyFromKCB(KeyBody,FALSE);
125 
126  //
127  // take additional precaution in the case the hive has been unloaded and this is the root
128  //
129  if( !KeyControlBlock->Delete ) {
130  CmHive = (PCMHIVE)CONTAINING_RECORD(KeyControlBlock->KeyHive, CMHIVE, Hive);
131  if( IsHiveFrozen(CmHive) ) {
132  //
133  // unload is pending for this hive;
134  //
135  DoUnloadCheck = TRUE;
136 
137  }
138  }
139 
140  CmpDelayDerefKeyControlBlock(KeyControlBlock);
141 
142  }
143  } else {
144  //
145  // This must be a predefined handle
146  // some sanity asserts
147  //
148  KeyControlBlock = KeyBody->KeyControlBlock;
149 
150  ASSERT( KeyBody->Type&REG_PREDEF_HANDLE_MASK);
151  ASSERT( KeyControlBlock->Flags&KEY_PREDEF_HANDLE );
152 
153  if( KeyControlBlock != NULL ) {
154  CmHive = (PCMHIVE)CONTAINING_RECORD(KeyControlBlock->KeyHive, CMHIVE, Hive);
155  if( IsHiveFrozen(CmHive) ) {
156  //
157  // unload is pending for this hive; we shouldn't put the kcb in the delay
158  // close table
159  //
160  DoUnloadCheck = TRUE;
161 
162  }
163  CmpDereferenceKeyControlBlock(KeyControlBlock);
164  }
165 
166  }
167 
168  //
169  // if a handle inside a frozen hive has been closed, we may need to unload the hive
170  //
171  if( DoUnloadCheck == TRUE ) {
172  CmpDoQueueLateUnloadWorker(CmHive);
173  }
174 
175  CmpUnlockRegistry();
176  END_LOCK_CHECKPOINT;
177 
178  //
179  // just a notification; disregard the return status
180  //
181  CmPostCallbackNotification(RegNtPostKeyHandleClose,NULL,STATUS_SUCCESS);
182  return;
183 }

185 VOID
186 CmpDoQueueLateUnloadWorker(IN PCMHIVE CmHive)
187 {
188  PWORK_QUEUE_ITEM WorkItem;
189 
190  CM_PAGED_CODE();
191 
192  ASSERT( CmHive->RootKcb != NULL );
193 
194  //
195  // NB: Hive lock has higher precedence; We don't need the kcb lock as we are only checking the refcount
196  //
197  CmLockHive(CmHive);
198 
199  if( (CmHive->RootKcb->RefCount == 1) && (CmHive->UnloadWorkItem == NULL) ) {
200  //
201  // the only reference on the rookcb is the one that we artificially created
202  // queue a work item to late unload the hive
203  //
204  WorkItem = ExAllocatePool(NonPagedPool, sizeof(WORK_QUEUE_ITEM));
205  if( InterlockedCompareExchangePointer(&(CmHive->UnloadWorkItem),WorkItem,NULL) == NULL ) {
206  ExInitializeWorkItem(CmHive->UnloadWorkItem,
207  CmpLateUnloadHiveWorker,
208  CmHive);
209  ExQueueWorkItem(CmHive->UnloadWorkItem, DelayedWorkQueue);
210  } else {
211  ExFreePool(WorkItem);
212  }
213 
214  }
215 
216  CmUnlockHive(CmHive);
217 }

219 VOID
220 CmpLateUnloadHiveWorker(
221  IN PVOID Hive
222  )
223 /*++
224 
225 Routine Description:
226 
227  "Late" unloads the hive; If nothing goes badly wrong (i.e. insufficient resources),
228  this function should succeed
229 
230 Arguments:
231 
232  CmHive - the frozen hive to be unloaded
233 
234 Return Value:
235 
236  NONE.
237 
238 --*/
239 {
240  NTSTATUS Status;
241  HCELL_INDEX Cell;
242  PCM_KEY_CONTROL_BLOCK RootKcb;
243  PCMHIVE CmHive;
244 
245  CM_PAGED_CODE();
246 
247  //
248  // first, load the registry exclusive
249  //
250  CmpLockRegistryExclusive();
251 
252  //
253  // hive is the parameter to this worker; make sure we free the work item
254  // allocated by CmpDeleteKeyObject
255  //
256  CmHive = (PCMHIVE)Hive;
257 
258  ASSERT( CmHive->UnloadWorkItem != NULL );
259  ExFreePool( CmHive->UnloadWorkItem );
260 
261  //
262  // if this attempt doesn't succeed, mark that we can try another
263  //
264  CmHive->UnloadWorkItem = NULL;
265 
266  ASSERT( !(CmHive->Hive.HiveFlags & HIVE_IS_UNLOADING) );
267  if( CmHive->Frozen == FALSE ) {
268  //
269  // another thread mounted the exact same hive in the exact same place, hence unfreezing the hive
270  // we've done the cleanup part (free the workitem) nothing more to do.
271  // or hive is already in process of being unloaded
272  //
273  ASSERT( CmHive->RootKcb == NULL );
274  CmpUnlockRegistry();
275  return;
276  }
277  //
278  // this is just about the only possible way the hive can get corrupted in between
279  //
280  if( HvShutdownComplete == TRUE ) {
281  // too late to do anything
282  CmpUnlockRegistry();
283  return;
284  }
285 
286  //
287  // hive should be frozen, otherwise we wouldn't get here
288  //
289  ASSERT( CmHive->Frozen == TRUE );
290 
291  RootKcb = CmHive->RootKcb;
292  //
293  // root kcb must be valid and has only our "artificial" refcount on it
294  //
295  ASSERT( RootKcb != NULL );
296 
297  if( RootKcb->RefCount > 1 ) {
298  //
299  // somebody else must've gotten in between dropping/reacquiring the reglock
300  // and opened a handle inside this hive; bad luck, we can't unload
301  //
302  CmpUnlockRegistry();
303  return;
304  }
305 
306  ASSERT_KCB(RootKcb);
307 
308  Cell = RootKcb->KeyCell;
309  Status = CmUnloadKey(RootKcb,0,CM_UNLOAD_REG_LOCKED_EX);
310  ASSERT( (Status != STATUS_CANNOT_DELETE) && (Status != STATUS_INVALID_PARAMETER) );
311 
312  if(NT_SUCCESS(Status)) {
313  // CmUnloadKey already released the lock
314  CmpLockRegistry();
315  CmpDereferenceKeyControlBlock(RootKcb);
316  }
317  CmpUnlockRegistry();
318 }
