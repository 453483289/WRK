37 NTSTATUS
38 RtlCreateEnvironment(
39  IN BOOLEAN CloneCurrentEnvironment OPTIONAL,
40  OUT PVOID *Environment
41  )
42 {
43  NTSTATUS Status;
44  MEMORY_BASIC_INFORMATION MemoryInformation;
45  PVOID pNew, pOld;
46  PPEB Peb;
47  PRTL_USER_PROCESS_PARAMETERS ProcessParameters;
48 
49  //
50  // If not cloning a copy of the current process's environment variable
51  // block, just allocate a block of committed memory and return its
52  // address.
53  //
54 
55  pNew = NULL;
56  if (!CloneCurrentEnvironment) {
57 createEmptyEnvironment:
58  MemoryInformation.RegionSize = 1;
59  Status = ZwAllocateVirtualMemory( NtCurrentProcess(),
60  &pNew,
61  0,
62  &MemoryInformation.RegionSize,
63  MEM_COMMIT,
64  PAGE_READWRITE
65  );
66  if (NT_SUCCESS( Status )) {
67  *Environment = pNew;
68  }
69 
70  return( Status );
71  }
72 
73  Peb = NtCurrentPeb ();
74 
75  ProcessParameters = Peb->ProcessParameters;
76 
77  //
78  // Acquire the Peb Lock for the duration while we munge the environment
79  // variable storage block.
80  //
81 
82  RtlAcquirePebLock();
83 
84  //
85  // Capture the pointer to the current process's environment variable
86  // block and initialize the new pointer to null for our finally clause.
87  //
88 
89  pOld = ProcessParameters->Environment;
90  if (pOld == NULL) {
91  RtlReleasePebLock();
92  goto createEmptyEnvironment;
93  }
94 
95  try {
96  try {
97  //
98  // Query the current size of the current process's environment
99  // variable block. Return status if failure.
100  //
101 
102  Status = ZwQueryVirtualMemory (NtCurrentProcess (),
103  pOld,
104  MemoryBasicInformation,
105  &MemoryInformation,
106  sizeof (MemoryInformation),
107  NULL);
108  if (!NT_SUCCESS (Status)) {
109  leave;
110  }
111 
112  //
113  // Allocate memory to contain a copy of the current process's
114  // environment variable block. Return status if failure.
115  //
116 
117  Status = ZwAllocateVirtualMemory (NtCurrentProcess (),
118  &pNew,
119  0,
120  &MemoryInformation.RegionSize,
121  MEM_COMMIT,
122  PAGE_READWRITE);
123  if (!NT_SUCCESS (Status)) {
124  leave;
125  }
126 
127  //
128  // Copy the current process's environment to the allocated memory
129  // and return a pointer to the copy.
130  //
131 
132  RtlCopyMemory (pNew, pOld, MemoryInformation.RegionSize);
133  *Environment = pNew;
134  } except (EXCEPTION_EXECUTE_HANDLER) {
135  Status = STATUS_ACCESS_VIOLATION;
136  }
137  } finally {
138  RtlReleasePebLock ();
139 
140  if (Status == STATUS_ACCESS_VIOLATION) {
141  if (pNew != NULL) {
142  ZwFreeVirtualMemory (NtCurrentProcess(),
143  &pNew,
144  &MemoryInformation.RegionSize,
145  MEM_RELEASE);
146  }
147  }
148 
149  }
150 
151  return (Status);
152 }

259 NTSTATUS
260 RtlQueryEnvironmentVariable_U(
261  IN PVOID Environment OPTIONAL,
262  IN PCUNICODE_STRING Name,
263  IN OUT PUNICODE_STRING Value
264  )
265 {
266  NTSTATUS Status;
267  UNICODE_STRING CurrentName;
268  UNICODE_STRING CurrentValue;
269  PWSTR p, q;
270  PPEB Peb;
271  BOOLEAN PebLockLocked = FALSE;
272  SIZE_T len;
273  SIZE_T NameLength, NameChars;
274  PRTL_USER_PROCESS_PARAMETERS ProcessParameters;
275 
276  Status = STATUS_VARIABLE_NOT_FOUND;
277 
278  Peb = NtCurrentPeb();
279 
280  ProcessParameters = Peb->ProcessParameters;
281 
282  try {
283  if (ARGUMENT_PRESENT (Environment)) {
284  p = Environment;
285  if (*p == UNICODE_NULL) {
286  leave;
287  }
288  } else {
289 
290 
291  //
292  // Acquire the Peb Lock for the duration while we munge the
293  // environment variable storage block.
294  //
295 
296  PebLockLocked = TRUE;
297 
298  RtlAcquirePebLock ();
299 
300  //
301  // Capture the pointer to the current process's environment variable
302  // block.
303  //
304 
305  p = ProcessParameters->Environment;
306 
307  }
308 
309  if (RtlpEnvironCacheValid && p == ProcessParameters->Environment) {
310  if (RtlEqualUnicodeString (Name, &RtlpEnvironCacheName, TRUE)) {
311 
312  //
313  // Names are equal. Always return the length of the
314  // value string, excluding the terminating null. If
315  // there is room in the caller's buffer, return a copy
316  // of the value string and success status. Otherwise
317  // return an error status. In the latter case, the caller
318  // can examine the length field of their value string
319  // so they can determine much memory is needed.
320  //
321 
322  Value->Length = RtlpEnvironCacheValue.Length;
323  if (Value->MaximumLength >= RtlpEnvironCacheValue.Length) {
324  RtlCopyMemory (Value->Buffer,
325  RtlpEnvironCacheValue.Buffer,
326  RtlpEnvironCacheValue.Length);
327  //
328  // Null terminate returned string if there is room.
329  //
330 
331  if (Value->MaximumLength > RtlpEnvironCacheValue.Length) {
332  Value->Buffer[RtlpEnvironCacheValue.Length/sizeof(WCHAR)] = L'\0';
333  }
334 
335  Status = STATUS_SUCCESS;
336  } else {
337  Status = STATUS_BUFFER_TOO_SMALL;
338  }
339  leave;
340  }
341  }
342 
343  //
344  // The environment variable block consists of zero or more null
345  // terminated UNICODE strings. Each string is of the form:
346  //
347  // name=value
348  //
349  // where the null termination is after the value.
350  //
351  NameLength = Name->Length;
352  NameChars = NameLength / sizeof (WCHAR);
353 
354  if (p != NULL) while (1) {
355 
356  //
357  // Get the length of the terminated string. This should be in the
358  // form 'keyword=value'
359  //
360  len = wcslen (p);
361  if (len == 0) {
362  break;
363  }
364 
365  //
366  // See if this environment variable is big enough to be our target.
367  // If must be at least one bigger than the name we are searching
368  // for since it must contain an '=' sign.
369  //
370 
371  if (NameChars < len) {
372  q = &p[NameChars];
373  //
374  // We have a possible match. See if there is an '=' at the correct point.
375  //
376  if (*q == L'=') {
377  //
378  // We have a possible match. Now compare the string out right
379  //
380  CurrentName.Length = (USHORT) NameLength;
381  CurrentName.Buffer = p;
382 
383  //
384  // After comparing the strings we want to make sure we are not
385  // matching something that we shouldn't. For example if somebody
386  // did a lookup of "FRED=BOB" we don't want this to match "FRED=BOB=ALBERT".
387  // The lookup name is invalid and the environment variable is really FRED here
388  // not "FRED=BOB". To eliminate this case we make sure that the "=" character
389  // is at the appropriate place
390  // There are environment variable of the for =C:
391  // In order not to eliminate these we skip the first character in our search.
392  //
393  if (RtlEqualUnicodeString (Name, &CurrentName, TRUE) &&
394  (wcschr (p+1, L'=') == q)) {
395  //
396  // Names are equal. Always return the length of the
397  // value string, excluding the terminating null. If
398  // there is room in the caller's buffer, return a copy
399  // of the value string and success status. Otherwise
400  // return an error status. In the latter case, the caller
401  // can examine the length field of their value string
402  // so they can determine much memory is needed.
403  //
404  CurrentValue.Buffer = q+1;
405  CurrentValue.Length = (USHORT) ((len - 1) * sizeof (WCHAR) - NameLength);
406 
407  Value->Length = CurrentValue.Length;
408  if (Value->MaximumLength >= CurrentValue.Length) {
409  RtlCopyMemory( Value->Buffer,
410  CurrentValue.Buffer,
411  CurrentValue.Length
412  );
413  //
414  // Null terminate returned string if there is room.
415  //
416 
417  if (Value->MaximumLength > CurrentValue.Length) {
418  Value->Buffer[ CurrentValue.Length/sizeof(WCHAR) ] = L'\0';
419  }
420 
421  if (Environment == ProcessParameters->Environment) {
422  RtlpEnvironCacheValid = TRUE;
423  RtlpEnvironCacheName = CurrentName;
424  RtlpEnvironCacheValue = CurrentValue;
425  }
426 
427  Status = STATUS_SUCCESS;
428  } else {
429  Status = STATUS_BUFFER_TOO_SMALL;
430  }
431 
432  break;
433  }
434 
435  }
436  }
437  p += len + 1;
438  }
439 
440  // If it's not in the real env block, let's see if it's a pseudo environment variable
441  if (Status == STATUS_VARIABLE_NOT_FOUND) {
442  static const UNICODE_STRING CurrentWorkingDirectoryPseudoVariable = RTL_CONSTANT_STRING(L"__CD__");
443  static const UNICODE_STRING ApplicationDirectoryPseudoVariable = RTL_CONSTANT_STRING(L"__APPDIR__");
444 
445  if (RtlEqualUnicodeString(Name, &CurrentWorkingDirectoryPseudoVariable, TRUE)) {
446  // Get the PEB lock if we don't already have it.
447  if (!PebLockLocked) {
448  PebLockLocked = TRUE;
449  RtlAcquirePebLock();
450  }
451 
452  // get cdw here...
453  CurrentValue = ProcessParameters->CurrentDirectory.DosPath;
454  Status = STATUS_SUCCESS;
455  } else if (RtlEqualUnicodeString(Name, &ApplicationDirectoryPseudoVariable, TRUE)) {
456  USHORT PrefixLength = 0;
457 
458  if (!PebLockLocked) {
459  PebLockLocked = TRUE;
460  RtlAcquirePebLock();
461  }
462 
463  // get appdir here
464  CurrentValue = ProcessParameters->ImagePathName;
465 
466  Status = RtlFindCharInUnicodeString(
467  RTL_FIND_CHAR_IN_UNICODE_STRING_START_AT_END,
468  &CurrentValue,
469  &RtlDosPathSeparatorsString,
470  &PrefixLength);
471  if (NT_SUCCESS(Status)) {
472  CurrentValue.Length = PrefixLength + sizeof(WCHAR);
473  } else if (Status == STATUS_NOT_FOUND) {
474  // Use the whole thing; just translate the status to success.
475  Status = STATUS_SUCCESS;
476  }
477  }
478 
479  if (NT_SUCCESS(Status)) {
480  Value->Length = CurrentValue.Length;
481  if (Value->MaximumLength >= CurrentValue.Length) {
482  RtlCopyMemory(Value->Buffer, CurrentValue.Buffer, CurrentValue.Length);
483 
484  //
485  // Null terminate returned string if there is room.
486  //
487 
488  if (Value->MaximumLength > CurrentValue.Length)
489  Value->Buffer[ CurrentValue.Length/sizeof(WCHAR) ] = L'\0';
490  }
491  }
492  }
493 
494 
495 
496  } except (EXCEPTION_EXECUTE_HANDLER) {
497  Status = STATUS_ACCESS_VIOLATION;
498  }
499 
500 
501  //
502  // Release the Peb lock.
503  //
504 
505  if (PebLockLocked) {
506  RtlReleasePebLock();
507  }
508 
509  //
510  // Return status.
511  //
512 
513  return Status;
514 }

517 NTSTATUS
518 RtlSetEnvironmentVariable(
519  IN OUT PVOID *Environment OPTIONAL,
520  IN PCUNICODE_STRING Name,
521  IN PCUNICODE_STRING Value OPTIONAL
522  )
523 {
524  NTSTATUS Status;
525  MEMORY_BASIC_INFORMATION MemoryInformation;
526  UNICODE_STRING CurrentName;
527  UNICODE_STRING CurrentValue;
528  PVOID pOld, pNew;
529  ULONG n, Size;
530  SIZE_T NewSize;
531  LONG CompareResult;
532  PWSTR p, pStart, pEnd;
533  PWSTR InsertionPoint;
534  PPEB Peb;
535  PRTL_USER_PROCESS_PARAMETERS ProcessParameters;
536 
537  //
538  // Validate passed in name and reject if zero length or anything but the first
539  // character is an equal sign.
540  //
541  n = Name->Length / sizeof( WCHAR );
542  if (n == 0) {
543  return STATUS_INVALID_PARAMETER;
544  }
545 
546  try {
547  p = Name->Buffer;
548  while (--n) {
549  if (*++p == L'=') {
550  return STATUS_INVALID_PARAMETER;
551  }
552  }
553  } except (EXCEPTION_EXECUTE_HANDLER) {
554  return GetExceptionCode();
555  }
556 
557  Peb = NtCurrentPeb ();
558 
559  ProcessParameters = Peb->ProcessParameters;
560 
561  Status = STATUS_VARIABLE_NOT_FOUND;
562 
563  pNew = NULL;
564  InsertionPoint = NULL;
565 
566  if (ARGUMENT_PRESENT (Environment)) {
567  pOld = *Environment;
568  } else {
569  //
570  // Acquire the Peb Lock for the duration while we munge the
571  // environment variable storage block.
572  //
573 
574  RtlAcquirePebLock();
575 
576  //
577  // Capture the pointer to the current process's environment variable
578  // block.
579  //
580 
581  pOld = ProcessParameters->Environment;
582  }
583 
584  RtlpEnvironCacheValid = FALSE;
585 
586  try {
587  try {
588 
589  //
590  // The environment variable block consists of zero or more null
591  // terminated UNICODE strings. Each string is of the form:
592  //
593  // name=value
594  //
595  // where the null termination is after the value.
596  //
597 
598  p = pOld;
599  pEnd = NULL;
600  if (p != NULL) while (*p) {
601  //
602  // Determine the size of the name and value portions of
603  // the current string of the environment variable block.
604  //
605 
606  CurrentName.Buffer = p;
607  CurrentName.Length = 0;
608  CurrentName.MaximumLength = 0;
609  while (*p) {
610  //
611  // If we see an equal sign, then compute the size of
612  // the name portion and scan for the end of the value.
613  //
614 
615  if (*p == L'=' && p != CurrentName.Buffer) {
616  CurrentName.Length = (USHORT)(p - CurrentName.Buffer) * sizeof(WCHAR);
617  CurrentName.MaximumLength = (USHORT)(CurrentName.Length+sizeof(WCHAR));
618  CurrentValue.Buffer = ++p;
619 
620  while(*p) {
621  p++;
622  }
623  CurrentValue.Length = (USHORT)(p - CurrentValue.Buffer) * sizeof(WCHAR);
624  CurrentValue.MaximumLength = (USHORT)(CurrentValue.Length+sizeof(WCHAR));
625 
626  //
627  // At this point we have the length of both the name
628  // and value portions, so exit the loop so we can
629  // do the compare.
630  //
631  break;
632  }
633  else {
634  p++;
635  }
636  }
637 
638  //
639  // Skip over the terminating null character for this name=value
640  // pair in preparation for the next iteration of the loop.
641  //
642 
643  p++;
644 
645  //
646  // Compare the current name with the one requested, ignore
647  // case.
648  //
649 
650  if (!(CompareResult = RtlCompareUnicodeString( Name, &CurrentName, TRUE ))) {
651  //
652  // Names are equal. Now find the end of the current
653  // environment variable block.
654  //
655 
656  pEnd = p;
657  while (*pEnd) {
658  while (*pEnd++) {
659  }
660  }
661  pEnd++;
662 
663  if (!ARGUMENT_PRESENT( Value )) {
664  //
665  // If the caller did not specify a new value, then delete
666  // the entire name=value pair by copying up the remainder
667  // of the environment variable block.
668  //
669 
670  RtlMoveMemory( CurrentName.Buffer,
671  p,
672  (ULONG) ((pEnd - p)*sizeof(WCHAR))
673  );
674  Status = STATUS_SUCCESS;
675 
676  } else if (Value->Length <= CurrentValue.Length) {
677  //
678  // New value is smaller, so copy new value, then null
679  // terminate it, and then move up the remainder of the
680  // variable block so it is immediately after the new
681  // null terminated value.
682  //
683 
684  pStart = CurrentValue.Buffer;
685  RtlMoveMemory( pStart, Value->Buffer, Value->Length );
686  pStart += Value->Length/sizeof(WCHAR);
687  *pStart++ = L'\0';
688 
689  RtlMoveMemory( pStart, p,(ULONG)((pEnd - p)*sizeof(WCHAR)) );
690  Status = STATUS_SUCCESS;
691  } else {
692  //
693  // New value is larger, so query the current size of the
694  // environment variable block. Return status if failure.
695  //
696 
697  Status = ZwQueryVirtualMemory( NtCurrentProcess(),
698  pOld,
699  MemoryBasicInformation,
700  &MemoryInformation,
701  sizeof( MemoryInformation ),
702  NULL
703  );
704  if (!NT_SUCCESS( Status )) {
705  leave;
706  }
707 
708  //
709  // See if there is room for new, larger value. If not
710  // allocate a new copy of the environment variable
711  // block.
712  //
713 
714  NewSize = (pEnd - (PWSTR)pOld)*sizeof(WCHAR) +
715  Value->Length - CurrentValue.Length;
716  if (NewSize >= MemoryInformation.RegionSize) {
717  //
718  // Allocate memory to contain a copy of the current
719  // process's environment variable block. Return
720  // status if failure.
721  //
722 
723  Status = ZwAllocateVirtualMemory( NtCurrentProcess(),
724  &pNew,
725  0,
726  &NewSize,
727  MEM_COMMIT,
728  PAGE_READWRITE
729  );
730  if (!NT_SUCCESS( Status )) {
731  leave;
732  }
733 
734  //
735  // Copy the current process's environment to the allocated memory
736  // inserting the new value as we do the copy.
737  //
738 
739  Size = (ULONG) (CurrentValue.Buffer - (PWSTR)pOld);
740  RtlMoveMemory( pNew, pOld, Size*sizeof(WCHAR) );
741  pStart = (PWSTR)pNew + Size;
742  RtlMoveMemory( pStart, Value->Buffer, Value->Length );
743  pStart += Value->Length/sizeof(WCHAR);
744  *pStart++ = L'\0';
745  RtlMoveMemory( pStart, p,(ULONG)((pEnd - p)*sizeof(WCHAR)));
746 
747  if (ARGUMENT_PRESENT( Environment )) {
748  *Environment = pNew;
749  } else {
750  ProcessParameters->Environment = pNew;
751 
752 #if defined(BUILD_WOW6432)
753  ((PRTL_USER_PROCESS_PARAMETERS64)(NtCurrentPeb64()->ProcessParameters))->Environment = (ULONGLONG)(ULONG)pNew;
754 #endif
755  Peb->EnvironmentUpdateCount += 1;
756  }
757 
758  ZwFreeVirtualMemory (NtCurrentProcess(),
759  &pOld,
760  &MemoryInformation.RegionSize,
761  MEM_RELEASE);
762  pNew = pOld;
763  } else {
764  pStart = CurrentValue.Buffer + Value->Length/sizeof(WCHAR) + 1;
765  RtlMoveMemory (pStart, p, (ULONG)((pEnd - p)*sizeof(WCHAR)));
766  *--pStart = L'\0';
767 
768  RtlMoveMemory (pStart - Value->Length/sizeof(WCHAR),
769  Value->Buffer,
770  Value->Length);
771  }
772  }
773 
774  break;
775  } else if (CompareResult < 0) {
776  //
777  // Requested name is less than the current name. Save this
778  // spot in case the variable is not in a sorted position.
779  // The insertion point for the new variable is before the
780  // variable just examined.
781  //
782 
783  if (InsertionPoint == NULL) {
784  InsertionPoint = CurrentName.Buffer;
785  }
786  }
787  }
788 
789  //
790  // If we found an insertion point, reset the string
791  // pointer back to it.
792  //
793 
794  if (InsertionPoint != NULL) {
795  p = InsertionPoint;
796  }
797 
798  //
799  // If variable name not found and a new value parameter was specified
800  // then insert the new variable name and its value at the appropriate
801  // place in the environment variable block (i.e. where p points to).
802  //
803 
804  if (pEnd == NULL && ARGUMENT_PRESENT( Value )) {
805  if (p != NULL) {
806  //
807  // Name not found. Now find the end of the current
808  // environment variable block.
809  //
810 
811  pEnd = p;
812  while (*pEnd) {
813  while (*pEnd++) {
814  }
815  }
816  pEnd++;
817 
818  //
819  // New value is present, so query the current size of the
820  // environment variable block. Return status if failure.
821  //
822 
823  Status = ZwQueryVirtualMemory( NtCurrentProcess(),
824  pOld,
825  MemoryBasicInformation,
826  &MemoryInformation,
827  sizeof( MemoryInformation ),
828  NULL
829  );
830  if (!NT_SUCCESS( Status )) {
831  leave;
832  }
833 
834  //
835  // See if there is room for new, larger value. If not
836  // allocate a new copy of the environment variable
837  // block.
838  //
839 
840  NewSize = (pEnd - (PWSTR)pOld) * sizeof(WCHAR) +
841  Name->Length +
842  sizeof(WCHAR) +
843  Value->Length +
844  sizeof(WCHAR);
845  } else {
846  NewSize = Name->Length +
847  sizeof(WCHAR) +
848  Value->Length +
