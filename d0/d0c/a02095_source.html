245 VOID
246 DbgkpCloseObject (
247  IN PEPROCESS Process,
248  IN PVOID Object,
249  IN ACCESS_MASK GrantedAccess,
250  IN ULONG_PTR ProcessHandleCount,
251  IN ULONG_PTR SystemHandleCount
252  )
253 /*++
254 
255 Routine Description:
256 
257  Called by the object manager when a handle is closed to the object.
258 
259 Arguments:
260 
261  Process - Process doing the close
262  Object - Debug object being deleted
263  GrantedAccess - Access ranted for this handle
264  ProcessHandleCount - Unused and unmaintained by OB
265  SystemHandleCount - Current handle count for this object
266 
267 Return Value:
268 
269  None.
270 
271 --*/
272 {
273  PDEBUG_OBJECT DebugObject = Object;
274  PDEBUG_EVENT DebugEvent;
275  PLIST_ENTRY ListPtr;
276  BOOLEAN Deref;
277 
278  PAGED_CODE ();
279 
280  UNREFERENCED_PARAMETER (GrantedAccess);
281  UNREFERENCED_PARAMETER (ProcessHandleCount);
282 
283  //
284  // If this isn't the last handle then do nothing.
285  //
286  if (SystemHandleCount > 1) {
287  return;
288  }
289 
290  ExAcquireFastMutex (&DebugObject->Mutex);
291 
292  //
293  // Mark this object as going away and wake up any processes that are waiting.
294  //
295  DebugObject->Flags |= DEBUG_OBJECT_DELETE_PENDING;
296 
297  //
298  // Remove any events and queue them to a temporary queue
299  //
300  ListPtr = DebugObject->EventList.Flink;
301  InitializeListHead (&DebugObject->EventList);
302 
303  ExReleaseFastMutex (&DebugObject->Mutex);
304 
305  //
306  // Wake anyone waiting. They need to leave this object alone now as its deleting
307  //
308  KeSetEvent (&DebugObject->EventsPresent, 0, FALSE);
309 
310  //
311  // Loop over all processes and remove the debug port from any that still have it.
312  // Debug port propagation was disabled by setting the delete pending flag above so we only have to do this
313  // once. No more refs can appear now.
314  //
315  for (Process = PsGetNextProcess (NULL);
316  Process != NULL;
317  Process = PsGetNextProcess (Process)) {
318 
319  if (Process->DebugPort == DebugObject) {
320  Deref = FALSE;
321  ExAcquireFastMutex (&DbgkpProcessDebugPortMutex);
322  if (Process->DebugPort == DebugObject) {
323  Process->DebugPort = NULL;
324  Deref = TRUE;
325  }
326  ExReleaseFastMutex (&DbgkpProcessDebugPortMutex);
327 
328 
329  if (Deref) {
330  DbgkpMarkProcessPeb (Process);
331  //
332  // If the caller wanted process deletion on debugger dying (old interface) then kill off the process.
333  //
334  if (DebugObject->Flags&DEBUG_OBJECT_KILL_ON_CLOSE) {
335  PsTerminateProcess (Process, STATUS_DEBUGGER_INACTIVE);
336  }
337  ObDereferenceObject (DebugObject);
338  }
339  }
340  }
341  //
342  // Wake up all the removed threads.
343  //
344  while (ListPtr != &DebugObject->EventList) {
345  DebugEvent = CONTAINING_RECORD (ListPtr, DEBUG_EVENT, EventList);
346  ListPtr = ListPtr->Flink;
347  DebugEvent->Status = STATUS_DEBUGGER_INACTIVE;
348  DbgkpWakeTarget (DebugEvent);
349  }

457 NTSTATUS
458 NtCreateDebugObject (
459  OUT PHANDLE DebugObjectHandle,
460  IN ACCESS_MASK DesiredAccess,
461  IN POBJECT_ATTRIBUTES ObjectAttributes,
462  IN ULONG Flags
463  )
464 /*++
465 
466 Routine Description:
467 
468  Creates a new debug object that maintains the context for a single debug session. Multiple processes may be
469  associated with a single debug object.
470 
471 Arguments:
472 
473  DebugObjectHandle - Pointer to a handle to recive the output objects handle
474  DesiredAccess - Required handle access
475  ObjectAttributes - Standard object attributes structure
476  Flags - Only one flag DEBUG_KILL_ON_CLOSE
477 
478 Return Value:
479 
480  NTSTATUS - Status of call.
481 
482 --*/
483 {
484  NTSTATUS Status;
485  HANDLE Handle;
486  KPROCESSOR_MODE PreviousMode;
487  PDEBUG_OBJECT DebugObject;
488 
489  PAGED_CODE();
490 
491  //
492  // Get previous processor mode and probe output arguments if necessary.
493  // Zero the handle for error paths.
494  //
495 
496  PreviousMode = KeGetPreviousMode();
497 
498  try {
499  if (PreviousMode != KernelMode) {
500  ProbeForWriteHandle (DebugObjectHandle);
501  }
502  *DebugObjectHandle = NULL;
503 
504  } except (ExSystemExceptionFilter ()) { // If previous mode is kernel then don't handle the exception
505  return GetExceptionCode ();
506  }
507 
508  if (Flags & ~DEBUG_KILL_ON_CLOSE) {
509  return STATUS_INVALID_PARAMETER;
510  }
511 
512  //
513  // Create a new debug object and initialize it.
514  //
515 
516  Status = ObCreateObject (PreviousMode,
517  DbgkDebugObjectType,
518  ObjectAttributes,
519  PreviousMode,
520  NULL,
521  sizeof (DEBUG_OBJECT),
522  0,
523  0,
524  &DebugObject);
525 
526  if (!NT_SUCCESS (Status)) {
527  return Status;
528  }
529 
530  ExInitializeFastMutex (&DebugObject->Mutex);
531  InitializeListHead (&DebugObject->EventList);
532  KeInitializeEvent (&DebugObject->EventsPresent, NotificationEvent, FALSE);
533 
534  if (Flags & DEBUG_KILL_ON_CLOSE) {
535  DebugObject->Flags = DEBUG_OBJECT_KILL_ON_CLOSE;
536  } else {
537  DebugObject->Flags = 0;
538  }
539 
540  //
541  // Insert the object into the handle table
542  //
543  Status = ObInsertObject (DebugObject,
544  NULL,
545  DesiredAccess,
546  0,
547  NULL,
548  &Handle);
549 
550 
551  if (!NT_SUCCESS (Status)) {
552  return Status;
553  }
554 
555  try {
556  *DebugObjectHandle = Handle;
557  } except (ExSystemExceptionFilter ()) {
558  //
559  // The caller changed the page protection or deleted the memory for the handle.
560  // No point closing the handle as process rundown will do that and we don't know its still the same handle
561  //
562  Status = GetExceptionCode ();
563  }
564 
565  return Status;
566 }

597 NTSTATUS
598 DbgkpQueueMessage (
599  IN PEPROCESS Process,
600  IN PETHREAD Thread,
601  IN OUT PDBGKM_APIMSG ApiMsg,
602  IN ULONG Flags,
603  IN PDEBUG_OBJECT TargetDebugObject
604  )
605 /*++
606 
607 Routine Description:
608 
609  Queues a debug message to the port for a user mode debugger to get.
610 
611 Arguments:
612 
613  Process - Process being debugged
614  Thread - Thread making call
615  ApiMsg - Message being sent and received
616  NoWait - Don't wait for a response. Buffer message and return.
617  TargetDebugObject - Port to queue nowait messages to
618 
619 Return Value:
620 
621  NTSTATUS - Status of call.
622 
623 --*/
624 {
625  PDEBUG_EVENT DebugEvent;
626  DEBUG_EVENT StaticDebugEvent;
627  PDEBUG_OBJECT DebugObject;
628  NTSTATUS Status;
629 
630  PAGED_CODE ();
631 
632  if (Flags&DEBUG_EVENT_NOWAIT) {
633  DebugEvent = ExAllocatePoolWithQuotaTag (NonPagedPool|POOL_QUOTA_FAIL_INSTEAD_OF_RAISE,
634  sizeof (*DebugEvent),
635  'EgbD');
636  if (DebugEvent == NULL) {
637  return STATUS_INSUFFICIENT_RESOURCES;
638  }
639  DebugEvent->Flags = Flags|DEBUG_EVENT_INACTIVE;
640  ObReferenceObject (Process);
641  ObReferenceObject (Thread);
642  DebugEvent->BackoutThread = PsGetCurrentThread ();
643  DebugObject = TargetDebugObject;
644  } else {
645  DebugEvent = &StaticDebugEvent;
646  DebugEvent->Flags = Flags;
647 
648  ExAcquireFastMutex (&DbgkpProcessDebugPortMutex);
649 
650  DebugObject = Process->DebugPort;
651 
652  //
653  // See if this create message has already been sent.
654  //
655  if (ApiMsg->ApiNumber == DbgKmCreateThreadApi ||
656  ApiMsg->ApiNumber == DbgKmCreateProcessApi) {
657  if (Thread->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_SKIP_CREATION_MSG) {
658  DebugObject = NULL;
659  }
660  }
661 
662  //
663  // See if this exit message is for a thread that never had a create
664  //
665  if (ApiMsg->ApiNumber == DbgKmExitThreadApi ||
666  ApiMsg->ApiNumber == DbgKmExitProcessApi) {
667  if (Thread->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_SKIP_TERMINATION_MSG) {
668  DebugObject = NULL;
669  }
670  }
671  }
672 
673  KeInitializeEvent (&DebugEvent->ContinueEvent, SynchronizationEvent, FALSE);
674 
675  DebugEvent->Process = Process;
676  DebugEvent->Thread = Thread;
677  DebugEvent->ApiMsg = *ApiMsg;
678  DebugEvent->ClientId = Thread->Cid;
679 
680  if (DebugObject == NULL) {
681  Status = STATUS_PORT_NOT_SET;
682  } else {
683 
684  //
685  // We must not use a debug port thats got no handles left.
686  //
687  ExAcquireFastMutex (&DebugObject->Mutex);
688 
689  //
690  // If the object is delete pending then don't use this object.
691  //
692  if ((DebugObject->Flags&DEBUG_OBJECT_DELETE_PENDING) == 0) {
693  InsertTailList (&DebugObject->EventList, &DebugEvent->EventList);
694  //
695  // Set the event to say there is an unread event in the object
696  //
697  if ((Flags&DEBUG_EVENT_NOWAIT) == 0) {
698  KeSetEvent (&DebugObject->EventsPresent, 0, FALSE);
699  }
700  Status = STATUS_SUCCESS;
701  } else {
702  Status = STATUS_DEBUGGER_INACTIVE;
703  }
704 
705  ExReleaseFastMutex (&DebugObject->Mutex);
706  }
707 
708 
709  if ((Flags&DEBUG_EVENT_NOWAIT) == 0) {
710  ExReleaseFastMutex (&DbgkpProcessDebugPortMutex);
711 
712  if (NT_SUCCESS (Status)) {
713  KeWaitForSingleObject (&DebugEvent->ContinueEvent,
714  Executive,
715  KernelMode,
716  FALSE,
717  NULL);
718 
719  Status = DebugEvent->Status;
720  *ApiMsg = DebugEvent->ApiMsg;
721  }
722  } else {
723  if (!NT_SUCCESS (Status)) {
724  ObDereferenceObject (Process);
725  ObDereferenceObject (Thread);
726  ExFreePool (DebugEvent);
727  }
728  }
729 
730  return Status;
731 }

764

766  if (DebugObject == NULL || (DebugObject != SourceDebugObject && SourceDebugObject != NULL)) {

783  //

785  //
786  InitializeListHead (&TempList);
787

1194  }
1195  } else {

1201  }
1202  }
