38 ObpCloseHandleTableEntry (
39  IN PHANDLE_TABLE ObjectTable,
40  IN PHANDLE_TABLE_ENTRY ObjectTableEntry,
41  IN HANDLE Handle,
42  IN KPROCESSOR_MODE PreviousMode,
43  IN BOOLEAN Rundown
44  )
45 /*++
46 
47 Routine Description:
48 
49  This function is used to close a handle table entry
50 
51 Arguments:
52 
53  ObjectTableEntry - Supplies the entry being closed. It must be locked
54  PreviousMode - Mode of caller
55  Rundown - Called as part of process rundown, ignore protected handles in this mode
56 
57 Return Value:
58 
59  NTSTATUS.
60 
61 --*/
62 {
63  POBJECT_HEADER ObjectHeader;
64  POBJECT_TYPE ObjectType;
65  PVOID Object;
66  ULONG CapturedGrantedAccess;
67  ULONG CapturedAttributes;
68  #if DBG
69  KIRQL SaveIrql;
70  #endif // DBG
71 
72  //
73  // From the object table entry we can grab a pointer to the object
74  // header, get its type and its body
75  //
76 
77  ObjectHeader = (POBJECT_HEADER)(((ULONG_PTR)(ObjectTableEntry->Object)) & ~OBJ_HANDLE_ATTRIBUTES);
78  ObjectType = ObjectHeader->Type;
79  Object = &ObjectHeader->Body;
80 
81  //
82  // If the object type specifies an okay to close procedure then we
83  // need to invoke that callback. If the callback doesn't want us to
84  // close handle then unlock the object table and return the error
85  // to our caller
86  //
87 
88  if (ObjectType->TypeInfo.OkayToCloseProcedure != NULL) {
89 
90  ObpBeginTypeSpecificCallOut( SaveIrql );
91 
92  if (!(*ObjectType->TypeInfo.OkayToCloseProcedure)( PsGetCurrentProcess(),
93  Object,
94  Handle,
95  PreviousMode )) {
96 
97  ObpEndTypeSpecificCallOut( SaveIrql, "NtClose", ObjectType, Object );
98 
99  ExUnlockHandleTableEntry( ObjectTable, ObjectTableEntry );
100 
101  return STATUS_HANDLE_NOT_CLOSABLE;
102  }
103 
104  ObpEndTypeSpecificCallOut( SaveIrql, "NtClose", ObjectType, Object );
105  }
106 
107  CapturedAttributes = ObpGetHandleAttributes(ObjectTableEntry);
108 
109  //
110  // If the previous mode was user and the handle is protected from
111  // being closed, then we'll either raise or return an error depending
112  // on the global flags and debugger port situation.
113  //
114 
115  if ((CapturedAttributes & OBJ_PROTECT_CLOSE) != 0 && Rundown == FALSE) {
116 
117  if (PreviousMode != KernelMode) {
118 
119  ExUnlockHandleTableEntry( ObjectTable, ObjectTableEntry );
120 
121  if (!KeIsAttachedProcess() &&
122  ((NtGlobalFlag & FLG_ENABLE_CLOSE_EXCEPTIONS) ||
123  (PsGetCurrentProcess()->DebugPort != NULL) ||
124  (ObjectTable->DebugInfo != NULL))) {
125 
126  //
127  // Raise and exception in user mode
128  //
129  return KeRaiseUserException(STATUS_HANDLE_NOT_CLOSABLE);
130 
131  } else {
132 
133  return STATUS_HANDLE_NOT_CLOSABLE;
134  }
135 
136  } else {
137  KeBugCheckEx(INVALID_KERNEL_HANDLE, (ULONG_PTR)Handle, 0, 0, 0);
138  }
139  }
140 
141  //
142  // Get the granted access for the handle
143  //
144 
145 #if i386
146 
147  if (NtGlobalFlag & FLG_KERNEL_STACK_TRACE_DB) {
148 
149  CapturedGrantedAccess = ObpTranslateGrantedAccessIndex( ObjectTableEntry->GrantedAccessIndex );
150 
151  } else {
152 
153  CapturedGrantedAccess = ObpDecodeGrantedAccess(ObjectTableEntry->GrantedAccess);
154  }
155 
156 #else
157 
158  CapturedGrantedAccess = ObpDecodeGrantedAccess(ObjectTableEntry->GrantedAccess);
159 
160 #endif // i386
161 
162  //
163  // Now remove the handle from the handle table
164  //
165 
166  ExDestroyHandle( ObjectTable,
167  Handle,
168  ObjectTableEntry );
169 
170  //
171  // perform any auditing required
172  //
173 
174  //
175  // Extract the value of the GenerateOnClose bit stored
176  // after object open auditing is performed. This value
177  // was stored by a call to ObSetGenerateOnClosed.
178  //
179 
180  if (CapturedAttributes & OBJ_AUDIT_OBJECT_CLOSE) {
181 
182  if ( SepAdtAuditingEnabled ) {
183 
184  SeCloseObjectAuditAlarm( Object,
185  (HANDLE)((ULONG_PTR)Handle & ~OBJ_HANDLE_TAGBITS), // Mask off the tagbits defined for OB objects.
186  TRUE );
187  }
188  }
189 
190  //
191  // Since we took the handle away we need to decrement the objects
192  // handle count, and remove a reference
193  //
194 
195  ObpDecrementHandleCount( PsGetCurrentProcess(),
196  ObjectHeader,
197  ObjectType,
198  CapturedGrantedAccess );
199 
200  ObDereferenceObject( Object );
201 
202  //
203  // And return to our caller
204  //
205 
206  return STATUS_SUCCESS;
207 }

210 NTSTATUS
211 ObpCloseHandle (
212  IN HANDLE Handle,
213  IN KPROCESSOR_MODE PreviousMode
214  )
215 /*++
216 
217 Routine Description:
218 
219  This function is used to close access to the specified handle with the given mode
220 
221 Arguments:
222 
223  Handle - Supplies the handle being closed
224  PreviousMode - Processor mode to be used in the handle access checks.
225  CanNotRaise - We are not allowed to do a user mode raise.
226 
227 Return Value:
228 
229  An appropriate status value
230 
231 --*/
232 {
233  PHANDLE_TABLE ObjectTable;
234  PHANDLE_TABLE_ENTRY ObjectTableEntry;
235  NTSTATUS Status;
236  BOOLEAN AttachedToProcess = FALSE;
237  KAPC_STATE ApcState;
238  PETHREAD CurrentThread;
239  PEPROCESS CurrentProcess;
240 
241 
242  ObpValidateIrql( "NtClose" );
243 
244  CurrentThread = PsGetCurrentThread ();
245  CurrentProcess = PsGetCurrentProcessByThread (CurrentThread);
246  //
247  // For the current process we will grab its handle/object table and
248  // translate the handle to its corresponding table entry. If the
249  // call is successful it also lock down the handle table. But first
250  // check for a kernel handle and attach and use that table if so.
251  //
252 
253  if (IsKernelHandle( Handle, PreviousMode )) {
254 
255  Handle = DecodeKernelHandle( Handle );
256 
257  ObjectTable = ObpKernelHandleTable;
258 
259  //
260  // Go to the system process if we have to
261  //
262  if (CurrentProcess != PsInitialSystemProcess) {
263  KeStackAttachProcess (&PsInitialSystemProcess->Pcb, &ApcState);
264  AttachedToProcess = TRUE;
265  }
266 
267  } else {
268 
269  ObjectTable = CurrentProcess->ObjectTable;
270  }
271 
272  //
273  // Protect ourselves from being interrupted while we hold a handle table
274  // entry lock
275  //
276 
277  KeEnterCriticalRegionThread(&CurrentThread->Tcb);
278 
279  ObjectTableEntry = ExMapHandleToPointer( ObjectTable,
280  Handle );
281 
282  //
283  // Check that the specified handle is legitimate otherwise we can
284  // assume the caller just passed in some bogus handle value
285  //
286 
287  if (ObjectTableEntry != NULL) {
288 
289  Status = ObpCloseHandleTableEntry (ObjectTable, ObjectTableEntry, Handle, PreviousMode, FALSE);
290 
291  KeLeaveCriticalRegionThread(&CurrentThread->Tcb);
292  //
293  // If we are attached to the system process then detach
294  //
295  if (AttachedToProcess) {
296 
297  KeUnstackDetachProcess(&ApcState);
298  AttachedToProcess = FALSE;
299  }
300 
301 
302  } else {
303 
304  KeLeaveCriticalRegionThread(&CurrentThread->Tcb);
305 
306  //
307  // At this point the input handle did not translate to a valid
308  // object table entry
309  //
310 
311  //
312  // If we are attached to the system process then return
313  // back to our caller
314  //
315 
316  if (AttachedToProcess) {
317  KeUnstackDetachProcess(&ApcState);
318  AttachedToProcess = FALSE;
319  }
320 
321  //
322  // Now if the handle is not null and it does not represent the
323  // current thread or process then if we're user mode we either raise
324  // or return an error
325  //
326 
327  if ((Handle != NULL) &&
328  (Handle != NtCurrentThread()) &&
329  (Handle != NtCurrentProcess())) {
330 
331  if (PreviousMode != KernelMode) {
332 
333  if ((NtGlobalFlag & FLG_ENABLE_CLOSE_EXCEPTIONS) ||
334  (CurrentProcess->DebugPort != NULL) ||
335  (ObjectTable->DebugInfo != NULL)) {
336 
337  if (!KeIsAttachedProcess()) {
338  return KeRaiseUserException (STATUS_INVALID_HANDLE);
339  } else {
340  return STATUS_INVALID_HANDLE;
341  }
342 
343  }
344 
345  } else {
346 
347  //
348  // bugcheck here if kernel debugger is enabled and if kernel mode code is
349  // closing a bogus handle and process is not exiting. Ignore
350  // if no PEB as this occurs if process is killed before
351  // really starting.
352  //
353 
354  if ((!PsIsThreadTerminating(CurrentThread)) &&
355  (CurrentProcess->Peb != NULL)) {
356 
357  if (KdDebuggerEnabled) {
358  KeBugCheckEx(INVALID_KERNEL_HANDLE, (ULONG_PTR)Handle, 1, 0, 0);
359  }
360  }
361 
362  }
363  }
364 
365  Status = STATUS_INVALID_HANDLE;
366  }
367 
368 
369  return Status;
370 }

429 /*++

450

458

460  DELETE,
