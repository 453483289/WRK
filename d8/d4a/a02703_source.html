199 
200 Routine Description:
201 
202  Determines if the process token is allowed to impersonate the
203  second token, assuming that the access rights check has already passed.
204 
205 Arguments:
206 
207  Token - Points to the token.
208 
209 Return Value:
210 
211  TRUE - Token contains the local admin group
212  FALSE - no admin.
213 
214 --*/
215 
216 {
217  PTOKEN PrimaryToken = (PTOKEN) ProcessToken ;
218  PTOKEN ImpToken = (PTOKEN) Token ;
219  PSID PrimaryUserSid ;
220  PSID ImpUserSid ;
221  NTSTATUS Status ;
222 
223  PAGED_CODE();
224 
225  if ( ImpersonationLevel < SecurityImpersonation )
226  {
227  return STATUS_SUCCESS ;
228  }
229 
230  //
231  // allow impersonating anonymous tokens
232  //
233 
234  if (RtlEqualLuid(&ImpToken->AuthenticationId, &SeAnonymousAuthenticationId))
235  {
236  return STATUS_SUCCESS ;
237  }
238 
239  SepAcquireTokenReadLock( PrimaryToken );
240 
241  if ((PrimaryToken->TokenFlags & TOKEN_HAS_IMPERSONATE_PRIVILEGE) != 0 )
242  {
243  SepReleaseTokenReadLock( PrimaryToken );
244 
245  return STATUS_SUCCESS ;
246  }
247 
248  SepAcquireTokenReadLock( ImpToken );
249 
250  Status = STATUS_PRIVILEGE_NOT_HELD ;
251 
252  if ( RtlEqualLuid( &PrimaryToken->AuthenticationId, &ImpToken->OriginatingLogonSession ) )
253  {
254  Status = STATUS_SUCCESS ;
255 
256  }
257  else
258  {

260  ImpUserSid = ImpToken->UserAndGroups[0].Sid ;

267  // is attempting to elevate to the nonrestricted version of itself. Do
268  // not allow this.
269  //

271  Status = STATUS_PRIVILEGE_NOT_HELD;
272  } else {
273  Status = STATUS_SUCCESS ;
274  }
275 
276  }
277 
278  }
279 
280  SepReleaseTokenReadLock( ImpToken );
281  SepReleaseTokenReadLock( PrimaryToken );
282 
283 #if DBG
284  if ( !NT_SUCCESS( Status ) )
285  {
286  DbgPrint( "Process %x.%x not allowed to impersonate! Returning %x\n", PsGetCurrentThread()->Cid.UniqueProcess,

507 
508 /*++
509 
510 Routine Description:
511 
512  This function establishes a primary token for a process.
513 
514 Arguments:
515 
516  Token - Points to the new primary token.
517 
518 Return Value:
519 
520  None.
521 
522 --*/
523 
524 {
525  NTSTATUS
526  Status;
527 
528  PTOKEN
529  NewToken = (PTOKEN)Token;
530 
531  PAGED_CODE();
532 
533  ASSERT(NewToken->TokenType == TokenPrimary);
534  ASSERT( !NewToken->TokenInUse );
535 
536 
537  //
538  // audit the assignment of a primary token, if requested
539  //
540 
541  if (SeDetailedAuditingWithToken(NULL)) {
542  SepAuditAssignPrimaryToken( Process, Token );
543  }
544 
545  //
546  // If the token being assigned to the child process has
547  // any one of the following SIDs, then the process
548  // is considered to be a system process:
549  // -- SeLocalSystemSid
550  // -- SeLocalServiceSid
551  // -- SeNetworkServiceSid
552  //
553  // For such a process, add SACL to its security descriptor
554  // if that option is enabled. If the option is disabled,
555  // this function returns very quickly.
556  //
557 
558  //
559  // Dereference the old token if there is one.
560  //
561  // Processes typically already have a token that must be
562  // dereferenced. There are two cases where this may not
563  // be the situation. First, during phase 0 system initialization,
564  // the initial system process starts out without a token. Second,
565  // if an error occurs during process creation, we may be cleaning
566  // up a process that hasn't yet had a primary token assigned.
567  //
568 
569  if (!ExFastRefObjectNull (Process->Token)) {
570  SeDeassignPrimaryToken( Process );
571  }
572 
573  ObReferenceObject(NewToken);
574  NewToken->TokenInUse = TRUE;
575 
576  ObInitializeFastReference (&Process->Token, Token);
577  return;

626 NTSTATUS
627 SeExchangePrimaryToken(
628  __in PEPROCESS Process,
629  __in PACCESS_TOKEN NewAccessToken,
630  __deref_out PACCESS_TOKEN *OldAccessToken
631  )
632 
633 
634 /*++
635 
636 Routine Description:
637 
638  This function is used to perform the portions of changing a primary
639  token that reference the internals of token structures.
640 
641  The new token is checked to make sure it is not already in use.
642 
643 
644 Arguments:
645 
646  Process - Points to the process whose primary token is being exchanged.
647 
648  NewAccessToken - Points to the process's new primary token.
649 
650  OldAccessToken - Receives a pointer to the process's current token.
651  The caller is responsible for dereferencing this token when
652  it is no longer needed. This can't be done while the process
653  security locks are held.
654 
655 
656 Return Value:
657 
658  STATUS_SUCCESS - Everything has been updated.
659 
660  STATUS_TOKEN_ALREADY_IN_USE - A primary token can only be used by a
661  single process. That is, each process must have its own primary
662  token. The token passed to be assigned as the primary token is
663  already in use as a primary token.
664 
665  STATUS_BAD_TOKEN_TYPE - The new token is not a primary token.
666 
667  STATUS_NO_TOKEN - The process did not have any existing token. This should never happen.
668 
669 --*/
670 
671 {
672  NTSTATUS
673  Status;
674 
675  PTOKEN
676  OldToken;
677 
678  PTOKEN
679  NewToken = (PTOKEN)NewAccessToken;
680 
681  ULONG SessionId;
682 
683  PAGED_CODE();
684 
685 
686  //
687  // Make sure the new token is a primary token...
688  //
689 
690  if (NewToken->TokenType != TokenPrimary) {
691  return (STATUS_BAD_TOKEN_TYPE);
692  }
693 
694  SessionId = MmGetSessionId (Process);
695 
696  //
697  // Lock the new token so we can atomically test and set the InUse flag
698  //
699 
700  SepAcquireTokenWriteLock (NewToken);
701 
702  //
703  // and that it is not already in use...
704  //
705 
706  if (NewToken->TokenInUse) {
707  SepReleaseTokenWriteLock (NewToken, FALSE);
708  return (STATUS_TOKEN_ALREADY_IN_USE);
709  }
710 
711  NewToken->TokenInUse = TRUE;
712 
713  //
714  // Ensure SessionId consistent for hydra
715  //
716 
717  NewToken->SessionId = SessionId;
718 
719  SepReleaseTokenWriteLock (NewToken, FALSE);
720 
721  //
722  // audit the assignment of a primary token, if requested
723  //
724 
725  if (SeDetailedAuditingWithToken (NULL)) {
726  SepAuditAssignPrimaryToken (Process, NewToken);
727  }
728 
729  //
730  // If the token being assigned to this process has
731  // any one of the following SIDs, then the process
732  // is considered to be a system process:
733  // -- SeLocalSystemSid
734  // -- SeLocalServiceSid
735  // -- SeNetworkServiceSid
736  //
737 
738  //
739  // Switch the tokens
740  //
741 
742  ObReferenceObject (NewToken);
743 
744  OldToken = ObFastReplaceObject (&Process->Token, NewToken);
745 
746  if (NULL == OldToken){
747  return (STATUS_NO_TOKEN);
748  }
749 
750  ASSERT (OldToken->TokenType == TokenPrimary);
751 
752  //
753  // Lock the old token to clkear the InUse flag
754  //
755 
756  SepAcquireTokenWriteLock (OldToken);
757 
758  ASSERT (OldToken->TokenInUse);
759 
760  //
761  // Mark the token as "NOT USED"
762  //
763 
764  OldToken->TokenInUse = FALSE;
765 
766  SepReleaseTokenWriteLock (OldToken, FALSE);
767 
768  //
769  // Return the pointer to the old token. The caller
770  // is responsible for dereferencing it if they don't need it.
771  //
772 
773  (*OldAccessToken) = OldToken;
774 
775  return (STATUS_SUCCESS);
776 }
