26

28  // The thread that currently owns the memory.
29  //

55 NTSTATUS
56 FsFilterInit(
57  )
58 
59 /*++
60 
61 Routine Description:
62 
63  This routine initializes the reserve pool the FsFilter routine need to use
64  when the system is in low memory conditions.
65 
66 Arguments:
67 
68  None.
69 
70 Return Value:
71 
72  Returns STATUS_SUCCESS if the initialization was successful, or
73  STATUS_INSUFFICIENT_RESOURCES otherwise.
74 
75 --*/
76 
77 {
78  NTSTATUS Status = STATUS_SUCCESS;
79 
80  AcquireOpsReservePool = ExAllocatePoolWithTag( NonPagedPool,
81  sizeof( FS_FILTER_RESERVE ),
82  FSRTL_FILTER_MEMORY_TAG );
83 
84  if (AcquireOpsReservePool == NULL) {
85 
86  Status = STATUS_INSUFFICIENT_RESOURCES;
87  }
88 
89  ReleaseOpsReservePool = ExAllocatePoolWithTag( NonPagedPool,
90  sizeof( FS_FILTER_RESERVE ),
91  FSRTL_FILTER_MEMORY_TAG );
92 
93  if (ReleaseOpsReservePool == NULL) {
94 
95  ExFreePoolWithTag( AcquireOpsReservePool,
96  FSRTL_FILTER_MEMORY_TAG );
97  Status = STATUS_INSUFFICIENT_RESOURCES;
98  }
99 
100 
101  KeInitializeEvent( &AcquireOpsEvent, SynchronizationEvent, TRUE );
102  KeInitializeEvent( &ReleaseOpsEvent, SynchronizationEvent, TRUE );
103 
104  return Status;
105 }

107 NTSTATUS
108 FsFilterAllocateCompletionStack (
109  IN PFS_FILTER_CTRL FsFilterCtrl,
110  IN BOOLEAN CanFail,
111  OUT PULONG AllocationSize
112  )
113 
114 /*++
115 
116 Routine Description:
117 
118  This routine allocates a completion stack for the given FsFilterCtrl. If
119  this allocation cannot fail, then this routine will wait to allocation
120  the memory from the FsFilter reserved pool.
121 
122  This routine initialized the appropriate CompletionStack parameters and
123  FsFilterCtrl flags to reflect the allocation made.
124 
125 Arguments:
126 
127  FsFilterCtrl - The FsFilterCtrl structure for which the completion stack
128  must be allocated.
129  CanFail - TRUE if the allocation is allowed to fail, FALSE otherwise.
130  AllocationSize - Set to the nuber of bytes of memory allocated for the
131  completion stack for this FsFilterCtrl.
132 
133 Return Value:
134 
135  Returns STATUS_SUCCESS if the memory was successfully allocated for the
136  completion stack, or STATUS_INSUFFICIENT_RESOURCES otherwise.
137 
138 --*/
139 
140 {
141  PFS_FILTER_COMPLETION_NODE Stack = NULL;
142  PFS_FILTER_RESERVE ReserveBlock = NULL;
143  PKEVENT Event = NULL;
144 
145  ASSERT( FsFilterCtrl != NULL );
146  ASSERT( AllocationSize != NULL );
147 
148  *AllocationSize = FsFilterCtrl->CompletionStack.StackLength *
149  sizeof( FS_FILTER_COMPLETION_NODE );
150 
151  Stack = ExAllocatePoolWithTag( NonPagedPool,
152  *AllocationSize,
153  FSRTL_FILTER_MEMORY_TAG );
154 
155  if (Stack == NULL) {
156 
157  if (CanFail) {
158 
159  return STATUS_INSUFFICIENT_RESOURCES;
160 
161  } else {
162 
163  //
164  // This allocation cannot fail, so get the needed memory from our
165  // private stash of pool.
166  //
167 
168  switch (FsFilterCtrl->Data.Operation) {
169 
170  case FS_FILTER_ACQUIRE_FOR_SECTION_SYNCHRONIZATION:
171  case FS_FILTER_ACQUIRE_FOR_MOD_WRITE:
172  case FS_FILTER_ACQUIRE_FOR_CC_FLUSH:
173 
174  ReserveBlock = AcquireOpsReservePool;
175  Event = &AcquireOpsEvent;
176  break;
177 
178  case FS_FILTER_RELEASE_FOR_SECTION_SYNCHRONIZATION:
179  case FS_FILTER_RELEASE_FOR_MOD_WRITE:
180  case FS_FILTER_RELEASE_FOR_CC_FLUSH:
181 
182  ReserveBlock = ReleaseOpsReservePool;
183  Event = &ReleaseOpsEvent;
184  break;
185 
186  default:
187 
188  //
189  // This shouldn't happen since we should always cover all
190  // possible types of operations in the above cases.
191  //
192 
193  ASSERTMSG( "FsFilterAllocateMemory: Unknown operation type\n",
194  FALSE );
195  }
196 
197  //
198  // Wait to get on the appropriate event so that we know the reserve
199  // memory is available for use.
200  //
201 
202  KeWaitForSingleObject( Event,
203  Executive,
204  KernelMode,
205  FALSE,
206  NULL );
207 
208  //
209  // We've been signaled, so the reserved block is available.
210  //
211 
212  ReserveBlock->Owner = PsGetCurrentThread();
213  Stack = ReserveBlock->Stack;
214  SetFlag( FsFilterCtrl->Flags, FS_FILTER_USED_RESERVE_POOL );
215  }
216  }
217 
218  ASSERT( Stack != NULL );
219 
220  //
221  // We've now got our block of memory, so initialize the completion stack.
222  //
223 
224  SetFlag( FsFilterCtrl->Flags, FS_FILTER_ALLOCATED_COMPLETION_STACK );
225  FsFilterCtrl->CompletionStack.Stack = Stack;
226 
227  return STATUS_SUCCESS;
228 }

230 VOID
231 FsFilterFreeCompletionStack (
232  IN PFS_FILTER_CTRL FsFilterCtrl
233  )
234 
235 /*++
236 
237 Routine Description:
238 
239  This routine frees the allocated completion stack in the FsFilterCtrl
240  parameter.
241 
242 Arguments:
243 
244  FsFilterCtrl - The FsFilterCtrl structure for which the completion stack
245  must be freed.
246 
247 Return Value:
248 
249  None.
250 
251 --*/
252 
253 {
254  PKEVENT Event = NULL;
255  PFS_FILTER_RESERVE ReserveBlock = NULL;
256 
257  ASSERT( FsFilterCtrl != NULL );
258  ASSERT( FlagOn( FsFilterCtrl->Flags, FS_FILTER_ALLOCATED_COMPLETION_STACK ) );
259 
260  if (!FlagOn( FsFilterCtrl->Flags, FS_FILTER_USED_RESERVE_POOL )) {
261 
262  //
263  // We were able to allocate this from the generate pool of memory,
264  // so just free the memory block used for the completion stack.
265  //
266 
267  ExFreePoolWithTag( FsFilterCtrl->CompletionStack.Stack,
268  FSRTL_FILTER_MEMORY_TAG );
269 
270  } else {
271 
272  //
273  // This allocation from our private pool stash, so use the operation
274  // to figure out which private stash.
275  //
276 
277  switch (FsFilterCtrl->Data.Operation) {
278  case FS_FILTER_ACQUIRE_FOR_SECTION_SYNCHRONIZATION:
279  case FS_FILTER_ACQUIRE_FOR_MOD_WRITE:
280  case FS_FILTER_ACQUIRE_FOR_CC_FLUSH:
281 
282  Event = &AcquireOpsEvent;
283  break;
284 
285  case FS_FILTER_RELEASE_FOR_SECTION_SYNCHRONIZATION:
286  case FS_FILTER_RELEASE_FOR_MOD_WRITE:
287  case FS_FILTER_RELEASE_FOR_CC_FLUSH:
288 
289  Event = &ReleaseOpsEvent;
290  break;
291 
292  default:
293 
294  //
295  // This shouldn't happen since we should always cover all
296  // possible types of operations in the above cases.
297  //
298 
299  ASSERTMSG( "FsFilterAllocateMemory: Unknown operation type\n",
300  FALSE );
301  }
302 
303  ASSERT( Event != NULL );
304 
305  //
306  // Clear out the owner of the reserve block before setting the event.
307  //
308 
309  ReserveBlock = CONTAINING_RECORD( FsFilterCtrl->CompletionStack.Stack,
310  FS_FILTER_RESERVE,
311  Stack );
312  ReserveBlock->Owner = NULL;
313 
314  //
315  // Now we are ready to release the reserved block to the next thread
316  // that needs it.
317  //
318 
319  KeSetEvent( Event, IO_NO_INCREMENT, FALSE );
320  }
321 }

323 NTSTATUS
324 FsFilterCtrlInit (
325  IN OUT PFS_FILTER_CTRL FsFilterCtrl,
326  IN UCHAR Operation,
327  IN PDEVICE_OBJECT DeviceObject,
328  IN PDEVICE_OBJECT BaseFsDeviceObject,
329  IN PFILE_OBJECT FileObject,
330  IN BOOLEAN CanFail
331  )
332 
333 /*++
334 
335 Routine Description:
336 
337  This routine initializes the FsFilterCtrl structure used to store the
338  contexts and post-operation callbacks for this operation. If the
339  default completion stack is not large enough, this routine will need to
340  allocate a completion stack of sufficient size to store all the possible
341  contexts and post-operation callbacks.
342 
343 Arguments:
344 
345  FsFilterCtrl - The FsFilterCtrl to initialize.
346  Operation - The operation this FsFilterCtrl is going to be used for.
347  DeviceObject - The device object to which this operation will be targeted.
348  BaseFsDeviceObject - The device object for the base file system at the
349  bottom on this filter stack.
350  FileObject - The file object to which this operation will be targeted.
351  CanFail - TRUE if the call can deal with memory allocations failing,
352  FALSE otherwise.
353 
354 Return Value:
355 
356  STATUS_SUCCESS if the FsFilterCtrl structure could be initialized,
357  STATUS_INSUFFICIENT_RESOURCES if the routine cannot allocate the needed
358  memory to initialize this structure.
359 
360 --*/
361 
362 {
363  PFS_FILTER_CALLBACK_DATA Data;
364  ULONG AllocationSize;
365  NTSTATUS Status = STATUS_SUCCESS;
366 
367  UNREFERENCED_PARAMETER( BaseFsDeviceObject );
368 
369  ASSERT( FsFilterCtrl != NULL );
370  ASSERT( DeviceObject != NULL );
371 
372  FsFilterCtrl->Flags = 0;
373 
374  Data = &(FsFilterCtrl->Data);
375 
376  Data->SizeOfFsFilterCallbackData = sizeof( FS_FILTER_CALLBACK_DATA );
377  Data->Operation = Operation;
378  Data->DeviceObject = DeviceObject;
379  Data->FileObject = FileObject;
380 
381  //
382  // Since it is possible for a filter to redirect this operation to another
383  // stack, we must assume that the stack size of their device object is
384  // large enough to account for the large stack they would need in the
385  // redirection. It is the stack size of the top device object that we
386  // will use to determine the size of our completion stack.
387  //
388 
389  FsFilterCtrl->CompletionStack.StackLength = DeviceObject->StackSize;
390  FsFilterCtrl->CompletionStack.NextStackPosition = 0;
391 
392  if (FsFilterCtrl->CompletionStack.StackLength > FS_FILTER_DEFAULT_STACK_SIZE) {
393 
394  //
395  // The stack isn't big enough, so we must dynamically allocate
396  // the completion stack. This should happen VERY rarely.
397  //
398 
399  Status = FsFilterAllocateCompletionStack( FsFilterCtrl,
400  CanFail,
401  &AllocationSize );
402 
403  //
404  // If the above allocation failed and we cannot fail this allocation,
405  // use our private pool stash.
406  //
407 
408  if (!NT_SUCCESS( Status )) {
409 
410  ASSERT( CanFail );
411  return Status;
412  }
413 
414  ASSERT( FsFilterCtrl->CompletionStack.Stack );
415 
416  } else {
417 
418  //
419  // The default completion noded array allocated for the stack
420  // is large enough, so set Stack to point to that array.
421  //
422 
423  FsFilterCtrl->CompletionStack.Stack = &(FsFilterCtrl->CompletionStack.DefaultStack[0]);
424  AllocationSize = sizeof( FS_FILTER_COMPLETION_NODE ) * FS_FILTER_DEFAULT_STACK_SIZE;
425  FsFilterCtrl->CompletionStack.StackLength = FS_FILTER_DEFAULT_STACK_SIZE;
426  }
427 
428  RtlZeroMemory( FsFilterCtrl->CompletionStack.Stack, AllocationSize );
429 
430  return Status;
431 }

467 VOID
468 FsFilterGetCallbacks (
469  IN UCHAR Operation,
470  IN PDEVICE_OBJECT DeviceObject,
471  OUT PFS_FILTER_CALLBACK *PreOperationCallback,
472  OUT PFS_FILTER_COMPLETION_CALLBACK *PostOperationCallback
473  )
474 
475 /*++
476 
477 Routine Description:
478 
479  This routine looks up the PreOperationCallback and the PostOperationCallback
480  that the filter has registered for this operation if it has registered one.
481 Arguments:
482 
483  Operation - The current operation of interest.
484  DeviceObject - The device object that the filter attached to the file system
485  filter stack.
486  PreOperationCallback - Set to the PreOperationCallback that the filter
487  registered for this operation if one was registered. Otherwise, this
488  is set to NULL.
489  PostOperationCallback - Set to the PostOperationCallback that the filter
490  registered for this operation if one was registered. Otherwise, this
491  is set to NULL.
492 
493 Return Value:
494 
495  NONE
496 
497 --*/
498 
499 {
500 
501  PFS_FILTER_CALLBACKS FsFilterCallbacks;
502 
503  //
504  // Initialize the pre and post callbacks to NULL. If
505  // we have valid callbacks, these output parameters will
506  // get set to the appropriate function pointers.
507  //
508 
509  *PreOperationCallback = NULL;
510  *PostOperationCallback = NULL;
511 
512  FsFilterCallbacks =
513  DeviceObject->DriverObject->DriverExtension->FsFilterCallbacks;
514 
515  if (FsFilterCallbacks == NULL) {
516 
517  //
518  // This filter didn't register any callbacks,
519  // so just return and save switch logic that follows.
520  //
521 
522  return;
523  }
524 
525  //
526  // This device did register at least some callbacks, so see
527  // if there are callbacks for the current operation.
528  //
529 
530  switch (Operation) {
531 
532  case FS_FILTER_ACQUIRE_FOR_SECTION_SYNCHRONIZATION:
533 
534  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
535  PreAcquireForSectionSynchronization )) {
536 
537  *PreOperationCallback = FsFilterCallbacks->PreAcquireForSectionSynchronization;
538 
539  }
540 
541  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
542  PostAcquireForSectionSynchronization)) {
543 
544  *PostOperationCallback = FsFilterCallbacks->PostAcquireForSectionSynchronization;
545 
546  }
547 
548  break;
549 
550  case FS_FILTER_RELEASE_FOR_SECTION_SYNCHRONIZATION:
551 
552  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
553  PreReleaseForSectionSynchronization )) {
554 
555  *PreOperationCallback = FsFilterCallbacks->PreReleaseForSectionSynchronization;
556  }
557 
558  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
559  PostReleaseForSectionSynchronization )) {
560 
561  *PostOperationCallback = FsFilterCallbacks->PostReleaseForSectionSynchronization;
562  }
563 
564  break;
565 
566  case FS_FILTER_ACQUIRE_FOR_MOD_WRITE:
567 
568  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
569  PreAcquireForModifiedPageWriter )) {
570 
571  *PreOperationCallback = FsFilterCallbacks->PreAcquireForModifiedPageWriter;
572  }
573 
574  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
575  PostAcquireForModifiedPageWriter )) {
576 
577  *PostOperationCallback = FsFilterCallbacks->PostAcquireForModifiedPageWriter;
578  }
579 
580  break;
581 
582  case FS_FILTER_RELEASE_FOR_MOD_WRITE:
583 
584  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
585  PreReleaseForModifiedPageWriter )) {
586 
587  *PreOperationCallback = FsFilterCallbacks->PreReleaseForModifiedPageWriter;
588  }
589 
590  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
591  PostReleaseForModifiedPageWriter )) {
592 
593  *PostOperationCallback = FsFilterCallbacks->PostReleaseForModifiedPageWriter;
594  }
595 
596  break;
597 
598  case FS_FILTER_ACQUIRE_FOR_CC_FLUSH:
599 
600  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
601  PreAcquireForCcFlush )) {
602 
603  *PreOperationCallback = FsFilterCallbacks->PreAcquireForCcFlush;
604  }
605 
606  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
607  PostAcquireForCcFlush )) {
608 
609  *PostOperationCallback = FsFilterCallbacks->PostAcquireForCcFlush;
610  }
611 
612  break;
613 
614  case FS_FILTER_RELEASE_FOR_CC_FLUSH:
615 
616  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
617  PreReleaseForCcFlush )) {
618 
619  *PreOperationCallback = FsFilterCallbacks->PreReleaseForCcFlush;
620  }
621 
622  if (VALID_FS_FILTER_CALLBACK_HANDLER( FsFilterCallbacks,
623  PostReleaseForCcFlush )) {
624 
625  *PostOperationCallback = FsFilterCallbacks->PostReleaseForCcFlush;
626  }
627 
628  break;
629 
630  default:
631 
632  ASSERT( FALSE );
633  *PreOperationCallback = NULL;
634  *PostOperationCallback = NULL;
635  }
636 }

638 NTSTATUS
639 FsFilterPerformCallbacks (
640  IN PFS_FILTER_CTRL FsFilterCtrl,
641  IN BOOLEAN AllowFilterToFail,
642  IN BOOLEAN AllowBaseFsToFail,
643  OUT BOOLEAN *BaseFsFailedOp
644  )
645 
646 /*++
647 
648 Routine Description:
649 
650  This routine calls all the file system filters that have registered
651  to see the operation described by the FsFilterCtrl. If this
652  routine returns a successful Status, the operation should be
653  passed onto the base file system. If an error Status is returned,
654  the caller is responsible for call FsFilterPerformCompletionCallbacks
655  to unwind any post-operations that need to be called.
656 
657 Arguments:
658 
659  FsFilterCtrl - The structure describing the control information
660  needed to pass this operation to each filter registered to
661  see this operation.
662 
663  AllowFilterToFail - TRUE if the filter is allowed to fail this
664  operation, FALSE otherwise.
665 
666  AllowBaseFsToFail - TRUE if the base file system is allowed to fail this
667  operation, FALSE otherwise.
668 
669  BaseFsFailedOp - Set to TRUE if the base file system failed
670  this operation, FALSE, otherwise.
671 
672 Return Value:
673 
674  STATUS_SUCCESS - All filters that are interested saw the operation
675  and none failed this operation.
676 
677  STATUS_INSUFFICIENT_RESOURCES - There is not enough memory
678  to allocate the completion node, so this operation
679  is failing.
680 
681  Other error Status - Could be returned from a filter's
682  pre-operation callback if it wants to fail this operation.
683 */
684 
685 {
686  PFS_FILTER_CALLBACK_DATA Data = &(FsFilterCtrl->Data);
687  PFS_FILTER_COMPLETION_STACK CompletionStack = &(FsFilterCtrl->CompletionStack);
688  PFS_FILTER_CALLBACK PreOperationCallback;
689  PFS_FILTER_COMPLETION_CALLBACK PostOperationCallback;
690  PFS_FILTER_COMPLETION_NODE CompletionNode;
691  PDEVICE_OBJECT CurrentDeviceObject;
692  NTSTATUS Status = STATUS_SUCCESS;
693  BOOLEAN isFilter = TRUE;
694 
695  //
696  // We should never be in the scenario where a filter can fail the operation
697  // but the base file system cannot.
698  //
699 
700  ASSERT( !(AllowFilterToFail && !AllowBaseFsToFail) );
701 
702  //
703  // Initialize output parameters if present.
704  //
705 
706  *BaseFsFailedOp = FALSE;
707 
708  //
709  // As we iterate through the device objects, we use the local
710  // CurrentDeviceObject to iterate through the list because we want
711  // Data->DeviceObject to be set to the last device object when we are
712  // finished iterating.
713  //
714 
715  CurrentDeviceObject = Data->DeviceObject;
716 
717  while (CurrentDeviceObject != NULL) {
718 
719  //
720  // First remember if this device object represents a filter or a file
721  // system.
722  //
723 
724  if (CurrentDeviceObject->DeviceObjectExtension->AttachedTo != NULL) {
725 
726  isFilter = TRUE;
727 
728  } else {
729 
730  isFilter = FALSE;
731  }
732 
733  //
734  // Now get the callbacks for this device object
735  //
736 
737  Data->DeviceObject = CurrentDeviceObject;
738 
739  FsFilterGetCallbacks( Data->Operation,
740  Data->DeviceObject,
741  &PreOperationCallback,
742  &PostOperationCallback );
743 
744  //
745  // If this device object has either a callback or completion callback
746  // for this operation, allocate a CompletionNode for it.
747  //
748 
749  if ((PreOperationCallback == NULL) && (PostOperationCallback == NULL)) {
750 
751  //
752  // This device object does not have any callbacks for this operation
753  // so move onto the next device.
754  //
755 
756  CurrentDeviceObject = Data->DeviceObject->DeviceObjectExtension->AttachedTo;
757  CompletionNode = NULL;
758  continue;
759 
760  } else if (PostOperationCallback != NULL) {
761 
762  //
763  // Since there is a PostOperationCallback, we will need to allocate
764  // a CompletionNode for this device.
765  //
766 
767  CompletionNode = PUSH_COMPLETION_NODE( CompletionStack );
768 
769  if (CompletionNode == NULL) {
770 
771  //
772  // This case shouldn't happen since we should ensure
773  // that our completion stack is large enough when
774  // we first saw this operation.
775  //
776 
777  if (!AllowFilterToFail) {
778 
779  //
780  // We cannot fail this operation, so bugcheck.
781  //
782 
783  KeBugCheckEx( FILE_SYSTEM, 0, 0, 0, 0 );
784  }
785 
786  return STATUS_INSUFFICIENT_RESOURCES;
787 
788  } else {
789 
790  CompletionNode->DeviceObject = Data->DeviceObject;
791  CompletionNode->FileObject = Data->FileObject;
792  CompletionNode->CompletionContext = NULL;
793  CompletionNode->CompletionCallback = PostOperationCallback;
794  }
795 
796  } else {
797 
798  //
799  // We just have a pre-operation, so just set the CompletionNode to
800  // NULL.
801  //
802 
803  CompletionNode = NULL;
804  }
805 
806  if (PreOperationCallback != NULL) {
807 
808  if (CompletionNode == NULL) {
809 
810  Status = PreOperationCallback( Data,
811  NULL );
812 
813  } else {
814 
815  Status = PreOperationCallback( Data,
816  &(CompletionNode->CompletionContext) );
817  }
818 
819  ASSERT( Status == STATUS_SUCCESS ||
820  Status == STATUS_FSFILTER_OP_COMPLETED_SUCCESSFULLY ||
821  !NT_SUCCESS( Status ) );
822 
823  if (Status == STATUS_FSFILTER_OP_COMPLETED_SUCCESSFULLY) {
824 
825  //
826  // The filter/file system completed the operation successfully, so just
827  // return the completion status.
828  //
829 
830  //
831  // If we allocated a completion node for this device,
832  // pop it now since we don't want to call it when we
833  // process the completions.
834  //
835 
836  if (CompletionNode != NULL) {
837 
838  POP_COMPLETION_NODE( CompletionStack );
839  }
840 
841  return Status;
842 
843  } else if (!NT_SUCCESS( Status )) {
844 
845  //

1010 
1011 /*++
1012 
1013 Routine Description:
1014 
1015  This routine creates a section for given a file object. It returns the
1016  section object, the handle which references the section object, and
1017  (optionally) the size of the file when it was mapped if the section was
1018  able to be created successfully.
1019 
1020  This routine functions like NtCreateSection except that it takes a
1021  file object as the parameter to describe which stream to use to back the
1022  file. It properly synchronizes the section creation with other file system
1023  operations.
1024 
1025  NOTE: At this point, this routine can only be used to create a DATA section.
1026  In the future, this routine may change to support that functionality.
1027 
1028 Arguments:
1029 
1030  SectionHandle - Set to the handle created to represent the section returned.
1031 
1032  SectionObject - Set to the section object created.
1033 
1034  SectionFileSize - If non-NULL, set to the file's size at the time the
1035  section was created.
1036 
1037  FileObject - The file object representing the stream to back this section.
1038 
1039  DesiredAccess - The desired types of access for the section.
1040 
1041  ObjectAttributes - Supplies a pointer to an object attributes structure.
1042 
1043  MaximumSize - Supplies the maximum size of the section in bytes.
1044  This value is rounded up to the host page size and
1045  specifies the size of the section (page file
1046  backed section) or the maximum size to which a
1047  file can be extended or mapped (file backed
1048  section).
1049  NOTE: This value is ignored for now since we will be passing a file object
1050  to MmCreateSection (Mm will query the file system itself to find the size).
1051 
1052  SectionPageProtection - Supplies the protection to place on each page
1053  in the section. One of PAGE_READ, PAGE_READWRITE,
1054  PAGE_EXECUTE, or PAGE_WRITECOPY and, optionally,
1055  PAGE_NOCACHE or PAGE_WRITECOMBINE
1056  may be specified.
1057 
1058  AllocationAttributes - Supplies a set of flags that describe the
1059  allocation attributes of the section.
1060  NOTE: This routine only supports a subset of AllocationAttributes. For
1061  the complete list of AllocationAttributes, see NtCreateSection.
1062 
1063  AllocationAttributes Flags
1064 
1065  SEC_COMMIT - All pages of the section are set to the commit state.
1066 
1067  SEC_FILE - The file specified by the file handle is a mapped
1068  file. If a file handle is supplied and neither
1069  SEC_IMAGE or SEC_FILE is supplied, SEC_FILE is
1070  assumed.
1071 
1072  Flags - Reserved for future use.
1073 
1074 Return Value:
1075 
1076  Returns STATUS_SUCCESS if the section could be created, or the appropriate
1077  error otherwise.
1078 
1079  If this routine is successful, the caller must close the SectionHandle
1080  and dereference the SectionObject when it is done using these values.
1081 
1082 --*/
1083 
1084 {
1085  NTSTATUS Status;
1086  LARGE_INTEGER FileSize;
1087  PVOID Object = NULL;
1088  HANDLE Handle = NULL;
1089  ULONG RetryCount = 0;
1090 
1091  UNREFERENCED_PARAMETER( MaximumSize );
1092  UNREFERENCED_PARAMETER( Flags );
1093 
1094  //
1095  // At this point, there are some options which this API does not support.
1096  // If those options are passed in, FAIL the operation here.
1097  //
1098  // For SectionPageProtection, one of these flags must be set:
1099  // PAGE_READONLY
1100  // PAGE_READWRITE
1101  //
1102  // For AllocationAttributes, only these flag combinations are allowed:
1103  // SEC_COMMIT must be set
1104  // SEC_FILE is allowed to be set (assumed otherwise)
1105  //
1106 
1107  if (FlagOn(SectionPageProtection, ~(PAGE_READONLY | PAGE_READWRITE)) ||
1108  (SectionPageProtection == 0)) {
1109 
1110  return STATUS_INVALID_PARAMETER_8;
1111  }
1112 
1113  if ((!FlagOn( AllocationAttributes, SEC_COMMIT )) ||

1115 
1116  return STATUS_INVALID_PARAMETER_9;
1117  }

1121  // size. Note that it can't change while we have its resource exclusive.
1122  //

1124  // handle is passed in, which we don't have.
1125  //
