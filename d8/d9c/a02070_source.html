54 ULONG
55 CmpSecConvKey(
56  IN ULONG DescriptorLength,
57  IN PULONG Descriptor
58  )
59 /*++
60 
61 Routine Description:
62 
63  Computes the ConvKey for the given security descriptor.
64  The algorithm is taken from the NTFS security hash.
65  (it was proven to be efficient there; why shouldn't do the same ?)
66 
67 
68  For speed in the hash, we consider the security descriptor as an array
69  of ULONGs. The fragment at the end that is ignored should not affect
70  the collision nature of this hash.
71 
72 Arguments:
73 
74  DescriptorLength - length (in bytes) of the sd
75 
76  Descriptor - actual sd to cache
77 
78 Return Value:
79 
80  ConvKey
81 
82 Note:
83 
84  We may want to convert this to a macro
85 --*/
86 
87 {
88  ULONG Count;
89  ULONG Hash = 0;
90 
91  PAGED_CODE();
92 
93  Count = DescriptorLength / 4;
94 
95  while (Count--) {
96  Hash = ((Hash << 3) | (Hash >> (32-3))) + *Descriptor++;
97  }
98 
99  return Hash;
100 }

121 NTSTATUS
122 CmpAddSecurityCellToCache (
123  IN OUT PCMHIVE CmHive,
124  IN HCELL_INDEX SecurityCell,
125  IN BOOLEAN BuildUp,
126  IN PCM_KEY_SECURITY_CACHE SecurityCached
127  )
128 
129 /*++
130 
131 Routine Description:
132 
133  This routine adds the specified security cell to the cache of the
134  specified hive. It takes care of cache allocation (grow) as well.
135  At build up time, cache size grows with a PAGE_SIZE, to avoid memory
136  fragmentation. After the table is built, it's size is adjusted (most
137  of the hives never add new security cells). Then, at run-time, the size
138  grows with 16 entries at a time (same reason)
139  The cache is ordered by the cell's index, so we can do a binary search on
140  cells retrieval.
141 
142 Arguments:
143 
144  CmHive - the hive to which the security cell belongs
145 
146  SecurityCell - the security cell to be added to the cache
147 
148  BuildUp - specifies that this is build up time
149 
150  SecurityCached - if not NULL it means we have it already allocated (this happens when rebuilding the cache).
151 
152 Return Value:
153 
154  NTSTATUS - STATUS_SUCCESS if the operation is successful and an
155  appropriate error status otherwise (i.e STATUS_INSUFFICIENT_RESOURCES).
156 
157 Note:
158 
159  If the security cell is already IN the cache; this function will return TRUE.
160 --*/
161 {
162  ULONG Index;
163  PCM_KEY_SECURITY Security;
164 
165  PAGED_CODE();
166 
167  if( CmpFindSecurityCellCacheIndex (CmHive,SecurityCell,&Index) == TRUE ) {
168  //
169  // cell already exist in the cache; return;
170  //
171  return STATUS_SUCCESS;
172  }
173 
174  //
175  // if this fails, we're doomed !
176  //
177  ASSERT( (PAGE_SIZE % sizeof(CM_KEY_SECURITY_CACHE_ENTRY)) == 0 );
178 
179  //
180  // check if the cache can accommodate a new cell
181  //
182  if( CmHive->SecurityCount == CmHive->SecurityCacheSize ) {
183  //
184  // We're at the limit with the cache; we need to extend it by a page
185  //
186  // OBS: this takes care of the first allocation too, as SecurityCount
187  // and SecurityCacheSize are both initialized with 0
188  //
189  PCM_KEY_SECURITY_CACHE_ENTRY Temp;
190 
191  // store the actual buffer
192  Temp = CmHive->SecurityCache;
193 
194  //
195  // compute the new size and allocate a new buffer
196  //
197  if( BuildUp == TRUE ) {
198  //
199  // We are building up the cache; grow the table in page increments
200  //
201  ASSERT( ((CmHive->SecurityCacheSize * sizeof(CM_KEY_SECURITY_CACHE_ENTRY)) % PAGE_SIZE) == 0 );
202  CmHive->SecurityCacheSize += (PAGE_SIZE / sizeof(CM_KEY_SECURITY_CACHE_ENTRY));
203  } else {
204  //
205  // normal case (running time); a new security cell is added; grow the
206  // table with a fixed number of increments (to avoid fragmentation, in
207  // case of an Office install :-) )
208  //
209  CmHive->SecurityCacheSize += SECURITY_CACHE_GROW_INCREMENTS;
210 
211  }
212  CmRetryExAllocatePoolWithTag(PagedPool, CmHive->SecurityCacheSize * sizeof(CM_KEY_SECURITY_CACHE_ENTRY),
213  CM_SECCACHE_TAG|PROTECTED_POOL,CmHive->SecurityCache);
214  if( CmHive->SecurityCache == NULL ) {
215  //
216  // bad luck; bail out
217  //
218  CmHive->SecurityCache = Temp;
219  CmHive->SecurityCacheSize = CmHive->SecurityCount;
220  return STATUS_INSUFFICIENT_RESOURCES;
221  }
222  //
223  // copy existing data in the new location and free the old buffer
224  //
225  RtlCopyMemory(CmHive->SecurityCache,Temp,CmHive->SecurityCount*sizeof(CM_KEY_SECURITY_CACHE_ENTRY));
226  if( Temp != NULL ) {
227  ExFreePoolWithTag(Temp, CM_SECCACHE_TAG|PROTECTED_POOL );
228  } else {
229  ASSERT( CmHive->SecurityCount == 0 );
230  }
231  }
232 
233  //
234  // try first to get the security cell from the hive; if this fails, there is no point to go on
235  //
236  Security = (PCM_KEY_SECURITY)HvGetCell(&(CmHive->Hive),SecurityCell);
237  if( Security == NULL ){
238  //
239  // we failed to map the view containing this cell; bail out
240  //
241  return STATUS_INSUFFICIENT_RESOURCES;
242  }
243 
244  if( !SecurityCached ) {
245  ULONG Size;
246  //
247  // compute the size for the cached security structure
248  //
249  Size = FIELD_OFFSET(CM_KEY_SECURITY_CACHE,Descriptor) + Security->DescriptorLength;
250 
251  //
252  // think forward: allocate and initialize a copy for the security cell, in order to store it in the cache
253  //
254  CmRetryExAllocatePoolWithTag(PagedPool,Size,CM_SECCACHE_TAG|PROTECTED_POOL,SecurityCached);
255  if(SecurityCached == NULL) {
256  //
257  // bad luck; bail out
258  //
259  HvReleaseCell(&(CmHive->Hive),SecurityCell);
260  return STATUS_INSUFFICIENT_RESOURCES;
261  }
262  }
263  //
264  // from now on, nothing can go wrong !
265  //
266  RtlCopyMemory(&(SecurityCached->Descriptor),&(Security->Descriptor),Security->DescriptorLength);
267  SecurityCached->Cell = SecurityCell;
268  SecurityCached->DescriptorLength = Security->DescriptorLength;
269 
270  //
271  // now add this to the hash table
272  //
273  SecurityCached->ConvKey = CmpSecConvKey(Security->DescriptorLength,(PULONG)(&(Security->Descriptor)));
274  // add it to the end of the list with this conv key
275  InsertTailList( &(CmHive->SecurityHash[SecurityCached->ConvKey % CmpSecHashTableSize]),
276  &(SecurityCached->List)
277  );
278 
279  HvReleaseCell(&(CmHive->Hive),SecurityCell);
280 
281  //
282  // At this point we are sure we have space for at least one more entry
283  // Move data to make room for the new entry
284  //
285  if( Index < CmHive->SecurityCount ) {
286  //
287  // RtlMoveMemory will take care of the overlapping problem
288  //
289  RtlMoveMemory( ((PUCHAR)CmHive->SecurityCache) + (Index+1)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // destination
290  ((PUCHAR)CmHive->SecurityCache) + Index*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // source
291  (CmHive->SecurityCount - Index)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY) // size
292  );
293  }
294 
295  //
296  // setup the new entry
297  //
298  CmHive->SecurityCache[Index].Cell = SecurityCell;
299  CmHive->SecurityCache[Index].CachedSecurity = SecurityCached;
300 
301  // update the count
302  CmHive->SecurityCount++;
303 
304  return STATUS_SUCCESS;
305 }

307 BOOLEAN
308 CmpFindSecurityCellCacheIndex (
309  IN PCMHIVE CmHive,
310  IN HCELL_INDEX SecurityCell,
311  OUT PULONG Index
312  )
313 
314 /*++
315 
316 Routine Description:
317 
318  Search (binary) for the specified cellindex in the security cache.
319  Returns the index of the cache entry where the cell is cached or
320  it should be added
321 
322 Arguments:
323 
324  CmHive - the hive to which the security cell belongs
325 
326  SecurityCell - the security cell to search for
327 
328  Index - out param to pass the index at which the cell is (or it should be)
329 
330 Return Value:
331 
332  TRUE - the cell was found (at *Index)
333  FALSE - the cell is not in the cache (it should be added at *Index)
334 
335 --*/
336 {
337  ULONG High;
338  ULONG Low;
339  ULONG Current;
340  USHORT State = 0; // state of the operation: 0 - normal binary search
341  // 1 - last low
342  // 2 - last high
343  LONG Result;
344  LONG Tmp1,Tmp2;
345 
346  PAGED_CODE();
347 
348  if( CmHive->SecurityCount == 0 ) {
349  //
350  // there is no cell in the security cache
351  //
352  *Index = 0;
353  return FALSE;
354  }
355 
356  // sanity asserts
357  ASSERT( CmHive->SecurityCount <= CmHive->SecurityCacheSize );
358  ASSERT( CmHive->SecurityCache != NULL );
359 
360 
361  High = CmHive->SecurityCount - 1;
362  Low = 0;
363  if( (CmHive->SecurityHitHint >= 0) && ( (ULONG)CmHive->SecurityHitHint <= High) ) {
364  //
365  // try the last search
366  //
367  Current = CmHive->SecurityHitHint;
368  } else {
369  Current = High/2;
370  }
371 
372  // sign adjustment
373  Tmp1 = SecurityCell & ~HCELL_TYPE_MASK;
374  if( SecurityCell & HCELL_TYPE_MASK ) {
375  Tmp1 = -Tmp1;
376  }
377 
378  while( TRUE ) {
379 
380  Tmp2 = CmHive->SecurityCache[Current].Cell & ~HCELL_TYPE_MASK;
381  // sign adjustment
382  if( CmHive->SecurityCache[Current].Cell & HCELL_TYPE_MASK ) {
383  Tmp2 = -Tmp2;
384  }
385 
386  Result = Tmp1 - Tmp2;
387 
388  if (Result == 0) {
389  //
390  // Success, return data to caller and exit
391  //
392 
393  *Index = Current;
394  //
395  // we have a hit! update the count and exit
396  //
397  CmHive->SecurityHitHint = Current;
398  return TRUE;
399  }
400  //
401  // compute the next index to try
402  //
403  switch(State) {
404  case 0:
405  //
406  // normal binary search state
407  //
408  if( Result < 0 ) {
409  High = Current;
410  } else {
411  Low = Current;
412  }
413  if ((High - Low) <= 1) {
414  //
415  // advance to the new state
416  //
417  Current = Low;
418  State = 1;
419  } else {
420  Current = Low + ( (High-Low) / 2 );
421  }
422  break;
423  case 1:
424  //
425  // last low state
426  //
427 
428  // this should be true
429  ASSERT( Current == Low );
430  if (Result < 0) {
431  //
432  // does not exist, under
433  //
434 
435  *Index = Current;
436  return FALSE;
437  } else if( Low == High ) {
438  //
439  // low and high are identical; but current is bigger than them; insert after
440  //
441 
442  *Index = Current + 1;
443  return FALSE;
444  } else {
445  //
446  // advance to the new state; i.e. look at high
447  //
448  State = 2;
449  Current = High;
450  }
451 
452  break;
453  case 2:
454  //
455  // last high state; if we got here, High = Low +1 and Current == High
456  //
457  ASSERT( Current == High);
458  ASSERT( High == (Low + 1) );
459  if( Result < 0 ) {
460  //
461  // under High, but above Low; we should insert it here
462  //
463 
464  *Index = Current;
465  return FALSE;
466  } else {
467  //
468  // above High;
469  //
470 
471  *Index = Current + 1;
472  return FALSE;
473  }
474  break;
475  default:
476  ASSERT( FALSE );
477  break;
478  }
479  }
480 
481  //
482  // we shouldn't get here !!!
483  //
484  ASSERT( FALSE );
485  return FALSE;
486 }

489 CmpAdjustSecurityCacheSize (
490  IN PCMHIVE CmHive
491  )
492 
493 /*++
494 
495 Routine Description:
496 
497  Adjust the scusrity cache size for the specified hive. This function
498  should be called after all the security cells for the hive were cached,
499  in order to give back extra memory used in the process.
500 
501 
502 Arguments:
503 
504  CmHive - the hive to which the security cell belongs
505 
506 Return Value:
507 
508  TRUE - success
509  FALSE - failure - the size remains the same
510 
511 --*/
512 {
513  PCM_KEY_SECURITY_CACHE_ENTRY Buffer;
514 
515  PAGED_CODE();
516 
517  if( CmHive->SecurityCount < CmHive->SecurityCacheSize ) {
518  //
519  // cache size is bigger than what we need; there is a good chance
520  // nobody will ever add new security cells to this hive, so go on
521  // and free the extra space
522  //
523 
524  //
525  // allocate a new buffer with the exact size we need
526  //
527  CmRetryExAllocatePoolWithTag(PagedPool, CmHive->SecurityCount * sizeof(CM_KEY_SECURITY_CACHE_ENTRY),
528  CM_SECCACHE_TAG|PROTECTED_POOL,Buffer);
529 
530  if( Buffer == NULL ) {
531  //
532  // the system is low on resources; leave the cache as it is
533  //
534  return FALSE;
535  }
536 
537  //
538  // copy significant data inot the new buffer
539  //
540  RtlCopyMemory(Buffer,CmHive->SecurityCache,CmHive->SecurityCount*sizeof(CM_KEY_SECURITY_CACHE_ENTRY));
541 
542  //
543  // free the old buffer and update cache members
544  //
545  ExFreePoolWithTag(CmHive->SecurityCache, CM_SECCACHE_TAG|PROTECTED_POOL );
546 
547  CmHive->SecurityCache = Buffer;
548  CmHive->SecurityCacheSize = CmHive->SecurityCount;
549  }
550 
551  return TRUE;
552 }

554 VOID
555 CmpRemoveFromSecurityCache (
556  IN OUT PCMHIVE CmHive,
557  IN HCELL_INDEX SecurityCell
558  )
559 
560 /*++
561 
562 Routine Description:
563 
564  Removes the specified security cell from the security cache.
565  (only if present !)
566  For performance (and memory fragmentation) reasons, it does not
567  change (shrink) the cache size.
568 
569 Arguments:
570 
571  CmHive - the hive to which the security cell belongs
572 
573  SecurityCell - the security cell to be removed from the cache
574 
575 Return Value:
576 
577  <none>
578 --*/
579 {
580  ULONG Index;
581 
582  PAGED_CODE();
583 
584  if( CmpFindSecurityCellCacheIndex (CmHive,SecurityCell,&Index) == FALSE ) {
585  //
586  // cell is not in the cache
587  //
588  return;
589  }
590 
591  ASSERT( CmHive->SecurityCache[Index].Cell == SecurityCell );
592  ASSERT( CmHive->SecurityCache[Index].CachedSecurity->Cell == SecurityCell );
593 
594  //
595  // remove the cached structure from the hash
596  //
597  CmpRemoveEntryList(&(CmHive->SecurityCache[Index].CachedSecurity->List));
598 
599  //
600  // free up the cached security cell;
601  //
602  ExFreePoolWithTag(CmHive->SecurityCache[Index].CachedSecurity, CM_SECCACHE_TAG|PROTECTED_POOL );
603 
604  //
605  // move memory to reflect the new size, and update the cache count
606  //
607  RtlMoveMemory( ((PUCHAR)CmHive->SecurityCache) + Index*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // destination
608  ((PUCHAR)CmHive->SecurityCache) + (Index+1)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // source
609  (CmHive->SecurityCount - Index - 1)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY) // size
610  );
611 
612  CmHive->SecurityCount--;
613 }

652 PCM_KEY_SECURITY_CACHE
653 CmpFindReusableCellFromCache(IN PCMHIVE CmHive,
654  IN HCELL_INDEX SecurityCell,
655  IN ULONG PreviousCount)
656 /*++
657 
658 Routine Description:
659 
660  Attempts to find the smallest cell which can accommodate the current security cell.
661  Then moves it at the end and return a pointer to it. Shifts the array towards the end
662  as we are going to extend the cache
663 
664  Security doesn't change too often, so chances are that we will be able to reuse the cells 90%
665  of the time.
666 
667  If one cannot be found, the last cell in the array is freed. A new one will be allocated inside
668  CmpAddSecurityCellToCache
669 
670 Arguments:
671 
672  CmHive - the hive to which the security cell belongs
673 
674  SecurityCell - the security cell
675 
676  PreviousCount - the end of the array
677 
678 Return Value:
679 
680  the cached cell or NULL
681 --*/
682 {
683  ULONG Size;
684  PCM_KEY_SECURITY Security;
685  PCM_KEY_SECURITY_CACHE SecurityCached;
686  ULONG i;
687  ULONG SmallestSize = 0;
688  ULONG SmallestIndex = 0;
689 
690  PAGED_CODE();
691 
692  ASSERT_CM_LOCK_OWNED_EXCLUSIVE();
693 
694  //
695  // try first to get the security cell from the hive; if this fails, there is no point to go on
696  //
697  Security = (PCM_KEY_SECURITY)HvGetCell(&(CmHive->Hive),SecurityCell);
698  if( Security == NULL ){
699  //
700  // we failed to map the view containing this cell; bail out
701  //
702  goto ErrorExit;
703  }
704 
705  //
706  // compute the size for the cached security structure
707  //
708  Size = Security->DescriptorLength;
709  HvReleaseCell(&(CmHive->Hive),SecurityCell);
710 
711  //
712  // Now that we know the desired size, start iterating the array to find a suitable entry.
713  //
714  for(i = CmHive->SecurityCount; i < PreviousCount; i++) {
715  SecurityCached = CmHive->SecurityCache[i].CachedSecurity;
716  if( SecurityCached->DescriptorLength == Size ) {
717 Found:
718  //
719  // we have found one matching the exact size; move it at the end
720  // shift to the end one entry as we are going to extend the cache
721  //
722  // this factored down translates to:
723  //
724  RtlMoveMemory( ((PUCHAR)CmHive->SecurityCache) + (CmHive->SecurityCount+1)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // destination
725  ((PUCHAR)CmHive->SecurityCache) + CmHive->SecurityCount*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // source
726  (i - CmHive->SecurityCount)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY) // size
727  );
728 
729  return SecurityCached;
730  }
731 
732  //
733  // if not; record the smallest suitable entry
734  //
735  if( SecurityCached->DescriptorLength > Size ) {
736  if( (SmallestSize == 0) ||
737  (SmallestSize > SecurityCached->DescriptorLength)
738  ) {
739  //
740  // first one or this one is smaller
741  //
742  SmallestSize = SecurityCached->DescriptorLength;
743  SmallestIndex = i;
744  }
745  }
746  }
747 
748  if( SmallestSize != 0 ) {
749  SecurityCached = CmHive->SecurityCache[SmallestIndex].CachedSecurity;
750  ASSERT( SecurityCached->DescriptorLength == SmallestSize );
751  ASSERT( Size < SmallestSize );
752  i = SmallestIndex;
753  goto Found;
754  }
755 
756 ErrorExit:
757  ExFreePoolWithTag(CmHive->SecurityCache[PreviousCount - 1].CachedSecurity, CM_SECCACHE_TAG|PROTECTED_POOL );
758  //
759  // shift to the end one entry as we are going to extend the cache
760  //
761  RtlMoveMemory( ((PUCHAR)CmHive->SecurityCache) + (CmHive->SecurityCount+1)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // destination
762  ((PUCHAR)CmHive->SecurityCache) + CmHive->SecurityCount*sizeof(CM_KEY_SECURITY_CACHE_ENTRY), // source
763  (PreviousCount - CmHive->SecurityCount - 1)*sizeof(CM_KEY_SECURITY_CACHE_ENTRY) // size
764  );
765  return NULL;
766 }

768 BOOLEAN
769 CmpRebuildSecurityCache(
770  IN OUT PCMHIVE CmHive
771  )
772 /*++
773 
774 Routine Description:
775 
776  Rebuilds the security cache by reiterating all security cells
777  and adding them to the cache; this routine is intended for hive
778  refresh operations
779 
780 Arguments:
781 
782  CmHive - the hive to which the security cell belongs
783 
784 Return Value:
785 
786  TRUE or FALSE
787 --*/
788 {
789  PCM_KEY_NODE RootNode;

870  Result = FALSE;
871  goto JustReturn;
872  }
873  }

901 CmpFindMatchingDescriptorCell(
902  IN PCMHIVE CmHive,
903  IN PSECURITY_DESCRIPTOR SecurityDescriptor,

906  OUT OPTIONAL PCM_KEY_SECURITY_CACHE *CachedSecurityPointer
907  )

909 /*++
910 
911 Routine Description:
912 
913  This routine attempts to find a security descriptor in the hive that
914  is identical to the one passed in. If it finds one, it returns its
915  cell index.
916

918  Currently, this routine checks the security descriptors of the parent
919  and siblings of the node to find a match.
920

922  It looks for the sd in the security cache for this hive. This will
923  eliminate duplicates and make the search process faster.
924 
925 Arguments:
926 
927  CmHive - Supplies a pointer to the hive control structure for the node.
928  Needed to get access to the cache
929 
930  SecurityDescriptor - Supplies the cooked security descriptor which
931  should be searched for.
932 
933  Type - Indicates whether the Security Descriptor that matches must
934  be in Stable or Volatile store
935 
936  MatchingCell - Returns the cell index of a security cell whose
937  security descriptor is identical to SecurityDescriptor.
938  Valid only if TRUE is returned.
939 
940  CachedSecurityPointer - pointer to the cached security (for update reasons)
941 
942 Return Value:
943 
944  TRUE - Matching security descriptor found. MatchingCell returns the
945  cell index of the matching security descriptor.
946 
947  FALSE - No matching security descriptor found. MatchingCell is invalid.
948 
949 --*/
950 
951 {
952  ULONG DescriptorLength;
953  ULONG ConvKey;
954  PLIST_ENTRY ListAnchor;
955  PLIST_ENTRY Current;
956  PCM_KEY_SECURITY_CACHE CachedSecurity;
957 
958  PAGED_CODE();
959 
960  DescriptorLength = RtlLengthSecurityDescriptor(SecurityDescriptor);
961 
962  //
963  // calculate the conv key
964  //
965  ConvKey = CmpSecConvKey(DescriptorLength,(PULONG)SecurityDescriptor);
966 
967  ListAnchor = &(CmHive->SecurityHash[ConvKey % CmpSecHashTableSize]);
968  if( IsListEmpty(ListAnchor) == TRUE ) {
969  return FALSE;
970  }
971 
972  //
973  // iterate through the list of colisions for this convkey
974  // start with teh first element in list
975  //
976  Current = (PLIST_ENTRY)(ListAnchor->Flink);
977  while( Current != ListAnchor ){
978  //
979  // get the current cached security
980  //
981  CachedSecurity = CONTAINING_RECORD(Current,
982  CM_KEY_SECURITY_CACHE,
983  List);
984 
985  //
986  // see if it matches with the given descriptor;
987  //
988  if( (CachedSecurity->ConvKey == ConvKey) && // same convkey
989  (Type == HvGetCellType(CachedSecurity->Cell)) && // same cell type
990  (DescriptorLength == CachedSecurity->DescriptorLength) && // same length
991  (RtlEqualMemory(SecurityDescriptor, // and, finally, bit-wise identical
992  &(CachedSecurity->Descriptor),
993  DescriptorLength))
994  ) {
995  //
996  // we have found a match
997  //
998  *MatchingCell = CachedSecurity->Cell;
999  if (ARGUMENT_PRESENT(CachedSecurityPointer)) {
1000  *CachedSecurityPointer = CachedSecurity;
1001  }
1002  return TRUE;
1003  }
1004 
1005  //
1006  // advance to the next element
1007  //
1008  Current = (PLIST_ENTRY)(Current->Flink);
1009  }
1010 
1011  // sorry, no match
1012  return FALSE;
1013 }
