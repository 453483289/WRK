315 /*++
316 
317 Routine Description:
318 
319  This function returns the base of an image that contains the
320  specified PcValue. An image contains the PcValue if the PcValue
321  is within the ImageBase, and the ImageBase plus the size of the
322  virtual image.
323 
324 Arguments:
325 
326  PcValue - Supplies a PcValue.
327 
328  DataTableEntry - Supplies a pointer to a variable that receives the
329  address of the data table entry that describes the image.
330 
331  DriversOnly - Supplies TRUE if the kernel and HAL should be skipped.
332 
333  InKernelOrHal - Set to TRUE if the PcValue is in the kernel or the HAL.
334  This only has meaning if DriversOnly is FALSE.
335 
336 Return Value:
337 
338  NULL - No image was found that contains the PcValue.
339 
340  NON-NULL - Returns the base address of the image that contains the
341  PcValue.
342 
343 --*/
344 
345 {
346 
347  ULONG i;
348  PLIST_ENTRY ModuleListHead;
349  PLDR_DATA_TABLE_ENTRY Entry;
350  PLIST_ENTRY Next;
351  ULONG_PTR Bounds;
352  PVOID ReturnBase, Base;
353 
354  //
355  // If the module list has been initialized, then scan the list to
356  // locate the appropriate entry.
357  //
358 
359  if (KeLoaderBlock != NULL) {
360  ModuleListHead = &KeLoaderBlock->LoadOrderListHead;
361 
362  } else {
363  ModuleListHead = &PsLoadedModuleList;
364  }
365 
366  *InKernelOrHal = FALSE;
367  ReturnBase = NULL;
368  Next = ModuleListHead->Flink;
369  if (Next != NULL) {
370  i = 0;
371  while (Next != ModuleListHead) {
372  if (MmIsAddressValid(Next) == FALSE) {
373  return NULL;
374  }
375 
376  i += 1;
377  if ((i <= 2) && (DriversOnly == TRUE)) {
378  Next = Next->Flink;
379  continue;
380  }
381 
382  Entry = CONTAINING_RECORD(Next,
383  LDR_DATA_TABLE_ENTRY,
384  InLoadOrderLinks);
385 
386  Next = Next->Flink;
387  Base = Entry->DllBase;
388  Bounds = (ULONG_PTR)Base + Entry->SizeOfImage;
389  if ((ULONG_PTR)PcValue >= (ULONG_PTR)Base && (ULONG_PTR)PcValue < Bounds) {
390  *DataTableEntry = Entry;
391  ReturnBase = Base;
392  if (i <= 2) {
393  *InKernelOrHal = TRUE;
394  }
395 
396  break;
397  }
398  }
399  }
400 
401  return ReturnBase;
402 }

1476  callback routine so it can dump component specific information
1477  that may identify the cause of the bugcheck.
1478

1480  carefully. Bugcheck callback routines are called at HIGH_LEVEL

1482 
1483 Arguments:
1484 
1485  None.
1486 
1487 Return Value:
1488 
1489  None.
1490 
1491 --*/
1492 
1493 {
1494

1496  ULONG_PTR Checksum;
1497  ULONG Index;
1498  PLIST_ENTRY LastEntry;
1499  PLIST_ENTRY ListHead;
1500  PLIST_ENTRY NextEntry;

1502

1504  // If the bugcheck callback listhead is not initialized, then the
1505  // bugcheck has occured before the system has gotten far enough
1506  // in the initialization code to enable anyone to register a callback.
1507  //
1508 
1509  ListHead = &KeBugCheckCallbackListHead;
1510  if ((ListHead->Flink != NULL) && (ListHead->Blink != NULL)) {
1511 
1512  //
1513  // Scan the bugcheck callback list.
1514  //
1515 
1516  LastEntry = ListHead;
1517  NextEntry = ListHead->Flink;
1518  while (NextEntry != ListHead) {
1519

1524  //

1527  return;

1536  if (MmIsAddressValid((PVOID)Source) == FALSE) {
1537  return;
1538  }
1539 
1540  Source += 1;
1541  }
1542 
1543  if (CallbackRecord->Entry.Blink != LastEntry) {
1544  return;
1545  }
1546

1548  // If the callback record has a state of inserted and the
1549  // computed checksum matches the callback record checksum,
1550  // then call the specified bugcheck callback routine.
1551  //
1552 
1553  Checksum = (ULONG_PTR)CallbackRecord->CallbackRoutine;
1554  Checksum += (ULONG_PTR)CallbackRecord->Buffer;
1555  Checksum += CallbackRecord->Length;
1556  Checksum += (ULONG_PTR)CallbackRecord->Component;

1562  // handle any exceptions that occur.

1574  }
1575  }
1576  }

1579  NextEntry = NextEntry->Flink;
