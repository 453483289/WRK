433 PVOID
434 ExRegisterCallback (
435  __inout PCALLBACK_OBJECT CallbackObject,
436  __in PCALLBACK_FUNCTION CallbackFunction,
437  __in_opt PVOID CallbackContext
438  )
439 
440 /*++
441 
442 Routine Description:
443 
444  This routine allows a caller to register that it would like to have its
445  callback Function invoked when the callback notification call occurs.
446 
447 Arguments:
448 
449  CallbackObject - Supplies a pointer to a CallbackObject.
450 
451  CallbackFunction - Supplies a pointer to a function which is to
452  be executed when the Callback notification occurs.
453 
454  CallbackContext - Supplies a pointer to an arbitrary data structure
455  that will be passed to the function specified by the CallbackFunction
456  parameter.
457 
458 Return Value:
459 
460  Returns handle to callback registration.
461 
462 --*/
463 {
464  PCALLBACK_REGISTRATION CallbackRegistration;
465  BOOLEAN Inserted;
466  KIRQL OldIrql;
467 
468  ASSERT (CallbackFunction);
469  ASSERT (KeGetCurrentIrql() < DISPATCH_LEVEL);
470 
471  //
472  // Add reference to object
473  //
474 
475  ObReferenceObject (CallbackObject);
476 
477  //
478  // Begin by attempting to allocate storage for the CallbackRegistration.
479  // one cannot be allocated, return the error status.
480  //
481 
482  CallbackRegistration = ExAllocatePoolWithTag(
483  NonPagedPool,
484  sizeof( CALLBACK_REGISTRATION ),
485  'eRBC'
486  );
487 
488 
489  if( !CallbackRegistration ) {
490  ObDereferenceObject (CallbackObject);
491  return NULL;
492  }
493 
494 
495  //
496  // Initialize the callback packet
497  //
498 
499  CallbackRegistration->CallbackObject = CallbackObject;
500  CallbackRegistration->CallbackFunction = CallbackFunction;
501  CallbackRegistration->CallbackContext = CallbackContext;
502  CallbackRegistration->Busy = 0;
503  CallbackRegistration->UnregisterWaiting = FALSE;
504 
505 
506  Inserted = FALSE;
507  KeAcquireSpinLock (&CallbackObject->Lock, &OldIrql);
508 
509  if( CallbackObject->AllowMultipleCallbacks ||
510  IsListEmpty( &CallbackObject->RegisteredCallbacks ) ) {
511 
512  //
513  // add CallbackRegistration to tail
514  //
515 
516 
517  Inserted = TRUE;
518  InsertTailList( &CallbackObject->RegisteredCallbacks,
519  &CallbackRegistration->Link );
520  }
521 
522  KeReleaseSpinLock (&CallbackObject->Lock, OldIrql);
523 
524  if (!Inserted) {
525  ExFreePool (CallbackRegistration);
526 
527  ObDereferenceObject (CallbackObject);
528 
529  CallbackRegistration = NULL;
530  }
531 
532  return (PVOID) CallbackRegistration;
533 }

542 
543 Routine Description:
544 
545  This function removes the callback registration for the callbacks
546  from the list of callback object .
547 
548 Arguments:
549 
550  CallbackRegistration - Pointer to device object for the file system.

559  PCALLBACK_REGISTRATION CallbackRegistration;
560  PCALLBACK_OBJECT CallbackObject;
561  KIRQL OldIrql;
562 
563  ASSERT (KeGetCurrentIrql() < DISPATCH_LEVEL);
564 
565  CallbackRegistration = (PCALLBACK_REGISTRATION) CbRegistration;
566  CallbackObject = CallbackRegistration->CallbackObject;
567 
568  KeAcquireSpinLock (&CallbackObject->Lock, &OldIrql);
569 
570  //
571  // Wait for registration
572  //
573 
574  while (CallbackRegistration->Busy) {
575 
576  //
577  // Set waiting flag, then wait. (not performance critical - use
578  // single global event to wait for any and all unregister waits)
579  //
580 
581  CallbackRegistration->UnregisterWaiting = TRUE;
582  KeClearEvent (&ExpCallbackEvent);
583  KeReleaseSpinLock (&CallbackObject->Lock, OldIrql);
584 
585  KeWaitForSingleObject (
586  &ExpCallbackEvent,
587  Executive,
588  KernelMode,
589  FALSE,
590  NULL
591  );
592 
593  //
594  // Synchronize with callback object and recheck registration busy
595  //
596 
597  KeAcquireSpinLock (&CallbackObject->Lock, &OldIrql);
598  }
599 
600  //
601  // Registration not busy, remove it from the callback object
602  //
603 
604  RemoveEntryList (&CallbackRegistration->Link);
605  KeReleaseSpinLock (&CallbackObject->Lock, OldIrql);
606 
607  //
608  // Free memory used for CallbackRegistration
609  //
610 
611  ExFreePool (CallbackRegistration);
612 
613  //
614  // Remove reference count on CallbackObject
615  //
616 
617  ObDereferenceObject (CallbackObject);
618 }

627 /*++
628 
629 Routine Description:
630 
631  This function notifies all registered callbacks .
632 
633 Arguments:
634 
635  CallbackObject - supplies a pointer to the callback object should be
636  notified.
637 
638  SystemArgument1 - supplies a pointer will be passed to callback function.
639 
640  SystemArgument2 - supplies a pointer will be passed to callback function.
641 
642 Return Value:
643 
644  None.
645 
646 --*/
647 
648 {
649  PLIST_ENTRY Link;
650  PCALLBACK_REGISTRATION CallbackRegistration;
651  KIRQL OldIrql;
652 
653  if (CallbackObject == NULL || IsListEmpty (&CallbackObject->RegisteredCallbacks)) {
654  return ;
655  }
656 
657  //
658  // Synchronize with callback object
659  //
660 
661  KeAcquireSpinLock (&CallbackObject->Lock, &OldIrql);
662 
663  //
664  // call registered callbacks at callers IRQL level
665  // ( done if FIFO order of registration )
666  //
667 
668  if (OldIrql == DISPATCH_LEVEL) {
669 
670  //
671  // OldIrql is DISPATCH_LEVEL, just invoke all callbacks without
672  // releasing the lock
673  //
674 
675  for (Link = CallbackObject->RegisteredCallbacks.Flink;
676  Link != &CallbackObject->RegisteredCallbacks;
677  Link = Link->Flink) {
678 
679  //
680  // Get current registration to notify
681  //
682 
683  CallbackRegistration = CONTAINING_RECORD (Link,
684  CALLBACK_REGISTRATION,
685  Link);
686 
687  //
688  // Notify registration
689  //
690 
691  CallbackRegistration->CallbackFunction(
692  CallbackRegistration->CallbackContext,
693  Argument1,
694  Argument2
695  );
696 
697  } // next registration
698 
699  } else {
700 
701  //
702  // OldIrql is < DISPATCH_LEVEL, the code being called may be pageable
703  // and the callback object spinlock needs to be released around
704  // each registration callback.
705  //
706 
707  for (Link = CallbackObject->RegisteredCallbacks.Flink;
708  Link != &CallbackObject->RegisteredCallbacks;
709  Link = Link->Flink ) {
710 
711  //
712  // Get current registration to notify
713  //
714 
715  CallbackRegistration = CONTAINING_RECORD (Link,
716  CALLBACK_REGISTRATION,
717  Link);
718 
719  //
720  // If registration is being removed, don't bother calling it
721  //
722 
723  if (!CallbackRegistration->UnregisterWaiting) {
724 
725  //
726  // Set registration busy
727  //
728 
729  CallbackRegistration->Busy += 1;
730 
731  //
732  // Release SpinLock and notify this callback
733  //
734 
735  KeReleaseSpinLock (&CallbackObject->Lock, OldIrql);
736 
737  CallbackRegistration->CallbackFunction(
738  CallbackRegistration->CallbackContext,
739  Argument1,
740  Argument2
741  );
742 
743  //
744  // Synchronize with CallbackObject
745  //
746 
747  KeAcquireSpinLock (&CallbackObject->Lock, &OldIrql);
748 
749  //
750  // Remove our busy count
751  //
752 
753  CallbackRegistration->Busy -= 1;
754 
755  //
756  // If the registration removal is pending, kick global
757  // event let unregister continue
758  //
759 
760  if (CallbackRegistration->UnregisterWaiting &&
761  CallbackRegistration->Busy == 0) {
762  KeSetEvent (&ExpCallbackEvent, 0, FALSE);
763  }
764  }
765  }
766  }
767 
768 
769  //
770  // Release callback
771  //
772 
773  KeReleaseSpinLock (&CallbackObject->Lock, OldIrql);
774 }

887 BOOLEAN
888 ExCompareExchangeCallBack (
889  IN OUT PEX_CALLBACK CallBack,
890  IN PEX_CALLBACK_ROUTINE_BLOCK NewBlock,
891  IN PEX_CALLBACK_ROUTINE_BLOCK OldBlock
892  )
893 /*++
894 
895 Routine Description:
896 
897  This function assigns, removes or swaps a low overhead callback function.
898 
899 Arguments:
900 
901  CallBack - Callback structure to be modified
902 
903  NewBlock - New block to be installed in the callback
904 
905  OldBlock - The old block that must be there now to be replaced
906 
907 Return Value:
908 
909  BOOLEAN - TRUE: The swap occured, FALSE: The swap failed
910 
911 --*/
912 {
913  EX_FAST_REF OldRef;
914  PEX_CALLBACK_ROUTINE_BLOCK ReplacedBlock;
915 
916  if (NewBlock != NULL) {
917  //
918  // Add the additional references to the routine block
919  //
920  if (!ExAcquireRundownProtectionEx (&NewBlock->RundownProtect,
921  ExFastRefGetAdditionalReferenceCount () + 1)) {
922  ASSERTMSG ("Callback block is already undergoing rundown", FALSE);
923  return FALSE;
924  }
925  }
926 
927  //
928  // Attempt to replace the existing object and balance all the reference counts
929  //
930  OldRef = ExFastRefCompareSwapObject (&CallBack->RoutineBlock,
931  NewBlock,
932  OldBlock);
933 
934  ReplacedBlock = ExFastRefGetObject (OldRef);
935 
936  //
937  // See if the swap occured. If it didn't undo the original references we added.
938  // If it did then release remaining references on the original
939  //
940  if (ReplacedBlock == OldBlock) {
941  PKTHREAD CurrentThread;
942  //
943  // We need to flush out any slow referencers at this point. We do this by
944  // acquiring and releasing a lock.
945  //
946  if (ReplacedBlock != NULL) {
947  CurrentThread = KeGetCurrentThread ();
948 
949  KeEnterCriticalRegionThread (CurrentThread);
950 
951  ExAcquireReleasePushLockExclusive (&ExpCallBackFlush);
952 
953  KeLeaveCriticalRegionThread (CurrentThread);
954 
955  ExReleaseRundownProtectionEx (&ReplacedBlock->RundownProtect,
956  ExFastRefGetUnusedReferences (OldRef) + 1);
957 
958  }
959  return TRUE;
960  } else {
961  //
962  // The swap failed. Remove the addition references if we had added any.
963  //
964  if (NewBlock != NULL) {
965  ExReleaseRundownProtectionEx (&NewBlock->RundownProtect,
966  ExFastRefGetAdditionalReferenceCount () + 1);
