9 Module Name:
10 
11  balmgr.c
12 
13 Abstract:
14 
15  This module implements the NT balance set manager. Normally the kernel
16  does not contain "policy" code. However, the balance set manager needs
17  to be able to traverse the kernel data structures and, therefore, the
18  code has been located as logically part of the kernel.
19 
20  The balance set manager performs the following operations:
21 
22  1. Makes the kernel stack of threads that have been waiting for a
23  certain amount of time, nonresident.
24 
25  2. Removes processes from the balance set when memory gets tight
26  and brings processes back into the balance set when there is
27  more memory available.
28 
29  3. Makes the kernel stack resident for threads whose wait has been
30  completed, but whose stack is nonresident.
31 
32  4. Arbitrarily boosts the priority of a selected set of threads
33  to prevent priority inversion in variable priority levels.
34 
35  In general, the balance set manager only is active during periods when
36  memory is tight.
37 
38 --*/

148 
149 Routine Description:
150 
151  This function is the startup code for the balance set manager. The
152  balance set manager thread is created during system initialization
153  and begins execution in this function.
154 
155 Arguments:
156 
157  Context - Supplies a pointer to an arbitrary data structure (NULL).
158

160 
161  None.
162

333 /*++
334 
335 Routine Description:
336 
337  This thread controls the swapping of processes and kernel stacks. The
338  order of evaluation is:
339 
340  Outswap kernel stacks
341  Outswap processes
342  Inswap processes
343  Inswap kernel stacks
344 
345 Arguments:
346 
347  Context - Supplies a pointer to the routine context - not used.
348 
349 Return Value:
350 
351  None.
352 
353 --*/

356 
357  PSINGLE_LIST_ENTRY SwapEntry;
358 
359  UNREFERENCED_PARAMETER(Context);
360 
361  //
362  // Set address of swap thread object and raise the thread priority to
363  // the lowest realtime level + 7 (i.e., priority 23).
364  //
365 
366  KiSwappingThread = KeGetCurrentThread();
367  KeSetPriorityThread(KeGetCurrentThread(), LOW_REALTIME_PRIORITY + 7);
368 
369  //
370  // Loop for ever processing swap events.
371  //
372  // N.B. This is the ONLY thread that processes swap events.
373  //
374

376 
377  //
378  // Wait for a swap event to occur.
379  //

381  KeWaitForSingleObject(&KiSwapEvent,

383  KernelMode,
384  FALSE,
385  NULL);
386 
387  //
388  // The following events are processed one after the other. If
389  // another event of a particular type arrives after having
390  // processed the respective event type, them the swap event
391  // will have been set and the above wait will immediately be
392  // satisfied.
393  //

396  //

416  //
417

419  if (SwapEntry != NULL) {

421  }

424  // Check if there are any kernel stack in swap requests pending.
425  //
426

430  }

432  } while (TRUE);

439 VOID
440 KiAdjustIrpCredits (
441  VOID
442  )
443 
444 /*++
445 
446 Routine Description:
447 
448  This function adjusts the lookaside IRP float credits for two processors
449  during each one second scan interval. IRP credits are adjusted by using
450  a moving average of two processors. It is possible for the IRP credits
451  for a processor to become negative, but this condition will be self
452  correcting.
453 
454 Arguments:
455 
456  None.
457 
458 Return Value:
459 
460  None.
461 
462 --*/
463 
464 {
465 
466  LONG Average;
467  LONG Adjust;
468  ULONG Index;
469  ULONG Number;
470  PKPRCB Prcb;
471  LONG TotalAdjust;
472 
473  //
474  // Adjust IRP credits if there are two or more processors.
475  //
476 
477  Number = KeNumberProcessors;
478  if (Number > 1) {
479 
480  //
481  // Compute the target average value by averaging the IRP credits
482  // across all processors.
483  //
484 
485  Index = 0;
486  Average = 0;
487  do {
488  Average += KiProcessorBlock[Index]->LookasideIrpFloat;
489  Index += 1;
490  } while (Index < Number);
491 
492 
493  //
494  // Adjust IRP credits for processor 0..n-1.
495  //
496 
497  Average /= (LONG)Number;
498  Number -= 1;
499  Index = 0;
500  TotalAdjust = 0;
501  do {
502  Prcb = KiProcessorBlock[Index];
503  Adjust = Average - Prcb->LookasideIrpFloat;
504  if (Adjust != 0) {
505  InterlockedExchangeAdd(&Prcb->LookasideIrpFloat, Adjust);
506  TotalAdjust += Adjust;
507  }
508 
509  Index += 1;
510  } while (Index < Number);
511 
512  //
513  // Adjust IRP credit for last processor.
514  //
515 
516  if (TotalAdjust != 0) {
517  Prcb = KiProcessorBlock[Index];
518  InterlockedExchangeAdd(&Prcb->LookasideIrpFloat, -TotalAdjust);
519  }
520  }
521 
522  return;
523 }

532 /*++
533 
534 Routine Description:
535 
536  This function in swaps the kernel stack for threads whose wait has been

538 
539 Arguments:
540 
541  SwapEntry - Supplies a pointer to the first entry in the in swap list.

630 VOID
631 KiOutSwapKernelStacks (
632  VOID
633  )
634 
635 /*++
636 
637 Routine Description:
638 
639  This function attempts to out swap the kernel stack for threads whose
640  wait mode is user and which have been waiting longer than the stack
641  protect time.
642 
643 Arguments:
644 
645  None.
646 
647 Return Value:
648 
649  None.
650 
651 --*/
652 
653 {
654 
655  PLIST_ENTRY NextEntry;
656  ULONG NumberOfThreads;
657  KIRQL OldIrql;
658  PKPRCB Prcb;
659  PKPROCESS Process;
660  PKTHREAD Thread;
661  PKTHREAD ThreadObjects[MAXIMUM_THREAD_STACKS];
662  ULONG WaitLimit;
663 
664  //
665  // Scan the waiting in list and check if the wait time exceeds the
666  // stack protect time. If the protect time is exceeded, then make
667  // the kernel stack of the waiting thread nonresident. If the count
668  // of the number of stacks that are resident for the process reaches
669  // zero, then insert the process in the outswap list and set its state
670  // to transition.
671  //
672  // Raise IRQL and lock the dispatcher database.
673  //
674 
675  NumberOfThreads = 0;
676  Prcb = KiProcessorBlock[KiLastProcessor];
677  WaitLimit = KiQueryLowTickCount() - KiStackProtectTime;
678  KiLockDispatcherDatabase(&OldIrql);
679  NextEntry = Prcb->WaitListHead.Flink;
680  while ((NextEntry != &Prcb->WaitListHead) &&
681  (NumberOfThreads < MAXIMUM_THREAD_STACKS)) {
682 
683  Thread = CONTAINING_RECORD(NextEntry, KTHREAD, WaitListEntry);
684 
685  ASSERT(Thread->WaitMode == UserMode);
686 
687  NextEntry = NextEntry->Flink;
688 
689  //
690  // Threads are inserted at the end of the wait list in very nearly
691  // reverse time order, i.e., the longest waiting thread is at the
692  // beginning of the list followed by the next oldest, etc. Thus when
693  // a thread is encountered which still has a protected stack it is
694  // known that all further threads in the wait also have protected
695  // stacks, and therefore, the scan can be terminated.
696  //
697  // N.B. It is possible due to a race condition in wait that a high
698  // priority thread was placed in the wait list. If this occurs,
699  // then the thread is removed from the wait list without swapping
700  // the stack.
701  //
702 
703  if (WaitLimit < Thread->WaitTime) {
704  break;
705 
706  } else if (Thread->Priority >= (LOW_REALTIME_PRIORITY + 9)) {
707  RemoveEntryList(&Thread->WaitListEntry);
708  Thread->WaitListEntry.Flink = NULL;
709 
710  } else if (KiIsThreadNumericStateSaved(Thread)) {
711  Thread->KernelStackResident = FALSE;
712  ThreadObjects[NumberOfThreads] = Thread;
713  NumberOfThreads += 1;
714  RemoveEntryList(&Thread->WaitListEntry);
715  Thread->WaitListEntry.Flink = NULL;
716  Process = Thread->ApcState.Process;
717 
718  ASSERT(Process->StackCount != 0);
719 
720  ASSERT(Process->State == ProcessInMemory);
721 
722  Process->StackCount -= 1;
723  if (Process->StackCount == 0) {
724  Process->State = ProcessOutTransition;
725  InterlockedPushEntrySingleList(&KiProcessOutSwapListHead,
726  &Process->SwapListEntry);
727 
728  KiSwapEvent.Header.SignalState = 1;
729  }
730  }
731  }
732 
733  //
734  // Unlock the dispatcher database and lower IRQL to its previous
735  // value.
736  //
737 
738  KiUnlockDispatcherDatabase(OldIrql);
739 
740  //
741  // Increment the last processor number.
742  //
743 
744  KiLastProcessor += 1;
745  if (KiLastProcessor == (ULONG)KeNumberProcessors) {
746  KiLastProcessor = 0;
747  }
748 
749  //
750  // Out swap the kernel stack for the selected set of threads.
751  //
752 
753  while (NumberOfThreads > 0) {
754  NumberOfThreads -= 1;
755  Thread = ThreadObjects[NumberOfThreads];
756 
757  //
758  // Wait until the context has been swapped for the thread and outswap
759  // the thread stack.
760  //
761 
762  KeWaitForContextSwap(Thread);
763  MmOutPageKernelStack(Thread);
764  }
765 
766  return;
767 }

769 VOID
770 KiOutSwapProcesses (
771  IN PSINGLE_LIST_ENTRY SwapEntry
772  )
773 
774 /*++
775 
776 Routine Description:
777 
778  This function out swaps processes.
779 
780 Arguments:
781 
782  SwapEntry - Supplies a pointer to the first entry in the SLIST.
783 
784 Return Value:
785 
786  None.
787 
788 --*/
789 
790 {
791 
792  PLIST_ENTRY NextEntry;
793  KIRQL OldIrql;
794  PKPROCESS Process;
795  PKTHREAD Thread;
796 
797  //
798  // Process the process out swap list and for each process removed from
799  // the list, make the process nonresident, and process its ready list.
800  //
801 
802  do {
803  Process = CONTAINING_RECORD(SwapEntry, KPROCESS, SwapListEntry);
804  SwapEntry = SwapEntry->Next;
805 
806  //
807  // If there are any threads in the process ready list, then don't
808  // out swap the process and ready all threads in the process ready
809  // list. Otherwise, out swap the process.
810  //
811 
812  KiLockDispatcherDatabase(&OldIrql);
813  NextEntry = Process->ReadyListHead.Flink;
814  if (NextEntry != &Process->ReadyListHead) {
815  Process->State = ProcessInMemory;
816  while (NextEntry != &Process->ReadyListHead) {
817  Thread = CONTAINING_RECORD(NextEntry, KTHREAD, WaitListEntry);
818  RemoveEntryList(NextEntry);
819  Thread->ProcessReadyQueue = FALSE;
820  KiReadyThread(Thread);
821  NextEntry = Process->ReadyListHead.Flink;
822  }
823 
824  KiUnlockDispatcherDatabase(OldIrql);
825 
826  } else {
827  Process->State = ProcessOutSwap;
828  KiUnlockDispatcherDatabase(OldIrql);
829  MmOutSwapProcess(Process);
830 
831  //
832  // While the process was being outswapped there may have been one
833  // or more threads that attached to the process. If the process
834  // ready list is not empty, then in swap the process. Otherwise,
835  // mark the process as out of memory.
836  //
837 
838  KiLockDispatcherDatabase(&OldIrql);
839  NextEntry = Process->ReadyListHead.Flink;
840  if (NextEntry != &Process->ReadyListHead) {
841  Process->State = ProcessInTransition;
842  InterlockedPushEntrySingleList(&KiProcessInSwapListHead,
843  &Process->SwapListEntry);
844 
845  KiSwapEvent.Header.SignalState = 1;
846 
847  } else {
848  Process->State = ProcessOutOfMemory;
849  }
850 
851  KiUnlockDispatcherDatabase(OldIrql);
852  }
853 
854  } while (SwapEntry != NULL);
855 
856  return;
857 }

877 Arguments:
878 
879  Dpc - Supplies a pointer to a DPC object - not used.
880 
881  DeferredContext - Supplies the DPC context - not used.
882 
883  SystemArgument1 - Supplies the first system argument - note used.
884 
885  SystemArgument2 - Supplies the second system argument - note used.
886 
887 Return Value:
888 
889  None.
890 
891 --*/
892 
893 {
894 
895  ULONG Count = 0;
896  PLIST_ENTRY Entry;
897  ULONG Index;
898  PLIST_ENTRY ListHead;
899  ULONG Number = 0;
900  KIRQL OldIrql;
901  PKPRCB Prcb;
902  ULONG ScanIndex;
903  PULONG ScanLast;
904  ULONG Summary;
905  PKTHREAD Thread;
906  ULONG WaitLimit;
907

909  UNREFERENCED_PARAMETER(SystemArgument2);
910 
911  //
912  // Get the address of the queue index variable.
913  //
914  // N.B. If a fault occurs accessing queue index value, then the exception
915  // handler is either executed or a bugcheck occurs.
916  //
