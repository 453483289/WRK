53 //
54 // The flags in the security operational mode are defined
55 // as:
56 //
57 // PasswordProtected - Some level of authentication (such as
58 // a password) must be provided by users before they are
59 // allowed to use the system. Once set, this value will
60 // not be cleared without re-booting the system.
61 //
62 // IndividualAccounts - Each user must identify an account to
63 // logon to. This flag is only meaningful if the
64 // PasswordProtected flag is also set. If this flag is
65 // not set and the PasswordProtected flag is set, then all
66 // users may logon to the same account. Once set, this value
67 // will not be cleared without re-booting the system.
68 //
69 // MandatoryAccess - Indicates the system is running in a mandatory
70 // access control mode (e.g., B-level as defined by the U.S.A's
71 // Department of Defense's "Orange Book"). This is not utilized
72 // in the current release of NT. This flag is only meaningful
73 // if both the PasswordProtected and IndividualAccounts flags are
74 // set. Once set, this value will not be cleared without
75 // re-booting the system.
76 //
77 // LogFull - Indicates the system has been brought up in a mode in
78 // which if must perform security auditing, but its audit log
79 // is full. This may (should) restrict the operations that
80 // can occur until the audit log is made not-full again. THIS
81 // VALUE MAY BE CLEARED WHILE THE SYSTEM IS RUNNING (I.E., WITHOUT
82 // REBOOTING).
83 //
84 // If the PasswordProtected flag is not set, then the system is running
85 // without security, and user interface should be adjusted appropriately.
86 //
87

89 #define LSA_MODE_INDIVIDUAL_ACCOUNTS (0x00000002L)
90 #define LSA_MODE_MANDATORY_ACCESS (0x00000004L)
91 #define LSA_MODE_LOG_FULL (0x00000008L)
