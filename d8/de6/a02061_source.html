38 /*++
39 
40 Routine Description:
41 
42  Determines the space needed to store a given string in the registry. May apply
43  any relevant compression to compute the length, but the compression used is
44  guaranteed to be the same as CmpCopyName.
45 
46 Arguments:
47 
48  Hive - supplies the hive control structure (for version checking)
49 
50  Name - Supplies the unicode string to be copied into the registry.
51 
52 Return Value:
53 
54  The number of bytes of storage required to store this name.
55 
56 --*/
57 
58 {
59  ULONG i;
60 
61  if (Hive->Version == 1) {
62  return(Name->Length);
63  }
64  for (i=0;i<Name->Length/sizeof(WCHAR);i++) {
65  if ((USHORT)Name->Buffer[i] > (UCHAR)-1) {
66  return(Name->Length);
67  }
68  }
69  return(Name->Length / sizeof(WCHAR));
70 
71 }

80 /*++
81 
82 Routine Description:
83 
84  Copies the given unicode name into the registry, applying any relevant compression
85  at the same time.
86 
87 Arguments:
88 
89  Hive - supplies the hive control structure (For version checking)
90 
91  Destination - Supplies the destination of the given string.
92 
93  Source - Supplies the unicode string to copy into the registry.
94 
95 Return Value:
96 
97  Number of bytes of storage copied
98 
99 --*/
100 
101 {
102  ULONG i;
103 
104  if (Hive->Version==1) {
105  RtlCopyMemory(Destination,Source->Buffer, Source->Length);
106  return(Source->Length);
107  }
108 
109  for (i=0;i<Source->Length/sizeof(WCHAR);i++) {
110  if ((USHORT)Source->Buffer[i] > (UCHAR)-1) {
111  RtlCopyMemory(Destination,Source->Buffer, Source->Length);
112  return(Source->Length);
113  }
114  ((PUCHAR)Destination)[i] = (UCHAR)(Source->Buffer[i]);
115  }
116  return(Source->Length / sizeof(WCHAR));
117 }

203 /*++
204 
205 Routine Description:
206 
207  Compares a compressed registry string to a Unicode string. Does a case-insensitive
208  comparison.
209 
210 Arguments:
211 
212  SearchName - Supplies the Unicode string to be compared
213 
214  CompressedName - Supplies the compressed string to be compared
215 
216  NameLength - Supplies the length of the compressed string
217 
218 Return Value:
219 
220  0 = SearchName == CompressedName (of Cell)
221 
222  < 0 = SearchName < CompressedName
223 
224  > 0 = SearchName > CompressedName
225 
226 --*/
227 
228 {
229  WCHAR *s1;
230  UCHAR *s2;
231  USHORT n1, n2;
232  WCHAR c1;
233  WCHAR c2;
234  LONG cDiff;
235 
236  s1 = SearchName->Buffer;
237  s2 = (UCHAR *)CompressedName;
238  n1 = (USHORT )(SearchName->Length / sizeof(WCHAR));
239  n2 = (USHORT )(NameLength);
240  while (n1 && n2) {
241  c1 = *s1++;
242  c2 = (WCHAR)(*s2++);
243 
244  //
245  // there is a 2/3 chance they match without doing the uppercase comparison.
246  //
247  if( c1 != c2 ) {
248  c1 = (CompareFlags&CMP_SOURCE_UP)?c1:CmUpcaseUnicodeChar(c1);
249  c2 = (CompareFlags&CMP_DEST_UP)?c2:CmUpcaseUnicodeChar(c2);
250 
251  if ((cDiff = ((LONG)c1 - (LONG)c2)) != 0) {
252  return( cDiff );
253  }
254  }
255 
256  n1--;
257  n2--;
258  }
259 
260  return( n1 - n2 );
261 }
