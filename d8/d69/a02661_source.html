46 /*++
47 
48 Routine Description:
49 
50  Converts a security descriptor from self-relative format to absolute
51  format
52 
53 Arguments:
54 
55  SecurityDescriptor - Supplies a pointer to a security descriptor in
56  Self-Relative format
57 
58  AbsoluteSecurityDescriptor - A pointer to a buffer in which will be
59  placed the main body of the Absolute format security descriptor.
60 
61  Dacl - Supplies a pointer to a buffer that will contain the Dacl of the
62  output descriptor. This pointer will be referenced by, not copied
63  into, the output descriptor.
64 
65  DaclSize - Supplies the size of the buffer pointed to by Dacl. In case
66  of error, it will return the minimum size necessary to contain the
67  Dacl.
68 
69  Sacl - Supplies a pointer to a buffer that will contain the Sacl of the
70  output descriptor. This pointer will be referenced by, not copied
71  into, the output descriptor.
72 
73  SaclSize - Supplies the size of the buffer pointed to by Sacl. In case
74  of error, it will return the minimum size necessary to contain the
75  Sacl.
76 
77  Owner - Supplies a pointer to a buffer that will contain the Owner of
78  the output descriptor. This pointer will be referenced by, not
79  copied into, the output descriptor.
80 
81  OwnerSize - Supplies the size of the buffer pointed to by Owner. In
82  case of error, it will return the minimum size necessary to contain
83  the Owner.
84 
85  PrimaryGroup - Supplies a pointer to a buffer that will contain the
86  PrimaryGroup of the output descriptor. This pointer will be
87  referenced by, not copied into, the output descriptor.
88 
89  PrimaryGroupSize - Supplies the size of the buffer pointed to by
90  PrimaryGroup. In case of error, it will return the minimum size
91  necessary to contain the PrimaryGroup.
92 
93 
94 Return Value:
95 
96  STATUS_SUCCESS - Success
97 
98  STATUS_BUFFER_TOO_SMALL - One of the buffers passed was too small.
99 
100  STATUS_INVALID_OWNER - There was not a valid owner in the passed
101  security descriptor.
102 
103 --*/
104 
105 {
106 
107  ULONG NewDaclSize;
108  ULONG NewSaclSize;
109  ULONG NewBodySize;
110  ULONG NewOwnerSize;
111  ULONG NewGroupSize;
112  PSID NewOwner;
113  PSID NewGroup;
114  PACL NewDacl;
115  PACL NewSacl;
116 
117  //
118  // typecast security descriptors so we don't have to cast all over the place.
119  //
120 
121  PISECURITY_DESCRIPTOR OutSD = AbsoluteSecurityDescriptor;
122 
123  PISECURITY_DESCRIPTOR InSD =
124  (PISECURITY_DESCRIPTOR)SelfRelativeSecurityDescriptor;
125 
126  RTL_PAGED_CODE();
127 
128  if ( !RtlpAreControlBitsSet( InSD, SE_SELF_RELATIVE) ) {
129  return( STATUS_BAD_DESCRIPTOR_FORMAT );
130  }
131 
132  NewBodySize = sizeof(SECURITY_DESCRIPTOR);
133  RtlpQuerySecurityDescriptor(
134  InSD,
135  &NewOwner,
136  &NewOwnerSize,
137  &NewGroup,
138  &NewGroupSize,
139  &NewDacl,
140  &NewDaclSize,
141  &NewSacl,
142  &NewSaclSize
143  );
144 
145  if ( (OutSD == NULL) ||
146  (NewBodySize > *AbsoluteSecurityDescriptorSize) ||
147  (NewOwnerSize > *OwnerSize ) ||
148  (NewDaclSize > *DaclSize ) ||
149  (NewSaclSize > *SaclSize ) ||
150  (NewGroupSize > *PrimaryGroupSize ) ) {
151 
152  *AbsoluteSecurityDescriptorSize = sizeof(SECURITY_DESCRIPTOR);
153  *PrimaryGroupSize = NewGroupSize;
154  *OwnerSize = NewOwnerSize;
155  *SaclSize = NewSaclSize;
156  *DaclSize = NewDaclSize;
157  return( STATUS_BUFFER_TOO_SMALL );
158  }
159 
160  RtlMoveMemory( OutSD,
161  InSD,
162  sizeof(SECURITY_DESCRIPTOR_RELATIVE) );
163 
164  OutSD->Owner = NULL;
165  OutSD->Group = NULL;
166  OutSD->Sacl = NULL;
167  OutSD->Dacl = NULL;
168  RtlpClearControlBits( OutSD, SE_SELF_RELATIVE );
169  if (NewOwner != NULL) {
170  RtlMoveMemory( Owner, NewOwner, SeLengthSid( NewOwner ));
171  OutSD->Owner = Owner;
172  }
173 
174  if (NewGroup != NULL) {
175  RtlMoveMemory( PrimaryGroup, NewGroup, SeLengthSid( NewGroup ));
176  OutSD->Group = PrimaryGroup;
177  }
178 
179  if (NewSacl != NULL) {
180  RtlMoveMemory( Sacl, NewSacl, NewSacl->AclSize );
181  OutSD->Sacl = Sacl;
182  }
183 
184  if (NewDacl != NULL) {
185  RtlMoveMemory( Dacl, NewDacl, NewDacl->AclSize );
186  OutSD->Dacl = Dacl;
187  }
188 
189  return( STATUS_SUCCESS );
190 }

193 RtlMakeSelfRelativeSD (
194  __in PSECURITY_DESCRIPTOR SecurityDescriptor,
195  __out_bcount(*BufferLength) PSECURITY_DESCRIPTOR SelfRelativeSecurityDescriptor,
196  __inout PULONG BufferLength
197  )
198 
199 /*++
200 
201 Routine Description:
202 
203  Makes a copy of a security descriptor. The produced copy will be in self-relative
204  form.
205 
206  The security descriptor to be copied may be in either absolute or self-relative
207  form.
208 
209 Arguments:
210 
211  SecurityDescriptor - Pointer to a security descriptor. This descriptor will not
212  be modified.
213 
214  SelfRelativeSecurityDescriptor - Pointer to a buffer that will contain
215  the returned self-relative security descriptor.
216 
217  BufferLength - Supplies the length of the buffer. If the supplied
218  buffer is not large enough to hold the self-relative security
219  descriptor, an error will be returned, and this field will return
220  the minimum size required.
221 
222 Return Value:
223 
224  STATUS_BUFFER_TOO_SMALL - The supplied buffer was too small to contain
225  the resultant security descriptor.
226 
227 --*/
228 
229 {
230 
231  ULONG NewDaclSize;
232  ULONG NewSaclSize;
233  ULONG NewOwnerSize;
234  ULONG NewGroupSize;
235  ULONG AllocationSize;
236  PSID NewOwner;
237  PSID NewGroup;
238  PACL NewDacl;
239  PACL NewSacl;
240  PCHAR Field;
241  PCHAR Base;
242 
243  //
244  // Convert security descriptors to new data type so we don't
245  // have to cast all over the place.
246  //
247 
248  PISECURITY_DESCRIPTOR_RELATIVE IResultantDescriptor =
249  (PISECURITY_DESCRIPTOR_RELATIVE)SelfRelativeSecurityDescriptor;
250 
251  PISECURITY_DESCRIPTOR IPassedSecurityDescriptor =
252  (PISECURITY_DESCRIPTOR)SecurityDescriptor;
253 
254  RtlpQuerySecurityDescriptor(
255  IPassedSecurityDescriptor,
256  &NewOwner,
257  &NewOwnerSize,
258  &NewGroup,
259  &NewGroupSize,
260  &NewDacl,
261  &NewDaclSize,
262  &NewSacl,
263  &NewSaclSize
264  );
265 
266  RTL_PAGED_CODE();
267 
268  AllocationSize = sizeof(SECURITY_DESCRIPTOR_RELATIVE) +
269  NewOwnerSize +
270  NewGroupSize +
271  NewDaclSize +
272  NewSaclSize ;
273 
274  if (AllocationSize > *BufferLength) {
275  *BufferLength = AllocationSize;
276  return( STATUS_BUFFER_TOO_SMALL );
277  }
278 
279  RtlZeroMemory( IResultantDescriptor, AllocationSize );
280  RtlCopyMemory( IResultantDescriptor,
281  IPassedSecurityDescriptor,
282  FIELD_OFFSET( SECURITY_DESCRIPTOR_RELATIVE, Owner ));
283 
284  Base = (PCHAR)(IResultantDescriptor);
285  Field = Base + (ULONG)sizeof(SECURITY_DESCRIPTOR_RELATIVE);
286  if (NewSaclSize > 0) {
287  RtlCopyMemory( Field, NewSacl, NewSaclSize );
288  IResultantDescriptor->Sacl = RtlPointerToOffset(Base, Field);
289  Field += NewSaclSize;
290 
291  } else {
292  IResultantDescriptor->Sacl = 0;
293  }
294 
295  if (NewDaclSize > 0) {
296  RtlCopyMemory( Field, NewDacl, NewDaclSize );
297  IResultantDescriptor->Dacl = RtlPointerToOffset(Base, Field);
298  Field += NewDaclSize;
299 
300  } else {
301  IResultantDescriptor->Dacl = 0;
302  }
303 
304  if (NewOwnerSize > 0) {
305  RtlCopyMemory( Field, NewOwner, NewOwnerSize );
306  IResultantDescriptor->Owner = RtlPointerToOffset(Base, Field);
307  Field += NewOwnerSize;
308  }
309 
310  if (NewGroupSize > 0) {
311  RtlCopyMemory( Field, NewGroup, NewGroupSize );
312  IResultantDescriptor->Group = RtlPointerToOffset(Base, Field);
313  }
314 
315  RtlpSetControlBits(IResultantDescriptor, SE_SELF_RELATIVE);
316  return( STATUS_SUCCESS );
317 }

385 RtlpQuerySecurityDescriptor (

397 /*++
398 
399 Routine Description:
400 
401  Returns the pieces of a security descriptor structure.
402 
403 Arguments:
404 
405 
406  SecurityDescriptor - Provides the security descriptor of interest.
407 
408  Owner - Returns a pointer to the owner information contained in the
409  security descriptor.
410 
411  OwnerSize - Returns the size of the owner information.
412 
413  PrimaryGroup - Returns a pointer to the primary group information.
414 
415  PrimaryGroupSize - Returns the size of the primary group information.
416 
417  Dacl - Returns a pointer to the Dacl.
418 
419  DaclSize - Returns the size of the Dacl.
420 
421  Sacl - Returns a pointer to the Sacl.
422 
423  SaclSize - Returns the size of the Sacl.
424 
425 Return Value:
426 
427  None.
428 
429 --*/
430 
431 {
432 
433  RTL_PAGED_CODE();
434 
435  *Owner = RtlpOwnerAddrSecurityDescriptor( SecurityDescriptor );
436  if (*Owner != NULL) {
437  *OwnerSize = LongAlignSize(SeLengthSid(*Owner));
438 
439  } else {
440  *OwnerSize = 0;
441  }
442 
443  *Dacl = RtlpDaclAddrSecurityDescriptor ( SecurityDescriptor );
444  if (*Dacl !=NULL) {
445  *DaclSize = LongAlignSize((*Dacl)->AclSize);
446 
447  } else {
448  *DaclSize = 0;
449  }
450 
451  *PrimaryGroup = RtlpGroupAddrSecurityDescriptor( SecurityDescriptor );
452  if (*PrimaryGroup != NULL) {
453  *PrimaryGroupSize = LongAlignSize(SeLengthSid(*PrimaryGroup));

455  } else {
456  *PrimaryGroupSize = 0;
457  }
458 
459  *Sacl = RtlpSaclAddrSecurityDescriptor( SecurityDescriptor );
460  if (*Sacl != NULL) {
461  *SaclSize = LongAlignSize((*Sacl)->AclSize);

463  } else {

468 NTSTATUS
469 RtlSelfRelativeToAbsoluteSD2 (
470  __inout_bcount(*pBufferSize) PSECURITY_DESCRIPTOR pSelfRelativeSecurityDescriptor,
471  __inout PULONG pBufferSize
472  )
473 
474 /*++
475 
476 Routine Description:
477 
478  Converts a security descriptor from self-relative format to absolute
479  format using the memory allocated for the SelfRelativeSecurityDescriptor
480 
481 Arguments:
482 
483  pSecurityDescriptor - Supplies a pointer to a security descriptor in
484  Self-Relative format. If success, we return a absolute security
485  descriptor where this pointer pointings.
486 
487  pBufferSize - Supplies a pointer to the size of the
488  buffer.
489 
490 Return Value:
491 
492  STATUS_SUCCESS - Success
493 
494  STATUS_BAD_DESCRIPTOR_FORMAT - The passed descriptor is not a self-relative
495  security descriptor.
496 
497  STATUS_BUFFER_TOO_SMALL - The passed buffer is too small.
498 
499  STATUS_INVALID_OWNER - There was not a valid owner in the passed
500  security descriptor.
501 
502 Notes: Despite some attempts to make this code as portable as possible and the
503  utilization of C_ASSERT or ASSERT to detect the respect of these assumptions,
504  this code is still making several assumptions about the format of the absolute
505  and self-relative descriptors and their relationships: in terms of packing,
506  fields definitions and locations in their respective structures.
507  In particular, this code assumes that the only differences are due to differences
508  in the types of the structure members and in the behavior of the security descriptor
509  query API.
510  At this time, the only structure members that get read/updated are Owner, Group,
511  Dacl and Sacl. If more members are added or displaced in the definitions of these
512  structures, this code may have to be modified.
513 
514 --*/
515 
516 {
517 
518  ULONG_PTR ptr;
519  PSID owner;
520  PSID group;
521  PACL dacl;
522  PACL sacl;
523  ULONG daclSize;
524  ULONG saclSize;
525  ULONG newBodySize;
526  ULONG ownerSize;
527  ULONG groupSize;
528  ULONG newBufferSize;
529  LONG deltaSize;
530 
531  //
532  // Typecast security descriptors so we don't have to cast all over the place.
533  //
534 
535  PISECURITY_DESCRIPTOR psd = (PISECURITY_DESCRIPTOR) pSelfRelativeSecurityDescriptor;
536  PISECURITY_DESCRIPTOR_RELATIVE psdr = (PISECURITY_DESCRIPTOR_RELATIVE)pSelfRelativeSecurityDescriptor;
537 
538  //
539  // This code uses several assumptions about the absolute and self-relative formats of
540  // security descriptors and the way they are packing in memory.
541  // See Routine Description Notes.
542  //
543 
544  C_ASSERT( sizeof( SECURITY_DESCRIPTOR ) >= sizeof( SECURITY_DESCRIPTOR_RELATIVE ) );
545  C_ASSERT( sizeof( psd->Control ) == sizeof( psdr->Control ) );
546  C_ASSERT( FIELD_OFFSET( SECURITY_DESCRIPTOR, Control ) == FIELD_OFFSET( SECURITY_DESCRIPTOR_RELATIVE, Control ) );
547 
548  RTL_PAGED_CODE();
549 
550  //
551  // Parameters check point
552  //
553 
554  if ( psd == (PISECURITY_DESCRIPTOR)0 ) {
555  return( STATUS_INVALID_PARAMETER_1 );
556  }
557 
558  if ( pBufferSize == (PULONG)0 ) {
559  return( STATUS_INVALID_PARAMETER_2 );
560  }
561 
562  //
563  // If the passed security descriptor is not self-relative, we return
564  // an format error.
565  //
566 
567  if ( !RtlpAreControlBitsSet( psd, SE_SELF_RELATIVE) ) {
568  return( STATUS_BAD_DESCRIPTOR_FORMAT );
569  }
570 
571  //
572  // Update local variables by querying the self-relative descriptor.
573  //
574  // Note that the returned size values are long-aligned.
575  //
576 
577  RtlpQuerySecurityDescriptor(
578  psd,
579  &owner,
580  &ownerSize,
581  &group,
582  &groupSize,
583  &dacl,
584  &daclSize,
585  &sacl,
586  &saclSize
587  );
588 
589  //
590  // Identical formats check:
591  //
592  // Determine the delta in size between the two formats of security descriptors
593  //
594 
595  deltaSize = sizeof( SECURITY_DESCRIPTOR ) - sizeof( SECURITY_DESCRIPTOR_RELATIVE );
596 
597  //
598  // If identical format:
599  // - clear the SELF_RELATIVE flag
600  // - update absolute descriptor members
601  // - return SUCCESS.
602  //
603 
604  if ( deltaSize == 0 ) {
605 
606  RtlpClearControlBits( psd, SE_SELF_RELATIVE );
607 
608  //
609  // Only the following fields are updated.
610  //
611 
612  ASSERT( sizeof( psd->Owner ) == sizeof( psdr->Owner ) );
613  ASSERT( sizeof( psd->Group ) == sizeof( psdr->Group ) );
614  ASSERT( sizeof( psd->Sacl ) == sizeof( psdr->Sacl ) );
615  ASSERT( sizeof( psd->Dacl ) == sizeof( psdr->Dacl ) );
616 
617  psd->Owner = owner;
618  psd->Group = group;
619  psd->Sacl = sacl;
620  psd->Dacl = dacl;
621 
622  return( STATUS_SUCCESS );
623  }
624 
625  //
626  // Determine the required size for the absolute format:
627  //
628 
629 #define ULONG_PTR_SDEND( _Adr ) ( (ULONG_PTR)(_Adr) + (ULONG_PTR)(_Adr##Size) )
630 
631  ptr = owner > group ? ULONG_PTR_SDEND( owner ) : ULONG_PTR_SDEND( group );
632  ptr = ptr > (ULONG_PTR)dacl ? ptr : ULONG_PTR_SDEND( dacl );
633  ptr = ptr > (ULONG_PTR)sacl ? ptr : ULONG_PTR_SDEND( sacl );
634 
635  newBufferSize = sizeof( SECURITY_DESCRIPTOR );
636  if ( ptr ) {
637 
638 #define ULONG_ROUND_UP( x, y ) ((ULONG)(x) + ((y)-1) & ~((y)-1))
639 
640  newBufferSize += ULONG_ROUND_UP( (ULONG_PTR)ptr - (ULONG_PTR)(psdr + 1), sizeof(PVOID) );
641  }
642 
643  //
644  // If the specified buffer size is not big enough, let the caller know about
645  // the minimum size and return STATUS_BUFFER_TOO_SMALL.
646  //
647 
648  if ( newBufferSize > *pBufferSize ) {
649  *pBufferSize = newBufferSize;
650  return( STATUS_BUFFER_TOO_SMALL );
651  }
652 
653  //
654  // Update absolute security descriptor:
655  //
656 
657  //
658  // Move the members of self-relative security descriptor in their
659  // absolute format locations.
660  //
661 
662  if ( ptr ) {
663  RtlMoveMemory( (PVOID)(psd + 1), (PVOID)(psdr + 1), newBufferSize - sizeof( SECURITY_DESCRIPTOR) );
664  }
665 
666  //
667  // Clear the self-relative flag
668  //
669 
670  RtlpClearControlBits( psd, SE_SELF_RELATIVE );
671 
672  //
673  // Only the following fields are updated.
674  //
675 
676  psd->Owner = (PSID)( owner ? (ULONG_PTR)owner + deltaSize : 0 );
677  psd->Group = (PSID)( group ? (ULONG_PTR)group + deltaSize : 0 );
678  psd->Sacl = (PACL)( sacl ? (ULONG_PTR)sacl + deltaSize : 0 );
679  psd->Dacl = (PACL)( dacl ? (ULONG_PTR)dacl + deltaSize : 0 );
680  return( STATUS_SUCCESS );
681 
682 } // RtlSelfRelativeToAbsoluteSD2()
