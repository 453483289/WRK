156 // Token Object Body //
157 // //
159 
160 //
161 // Tokens have three parts:
162 //
163 // Fixed part of body,
164 // Variable part of body,
165 // Dynamic part (not in body).
166 //
167 // The fixed and variable parts are allocated in a single block of memory.
168 // The dynamic part is a separately allocated block of memory.
169 //
170 // The fixed part of the body contains the fixed length fields. These
171 // are defined in the TOKEN data type.
172 //
173 // The variable part of the body is variable in length and contains
174 // privileges and user/group SIDs. This part is variable in length
175 // between different token objects, but does not change once established
176 // for an individual token.
177 //
178 // The dynamic part is used to house default discretionary ACL information
179 // and the primary group ID.
180 //
181 // Pictorially, a token looks like:
182 //
183 // ============== +---------------+
184 // ^ | |
185 // | | |
186 // | | |
187 // | | DynamicPart o-----------+
188 // | |- - - - - - - -| |
189 // +-----o Privileges | |
190 // Token | |- - - - - - - -| |
191 // Body | +--o UserAndGroups | |
192 // | | | |- - - - - - - -| |
193 // | | +--o RestrictedSids| \|/
194 // | | | |- - - - - - - -| +---------------------+
195 // | | | | PrimaryGroup o------->| [Primary Group SID] |
196 // | | | |- - - - - - - -| | o |
197 // | | | | DefaultAcl o---+ | o |
198 // | | | |- - - - - - - -| | | o |
199 // | | | | o | | |- - - - - - - - - - -|
200 // | | | | o | +--->| [ Default Acl ] |
201 // v | | | o | | o |
202 // ==============| | |===============| | o |
203 // ^ | +->| SIDs Array | | o |
204 // | | | [User SID ] | +---------------------+
205 // | | | [Group SID ] |
206 // | | [Group SID ] |
207 // Variable | | [Rest. Sid ] |
208 // Part | | o |
209 // | |- - - - - - - -|
210 // | +---->| Privileges |
211 // | | Array |
212 // v | |
213 // ============== +---------------+
214 //
215 // WARNING: The positions of fields illustrated in this picture are not
216 // intented to reflect their actual or even relative positions
217 // within the real data structures. The exception to this is
218 // that THE USER SID IS THE FIRST SID IN THE UserAndGroups
219 // ARRAY.
220 //
221 
222 
223 //
224 // ! ! ! ! IMPORTANT ! ! ! !
225 //
226 // The access validation routines assume the SIDs are arranged
227 // in a particular order within the variable part of the token.
228 // Any changes to the order of the SIDs must be coordinated with
229 // corresponding changes to the access validation routines.
230 //
231 // ! ! ! ! ! ! ! ! ! ! !

339 //
340 // Where:
341 //
342 // TokenSource - Information provided by the executive component that
343 // requested the logon that the token represents.
344 //
345 //
346 // TokenId - Is an LUID value. Each token object has a uniquely
347 // assigned LUID.
348 //
349 //
350 // AuthenticationId - Is the LUID assigned by the domain controller for
351 // the logon session.
352 //
353 //
354 // ExpirationTime - Not yet supported in NT.
355 //
356 //
357 // ModifiedId - Is an LUID which is changed each time a modification is
358 // made to this token which changes the security semantics of the
359 // token. This includes enabling/disabling privileges and groups
360 // and changing default ACLs, et cetera. Any token which is a
361 // duplicate of this token will have the same ModifiedId (until
362 // one or the other is changed). This does not cover changes to
363 // non-security semantics fields, like TokenInUse.
364 //
365 //
366 // UserAndGroupCount - Indicates the number of user/group IDs in this token.
367 // This value must be at least 1. A value of 1 indicates a user
368 // ID with no supplemental group IDs. A value of 5 indicates a
369 // user ID and 4 supplemental group IDs.
370 //
371 // PrivilegeCount - Indicates how many privileges are included in
372 // this token. May be zero or larger.
373 //
374 // TokenType - Indicates which type of token this token object is.
375 //
376 // ImpersonationLevel - For TokenImpersonation type tokens, this field
377 // indicates the impersonation level. For TokenPrimary type tokens,
378 // this field is ignored.
379 //
380 // DynamicCharged - Indicates how much pool quota has been charged
381 // for the dynamic portion of this token.
382 //
383 // DynamicAvailable - Indicates how much of the charged quota is still
384 // available for use. This is modified when pool associated
385 // with the dynamic portion of the token is allocated or freed,
386 // such as when the default DACL or primary group is replaced.
387 //
388 //
389 // DefaultOwnerIndex - If non-zero, identifies an ID that has explicitly
390 // been established as the default owner for this token. If it is zero,
391 // the standard default (user ID) is used as the default owner.
392 //
393 // UserAndGroups - Points to an array of SID_AND_ATTRIBUTES. The first
394 // element in this array is the token's user ID. Any additional
395 // elements are those of groups. The number of entries in this
396 // array is one greater than
397 //
398 // PrimaryGroup - Points to an SID that is to be used as the primary
399 // group of the token. There are no value restrictions
400 // placed upon what can be used as a primary group. This
401 // SID is not one of user or group IDs (although it may have the
402 // same value as one of those IDs).
403 //
404 // Privileges - Points to an array of privileges represented as
405 // LUID_AND_ATTRIBUTES. The number of elements in this array
406 // is contained in the PrivilegesCount field.
407 //
408 // TokenInUse - Is a boolean that indicates whether a primary token
409 // is already in use by a process. This field value is only
410 // valid for primary tokens.
411 //
412 // ProxyData - Optionally points to a Proxy data structure, containing
413 // the information to be passed to AVR routines by file systems.
414 // This field being non-null identifies the token as a proxy token.
415 //
416 // AuditData - Optionally points to an Audit data structure, containing
417 // global auditing data for this subject.
418 //
419 // NOTE: Access to this field is guarded by the global
420 // PROCESS SECURITY FIELDS LOCK.
421 // VariablePart - Is the beginning of the variable part of the token.
422 //

726 NTSTATUS
727 FORCEINLINE
728 SepDuplicateLogonSessionReference(
729  IN PTOKEN NewToken,
730  IN PTOKEN ExistingToken
731  )
732 {
733  PSEP_LOGON_SESSION_REFERENCES LogonSession;
734  LONG NewRef;
735  NTSTATUS Status;
736 
737  //
738  // Obtain the logon session reference. If the existing token
739  // has a reference then use that to obtain a new one. Otherwise
740  // Look up the session the slow way.
741  //
742  if ((ExistingToken->TokenFlags & TOKEN_SESSION_NOT_REFERENCED) == 0) {
743  LogonSession = ExistingToken->LogonSession;
744  NewToken->LogonSession = LogonSession;
745  NewRef = InterlockedIncrement (&LogonSession->ReferenceCount);
746  ASSERT (NewRef > 1);
