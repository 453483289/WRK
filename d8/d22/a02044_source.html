112  PCMHIVE CmHive,

115 /*++
116 
117 Routine Description:
118 
119  Check consistency of the registry within a given hive. Start from
120  root, and check that:
121  . Each child key points back to its parent.
122  . All allocated cells are referred to exactly once
123  (requires looking inside the hive structure...)
124  [This also detects space leaks.]
125  . All allocated cells are reachable from the root.
126 
127  NOTE: Exactly 1 ref rule may change with security.
128 
129 Arguments:
130 
131  CmHive - supplies a pointer to the CM hive control structure for the
132  hive of interest.
133 
134  Clean - if TRUE, references to volatile cells will be zapped
135  (done at startup only to avoid groveling hives twice.)
136  if FALSE, nothing will be changed.
137 
138  HiveCheck - If TRUE, performs hive consistency check too (i.e. checks
139  the bins)
140 
141 Return Value:
142 
143  0 if Hive is OK. Error return indicator if not.
144 
145  RANGE: 3000 - 3999
146 
147 --*/
148 {
149  PHHIVE Hive;
150  ULONG rc = 0;
151  ULONG Storage;
152  PRELEASE_CELL_ROUTINE ReleaseCellRoutine;
153  BOOLEAN ResetSD = FALSE;
154 
155  if (CmHive == CmpMasterHive) {
156  return(0);
157  }
158 
159  CmCheckRegistryDebug.Hive = (PHHIVE)CmHive;
160  CmCheckRegistryDebug.Status = 0;
161 
162 
163  //
164  // check the underlying hive and get storage use
165  //
166  Hive = &CmHive->Hive;
167 
168  if( Flags & CM_CHECK_REGISTRY_HIVE_CHECK ) {
169  rc = HvCheckHive(Hive, &Storage);
170  if (rc != 0) {
171  CmCheckRegistryDebug.Status = rc;
172  return rc;
173  }
174  }
175 
176  //
177  // Store the release cell procedure so we can restore at the end;
178  // Set it to NULL so we don't count : this saves us some pain during the check
179  //
180  ReleaseCellRoutine = Hive->ReleaseCellRoutine;
181  Hive->ReleaseCellRoutine = NULL;
182 
183  //
184  // Validate all the security descriptors in the hive
185  //
186  if (!CmpValidateHiveSecurityDescriptors(Hive,&ResetSD)) {
187  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"CmCheckRegistry:"));
188  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL," CmpValidateHiveSecurityDescriptors failed\n"));
189  rc = 3040;
190  CmCheckRegistryDebug.Status = rc;
191  Hive->ReleaseCellRoutine = ReleaseCellRoutine;
192  return rc;
193  }
194 
195  rc = CmpCheckRegistry2((PHHIVE)CmHive,Flags,Hive->BaseBlock->RootCell, HCELL_NIL,ResetSD);
196 
197  //
198  // Print a bit of a summary (make sure this data avail in all error cases)
199  //
200  if (rc > 0) {
201  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"CmCheckRegistry Failed (%d): CmHive:%p\n", rc, CmHive));
202  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL," Hive:%p Root:%08lx\n", Hive, Hive->BaseBlock->RootCell));
203  }
204 
205  //
206  // restore the release cell routine
207  // this saves us some pain during the check
208  //
209  Hive->ReleaseCellRoutine = ReleaseCellRoutine;
210 
211  return rc;
212 }

215 CmpCheckRegistry2(
216  PHHIVE HiveToCheck,
217  ULONG CheckFlags,
218  HCELL_INDEX Cell,
219  HCELL_INDEX ParentCell,
220  BOOLEAN ResetSD
221  )
222 /*++
223 
224 Routine Description:
225 
226  Check consistency of the registry, from a particular cell on down.
227 
228  . Check that the cell's value list, child key list, class,
229  security are OK.
230  . Check that each value entry IN the list is OK.
231  . Apply self to each child key list.
232 
233 
234  This version uses a stack in order to parse the tree "in-depth",
235  but not to touch any key_node.
236 
237 Arguments:
238 
239  Cell - HCELL_INDEX of subkey to work on.
240 
241  ParentCell - expected value of parent cell for Cell, unless
242  HCELL_NIL, in which case ignore.
243 
244 Return Value:
245 
246  0 if Hive is OK. Error return indicator if not.
247 
248  RANGE: 4000 - 4999
249 
250 --*/
251 {
252  PCMP_CHECK_REGISTRY_STACK_ENTRY CheckStack;
253  LONG StackIndex;
254  PCM_KEY_NODE Node;
255  ULONG rc = 0;
256  HCELL_INDEX SubKey;
257 
258 
259  CmpCheckRegistry2Debug.Hive = HiveToCheck;
260  CmpCheckRegistry2Debug.Status = 0;
261 
262  ASSERT( HiveToCheck->ReleaseCellRoutine == NULL );
263 
264  //
265  // Initialize the stack to simulate recursion here
266  //
267 
268  CmRetryExAllocatePoolWithTag(PagedPool,sizeof(CMP_CHECK_REGISTRY_STACK_ENTRY)*CMP_MAX_REGISTRY_DEPTH,CM_POOL_TAG|PROTECTED_POOL,CheckStack);
269  if (CheckStack == NULL) {
270  CmpCheckRegistry2Debug.Status = 4099;
271  return 4099;
272  }
273 
274 Restart:
275 
276  CheckStack[0].Cell = Cell;
277  CheckStack[0].ParentCell = ParentCell;
278  CheckStack[0].PriorSibling = HCELL_NIL;
279  CheckStack[0].ChildIndex = 0;
280  CheckStack[0].CellChecked = FALSE;
281  StackIndex = 0;
282 
283 
284  while(StackIndex >=0) {
285  //
286  // first check the current cell
287  //
288  if( CheckStack[StackIndex].CellChecked == FALSE ) {
289  CheckStack[StackIndex].CellChecked = TRUE;
290 
291  rc = CmpCheckKey(HiveToCheck,CheckFlags,CheckStack[StackIndex].Cell, CheckStack[StackIndex].ParentCell,ResetSD);
292  if (rc != 0) {
293  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"\tChild is list entry #%08lx\n", CheckStack[StackIndex].ChildIndex));
294  CmpCheckRegistry2Debug.Status = rc;
295 YankKey:
296  if( CmDoSelfHeal() && StackIndex ) { // root cell damage is fatal.
297  //
298  // delete this key from the parent's list and restart the whole iteration (not best performance, but safest).
299  //
300  if( !CmpRemoveSubKeyCellNoCellRef(HiveToCheck,CheckStack[StackIndex].ParentCell,CheckStack[StackIndex].Cell) ) {
301  //
302  // unable to delete subkey; punt.
303  //
304  break;
305  }
306  CmMarkSelfHeal(HiveToCheck);
307  rc = 0;
308  goto Restart;
309  } else {
310  // bail out
311  break;
312  }
313  } else if( StackIndex > 0 ) {
314  //
315  // key is OK, check lexicographical order with PriorSibling
316  //
317  if( CheckStack[StackIndex-1].PriorSibling != HCELL_NIL ) {
318 
319  ASSERT( CheckStack[StackIndex-1].Cell == CheckStack[StackIndex].ParentCell );
320 
321  if( !CmpCheckLexicographicalOrder(HiveToCheck,CheckStack[StackIndex-1].PriorSibling,CheckStack[StackIndex].Cell) ) {
322  //
323  // invalid order
324  //
325  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"\t Invalid subkey ordering key #%08lx\n", CheckStack[StackIndex].Cell));
326  CmpCheckRegistry2Debug.Status = 4091;
327  rc = 4091;
328  // attempt to yank the key
329  goto YankKey;
330  }
331  }
332  CheckStack[StackIndex-1].PriorSibling = CheckStack[StackIndex].Cell;
333  }
334  }
335 
336  Node = (PCM_KEY_NODE)HvGetCell(HiveToCheck, CheckStack[StackIndex].Cell);
337  if( Node == NULL ) {
338  //
339  // we couldn't map a view for the bin containing this cell
340  //
341  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"\tCould not map cell #%08lx\n", CheckStack[StackIndex].Cell));
342  CmpCheckRegistry2Debug.Status = 4098;
343  rc = 4098;
344  // bail out
345  break;
346  }
347 
348  if( CheckStack[StackIndex].ChildIndex < Node->SubKeyCounts[Stable] ) {
349  //
350  // we still have childs to check; add another entry for them and advance the
351  // StackIndex
352  //
353  SubKey = CmpFindSubKeyByNumber(HiveToCheck,
354  Node,
355  CheckStack[StackIndex].ChildIndex);
356  if( SubKey == HCELL_NIL ) {
357  //
358  // we couldn't map cell;bail out
359  //
360  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"\tCould not map cell #%08lx\n", CheckStack[StackIndex].Cell));
361  CmpCheckRegistry2Debug.Status = 4097;
362  rc = 4097;
363  break;
364  }
365  //
366  // next iteration will check the next child
367  //
368  CheckStack[StackIndex].ChildIndex++;
369 
370  StackIndex++;
371  if( StackIndex == CMP_MAX_REGISTRY_DEPTH ) {
372  //
373  // we've run out of stack; registry tree has too many levels
374  //
375  CmpCheckRegistry2Debug.Status = 4096;
376  rc = 4096;
377  // bail out
378  break;
379  }
380  CheckStack[StackIndex].Cell = SubKey;
381  CheckStack[StackIndex].ParentCell = CheckStack[StackIndex-1].Cell;
382  CheckStack[StackIndex].PriorSibling = HCELL_NIL;
383  CheckStack[StackIndex].ChildIndex = 0;
384  CheckStack[StackIndex].CellChecked = FALSE;
385 
386  } else {
387  //
388  // we have checked all childs for this node; go back
389  //
390  StackIndex--;
391 
392  }
393 
394  }
395 
396  ExFreePoolWithTag(CheckStack, CM_POOL_TAG|PROTECTED_POOL);
397  return rc;
398 }
