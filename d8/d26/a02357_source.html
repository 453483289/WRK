529  }
530  }
531  }
532 
533  } else if (KiNode0.ProcessorMask != KeActiveProcessors) {
534  goto StartFailure;
535  }

538  // Clear node structure address for nonexistent nodes.
539  //
540 
541  for (Index = KeNumberNodes; Index < MAXIMUM_CCNUMA_NODES; Index += 1) {
542  KeNodeBlock[Index] = NULL;
543  }

548
