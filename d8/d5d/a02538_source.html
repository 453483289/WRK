73 
74 Routine Description:
75 
76  This routine zeroes all inactive pagefile blocks in the specified paging
77  file.

95  PFN_NUMBER MaxPagesToWrite;
96  PMMPFN Pfn1;
97  PPFN_NUMBER Page;
98  PFN_NUMBER MdlHack[(sizeof(MDL)/sizeof(PFN_NUMBER)) + MM_MAXIMUM_WRITE_CLUSTER];

115  ZeroContext = (PMM_ZERO_PAGEFILE_CONTEXT) Context;
116

118  ZeroedPageFrame = ZeroContext->ZeroedPageFrame;
119  AllDone = ZeroContext->AllDone;
120 
121  ExFreePool (Context);
122 
123  NumberOfBytes = MmModifiedWriteClusterSize << PAGE_SHIFT;
124  MaxPagesToWrite = NumberOfBytes >> PAGE_SHIFT;
125 
126  Mdl = (PMDL) MdlHack;
127  Page = (PPFN_NUMBER)(Mdl + 1);
128 
129  KeInitializeEvent (&IoEvent, NotificationEvent, FALSE);
130 
131  MmInitializeMdl (Mdl, NULL, PAGE_SIZE);
132 
133  Mdl->MdlFlags |= MDL_PAGES_LOCKED;
134 
135  Mdl->StartVa = NULL;
136 
137  i = 0;
138  Page = (PPFN_NUMBER)(Mdl + 1);
139 
140  for (i = 0; i < MaxPagesToWrite; i += 1) {
141  *Page = ZeroedPageFrame;
142  Page += 1;
143  }
144 
145  count = 0;
146  write = FALSE;
147 
148  SATISFY_OVERZEALOUS_COMPILER (first = 0);
149 
150  LOCK_PFN (OldIrql);
151 
152  for (i = 1; i < PagingFile->Size; i += 1) {
153 
154  if (RtlCheckBit (PagingFile->Bitmap, (ULONG) i) == 0) {
155 
156  //
157  // Claim the pagefile location as the modified writer
158  // may already be scanning.
159  //
160 
161  RtlSetBit (PagingFile->Bitmap, (ULONG) i);
162 
163  if (count == 0) {
164  first = i;
165  }
166 
167  count += 1;
168 
169  if ((count == MaxPagesToWrite) || (i == PagingFile->Size - 1)) {
170  write = TRUE;
171  }
172  }
173  else {
174  if (count != 0) {
175 
176  //
177  // Issue a write.
178  //
179 
180  write = TRUE;
181  }
182  }
183 
184  if (write) {
185 
186  UNLOCK_PFN (OldIrql);
187 
188  StartingOffset.QuadPart = (LONGLONG)first << PAGE_SHIFT;
189  Mdl->ByteCount = count << PAGE_SHIFT;
190  KeClearEvent (&IoEvent);
191 
192  Status = IoSynchronousPageWrite (PagingFile->File,
193  Mdl,
194  &StartingOffset,
195  &IoEvent,
196  &IoStatus);
197 
198  //
199  // Ignore all I/O failures - there is nothing that can
200  // be done at this point.
201  //
202 
203  if (!NT_SUCCESS (Status)) {
204  KeSetEvent (&IoEvent, 0, FALSE);
205  }
206 
207  Status = KeWaitForSingleObject (&IoEvent,
208  WrPageOut,
209  KernelMode,
210  FALSE,
211  (PLARGE_INTEGER)&MmTwentySeconds);
212 
213  if (Status == STATUS_TIMEOUT) {
214 
215  //
216  // The write did not complete in 20 seconds, assume
217  // that the file systems are hung and return an error.
218  //
219  // Note the zero page (and any MDL system virtual address a
220  // driver may have created) is leaked because we don't know
221  // what the filesystem or storage stack might (still) be
222  // doing to them.
223  //
224 
225  Pfn1 = MI_PFN_ELEMENT (ZeroedPageFrame);
226 
227  LOCK_PFN (OldIrql);
228 
229  //
230  // Increment the reference count on the zeroed page to ensure
231  // it is never freed.
232  //
233 
234  InterlockedIncrementPfn ((PSHORT)&Pfn1->u3.e2.ReferenceCount);
235 
236  RtlClearBits (PagingFile->Bitmap, (ULONG) first, count);
237 
238  break;
239  }
240 
241  if (Mdl->MdlFlags & MDL_MAPPED_TO_SYSTEM_VA) {
242  MmUnmapLockedPages (Mdl->MappedSystemVa, Mdl);
243  }
244 
245  write = FALSE;
246  LOCK_PFN (OldIrql);
247  RtlClearBits (PagingFile->Bitmap, (ULONG) first, count);
248  count = 0;
249  }
250  }
251 
252  UNLOCK_PFN (OldIrql);
253 
254  KeSetEvent (AllDone, 0, FALSE);
255  return;
256 }

399 MiShutdownSystem (
400  VOID
401  )
402 
403 /*++
404 
405 Routine Description:
406 
407  This function performs the shutdown of memory management. This
408  is accomplished by writing out all modified pages which are
409  destined for files other than the paging file.
410 
411  All processes have already been killed, the registry shutdown and
412  shutdown IRPs already sent. On return from this phase all mapped
413  file data must be flushed and the unused segment list emptied.
414  This releases all the Mm references to file objects, allowing many
415  drivers (especially the network) to unload.
416 
417 Arguments:
418 
419  None.
420 
421 Return Value:
422 
423  TRUE if the pages were successfully written, FALSE otherwise.
424 
425 --*/
426 
427 {
428  SIZE_T ImportListSize;
429  PLOAD_IMPORTS ImportList;
430  PLOAD_IMPORTS ImportListNonPaged;
431  PLIST_ENTRY NextEntry;
432  PKLDR_DATA_TABLE_ENTRY DataTableEntry;
433  PFN_NUMBER ModifiedPage;
434  PMMPFN Pfn1;
435  PSUBSECTION Subsection;
436  PCONTROL_AREA ControlArea;
437  PPFN_NUMBER Page;
438  PFILE_OBJECT FilePointer;
439  ULONG ConsecutiveFileLockFailures;
440  PFN_NUMBER MdlHack[(sizeof(MDL)/sizeof(PFN_NUMBER)) + MM_MAXIMUM_WRITE_CLUSTER];
441  PMDL Mdl;
442  NTSTATUS Status;
443  KEVENT IoEvent;
444  IO_STATUS_BLOCK IoStatus;
445  KIRQL OldIrql;
446  LARGE_INTEGER StartingOffset;
447  ULONG count;
448  ULONG i;
449 
450  //
451  // Don't do this more than once.
452  //
453 
454  if (MmSystemShutdown == 0) {
455 
456  Mdl = (PMDL) MdlHack;
457  Page = (PPFN_NUMBER)(Mdl + 1);
458 
459  KeInitializeEvent (&IoEvent, NotificationEvent, FALSE);
460 
461  MmInitializeMdl (Mdl, NULL, PAGE_SIZE);
462 
463  Mdl->MdlFlags |= MDL_PAGES_LOCKED;
464 
465  MmLockPageableSectionByHandle (ExPageLockHandle);
466 
467  LOCK_PFN (OldIrql);
468 
469  ModifiedPage = MmModifiedPageListHead.Flink;
470 
471  while (ModifiedPage != MM_EMPTY_LIST) {
472 
473  //
474  // There are modified pages.
475  //
476 
477  Pfn1 = MI_PFN_ELEMENT (ModifiedPage);
478 
479  if (Pfn1->OriginalPte.u.Soft.Prototype == 1) {
480 
481  //
482  // This page is destined for a file.
483  //
484 
485  Subsection = MiGetSubsectionAddress (&Pfn1->OriginalPte);
486  ControlArea = Subsection->ControlArea;
487  if ((!ControlArea->u.Flags.Image) &&
488  (!ControlArea->u.Flags.NoModifiedWriting)) {
489 
490  MiUnlinkPageFromList (Pfn1);
491 
492  //
493  // Issue the write.
494  //
495 
496  MI_SET_MODIFIED (Pfn1, 0, 0x28);
497 
498  //
499  // Up the reference count for the physical page as there
500  // is I/O in progress.
501  //
502 
503  MI_ADD_LOCKED_PAGE_CHARGE_FOR_MODIFIED_PAGE (Pfn1);
504 
505  *Page = ModifiedPage;
506  ControlArea->NumberOfMappedViews += 1;
507  ControlArea->NumberOfPfnReferences += 1;
508 
509  UNLOCK_PFN (OldIrql);
510 
511  StartingOffset.QuadPart = MiStartingOffset (Subsection,
512  Pfn1->PteAddress);
513  Mdl->StartVa = NULL;
514 
515  ConsecutiveFileLockFailures = 0;
516  FilePointer = ControlArea->FilePointer;
517 
518 retry:
519  KeClearEvent (&IoEvent);
520 
521  Status = FsRtlAcquireFileForCcFlushEx (FilePointer);
522 
523  if (NT_SUCCESS(Status)) {
524  Status = IoSynchronousPageWrite (FilePointer,
525  Mdl,
526  &StartingOffset,
527  &IoEvent,
528  &IoStatus);
529 
530  //
531  // Release the file we acquired.
532  //
533 
534  FsRtlReleaseFileForCcFlush (FilePointer);
535  }
536 
537  if (!NT_SUCCESS(Status)) {
538 
539  //
540  // Only try the request more than once if the
541  // filesystem said it had a deadlock.
542  //
543 
544  if (Status == STATUS_FILE_LOCK_CONFLICT) {
545  ConsecutiveFileLockFailures += 1;
546  if (ConsecutiveFileLockFailures < 5) {
547  KeDelayExecutionThread (KernelMode,
548  FALSE,
549  (PLARGE_INTEGER)&MmShortTime);
550  goto retry;
551  }
552  goto wait_complete;
553  }
554 
555  //
556  // Ignore all I/O failures - there is nothing that
557  // can be done at this point.
558  //
559 
560  KeSetEvent (&IoEvent, 0, FALSE);
561  }
562 
563  Status = KeWaitForSingleObject (&IoEvent,
564  WrPageOut,
565  KernelMode,
566  FALSE,
567  (PLARGE_INTEGER)&MmTwentySeconds);
568 
569 wait_complete:
570 
571  if (Mdl->MdlFlags & MDL_MAPPED_TO_SYSTEM_VA) {
572  MmUnmapLockedPages (Mdl->MappedSystemVa, Mdl);
573  }
574 
575  if (Status == STATUS_TIMEOUT) {
576 
577  //
578  // The write did not complete in 20 seconds, assume
579  // that the file systems are hung and return an
580  // error.
581  //
582 
583  LOCK_PFN (OldIrql);
584 
585  MI_SET_MODIFIED (Pfn1, 1, 0xF);
586 
587  MI_REMOVE_LOCKED_PAGE_CHARGE_AND_DECREF (Pfn1);
588  ControlArea->NumberOfMappedViews -= 1;
589  ControlArea->NumberOfPfnReferences -= 1;
590 
591  //
592  // This routine returns with the PFN lock released!
593  //
594 
595  MiCheckControlArea (ControlArea, OldIrql);
596 
597  MmUnlockPageableImageSection (ExPageLockHandle);
598 
599  return FALSE;
600  }
601 
602  LOCK_PFN (OldIrql);
603  MI_REMOVE_LOCKED_PAGE_CHARGE_AND_DECREF (Pfn1);
604  ControlArea->NumberOfMappedViews -= 1;
605  ControlArea->NumberOfPfnReferences -= 1;
606 
607  //
608  // This routine returns with the PFN lock released!
609  //
610 
611  MiCheckControlArea (ControlArea, OldIrql);
612  LOCK_PFN (OldIrql);
613 
614  //
615  // Restart scan at the front of the list.
616  //
617 
618  ModifiedPage = MmModifiedPageListHead.Flink;
619  continue;
620  }
621  }
622  ModifiedPage = Pfn1->u1.Flink;
623  }
624 
625  UNLOCK_PFN (OldIrql);
626 
627  //
628  // Indicate to the modified page writer that the system has
629  // shutdown.
630  //
631 
632  MmSystemShutdown = 1;
633 
634  //
635  // Check to see if the paging file should be overwritten.
636  // Only free blocks are written.
637  //
638 
639  if (MmZeroPageFile) {
640  MiZeroAllPageFiles ();
641  }
642 
643  MmUnlockPageableImageSection (ExPageLockHandle);
644  }
645 
646  if (PoCleanShutdownEnabled ()) {
647 
648  //
649  // Empty the unused segment list.
650  //
651 
652  LOCK_PFN (OldIrql);
653  MmUnusedSegmentForceFree = (ULONG)-1;
654  KeSetEvent (&MmUnusedSegmentCleanup, 0, FALSE);
655 
656  //
657  // Give it 5 seconds to empty otherwise assume the filesystems are
658  // hung and march on.
659  //
660 
661  for (count = 0; count < 500; count += 1) {
662 
663  if (IsListEmpty(&MmUnusedSegmentList)) {
664  break;
665  }
666 
667  UNLOCK_PFN (OldIrql);
668 
669  KeDelayExecutionThread (KernelMode,
670  FALSE,
671  (PLARGE_INTEGER)&MmShortTime);
672  LOCK_PFN (OldIrql);
673 
674 #if DBG
675  if (count == 400) {
676 
677  //
678  // Everything should have been flushed by now. Give the
679  // filesystem team a chance to debug this on checked builds.
680  //
681 
682  ASSERT (FALSE);
683  }
684 #endif
685 
686  //
687  // Resignal if needed in case more closed file objects triggered
688  // additional entries.
689  //
690 
691  if (MmUnusedSegmentForceFree == 0) {
692  MmUnusedSegmentForceFree = (ULONG)-1;
693  KeSetEvent (&MmUnusedSegmentCleanup, 0, FALSE);
694  }
695  }
696 
697  UNLOCK_PFN (OldIrql);
698 
699  //
700  // Get rid of any paged pool references as they will be illegal
701  // by the time MmShutdownSystem is called again since the filesystems
702  // will have shutdown.
703  //
704 
705  KeWaitForSingleObject (&MmSystemLoadLock,
706  WrVirtualMemory,
707  KernelMode,
708  FALSE,
709  (PLARGE_INTEGER)NULL);
710 
711  NextEntry = PsLoadedModuleList.Flink;
712  while (NextEntry != &PsLoadedModuleList) {
713 
714  DataTableEntry = CONTAINING_RECORD (NextEntry,
715  KLDR_DATA_TABLE_ENTRY,
716  InLoadOrderLinks);
717 
718  ImportList = (PLOAD_IMPORTS)DataTableEntry->LoadedImports;
719 
720  if ((ImportList != (PVOID)LOADED_AT_BOOT) &&
721  (ImportList != (PVOID)NO_IMPORTS_USED) &&
722  (!SINGLE_ENTRY(ImportList))) {
723 
724  ImportListSize = ImportList->Count * sizeof(PVOID) + sizeof(SIZE_T);
725  ImportListNonPaged = (PLOAD_IMPORTS) ExAllocatePoolWithTag (NonPagedPool,
726  ImportListSize,
727  'TDmM');
728 
729  if (ImportListNonPaged != NULL) {
730  RtlCopyMemory (ImportListNonPaged, ImportList, ImportListSize);
731  ExFreePool (ImportList);
732  DataTableEntry->LoadedImports = ImportListNonPaged;
733  }
734  else {
735 
736  //
737  // Don't bother with the clean shutdown at this point.
738  //
739 
740  PopShutdownCleanly = FALSE;
741  break;
742  }
743  }
744

747  //
748

750  ExFreePool (DataTableEntry->FullDllName.Buffer);
751  DataTableEntry->FullDllName.Buffer = NULL;
752  }
753 
754  NextEntry = NextEntry->Flink;
755  }
756

758

761  // reference to each keeping the underlying object resident.
762  // At the end of Phase1 shutdown we'll release those references
763  // to trigger the storage stack unload. The handle close must be
764  // done here however as it will reference pageable structures.
765  //
766 
767  for (i = 0; i < MmNumberOfPagingFiles; i += 1) {
768 
769  //
770  // Free each pagefile name now as it resides in paged pool and
771  // may need to be inpaged to be freed. Since the paging files
772  // are going to be shutdown shortly, now is the time to access
773  // pageable stuff and get rid of it. Zeroing the buffer pointer
774  // is sufficient as the only accesses to this are from the
775  // try-except-wrapped GetSystemInformation APIs and all the
776  // user processes are gone already.
777  //
778 
779  ASSERT (MmPagingFile[i]->PageFileName.Buffer != NULL);
780  ExFreePool (MmPagingFile[i]->PageFileName.Buffer);
781  MmPagingFile[i]->PageFileName.Buffer = NULL;
782 
783  ZwClose (MmPagingFile[i]->FileHandle);
784  }
785  }
786 
787  return TRUE;
788 }

790 BOOLEAN
791 MmShutdownSystem (
792  IN ULONG Phase
793  )
794 
795 /*++
796 
797 Routine Description:
798 
799  This function performs the shutdown of memory management. This
800  is accomplished by writing out all modified pages which are
801  destined for files other than the paging file.
802 
803 Arguments:
804 
805  Phase - Supplies 0 on the initiation of shutdown. All processes have
806  already been killed, the registry shutdown and shutdown IRPs already
807  sent. On return from this phase all mapped file data must be
808  flushed and the unused segment list emptied. This releases all
809  the Mm references to file objects, allowing many drivers (especially
810  the network) to unload.
811 
812  Supplies 1 on the initiation of shutdown. The filesystem stack
813  has received its shutdown IRPs (the stack must free its paged pool
814  allocations here and lock down any pageable code it intends to call)
815  as no more references to pageable code or data are allowed on return.
816  ie: Any IoPageRead at this point is illegal.
817  Close the pagefile handles here so the filesystem stack will be
818  dereferenced causing those drivers to unload as well.
819 
820  Supplies 2 on final shutdown of the system. Any resources not
821  freed by this point are treated as leaks and cause a bugcheck.
822 
823 Return Value:
824 
825  TRUE if the pages were successfully written, FALSE otherwise.
826 
827 --*/
828 
829 {
830  ULONG i;
831
