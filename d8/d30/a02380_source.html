37 /*++
38 
39 Routine Description:
40 
41  This function initializes a kernel interrupt object. The service routine,
42  service context, spin lock, vector, IRQL, SynchronizeIrql, and floating
43  context save flag are initialized.
44 
45 Arguments:
46 
47  Interrupt - Supplies a pointer to a control object of type interrupt.
48 
49  ServiceRoutine - Supplies a pointer to a function that is to be
50  executed when an interrupt occurs via the specified interrupt
51  vector.
52 
53  ServiceContext - Supplies a pointer to an arbitrary data structure which is
54  to be passed to the function specified by the ServiceRoutine parameter.
55 
56  SpinLock - Supplies a pointer to an executive spin lock. There are two
57  distinguished values recognized for SpinLock:
58 
59  NO_INTERRUPT_SPINLOCK - The kernel does not manage a spinlock
60  associated with this interrupt.
61 
62  NO_END_OF_INTERRUPT - The interrupt represents a spurious interrupt
63  vector, which is handled with a special, non-EOI interrupt
64  routine.
65 
66  Vector - Supplies the HAL-generated interrupt vector. Note that this
67  is not be directly used as an index into the Interrupt Dispatch Table.
68 
69  Irql - Supplies the request priority of the interrupting source.
70 
71  SynchronizeIrql - Supplies the request priority that the interrupt should be
72  synchronized with.
73 
74  InterruptMode - Supplies the mode of the interrupt; LevelSensitive or
75 
76  ShareVector - Supplies a boolean value that specifies whether the
77  vector can be shared with other interrupt objects or not. If FALSE
78  then the vector may not be shared, if TRUE it may be.
79  Latched.
80 
81  ProcessorNumber - Supplies the number of the processor to which the
82  interrupt will be connected.
83 
84  FloatingSave - Supplies a boolean value that determines whether the
85  floating point registers are to be saved before calling the service
86  routine function. N.B. This argument is ignored.
87 
88 Return Value:
89 
90  None.
91 
92 --*/
93 
94 {
95  LONG64 Index;
96  PULONG InterruptTemplate;
97 
98  UNREFERENCED_PARAMETER(FloatingSave);
99 
100  //
101  // Initialize standard control object header.
102  //
103 
104  Interrupt->Type = InterruptObject;
105  Interrupt->Size = sizeof(KINTERRUPT);
106 
107  //
108  // Initialize the address of the service routine, the service context,
109  // the address of the spin lock, the address of the actual spinlock
110  // that will be used, the vector number, the IRQL of the interrupting
111  // source, the IRQL used for synchronize execution, the interrupt mode,
112  // the processor number, and the floating context save flag.
113  //
114 
115  Interrupt->ServiceRoutine = ServiceRoutine;
116  Interrupt->ServiceContext = ServiceContext;
117  if (ARGUMENT_PRESENT(SpinLock)) {
118  Interrupt->ActualLock = SpinLock;
119 
120  } else {
121  KeInitializeSpinLock (&Interrupt->SpinLock);
122  Interrupt->ActualLock = &Interrupt->SpinLock;
123  }
124 
125  Interrupt->Vector = Vector;
126  Interrupt->Irql = Irql;
127  Interrupt->SynchronizeIrql = SynchronizeIrql;
128  Interrupt->Mode = InterruptMode;
129  Interrupt->ShareVector = ShareVector;
130  Interrupt->Number = ProcessorNumber;
131 
132  //
133  // Copy the interrupt dispatch code template into the interrupt object.
134  //
135 
136  if (SpinLock == NO_END_OF_INTERRUPT) {
137  InterruptTemplate = KiSpuriousInterruptTemplate;
138 
139  } else {
140  InterruptTemplate = KiInterruptTemplate;
141  }
142 
143  for (Index = 0; Index < NORMAL_DISPATCH_LENGTH; Index += 1) {
144  Interrupt->DispatchCode[Index] = InterruptTemplate[Index];
145  }
146 
147  //
148  // If this is the performance interrupt, and we're on a processor that
149  // supports the LB MSR, then route the interrupt to the special interrupt
150  // handler. Otherwise, route to the standard no lock interrupt handler.
151  //
152 
153  if ((SpinLock == INTERRUPT_PERFORMANCE) &&
154  (KeGetCurrentPrcb()->CpuVendor != CPU_INTEL)) {
155 
156  SpinLock = NO_INTERRUPT_SPINLOCK;
157  }
158 
159  //
160  // Set DispatchAddress to KiInterruptDispatch as a default value.
161  // The AMD64 HAL expects this to be set here. Other clients will
162  // overwrite this value as appropriate via KeConnectInterrupt().
163  //
164 
165  if (SpinLock == NO_INTERRUPT_SPINLOCK) {
166  Interrupt->DispatchAddress = &KiInterruptDispatchNoLock;
167 
168  } else if (SpinLock == INTERRUPT_PERFORMANCE) {
169  Interrupt->DispatchAddress = &KiInterruptDispatchLBControl;
170 
171  } else {
172  Interrupt->DispatchAddress = &KiInterruptDispatch;
173  }
174 
175  //
176  // Set the connected state of the interrupt object to FALSE.
177  //
178 
179  Interrupt->Connected = FALSE;
180  return;
181 }

183 BOOLEAN
184 KeConnectInterrupt (
185  __inout PKINTERRUPT Interrupt
186  )
187 
188 /*++
189 
190 Routine Description:
191 
192  This function connects an interrupt object to the interrupt vector
193  specified by the interrupt object.
194 
195 Arguments:
196 
197  Interrupt - Supplies a pointer to a control object of type interrupt.
198 
199 Return Value:
200 
201  If the interrupt object is already connected or an attempt is made to
202  connect to an interrupt vector that cannot be connected, then a value
203  of FALSE is returned. Otherwise, a value of TRUE is returned.
204 
205 --*/
206 
207 {
208 
209  BOOLEAN Connected;
210  PVOID Dispatch;
211  ULONG IdtIndex;
212  PKINTERRUPT Interruptx;
213  KIRQL Irql;
214  CCHAR Number;
215  KIRQL OldIrql;
216  PVOID Unexpected;
217  ULONG Vector;
218 
219  //
220  // If the interrupt object is already connected, the interrupt vector
221  // number is invalid, an attempt is being made to connect to a vector
222  // that cannot be connected, the interrupt request level is invalid, or
223  // the processor number is invalid, then do not connect the interrupt
224  // object. Otherwise, connect the interrupt object to the specified
225  // vector and establish the proper interrupt dispatcher.
226  //
227 
228  Connected = FALSE;
229  Irql = Interrupt->Irql;
230  Number = Interrupt->Number;
231  Vector = Interrupt->Vector;
232  IdtIndex = HalVectorToIDTEntry(Vector);
233  if (((IdtIndex > MAXIMUM_PRIMARY_VECTOR) ||
234  (Irql > HIGH_LEVEL) ||
235  (Irql != (IdtIndex >> 4)) ||
236  (Number >= KeNumberProcessors) ||
237  (Interrupt->SynchronizeIrql < Irql)) == FALSE) {
238 
239  //
240  // Set the system affinity to the specified processor, raise IRQL to
241  // dispatcher level, and lock the dispatcher database.
242  //
243 
244  KeSetSystemAffinityThread(AFFINITY_MASK(Number));
245  KiLockDispatcherDatabase(&OldIrql);
246 
247  //
248  // If the specified interrupt vector is not connected, then
249  // connect the interrupt vector to the interrupt object dispatch
250  // code, establish the dispatcher address, and set the new
251  // interrupt mode and enable masks. Otherwise, if the interrupt is
252  // already chained, then add the new interrupt object at the end
253  // of the chain. If the interrupt vector is not chained, then
254  // start a chain with the previous interrupt object at the front
255  // of the chain. The interrupt mode of all interrupt objects in
256  // a chain must be the same.
257  //
258 
259  if (Interrupt->Connected == FALSE) {
260  KeGetIdtHandlerAddress(Vector, &Dispatch);
261  Unexpected = &KxUnexpectedInterrupt0[IdtIndex];
262  if (Unexpected == Dispatch) {
263 
264  KIRQL OldIrql;
265 
266  //
267  // The interrupt vector is not connected.
268  //
269  // Raise IRQL to high level in order to prevent a pending
270  // interrupt from firing before the IDT is set up.
271  //
272 
273  KeRaiseIrql(HIGH_LEVEL,&OldIrql);
274  Connected = HalEnableSystemInterrupt(Vector,
275  Irql,
276  Interrupt->Mode);
277 
278  if (Connected != FALSE) {
279  Interrupt->DispatchAddress = &KiInterruptDispatch;
280  KeSetIdtHandlerAddress(Vector, &Interrupt->DispatchCode[0]);
281  }
282 
283  KeLowerIrql(OldIrql);
284 
285  } else if (IdtIndex >= PRIMARY_VECTOR_BASE) {
286 
287  //
288  // The interrupt vector is connected. Make sure the interrupt
289  // mode matchs and that both interrupt objects allow sharing
290  // of the interrupt vector.
291  //
292 
293  Interruptx = CONTAINING_RECORD(Dispatch,
294  KINTERRUPT,
295  DispatchCode[0]);
296 
297  if ((Interrupt->Mode == Interruptx->Mode) &&
298  (Interrupt->ShareVector != FALSE) &&
299  (Interruptx->ShareVector != FALSE) &&
300  ((Interruptx->DispatchAddress == KiInterruptDispatch) ||
301  (Interruptx->DispatchAddress == KiChainedDispatch))) {
302 
303  Connected = TRUE;
304 
305  //
306  // If the chained dispatch routine is not being used,
307  // then switch to chained dispatch.
308  //
309 
310  if (Interruptx->DispatchAddress != &KiChainedDispatch) {
311  InitializeListHead(&Interruptx->InterruptListEntry);
312  Interruptx->DispatchAddress = &KiChainedDispatch;
313  }
314 
315  InsertTailList(&Interruptx->InterruptListEntry,
316  &Interrupt->InterruptListEntry);
317  }
318  }
319  }
320 
321  //
322  // Unlock dispatcher database, lower IRQL to its previous value, and
323  // set the system affinity back to the original value.
324  //
325 
326  KiUnlockDispatcherDatabase(OldIrql);
327  KeRevertToUserAffinityThread();
328  }
329 
330  //
331  // Return whether interrupt was connected to the specified vector.
332  //
333 
334  Interrupt->Connected = Connected;
335  return Connected;
336 }

339 KeDisconnectInterrupt (

343 /*++
344 
345 Routine Description:
346 
347  This function disconnects an interrupt object from the interrupt vector
348  specified by the interrupt object.
349 
350 Arguments:
351 
352  Interrupt - Supplies a pointer to a control object of type interrupt.
353 
354 Return Value:
355 
356  If the interrupt object is not connected, then a value of FALSE is
357  returned. Otherwise, a value of TRUE is returned.
358 
359 --*/
360 
361 {
362 
363  BOOLEAN Disconnected;
364  PVOID Dispatch;
365  ULONG IdtIndex;
366  PKINTERRUPT Interruptx;
367  PKINTERRUPT Interrupty;
368  KIRQL Irql;
369  KIRQL OldIrql;
370  PVOID Unexpected;
371  ULONG Vector;
372 
373  //
374  // Set the system affinity to the specified processor, raise IRQL to
375  // dispatcher level, and lock dispatcher database.
376  //
377 
378  KeSetSystemAffinityThread(AFFINITY_MASK(Interrupt->Number));
379  KiLockDispatcherDatabase(&OldIrql);
380 
381  //
382  // If the interrupt object is connected, then disconnect it from the
383  // specified vector.
384  //
385 
386  Disconnected = Interrupt->Connected;
387  if (Disconnected != FALSE) {
388  Irql = Interrupt->Irql;
389  Vector = Interrupt->Vector;
390  IdtIndex = HalVectorToIDTEntry(Vector);
391 
392  //
393  // If the specified interrupt vector is not connected to the chained
394  // interrupt dispatcher, then disconnect it by setting its dispatch
395  // address to the unexpected interrupt routine. Otherwise, remove the
396  // interrupt object from the interrupt chain. If there is only
397  // one entry remaining in the list, then reestablish the dispatch
398  // address.
399  //
400 
401  KeGetIdtHandlerAddress(Vector, &Dispatch);
402  Interruptx = CONTAINING_RECORD(Dispatch, KINTERRUPT, DispatchCode[0]);
403  if (Interruptx->DispatchAddress == &KiChainedDispatch) {
404 
405  //
406  // The interrupt object is connected to the chained dispatcher.
407  //
408 
409  if (Interrupt == Interruptx) {
410  Interruptx = CONTAINING_RECORD(Interruptx->InterruptListEntry.Flink,
411  KINTERRUPT,
412  InterruptListEntry);
413 
414  Interruptx->DispatchAddress = &KiChainedDispatch;
415  KeSetIdtHandlerAddress(Vector, &Interruptx->DispatchCode[0]);
416  }
417 
418  RemoveEntryList(&Interrupt->InterruptListEntry);
419  Interrupty = CONTAINING_RECORD(Interruptx->InterruptListEntry.Flink,
420  KINTERRUPT,
421  InterruptListEntry);
422 
423  if (Interruptx == Interrupty) {
424  Interrupty->DispatchAddress = KiInterruptDispatch;
425  KeSetIdtHandlerAddress(Vector, &Interrupty->DispatchCode[0]);
426  }
427 
428  } else {
429 
430  //
431  // The interrupt object is not connected to the chained interrupt
432  // dispatcher.
433  //
434 
435  HalDisableSystemInterrupt(Vector, Irql);
436  Unexpected = &KxUnexpectedInterrupt0[IdtIndex];
437  KeSetIdtHandlerAddress(Vector, Unexpected);
438  }
439 
440  Interrupt->Connected = FALSE;
441  }
442 
443  //
444  // Unlock dispatcher database, lower IRQL to its previous value, and
445  // set the system affinity back to the original value.
446  //
447 
448  KiUnlockDispatcherDatabase(OldIrql);
449  KeRevertToUserAffinityThread();
450 
451  //
452  // Return whether interrupt was disconnected from the specified vector.
453  //
454 
455  return Disconnected;
456 }
