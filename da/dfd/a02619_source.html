34 ; PSINGLE_LIST_ENTRY
35 ; FirstEntrySList (
36 ; IN PSLIST_HEADER SListHead
37 ; )
38 ;
39 ; Routine Description:
40 ;
41 ; This function returns the address of the fisrt entry in the SLIST or
42 ; NULL.
43 ;
44 ; Arguments:
45 ;
46 ; ListHead (rcx) - Supplies a pointer to the sequenced listhead from
47 ; which the first entry address is to be computed.
48 ;
49 ; Return Value:
50 ;
51 ; The address of the first entry is the specified, or NULL if the list is
52 ; empty.
53 ;
54 ;--
55 
56  LEAF_ENTRY FirstEntrySList, _TEXT$00
57 
58  mov rax, [rcx] ; get address, sequence, and depth
59  and rax, 0fe000000H ; isolate packed address
60 
61 ;
62 ; The following code takes advantage of the fact that the high order bit
63 ; for user mode addresses is zero and for system addresses is one.
64 ;
65 
66  cmp rax, 1 ; set carry if address is zero
67  cmc ; set carry if address is not zero
68  rcr rax, 1 ; rotate carry into high bit
69  sar rax, 63 - 43 ; extract first entry address
70 
71  ret ; return
72 
73  LEAF_END FirstEntrySList, _TEXT$00

79 ; RtlpInterlockedPopEntrySList (

83 ; Routine Description:
84 ;
85 ; This function removes an entry from the front of a sequenced singly

87 ; If there are no entries in the list, then a value of NULL is returned.
88 ; Otherwise, the address of the entry that is removed is returned as the
89 ; function value.
90 ;
91 ; Arguments:
92 ;
93 ; ListHead (rcx) - Supplies a pointer to the sequenced listhead from
94 ; which an entry is to be removed.
95 ;
96 ; Return Value:
97 ;
98 ; The address of the entry removed from the list, or NULL if the list is
99 ; empty.
100 ;
101 ;--
102

104

106  ALTERNATE_ENTRY RtlInterlockedPopEntrySList
107 
108 ;
109 ; N.B. The following code is the continuation address should a fault occur
110 ; in the rare case described below.
111 ;
112 
113  ALTERNATE_ENTRY ExpInterlockedPopEntrySListResume
114 
115 Pop10: prefetchw [rcx] ; prefetch entry for write
116  mov rax, [rcx] ; get address, sequence, and depth
117  mov rdx, rax ; make a copy
118  and rdx, 0fe000000H ; isolate packed address
119  jz short Pop20 ; if z, list is empty
120 
121 ;
122 ; The following code takes advantage of the fact that the high order bit
123 ; for user mode addresses is zero and for system addresses is one.
124 ;
125  or rdx, 01fffffh ; sign-extend resultant address
126 
127  ror rdx, 63 - 42 ; extract next entry address
128 
129 ;
130 ; N.B. It is possible for the following instruction to fault in the rare
131 ; case where the first entry in the list is allocated on another
132 ; processor and free between the time the free pointer is read above
133 ; and the following instruction. When this happens, the access fault
134 ; code continues execution by skipping the following instruction.
135 ; This results in the compare failing and the entire operation is
136 ; retried.
137 ;
138 
139  ALTERNATE_ENTRY ExpInterlockedPopEntrySListFault
140 
141  mov r8, [rdx] ; get address of successor entry
142  shl r8, 63 - 42 ; shift address into position
143  mov r9, rax ; adjust depth but not sequence
144  dec r9w ;
145  and r9, 01ffffffh ; isolate sequence and depth
146  or r8, r9 ; merge address, sequence, and depth
147 
148  ALTERNATE_ENTRY ExpInterlockedPopEntrySListEnd
149 
150 ifndef NT_UP
151 
152  lock cmpxchg [rcx], r8 ; compare and exchange
153 
154 else
155 
156  cmpxchg [rcx], r8 ; compare and exchange
157 
158 endif
159 
160  jnz short Pop10 ; if nz, exchange failed
161 Pop20: mov rax, rdx ; set address of next entry
162  ret ;
163 
164  LEAF_END RtlpInterlockedPopEntrySList, _TEXT$00

170 ; RtlpInterlockedPushEntrySList (
171 ; IN PSINGLE_LIST_ENTRY ListHead,
172 ; IN PSINGLE_LIST_ENTRY ListEntry

174 ;
175 ; Routine Description:
176 ;
177 ; This function inserts an entry at the head of a sequenced singly linked
178 ; list so that access to the list is synchronized in an MP system.
179 ;
180 ; Arguments:
181 ;
182 ; ListHead (rcx) - Supplies a pointer to the sequenced listhead into which
183 ; an entry is to be inserted.
184 ;
185 ; ListEntry (rdx) - Supplies a pointer to the entry to be inserted at the
186 ; head of the list.
187 ;
188 ; Return Value:
189 ;
190 ; Previous contents of list head. NULL implies list went from empty to not
191 ; empty.
192 ;
193 ;--

195  LEAF_ENTRY RtlpInterlockedPushEntrySList, _TEXT$00
196 
197  ALTERNATE_ENTRY ExpInterlockedPushEntrySList
198 
199  prefetchw [rcx] ; prefetch entry for write
200  mov rax, [rcx] ; get address, sequence, and depth
201  mov r9, rdx ; make copy of list entry pointer
202  shl r9, 63 - 42 ; shift address into position
203 
204 if DBG
205 
206  test dl, 0fh ; test if entry 16-byte aligned
207  jz short Push10 ; if z, entry is 16-byte aligned
208  int 3 ; break into debugger
209 
210 endif
211 
212 Push10: mov r8, rax ; copy address, sequence, and depth
213  mov r10, rax ; copy address, sequence, and depth
214  and r8, 0fe000000h ; isolate packed address
215 
216 ;
217 ; The following code takes advantage of the fact that the high order bit
218 ; for user mode addresses is zero and for system addresses is one.
219 ;
220 
221  cmp r8, 1 ; set carry if address is zero
222  cmc ; set carry if address is not zero
223  rcr r8, 1 ; rotate carry into high bit

316 ; PSINGLE_LIST_ENTRY
317 ; InterlockedPushListSList (
318 ; IN PSLIST_HEADER ListHead,
319 ; IN PSINGLE_LIST_ENTRY List,
320 ; IN PSINGLE_LIST_ENTRY ListEnd,
321 ; IN ULONG Count
322 ; )
323 ;
324 ; Routine Description:
325 ;
326 ; This function pushes the specified singly linked list onto the front of
327 ; a sequenced list.
328 ;
329 ; Arguments:
330 ;
331 ; ListHead (rcx) - Supplies a pointer to the sequenced listhead into which
332 ; the specified list is inserted.
333 ;
334 ; List (rdx) - Supplies a pointer to the first entry in the list to be
335 ; pushed onto the front of the specified sequenced list.
336 ;
337 ; ListEnd (r8) - Supplies a pointer to the last entry in the list to be
338 ; pushed onto the front of the specified sequence list.
339 ;
340 ; Count (r9) - Supplies the number of entries in the list.
341 ;
342 ; Return Value:
343 ;
344 ; Previous contents of list head. NULL implies list went from empty to not
345 ; empty.
346 ;
347 ;--
348 
349  NESTED_ENTRY InterlockedPushListSList, _TEXT$00
350 
351  alloc_stack 8 ; allocate stack frame
352  save_reg rsi, 0 ; save nonvolatile register
353 
354  END_PROLOGUE
355 
356  prefetchw [rcx] ; prefetch entry for write
357  mov rax, [rcx] ; get address, sequence, and depth
358 
359 if DBG
360 
361  test dl, 0fh ; test if entry 16-byte aligned
362  jnz short Pshl10 ; if nz, entry not 16-byte aligned
363  test r8b, 0fh ; test if entry 16-byte aligned
364  jz short Pshl20 ; if z, entry is 16-byte aligned
365 Pshl10: int 3 ; break into debugger
366 
367 endif
368 
369 Pshl20: shl rdx, 63 - 42 ; shift first address into position
370 Pshl30: mov r10, rax ; make a copy
371  and r10, 0fe000000H ; isolate packed address
372 
373 ;
374 ; The following code takes advantage of the fact that the high order bit
375 ; for user mode addresses is zero and for system addresses is one.
376 ;
377 
378  cmp r10, 1 ; set carry if address is zero
379  cmc ; set carry if address is not zero
380  rcr r10, 1 ; rotate carry into high bit
381  sar r10, 63 - 43 ; extract next entry address
382 
383  mov [r8], r10 ; link old first to last in list
384  lea r11d, [rax][r9] ; add length of list to depth
385  and r11d, 0ffffh ; wrap depth if overflow
386  lea esi, 010000h[rax] ; increment sequence
387  and esi, 01ff0000h ; wrap sequence if overflow
