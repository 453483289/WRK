1669 WmipAllocateTraceBufferPool(
1670  IN PWMI_LOGGER_CONTEXT LoggerContext
1671  )
1672 
1673 /*++
1674 
1675 Routine Description:
1676  This routine is used to set up the circular trace buffers
1677 
1678 Arguments:
1679 
1680  LoggerContext Context of the logger to own the buffers.
1681 
1682 Returned Value:
1683 
1684  STATUS_SUCCESS if the initialization is successful
1685 
1686 --*/
1687 
1688 {
1689  ULONG NumberProcessors, SysMax_Buffers, SysMin_Buffers;
1690  LONG i;
1691  PWMI_BUFFER_HEADER Buffer;
1692  ULONG AllocatedBuffers, NumberOfBuffers;
1693  SIZE_T WmiMaximumPoolInBytes;
1694 
1695  PAGED_CODE();
1696 //
1697 // Allocate the pointers the each buffer here by sharing the same page
1698 // with LoggerContext context pointer
1699 //
1700  NumberProcessors = (ULONG) KeNumberProcessors;
1701 
1702  // This does not keep track of the amount already used by other sessions
1703  if (LoggerContext->LoggerMode & EVENT_TRACE_USE_PAGED_MEMORY) {
1704  WmiMaximumPoolInBytes = MmSizeOfPagedPoolInBytes;
1705  }
1706  else {
1707  WmiMaximumPoolInBytes = MmMaximumNonPagedPoolInBytes;
1708  }
1709 
1710  // Compute System limits for min and max
1711 
1712  // This is the absolute maximum that ANYONE can use
1713  SysMax_Buffers = (ULONG) (WmiMaximumPoolInBytes
1714  / TRACE_MAXIMUM_NP_POOL_USAGE
1715  / LoggerContext->BufferSize);
1716 
1717  // This is the absolute minimum that ANYONE MUST have
1718  SysMin_Buffers = NumberProcessors + DEFAULT_BUFFERS;
1719 
1720  // Sanity check to ensure that we have at least the minimum available
1721  if (SysMin_Buffers > SysMax_Buffers) {
1722  return STATUS_NO_MEMORY;
1723  }
1724 
1725 
1726  // Cover the case if the caller did not specify any values
1727  LoggerContext->MaximumBuffers = max(LoggerContext->MaximumBuffers,
1728  NumberProcessors + DEFAULT_BUFFERS +
1729  DEFAULT_MAX_BUFFERS);
1730 
1731  LoggerContext->MinimumBuffers = max(LoggerContext->MinimumBuffers,
1732  SysMin_Buffers);
1733 
1734 
1735  // Ensure each parameter is in range of SysMin and SysMax
1736 
1737  LoggerContext->MaximumBuffers = max (LoggerContext->MaximumBuffers,
1738  SysMin_Buffers);
1739  LoggerContext->MaximumBuffers = min (LoggerContext->MaximumBuffers,
1740  SysMax_Buffers);
1741 
1742  LoggerContext->MinimumBuffers = max (LoggerContext->MinimumBuffers,
1743  SysMin_Buffers);
1744  LoggerContext->MinimumBuffers = min (LoggerContext->MinimumBuffers,
1745  SysMax_Buffers);
1746 
1747  // In case the MaximumBuffers and MinimumBuffers got reversed pick the
1748  // larger value
1749 
1750  if (LoggerContext->MinimumBuffers > LoggerContext->MaximumBuffers) {
1751  LoggerContext->MaximumBuffers = LoggerContext->MinimumBuffers;
1752  }
1753 
1754  // NOTE: We do not return anything if we reset MaximumBuffers or MinimumBuffers
1755  // provided by the caller.
1756 
1757  LoggerContext->NumberOfBuffers = (LONG) LoggerContext->MinimumBuffers;
1758  LoggerContext->BuffersAvailable = LoggerContext->NumberOfBuffers;
1759 
1760  //
1761  // Allocate the buffers now
1762  //
1763  //
1764  // Now determine the initial number of buffers
1765  //
1766  NumberOfBuffers = LoggerContext->NumberOfBuffers;

1768  LoggerContext->BuffersAvailable = 0;
1769 
1770  AllocatedBuffers = WmipAllocateFreeBuffers(LoggerContext,
1771  NumberOfBuffers);
1772 
1773  if (AllocatedBuffers < NumberOfBuffers) {
1774  //
1775  // No enough buffer is allocated.
1776  //
1777  WmipFreeTraceBufferPool(LoggerContext);
1778  return STATUS_NO_MEMORY;
1779  }
1780

1782 // Allocate Per Processor Buffer pointers
1783 //
1784

1786  = (PWMI_BUFFER_HEADER *)
1787  WmipExtendBase(LoggerContext,
1788  sizeof(PWMI_BUFFER_HEADER)*NumberProcessors);
1789 
1790 
1791  if (LoggerContext->ProcessorBuffers == NULL) {
1792  WmipFreeTraceBufferPool(LoggerContext);
1793  return STATUS_NO_MEMORY;
1794  }
1795 
1796  //
1797  // NOTE: We already know that we have allocated > number of processors
1798  // buffers
1799  //
1800  for (i=0; i<(LONG)NumberProcessors; i++) {
1801  Buffer = (PWMI_BUFFER_HEADER) WmipGetFreeBuffer(LoggerContext);
1802  LoggerContext->ProcessorBuffers[i] = Buffer;
1803  Buffer->ClientContext.ProcessorNumber = (UCHAR)i;
1804  }
1805 
1806  return STATUS_SUCCESS;
1807 }
