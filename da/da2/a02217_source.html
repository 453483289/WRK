157 //
158 // Macro that defines what a "drive letter" mount point is. This macro can
159 // be used to scan the result from QUERY_POINTS to discover which mount points
160 // are find "drive letter" mount points.
161 //
162 
163 #define MOUNTMGR_IS_DRIVE_LETTER(s) ( \
164  (s)->Length == 28 && \
165  (s)->Buffer[0] == '\\' && \
166  (s)->Buffer[1] == 'D' && \
167  (s)->Buffer[2] == 'o' && \
168  (s)->Buffer[3] == 's' && \
169  (s)->Buffer[4] == 'D' && \
170  (s)->Buffer[5] == 'e' && \
171  (s)->Buffer[6] == 'v' && \
172  (s)->Buffer[7] == 'i' && \
173  (s)->Buffer[8] == 'c' && \
174  (s)->Buffer[9] == 'e' && \
175  (s)->Buffer[10] == 's' && \
176  (s)->Buffer[11] == '\\' && \
177  (s)->Buffer[12] >= 'A' && \
178  (s)->Buffer[12] <= 'Z' && \
179  (s)->Buffer[13] == ':')
180 
181 //
182 // Macro that defines what a "volume name" mount point is. This macro can
183 // be used to scan the result from QUERY_POINTS to discover which mount points
184 // are "volume name" mount points.
185 //
186 
187 #define MOUNTMGR_IS_VOLUME_NAME(s) ( \
188  ((s)->Length == 96 || ((s)->Length == 98 && (s)->Buffer[48] == '\\')) && \
189  (s)->Buffer[0] == '\\' && \
190  ((s)->Buffer[1] == '?' || (s)->Buffer[1] == '\\') && \
191  (s)->Buffer[2] == '?' && \
192  (s)->Buffer[3] == '\\' && \
193  (s)->Buffer[4] == 'V' && \
194  (s)->Buffer[5] == 'o' && \
195  (s)->Buffer[6] == 'l' && \
196  (s)->Buffer[7] == 'u' && \
197  (s)->Buffer[8] == 'm' && \
198  (s)->Buffer[9] == 'e' && \
199  (s)->Buffer[10] == '{' && \
200  (s)->Buffer[19] == '-' && \
201  (s)->Buffer[24] == '-' && \
202  (s)->Buffer[29] == '-' && \
203  (s)->Buffer[34] == '-' && \
204  (s)->Buffer[47] == '}' \
205  )
