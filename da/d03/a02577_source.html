144  structure that was specified when the APC was initialized.
145 
146  SystemArgument1, SystemArgument2 - Supplies a set of two pointer to two
147  arguments that contain untyped data.
148 
149 Return Value:
150 
151  None.
152 
153 --*/
154 
155 {
156  PGETSETCONTEXT Ctx;
157  PKTRAP_FRAME TrapFrame;
158  PETHREAD Thread;
159 
160  PAGED_CODE();
161 
162  UNREFERENCED_PARAMETER( NormalRoutine );
163  UNREFERENCED_PARAMETER( NormalContext );
164  UNREFERENCED_PARAMETER( SystemArgument1 );
165  UNREFERENCED_PARAMETER( SystemArgument2 );
