52 //
53 // Executive macros
54 //
55 //
56 
57 // This macro returns the address of a field given the type of the structure, pointer to
58 // the base of the structure and the name of the field whose address is to be computed.
59 // This returns the same value as &Base->Member
60 //
61 
62 #define EX_FIELD_ADDRESS(Type, Base, Member) ((PUCHAR)Base + FIELD_OFFSET(Type, Member))
63 
64 //
65 // This macro is used like a C for loop and iterates over all the entries in the list.
66 //
67 // _Type - Name of the data type for each element in the list
68 // _Link - Name of the field of type LIST_ENTRY used to link the elements.
69 // _Head - Pointer to the head of the list.
70 // _Current - Points to the iterated element.
71 //
72 // Example: EX_FOR_EACH_IN_LIST(SYSTEM_FIRMWARE_TABLE_HANDLER_NODE,
73 // FirmwareTableProviderList,
74 // &ExpFirmwareTableProviderListHead,
75 // HandlerListCurrent) {
76 //
77 // /* Do something with Current */
78 //
79 // }
80 //
81 
82 #define EX_FOR_EACH_IN_LIST(_Type, _Link, _Head, _Current) \
83  for((_Current) = CONTAINING_RECORD((_Head)->Flink, _Type, _Link); \
84  (_Head) != (PLIST_ENTRY)EX_FIELD_ADDRESS(_Type, _Current, _Link); \
85  (_Current) = CONTAINING_RECORD(((PLIST_ENTRY)EX_FIELD_ADDRESS(_Type, _Current, _Link))->Flink, \
86  _Type, \
87  _Link) \
88  )
