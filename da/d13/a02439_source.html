31 //
32 // VOID
33 // KiIpiSendSynchronousPacket (
34 // IN PKPRCB Prcb,
35 // IN KAFFINITY TargetProcessors,
36 // IN PKIPI_WORKER WorkerFunction,
37 // IN PVOID Parameter1,
38 // IN PVOID Parameter2,
39 // IN PVOID Parameter3
40 // )
41 //
42 // Routine Description:
43 //
44 // Similar to KiIpiSendPacket except that the pointer to the
45 // originating PRCB (SignalDone) is kept in the global variable
46 // KiSynchPacket and is protected by the context swap lock. The
47 // actual IPI is sent via KiIpiSend with a request type of
48 // IPI_SYNCH_REQUEST. This mechanism is used to send IPI's that
49 // (reverse) stall until released by the originator. This avoids
50 // a deadlock that can occur if two processors are trying to deliver
51 // IPI packets at the same time and one of them is a reverse stall.
52 //
53 // N.B. The low order bit of the packet address is set if there is
54 // exactly one target recipient. Otherwise, the low order bit
55 // of the packet address is clear.
56 //
57 
58 #define KiIpiSendSynchronousPacket(Prcb,Target,Function,P1,P2,P3) \
59  { \
60  extern PKPRCB KiSynchPacket; \
61  \
62  Prcb->CurrentPacket[0] = (PVOID)(P1); \
63  Prcb->CurrentPacket[1] = (PVOID)(P2); \
64  Prcb->CurrentPacket[2] = (PVOID)(P3); \
65  Prcb->TargetSet = (Target); \
66  Prcb->WorkerRoutine = (Function); \
67  if (((Target) & ((Target) - 1)) == 0) { \
68  KiSynchPacket = (PKPRCB)((ULONG_PTR)(Prcb) | 1); \
69  } else { \
70  KiSynchPacket = (Prcb); \
71  Prcb->PacketBarrier = 1; \
72  } \
73  KiIpiSend((Target),IPI_SYNCH_REQUEST); \
74  }
