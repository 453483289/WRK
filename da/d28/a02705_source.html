48 /*++
49 
50 
51 Routine Description:
52 
53  Create a new token that is a duplicate of an existing token.
54 
55 Arguments:
56 
57  ExistingTokenHandle - Is a handle to a token already open for
58  TOKEN_DUPLICATE access.
59 
60  DesiredAccess - Is an access mask indicating which access types
61  are desired to the newly created token. If specified as zero,
62  the granted access mask of the existing token handle
63  is used as the desired access mask for the new token.
64 
65  ObjectAttributes - Points to the standard object attributes data
66  structure. Refer to the NT Object Management
67  Specification for a description of this data structure.
68 
69  If the new token type is TokenImpersonation, then this
70  parameter may be used to specify the impersonation level
71  of the new token. If no value is provided, and the source
72  token is an impersonation token, then the impersonation level
73  of the source will become that of the target as well. If the
74  source token is a primary token, then an impersonation level
75  must be explicitly provided.
76 
77  If the token being duplicated is an impersonation token, and
78  an impersonation level is explicitly provided for the target,
79  then the value provided must not be greater than that of the
80  source token. For example, an Identification level token can
81  not be duplicated to produce a Delegation level token.
82 
83  EffectiveOnly - Is a boolean flag indicating whether the entire
84  source token should be duplicated into the target token or
85  just the effective (currently enabled) part of the token.
86  This provides a means for a caller of a protected subsystem
87  to limit which privileges and optional groups are made
88  available to the protected subsystem. A value of TRUE
89  indicates only the currently enabled parts of the source
90  token are to be duplicated. Otherwise, the entire source
91  token is duplicated.
92 
93  TokenType - Indicates which type of object the new object is to
94  be created as (primary or impersonation). If you are duplicating
95  an Impersonation token to produce a Primary token, then
96  the Impersonation token must have an impersonation level of
97  either DELEGATE or IMPERSONATE.
98 
99 
100  NewTokenHandle - Receives the handle of the newly created token.
101 
102 Return Value:
103 
104  STATUS_SUCCESS - Indicates the operation was successful.
105 
106  STATUS_INVALID_PARAMETER - Indicates one or more of the parameter values
107  was invalid. This value is returned if the target token is not
108  an impersonation token.
109 
110  STATUS_BAD_IMPERSONATION_LEVEL - Indicates the impersonation level
111  requested for the duplicate token is not compatible with the
112  level of the source token. The duplicate token may not be assigned
113  a level greater than that of the source token.
114 
115 --*/

272  }

278  //
279 
280  NewToken = NULL;

282  Token,
283  ObjectAttributes,
284  EffectiveOnly,
285  TokenType,
286  SecurityQos.ImpersonationLevel,
287  PreviousMode,
288  &NewToken
289  );
290 
291 
292  if (NT_SUCCESS(Status)) {
293 
294  //
295  // Insert the new token
296  //
297 
298  Status = ObInsertObject( NewToken,
299  NULL,
300  EffectiveDesiredAccess,
301  0,
302  (PVOID *)NULL,
303  &LocalHandle
304  );
305  }
306 
307  //
308  // We no longer need our reference to the source token
309  //
310 
311  ObDereferenceObject( (PVOID)Token );
312 
313  if (SecurityQosPresent) {
314  SeFreeCapturedSecurityQos( &SecurityQos );
315  }
316 
317  //
318  // Return the new handle
319  //
320 
321  if (NT_SUCCESS(Status)) {
322  try {
323  *NewTokenHandle = LocalHandle;
324  } except(EXCEPTION_EXECUTE_HANDLER) {
325  return GetExceptionCode();
326  }
327  }
328 
329  return Status;
330 }

345 /*++
346 
347 
348 Routine Description:
349 
350  This routine does the bulk of the work to actually duplicate
351  a token. This routine assumes all access validation and argument
352  probing (except the ObjectAttributes) has been performed.
353 
354  THE CALLER IS RESPONSIBLE FOR CHECKING SUBJECT RIGHTS TO CREATE THE
355  TYPE OF TOKEN BEING CREATED.
356 
357  This routine acquires a read lock on the token being duplicated.
358 
359 Arguments:
360 
361  ExistingToken - Points to the token to be duplicated.
362 
363  ObjectAttributes - Points to the standard object attributes data
364  structure. Refer to the NT Object Management
365  Specification for a description of this data structure.
366 
367  The security Quality Of Service of the object attributes are ignored.
368  This information must be specified using parameters to this
369  routine.
370 
371  EffectiveOnly - Is a boolean flag indicating whether the entire
372  source token should be duplicated into the target token or
373  just the effective (currently enabled) part of the token.
374  This provides a means for a caller of a protected subsystem
375  to limit which privileges and optional groups are made
376  available to the protected subsystem. A value of TRUE
377  indicates only the currently enabled parts of the source
378  token are to be duplicated. Otherwise, the entire source
379  token is duplicated.
380 
381  TokenType - Indicates the type of token to make the duplicate token.
382 
383  ImpersonationLevel - This value specifies the impersonation level
384  to assign to the duplicate token. If the TokenType of the
385  duplicate is not TokenImpersonation then this parameter is
386  ignored. Otherwise, it is must be provided.
387 
388  RequestorMode - Mode of client requesting the token be duplicated.
389 
390  DuplicateToken - Receives a pointer to the duplicate token.
391  The token has not yet been inserted into any object table.
392  No exceptions are expected when trying to set this OUT value.
393 
394 Return Value:
395 
396  STATUS_SUCCESS - The service successfully completed the requested
397  operation.
398 
399 
400 --*/

402  NTSTATUS Status;
403 
404  PTOKEN NewToken;
405  PULONG DynamicPart;
406  ULONG PagedPoolSize;
407  ULONG NonPagedPoolSize;
408  ULONG TokenBodyLength;
409  ULONG_PTR FieldOffset;
410  ULONG DynamicSize;
411  ULONG Index;
412 
413  PSECURITY_TOKEN_PROXY_DATA NewProxyData = NULL;
414  PSECURITY_TOKEN_AUDIT_DATA NewAuditData = NULL;
415  PSID_AND_ATTRIBUTES UserAndGroups;
416  PSID_AND_ATTRIBUTES RestrictedSids;
417 
418  PERESOURCE TokenLock;
419 
420 #if DBG || TOKEN_LEAK_MONITOR
421  ULONG Frames;
422 #endif
423 
424  PAGED_CODE();
425 
426  ASSERT( sizeof(SECURITY_IMPERSONATION_LEVEL) <= sizeof(ULONG) );
427 
428 
429  if ( TokenType == TokenImpersonation ) {
430 
431  ASSERT( SecurityDelegation > SecurityImpersonation );
432  ASSERT( SecurityImpersonation > SecurityIdentification );
433  ASSERT( SecurityIdentification > SecurityAnonymous );
434 
435  if ( (ImpersonationLevel > SecurityDelegation) ||
436  (ImpersonationLevel < SecurityAnonymous) ) {
437 
438  return STATUS_BAD_IMPERSONATION_LEVEL;
439  }
440  }
441 
442  if (ARGUMENT_PRESENT(ExistingToken->ProxyData)) {
443 
444  Status = SepCopyProxyData(
445  &NewProxyData,
446  ExistingToken->ProxyData
447  );
448 
449  if (!NT_SUCCESS(Status)) {
450  return( Status );
451  }
452 
453  } else {
454 
455  NewProxyData = NULL;
456  }
457 
458  if (ARGUMENT_PRESENT( ExistingToken->AuditData )) {
459 
460  NewAuditData = ExAllocatePool( PagedPool, sizeof( SECURITY_TOKEN_AUDIT_DATA ));
461 
462  if (NewAuditData == NULL) {
463 
464  SepFreeProxyData( NewProxyData );
465 
466  return( STATUS_INSUFFICIENT_RESOURCES );
467 
468  } else {
469

471  }
472 
473  } else {

482

484  ExFreePool( NewAuditData );
485  }

488

490  }

494  //

498

503  RequestorMode, // ProbeMode
504  SeTokenObjectType, // ObjectType
505  ObjectAttributes, // ObjectAttributes
506  RequestorMode, // OwnershipMode
507  NULL, // ParseContext

509  PagedPoolSize, // PagedPoolCharge
510  NonPagedPoolSize, // NonPagedPoolCharge

512  );
513 
514  if (!NT_SUCCESS(Status)) {
515  SepFreeProxyData( NewProxyData );
516  ExFreePool( TokenLock );
517

519  ExFreePool( NewAuditData );
520  }

527  //
528

560  //

564  NewToken->DefaultOwnerIndex = ExistingToken->DefaultOwnerIndex;
565  NewToken->UserAndGroupCount = ExistingToken->UserAndGroupCount;
566  NewToken->RestrictedSidCount = ExistingToken->RestrictedSidCount;
567  NewToken->PrivilegeCount = ExistingToken->PrivilegeCount;

571  NewToken->AuditPolicy = ExistingToken->AuditPolicy;

573

576  // This can not fail, since there is already a token in this logon
577  // session.
578  //
579

584  if (!NT_SUCCESS (Status)) {
585  SepReleaseTokenReadLock( ExistingToken );
586 
587  NewToken->DynamicPart = NULL;
588  ObDereferenceObject (NewToken);
589  return Status;
590  }
591 
592 
593 #if DBG || TOKEN_LEAK_MONITOR
594 
595  NewToken->ProcessCid = PsGetCurrentThread()->Cid.UniqueProcess;
596  NewToken->ThreadCid = PsGetCurrentThread()->Cid.UniqueThread;
597  NewToken->CreateMethod = 0xD; // Duplicate
598  NewToken->Count = 0;
599  NewToken->CaptureCount = 0;

603  PsGetCurrentProcess()->ImageFileName,
604  min(sizeof(NewToken->ImageFileName), sizeof(PsGetCurrentProcess()->ImageFileName))
605  );
606

608  (PVOID)NewToken->CreateTrace,
609  TRACE_SIZE,
610  0
611  );
612

614

616  (PVOID)&NewToken->CreateTrace[Frames],

623

626  //
627  // Copy and initialize the variable part.
628  // The variable part is assumed to be position independent.
629  //
630

632  (PVOID)&(ExistingToken->VariablePart),

634  );
635

637  // Set the address of the UserAndGroups array.
638  //
639 
640  ASSERT( ARGUMENT_PRESENT(ExistingToken->UserAndGroups ) );

643

647  //

651

657  //

660

664 
665  Index -= 1;
666  UserAndGroups++;
667 
668  }
669 
670  //
671  // Set the address of the RestrictedSids array.
672  //
673 
674  RestrictedSids = ExistingToken->RestrictedSids;
675  NewToken->RestrictedSids = RestrictedSids;
676 
677  if (ARGUMENT_PRESENT(RestrictedSids) ) {
678  ASSERT( (ULONG_PTR)(ExistingToken->RestrictedSids) >=
679  (ULONG_PTR)(&(ExistingToken->VariablePart)) );
680 
681 
682  (ULONG_PTR) RestrictedSids += FieldOffset;
683  NewToken->RestrictedSids = RestrictedSids;
684 
685  //
686  // Now go through and change the address of each SID pointer
687  // for the user and groups
688  //
689 
690  Index = NewToken->RestrictedSidCount;
691 
692  while (Index > 0) {
693 
694  (ULONG_PTR) RestrictedSids->Sid += FieldOffset;
695  RestrictedSids++;
696 
697  Index -= 1;
698 
699  }
700  }
701 
702  //
703  // If present, set the address of the privileges
704  //
705 
706  if (ExistingToken->PrivilegeCount > 0) {
707  ASSERT( ARGUMENT_PRESENT(ExistingToken->Privileges ) );
708  ASSERT( (ULONG_PTR)(ExistingToken->Privileges) >=
709  (ULONG_PTR)(&(ExistingToken->VariablePart)) );
710 
711  (ULONG_PTR) NewToken->Privileges = (ULONG_PTR) ExistingToken->Privileges + FieldOffset;
712 
713  } else {
714 
715  NewToken->Privileges = NULL;
716 
717  }
718 
719  //
720  // Allocate the dynamic portion
721  //
722  DynamicSize = SeLengthSid( ExistingToken->PrimaryGroup );
723  if (ExistingToken->DefaultDacl) {
724  DynamicSize += ExistingToken->DefaultDacl->AclSize;
725  }
726 
727  DynamicPart = (PULONG)ExAllocatePoolWithTag(
728  PagedPool,
729  DynamicSize,
730  'dTeS'
731  );
732 
733  NewToken->DynamicPart = DynamicPart;
734 
735  if (DynamicPart == NULL) {
736  SepReleaseTokenReadLock( ExistingToken );
737  ObDereferenceObject (NewToken);
738  return( STATUS_INSUFFICIENT_RESOURCES );
739  }
740  //
741  // Copy and initialize the dynamic part.
742  // The dynamic part is assumed to be position independent.
743  //
744 
745  RtlCopyMemory( (PVOID)DynamicPart,
746  (PVOID)(ExistingToken->DynamicPart),
747  DynamicSize
748  );
749 
750  FieldOffset = (ULONG_PTR) DynamicPart - (ULONG_PTR) ExistingToken->DynamicPart;
751  //
752  // If present, set the address of the default Dacl
753  //
754 
755  NewToken->DefaultDacl = ExistingToken->DefaultDacl;
756  if (ARGUMENT_PRESENT(NewToken->DefaultDacl)) {
757 
758  ASSERT( (ULONG_PTR)(ExistingToken->DefaultDacl) >=
759  (ULONG_PTR)(ExistingToken->DynamicPart) );
760 
761 
762  (ULONG_PTR) NewToken->DefaultDacl += FieldOffset;
763 
764  }
765 
766  //
767  // Set the address of the primary group
768  //
769 
770  ASSERT(ARGUMENT_PRESENT(ExistingToken->PrimaryGroup));
771 
772  ASSERT( (ULONG_PTR)(ExistingToken->PrimaryGroup) >=
773  (ULONG_PTR)(ExistingToken->DynamicPart) );
774 
775 
776  (ULONG_PTR) NewToken->PrimaryGroup = (ULONG_PTR) ExistingToken->PrimaryGroup + FieldOffset;
777 
778  //
779  // Release the source token.
780  //
781 
782  SepReleaseTokenReadLock( ExistingToken );
783 
784 
785 
786 
787  //
788  // For the time being, take the easy way to generating an "EffectiveOnly"
789  // duplicate. That is, use the same space required of the original, just
790  // eliminate any IDs or privileges not active.
791  //
792  // Ultimately, if duplication becomes a common operation, then it will be
793  // worthwhile to recalculate the actual space needed and copy only the
794  // effective IDs/privileges into the new token.
795  //
796 
797  if (EffectiveOnly) {
798  SepMakeTokenEffectiveOnly( NewToken );
799  }
800 
801 
802  //
803  // If the NewToken inherited an active SEP_AUDIT_POLICY from ExistingToken,
804  // then increment the counter of tokens with policies.
805  //
806 
807  if ( NewToken->AuditPolicy.Overlay ) {
808  SepModifyTokenPolicyCounter(&NewToken->AuditPolicy, TRUE);
809  }
810 
811 #if DBG || TOKEN_LEAK_MONITOR
812  if (SepTokenLeakTracking && SepTokenLeakMethodWatch == 0xD && PsGetCurrentProcess()->UniqueProcessId == SepTokenLeakProcessCid) {
813 
814  NewToken->Count = InterlockedIncrement(&SepTokenLeakMethodCount);
815  if (NewToken->Count >= SepTokenLeakBreakCount) {
816 
817  DbgPrint("\nToken number 0x%x = 0x%x\n", NewToken->Count, NewToken);
818  DbgBreakPoint();
819  }
820  }
821 #endif
822 
823  (*DuplicateToken) = NewToken;
824  return Status;
825 }
