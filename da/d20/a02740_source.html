69 WmipConstructMCAErrorEvent(
70  IN PMCA_EXCEPTION McaException,
71  IN ULONG ErrorLogSize,
72  IN OUT PWNODE_SINGLE_INSTANCE Wnode,
73  IN OUT PMSMCAEvent_Header Header,
74  IN OUT PUCHAR *RawPtr,
75  IN OUT BOOLEAN *IsFatal
76  )
77 /*++
78 
79 Routine Description:
80 
81  This routine attempts to build an MCA error log event from the information
82  in the supplied MCA_EXCEPTION.
83 
84 Arguments:
85 
86  McaException - Pointer to a MCA_EXCEPTION that encapsulates the machine
87  check event.
88 
89  ErrorLogSize - The total size in bytes of the raw error information.
90 
91  Wnode - Pointer to the error event WNODE.
92 
93  Header - Pointer to the common header for all MCA events.
94 
95  RawPtr - Address into which a pointer to the raw MCA data is to
96  be copied.
97 
98 Return Value:
99 
100  STATUS_SUCCESS - If an event is successfully constructed.
101 
102  STATUS_NOT_SUPPORTED - If the routine does not create an event.
103 
104 --*/
105 {
106  PKPRCB prcb;
107  NTSTATUS status;
108 
109  PAGED_CODE();
110 
111  if (*IsFatal) {
112  if (McaException->ExceptionType == HAL_MCA_RECORD) {
113  if (!McaException->u.Mca.Status.MciStatus.UncorrectedError) {
114  *IsFatal = FALSE;
115  }
116  }
117  }
118 
119  prcb = KeGetCurrentPrcb();
120 
121  if (prcb->CpuVendor == CPU_AMD) {
122 
123  status = WmipConstructMCAErrorEventAMD(
124  McaException,
125  ErrorLogSize,
126  Wnode,
127  Header,
128  RawPtr
129  );
130 
131  } else if (prcb->CpuVendor == CPU_INTEL) {
132 
133  status = WmipConstructMCAErrorEventIntel(
134  McaException,
135  ErrorLogSize,
136  Wnode,
137  Header,
138  RawPtr
139  );
140 
141  } else {
142 
143  status = STATUS_NOT_SUPPORTED;
144 
145  }
146 
147  return status;
148 }

150 NTSTATUS
151 WmipConstructMCAErrorEventIntel(
152  IN PMCA_EXCEPTION McaException,
153  IN ULONG ErrorLogSize,
154  IN OUT PWNODE_SINGLE_INSTANCE Wnode,
155  IN OUT PMSMCAEvent_Header Header,
156  IN OUT PUCHAR *RawPtr
157  )
158 {
159  PMSMCAEvent_MemoryHierarchyError memoryEvent;
160  PMSMCAEvent_InvalidError invalidEvent;
161  PMSMCAEvent_TLBError tlbEvent;
162  PMSMCAEvent_BusError busEvent;
163  NTSTATUS status;
164  ULONG mcaCode;
165 
166  PAGED_CODE();
167 
168  //
169  // Sanity checks on the size of the event structures. None should be
170  // larger than the size of a memory error event.
171  //
172  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
173  sizeof(MSMCAEvent_InvalidError));
174  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
175  sizeof(MSMCAEvent_MemoryHierarchyError));
176  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
177  sizeof(MSMCAEvent_TLBError));
178  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
179  sizeof(MSMCAEvent_BusError));
180 
181  //
182  // This routine is only to be called when the MCA status bits are valid.
183  //
184  WmipAssert(McaException->u.Mca.Status.MciStatus.Valid);
185  mcaCode = McaException->u.Mca.Status.MciStatus.McaErrorCode;
186  status = STATUS_SUCCESS;
187 
188  if (mcaCode == MCA_CODE_MICROCODE_ROM_PARITY_ERROR ||
189  mcaCode == MCA_CODE_EXTERNAL_ERROR ||
190  mcaCode == MCA_CODE_FRC_ERROR ||
191  mcaCode == MCA_CODE_INTERNAL_TIMER_ERROR) {
192 
193  //
194  // For Microcode ROM parity errors, external errors, FRC errors, and
195  // internal timer errors, we will use the InvalidError event since it
196  // is useful for generating error events with no parameters.
197  //
198  invalidEvent = (PMSMCAEvent_InvalidError)Header;
199 
200  //
201  // Indicate the error type.
202  //
203  if (mcaCode == MCA_CODE_MICROCODE_ROM_PARITY_ERROR) {
204  invalidEvent->Type = (ULONG)MCA_MICROCODE_ROM_PARITY_ERROR;
205  } else if (mcaCode == MCA_CODE_EXTERNAL_ERROR) {
206  invalidEvent->Type = (ULONG)MCA_EXTERNAL_ERROR;
207  } else if (mcaCode == MCA_CODE_FRC_ERROR) {
208  invalidEvent->Type = (ULONG)MCA_FRC_ERROR;
209  } else {
210  invalidEvent->Type = (ULONG)MCA_INTERNALTIMER_ERROR;
211  }
212 
213  //
214  // Fill in the GUID and the size of the data block.
215  //
216  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_InvalidErrGuid;
217  Wnode->SizeDataBlock =
218  FIELD_OFFSET(MSMCAEvent_InvalidError, RawRecord) + ErrorLogSize;
219 
220  //
221  // Copy the address of the raw record into the supplied address.
222  //
223  *RawPtr = invalidEvent->RawRecord;
224 
225  //
226  // Indicate the size of the entire record.
227  //
228  invalidEvent->Size = ErrorLogSize;
229 
230  } else if (mcaCode & MCA_CODE_BUS_ERR_MASK) {
231 
232  //
233  // Bus/Interconnect error. Extract the details of the error information
234  // from the preserved MCI_STATS and save it in the bus error event.
235  //
236 
237  busEvent = (PMSMCAEvent_BusError)Header;
238 
239  if (mcaCode & 0x0100) {
240  busEvent->Type = (ULONG)MCA_BUS_TIMEOUT_ERROR;
241  } else {
242  busEvent->Type = (ULONG)MCA_BUS_ERROR;
243  }
244 
245  busEvent->Participation = ((mcaCode & 0x00000600) >> 9);
246  busEvent->MemoryHierarchyLevel = (mcaCode & 0x00000003);
247  busEvent->RequestType = ((mcaCode & 0x000000F0) >> 4);
248  busEvent->MemOrIo = ((mcaCode & 0x0000000C) >> 2);
249 
250  if (McaException->u.Mca.Status.MciStatus.AddressValid) {
251  busEvent->Address = McaException->u.Mca.Address.QuadPart;
252  } else {
253  busEvent->Address = (ULONG64)0;
254  }
255 
256  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_BusErrorGuid;
257  Wnode->SizeDataBlock = FIELD_OFFSET(MSMCAEvent_BusError,
258  RawRecord) + ErrorLogSize;
259 
260  busEvent->Size = ErrorLogSize;
261  *RawPtr = busEvent->RawRecord;
262 
263  } else if (mcaCode & MCA_CODE_MEM_HIERARCHY_ERR_MASK) {
264 
265  //
266  // Memory hierarchy error.
267  //
268 
269  memoryEvent = (PMSMCAEvent_MemoryHierarchyError)Header;
270 
271  memoryEvent->Type = (ULONG)MCA_MEMORYHIERARCHY_ERROR;
272  memoryEvent->TransactionType = ((mcaCode & 0x0000000C) >> 2);
273  memoryEvent->MemoryHierarchyLevel = (mcaCode & 0x00000003);
274  memoryEvent->RequestType = ((mcaCode & 0x000000F0) >> 4);
275 
276  if (McaException->u.Mca.Status.MciStatus.AddressValid) {
277  memoryEvent->Address = McaException->u.Mca.Address.QuadPart;
278  } else {
279  memoryEvent->Address = (ULONG64)0;
280  }
281 
282  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_MemoryHierarchyErrorGuid;
283  Wnode->SizeDataBlock = FIELD_OFFSET(MSMCAEvent_MemoryHierarchyError,
284  RawRecord) + ErrorLogSize;
285 
286  memoryEvent->Size = ErrorLogSize;
287  *RawPtr = memoryEvent->RawRecord;
288 
289  } else if (mcaCode & MCA_CODE_TLB_ERR_MASK) {
290 
291  //
292  // TLB error.
293  //
294 
295  tlbEvent = (PMSMCAEvent_TLBError)Header;
296 
297  tlbEvent->Type = (ULONG)MCA_TLB_ERROR;
298  tlbEvent->TransactionType = ((mcaCode & 0x0000000C) >> 2);
299  tlbEvent->MemoryHierarchyLevel = (mcaCode & 0x00000003);
300 
301  if (McaException->u.Mca.Status.MciStatus.AddressValid) {
302  tlbEvent->Address = McaException->u.Mca.Address.QuadPart;
303  } else {
304  tlbEvent->Address = (ULONG64)0;
305  }
306 
307  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_TLBErrorGuid;
308  Wnode->SizeDataBlock = FIELD_OFFSET(MSMCAEvent_TLBError,
309  RawRecord) + ErrorLogSize;
310 
311  tlbEvent->Size = ErrorLogSize;
312  *RawPtr = tlbEvent->RawRecord;
313 
314  } else {
315 
316  //
317  // No error, unclassified errors, and internal unclassified errors are
318  // treated the same way... we ignore them.
319  //
320  status = STATUS_NOT_SUPPORTED;
321  }
322 
323  return status;
324 }
