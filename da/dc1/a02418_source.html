48 KiExecuteDpc (

51 
52 /*++
53 
54 Routine Description:
55 
56  This function is executed by the DPC thread for each processor. DPC
57  threads are started during kernel initialization after having started
58  all processors and it is determined that the host configuation should
59  execute threaded DPCs in a DPC thread.
60 
61 Arguments:
62 
63  Context - Supplies a pointer to the processor control block for the
64  processor on which the DPC thread is to run.
65 
66 Return Value:
67 
68  None.
69 
70 --*/
71 
72 {
73 
74  PKDPC Dpc;
75  PVOID DeferredContext;
76  PKDEFERRED_ROUTINE DeferredRoutine;
77  PERFINFO_DPC_INFORMATION DpcInformation;
78  PLIST_ENTRY Entry;
79  PLIST_ENTRY ListHead;
80  LOGICAL Logging;
81  KIRQL OldIrql;
82  PKPRCB Prcb;
83  PVOID SystemArgument1;
84  PVOID SystemArgument2;
85  PKTHREAD Thread;
86  LARGE_INTEGER TimeStamp = {0};
87 
88  //
89  // Get PRCB and set the DPC thread address.
90  //
91 
92  Prcb = Context;
93  Thread = KeGetCurrentThread();
94  Prcb->DpcThread = Thread;
95 
96  //
97  // Set the DPC thread priority to the highest level, set the thread
98  // affinity, and enable threaded DPCs on this processor.
99  //
100 
101  KeSetPriorityThread(Thread, HIGH_PRIORITY);
102  KeSetSystemAffinityThread(Prcb->SetMember);
103  Prcb->ThreadDpcEnable = TRUE;
104 
105  //
106  // Loop processing DPC list entries until the specified DPC list is empty.
107  //
108  // N.B. This following code appears to have a redundant loop, but it does
109  // not. The point of this code is to avoid as many dispatch interrupts
110  // as possible.
111  //
112 
113  ListHead = &Prcb->DpcData[DPC_THREADED].DpcListHead;
114  do {
115  Prcb->DpcThreadActive = TRUE;
116 
117  //
118  // If the DPC list is not empty, then process the DPC list.
119  //
120 
121  if (Prcb->DpcData[DPC_THREADED].DpcQueueDepth != 0) {
122 
123  //
124  // Acquire the DPC lock for the current processor and check if
125  // the DPC list is empty. If the DPC list is not empty, then
126  // remove the first entry from the DPC list, capture the DPC
127  // parameters, set the DPC inserted state false, decrement the
128  // DPC queue depth, release the DPC lock, enable interrupts, and
129  // call the specified DPC routine. Otherwise, release the DPC
130  // lock and enable interrupts.
131  //
132 
133  Logging = PERFINFO_IS_GROUP_ON(PERF_DPC);
134  do {
135  KeRaiseIrql(HIGH_LEVEL, &OldIrql);
136  KeAcquireSpinLockAtDpcLevel(&Prcb->DpcData[DPC_THREADED].DpcLock);
137  Entry = ListHead->Flink;
138  if (Entry != ListHead) {
139  RemoveEntryList(Entry);
140  Dpc = CONTAINING_RECORD(Entry, KDPC, DpcListEntry);
141  DeferredRoutine = Dpc->DeferredRoutine;
142  DeferredContext = Dpc->DeferredContext;
143  SystemArgument1 = Dpc->SystemArgument1;
144  SystemArgument2 = Dpc->SystemArgument2;
145  Dpc->DpcData = NULL;
146  Prcb->DpcData[DPC_THREADED].DpcQueueDepth -= 1;
147  KeReleaseSpinLockFromDpcLevel(&Prcb->DpcData[DPC_THREADED].DpcLock);
148  KeLowerIrql(OldIrql);
149 
150  //
151  // If event tracing is enabled, capture the start time.
152  //
153 
154  if (Logging != FALSE) {
155  PerfTimeStamp(TimeStamp);
156  }
157 
158  //
159  // Call the DPC routine.
160  //
161 
162  (DeferredRoutine)(Dpc,
163  DeferredContext,
164  SystemArgument1,
165  SystemArgument2);
166 
167  ASSERT(KeGetCurrentIrql() == PASSIVE_LEVEL);
168 
169  ASSERT(Thread->Affinity == Prcb->SetMember);
170 
171  ASSERT(Thread->Priority == HIGH_PRIORITY);
172 
173  //
174  // If event tracing is enabled, then log the start time
175  // and routine address.
176  //
177 
178  if (Logging != FALSE) {
179  DpcInformation.InitialTime = TimeStamp.QuadPart;
180  DpcInformation.DpcRoutine = (PVOID)(ULONG_PTR)DeferredRoutine;
181  PerfInfoLogBytes(PERFINFO_LOG_TYPE_DPC,
182  &DpcInformation,
183  sizeof(DpcInformation));
184  }
185 
186  } else {
187 
188  ASSERT(Prcb->DpcData[DPC_THREADED].DpcQueueDepth == 0);
189 
190  KeReleaseSpinLockFromDpcLevel(&Prcb->DpcData[DPC_THREADED].DpcLock);
191  KeLowerIrql(OldIrql);
192  }
193 
194  } while (Prcb->DpcData[DPC_THREADED].DpcQueueDepth != 0);
195  }
196 
197  Prcb->DpcThreadActive = FALSE;
198  Prcb->DpcThreadRequested = FALSE;
199  KeMemoryBarrier();
200 
201  //
202  // If the thread DPC list is empty, then wait until the DPC event
203  // for the current processor is set.
204  //
205 
206  if (Prcb->DpcData[DPC_THREADED].DpcQueueDepth == 0) {
207  KeWaitForSingleObject(&Prcb->DpcEvent,
208  Suspended,
209  KernelMode,
210  FALSE,
211  NULL);
212 
213  }
214 
215  } while (TRUE);
216 
217  return;
218 }

220 VOID
221 KiQuantumEnd (
222  VOID
223  )
224 
225 /*++
226 
227 Routine Description:
228 
229  This function is called when a quantum end event occurs on the current
230  processor. Its function is to determine whether the thread priority should
231  be decremented and whether a redispatch of the processor should occur.
232 
233  N.B. This function is called at DISPATCH level and returns at DISPATCH
234  level.
235 
236 Arguments:
237 
238  None.
239 
240 Return Value:
241 
242  None.
243 
244 --*/
245 
246 {
247 
248  PKPRCB Prcb;
249  PKPROCESS Process;
250  PRKTHREAD Thread;
251  PRKTHREAD NewThread;
252 
253  //
254  // If DPC thread activation is requested, then set the DPC event.
255  //
256 
257  Prcb = KeGetCurrentPrcb();
258  Thread = KeGetCurrentThread();
259  if (InterlockedExchange(&Prcb->DpcSetEventRequest, FALSE) == TRUE) {
260  KeSetEvent(&Prcb->DpcEvent, 0, FALSE);
261  }
262 
263  //
264  // Raise IRQL to SYNCH level, acquire the thread lock, and acquire the
265  // PRCB lock.
266  //
267  // If the quantum has expired for the current thread, then update its
268  // quantum and priority.
269  //
270 
271  KeRaiseIrqlToSynchLevel();
272  KiAcquireThreadLock(Thread);
273  KiAcquirePrcbLock(Prcb);
274  if (Thread->Quantum <= 0) {
275 
276  //
277  // If quantum runout is disabled for the thread's process and
278  // the thread is running at a realtime priority, then set the
279  // thread quantum to the highest value and do not round robin
280  // at the thread's priority level. Otherwise, reset the thread
281  // quantum and decay the thread's priority as appropriate.
282  //
283 
284  Process = Thread->ApcState.Process;
285  if ((Process->DisableQuantum != FALSE) &&
286  (Thread->Priority >= LOW_REALTIME_PRIORITY)) {
287 
288  Thread->Quantum = MAXCHAR;
289 
290  } else {
291  Thread->Quantum = Thread->QuantumReset;
292 
293  //
294  // Compute the new thread priority and attempt to reschedule the
295  // current processor.
296  //
297  // N.B. The new priority will never be greater than the previous
298  // priority.
299  //
300 
301  Thread->Priority = KiComputeNewPriority(Thread, 1);
302  if (Prcb->NextThread == NULL) {
303  if ((NewThread = KiSelectReadyThread(Thread->Priority, Prcb)) != NULL) {
304  NewThread->State = Standby;
305  Prcb->NextThread = NewThread;
306  }
307 
308  } else {
309  Thread->Preempted = FALSE;
310  }
311  }
312  }
313 
314  //
315  // Release the thread lock.
316  //
317  // If a thread was scheduled for execution on the current processor, then
318  // acquire the PRCB lock, set the current thread to the new thread, set
319  // next thread to NULL, set the thread state to running, release the PRCB
320  // lock, set the wait reason, ready the old thread, and swap context to
321  // the new thread.
322  //
323 
324  KiReleaseThreadLock(Thread);
325  if (Prcb->NextThread != NULL) {
326  KiSetContextSwapBusy(Thread);
327  NewThread = Prcb->NextThread;
328  Prcb->NextThread = NULL;
329  Prcb->CurrentThread = NewThread;
330  NewThread->State = Running;
331  Thread->WaitReason = WrQuantumEnd;
332  KxQueueReadyThread(Thread, Prcb);
333  Thread->WaitIrql = APC_LEVEL;
334  KiSwapContext(Thread, NewThread);
335 
336  } else {
337  KiReleasePrcbLock(Prcb);
338  }
339 
340  //
341  // Lower IRQL to DISPATCH level and return.
342  //
343 
344  KeLowerIrql(DISPATCH_LEVEL);
345  return;
346 }

503 VOID
504 KiTimerExpiration (
505  IN PKDPC TimerDpc,
506  IN PVOID DeferredContext,
507  IN PVOID SystemArgument1,
508  IN PVOID SystemArgument2
509  )
510 
511 /*++
512 
513 Routine Description:
514 
515  This function is called when the clock interupt routine discovers that
516  a timer has expired.
517 
518  N.B. This function executes on the same processor that receives clock
519  interrupts.
520 
521 Arguments:
522 
523  TimerDpc - Not used.
524 
525  DeferredContext - Not used.
526 
527  SystemArgument1 - Supplies the starting timer table index value to
528  use for the timer table scan.
529 
530  SystemArgument2 - Not used.
531 
532 Return Value:
533 
534  None.
535 
536 --*/
537 
538 {
539 
540  ULARGE_INTEGER CurrentTime;
541  ULONG DpcCount;
542  PKDPC Dpc;
543  DPC_ENTRY DpcTable[MAXIMUM_TIMERS_PROCESSED];
544  KIRQL DummyIrql;
545  LONG HandLimit;
546  LONG Index;
547  LARGE_INTEGER Interval;
548  PLIST_ENTRY ListHead;
549  PKSPIN_LOCK_QUEUE LockQueue;
550  PLIST_ENTRY NextEntry;
551  KIRQL OldIrql;
552  LONG Period;
553 
554 #if !defined(NT_UP) || defined(_WIN64)
555 
556  PKPRCB Prcb = KeGetCurrentPrcb();
557 
558 #endif
559 
560  ULARGE_INTEGER SystemTime;
561  PKTIMER Timer;
562  ULONG TimersExamined;
563  ULONG TimersProcessed;
564 
565  UNREFERENCED_PARAMETER(TimerDpc);
566  UNREFERENCED_PARAMETER(DeferredContext);
567  UNREFERENCED_PARAMETER(SystemArgument2);
568 
569  //
570  // Capture the timer expiration time, the current interrupt time, and
571  // the low tick count.
572  //
573  // N.B. Interrupts are disabled to ensure that interrupt activity on the
574  // current processor does not cause the values read to be skewed.
575  //
576 
577  _disable();
578  KiQuerySystemTime((PLARGE_INTEGER)&SystemTime);
579  KiQueryInterruptTime((PLARGE_INTEGER)&CurrentTime);
580  HandLimit = (LONG)KiQueryLowTickCount();
581  _enable();
582 
583  //
584  // If the timer table has not wrapped, then start with the specified
585  // timer table index value, and scan for timer entries that have expired.
586  // Otherwise, start with the specified timer table index value and scan
587  // the entire table for timer entries that have expired.
588  //
589  // N.B. This later condition exists when DPC processing is blocked for a
590  // period longer than one round trip throught the timer table.
591  //
592  // N.B. The current instance of the timer expiration execution will only
593  // process the timer queue entries specified by the computed index
594  // and hand limit. If another timer expires while the current scan
595  // is in progress, then another scan will occur when the current one
596  // is finished.
597  //
598 
599  Index = PtrToLong(SystemArgument1);
600  if ((ULONG)(HandLimit - Index) >= TIMER_TABLE_SIZE) {
601  HandLimit = Index + TIMER_TABLE_SIZE - 1;
602  }
603 
604  Index -= 1;
605  HandLimit &= (TIMER_TABLE_SIZE - 1);
606 
607  //
608  // Acquire the dispatcher database lock and read the current interrupt
609  // time to determine which timers have expired.
610  //
611 
612  DpcCount = 0;
613  TimersExamined = MAXIMUM_TIMERS_EXAMINED;
614  TimersProcessed = MAXIMUM_TIMERS_PROCESSED;
615  KiLockDispatcherDatabase(&OldIrql);
616  do {
617  Index = (Index + 1) & (TIMER_TABLE_SIZE - 1);
618  ListHead = &KiTimerTableListHead[Index].Entry;
619  while (ListHead != ListHead->Flink) {
620  LockQueue = KiAcquireTimerTableLock(Index);
621  NextEntry = ListHead->Flink;
622  Timer = CONTAINING_RECORD(NextEntry, KTIMER, TimerListEntry);
623  TimersExamined -= 1;
624  if ((NextEntry != ListHead) &&
625  (Timer->DueTime.QuadPart <= CurrentTime.QuadPart)) {
626 
627  //
628  // The next timer in the current timer list has expired.
629  // Remove the entry from the timer tree and set the signal
630  // state of the timer.
631  //
632 
633  TimersProcessed -= 1;
634  KiRemoveEntryTimer(Timer);
635  Timer->Header.Inserted = FALSE;
636  KiReleaseTimerTableLock(LockQueue);
637  Timer->Header.SignalState = 1;
638 
639  //
640  // Capture the DPC and period fields from the timer object.
641  // Once wait test is called, the timer must not be touched
642  // again unless it is periodic. The reason for this is that
643  // a thread may allocate a timer on its local stack and wait
644  // on it. Wait test can cause that thread to immediately
645  // start running on another processor on an MP system. If
646  // the thread returns, then the timer will be corrupted.
647  //
648 
649  Dpc = Timer->Dpc;
650  Period = Timer->Period;
651  if (IsListEmpty(&Timer->Header.WaitListHead) == FALSE) {
652  if (Timer->Header.Type == TimerNotificationObject) {
653  KiWaitTestWithoutSideEffects(Timer, TIMER_EXPIRE_INCREMENT);
654 
655  } else {
656  KiWaitTestSynchronizationObject(Timer, TIMER_EXPIRE_INCREMENT);
657  }
658  }
659 
660  //
661  // If the timer is periodic, then compute the next interval
662  // time and reinsert the timer in the timer tree.
663  //
664  // N.B. Even though the timer insertion is relative, it can
665  // still fail if the period of the timer elapses in
666  // between computing the time and inserting the timer.
667  // If this happens, then the insertion is retried.
668  //
669 
670  if (Period != 0) {
671  Interval.QuadPart = Int32x32To64(Period, - 10 * 1000);
672  do {
673  } while (KiInsertTreeTimer(Timer, Interval) == FALSE);
674  }
675 
676  //
677  // If a DPC is specified, then insert it in the target
678  // processor's DPC queue or capture the parameters in
679  // the DPC table for subsequent execution on the current
680  // processor.
681  //
682 
683  if (Dpc != NULL) {
684 
685 #if defined(NT_UP)
686 
687  DpcTable[DpcCount].Dpc = Dpc;
688  DpcTable[DpcCount].Routine = Dpc->DeferredRoutine;
689  DpcTable[DpcCount].Context = Dpc->DeferredContext;
690  DpcCount += 1;
691 
692 #else
693 
694  if (((Dpc->Number >= MAXIMUM_PROCESSORS) &&
695  (((LONG)Dpc->Number - MAXIMUM_PROCESSORS) != Prcb->Number)) ||
696  ((Dpc->Type == (UCHAR)ThreadedDpcObject) &&
697  (Prcb->ThreadDpcEnable != FALSE))) {
698 
699  KeInsertQueueDpc(Dpc,
700  ULongToPtr(SystemTime.LowPart),
701  ULongToPtr(SystemTime.HighPart));
702 
703  } else {
704  DpcTable[DpcCount].Dpc = Dpc;
705  DpcTable[DpcCount].Routine = Dpc->DeferredRoutine;
706  DpcTable[DpcCount].Context = Dpc->DeferredContext;
707  DpcCount += 1;
708  }
709 
710 #endif
711 
712  }
713 
714  //
715  // If the maximum number of timers have been processed or
716  // the maximum number of timers have been examined, then
717  // drop the dispatcher lock and process the DPC table.
718  //
719 
720  if ((TimersProcessed == 0) || (TimersExamined == 0)) {
721  KiProcessTimerDpcTable(&SystemTime, &DpcTable[0], DpcCount);
722 
723  //
724  // Initialize the DPC count, the scan counters, and
725  // acquire the dispatcher database lock.
726  //
727  // N.B. Control is returned with the dispatcher database
728  // unlocked.
729  //
730 
731  DpcCount = 0;
732  TimersExamined = MAXIMUM_TIMERS_EXAMINED;
733  TimersProcessed = MAXIMUM_TIMERS_PROCESSED;
734 
735 #if defined(_WIN64)
736 
737  if (KiTryToLockDispatcherDatabase(&DummyIrql) == FALSE) {
738  Prcb->TimerHand = 0x100000000I64 + Index;
739  return;
740  }
741 
742 #else
743 
744  KiLockDispatcherDatabase(&DummyIrql);
745 
746 #endif
747 
748  }
749 
750  } else {
751 
752  //
753  // If the timer table list is not empty, then set the due time
754  // to the first entry in the respective timer table entry.
755  //
756  // N.B. On the x86 the write of the due time is not atomic.
757  // Therefore, interrupts must be disabled to synchronize
758  // with the clock interrupt so the clock interupt code
759  // does not observe a partial update of the due time.
760  //
761  // Release the timer table lock.
762  //
763 
764  if (NextEntry != ListHead) {
765 
766  ASSERT(KiTimerTableListHead[Index].Time.QuadPart <= Timer->DueTime.QuadPart);
767 
768 #if defined(_X86_)
769 
770  _disable();
771  KiTimerTableListHead[Index].Time.QuadPart = Timer->DueTime.QuadPart;
772  _enable();
773 
774 #else
775 
776  KiTimerTableListHead[Index].Time.QuadPart = Timer->DueTime.QuadPart;
777 
778 #endif
779 
780  }
781 
782  KiReleaseTimerTableLock(LockQueue);
783 
784  //
785  // If the maximum number of timers have been scanned, then
786  // drop the dispatcher lock and process the DPC table.
787  //
788 
789  if (TimersExamined == 0) {
790  KiProcessTimerDpcTable(&SystemTime, &DpcTable[0], DpcCount);
791 
792  //
793  // Initialize the DPC count, the scan counters, and
794  // acquire the dispatcher database lock.
795  //
796  // N.B. Control is returned with the dispatcher database
797  // unlocked.
798  //
799 
800  DpcCount = 0;
801  TimersExamined = MAXIMUM_TIMERS_EXAMINED;
802  TimersProcessed = MAXIMUM_TIMERS_PROCESSED;
803 
804 #if defined(_WIN64)
805 
806  if (KiTryToLockDispatcherDatabase(&DummyIrql) == FALSE) {
807  Prcb->TimerHand = 0x100000000I64 + Index;
808  return;
809  }
810 
811 #else
812 
813  KiLockDispatcherDatabase(&DummyIrql);
814 
815 #endif
816 
817  }
818 
819  break;
820  }
821  }
822 
823  } while(Index != HandLimit);
824 
825 #if DBG
826 
827  if (KeNumberProcessors == 1) {
828  KiCheckTimerTable(CurrentTime);
829  }
830 
831 #endif
832 
833  //
834  // If the DPC table is not empty, then process the remaining DPC table
835  // entries and lower IRQL. Otherwise, unlock the dispatcher database.
836  //
837  // N.B. Control is returned from the DPC processing routine with the
838  // dispatcher database unlocked.
839  //
