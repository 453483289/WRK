46 /*++
47 
48 Routine Description:
49 
50  A client and server process send datagram messages using this procedure.
51 
52  The message pointed to by the RequestMessage parameter is placed in the
53  message queue of the port connected to the communication port specified
54  by the PortHandle parameter. This service returns an error if PortHandle
55  is invalid or if the MessageId field of the PortMessage structure is
56  non-zero.
57 
58 Arguments:
59 
60  PortHandle - Specifies the handle of the communication port to send
61  the request message to.
62 
63  RequestMessage - Specifies a pointer to the request message. The Type
64  field of the message is set to LPC_DATAGRAM by the service.
65 
66 Return Value:
67 
68  NTSTATUS - A status code that indicates whether or not the operation was
69  successful.
70 
71 --*/

74  PETHREAD CurrentThread;
75  PLPCP_PORT_OBJECT PortObject;
76  PLPCP_PORT_OBJECT QueuePort;
77  PORT_MESSAGE CapturedRequestMessage;
78  ULONG MsgType;
79  KPROCESSOR_MODE PreviousMode;

86  //
87  // Get previous processor mode and validate parameters
88  //
89

91 
92  if (PreviousMode != KernelMode) {
93 
94  try {
95 
96  ProbeForReadSmallStructure( RequestMessage,
97  sizeof( *RequestMessage ),
98  PROBE_ALIGNMENT (PORT_MESSAGE));
99

101  CapturedRequestMessage.u2.s2.Type &= ~LPC_KERNELMODE_MESSAGE;
102 
103  } except( EXCEPTION_EXECUTE_HANDLER ) {
104 
105  return GetExceptionCode();
106  }
107 
108  if (CapturedRequestMessage.u2.s2.Type != 0) {
109 
110  return STATUS_INVALID_PARAMETER;
111  }
112

114 
115  //
116  // This is a kernel mode caller
117  //
118 
119  CapturedRequestMessage = *RequestMessage;
120 
121  if ((CapturedRequestMessage.u2.s2.Type & ~LPC_KERNELMODE_MESSAGE) != 0) {
122 
123  return STATUS_INVALID_PARAMETER;
124  }
125  }
126

128  // Make sure that the caller has given us some data to send
129  //
130

132 
133  return STATUS_INVALID_PARAMETER;
134  }
135

137  // Make sure DataLength is valid with respect to header size and total length
138  //
139

141  ((CLONG)CapturedRequestMessage.u1.s1.TotalLength)) {
142 
143  return STATUS_INVALID_PARAMETER;
144  }
145

147  // Reference the communication port object by handle. Return status if
148  // unsuccessful.
149  //
150 
151  Status = LpcpReferencePortObject( PortHandle,
152  0,
153  PreviousMode,
154  &PortObject );
155 
156  if (!NT_SUCCESS( Status )) {
157 
158  return Status;
159  }
160

162  // Validate the message length
163  //
164

166  ((ULONG)CapturedRequestMessage.u1.s1.TotalLength <= (ULONG)CapturedRequestMessage.u1.s1.DataLength)) {
167 
168  ObDereferenceObject( PortObject );
169 
170  return STATUS_PORT_MESSAGE_TOO_LONG;
171  }

175  // port context if client sending to server. Also validate
176  // length of message being sent.
177  //
178 
179  //
180  // Allocate and initialize the LPC message to send off
181  //

188

193  Msg->PortContext = NULL;

197

199 
200  try {
201 
202  LpcpMoveMessage( &Msg->Request,
203  &CapturedRequestMessage,
204  (RequestMessage + 1),
205  MsgType,
206  &CurrentThread->Cid );
207 
208  } except( EXCEPTION_EXECUTE_HANDLER ) {
209 
210  Status = GetExceptionCode();
211 
212  LpcpFreeToPortZone( Msg, 0 );
213 
214  ObDereferenceObject( PortObject );
215

221 
222  LpcpMoveMessage( &Msg->Request,

224  (RequestMessage + 1),
225  MsgType,

231  // field of the thread and the request message queue. Stamp the
232  // request message with a serial number, insert the message at
233  // the tail of the request message queue and remember the address
234  // of the message in the LpcReplyMessage field for the current thread.
235  //

241  // the port to actually queue the message off to.
242  //
243

245 
246  //
247  // The caller didn't give us a connection port so find the
248  // connection port for this port. If it is null then we'll
249  // fall through without sending a message
250  //
251 
252  QueuePort = PortObject->ConnectedPort;
253 
254  //
255  // Check if the queue port is in process of going away
256  //
257

259

261  // If the port is a client communication port then give the

263  //

275 
276  return STATUS_PORT_DISCONNECTED;

282  // to queue messages.
283  //
284 
285  } else if ((PortObject->Flags & PORT_TYPE) != SERVER_COMMUNICATION_PORT) {
286 
287  ConnectionPort = QueuePort = PortObject->ConnectionPort;
288 
289  if (ConnectionPort == NULL) {
290 
291  LpcpFreeToPortZone( Msg, LPCP_MUTEX_OWNED | LPCP_MUTEX_RELEASE_ON_RETURN );
292 
293  ObDereferenceObject( PortObject );
294 
295  return STATUS_PORT_DISCONNECTED;
296  }
297  }
298 
299  if (ConnectionPort) {
300 
301  ObReferenceObject( ConnectionPort );
302  }
303  }
304 
305  } else {
306 
307  //
308  // The caller supplied a server connection port so that is the port
309  // we queue off to
310  //
311 
312  QueuePort = PortObject;
313  }
314 
315  //
316  // At this point we have an LPC message ready to send and if queue port is
317  // not null then we have a port to actually send the message off to
318  //
319 
320  if (QueuePort != NULL) {
321 
322  //
323  // Reference the QueuePort to prevent this port evaporating under us
324  // Test if the QueuePort isn't in the process of going away
325  // (i.e. we need to have at least 2 references for this object when
326  // ObReferenceObject returns). Note the LPC lock is still held.
327  //
328 
329  if ( ObReferenceObjectSafe( QueuePort ) ) {
330 
331  //
332  // Finish filling in the message and then insert it in the queue
333  //
334 
335  Msg->Request.MessageId = LpcpGenerateMessageId();
336  Msg->Request.CallbackId = 0;
337  Msg->SenderPort = PortObject;
338 
339  CurrentThread->LpcReplyMessageId = 0;
340 
341  InsertTailList( &QueuePort->MsgQueue.ReceiveHead, &Msg->Entry );
342 
343  LpcpTrace(( "%s Send DataGram (%s) Msg %lx [%08x %08x %08x %08x] to Port %lx (%s)\n",
344  PsGetCurrentProcess()->ImageFileName,
345  LpcpMessageTypeName[ Msg->Request.u2.s2.Type & ~LPC_KERNELMODE_MESSAGE ],
346  Msg,
347  *((PULONG)(Msg+1)+0),
348  *((PULONG)(Msg+1)+1),
349  *((PULONG)(Msg+1)+2),
350  *((PULONG)(Msg+1)+3),
351  QueuePort,
352  LpcpGetCreatorName( QueuePort )));
353 
354  //
355  // Release the mutex, increment the request message queue
356  // semaphore by one for the newly inserted request message
357  // then exit the critical region.
358  //
359  // Disable APCs to prevent this thread from being suspended
360  // before being able to release the semaphore.
361  //
362 
363  KeEnterCriticalRegionThread(&CurrentThread->Tcb);
364 
365  LpcpReleaseLpcpLock();
366 
367  KeReleaseSemaphore( QueuePort->MsgQueue.Semaphore,
368  LPC_RELEASE_WAIT_INCREMENT,
369  1L,
370  FALSE );
371 
372  //
373  // If this is a waitable port then we'll need to set the event for
374  // anyone that was waiting on the port
375  //
376 
377  if ( QueuePort->Flags & PORT_WAITABLE ) {
378 
379  KeSetEvent( &QueuePort->WaitEvent,
380  LPC_RELEASE_WAIT_INCREMENT,
381  FALSE );
382  }
383 
384  //
385  // Exit the critical region and release the port object
386  //
387 
388  KeLeaveCriticalRegionThread(&CurrentThread->Tcb);
389 
390  if (ConnectionPort) {
391 
392  ObDereferenceObject( ConnectionPort );
393  }
394 
395  ObDereferenceObject( QueuePort );
396  ObDereferenceObject( PortObject );
397 
398  //
399  // And return to our caller. This is the only successful way out
400  // of this routine
401  //
402 
403  return Status;
404  }
405  }
406 
407  //
408  // At this point we have a message but not a valid port to queue it off
409  // to so we'll free up the port object and release the unused message.
410  //
411 
412  LpcpFreeToPortZone( Msg, LPCP_MUTEX_OWNED | LPCP_MUTEX_RELEASE_ON_RETURN );
413 
414  ObDereferenceObject( PortObject );
415 
416  if (ConnectionPort) {
417 
418  ObDereferenceObject( ConnectionPort );
419  }
420 
421  //
422  // And return the error status to our caller
423  //
424 
425  return STATUS_PORT_DISCONNECTED;
426 }

430 NtRequestWaitReplyPort (
431  __in HANDLE PortHandle,
432  __in PPORT_MESSAGE RequestMessage,
433  __out PPORT_MESSAGE ReplyMessage
434  )
435 
436 /*++
437 
438 Routine Description:
439 
440  A client and server process can send a request and wait for a reply using
441  the NtRequestWaitReplyPort service.
442 
443  If the Type field of the RequestMessage structure is euqal to LPC_REQUEST,
444  then this is identified as a callback request. The ClientId and MessageId
445  fields are used to identify the thread that is waiting for a reply. This
446  thread is unblocked and the current thread that called this service then
447  blocks waiting for a reply.
448 
449  The Type field of the message is set to LPC_REQUEST by the service.
450  Otherwise the Type field of the message must be zero and it will be set to
451  LPC_REQUEST by the service. The message pointed to by the RequestMessage
452  parameter is placed in the message queue of the port connected to the
453  communication port specified by the PortHandle parameter. This service
454  returns an error if PortHandle is invalid. The calling thread then blocks
455  waiting for a reply.
456 
457  The reply message is stored in the location pointed to by the ReplyMessage
458  parameter. The ClientId, MessageId and message type fields will be filled
459  in by the service.
460 
461 Arguments:
462 
463  PortHandle - Specifies the handle of the communication port to send the
464  request message to.
465 
466  RequestMessage - Specifies a pointer to a request message to send.
467 
468  ReplyMessage - Specifies the address of a variable that will receive the
469  reply message. This parameter may point to the same buffer as the
470  RequestMessage parameter.
471 
472 Return Value:
473 
474  NTSTATUS - A status code that indicates whether or not the operation was
475  successful.
476 
477 --*/
478 
479 {
480  PLPCP_PORT_OBJECT PortObject;
481  PLPCP_PORT_OBJECT QueuePort;
482  PLPCP_PORT_OBJECT RundownPort;
483  PORT_MESSAGE CapturedRequestMessage;
484  ULONG MsgType;
485  PKSEMAPHORE ReleaseSemaphore;
486  KPROCESSOR_MODE PreviousMode;
487  NTSTATUS Status;
488  PLPCP_MESSAGE Msg;
489  PETHREAD CurrentThread;
490  PETHREAD WakeupThread;
491  BOOLEAN CallbackRequest;
492  PORT_DATA_INFORMATION CapturedDataInfo;
493  PLPCP_PORT_OBJECT ConnectionPort = NULL;
494  LOGICAL NoImpersonate;
495 
496  PAGED_CODE();
497 
498  //
499  // We cannot wait for a reply if the current thread is exiting
500  //
501 
502  CurrentThread = PsGetCurrentThread();
503 
504  if (CurrentThread->LpcExitThreadCalled) {
505 
506  return STATUS_THREAD_IS_TERMINATING;
507  }
508 
509  //
510  // Get previous processor mode and probe output arguments if necessary.
511  //
512 
513  PreviousMode = KeGetPreviousMode();
514 
515  if (PreviousMode != KernelMode) {
516 
517  try {
518 
519  ProbeForReadSmallStructure( RequestMessage,
520  sizeof( *RequestMessage ),
521  PROBE_ALIGNMENT (PORT_MESSAGE));
522 
523  CapturedRequestMessage = *RequestMessage;
524  CapturedRequestMessage.u2.s2.Type &= ~LPC_KERNELMODE_MESSAGE;
525 
526  ProbeForWriteSmallStructure( ReplyMessage,
527  sizeof( *ReplyMessage ),
528  PROBE_ALIGNMENT (PORT_MESSAGE));
529 
530  //
531  // Make sure that if this message has a data info offset that
532  // the port data information actually fits in the message.
533  //
534  // We first check that the DataInfoOffset doesn't put us beyond
535  // the end of the message.
536  //
537  // Then we capture the data info record and compute a pointer to
538  // the first unused data entry based on the supplied count. If
539  // the start of the message plus its total length doesn't come
540  // up to the first unused data entry then the last valid data
541  // entry doesn't fit in the message buffer. Also if the data
542  // entry pointer that we compute is less than the data info
543  // pointer then we must have wrapped.
544  //
545 
546  if (CapturedRequestMessage.u2.s2.DataInfoOffset != 0) {
547 
548  PPORT_DATA_INFORMATION DataInfo;
549  PPORT_DATA_ENTRY DataEntry;
550 
551  if (((ULONG)CapturedRequestMessage.u2.s2.DataInfoOffset) > (CapturedRequestMessage.u1.s1.TotalLength - sizeof(PORT_DATA_INFORMATION))) {
552 
553  return STATUS_INVALID_PARAMETER;
554  }
555 
556  if ((ULONG)CapturedRequestMessage.u2.s2.DataInfoOffset < sizeof(PORT_MESSAGE)) {
557 
558  return STATUS_INVALID_PARAMETER;
559  }
560 
561  DataInfo = (PPORT_DATA_INFORMATION)(((PUCHAR)RequestMessage) + CapturedRequestMessage.u2.s2.DataInfoOffset);
562 
563  ProbeForReadSmallStructure( DataInfo,
564  sizeof( *DataInfo ),
565  PROBE_ALIGNMENT (PORT_DATA_INFORMATION));
566 
567  CapturedDataInfo = *DataInfo;
568 
569  if (CapturedDataInfo.CountDataEntries > ((CapturedRequestMessage.u1.s1.TotalLength - CapturedRequestMessage.u2.s2.DataInfoOffset) / sizeof(PORT_DATA_ENTRY))) {
570 
571  return STATUS_INVALID_PARAMETER;
572  }
573 
574  DataEntry = &(DataInfo->DataEntries[CapturedDataInfo.CountDataEntries]);
575 
576  if ( ((PUCHAR)DataEntry < (PUCHAR)DataInfo)
577  ||
578  ((((PUCHAR)RequestMessage) + CapturedRequestMessage.u1.s1.TotalLength) < (PUCHAR)DataEntry)) {
579 
580  return STATUS_INVALID_PARAMETER;
581  }
582  }
583 
584  } except( EXCEPTION_EXECUTE_HANDLER ) {
585 
586  return GetExceptionCode();
587  }
588 
589  } else {
590 
591  CapturedRequestMessage = *RequestMessage;
592 
593  if (CapturedRequestMessage.u2.s2.DataInfoOffset != 0) {
594 
595  PPORT_DATA_INFORMATION DataInfo;
596 
597  if (((ULONG)CapturedRequestMessage.u2.s2.DataInfoOffset) > (CapturedRequestMessage.u1.s1.TotalLength - sizeof(PORT_DATA_INFORMATION))) {
598 
599  return STATUS_INVALID_PARAMETER;
600  }
601

603 
604  return STATUS_INVALID_PARAMETER;
605  }
606

608

610  }
611  }
612

615  //

618 
619  NoImpersonate = TRUE;
620  CapturedRequestMessage.u2.s2.Type &= ~LPC_NO_IMPERSONATE;
621 
622  } else {
623 
624  NoImpersonate = FALSE;
625  }

632  // but without the callback.

636

638

640 
641  return STATUS_INVALID_PARAMETER;
642 
643  } else {
644 
645  CapturedRequestMessage.u2.s2.Type |= LPC_REQUEST;
646  CallbackRequest = FALSE;
647  }
648 
649  //
650  // Make sure DataLength is valid with respect to header size and total length
651  //
652 
653  if ((((CLONG)CapturedRequestMessage.u1.s1.DataLength) + sizeof( PORT_MESSAGE )) >
654  ((CLONG)CapturedRequestMessage.u1.s1.TotalLength)) {
655 
656  return STATUS_INVALID_PARAMETER;
657  }
658 
659  //
660  // Reference the communication port object by handle. Return status if
661  // unsuccessful.
662  //
663 
664  Status = LpcpReferencePortObject( PortHandle,
665  0,
666  PreviousMode,
667  &PortObject );
668 
669  if (!NT_SUCCESS( Status )) {
670 
671  return Status;
672  }
673 
674  //
675  // Validate the message length
676  //
677 
678  if (((ULONG)CapturedRequestMessage.u1.s1.TotalLength > PortObject->MaxMessageLength) ||
679  ((ULONG)CapturedRequestMessage.u1.s1.TotalLength <= (ULONG)CapturedRequestMessage.u1.s1.DataLength)) {
680 
681  ObDereferenceObject( PortObject );
682 
683  return STATUS_PORT_MESSAGE_TOO_LONG;
684  }
685 
686  //
687  // Determine which port to queue the message to and get client
688  // port context if client sending to server. Also validate
689  // length of message being sent.
690  //
691 
692  //
693  // Allocate and initialize the LPC message to send off
694  //
695 
696 
697  Msg = (PLPCP_MESSAGE)LpcpAllocateFromPortZone( CapturedRequestMessage.u1.s1.TotalLength );
698 
699  if (Msg == NULL) {
700 
701  ObDereferenceObject( PortObject );
702 
703  return STATUS_NO_MEMORY;
704  }
705 
706  MsgType = CapturedRequestMessage.u2.s2.Type;
707 
708  //
709  // Check if we need to do a callback
710  //
711 
712  if (CallbackRequest) {
713 
714  //
715  // Check for a valid request message id
716  //
717 
718  if (CapturedRequestMessage.MessageId == 0) {
719 
720  LpcpFreeToPortZone( Msg, 0 );
721 
722  ObDereferenceObject( PortObject );
723 
724  return STATUS_INVALID_PARAMETER;
725  }
726 
727  //
728  // Translate the ClientId from the request into a
729  // thread pointer. This is a referenced pointer to keep the thread
730  // from evaporating out from under us.
731  //
732 
733  Status = PsLookupProcessThreadByCid( &CapturedRequestMessage.ClientId,
734  NULL,
735  &WakeupThread );
736 
737  if (!NT_SUCCESS( Status )) {
738 
739  LpcpFreeToPortZone( Msg, 0 );
740 
741  ObDereferenceObject( PortObject );
742 
743  return Status;
744  }
745 
746  //
747  // Acquire the mutex that guards the LpcReplyMessage field of
748  // the thread and get the pointer to the message that the thread
749  // is waiting for a reply to.
750  //
751 
752  LpcpAcquireLpcpLockByThread(CurrentThread);
753 
754  //
755  // See if the thread is waiting for a reply to the message
756  // specified on this call. If not then a bogus message has been
757  // specified, so release the mutex, dereference the thread
758  // and return failure.
759  //
760 
761  if ((WakeupThread->LpcReplyMessageId != CapturedRequestMessage.MessageId)
762 
763  ||
764 
765  ((LpcpGetThreadMessage(WakeupThread) != NULL) &&
766  (LpcpGetThreadMessage(WakeupThread)->Request.u2.s2.Type & ~LPC_KERNELMODE_MESSAGE) != LPC_REQUEST)
767 
768  ||
769 
770  (!LpcpValidateClientPort(WakeupThread, PortObject, LPCP_VALIDATE_REASON_REPLY)) ) {
771 
772  LpcpPrint(( "%s Attempted CallBack Request to Thread %lx (%s)\n",
773  PsGetCurrentProcess()->ImageFileName,
774  WakeupThread,
775  THREAD_TO_PROCESS( WakeupThread )->ImageFileName ));
776 
777  LpcpPrint(( "failed. MessageId == %u Client Id: %x.%x\n",
778  CapturedRequestMessage.MessageId,
779  CapturedRequestMessage.ClientId.UniqueProcess,
780  CapturedRequestMessage.ClientId.UniqueThread ));
781 
782  LpcpPrint(( " Thread MessageId == %u Client Id: %x.%x\n",
783  WakeupThread->LpcReplyMessageId,
784  WakeupThread->Cid.UniqueProcess,
785  WakeupThread->Cid.UniqueThread ));
786

789 
790  DbgBreakPoint();
791  }
792 #endif
793

795 
796  ObDereferenceObject( WakeupThread );
797  ObDereferenceObject( PortObject );
798 
799  return STATUS_REPLY_MESSAGE_MISMATCH;
800  }

803  // Copy over the text of the message
804  //
805 
806  try {
807 
808  LpcpMoveMessage( &Msg->Request,
