57 /*++
58 
59 Routine Description:
60 
61  This is the initialization routine for the Raw file system
62  device driver. This routine creates the device object for the FileSystem
63  device and performs all other driver initialization.
64 
65 Arguments:
66 
67  DriverObject - Pointer to driver object created by the system.
68 
69 Return Value:
70 
71  NTSTATUS - The function value is the final status from the initialization
72  operation.
73 
74 --*/
75 
76 {
77  NTSTATUS Status;
78  UNICODE_STRING NameString;
79 
80  UNREFERENCED_PARAMETER (RegistryPath);
81 
82  //
83  // First create a device object for the Disk file system queue
84  //
85 
86  RtlInitUnicodeString( &NameString, L"\\Device\\RawDisk" );
87  Status = IoCreateDevice( DriverObject,
88  0L,
89  &NameString,
90  FILE_DEVICE_DISK_FILE_SYSTEM,
91  0,
92  FALSE,
93  &RawDeviceDiskObject );
94  if (!NT_SUCCESS( Status )) {
95  return Status;
96  }
97 
98  DriverObject->DriverUnload = RawUnload;
99  //
100  // Now create one for the CD ROM file system queue
101  //
102 
103  RtlInitUnicodeString( &NameString, L"\\Device\\RawCdRom" );
104  Status = IoCreateDevice( DriverObject,
105  0L,
106  &NameString,
107  FILE_DEVICE_CD_ROM_FILE_SYSTEM,
108  0,
109  FALSE,
110  &RawDeviceCdRomObject );
111  if (!NT_SUCCESS( Status )) {
112  IoDeleteDevice (RawDeviceDiskObject);
113  return Status;
114  }
115 
116  //
117  // And now create one for the Tape file system queue
118  //
119 
120  RtlInitUnicodeString( &NameString, L"\\Device\\RawTape" );
121  Status = IoCreateDevice( DriverObject,
122  0L,
123  &NameString,
124  FILE_DEVICE_TAPE_FILE_SYSTEM,
125  0,
126  FALSE,
127  &RawDeviceTapeObject );
128  if (!NT_SUCCESS( Status )) {
129  IoDeleteDevice (RawDeviceCdRomObject);
130  IoDeleteDevice (RawDeviceDiskObject);
131  return Status;
132  }
133 
134  //
135  // Register a shutdown handler to enable us to unregister the file system objects
136  //
137  Status = IoRegisterShutdownNotification (RawDeviceTapeObject);
138  if (!NT_SUCCESS( Status )) {
139  IoDeleteDevice (RawDeviceTapeObject);
140  IoDeleteDevice (RawDeviceCdRomObject);
141  IoDeleteDevice (RawDeviceDiskObject);
142  return Status;
143  }
144  //
145  // Raw does direct IO
146  //
147 
148  RawDeviceDiskObject->Flags |= DO_DIRECT_IO;
149  RawDeviceCdRomObject->Flags |= DO_DIRECT_IO;
150  RawDeviceTapeObject->Flags |= DO_DIRECT_IO;
151 
152  //
153  // Initialize the driver object with this driver's entry points. Note
154  // that only a limited capability is supported by the raw file system.
155  //
156 
157  DriverObject->MajorFunction[IRP_MJ_CREATE] =
158  DriverObject->MajorFunction[IRP_MJ_CLEANUP] =
159  DriverObject->MajorFunction[IRP_MJ_CLOSE] =
160  DriverObject->MajorFunction[IRP_MJ_READ] =
161  DriverObject->MajorFunction[IRP_MJ_WRITE] =
162  DriverObject->MajorFunction[IRP_MJ_QUERY_INFORMATION] =
163  DriverObject->MajorFunction[IRP_MJ_SET_INFORMATION] =
164  DriverObject->MajorFunction[IRP_MJ_QUERY_VOLUME_INFORMATION] =
165  DriverObject->MajorFunction[IRP_MJ_CLEANUP] =
166  DriverObject->MajorFunction[IRP_MJ_FILE_SYSTEM_CONTROL] =
167  DriverObject->MajorFunction[IRP_MJ_DEVICE_CONTROL] =
168  DriverObject->MajorFunction[IRP_MJ_PNP] =
169 
170  (PDRIVER_DISPATCH)RawDispatch;
171  DriverObject->MajorFunction[IRP_MJ_SHUTDOWN] = RawShutdown;
172 
173 
174  //
175  // Finally, register this file system in the system.
176  //
177 
178  IoRegisterFileSystem( RawDeviceDiskObject );
179  IoRegisterFileSystem( RawDeviceCdRomObject );
180  IoRegisterFileSystem( RawDeviceTapeObject );
181  ObReferenceObject (RawDeviceDiskObject);
182  ObReferenceObject (RawDeviceCdRomObject);
183  ObReferenceObject (RawDeviceTapeObject);
184 
185  //
186  // And return to our caller
187  //
188 
189  return( STATUS_SUCCESS );
190 }
