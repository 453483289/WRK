61 /*++
62 
63 Routine Description:
64 
65  This routine fetches the correct esp from a trapframe, accounting
66  for whether the frame is a user or kernel mode frame, and whether
67  it has been edited.
68 
69 Arguments:
70 
71  TrapFrame - Supplies a pointer to a trap frame from which volatile context
72  should be copied into the context record.
73 
74 Return Value:
75 
76  Value of Esp.
77 
78 --*/
79 
80 {
81  if (((TrapFrame->SegCs & MODE_MASK) != KernelMode) ||
82  (TrapFrame->EFlags & EFLAGS_V86_MASK)) {
83 
84  // User mode frame, real value of Esp is always in HardwareEsp.
85 
86  return TrapFrame->HardwareEsp;
87 
88  } else {
89 
90  if ((TrapFrame->SegCs & FRAME_EDITED) == 0) {
91 
92  // Kernel mode frame which has had esp edited,
93  // value of Esp is in TempEsp.
94 
95  return TrapFrame->TempEsp;
96 
97  } else {
98 
99  // Kernel mode frame has has not had esp edited, compute esp.
100 
101  return (ULONG)&TrapFrame->HardwareEsp;
102  }
103  }
104 }

107 KiEspToTrapFrame(
108  IN PKTRAP_FRAME TrapFrame,
109  IN ULONG Esp
110  )
111 
112 /*++
113 
114 Routine Description:
115 
116  This routine sets the specified value Esp into the trap frame,
117  accounting for whether the frame is a user or kernel mode frame,
118  and whether it has been edited before.
119 
120 Arguments:
121 
122  TrapFrame - Supplies a pointer to a trap frame from which volatile context
123  should be copied into the context record.
124 
125  Esp - New value for Esp.
126 
127 Return Value:
128 
129  None.
130 
131 --*/
132 {
133  ULONG OldEsp;
134  KIRQL OldIrql;
135 
136  //
137  // Raise to at least APC_LEVEL so we can test SegCs and EFLAGS
138  // as an atomic operation since NtSetContextThread may modify these.
139  //
140  OldIrql = KeGetCurrentIrql ();
141  if (OldIrql < APC_LEVEL) {
142  KeRaiseIrql (APC_LEVEL, &OldIrql);
143  }
144 
145  OldEsp = KiEspFromTrapFrame(TrapFrame);
146 
147  if (((TrapFrame->SegCs & MODE_MASK) != KernelMode) ||
148  (TrapFrame->EFlags & EFLAGS_V86_MASK)) {
149 
150  //
151  // User mode trap frame
152  //
153 
154  TrapFrame->HardwareEsp = Esp;
155 
156  } else {
157 
158  //
159  // Kernel mode esp can't be lowered or iret emulation will fail
160  //
161 
162  if (Esp < OldEsp)
163  KeBugCheckEx(SET_OF_INVALID_CONTEXT,
164  Esp, OldEsp, (ULONG_PTR)TrapFrame, 0);
165 
166  //
167  // Edit frame, setting edit marker as needed.
168  //
169 
170  if ((TrapFrame->SegCs & FRAME_EDITED) == 0) {
171 
172  // Kernel frame that has already been edited,
173  // store value in TempEsp.
174 
175  TrapFrame->TempEsp = Esp;
176 
177  } else {
178 
179  // Kernel frame for which Esp is being edited first time.
180  // Save real SegCs, set marked in SegCs, save Esp value.
181 
182  if (OldEsp != Esp) {
183 
184  TrapFrame->TempSegCs = TrapFrame->SegCs;
185  TrapFrame->SegCs = TrapFrame->SegCs & ~FRAME_EDITED;
186  TrapFrame->TempEsp = Esp;
187  }
188  }
189  }
190 
191  //
192  // Restore original IRQL
193  //
194  if (OldIrql < APC_LEVEL) {
195  KeLowerIrql (OldIrql);
196  }
197 }

237 VOID
238 KiSegSsToTrapFrame(
239  IN PKTRAP_FRAME TrapFrame,
240  IN ULONG SegSs
241  )
242 
243 /*++
244 
245 Routine Description:
246 
247  It turns out that in a flat system there are only two legal values
248  for SS. Therefore, this procedure forces the appropriate one
249  of those values to be used. The legal SS value is a function of
250  which CS value is already set.
251 
252 Arguments:
253 
254  TrapFrame - Supplies a pointer to a trap frame from which volatile context
255  should be copied into the context record.
256 
257  SegSs - value of SS caller would like to set.
258 
259 Return Value:
260 
261  Nothing.
262 
263 --*/
264 
265 {
266  SegSs &= SEGMENT_MASK; // Throw away the high order trash bits
267 
268  if (TrapFrame->EFlags & EFLAGS_V86_MASK) {
269  TrapFrame->HardwareSegSs = SegSs;
270  } else if ((TrapFrame->SegCs & MODE_MASK) == UserMode) {
271 
272  //
273  // If user mode, we simply put SegSs to trapframe. If the SegSs
274  // is a bogus value. The trap0d handler will be able to detect
275  // this and handle it appropriately.
276  //
277 
278  TrapFrame->HardwareSegSs = SegSs | RPL_MASK;
279  }
280 
281  //
282  // else {
283  // The frame is a kernel mode frame, which does not have
284  // a place to store SS. Therefore, do nothing.
285  //
286 }

295 /*++
296 
297 Routine Description:
298 
299  This routine moves the selected contents of the specified trap and exception frames
300  frames into the specified context frame according to the specified context
301  flags.
302 
303 Arguments:
304 
305  TrapFrame - Supplies a pointer to a trap frame from which volatile context
306  should be copied into the context record.
307 
308  ExceptionFrame - Supplies a pointer to an exception frame from which context
309  should be copied into the context record. This argument is ignored since
310  there is no exception frame on NT386.
311 
312  ContextFrame - Supplies a pointer to the context frame that receives the
313  context copied from the trap and exception frames.
314 
315 Return Value:
316 
317  None.
318 
319 --*/
320 
321 {
322

324  BOOLEAN StateSaved;
325  ULONG i;
326  struct _FPSaveBuffer {
327  UCHAR Buffer[15];
328  FLOATING_SAVE_AREA SaveArea;

331  KIRQL OldIrql;
332 
333  UNREFERENCED_PARAMETER( ExceptionFrame );
334 
335  //
336  // This routine is called at both PASSIVE_LEVEL by exception dispatch
337  // and at APC_LEVEL by NtSetContextThread. We raise to APC_LEVEL to
338  // make the trap frame capture atomic.
339  //
340  OldIrql = KeGetCurrentIrql ();
341  if (OldIrql < APC_LEVEL) {
342  KeRaiseIrql (APC_LEVEL, &OldIrql);
343  }
344 
345  //
346  // Set control information if specified.
347  //
348 
349  if ((ContextFrame->ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL) {
350 
351  //
352  // Set registers ebp, eip, cs, eflag, esp and ss.
353  //
354 
355  ContextFrame->Ebp = TrapFrame->Ebp;
356  ContextFrame->Eip = TrapFrame->Eip;
357 
358  if (((TrapFrame->SegCs & FRAME_EDITED) == 0) &&
359  ((TrapFrame->EFlags & EFLAGS_V86_MASK) == 0)) {
360  ContextFrame->SegCs = TrapFrame->TempSegCs & SEGMENT_MASK;
361  } else {
362  ContextFrame->SegCs = TrapFrame->SegCs & SEGMENT_MASK;
363  }
364  ContextFrame->EFlags = TrapFrame->EFlags;
365  ContextFrame->SegSs = KiSegSsFromTrapFrame(TrapFrame);
366  ContextFrame->Esp = KiEspFromTrapFrame(TrapFrame);
367  }
368 
369  //
370  // Set segment register contents if specified.
371  //
372 
373  if ((ContextFrame->ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS) {
374 
375  //
376  // Set segment registers gs, fs, es, ds.
377  //
378  // These values are junk most of the time, but useful
379  // for debugging under certain conditions. Therefore,
380  // we report whatever was in the frame.
381  //
382  if (TrapFrame->EFlags & EFLAGS_V86_MASK) {
383  ContextFrame->SegGs = TrapFrame->V86Gs & SEGMENT_MASK;
384  ContextFrame->SegFs = TrapFrame->V86Fs & SEGMENT_MASK;
385  ContextFrame->SegEs = TrapFrame->V86Es & SEGMENT_MASK;
386  ContextFrame->SegDs = TrapFrame->V86Ds & SEGMENT_MASK;
387  }
388  else {
389  if (TrapFrame->SegCs == KGDT_R0_CODE) {
390  //
391  // Trap frames created from R0_CODE traps do not save
392  // the following selectors. Set them in the frame now.
393  //
394 
395  TrapFrame->SegGs = 0;
396  TrapFrame->SegFs = KGDT_R0_PCR;
397  TrapFrame->SegEs = KGDT_R3_DATA | RPL_MASK;
398  TrapFrame->SegDs = KGDT_R3_DATA | RPL_MASK;
399  }
400 
401  ContextFrame->SegGs = TrapFrame->SegGs & SEGMENT_MASK;
402  ContextFrame->SegFs = TrapFrame->SegFs & SEGMENT_MASK;
403  ContextFrame->SegEs = TrapFrame->SegEs & SEGMENT_MASK;
404  ContextFrame->SegDs = TrapFrame->SegDs & SEGMENT_MASK;
405  }
406 
407  }
408 
409  //
410  // Set integer register contents if specified.
411  //
412 
413  if ((ContextFrame->ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER) {
414 
415  //
416  // Set integer registers edi, esi, ebx, edx, ecx, eax
417  //
418 
419  ContextFrame->Edi = TrapFrame->Edi;
420  ContextFrame->Esi = TrapFrame->Esi;
421  ContextFrame->Ebx = TrapFrame->Ebx;
422  ContextFrame->Ecx = TrapFrame->Ecx;
423  ContextFrame->Edx = TrapFrame->Edx;
424  ContextFrame->Eax = TrapFrame->Eax;
425  }
426 
427  if (((ContextFrame->ContextFlags & CONTEXT_EXTENDED_REGISTERS) ==
428  CONTEXT_EXTENDED_REGISTERS) &&
429  ((TrapFrame->SegCs & MODE_MASK) == UserMode)) {
430 
431  //
432  // This is the base TrapFrame, and the NpxFrame is on the base
433  // of the kernel stack, just above it in memory.
434  //
435 
436  NpxFrame = (PFX_SAVE_AREA)(TrapFrame + 1);
437 
438  if (KeI386NpxPresent) {
439  KiFlushNPXState (NULL);
440  RtlCopyMemory( (PVOID)&(ContextFrame->ExtendedRegisters[0]),
441  (PVOID)&(NpxFrame->U.FxArea),
442  MAXIMUM_SUPPORTED_EXTENSION
443  );
444  }
445  }
446 
447  //
448  // Fetch floating register contents if requested, and type of target
449  // is user. (system frames have no fp state, so ignore request)
450  //
451  if ( ((ContextFrame->ContextFlags & CONTEXT_FLOATING_POINT) ==
452  CONTEXT_FLOATING_POINT) &&
453  ((TrapFrame->SegCs & MODE_MASK) == UserMode)) {
454 
455  //
456  // This is the base TrapFrame, and the NpxFrame is on the base
457  // of the kernel stack, just above it in memory.
458  //
459 
460  NpxFrame = (PFX_SAVE_AREA)(TrapFrame + 1);
461 
462  if (KeI386NpxPresent) {
463 
464  //
465  // Force the coprocessors state to the save area and copy it
466  // to the context frame.
467  //
468 
469  if (KeI386FxsrPresent == TRUE) {
470 
471  //
472  // FP state save was done using fxsave. Get the save
473  // area in fnsave format
474  //
475  // Save area must be 16 byte aligned so we cushion it with
476  // 15 bytes (in the locals declaration above) and round
477  // down to align.
478  //
479 
480  ULONG_PTR Temp;
481  Temp = (ULONG_PTR)&FloatSaveBuffer.SaveArea;
482  Temp &= ~0xf;
483  PSaveArea = (PFLOATING_SAVE_AREA)Temp;
484  KiFlushNPXState (PSaveArea);
485  } else {
486 
487  PSaveArea = (PFLOATING_SAVE_AREA)&(NpxFrame->U.FnArea);
488  KiFlushNPXState (NULL);
489 
490  }
491 
492  ContextFrame->FloatSave.ControlWord = PSaveArea->ControlWord;
493  ContextFrame->FloatSave.StatusWord = PSaveArea->StatusWord;
494  ContextFrame->FloatSave.TagWord = PSaveArea->TagWord;
495  ContextFrame->FloatSave.ErrorOffset = PSaveArea->ErrorOffset;
496  ContextFrame->FloatSave.ErrorSelector = PSaveArea->ErrorSelector;
497  ContextFrame->FloatSave.DataOffset = PSaveArea->DataOffset;
498  ContextFrame->FloatSave.DataSelector = PSaveArea->DataSelector;
499  ContextFrame->FloatSave.Cr0NpxState = NpxFrame->Cr0NpxState;
500 
501  for (i = 0; i < SIZE_OF_80387_REGISTERS; i++) {
502  ContextFrame->FloatSave.RegisterArea[i] = PSaveArea->RegisterArea[i];
503  }
504 
505  } else {
506 
507  //
508  // The 80387 is being emulated by the R3 emulator.
509  // ** The only time the Npx state is ever obtained or set is
510  // ** for userlevel handling. Current Irql must be 0 or 1.
511  // Go slurp the emulator's R3 data and generate the
512  // floating point context
513  //
514 
515  StateSaved = KiEm87StateToNpxFrame(&ContextFrame->FloatSave);
516  if (StateSaved) {
517  ContextFrame->FloatSave.Cr0NpxState = NpxFrame->Cr0NpxState;
518  } else {
519 
520  //
521  // The floatingpoint state can not be determined.
522  // Remove the floatingpoint flag from the context frame flags.
523  //
524 
525  ContextFrame->ContextFlags &= (~CONTEXT_FLOATING_POINT) | CONTEXT_i386;
526  }
527  }
528  }
529 
530  //
531  // Fetch Dr register contents if requested.
532  //
533 
534  if ((ContextFrame->ContextFlags & CONTEXT_DEBUG_REGISTERS) ==
535  CONTEXT_DEBUG_REGISTERS) {
536 
537  //
538  // Care is now taken to ensure that the DebugActive/Dr7 value is set on
539  // any valid set of a legal DR value, ensuring the values on the kernel
540  // stack cannot become trash.
541  //
542 
543  if ((TrapFrame->Dr7 & ~DR7_RESERVED_MASK) != 0) {
544  ContextFrame->Dr0 = TrapFrame->Dr0;
545  ContextFrame->Dr1 = TrapFrame->Dr1;
546  ContextFrame->Dr2 = TrapFrame->Dr2;
547  ContextFrame->Dr3 = TrapFrame->Dr3;
548  ContextFrame->Dr6 = TrapFrame->Dr6;
549  ContextFrame->Dr7 = KiUpdateDr7 (TrapFrame->Dr7);
550  } else {
551  ContextFrame->Dr0 = 0;
552  ContextFrame->Dr1 = 0;
553  ContextFrame->Dr2 = 0;
554  ContextFrame->Dr3 = 0;
555  ContextFrame->Dr6 = 0;
556  ContextFrame->Dr7 = 0;
557  }
558  }
559 
560  //
561  // Lower IRQL if we had to raise it
562  //
563  if (OldIrql < APC_LEVEL) {
564  KeLowerIrql (OldIrql);
565  }

569 KeContextToKframes (
570  __inout PKTRAP_FRAME TrapFrame,
571  __inout PKEXCEPTION_FRAME ExceptionFrame,
572  __in PCONTEXT ContextFrame,
573  __in ULONG ContextFlags,
574  __in KPROCESSOR_MODE PreviousMode
575  )

577 /*++
578 
579 Routine Description:
580 
581  This routine moves the selected contents of the specified context frame into
582  the specified trap and exception frames according to the specified context
583  flags.
584 
585 Arguments:
586 
587  TrapFrame - Supplies a pointer to a trap frame that receives the volatile
588  context from the context record.
589 
590  ExceptionFrame - Supplies a pointer to an exception frame that receives
591  the nonvolatile context from the context record. This argument is
592  ignored since there is no exception frame on NT386.
593 
594  ContextFrame - Supplies a pointer to a context frame that contains the
595  context that is to be copied into the trap and exception frames.
596 
597  ContextFlags - Supplies the set of flags that specify which parts of the
598  context frame are to be copied into the trap and exception frames.
599 
600  PreviousMode - Supplies the processor mode for which the trap and exception
601  frames are being built.
602 
603 Return Value:
604 
605  None.
606 
607 --*/
608 
609 {
610 
611  PFX_SAVE_AREA NpxFrame;
612  ULONG i;
613  ULONG j;
614  ULONG TagWord;
615  BOOLEAN StateSaved;
616  BOOLEAN ModeChanged;
617  KIRQL OldIrql;
618 #if DBG
619  PKPCR Pcr;
620 #endif
621 #if DBG
622  KIRQL OldIrql2;
623 #endif
624 
625  UNREFERENCED_PARAMETER( ExceptionFrame );
626 
627  //
628  // This routine is called at both PASSIVE_LEVEL by exception dispatch
629  // and at APC_LEVEL by NtSetContextThread. We raise to APC_LEVEL to
630  // make the trap frame modifications atomic.
631  //
632  OldIrql = KeGetCurrentIrql ();
633  if (OldIrql < APC_LEVEL) {
634  KeRaiseIrql (APC_LEVEL, &OldIrql);
635  }
636 
637  //
638  // Set control information if specified.
639  //
640 
641  if ((ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL) {
642 
643  if ((ContextFrame->EFlags & EFLAGS_V86_MASK) !=
644  (TrapFrame->EFlags & EFLAGS_V86_MASK)) {
645  ModeChanged = TRUE;
646  } else {
647  ModeChanged = FALSE;
648  }
649 
650 
651  //
652  // Set registers eflag, ebp, eip, cs, esp and ss.
653  // Eflags is set first, so that the auxiliary routines
654  // can check the v86 bit to determine as well as cs, to
655  // determine if the frame is kernel or user mode. (v86 mode cs
656  // can have any value)
657  //
658 
659  TrapFrame->EFlags = SANITIZE_FLAGS(ContextFrame->EFlags, PreviousMode);
660  TrapFrame->Ebp = ContextFrame->Ebp;
661  TrapFrame->Eip = ContextFrame->Eip;
662  if (TrapFrame->EFlags & EFLAGS_V86_MASK) {
663  TrapFrame->SegCs = ContextFrame->SegCs;
664  } else {
665  TrapFrame->SegCs = SANITIZE_SEG(ContextFrame->SegCs, PreviousMode);
666  if (PreviousMode != KernelMode && TrapFrame->SegCs < 8) {
667 
668  //
669  // If user mode and the selector value is less than 8, we
670  // know it is an invalid selector. Set it to flat user
671  // mode selector. Another reason we need to check for this
672  // is that any cs value less than 8 causes our exit kernel
673  // macro to treat its exit trap frame as an edited frame.
674  //
675 
676  TrapFrame->SegCs = KGDT_R3_CODE | RPL_MASK;
677  }
678  }
679 
680 
681  KiSegSsToTrapFrame(TrapFrame, ContextFrame->SegSs);
682  KiEspToTrapFrame(TrapFrame, ContextFrame->Esp);
683  if (ModeChanged) {
684  Ki386AdjustEsp0(TrapFrame); // realign esp0 in the tss
685  }
686  }
687 
688  //
689  // Set segment register contents if specified.
690  //
691 
692  if ((ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS) {
693 
694  //
695  // Set segment registers gs, fs, es, ds.
696  //
697 
698  //
699  // There's only one legal value for DS and ES, so simply set it.
700  // This allows KeContextFromKframes to report the real values in
701  // the frame. (which are junk most of the time, but sometimes useful
702  // for debugging.)
703  // Only 2 legal values for FS, let either one be set.
704  // Force GS to be 0 to deal with entry via SysCall and exit
705  // via exception.
706  //
707  // For V86 mode, the FS, GS, DS, and ES registers must be properly
708  // set from the supplied context.
709  //
710 
711  if (TrapFrame->EFlags & EFLAGS_V86_MASK) {
712  TrapFrame->V86Fs = ContextFrame->SegFs;
713  TrapFrame->V86Es = ContextFrame->SegEs;
714  TrapFrame->V86Ds = ContextFrame->SegDs;
715  TrapFrame->V86Gs = ContextFrame->SegGs;
716  } else if (((TrapFrame->SegCs & MODE_MASK) == KernelMode)) {
717 
718  //
719  // set up the standard selectors
720  //
721 
722  TrapFrame->SegFs = SANITIZE_SEG(ContextFrame->SegFs, PreviousMode);
723  TrapFrame->SegEs = KGDT_R3_DATA | RPL_MASK;
724  TrapFrame->SegDs = KGDT_R3_DATA | RPL_MASK;
725  TrapFrame->SegGs = 0;
726  } else {
727 
728  //
729  // If user mode, we simply return whatever left in context frame
730  // and let trap 0d handle it (if later we trap while popping the
731  // trap frame.) V86 mode also get handled here.
732  //
733 
734  TrapFrame->SegFs = ContextFrame->SegFs;
735  TrapFrame->SegEs = ContextFrame->SegEs;
736  TrapFrame->SegDs = ContextFrame->SegDs;
737  if (TrapFrame->SegCs == (KGDT_R3_CODE | RPL_MASK)) {
738  TrapFrame->SegGs = 0;
739  } else {
740  TrapFrame->SegGs = ContextFrame->SegGs;
741  }
742  }
743  }
744  //
745  // Set integer registers contents if specified.
746  //
747 
748  if ((ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER) {
749 
750  //
751  // Set integer registers edi, esi, ebx, edx, ecx, eax.
752  //
753  // Can NOT call RtlCopyMemory here because the regs aren't
754  // contiguous in pusha frame, and we don't want to export
755  // bits of junk into context record.
756  //
757 
758  TrapFrame->Edi = ContextFrame->Edi;
759  TrapFrame->Esi = ContextFrame->Esi;
760  TrapFrame->Ebx = ContextFrame->Ebx;
761  TrapFrame->Ecx = ContextFrame->Ecx;
762  TrapFrame->Edx = ContextFrame->Edx;
763  TrapFrame->Eax = ContextFrame->Eax;
764 
765  }
766 
767  //
768  // Set extended register contents if requested, and type of target

806  (CR0_EM | CR0_MP);
807  }
808  }

813  // is user. (system frames have no fp state, so ignore request)

820  // This is the base TrapFrame, and the NpxFrame is on the base
821  // of the kernel stack, just above it in memory.
822  //
823

825

827

830  //

838  //
839

842  NpxFrame->U.FxArea.StatusWord =

847  //

854

856

872  ContextFrame->FloatSave.DataOffset;

874  ContextFrame->FloatSave.DataSelector;

879  //

881  SIZE_OF_FX_REGISTERS

903  ContextFrame->FloatSave.DataOffset;

905  ContextFrame->FloatSave.DataSelector;

908  NpxFrame->U.FnArea.RegisterArea[i] =
909  ContextFrame->FloatSave.RegisterArea[i];
910  }

912  }

915  // Make sure only valid floating state bits are moved to Cr0NpxState.
916  //
917 
918  NpxFrame->Cr0NpxState &= ~(CR0_EM | CR0_MP | CR0_TS);
919 
920  //
921  // Only let VDMs turn on the EM bit. The kernel can't do
922  // anything for FLAT apps

927  }

930

935  // turn on/off it's CR0_EM bit.
936  //

940  VdmUserCr0MapIn[ContextFrame->FloatSave.Cr0NpxState & (CR0_EM | CR0_MP)];

946  // ** The only time the Npx state is ever obtained or set is
947  // ** for userlevel handling. Current Irql must be 0 or 1.
948  // And the context being set must be for the current thread.
949  // Go smash the floatingpoint context into the R3 emulator's
950  // data area.
951  //

954  Pcr = KeGetPcr();

957 #endif

965  // resetting CR0_EM.
966  //
967

969  NpxFrame->Cr0NpxState |=
970  ContextFrame->FloatSave.Cr0NpxState & CR0_MP;
971  }
972  }
973  }
974  }

981  //

986  ULONG DrValue;
987 
988  NewMask = 0;

992  //
993

995  DrValue = SANITIZE_DRADDR (*((PULONG)((ULONG_PTR)ContextFrame +
996  KiDebugRegisterContextOffsets[Index])), PreviousMode);
997 
998  if ((*((PULONG)((ULONG_PTR)TrapFrame + KiDebugRegisterTrapOffsets[Index])) = DrValue) != 0)
999  {
1000  NewMask |= DR_MASK (Index);
1001  }
1002  }
1003 
1004  if ((TrapFrame->Dr6 = SANITIZE_DR6 (ContextFrame->Dr6, PreviousMode)) != 0) {
1005  NewMask |= DR_MASK (6);
1006  }
1007 
1008  TrapFrame->Dr7 = SANITIZE_DR7(ContextFrame->Dr7, PreviousMode);
1009  (VOID) KiRecordDr7 (&TrapFrame->Dr7, &NewMask);
1010 
1011  if (PreviousMode != KernelMode) {
1012  KeGetCurrentThread()->Header.DebugActive = (BOOLEAN) NewMask;
1013  }
1014  }
1015 
1016  //
1017  // If thread is supposed to have IOPL, then force it on in eflags
1018  //
1019  if (KeGetCurrentThread()->Iopl) {
1020  TrapFrame->EFlags |= (EFLAGS_IOPL_MASK & -1); // IOPL = 3
1021  }
1022 
1023  //
1024  // Lower IRQL if we had to raise it
1025  //
1026  if (OldIrql < APC_LEVEL) {
1027  KeLowerIrql (OldIrql);
1028  }
1029 
1030  return;
1031 }

1038  IN KPROCESSOR_MODE PreviousMode,
1039  IN BOOLEAN FirstChance

1042 /*++
1043 
1044 Routine Description:
1045 
1046  This function is called to dispatch an exception to the proper mode and
1047  to cause the exception dispatcher to be called. If the previous mode is
1048  kernel, then the exception dispatcher is called directly to process the
1049  exception. Otherwise the exception record, exception frame, and trap
1050  frame contents are copied to the user mode stack. The contents of the
1051  exception frame and trap are then modified such that when control is
1052  returned, execution will commense in user mode in a routine which will
1053  call the exception dispatcher.
1054 
1055 Arguments:
1056 
1057  ExceptionRecord - Supplies a pointer to an exception record.
1058 
1059  ExceptionFrame - Supplies a pointer to an exception frame. For NT386,
1060  this should be NULL.
1061 
1062  TrapFrame - Supplies a pointer to a trap frame.
1063 
1064  PreviousMode - Supplies the previous processor mode.
1065 
1066  FirstChance - Supplies a boolean value that specifies whether this is
1067  the first (TRUE) or second (FALSE) chance for the exception.
1068 
1069 Return Value:
1070 
1071  None.
1072 
1073 --*/
1074 
1075 {
1076  CONTEXT ContextFrame;
1077  EXCEPTION_RECORD ExceptionRecord1, ExceptionRecord2;
1078  LONG Length;
1079  ULONG UserStack1;
1080  ULONG UserStack2;
1081 
1082  //
1083  // Move machine state from trap and exception frames to a context frame,
1084  // and increment the number of exceptions dispatched.
1085  //
1086 
1087  KeGetCurrentPrcb()->KeExceptionDispatchCount += 1;
1088  ContextFrame.ContextFlags = CONTEXT_FULL | CONTEXT_DEBUG_REGISTERS;
1089 
1090  if ((PreviousMode == UserMode) || KdDebuggerEnabled) {
1091  //
1092  // For usermode exceptions always try to dispatch the floating
1093  // point state. This allows exception handlers & debuggers to
1094  // examine/edit the npx context if required. Plus it allows
1095  // exception handlers to use fp instructions without destroying
1096  // the npx state at the time of the exception.
1097  //
1098  // Note: If there's no 80387, ContextTo/FromKFrames will use the
1099  // emulator's current state. If the emulator can not give the
1100  // current state, then the context_floating_point bit will be
1101  // turned off by ContextFromKFrames.
1102  //
1103 
1104  ContextFrame.ContextFlags |= CONTEXT_FLOATING_POINT;
1105  if (KeI386XMMIPresent) {
1106  ContextFrame.ContextFlags |= CONTEXT_EXTENDED_REGISTERS;
1107  }
1108  }
1109 
1110  KeContextFromKframes(TrapFrame, ExceptionFrame, &ContextFrame);
1111 
1112  //
1113  // if it is BREAK_POINT exception, we subtract 1 from EIP and report
1114  // the updated EIP to user. This is because Cruiser requires EIP
1115  // points to the int 3 instruction (not the instruction following int 3).
1116  // In this case, BreakPoint exception is fatal. Otherwise we will step
1117  // on the int 3 over and over again, if user does not handle it
1118  //
1119  // if the BREAK_POINT occured in V86 mode, the debugger running in the
1120  // VDM will expect CS:EIP to point after the exception (the way the
1121  // processor left it. this is also true for protected mode dos
1122  // app debuggers. We will need a way to detect this.
1123  //
1124  //
1125 
1126  switch (ExceptionRecord->ExceptionCode) {
1127  case STATUS_BREAKPOINT:
1128  ContextFrame.Eip--;
1129  break;
1130 
1131  case KI_EXCEPTION_ACCESS_VIOLATION:
1132  ExceptionRecord->ExceptionCode = STATUS_ACCESS_VIOLATION;
1133  if (PreviousMode == UserMode) {
1134  if (KiCheckForAtlThunk(ExceptionRecord,&ContextFrame) != FALSE) {
1135  goto Handled1;
1136  }
1137 
1138  if ((SharedUserData->ProcessorFeatures[PF_NX_ENABLED] == TRUE) &&
1139  (ExceptionRecord->ExceptionInformation [0] == EXCEPTION_EXECUTE_FAULT)) {
1140 
1141  if (((KeFeatureBits & KF_GLOBAL_32BIT_EXECUTE) != 0) ||
1142  (PsGetCurrentProcess()->Pcb.Flags.ExecuteEnable != 0) ||
1143  (((KeFeatureBits & KF_GLOBAL_32BIT_NOEXECUTE) == 0) &&
1144  (PsGetCurrentProcess()->Pcb.Flags.ExecuteDisable == 0))) {
1145  ExceptionRecord->ExceptionInformation [0] = 0;
1146  }
1147  }
1148  }
1149  break;
1150  }
1151 
1152  //
1153  // Select the method of handling the exception based on the previous mode.
1154  //
1155 
1156  ASSERT ((
1157  !((PreviousMode == KernelMode) &&
1158  (ContextFrame.EFlags & EFLAGS_V86_MASK))
1159  ));
1160 
1161  if (PreviousMode == KernelMode) {
1162 
1163  //
1164  // Previous mode was kernel.
1165  //
1166  // If the kernel debugger is active, then give the kernel debugger the
1167  // first chance to handle the exception. If the kernel debugger handles
1168  // the exception, then continue execution. Else attempt to dispatch the
1169  // exception to a frame based handler. If a frame based handler handles
1170  // the exception, then continue execution.
1171  //
1172  // If a frame based handler does not handle the exception,
1173  // give the kernel debugger a second chance, if it's present.
1174  //
1175  // If the exception is still unhandled, call KeBugCheck().
1176  //
1177 
1178  if (FirstChance == TRUE) {
1179 
1180  if ((KiDebugRoutine != NULL) &&
1181  (((KiDebugRoutine) (TrapFrame,
1182  ExceptionFrame,
1183  ExceptionRecord,
1184  &ContextFrame,
1185  PreviousMode,
1186  FALSE)) != FALSE)) {
1187 
1188  goto Handled1;
1189  }
1190 
1191  // Kernel debugger didn't handle exception.
1192 
1193  if (RtlDispatchException(ExceptionRecord, &ContextFrame) == TRUE) {
1194  goto Handled1;
1195  }
1196  }
1197 
1198  //
1199  // This is the second chance to handle the exception.
1200  //
1201 
1202  if ((KiDebugRoutine != NULL) &&
1203  (((KiDebugRoutine) (TrapFrame,
1204  ExceptionFrame,
1205  ExceptionRecord,
1206  &ContextFrame,
1207  PreviousMode,
1208  TRUE)) != FALSE)) {
1209 
1210  goto Handled1;
1211  }
1212 
1213  KeBugCheckEx(
1214  KERNEL_MODE_EXCEPTION_NOT_HANDLED,
1215  ExceptionRecord->ExceptionCode,
1216  (ULONG)ExceptionRecord->ExceptionAddress,
1217  (ULONG)TrapFrame,
1218  0);
1219 
1220  } else {
1221 
1222  //
1223  // Previous mode was user.
1224  //
1225  // If this is the first chance and the current process has a debugger
1226  // port, then send a message to the debugger port and wait for a reply.
1227  // If the debugger handles the exception, then continue execution. Else
1228  // transfer the exception information to the user stack, transition to
1229  // user mode, and attempt to dispatch the exception to a frame based
1230  // handler. If a frame based handler handles the exception, then continue
1231  // execution with the continue system service. Else execute the
1232  // NtRaiseException system service with FirstChance == FALSE, which
1233  // will call this routine a second time to process the exception.
1234  //
1235  // If this is the second chance and the current process has a debugger
1236  // port, then send a message to the debugger port and wait for a reply.
1237  // If the debugger handles the exception, then continue execution. Else
1238  // if the current process has a subsystem port, then send a message to
1239  // the subsystem port and wait for a reply. If the subsystem handles the
1240  // exception, then continue execution. Else terminate the process.
1241  //
1242 
1243 
1244  if (FirstChance == TRUE) {
1245 
1246  //
1247  // This is the first chance to handle the exception.
1248  //
1249 
1250  if ((KiDebugRoutine != NULL) &&
1251  ((PsGetCurrentProcess()->DebugPort == NULL &&
1252  !KdIgnoreUmExceptions) ||
1253  (KdIsThisAKdTrap(ExceptionRecord, &ContextFrame, UserMode)))) {
1254  //
1255  // Now dispatch the fault to the kernel debugger.
1256  //
1257 
1258  if ((((KiDebugRoutine) (TrapFrame,
1259  ExceptionFrame,
1260  ExceptionRecord,
1261  &ContextFrame,
1262  PreviousMode,
1263  FALSE)) != FALSE)) {
1264 
1265  goto Handled1;
1266  }
1267  }
1268 
1269  if (DbgkForwardException(ExceptionRecord, TRUE, FALSE)) {
1270  goto Handled2;
1271  }
1272 
1273  //
1274  // Transfer exception information to the user stack, transition
1275  // to user mode, and attempt to dispatch the exception to a frame
1276  // based handler.
1277 
1278  ExceptionRecord1.ExceptionCode = 0; // satisfy no_opt compilation
1279 
1280  repeat:
1281  try {
1282 
1283  //
1284  // If the SS segment is not 32 bit flat, there is no point
1285  // to dispatch exception to frame based exception handler.
1286  //
1287 
1288  if (TrapFrame->HardwareSegSs != (KGDT_R3_DATA | RPL_MASK) ||
1289  TrapFrame->EFlags & EFLAGS_V86_MASK ) {
1290  ExceptionRecord2.ExceptionCode = STATUS_ACCESS_VIOLATION;
1291  ExceptionRecord2.ExceptionFlags = 0;
1292  ExceptionRecord2.NumberParameters = 0;
1293  ExRaiseException(&ExceptionRecord2);
1294  }
1295 
1296  //
1297  // Compute length of context record and new aligned user stack
1298  // pointer.
1299  //
1300 
1301  UserStack1 = (ContextFrame.Esp & ~CONTEXT_ROUND) - CONTEXT_ALIGNED_SIZE;
1302 
1303  //
1304  // Probe user stack area for writability and then transfer the
1305  // context record to the user stack.
1306  //
1307 
1308  ProbeForWrite((PCHAR)UserStack1, CONTEXT_ALIGNED_SIZE, CONTEXT_ALIGN);
1309  RtlCopyMemory((PULONG)UserStack1, &ContextFrame, sizeof(CONTEXT));
1310 
1311  //
1312  // Compute length of exception record and new aligned stack
1313  // address.
1314  //
1315 
1316  Length = (sizeof(EXCEPTION_RECORD) - (EXCEPTION_MAXIMUM_PARAMETERS -
1317  ExceptionRecord->NumberParameters) * sizeof(ULONG) +3) &
1318  (~3);

1320

1322  // Probe user stack area for writeability and then transfer the
1323  // context record to the user stack area.
1324  // N.B. The probing length is Length+8 because there are two
1325  // arguments need to be pushed to user stack later.
1326  //
1327 
1328  ProbeForWrite((PCHAR)(UserStack2 - 8), Length + 8, sizeof(ULONG));
1329  RtlCopyMemory((PULONG)UserStack2, ExceptionRecord, Length);

1332  // Push address of exception record, context record to the
1333  // user stack. They are the two parameters required by
1334  // _KiUserExceptionDispatch.
1335  //
1336 
1337  *(PULONG)(UserStack2 - sizeof(ULONG)) = UserStack1;
1338  *(PULONG)(UserStack2 - 2*sizeof(ULONG)) = UserStack2;

1341  // Set new stack pointer to the trap frame.
1342  //
1343 
1344  KiSegSsToTrapFrame(TrapFrame, KGDT_R3_DATA);
1345  KiEspToTrapFrame(TrapFrame, (UserStack2 - sizeof(ULONG)*2));

1348  // Force correct R3 selectors into TrapFrame.
1349  //
1350

1352  TrapFrame->SegDs = SANITIZE_SEG(KGDT_R3_DATA, PreviousMode);
1353  TrapFrame->SegEs = SANITIZE_SEG(KGDT_R3_DATA, PreviousMode);
1354  TrapFrame->SegFs = SANITIZE_SEG(KGDT_R3_TEB, PreviousMode);
1355  TrapFrame->SegGs = 0;

1358  // Set the address of the exception routine that will call the
1359  // exception dispatcher and then return to the trap handler.
1360  // The trap handler will restore the exception and trap frame
1361  // context and continue execution in the routine that will
1362  // call the exception dispatcher.
1363  //
1364 
1365  TrapFrame->Eip = (ULONG)KeUserExceptionDispatcher;
1366  return;
1367 
1368  } except (KiCopyInformation(&ExceptionRecord1,
1369  (GetExceptionInformation())->ExceptionRecord)) {
1370 
1371  //
1372  // If the exception is a stack overflow, then attempt
1373  // to raise the stack overflow exception. Otherwise,
1374  // the user's stack is not accessible, or is misaligned,
1375  // and second chance processing is performed.
1376  //
1377 
1378  if (ExceptionRecord1.ExceptionCode == STATUS_STACK_OVERFLOW) {
1379  ExceptionRecord1.ExceptionAddress = ExceptionRecord->ExceptionAddress;
1380  RtlCopyMemory((PVOID)ExceptionRecord,
1381  &ExceptionRecord1, sizeof(EXCEPTION_RECORD));
1382  goto repeat;
1383  }
1384  }
1385  }
1386 
1387  //
1388  // This is the second chance to handle the exception.
1389  //
1390 
1391  if (DbgkForwardException(ExceptionRecord, TRUE, TRUE)) {
1392  goto Handled2;
1393  } else if (DbgkForwardException(ExceptionRecord, FALSE, TRUE)) {
1394  goto Handled2;
1395  } else {
1396  ZwTerminateProcess(NtCurrentProcess(), ExceptionRecord->ExceptionCode);
1397  KeBugCheckEx(
1398  KERNEL_MODE_EXCEPTION_NOT_HANDLED,
1399  ExceptionRecord->ExceptionCode,
1400  (ULONG)ExceptionRecord->ExceptionAddress,
1401  (ULONG)TrapFrame,
1402  0);
1403  }
1404  }
1405 
1406  //
1407  // Move machine state from context frame to trap and exception frames and
1408  // then return to continue execution with the restored state.
1409  //
1410 
1411 Handled1:
1412 
1413  KeContextToKframes(TrapFrame, ExceptionFrame, &ContextFrame,
1414  ContextFrame.ContextFlags, PreviousMode);
1415 
1416  //
1417  // Exception was handled by the debugger or the associated subsystem
1418  // and state was modified, if necessary, using the get state and set
1419  // state capabilities. Therefore the context frame does not need to
1420  // be transferred to the trap and exception frames.
1421  //
1422 
1423 Handled2:
1424  return;
1425 }
