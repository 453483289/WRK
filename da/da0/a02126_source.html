46 /*++
47 
48 Routine Description:
49 
50  This function probes a structure for write accessibility and ensures
51  correct alignment of the structure. If the structure is not accessible
52  or has incorrect alignment, then an exception is raised.
53 
54 Arguments:
55 
56  Address - Supplies a pointer to the structure to be probed.
57 
58  Length - Supplies the length of the structure.
59 
60  Alignment - Supplies the required alignment of the structure expressed
61  as the number of bytes in the primitive datatype (e.g., 1 for char,
62  2 for short, 4 for long, and 8 for quad).
63 
64 Return Value:
65 
66  None.
67 
68 --*/
69 
70 {
71 
72  ULONG_PTR EndAddress;
73  ULONG_PTR StartAddress;
74 
75 #define PageSize PAGE_SIZE
76 
77  //
78  // If the structure has zero length, then do not probe the structure for
79  // write accessibility or alignment.
80  //
81 
82  if (Length != 0) {
83 
84  //
85  // If the structure is not properly aligned, then raise a data
86  // misalignment exception.
87  //
88 
89  ASSERT((Alignment == 1) || (Alignment == 2) ||
90  (Alignment == 4) || (Alignment == 8) ||
91  (Alignment == 16));
92 
93  StartAddress = (ULONG_PTR)Address;
94  if ((StartAddress & (Alignment - 1)) == 0) {
95 
96  //
97  // Compute the ending address of the structure and probe for
98  // write accessibility.
99  //
100 
101  EndAddress = StartAddress + Length - 1;
102  if ((StartAddress <= EndAddress) &&
103  (EndAddress < MM_USER_PROBE_ADDRESS)) {
104 
105  //
106  // N.B. Only the contents of the buffer may be probed.
107  // Therefore the starting byte is probed for the
108  // first page, and then the first byte in the page
109  // for each succeeding page.
110  //
111  // If this is a Wow64 process, then the native page is 4K, which
112  // could be smaller than the native page size/
113  //
114 
115  EndAddress = (EndAddress & ~(PageSize - 1)) + PageSize;
116  do {
117  *(volatile CHAR *)StartAddress = *(volatile CHAR *)StartAddress;
118  StartAddress = (StartAddress & ~(PageSize - 1)) + PageSize;
119  } while (StartAddress != EndAddress);
120 
121  return;
122 
123  } else {
124  ExRaiseAccessViolation();
125  }
126 
127  } else {
128  ExRaiseDatatypeMisalignment();
129  }
130  }

132  return;
133 }
