33 BOOLEAN
34 SepPrivilegeCheck(
35  IN PTOKEN Token,
36  IN OUT PLUID_AND_ATTRIBUTES RequiredPrivileges,
37  IN ULONG RequiredPrivilegeCount,
38  IN ULONG PrivilegeSetControl,
39  IN KPROCESSOR_MODE PreviousMode
40  )
41 /*++
42 
43 Routine Description:
44 
45  Worker routine for SePrivilegeCheck
46 
47 Arguments:
48 
49  Token - The user's effective token.
50 
51  RequiredPrivileges - A privilege set describing the required
52  privileges. The UsedForAccess bits will be set in any privilege
53  that is actually used (usually all of them).
54 
55  RequiredPrivilegeCount - How many privileges are in the
56  RequiredPrivileges set.
57 
58  PrivilegeSetControl - Describes how many privileges are required.
59 
60  PreviousMode - The previous processor mode.
61 
62 Return Value:
63 
64  Returns TRUE if requested privileges are granted, FALSE otherwise.
65 
66 --*/
67 
68 {
69  PLUID_AND_ATTRIBUTES CurrentRequiredPrivilege;
70  PLUID_AND_ATTRIBUTES CurrentTokenPrivilege;
71 
72  BOOLEAN RequiredAll;
73 
74  ULONG TokenPrivilegeCount;
75  ULONG MatchCount = 0;
76 
77  ULONG i;
78  ULONG j;
79 
80  PAGED_CODE();
81 
82  //
83  // Take care of kernel callers first
84  //
85 
86  if (PreviousMode == KernelMode) {
87 
88  return(TRUE);
89 
90  }
91 
92  TokenPrivilegeCount = Token->PrivilegeCount;
93 
94  //
95  // Save whether we require ALL of them or ANY
96  //
97 
98  RequiredAll = (BOOLEAN)(PrivilegeSetControl & PRIVILEGE_SET_ALL_NECESSARY);
99 
100  SepAcquireTokenReadLock( Token );
101 
102 
103  for ( i = 0 , CurrentRequiredPrivilege = RequiredPrivileges ;
104  i < RequiredPrivilegeCount ;
105  i++, CurrentRequiredPrivilege++ ) {
106 
107  for ( j = 0, CurrentTokenPrivilege = Token->Privileges;
108  j < TokenPrivilegeCount ;
109  j++, CurrentTokenPrivilege++ ) {
110 
111  if ((CurrentTokenPrivilege->Attributes & SE_PRIVILEGE_ENABLED) &&
112  (RtlEqualLuid(&CurrentTokenPrivilege->Luid,
113  &CurrentRequiredPrivilege->Luid))
114  ) {
115 
116  CurrentRequiredPrivilege->Attributes |=
117  SE_PRIVILEGE_USED_FOR_ACCESS;
118  MatchCount++;
119  break; // start looking for next one
120  }
121 
122  }
123 
124  }
125 
126  SepReleaseTokenReadLock( Token );
127 
128  //
129  // If we wanted ANY and didn't get any, return failure.
130  //
131 
132  if (!RequiredAll && (MatchCount == 0)) {
133 
134  return (FALSE);
135 
136  }
137 
138  //
139  // If we wanted ALL and didn't get all, return failure.
140  //
141 
142  if (RequiredAll && (MatchCount != RequiredPrivilegeCount)) {
143 
144  return(FALSE);
145  }
146 
147  return(TRUE);
148 
149 }

155 SePrivilegeCheck(

157  __in PSECURITY_SUBJECT_CONTEXT SubjectSecurityContext,

160 /*++
161 
162 Routine Description:
163 
164  This routine checks to see if the token contains the specified
165  privileges.
166 
167 Arguments:
168 
169  RequiredPrivileges - Points to a set of privileges. The subject's
170  security context is to be checked to see which of the specified
171  privileges are present. The results will be indicated in the
172  attributes associated with each privilege. Note that
173  flags in this parameter indicate whether all the privileges listed
174  are needed, or any of the privileges.
175 
176  SubjectSecurityContext - A pointer to the subject's captured security
177  context.
178 
179  AccessMode - Indicates the access mode to use for access check. One of
180  UserMode or KernelMode. If the mode is kernel, then all privileges
181  will be marked as being possessed by the subject, and successful
182  completion status is returned.
183 
184 
185 Return Value:
186 
187  BOOLEAN - TRUE if all specified privileges are held by the subject,
188  otherwise FALSE.
189 
190 
191 --*/
192 
193 {
194  BOOLEAN Status;
195 
196  PAGED_CODE();
197 
198  //
199  // If we're impersonating a client, we have to be at impersonation level
200  // of SecurityImpersonation or above.
201  //
202 
203  if ( (SubjectSecurityContext->ClientToken != NULL) &&
204  (SubjectSecurityContext->ImpersonationLevel < SecurityImpersonation)
205  ) {
206 
207  return(FALSE);
208  }
209 
210  //
211  // SepPrivilegeCheck locks the passed token for read access
212  //
213 
214  Status = SepPrivilegeCheck(
215  EffectiveToken( SubjectSecurityContext ),
216  RequiredPrivileges->Privilege,
217  RequiredPrivileges->PrivilegeCount,
218  RequiredPrivileges->Control,
219  AccessMode
220  );
221 
222  return(Status);
223 }

228 NtPrivilegeCheck(
229  __in HANDLE ClientToken,
230  __inout PPRIVILEGE_SET RequiredPrivileges,
231  __out PBOOLEAN Result
232  )
233 
234 /*++
235 
236 Routine Description:
237 
238  This routine tests the caller's client's security context to see if it
239  contains the specified privileges.
240 
241 Arguments:
242 
243  ClientToken - A handle to a token object representing a client
244  attempting access. This handle must be obtained from a
245  communication session layer, such as from an LPC Port or Local
246  Named Pipe, to prevent possible security policy violations.
247 
248  RequiredPrivileges - Points to a set of privileges. The client's
249  security context is to be checked to see which of the specified
250  privileges are present. The results will be indicated in the
251  attributes associated with each privilege. Note that
252  flags in this parameter indicate whether all the privileges listed
253  are needed, or any of the privileges.
254 
255  Result - Receives a boolean flag indicating whether the client has all
256  the specified privileges or not. A value of TRUE indicates the
257  client has all the specified privileges. Otherwise a value of
258  FALSE is returned.
259 
260 
261 
262 Return Value:
263 
264  STATUS_SUCCESS - Indicates the call completed successfully.
265 
266  STATUS_PRIVILEGE_NOT_HELD - Indicates the caller does not have
267  sufficient privilege to use this privileged system service.
268 
269 --*/
270 
271

273 {
274  BOOLEAN BStatus;
275  KPROCESSOR_MODE PreviousMode;
276  NTSTATUS Status;
277  PLUID_AND_ATTRIBUTES CapturedPrivileges = NULL;
278  PTOKEN Token = NULL;

280  ULONG CapturedPrivilegesLength = 0;
281  ULONG ParameterLength = 0;
282  ULONG PrivilegeSetControl = 0;
283 
284  PAGED_CODE();
285 
286  PreviousMode = KeGetPreviousMode();
287 
288  Status = ObReferenceObjectByHandle(
289  ClientToken, // Handle
290  TOKEN_QUERY, // DesiredAccess
291  SeTokenObjectType, // ObjectType
292  PreviousMode, // AccessMode
293  (PVOID *)&Token, // Object
294  NULL // GrantedAccess
295  );
296 
297  if ( !NT_SUCCESS(Status) ) {
298  return Status;
299 
300  }
301 
302  //
303  // If the passed token is an impersonation token, make sure
304  // it is at SecurityIdentification or above.
305  //
306 
307  if (Token->TokenType == TokenImpersonation) {
308 
309  if (Token->ImpersonationLevel < SecurityIdentification) {
310 
311  ObDereferenceObject( (PVOID)Token );
312 
313  return( STATUS_BAD_IMPERSONATION_LEVEL );
314 
315  }
316  }
317 
318  try {
319 
320  //
321  // Capture passed Privilege Set
322  //
323 
324  ProbeForWriteSmallStructure(
325  RequiredPrivileges,
326  sizeof(PRIVILEGE_SET),
327  sizeof(ULONG)
328  );
329 
330  CapturedPrivilegeCount = RequiredPrivileges->PrivilegeCount;
331 
332  if (!IsValidElementCount(CapturedPrivilegeCount, LUID_AND_ATTRIBUTES)) {
333  Status = STATUS_INVALID_PARAMETER;
334  leave;
335  }
336  ParameterLength = (ULONG)sizeof(PRIVILEGE_SET) +
337  ((CapturedPrivilegeCount - ANYSIZE_ARRAY) *
338  (ULONG)sizeof(LUID_AND_ATTRIBUTES) );
339 
340  ProbeForWrite(
341  RequiredPrivileges,
342  ParameterLength,
343  sizeof(ULONG)
344  );
345 
346 
347  ProbeForWriteBoolean(Result);
348 
349  PrivilegeSetControl = RequiredPrivileges->Control;
350 
351 
352  } except(EXCEPTION_EXECUTE_HANDLER) {
353  Status = GetExceptionCode();
354  }
355 
356  if (!NT_SUCCESS(Status)) {
357  ObDereferenceObject( (PVOID)Token );
358  return Status;
359 
360  }
361 
362  Status = SeCaptureLuidAndAttributesArray(
363  (RequiredPrivileges->Privilege),
364  CapturedPrivilegeCount,
365  UserMode,
366  NULL, 0,
367  PagedPool,
368  TRUE,
369  &CapturedPrivileges,
370  &CapturedPrivilegesLength
371  );
372 
373  if (!NT_SUCCESS(Status)) {
374 
375  ObDereferenceObject( (PVOID)Token );
376  return Status;
377  }
378 
379  BStatus = SepPrivilegeCheck(
380  Token, // Token,
381  CapturedPrivileges, // RequiredPrivileges,
382  CapturedPrivilegeCount, // RequiredPrivilegeCount,
383  PrivilegeSetControl, // PrivilegeSetControl
384  PreviousMode // PreviousMode
385  );
386 
387  ObDereferenceObject( Token );
388 
389 
390  try {
391 
392  //
393  // copy the modified privileges buffer back to user
394  //
395 
396  RtlCopyMemory(
397  RequiredPrivileges->Privilege,
398  CapturedPrivileges,
399  CapturedPrivilegesLength
400  );
401 
402  *Result = BStatus;
403 
404  } except (EXCEPTION_EXECUTE_HANDLER) {
405 
406  SeReleaseLuidAndAttributesArray(
407  CapturedPrivileges,
408  PreviousMode,
409  TRUE
410  );
411 
412  return(GetExceptionCode());
413 
414  }
415 
416  SeReleaseLuidAndAttributesArray(
417  CapturedPrivileges,
418  PreviousMode,
419  TRUE
420  );
421 
422  return( STATUS_SUCCESS );
423 }

461  // call.

463 
464  RequiredPrivileges.PrivilegeCount = 1;
465  RequiredPrivileges.Control = PRIVILEGE_SET_ALL_NECESSARY;
466  RequiredPrivileges.Privilege[0].Luid = PrivilegeValue;
467  RequiredPrivileges.Privilege[0].Attributes = 0;
468 
469  SeCaptureSubjectContext( &SubjectSecurityContext );
470 
471  AccessGranted = SePrivilegeCheck(
472  &RequiredPrivileges,
473  &SubjectSecurityContext,
474  PreviousMode
