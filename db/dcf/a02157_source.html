102 FsRtlpRegisterProviderWithMUP
103 (
104  IN HANDLE mupHandle,
105  IN PUNICODE_STRING RedirDevName,
106  IN BOOLEAN MailslotsSupported
107 )
108 /*++
109 
110 Routine Description:
111 
112  This private routine does the FSCTL to the MUP to tell it about
113  a new redir
114 
115 Arguments:
116 
117  mupHandle - Handle to the MUP
118 
119  RedirDevName - The device name of the redir.
120 
121  MailslotsSupported - If TRUE, this redir supports mailslots.
122 
123 Return Value:
124 
125  NTSTATUS - The status of the operation.
126 
127 --*/
128 {
129  NTSTATUS status;
130  IO_STATUS_BLOCK ioStatusBlock;
131  ULONG paramLength;
132  PREDIRECTOR_REGISTRATION params;
133 
134  PAGED_CODE();
135 
136  paramLength = sizeof( REDIRECTOR_REGISTRATION ) +
137  RedirDevName->Length;
138 
139  params = ExAllocatePoolWithTag( NonPagedPool, paramLength, MODULE_POOL_TAG );
140  if( params == NULL )
141  return STATUS_INSUFFICIENT_RESOURCES;
142 
143  params->DeviceNameOffset = sizeof( REDIRECTOR_REGISTRATION );
144  params->DeviceNameLength = RedirDevName->Length;
145  params->MailslotsSupported = MailslotsSupported;
146 
147  RtlCopyMemory(
148  (PCHAR)params + params->DeviceNameOffset,
149  RedirDevName->Buffer,
150  RedirDevName->Length
151  );
152 
153  status = NtFsControlFile(
154  mupHandle,
155  0,
156  NULL,
157  NULL,
158  &ioStatusBlock,
159  FSCTL_MUP_REGISTER_UNC_PROVIDER,
160  params,
161  paramLength,
162  NULL,
163  0
164  );
165 
166  if ( status == STATUS_PENDING ) {
167  status = NtWaitForSingleObject( mupHandle, TRUE, NULL );
168  }
169 
170  if ( NT_SUCCESS( status ) ) {
171  status = ioStatusBlock.Status;
172  }
173 
174  ASSERT( NT_SUCCESS( status ) );
175 
176  ExFreePool( params );
177 
178  return status;
179 }

245 NTSTATUS
246 FsRtlRegisterUncProvider(
247  __inout PHANDLE MupHandle,
248  __in PUNICODE_STRING RedirDevName,
249  __in BOOLEAN MailslotsSupported
250  )
251 /*++
252 
253 Routine Description:
254 
255  This routine registers a redir as a UNC provider.
256 
257 Arguments:
258 
259  Handle - Pointer to a handle. The handle is returned by the routine
260  to be used when calling FsRtlDeregisterUncProvider.
261  It is valid only if the routines returns STATUS_SUCCESS.
262 
263  RedirDevName - The device name of the redir.
264 
265  MailslotsSupported - If TRUE, this redir supports mailslots.
266 
267 Return Value:
268 
269  NTSTATUS - The status of the operation.
270 
271 --*/
272 {
273  NTSTATUS status;
274  HANDLE mupHandle = (HANDLE)-1;
275  UNICODE_STRING mupDriverName;
276  BOOLEAN dfsEnabled;
277 
278  PAGED_CODE();
279 
280  KeWaitForSingleObject(&FsRtlpUncSemaphore, Executive, KernelMode, FALSE, NULL );
281 
282  if (FsRtlpRedirs == 0) {
283 
284  dfsEnabled = FsRtlpIsDfsEnabled();
285 
286  if (dfsEnabled) {
287  FsRtlpRedirs = 1;
288  RtlZeroMemory((PVOID) &FsRtlpDRD, sizeof(FsRtlpDRD));
289  }
290 
291  }
292 
293  switch( FsRtlpRedirs ) {
294  case 0:
295  //
296  // Ok, the MUP isn't there and we don't need to use the
297  // MUP for the first redir.
298  //
299  // We need to return a handle, but we're not really using the MUP yet.
300  // And we may never use it (if there's only 1 redir). Return
301  // a handle to the NULL device object, since we're committed to returning
302  // a handle to our caller. Our caller isn't supposed to do anything with
303  // the handle except to call FsRtlDeregisterUncProvider() with it.
304  //
305  status = FsRtlpOpenDev( &mupHandle, DevNull );
306 
307  if( !NT_SUCCESS( status ) )
308  break;
309 
310  //
311  // Save up enough state to allow us to call the MUP later with
312  // this registration info if necessary.
313  //
314  FsRtlpDRD.RedirDevName.Buffer = ExAllocatePoolWithTag( NonPagedPool,
315  RedirDevName->MaximumLength,
316  MODULE_POOL_TAG );
317 
318  if( FsRtlpDRD.RedirDevName.Buffer == NULL ) {
319  status = STATUS_INSUFFICIENT_RESOURCES;
320  break;
321  }
322 
323  FsRtlpDRD.RedirDevName.Length = RedirDevName->Length;
324  FsRtlpDRD.RedirDevName.MaximumLength = RedirDevName->MaximumLength;
325 
326  RtlCopyMemory(
327  (PCHAR)FsRtlpDRD.RedirDevName.Buffer,
328  RedirDevName->Buffer,
329  RedirDevName->MaximumLength
330  );
331 
332  FsRtlpDRD.MailslotsSupported = MailslotsSupported;
333  FsRtlpDRD.ReturnedHandle = mupHandle;
334  FsRtlpDRD.MupHandle = (HANDLE)-1;
335 
336  //
337  // Set the UNC symbolic link to point to the redir we just loaded
338  //
339  FsRtlpSetSymbolicLink( RedirDevName );
340 
341  break;
342 
343  default:
344  //
345  // This is the second or later redir load -- MUST use the MUP
346  //
347  status = FsRtlpOpenDev( &mupHandle, DevMup );
348 
349  if( !NT_SUCCESS( status ) ) {
350 
351  RtlInitUnicodeString( &mupDriverName, MupRegKey );
352 
353  (VOID)ZwLoadDriver( &mupDriverName );
354 
355  status = FsRtlpOpenDev( &mupHandle, DevMup );
356  if( !NT_SUCCESS( status ) )
357  break;
358  }
359 
360  //
361  // See if we need to tell the MUP about the first redir that registered
362  //
363  if( FsRtlpDRD.RedirDevName.Buffer ) {
364 
365  status = FsRtlpRegisterProviderWithMUP( mupHandle,
366  &FsRtlpDRD.RedirDevName,
367  FsRtlpDRD.MailslotsSupported );
368 
369  if( !NT_SUCCESS( status ) )
370  break;
371 
372  FsRtlpDRD.MupHandle = mupHandle;
373 
374  ExFreePool( FsRtlpDRD.RedirDevName.Buffer );
375  FsRtlpDRD.RedirDevName.Buffer = NULL;
376 
377  //
378  // Set the UNC symbolic link to point to the MUP
379  //
380  RtlInitUnicodeString( &mupDriverName, DevMup );
381  FsRtlpSetSymbolicLink( &mupDriverName );
382 
383  status = FsRtlpOpenDev( &mupHandle, DevMup );
384 
385  if( !NT_SUCCESS( status ) )
386  break;
387  }
388 
389  //
390  // Pass the request to the MUP for this redir
391  //
392  status = FsRtlpRegisterProviderWithMUP( mupHandle,
393  RedirDevName,
394  MailslotsSupported );
395  break;
396 
397  }
398 
399  if( NT_SUCCESS( status ) ) {
400  FsRtlpRedirs++;
401  *MupHandle = mupHandle;
402 
403  } else {
404  if( mupHandle != (HANDLE)-1 && mupHandle != NULL ) {
405  ZwClose( mupHandle );
406  }
407 
408  *MupHandle = (HANDLE)-1;
409  }
410 
411  KeReleaseSemaphore(&FsRtlpUncSemaphore, 0, 1, FALSE );
412  return status;
413 }

416 VOID
417 FsRtlDeregisterUncProvider(
418  __in HANDLE Handle
419  )
420 
421 /*++
422 
423 Routine Description:
424 
425  This routine deregisters a redir as a UNC provider.
426 
427 Arguments:
428 
429  Handle - A handle to the Multiple UNC router, returned by the
430  registration call.
431 
432 Return Value:
433 
434  None.
435 
436 --*/
437 
438 {
439  NTSTATUS status;
440 
441  PAGED_CODE();
442 
443  if( Handle == (HANDLE)-1 || Handle == NULL )
444  return;
445 
446  status = ZwClose( Handle );
447 
448  if( !NT_SUCCESS( status ) ) {
449  return;
450  }
451 
452  KeWaitForSingleObject(&FsRtlpUncSemaphore, Executive, KernelMode, FALSE, NULL );
453 
454  ASSERT( FsRtlpRedirs > 0 );
455 
456  if( Handle == FsRtlpDRD.ReturnedHandle ) {
457 
458  //
459  // The first redir in the system is closing. Release the state we saved
460  // for it, and pass the close on to the MUP if necessary
461  //
462 
463  if( FsRtlpDRD.RedirDevName.Buffer != NULL ) {
464  ExFreePool( FsRtlpDRD.RedirDevName.Buffer );
465  FsRtlpDRD.RedirDevName.Buffer = NULL;
466  }
467 
468  if( FsRtlpDRD.MupHandle != (HANDLE)-1 ) {
469  ZwClose( FsRtlpDRD.MupHandle );
470  FsRtlpDRD.MupHandle = (HANDLE)-1;
471  }
472 
473  FsRtlpDRD.ReturnedHandle = (HANDLE)-1;
474 
475  }
476 
477  if( --FsRtlpRedirs == 0 ) {
478  FsRtlpSetSymbolicLink( (PUNICODE_STRING)NULL );
479  }
480 
481  KeReleaseSemaphore(&FsRtlpUncSemaphore, 0, 1, FALSE );
482 }

486 FsRtlpIsDfsEnabled()
487 
488 /*++
489 
490 Routine Description:
491 
492  This routine checks a registry key to see if the Dfs client is enabled.
493  The client is assumed to be enabled by default, and disabled only if there
494  is a registry value indicating that it should be disabled.
495 
496 Arguments:
497 
498  None
499 
500 Return Value:
501 
502  TRUE if Dfs client is enabled, FALSE otherwise.
503 
504 --*/
505 
506 {
507  NTSTATUS status;
508  HANDLE mupRegHandle;
509  OBJECT_ATTRIBUTES objectAttributes;
510  ULONG valueSize;
511  BOOLEAN dfsEnabled = TRUE;
512 
513  UNICODE_STRING mupRegKey = {
514  sizeof(MupRegKey) - sizeof(WCHAR),
515  sizeof(MupRegKey),
516  MupRegKey};
517 
518 #define DISABLE_DFS_VALUE_NAME L"DisableDfs"
519 
520  UNICODE_STRING disableDfs = {
521  sizeof(DISABLE_DFS_VALUE_NAME) - sizeof(WCHAR),
522  sizeof(DISABLE_DFS_VALUE_NAME),
523  DISABLE_DFS_VALUE_NAME};
524 
525  struct {
526  KEY_VALUE_PARTIAL_INFORMATION Info;
527  ULONG Buffer;
528  } disableDfsValue;
529 
530 
531  InitializeObjectAttributes(
532  &objectAttributes,
533  &mupRegKey,
534  OBJ_CASE_INSENSITIVE,
535  0,
536  NULL
537  );
538 
539  status = ZwOpenKey(&mupRegHandle, KEY_READ, &objectAttributes);
540 
541  if (NT_SUCCESS(status)) {
542 
543  status = ZwQueryValueKey(
544  mupRegHandle,
545  &disableDfs,
546  KeyValuePartialInformation,
547  (PVOID) &disableDfsValue,
548  sizeof(disableDfsValue),
549  &valueSize);
550 
551  if (NT_SUCCESS(status) && disableDfsValue.Info.Type == REG_DWORD) {
552 
553  if ( (*((PULONG) disableDfsValue.Info.Data)) == 1 )
554  dfsEnabled = FALSE;
555 
556  }
557 
558  ZwClose( mupRegHandle );
559 
560  }
561 
562  return( dfsEnabled );
563 
564 }
