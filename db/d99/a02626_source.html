51 //
52 // Build a table of the worst case efficiency of a balanced binary tree, holding the
53 // fewest possible nodes that can possibly be contained in a balanced binary tree with
54 // the given number of levels. After the first two levels, each level n is obviously
55 // occupied by a root node, plus one subtree the size of level n-1, and another subtree
56 // which is the size of n-2, i.e.:
57 //
58 // WorstCaseFill[n] = 1 + WorstCaseFill[n-1] + WorstCaseFill[n-2]
59 //
60 // The efficiency of a typical balanced binary tree will normally fall between the two
61 // extremes, typically closer to the best case. Note however that even with the worst
62 // case, it only takes 32 compares to find an element in a worst case tree populated with
63 // ~3.5M nodes. Unbalanced trees and splay trees, on the other hand, can and will sometimes
64 // degenerate to a straight line, requiring on average n/2 compares to find a node.
65 //
66 // A specific case (that will frequently occur in TXF), is one where the nodes are inserted
67 // in collated order. In this case an unbalanced or a splay tree will generate a straight
68 // line, yet the balanced binary tree will always create a perfectly balanced tree (best-case
69 // fill) in this situation.
70 //
71 // (Used for debug only.)
72 //

121 /*++
122 
123 Routine Description:
124 
125  This routine is used by all of the routines of the generic
126  table package to locate the a node in the tree. It will
127  find and return (via the NodeOrParent parameter) the node
128  with the given key, or if that node is not in the tree it
129  will return (via the NodeOrParent parameter) a pointer to
130  the parent.
131 
132 Arguments:
133 
134  Table - The generic table to search for the key.
135 
136  Buffer - Pointer to a buffer holding the key. The table
137  package doesn't examine the key itself. It leaves
138  this up to the user supplied compare routine.
139 
140  NodeOrParent - Will be set to point to the node containing the
141  the key or what should be the parent of the node
142  if it were in the tree. Note that this will *NOT*
143  be set if the search result is TableEmptyTree.
144 
145 Return Value:
146 
147  TABLE_SEARCH_RESULT - TableEmptyTree: The tree was empty. NodeOrParent
148  is *not* altered.
149 
150  TableFoundNode: A node with the key is in the tree.
151  NodeOrParent points to that node.
152 
153  TableInsertAsLeft: Node with key was not found.
154  NodeOrParent points to what would be
155  parent. The node would be the left
156  child.
157 
158  TableInsertAsRight: Node with key was not found.
159  NodeOrParent points to what would be
160  parent. The node would be the right
161  child.
162 
163 --*/

166

169  return TableEmptyTree;
170 
171  } else {

174  // Used as the iteration variable while stepping through
175  // the generic table.
176  //
177

179 
180  //
181  // Just a temporary. Hopefully a good compiler will get
182  // rid of it.
183  //
184

186 
187  //

189  //

266 VOID
267 PromoteNode (
268  IN PRTL_BALANCED_LINKS C
269  )
270 
271 /*++
272 
273 Routine Description:
274 
275  This routine performs the fundamental adjustment required for balancing
276  the binary tree during insert and delete operations. Simply put, the designated
277  node is promoted in such a way that it rises one level in the tree and its parent
278  drops one level in the tree, becoming now the child of the designated node.
279  Generally the path length to the subtree "opposite" the original parent. Balancing
280  occurs as the caller chooses which nodes to promote according to the balanced tree
281  algorithms from Knuth.
282 
283  This is not the same as a splay operation, typically a splay "promotes" a designated
284  node twice.
285 
286  Note that the pointer to the root node of the tree is assumed to be contained in a
287  RTL_BALANCED_LINK structure itself, to allow the algorithms below to change the root
288  of the tree without checking for special cases. Note also that this is an internal
289  routine, and the caller guarantees that it never requests to promote the root itself.
290 
291  This routine only updates the tree links; the caller must update the balance factors
292  as appropriate.
293 
294 Arguments:
295 
296  C - pointer to the child node to be promoted in the tree.
297 
298 Return Value:
299 
300  None.
301 
302 --*/
303 
304 {
305  PRTL_BALANCED_LINKS P, G;
306 
307  //
308  // Capture the current parent and grandparent (may be the root).
309  //
310 
311  P = C->Parent;
312  G = P->Parent;
313 
314  //
315  // Break down the promotion into two cases based upon whether C is a left or right child.
316  //
317 
318  if (P->LeftChild == C) {
319 
320  //
321  // This promotion looks like this:
322  //
323  // G G
324  // | |
325  // P C
326  // / \ => / \
327  // C z x P
328  // / \ / \
329  // x y y z
330  //
331 
332  P->LeftChild = checkit(C->RightChild);
333 
334  if (P->LeftChild != NULL) {
335  P->LeftChild->Parent = checkit(P);
336  }
337 
338  C->RightChild = checkit(P);
339 
340  //
341  // Fall through to update parent and G <-> C relationship in common code.
342  //
343 
344  } else {
345 
346  ASSERT(P->RightChild == C);
347 
348  //
349  // This promotion looks like this:
350  //
351  // G G
352  // | |
353  // P C
354  // / \ => / \
355  // x C P z
356  // / \ / \
357  // y z x y
358  //
359 
360  P->RightChild = checkit(C->LeftChild);
361 
362  if (P->RightChild != NULL) {
363  P->RightChild->Parent = checkit(P);
364  }
365 
366  C->LeftChild = checkit(P);
367  }
368 
369  //
370  // Update parent of P, for either case above.
371  //
372 
373  P->Parent = checkit(C);
374 
375  //
376  // Finally update G <-> C links for either case above.
377  //
378 
379  if (G->LeftChild == P) {
380  G->LeftChild = checkit(C);
381  } else {
382  ASSERT(G->RightChild == P);
383  G->RightChild = checkit(C);
384  }
385  C->Parent = checkit(G);
386 }

389 ULONG
390 RebalanceNode (
391  IN PRTL_BALANCED_LINKS S
392  )
393 
394 /*++
395 
396 Routine Description:
397 
398  This routine performs a rebalance around the input node S, for which the
399  Balance factor has just effectively become +2 or -2. When called, the
400  Balance factor still has a value of +1 or -1, but the respective longer
401  side has just become one longer as the result of an insert or delete operation.
402 
403  This routine effectively implements steps A7.iii (test for Case 1 or Case 2) and
404  steps A8 and A9 of Knuths balanced insertion algorithm, plus it handles Case 3
405  identified in the delete section, which can only happen on deletes.
406 
407  The trick is, to convince yourself that while traveling from the insertion point
408  at the bottom of the tree up, that there are only these two cases, and that when
409  traveling up from the deletion point, that there are just these three cases.
410  Knuth says it is obvious!
411 
412 Arguments:
413 
414  S - pointer to the node which has just become unbalanced.
415 
416 Return Value:
417 
418  TRUE if Case 3 was detected (causes delete algorithm to terminate).
419 
420 --*/
421 
422 {
423  PRTL_BALANCED_LINKS R, P;
424  CHAR a;
425 
426  //
427  // Capture which side is unbalanced.
428  //
429 
430  a = S->Balance;
431  if (a == +1) {
432  R = S->RightChild;
433  } else {
434  R = S->LeftChild;
435  }
436 
437  //
438  // If the balance of R and S are the same (Case 1 in Knuth) then a single
439  // promotion of R will do the single rotation. (Step A8, A10)
440  //
441  // Here is a diagram of the Case 1 transformation, for a == +1 (a mirror
442  // image transformation occurs when a == -1), and where the subtree
443  // heights are h and h+1 as shown (++ indicates the node out of balance):
444  //
445  // | |
446  // S++ R
447  // / \ / \
448  // (h) R+ ==> S (h+1)
449  // / \ / \
450  // (h) (h+1) (h) (h)
451  //
452  // Note that on an insert we can hit this case by inserting an item in the
453  // right subtree of R. The original height of the subtree before the insert
454  // was h+2, and it is still h+2 after the rebalance, so insert rebalancing may
455  // terminate.
456  //
457  // On a delete we can hit this case by deleting a node from the left subtree
458  // of S. The height of the subtree before the delete was h+3, and after the
459  // rebalance it is h+2, so rebalancing must continue up the tree.
460  //
461 
462  if (R->Balance == a) {
463 
464  PromoteNode( R );
465  R->Balance = 0;
466  S->Balance = 0;
467  return FALSE;
468 
469  //
470  // Otherwise, we have to promote the appropriate child of R twice (Case 2
471  // in Knuth). (Step A9, A10)
472  //
473  // Here is a diagram of the Case 2 transformation, for a == +1 (a mirror
474  // image transformation occurs when a == -1), and where the subtree
475  // heights are h and h-1 as shown. There are actually two minor subcases,
476  // differing only in the original balance of P (++ indicates the node out
477  // of balance).
478  //
479  // | |
480  // S++ P
481  // / \ / \
482  // / \ / \
483  // / \ / \
484  // (h) R- ==> S- R
485  // / \ / \ / \
486  // P+ (h) (h)(h-1)(h) (h)
487  // / \
488  // (h-1) (h)
489  //
490  //
491  // | |
492  // S++ P
493  // / \ / \
494  // / \ / \
495  // / \ / \
496  // (h) R- ==> S R+
497  // / \ / \ / \
498  // P- (h) (h) (h)(h-1)(h)
499  // / \
500  // (h) (h-1)
501  //
502  // Note that on an insert we can hit this case by inserting an item in the
503  // left subtree of R. The original height of the subtree before the insert
504  // was h+2, and it is still h+2 after the rebalance, so insert rebalancing may
505  // terminate.
506  //
507  // On a delete we can hit this case by deleting a node from the left subtree
508  // of S. The height of the subtree before the delete was h+3, and after the
509  // rebalance it is h+2, so rebalancing must continue up the tree.
510  //
511 
512  } else if (R->Balance == -a) {
513 
514  //
515  // Pick up the appropriate child P for the double rotation (Link(-a,R)).
516  //
517 
518  if (a == 1) {
519  P = R->LeftChild;
520  } else {
521  P = R->RightChild;
522  }
523 
524  //
525  // Promote him twice to implement the double rotation.
526  //
527 
528  PromoteNode( P );
529  PromoteNode( P );
530 
531  //
532  // Now adjust the balance factors.
533  //
534 
535  S->Balance = 0;
536  R->Balance = 0;
537  if (P->Balance == a) {
538  S->Balance = -a;
539  } else if (P->Balance == -a) {
540  R->Balance = a;
541  }
542 
543  P->Balance = 0;
544  return FALSE;
545 
546  //
547  // Otherwise this is Case 3 which can only happen on Delete (identical to Case 1 except
548  // R->Balance == 0). We do a single rotation, adjust the balance factors appropriately,
549  // and return TRUE. Note that the balance of S stays the same.
550  //
551  // Here is a diagram of the Case 3 transformation, for a == +1 (a mirror
552  // image transformation occurs when a == -1), and where the subtree
553  // heights are h and h+1 as shown (++ indicates the node out of balance):
554  //
555  // | |
556  // S++ R-
557  // / \ / \
558  // (h) R ==> S+ (h+1)
559  // / \ / \
560  // (h+1)(h+1) (h) (h+1)
561  //
562  // This case can not occur on an insert, because it is impossible for a single insert to
563  // balance R, yet somehow grow the right subtree of S at the same time. As we move up
564  // the tree adjusting balance factors after an insert, we terminate the algorithm if a
565  // node becomes balanced, because that means the subtree length did not change!
566  //
567  // On a delete we can hit this case by deleting a node from the left subtree
568  // of S. The height of the subtree before the delete was h+3, and after the
569  // rebalance it is still h+3, so rebalancing may terminate in the delete path.
570  //
571 
572  } else {
573 
574  PromoteNode( R );
575  R->Balance = -a;
576  return TRUE;
577  }
578 }

581 VOID
582 DeleteNodeFromTree (
583  IN PRTL_AVL_TABLE Table,
584  IN PRTL_BALANCED_LINKS NodeToDelete
585  )
586 
587 /*++
588 
589 Routine Description:
590 
591  This routine deletes the specified node from the balanced tree, rebalancing
592  as necessary. If the NodeToDelete has at least one NULL child pointers, then
593  it is chosen as the EasyDelete, otherwise a subtree predecessor or successor
594  is found as the EasyDelete. In either case the EasyDelete is deleted
595  and the tree is rebalanced. Finally if the NodeToDelete was different
596  than the EasyDelete, then the EasyDelete is linked back into the tree in
597  place of the NodeToDelete.
598 
599 Arguments:
600 
601  Table - The generic table in which the delete is to occur.
602 
603  NodeToDelete - Pointer to the node which the caller wishes to delete.
604 
605 Return Value:
606 
607  None.
608 
609 --*/
610 
611 {
612  PRTL_BALANCED_LINKS EasyDelete;
613  PRTL_BALANCED_LINKS P;
614  CHAR a;
615 
616  //
617  // If the NodeToDelete has at least one NULL child pointer, then we can
618  // delete it directly.
619  //
620 
621  if ((NodeToDelete->LeftChild == NULL) || (NodeToDelete->RightChild == NULL)) {
622 
623  EasyDelete = NodeToDelete;
624 
625  //
626  // Otherwise, we may as well pick the longest side to delete from (if one is
627  // is longer), as that reduces the probability that we will have to rebalance.
628  //
629 
630  } else if (NodeToDelete->Balance >= 0) {
631 
632  //
633  // Pick up the subtree successor.
634  //
635 
636  EasyDelete = NodeToDelete->RightChild;
637  while (EasyDelete->LeftChild != NULL) {
638  EasyDelete = EasyDelete->LeftChild;
639  }
640  } else {
641 
642  //
643  // Pick up the subtree predecessor.
644  //
645 
646  EasyDelete = NodeToDelete->LeftChild;
647  while (EasyDelete->RightChild != NULL) {
648  EasyDelete = EasyDelete->RightChild;
649  }
650  }
651 
652  //
653  // Rebalancing must know which side of the first parent the delete occurred
654  // on. Assume it is the left side and otherwise correct below.
655  //
656 
657  a = -1;
658 
659  //
660  // Now we can do the simple deletion for the no left child case.
661  //
662 
663  if (EasyDelete->LeftChild == NULL) {
664 
665  if (RtlIsLeftChild(EasyDelete)) {
666  EasyDelete->Parent->LeftChild = checkit(EasyDelete->RightChild);
667  } else {
668  EasyDelete->Parent->RightChild = checkit(EasyDelete->RightChild);
669  a = 1;
670  }
671 
672  if (EasyDelete->RightChild != NULL) {
673  EasyDelete->RightChild->Parent = checkit(EasyDelete->Parent);
674  }
675 
676  //
677  // Now we can do the simple deletion for the no right child case,
678  // plus we know there is a left child.
679  //
680 
681  } else {
682 
683  if (RtlIsLeftChild(EasyDelete)) {
684  EasyDelete->Parent->LeftChild = checkit(EasyDelete->LeftChild);
685  } else {
686  EasyDelete->Parent->RightChild = checkit(EasyDelete->LeftChild);
687  a = 1;
688  }
689 
690  EasyDelete->LeftChild->Parent = checkit(EasyDelete->Parent);
691  }
692 
693  //
694  // For delete rebalancing, set the balance at the root to 0 to properly
695  // terminate the rebalance without special tests, and to be able to detect
696  // if the depth of the tree actually decreased.
697  //
698 
699  Table->BalancedRoot.Balance = 0;
700  P = EasyDelete->Parent;
701 
702  //
703  // Loop until the tree is balanced.
704  //
705 
706  while (TRUE) {
707 
708  //
709  // First handle the case where the tree became more balanced. Zero
710  // the balance factor, calculate a for the next loop and move on to
711  // the parent.
712  //
713 
714  if (P->Balance == a) {
715 
716  P->Balance = 0;
717 
718  //
719  // If this node is curently balanced, we can show it is now unbalanced
720  // and terminate the scan since the subtree length has not changed.
721  // (This may be the root, since we set Balance to 0 above!)
722  //
723 
724  } else if (P->Balance == 0) {
725 
726  P->Balance = -a;
727 
728  //
729  // If we shortened the depth all the way back to the root, then the tree really
730  // has one less level.
731  //
732 
733  if (Table->BalancedRoot.Balance != 0) {
734  Table->DepthOfTree -= 1;
735  }
736 
737  break;
738 
739  //
740  // Otherwise we made the short side 2 levels less than the long side,
741  // and rebalancing is required. On return, some node has been promoted
742  // to above node P. If Case 3 from Knuth was not encountered, then we
743  // want to effectively resume rebalancing from P's original parent which
744  // is effectively its grandparent now.
745  //
746 
747  } else {
748 
749  //
750  // We are done if Case 3 was hit, i.e., the depth of this subtree is
751  // now the same as before the delete.
752  //
753 
754  if (RebalanceNode(P)) {
755  break;
756  }
757 
758  P = P->Parent;
759  }
760 
761  a = -1;
762  if (RtlIsRightChild(P)) {
763  a = 1;
764  }
765  P = P->Parent;
766  }
767 
768  //
769  // Finally, if we actually deleted a predecessor/successor of the NodeToDelete,
770  // we will link him back into the tree to replace NodeToDelete before returning.
771  // Note that NodeToDelete did have both child links filled in, but that may no
772  // longer be the case at this point.
773  //
774 
775  if (NodeToDelete != EasyDelete) {
776  *EasyDelete = *NodeToDelete;
777  if (RtlIsLeftChild(NodeToDelete)) {
778  EasyDelete->Parent->LeftChild = checkit(EasyDelete);
779  } else {
780  ASSERT(RtlIsRightChild(NodeToDelete));
781  EasyDelete->Parent->RightChild = checkit(EasyDelete);
782  }
783  if (EasyDelete->LeftChild != NULL) {
784  EasyDelete->LeftChild->Parent = checkit(EasyDelete);
785  }
786  if (EasyDelete->RightChild != NULL) {
787  EasyDelete->RightChild->Parent = checkit(EasyDelete);
788  }
789  }
790 }
