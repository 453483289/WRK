32 CmpDeleteTree(
33  PHHIVE Hive,
34  HCELL_INDEX Cell
35  )
36 /*++
37 
38 Routine Description:
39 
40  Delete all child subkeys, recursively, of Hive.Cell. Delete all
41  value entries of Hive.Cell. Do NOT delete Hive.Cell itself.
42 
43  NOTE: If this call fails part way through, it will NOT undo
44  any successfully completed deletes.
45 
46  NOTE: This algorithm can deal with a hive of any depth, at the
47  cost of some "redundent" scaning and mapping.
48 
49 Arguments:
50 
51  Hive - pointer to hive control structure for hive to delete from
52 
53  Cell - index of cell at root of tree to delete
54 
55 Return Value:
56 
57  BOOLEAN - Result code from call, among the following:
58  TRUE - it worked
59  FALSE - the tree delete was not completed (though more than 0
60  keys may have been deleted)
61 
62 --*/
63 {
64  ULONG count;
65  HCELL_INDEX ptr1;
66  HCELL_INDEX ptr2;
67  HCELL_INDEX parent;
68  PCM_KEY_NODE Node;
69 
70  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_SAVRES,"CmpDeleteTree:\n"));
71  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_SAVRES,"\tHive=%p Cell=%08lx\n",Hive,Cell));
72 
73  //
74  // we have the lock exclusive or nobody is operating inside this hive
75  //
76  ASSERT_CM_EXCLUSIVE_HIVE_ACCESS(Hive);
77 
78  ptr1 = Cell;
79 
80  while(TRUE) {
81 
82  Node = (PCM_KEY_NODE)HvGetCell(Hive, ptr1);
83  if( Node == NULL ) {
84  //
85  // we couldn't map the bin containing this cell
86  // bad luck! we cannot delete this tree
87  //
88  return;
89  }
90  count = Node->SubKeyCounts[Stable] +
91  Node->SubKeyCounts[Volatile];
92  parent = Node->Parent;
93 
94  if (count > 0) { // ptr1->count > 0?
95 
96  //
97  // subkeys exist, find and delete them
98  //
99  ptr2 = CmpFindSubKeyByNumber(Hive, Node, 0);
100 
101  //
102  // release the cell here as we are overriding node
103  //
104  HvReleaseCell(Hive, ptr1);
105 
106  if( ptr2 == HCELL_NIL ) {
107  //
108  // we couldn't map view inside
109  // bad luck! we cannot delete this tree
110  //
111  return;
112  }
113 
114  Node = (PCM_KEY_NODE)HvGetCell(Hive, ptr2);
115  if( Node == NULL ) {
116  //
117  // we couldn't map the bin containing this cell
118  // bad luck! we cannot delete this tree
119  //
120  return;
121  }
122  count = Node->SubKeyCounts[Stable] +
123  Node->SubKeyCounts[Volatile];
124 
125  //
126  // release the cell here as we don't need it anymore
127  //
128  HvReleaseCell(Hive, ptr2);
129  if (count > 0) { // ptr2->count > 0?
130 
131  //
132  // subkey has subkeys, descend to next level
133  //
134  ptr1 = ptr2;
135  continue;
136 
137  } else {
138 
139  //
140  // have found leaf, delete it
141  //
142  CmpFreeKeyByCell(Hive, ptr2, TRUE);
143  continue;
144  }
145 
146  } else {
147  //
148  // release the cell here as we don't need it anymore
149  //
150  HvReleaseCell(Hive, ptr1);
151 
152  //
153  // no more subkeys at this level, we are now a leaf.
154  //
155  if (ptr1 != Cell) {
156 
157  //
158  // we are not at the root cell, so ascend to parent
159  //
160  ptr1 = parent; // ptr1 = ptr1->parent
161  continue;
162 
163  } else {
164 
165  //
166  // we are at the root cell, we are done
167  //
168  return;
169  }
170  } // outer if
171  } // while
172 }

181 /*++
182 
183 Routine Description:
184 
185  Actually free the storage for the specified cell. We will first
186  remove it from its parent's child key list, then free all of its
187  values, then the key body itself.
188 
189 Arguments:
190 
191  Hive - pointer to hive control structure for hive of interest
192 
193  Cell - index for cell to free storage for (the target)
194 
195  Unlink - if TRUE, target cell will be removed from parent cell's
196  subkeylist, if FALSE, it will NOT be.
197 
198 Return Value:
199 
200  NONE.
201 
202 --*/
203 {
204  PCELL_DATA ptarget;
205  PCELL_DATA pparent;
206  PCELL_DATA plist;
207  ULONG i;
208 
209  //
210  // we have the lock exclusive or nobody is operating inside this hive
211  //
212  ASSERT_HIVE_FLUSHER_LOCKED((PCMHIVE)Hive);
213 
214  //
215  // Mark dirty everything that we might touch
216  //
217  if (! CmpMarkKeyDirty(Hive, Cell
218 #if DBG
219  ,TRUE
220 #endif //DBG
221  )) {
222  return STATUS_NO_LOG_SPACE;
223  }
224 
225  //
226  // Map in the target
227  //
228  ptarget = HvGetCell(Hive, Cell);
229  if( ptarget == NULL ) {
230  //
231  // we couldn't map the bin containing this cell
232  // we shouldn't hit this as we just marked the cell dirty
233  //
234  ASSERT( FALSE );
235  return STATUS_INSUFFICIENT_RESOURCES;
236  }
237  //
238  // release the cell here as it is dirty(pinned); it cannot go anywhere
239  //
240  HvReleaseCell(Hive, Cell);
241 
242  ASSERT((ptarget->u.KeyNode.SubKeyCounts[Stable] +
243  ptarget->u.KeyNode.SubKeyCounts[Volatile]) == 0);
244 
245 
246  if (Unlink == TRUE) {
247  BOOLEAN Success;
248 
249  Success = CmpRemoveSubKey(Hive, ptarget->u.KeyNode.Parent, Cell);
250  if (!Success) {
251  return STATUS_INSUFFICIENT_RESOURCES;
252  }
253  pparent = HvGetCell(Hive, ptarget->u.KeyNode.Parent);
254  if( pparent == NULL ) {
255  //
256  // we couldn't map the bin containing this cell
257  // we shouldn't hit this as we just marked the cell dirty
258  //
259  ASSERT( FALSE );
260  return STATUS_INSUFFICIENT_RESOURCES;
261  }
262 
263  //
264  // release the cell here as it is dirty(pinned); it cannot go anywhere
265  //
266  HvReleaseCell(Hive, ptarget->u.KeyNode.Parent);
267 
268  if ( (pparent->u.KeyNode.SubKeyCounts[Stable] +
269  pparent->u.KeyNode.SubKeyCounts[Volatile]) == 0)
270  {
271  pparent->u.KeyNode.MaxNameLen = 0;
272  pparent->u.KeyNode.MaxClassLen = 0;
273  }
274  }
275 
276  //
277  // Target is now an unreferenced key, free it's actual storage
278  //
279 
280  //
281  // Free misc stuff
282  //
283  if (!(ptarget->u.KeyNode.Flags & KEY_HIVE_EXIT) &&
284  !(ptarget->u.KeyNode.Flags & KEY_PREDEF_HANDLE) ) {
285 
286  //
287  // First, free the value entries
288  //
289  if (ptarget->u.KeyNode.ValueList.Count > 0) {
290 
291  // target list
292  plist = HvGetCell(Hive, ptarget->u.KeyNode.ValueList.List);
293  if( plist == NULL ) {
294  //
295  // we couldn't map the bin containing this cell
296  // we shouldn't hit this as we just marked the cell dirty
297  //
298  ASSERT( FALSE );
299  return STATUS_INSUFFICIENT_RESOURCES;
300  }
301  //
302  // release the cell here as it is dirty(pinned); it cannot go anywhere
303  //
304  HvReleaseCell(Hive, ptarget->u.KeyNode.ValueList.List);
305 
306  for (i = 0; i < ptarget->u.KeyNode.ValueList.Count; i++) {
307  //
308  // even if we cannot free the value here, we ignore it.
309  // nothing bad happens (just some leaks)
310  //
311  if( CmpFreeValue(Hive, plist->u.KeyList[i]) == FALSE ) {
312  //
313  // we couldn't map view inside call above
314  // this shouldn't happen as we just marked the values dirty
315  // (i.e. they should be PINNED into memory by now)
316  //
317  ASSERT( FALSE );
318  }
319  }
320 
321  HvFreeCell(Hive, ptarget->u.KeyNode.ValueList.List);
322  }
323 
324  //
325  // Free the security descriptor
326  //
327  CmLockHiveSecurityExclusive((PCMHIVE)Hive);
328  CmpFreeSecurityDescriptor(Hive, Cell);
329  CmUnlockHiveSecurity((PCMHIVE)Hive);
330  }
331 
332  //
333  // Free the key body itself, and Class data.
334  //
335  if( CmpFreeKeyBody(Hive, Cell) == FALSE ) {
336  //
337  // couldn't map view inside
338  //
339  return STATUS_INSUFFICIENT_RESOURCES;
340  }
341 
342  return STATUS_SUCCESS;
343 }

347 CmpMarkKeyDirty(
348  PHHIVE Hive,
349  HCELL_INDEX Cell
350 #if DBG
351  ,
352  BOOLEAN CheckNoSubkeys
353 #endif

355 /*++
356 
357 Routine Description:
358 
359  Mark all of the cells related to a key being deleted dirty.
360  Includes the parent, the parent's child list, the key body,
361  class, security, all value entry bodies, and all of their data cells.
362 
363 Arguments:
364 
365  Hive - pointer to hive control structure for hive of interest
366 
367  Cell - index for cell holding keybody to make dirty
368 
369 Return Value:
370 
371  TRUE - it worked
372 
373  FALSE - some error, most likely cannot get log space
374 
375 --*/
376 {
377  PCELL_DATA ptarget;
378  PCELL_DATA plist;
379  PCELL_DATA security;
380  PCELL_DATA pvalue;
381  ULONG i;
382 
383  //
384  // we have the lock exclusive or nobody is operating inside this hive
385  //
386  ASSERT_HIVE_FLUSHER_LOCKED((PCMHIVE)Hive);
387 
388  //
389  // Map in the target
390  //
391  ptarget = HvGetCell(Hive, Cell);
392  if( ptarget == NULL ) {
393  //
394  // we couldn't map the bin containing this cell
395  //
396  return FALSE;
397  }
398 
399 #if DBG
400  if(CheckNoSubkeys == TRUE) {
401  ASSERT(ptarget->u.KeyNode.SubKeyCounts[Stable] == 0);
402  ASSERT(ptarget->u.KeyNode.SubKeyCounts[Volatile] == 0);
403  }
404 #endif //DBG
405 
406  if (ptarget->u.KeyNode.Flags & KEY_HIVE_EXIT) {
407 
408  //
409  // If this is a link node, we are done. Link nodes never have
410  // classes, values, subkeys, or security descriptors. Since
411  // they always reside in the master hive, they're always volatile.
412  //
413  HvReleaseCell(Hive, Cell);
414  return(TRUE);
415  }
416 
417  //
418  // mark cell itself
419  //
420  if (! HvMarkCellDirty(Hive, Cell, FALSE)) {
421  HvReleaseCell(Hive, Cell);
422  return FALSE;
423  }
424  //
425  // we can safely release it here, as it is now dirty/pinned
426  //
427  HvReleaseCell(Hive, Cell);
428 
429  //
430  // Mark the class
431  //
432  if (ptarget->u.KeyNode.Class != HCELL_NIL) {
433  if (! HvMarkCellDirty(Hive, ptarget->u.KeyNode.Class, FALSE)) {
434  return FALSE;
435  }
436  }
437 
438  //
439  // Mark security
440  //
441  if (ptarget->u.KeyNode.Security != HCELL_NIL) {
442  if (! HvMarkCellDirty(Hive, ptarget->u.KeyNode.Security, FALSE)) {
443  return FALSE;
444  }
445 
446  security = HvGetCell(Hive, ptarget->u.KeyNode.Security);
447  if( security == NULL ) {
448  //
449  // we couldn't map the bin containing this cell
450  // we shouldn't hit this as we just marked the cell dirty
451  // (i.e. dirty == PINNED in memory)
452  //
453  ASSERT( FALSE );
454  return FALSE;
455  }
456 
457  //
458  // we can safely release it here, as it is now dirty/pinned
459  //
460  HvReleaseCell(Hive, ptarget->u.KeyNode.Security);
461 
462  if (! (HvMarkCellDirty(Hive, security->u.KeySecurity.Flink, FALSE) &&
463  HvMarkCellDirty(Hive, security->u.KeySecurity.Blink, FALSE)))
464  {
465  return FALSE;
466  }
467  }
468 
469  //
470  // Mark the value entries and their data
471  //
472  if ( !(ptarget->u.KeyNode.Flags & KEY_PREDEF_HANDLE) &&
473  (ptarget->u.KeyNode.ValueList.Count > 0)
474  ) {
475 
476  // target list
477  if (! HvMarkCellDirty(Hive, ptarget->u.KeyNode.ValueList.List, FALSE)) {
478  return FALSE;
479  }
480  plist = HvGetCell(Hive, ptarget->u.KeyNode.ValueList.List);
481  if( plist == NULL ) {
482  //
483  // we couldn't map the bin containing this cell
484  // we shouldn't hit this as we just marked the cell dirty
485  // (i.e. dirty == PINNED in memory)
486  //
487  ASSERT( FALSE );
488  return FALSE;
489  }
490 
491  //
492  // we can safely release it here, as it is now dirty/pinned
493  //
494  HvReleaseCell(Hive, ptarget->u.KeyNode.ValueList.List);
495 
496  for (i = 0; i < ptarget->u.KeyNode.ValueList.Count; i++) {
497  if (! HvMarkCellDirty(Hive, plist->u.KeyList[i], FALSE)) {
498  return FALSE;
499  }
500 
501  pvalue = HvGetCell(Hive, plist->u.KeyList[i]);
502  if( pvalue == NULL ) {
503  //
504  // we couldn't map the bin containing this cell
505  // we shouldn't hit this as we just marked the cell dirty
506  // (i.e. dirty == PINNED in memory)
507  //
508  ASSERT( FALSE );
509  return FALSE;
510  }
511 
512  //
513  // we can safely release it here, as it is now dirty/pinned
514  //
515  HvReleaseCell(Hive,plist->u.KeyList[i]);
516 
517  if( !CmpMarkValueDataDirty(Hive,&(pvalue->u.KeyValue)) ) {
518  return FALSE;
519  }
520  }
521  }
522 
523  if (ptarget->u.KeyNode.Flags & KEY_HIVE_ENTRY) {
524 
525  //
526  // if this is an entry node, we are done. our parent will
527  // be in the master hive (and thus volatile)
528  //
529  return TRUE;
530  }
531 
532  //
533  // Mark the parent's Subkey list
534  //
535  if (! CmpMarkIndexDirty(Hive, ptarget->u.KeyNode.Parent, Cell)) {
536  return FALSE;
537  }
538 
539  //
540  // Mark the parent
541  //
542  if (! HvMarkCellDirty(Hive, ptarget->u.KeyNode.Parent, FALSE)) {
543  return FALSE;
544  }
545 
546  return TRUE;
547 }
