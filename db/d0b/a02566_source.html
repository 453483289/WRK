38 VOID
39 PerfInfoFlushProfileCache(
40  VOID
41  )
42 /*++
43 
44 Routine description:
45 
46  Flushes the profile cache to the log buffer. To make sure it get's valid data
47  we read the 2 separate version numbers (1 before and 1 after) to check if it's
48  been changed. If so, we just read again. If that fails often, then we disable
49  the cache. Once the cache is read, we clear it. This may cause samples to be
50  lost but that's ok as this is statistical and it won't matter.
51 
52 Arguments:
53  CheckVersion - If FALSE, the version is not checked. This used when the profile
54  interrupt code flushes the cache.
55 
56 Return Value:
57  None
58 
59 --*/
60 {
61  ULONG PreviousInProgress;
62 
63  if ((PerfProfileCache.Entries == 0) || (PerfInfoSampledProfileCaching == FALSE)) {
64  return;
65  }
66 
67  //
68  // Signal the interrupt not to mess with the cache
69  //
70  PreviousInProgress = InterlockedIncrement(&PerfInfoSampledProfileFlushInProgress);
71  if (PreviousInProgress != 1) {
72  //
73  // A flush is already in progress so just return.
74  //
75  InterlockedDecrement(&PerfInfoSampledProfileFlushInProgress);
76  return;
77  }
78 
79  //
80  // Log the portion of the cache that has valid data.
81  //
82  PerfInfoLogBytes(PERFINFO_LOG_TYPE_SAMPLED_PROFILE_CACHE,
83  &PerfProfileCache,
84  FIELD_OFFSET(PERFINFO_SAMPLED_PROFILE_CACHE, Sample) +
85  (PerfProfileCache.Entries *
86  sizeof(PERFINFO_SAMPLED_PROFILE_INFORMATION))
87  );
88 
89  //
90  // Clear the cache for the next set of entries.
91  //
92  PerfProfileCache.Entries = 0;
93 
94  //
95  // Let the interrupt fill the cache again.
96  //
97  InterlockedDecrement(&PerfInfoSampledProfileFlushInProgress);
98 }

103 PerfProfileInterrupt(
104  IN KPROFILE_SOURCE Source,
105  IN PVOID InstructionPointer
106  )
107 /*++
108 
109 Routine description:
110 
111  Implements instruction profiling. If the source is not the one we're sampling on,
112  we return. If caching is off, we write any samples coming from the immediately to
113  the log. If caching is on, wrap the cache update with writes to the two versions so
114  that the flush routine can know if it has a valid buffer.
115 
116 Arguments:
117 
118  Source - Type of profile interrupt
119 
120  InstructionPointer - IP at the time of the interrupt
121 
122 Return Value:
123  None
124 
125 --*/
126 {
127  ULONG i;
128  PERFINFO_SAMPLED_PROFILE_INFORMATION SampleData;
129 #ifdef _X86_
130  ULONG_PTR TwiddledIP;
131 #endif // _X86_
132  ULONG ThreadId;
133 
134  if (!PERFINFO_IS_GROUP_ON(PERF_PROFILE) &&
135  (Source != PerfInfoProfileSourceActive)
136  ) {
137  //
138  // We don't handle multple sources.
139  //
140  return;
141  }
142 
143  ThreadId = HandleToUlong(PsGetCurrentThread()->Cid.UniqueThread);
144 
145  if (!PerfInfoSampledProfileCaching ||
146  PerfInfoSampledProfileFlushInProgress != 0) {
147  //
148  // No caching. Log and return.
149  //
150  SampleData.ThreadId = ThreadId;
151  SampleData.InstructionPointer = InstructionPointer;
152  SampleData.Count = 1;
153 
154  PerfInfoLogBytes(PERFINFO_LOG_TYPE_SAMPLED_PROFILE,
155  &SampleData,
156  sizeof(PERFINFO_SAMPLED_PROFILE_INFORMATION)
157  );
158  return;
159  }
160 
161 #ifdef _X86_
162  //
163  // Clear the low two bits to have more cache hits for loops. Don't waste
164  // cycles on other architectures.
165  //
166  TwiddledIP = (ULONG_PTR)InstructionPointer & ~3;
167 #endif // _X86_
168 
169  //
170  // Initial walk thru Instruction Pointer Cache. Bump Count if address is in cache.
171  //
172  for (i = 0; i < PerfProfileCache.Entries ; i++) {
173 
174  if ((PerfProfileCache.Sample[i].ThreadId == ThreadId) &&
175 #ifdef _X86_
176  (((ULONG_PTR)PerfProfileCache.Sample[i].InstructionPointer & ~3) == TwiddledIP)
177 #else
178  (PerfProfileCache.Sample[i].InstructionPointer == InstructionPointer)
179 #endif // _X86_
180  ) {
181  //
182  // If we find the instruction pointer in the cache, bump the count
183  //
184 
185  PerfProfileCache.Sample[i].Count++;
186  return;
187  }
188  }
189  if (PerfProfileCache.Entries < PERFINFO_SAMPLED_PROFILE_CACHE_MAX) {
190  //
191  // If we find an empty spot in the cache, use it for this instruction pointer
192  //
193 
194  PerfProfileCache.Sample[i].ThreadId = ThreadId;
195  PerfProfileCache.Sample[i].InstructionPointer = InstructionPointer;
196  PerfProfileCache.Sample[i].Count = 1;
197  PerfProfileCache.Entries++;
198  return;
199  }
200 
201  //
202  // Flush the cache
203  //
204  PerfInfoLogBytes(PERFINFO_LOG_TYPE_SAMPLED_PROFILE_CACHE,
205  &PerfProfileCache,
206  sizeof(PERFINFO_SAMPLED_PROFILE_CACHE)
207  );
208 
209  PerfProfileCache.Sample[0].ThreadId = ThreadId;
210  PerfProfileCache.Sample[0].InstructionPointer = InstructionPointer;
211  PerfProfileCache.Sample[0].Count = 1;
212  PerfProfileCache.Entries = 1;
213  return;
214 }

370 PerfInfoCalcHashValue(
371  PVOID Key,
372  ULONG Len
373  )
374 
375 /*++
376 
377 Routine Description:
378 
379  Generic hash routine.
380 
381 Arguments:
382 
383  Key - Pointer to data to calculate a hash value for.
384 
385  Len - Number of bytes pointed to by key.
386 
387 Return Value:
388 
389  Hash value.
390 
391 --*/
392 
393 {
394  char *cp = Key;
395  ULONG i, ConvKey=0;
396  for(i = 0; i < Len; i++)
397  {
398  ConvKey = 37 * ConvKey + (unsigned int) *cp;
399  cp++;
400  }
401 
402  #define RNDM_CONSTANT 314159269
403  #define RNDM_PRIME 1000000007
404 
405  return (abs(RNDM_CONSTANT * ConvKey) % RNDM_PRIME);
406 }

409 BOOLEAN
410 PerfInfoAddToFileHash(
411  PPERFINFO_ENTRY_TABLE HashTable,
412  PFILE_OBJECT ObjectPointer
413  )
414 /*++
415 
416 Routine Description:
417 
418  This routine add a FileObject into the specified
419  hash table if it is not already there.
420 
421 Arguments:
422 
423  HashTable - pointer to a hash table to be used.
424 
425  ObjectPointer - This is used as a key to identify a mapping.
426 
427 Return Value:
428 
429  TRUE - If either the FileObject was in the table or we add it.
430  FALSE - If the table is full.
431 
432 --*/
433 {
434  ULONG HashIndex;
435  LONG i;
436  BOOLEAN Result = FALSE;
437  LONG TableSize = HashTable->NumberOfEntries;
438  PVOID *Table;
439 
440  ASSERT (ObjectPointer != NULL);
441 
442  Table = HashTable->Table;
443  //
444  // Get the hashed index into the table where the entry ideally
445  // should be at.
446  //
447 
448  HashIndex = PerfInfoCalcHashValue((PVOID)&ObjectPointer,
449  sizeof(ObjectPointer)) % TableSize;
450 
451  for (i = 0; i < TableSize; i++) {
452 
453  if(Table[HashIndex] == NULL) {
454  //
455  // Found a empty slot. Reference the object and insert
456  // it into the table.
457  //
458  ObReferenceObject(ObjectPointer);
459  Table[HashIndex] = ObjectPointer;
460 
461  Result = TRUE;
462  break;
463  } else if (Table[HashIndex] == ObjectPointer) {
464  //
465  // Found a slot. Reference the object and insert
466  // it into the table.
467  //
468  Result = TRUE;
469  break;
470  }
471 
472  //
473  // Try next slot.
474  //
475  HashIndex = (HashIndex + 1) % TableSize;
476  }
477  return Result;
478 }

482 PerfInfoFileNameRunDown (
483  )
484 /*++
485 
486 Routine Description:
487 
488  This routine walks through multiple lists to collect the names of all files.
489  It includes:
490  1. Handle table: for all file handles
491  2. Process Vad for all file objects mapped in VAD.
492  3. MmUnusedSegment List
493  4. CcDirtySharedCacheMapList & CcCleanSharedCacheMapList
494 
495 Arguments:
496 
497  None.
498 
499 --*/
500 {
501  PEPROCESS Process;
502  ULONG AllocateBytes;
503  PFILE_OBJECT *FileObjects;
504  PFILE_OBJECT *File;
505  PERFINFO_ENTRY_TABLE HashTable;
506  extern POBJECT_TYPE IoFileObjectType;
507  POBJECT_NAME_INFORMATION FileNameInfo;
508  ULONG ReturnLen;
509  NTSTATUS Status;
510  LONG i;
511 
512  //
513  // First create a temporary hash table to build the list of
514  // files to walk through
515  //
516  AllocateBytes = PAGE_SIZE + sizeof(PVOID) * IoFileObjectType->TotalNumberOfObjects;
517 
518  //
519  // Run up to page boundary
520  //
521  AllocateBytes = PERFINFO_ROUND_UP(AllocateBytes, PAGE_SIZE);
522 
523  HashTable.Table = ExAllocatePoolWithTag(NonPagedPool, AllocateBytes, PERFPOOLTAG);
524 
525  if (HashTable.Table == NULL) {
526  return STATUS_NO_MEMORY;
527  } else {
528  //
529  // Allocation Succeeded
530  //
531  HashTable.NumberOfEntries = AllocateBytes / sizeof(PVOID);
532  RtlZeroMemory(HashTable.Table, AllocateBytes);
533  }
534 
535  //
536  // Allocate Buffers for FileNames
537  //
538  FileNameInfo = ExAllocatePoolWithTag (NonPagedPool, MAX_FILENAME_TO_LOG, PERFPOOLTAG);
539 
540  if (FileNameInfo == NULL) {
541  ExFreePool(HashTable.Table);
542  return STATUS_NO_MEMORY;
543  }
544 
545  //
546  // Walk through the Cc SharedCacheMapList
547  //
548 
549  CcPerfFileRunDown(&HashTable);
550 
551  //
552  // Now, walk through each process
553  //
554  for (Process = PsGetNextProcess (NULL);
555  Process != NULL;
556  Process = PsGetNextProcess (Process)) {
557 
558  //
559  // First Walk the VAD tree
560  //
561 
562  FileObjects = MmPerfVadTreeWalk(Process);
563  if (FileObjects != NULL) {
564  File = FileObjects;
565  while (*File != NULL) {
566  PerfInfoAddToFileHash(&HashTable, *File);
567  ObDereferenceObject(*File);
568  File += 1;
569  }
570  ExFreePool(FileObjects);
571  }
572 
573  //
574  // Next, walk the handle Table
575  //
576  ObPerfHandleTableWalk (Process, &HashTable);
577  }
578 
579  //
580  // Walk through the kernel handle table;
581  //
582  ObPerfHandleTableWalk(NULL, &HashTable);
583 
584  //
585  // Walk through the MmUnusedSegmentList;
586  //
587 
588  FileObjects = MmPerfUnusedSegmentsEnumerate();
589 
590  if (FileObjects != NULL) {
591  File = FileObjects;
592  while (*File != NULL) {
593  PerfInfoAddToFileHash(&HashTable, *File);
594  ObDereferenceObject(*File);
595  File += 1;
596  }
597  ExFreePool(FileObjects);
598  }
599 
600  //
601  // Now we have walked through all list.
602  // Log the filenames and dereference the objects.
603  //
604 
605  for (i = 0; i < HashTable.NumberOfEntries; i++) {
606  if (HashTable.Table[i]) {
607  PFILE_OBJECT FileObject = HashTable.Table[i];
608 
609  Status = ObQueryNameString( FileObject,
610  FileNameInfo,
611  MAX_FILENAME_TO_LOG,
612  &ReturnLen
613  );
614 
615  if (NT_SUCCESS (Status)) {
616  PerfInfoLogFileName(FileObject, &FileNameInfo->Name);
617  }
618  ObDereferenceObject(FileObject);
619  }
620  }
621 
622  //
623  // Free the pool reserved.
624  //
625  ExFreePool(HashTable.Table);
626  ExFreePool(FileNameInfo);
627 
628  return STATUS_SUCCESS;
629 }

633 PerfInfoProcessRunDown (
634  )
635 /*++
636 
637 Routine Description:
638 
639  This routine does the Process and thread rundown in the kernel mode.
640  Since this routine is called only by global logger (i.e., trace from boot),
641  no Sid info is collected.
642 
643 Arguments:
644 
645  None.
646 
647 Return Value:
648 
649  Status
650 
651 --*/
652 {
653  NTSTATUS Status;
654  PSYSTEM_PROCESS_INFORMATION ProcessInfo;
655  PSYSTEM_EXTENDED_THREAD_INFORMATION ThreadInfo;
656  PCHAR Buffer;
657  ULONG BufferSize = 4096;
658  ULONG ReturnLength;
659 
660 retry:
661  Buffer = ExAllocatePoolWithTag(NonPagedPool, BufferSize, PERFPOOLTAG);
662 
663  if (!Buffer) {
664  return STATUS_NO_MEMORY;
665  }
666  Status = ZwQuerySystemInformation( SystemExtendedProcessInformation,
667  Buffer,
668  BufferSize,
669  &ReturnLength
670  );
671 
672  if (Status == STATUS_INFO_LENGTH_MISMATCH) {
673  ExFreePool(Buffer);
674  BufferSize = ReturnLength;
675  goto retry;
676  }
677 
678  if (NT_SUCCESS(Status)) {
679  ULONG TotalOffset = 0;
680  ProcessInfo = (PSYSTEM_PROCESS_INFORMATION) Buffer;
681  while (TRUE) {
682  PWMI_PROCESS_INFORMATION WmiProcessInfo;
683  PWMI_EXTENDED_THREAD_INFORMATION WmiThreadInfo;
684  PERFINFO_HOOK_HANDLE Hook;
685  ANSI_STRING ProcessName;
686  PCHAR AuxPtr;
687  ULONG NameLength;
688  ULONG ByteCount;
689  ULONG SidLength = sizeof(ULONG);
690  ULONG TmpSid = 0;
691  ULONG i;
692 
693  //
694  // Process Information
695  //
696  if ( ProcessInfo->ImageName.Buffer && ProcessInfo->ImageName.Length > 0 ) {
697  NameLength = ProcessInfo->ImageName.Length / sizeof(WCHAR) + 1;
698  }
699  else {
700  NameLength = 1;
701  }
702  ByteCount = FIELD_OFFSET(WMI_PROCESS_INFORMATION, Sid) + SidLength + NameLength;
703 
704  Status = PerfInfoReserveBytes(&Hook,
705  WMI_LOG_TYPE_PROCESS_DC_START,
706  ByteCount);
707 
708  if (NT_SUCCESS(Status)){
709  WmiProcessInfo = PERFINFO_HOOK_HANDLE_TO_DATA(Hook, PWMI_PROCESS_INFORMATION);
710 
711  WmiProcessInfo->ProcessId = HandleToUlong(ProcessInfo->UniqueProcessId);
712  WmiProcessInfo->ParentId = HandleToUlong(ProcessInfo->InheritedFromUniqueProcessId);
713  WmiProcessInfo->SessionId = ProcessInfo->SessionId;
714  WmiProcessInfo->PageDirectoryBase = ProcessInfo->PageDirectoryBase;
715 
716  AuxPtr = (PCHAR) &WmiProcessInfo->Sid;
717  RtlCopyMemory(AuxPtr, &TmpSid, SidLength);
718 
719  AuxPtr += SidLength;
720  if (NameLength > 1) {
721 
722  ProcessName.Buffer = AuxPtr;
723  ProcessName.MaximumLength = (USHORT) NameLength;
724 
725  RtlUnicodeStringToAnsiString( &ProcessName,
726  (PUNICODE_STRING) &ProcessInfo->ImageName,
727  FALSE);
728  AuxPtr += NameLength - 1; //point to the place for the '\0'
729  }
730  *AuxPtr = '\0';
731 
732  PERF_FINISH_HOOK(Hook);
733  }
734 
735  //
736  // Thread Information
737  //
738  ThreadInfo = (PSYSTEM_EXTENDED_THREAD_INFORMATION) (ProcessInfo + 1);
739 
740  for (i=0; i < ProcessInfo->NumberOfThreads; i++) {
741  Status = PerfInfoReserveBytes(&Hook,
742  WMI_LOG_TYPE_THREAD_DC_START,
743  sizeof(WMI_EXTENDED_THREAD_INFORMATION));
744  if (NT_SUCCESS(Status)){
745  WmiThreadInfo = PERFINFO_HOOK_HANDLE_TO_DATA(Hook, PWMI_EXTENDED_THREAD_INFORMATION);
746  WmiThreadInfo->ProcessId = HandleToUlong(ThreadInfo->ThreadInfo.ClientId.UniqueProcess);
747  WmiThreadInfo->ThreadId = HandleToUlong(ThreadInfo->ThreadInfo.ClientId.UniqueThread);
748  WmiThreadInfo->StackBase = ThreadInfo->StackBase;
749  WmiThreadInfo->StackLimit = ThreadInfo->StackLimit;
750 
751  WmiThreadInfo->UserStackBase = NULL;
752  WmiThreadInfo->UserStackLimit = NULL;
753  WmiThreadInfo->StartAddr = ThreadInfo->ThreadInfo.StartAddress;
754  WmiThreadInfo->Win32StartAddr = ThreadInfo->Win32StartAddress;
755  WmiThreadInfo->WaitMode = -1;
756  PERF_FINISH_HOOK(Hook);
757  }
758 
759  ThreadInfo += 1;
760  }
761 
762  if (ProcessInfo->NextEntryOffset == 0) {
763  break;
764  } else {
765  TotalOffset += ProcessInfo->NextEntryOffset;
766  ProcessInfo = (PSYSTEM_PROCESS_INFORMATION) &Buffer[TotalOffset];
767  }
768  }
769  }
770 
771  ExFreePool(Buffer);
772  return Status;
773 
774 }
