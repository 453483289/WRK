235 /*++
236 
237 Routine Description:
238 
239  This function checks the supplied worker queue and determines whether
240  it is appropriate to spin up a dynamic worker thread for that queue.
241 
242 Arguments:
243 
244  Queue - Supplies the queue that should be examined.
245 
246 Return Value:
247 
248  TRUE if the given work queue would benefit from the creation of an
249  additional thread, FALSE if not.
250 
251 --*/
252 {
253  if ((Queue->Info.MakeThreadsAsNecessary == 1) &&
254  (IsListEmpty (&Queue->WorkerQueue.EntryListHead) == FALSE) &&
255  (Queue->WorkerQueue.CurrentCount < Queue->WorkerQueue.MaximumCount) &&
256  (Queue->DynamicThreadCount < MAX_ADDITIONAL_DYNAMIC_THREADS)) {
257 
258  //
259  // We know these things:
260  //
261  // - This queue is eligible for dynamic creation of threads to try
262  // to keep the CPUs busy,
263  //
264  // - There are work items waiting in the queue,
265  //
266  // - The number of runable worker threads for this queue is less than
267  // the number of processors on this system, and
268  //
269  // - We haven't reached the maximum dynamic thread count.
270  //
271  // An additional worker thread at this point will help clear the
272  // backlog.
273  //
274 
275  return TRUE;
276  }
277 
278  //
279  // One of the above conditions is false.
280  //
281 
282  return FALSE;

286 ExpWorkerInitialization (
287  VOID
288  )
289 {
290  ULONG Index;
291  OBJECT_ATTRIBUTES ObjectAttributes;
292  ULONG NumberOfDelayedThreads;
293  ULONG NumberOfCriticalThreads;
294  ULONG NumberOfThreads;
295  NTSTATUS Status;
296  HANDLE Thread;
297  BOOLEAN NtAs;
298  WORK_QUEUE_TYPE WorkQueueType;
299 
300  ExInitializeFastMutex (&ExpWorkerSwapinMutex);
301  InitializeListHead (&ExpWorkerListHead);
302  ExpWorkersCanSwap = TRUE;
303 
304  //
305  // Set the number of worker threads based on the system size.
306  //
307 
308  NtAs = MmIsThisAnNtAsSystem();
309 
310  NumberOfCriticalThreads = MEDIUM_NUMBER_OF_THREADS;
311 
312  //
313  // Incremented boot time number of delayed threads.
314  // We did this in Windows XP, because 3COM NICs would take a long
315  // time with the network stack tying up the delayed worker threads.
316  // When Mm would need a worker thread to load a driver on the critical
317  // path of boot, it would also get stuck for a few seconds and hurt
318  // boot times. Ideally we'd spawn new delayed threads as necessary as
319  // well to prevent such contention from hurting boot and resume.
320  //
321 
322  NumberOfDelayedThreads = MEDIUM_NUMBER_OF_THREADS + 4;
323 
324  switch (MmQuerySystemSize()) {
325 
326  case MmSmallSystem:
327  break;
328 
329  case MmMediumSystem:
330  if (NtAs) {
331  NumberOfCriticalThreads += MEDIUM_NUMBER_OF_THREADS;
332  }
333  break;
334 
335  case MmLargeSystem:
336  NumberOfCriticalThreads = LARGE_NUMBER_OF_THREADS;
337  if (NtAs) {
338  NumberOfCriticalThreads += LARGE_NUMBER_OF_THREADS;
339  }
340  break;
341 
342  default:
343  break;
344  }
345 
346  //
347  // Initialize the work Queue objects.
348  //
349 
350  if (ExpAdditionalCriticalWorkerThreads > MAX_ADDITIONAL_THREADS) {
351  ExpAdditionalCriticalWorkerThreads = MAX_ADDITIONAL_THREADS;
352  }
353 
354  if (ExpAdditionalDelayedWorkerThreads > MAX_ADDITIONAL_THREADS) {
355  ExpAdditionalDelayedWorkerThreads = MAX_ADDITIONAL_THREADS;
356  }
357 
358  //
359  // Initialize the ExWorkerQueue[] array.
360  //
361 
362  RtlZeroMemory (&ExWorkerQueue[0], MaximumWorkQueue * sizeof(EX_WORK_QUEUE));
363 
364  for (WorkQueueType = 0; WorkQueueType < MaximumWorkQueue; WorkQueueType += 1) {
365 
366  KeInitializeQueue (&ExWorkerQueue[WorkQueueType].WorkerQueue, 0);
367  ExWorkerQueue[WorkQueueType].Info.WaitMode = UserMode;
368  }
369 
370  //
371  // Always make stack for this thread resident
372  // so that worker pool deadlock magic can run
373  // even when what we are trying to do is inpage
374  // the hyper critical worker thread's stack.
375  // Without this fix, we hold the process lock
376  // but this thread's stack can't come in, and
377  // the deadlock detection cannot create new threads
378  // to break the system deadlock.
379  //
380 
381  ExWorkerQueue[HyperCriticalWorkQueue].Info.WaitMode = KernelMode;
382 
383  if (NtAs) {
384  ExWorkerQueue[CriticalWorkQueue].Info.WaitMode = KernelMode;
385  }
386 
387  //
388  // We only create dynamic threads for the critical work queue (note
389  // this doesn't apply to dynamic threads created to break deadlocks.)
390  //
391  // The rationale is this: folks who use the delayed work queue are
392  // not time critical, and the hypercritical queue is used rarely
393  // by folks who are non-blocking.
394  //
395 
396  ExWorkerQueue[CriticalWorkQueue].Info.MakeThreadsAsNecessary = 1;
397 
398  //
399  // Initialize the global thread set manager events
400  //
401 
402  KeInitializeEvent (&ExpThreadSetManagerEvent,
403  SynchronizationEvent,
404  FALSE);
405 
406  KeInitializeEvent (&ExpThreadSetManagerShutdownEvent,
407  SynchronizationEvent,
408  FALSE);
409 
410  //
411  // Create the desired number of executive worker threads for each
412  // of the work queues.
413  //
414 
415  //
416  // Create the builtin critical worker threads.
417  //
418 
419  NumberOfThreads = NumberOfCriticalThreads + ExpAdditionalCriticalWorkerThreads;
420  for (Index = 0; Index < NumberOfThreads; Index += 1) {
421 
422  //
423  // Create a worker thread to service the critical work queue.
424  //
425 
426  Status = ExpCreateWorkerThread (CriticalWorkQueue, FALSE);
427 
428  if (!NT_SUCCESS(Status)) {
429  break;
430  }
431  }
432 
433  ExCriticalWorkerThreads += Index;
434 
435  //
436  // Create the delayed worker threads.
437  //
438 
439  NumberOfThreads = NumberOfDelayedThreads + ExpAdditionalDelayedWorkerThreads;
440  for (Index = 0; Index < NumberOfThreads; Index += 1) {
441 
442  //
443  // Create a worker thread to service the delayed work queue.
444  //
445 
446  Status = ExpCreateWorkerThread (DelayedWorkQueue, FALSE);
447 
448  if (!NT_SUCCESS(Status)) {
449  break;
450  }
451  }
452 
453  ExDelayedWorkerThreads += Index;
454 
455  //
456  // Create the hypercritical worker thread.
457  //
458 
459  Status = ExpCreateWorkerThread (HyperCriticalWorkQueue, FALSE);
460 
461  //
462  // Create the worker thread set manager thread.
463  //
464 
465  InitializeObjectAttributes (&ObjectAttributes, NULL, 0, NULL, NULL);
466 
467  Status = PsCreateSystemThread (&Thread,
468  THREAD_ALL_ACCESS,
469  &ObjectAttributes,
470  0,
471  NULL,
472  ExpWorkerThreadBalanceManager,
473  NULL);
474 
475  if (NT_SUCCESS(Status)) {
476  Status = ObReferenceObjectByHandle (Thread,
477  SYNCHRONIZE,
478  NULL,
479  KernelMode,
480  &ExpWorkerThreadBalanceManagerPtr,
481  NULL);
482  ZwClose (Thread);
483  }
484 
485  return Status;

493 
494 /*++
495 
496 Routine Description:
497 
498  This function inserts a work item into a work queue that is processed
499  by a worker thread of the corresponding type.
500 
501 Arguments:
502 
503  WorkItem - Supplies a pointer to the work item to add the the queue.
504  This structure must be located in NonPagedPool. The work item
505  structure contains a doubly linked list entry, the address of a
506  routine to call and a parameter to pass to that routine.
507 
508  QueueType - Specifies the type of work queue that the work item
509  should be placed in.
510 
511 Return Value:
512 
513  None.
514 
515 --*/
516 
517 {
518  PEX_WORK_QUEUE Queue;
519 
520  ASSERT (QueueType < MaximumWorkQueue);
521  ASSERT (WorkItem->List.Flink == NULL);
522 
523  //
524  // Perform a rudimentary validation on the worker routine.
525  //
526 
527  if ((ULONG64)WorkItem->WorkerRoutine <= MmUserProbeAddress) {
528 
529  KeBugCheckEx (WORKER_INVALID,
530  0x1,
531  (ULONG_PTR)WorkItem,
532  (ULONG_PTR)WorkItem->WorkerRoutine,
533  0);
534  }
535 
536  Queue = &ExWorkerQueue[QueueType];
537 
538  //
539  // Insert the work item in the appropriate queue object.
540  //
541 
542  KeInsertQueue (&Queue->WorkerQueue, &WorkItem->List);
543 
544  //
545  // We check the queue's shutdown state after we insert the work
546  // item to avoid the race condition when the queue's marked
547  // between checking the queue and inserting the item. It's
548  // possible for the queue to be marked for shutdown between the
549  // insert and this assert (so the insert would've barely sneaked
550  // in), but it's not worth guarding against this -- barely
551  // sneaking in is not a good design strategy, and at this point in
552  // the shutdown sequence, the caller simply should not be trying
553  // to insert new queue items.
554  //
555 
556  ASSERT (!Queue->Info.QueueDisabled);
557 
558  //
559  // Determine whether another thread should be created, and signal the
560  // thread set balance manager if so.
561  //
562 
563  if (ExpNewThreadNecessary (Queue) != FALSE) {
564  KeSetEvent (&ExpThreadSetManagerEvent, 0, FALSE);
565  }
566 
567  return;
568 }

572  IN PVOID StartContext
573  )
574 
575 /*++
576 
577 Routine Description:
578 
579  This function is the startup code for the worker thread manager thread.
580  The worker thread manager thread is created during system initialization
581  and begins execution in this function.
582 
583  This thread is responsible for detecting and breaking circular deadlocks
584  in the system worker thread queues. It will also create and destroy
585  additional worker threads as needed based on loading.
586 
587 Arguments:
588 
589  Context - Supplies a pointer to an arbitrary data structure (NULL).
590 
591 Return Value:
592 
593  None.
594 
595 --*/
596 {
597  KTIMER PeriodTimer;
598  LARGE_INTEGER DueTime;
599  PVOID WaitObjects[MaximumBalanceObject];
600  NTSTATUS Status;
601 
602  PAGED_CODE();
603 
604  UNREFERENCED_PARAMETER (StartContext);
605 
606  //
607  // Raise the thread priority to just higher than the priority of the
608  // critical work queue.
609  //
610 
611  KeSetBasePriorityThread (KeGetCurrentThread(),
612  CRITICAL_WORK_QUEUE_PRIORITY + 1);
613 
614  //
615  // Initialize the periodic timer and set the manager period.
616  //
617 
618  KeInitializeTimer (&PeriodTimer);
619  DueTime.QuadPart = - THREAD_SET_INTERVAL;
620 
621  //
622  // Initialize the wait object array.
623  //
624 
625  WaitObjects[TimerExpiration] = (PVOID)&PeriodTimer;
626  WaitObjects[ThreadSetManagerEvent] = (PVOID)&ExpThreadSetManagerEvent;
627  WaitObjects[ShutdownEvent] = (PVOID)&ExpThreadSetManagerShutdownEvent;
628 
629  //
630  // Loop forever processing events.
631  //
632 
633  while (TRUE) {
634 
635  //
636  // Set the timer to expire at the next periodic interval.
637  //
638 
639  KeSetTimer (&PeriodTimer, DueTime, NULL);
640 
641  //
642  // Wake up when the timer expires or the set manager event is
643  // signaled.
644  //
645 
646  Status = KeWaitForMultipleObjects (MaximumBalanceObject,
647  WaitObjects,
648  WaitAny,
649  Executive,
650  KernelMode,
651  FALSE,
652  NULL,
653  NULL);
654 
655  switch (Status) {
656 
657  case TimerExpiration:
658 
659  //
660  // Periodic timer expiration - go see if any work queues
661  // are deadlocked.
662  //
663 
664  ExpDetectWorkerThreadDeadlock ();
665  break;
666 
667  case ThreadSetManagerEvent:
668 
669  //
670  // Someone has asked us to check some metrics to determine
671  // whether we should create another worker thread.
672  //
673 
674  ExpCheckDynamicThreadCount ();
675  break;
676 
677  case ShutdownEvent:
678 
679  //
680  // Time to exit...
681  //
682 
683  KeCancelTimer (&PeriodTimer);
684 
685  ASSERT (ExpLastWorkerThread);
686 
687  //
688  // Wait for the last worker thread to terminate
689  //
690 
691  KeWaitForSingleObject (ExpLastWorkerThread,
692  Executive,
693  KernelMode,
694  FALSE,
695  NULL);
696 
697  ObDereferenceObject (ExpLastWorkerThread);
698 
699  PsTerminateSystemThread(STATUS_SYSTEM_SHUTDOWN);
700 
701  break;
702  }
703 
704  //
705  // Special debugger support.
706  //
707  // This checks if special debugging routines need to be run on the
708  // behalf of the debugger.
709  //
710 
711  if (ExpDebuggerWork == 1) {
712 
713  ExInitializeWorkItem(&ExpDebuggerWorkItem, ExpDebuggerWorker, NULL);
714  ExpDebuggerWork = 2;
715  ExQueueWorkItem(&ExpDebuggerWorkItem, DelayedWorkQueue);
716  }
717  }
718 }

778 /*++
779 
780 Routine Description:
781 
782  This function creates new work item threads if a possible deadlock is
783  detected.
784 
785 Arguments:
786 
787  None.
788 
789 Return Value:
790 
791  None
792 
793 --*/

795 {
796  ULONG Index;
797  PEX_WORK_QUEUE Queue;
798 
799  PAGED_CODE();
800 
801  //
802  // Process each queue type.
803  //
804 
805  for (Index = 0; Index < MaximumWorkQueue; Index += 1) {
806 
807  Queue = &ExWorkerQueue[Index];
808 
809  ASSERT( Queue->DynamicThreadCount <= MAX_ADDITIONAL_DYNAMIC_THREADS );
810 
811  if ((Queue->QueueDepthLastPass > 0) &&
812  (Queue->WorkItemsProcessed == Queue->WorkItemsProcessedLastPass) &&
813  (Queue->DynamicThreadCount < MAX_ADDITIONAL_DYNAMIC_THREADS)) {
814 
815  //
816  // These things are known:
817  //
818  // - There were work items waiting in the queue at the last pass.
819  // - No work items have been processed since the last pass.
820  // - We haven't yet created the maximum number of dynamic threads.
821  //
822  // Things look like they're stuck, create a new thread for this
823  // queue.
824  //
825  // We explicitly ignore an error from ExpCreateDynamicThread():
826  // we'll try again in another detection period if the queue looks
827  // like it's still stuck.
828  //
829 
830  ExpCreateWorkerThread (Index, TRUE);
831  }
832 
833  //
834  // Update some bookkeeping.
835  //
836  // Note that WorkItemsProcessed and the queue depth must be recorded
837  // in that order to avoid getting a false deadlock indication.
838  //
839

841  Queue->QueueDepthLastPass = KeReadStateQueue (&Queue->WorkerQueue);
