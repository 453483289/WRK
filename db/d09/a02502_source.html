67 MiFlushRangeFilter (
68  IN PEXCEPTION_POINTERS ExceptionPointers,
69  IN PVOID *BaseAddress,
70  IN PSIZE_T Length,
71  IN PLOGICAL Retry
72  )
73 
74 /*++
75 
76 Routine Description:
77 
78  This is the exception handler used by NtFlushInstructionCache to protect
79  against bad virtual addresses passed to KeSweepIcacheRange. If an
80  access violation occurs, this routine causes NtFlushInstructionCache to
81  restart the sweep at the page following the failing page.
82 
83 Arguments:
84 
85  ExceptionPointers - Supplies exception information.
86 
87  BaseAddress - Supplies a pointer to address the base of the region being
88  flushed. If the failing address is not in the last page of
89  the region, this routine updates BaseAddress to point to the
90  next page of the region.
91 
92  Length - Supplies a pointer the length of the region being flushed.
93  If the failing address is not in the last page of the region,
94  this routine updates Length to reflect restarting the flush at
95  the next page of the region.
96 
97  Retry - Supplies a pointer to a LOGICAL that the caller has initialized
98  to FALSE. This routine sets this LOGICAL to TRUE if an access
99  violation occurs in a page before the last page of the flush region.
100 
101 Return Value:
102 
103  EXCEPTION_EXECUTE_HANDLER.
104 
105 --*/
106 
107 {
108  PEXCEPTION_RECORD ExceptionRecord;
109  ULONG_PTR BadVa;
110  ULONG_PTR NextVa;
111  ULONG_PTR EndVa;
112 
113  ExceptionRecord = ExceptionPointers->ExceptionRecord;
114 
115  //
116  // If the exception was an access violation, skip the current page of the
117  // region and move to the next page.
118  //
119 
120  if ( ExceptionRecord->ExceptionCode == STATUS_ACCESS_VIOLATION ) {
121 
122  //
123  // Get the failing address, calculate the base address of the next page,
124  // and calculate the address at the end of the region.
125  //
126 
127  BadVa = ExceptionRecord->ExceptionInformation[1];
128  NextVa = ROUND_TO_PAGES( BadVa + 1 );
129  EndVa = *(PULONG_PTR)BaseAddress + *Length;
130 
131  //
132  // If the next page didn't wrap, and the next page is below the end of
133  // the region, update Length and BaseAddress appropriately and set Retry
134  // to TRUE to indicate to NtFlushInstructionCache that it should call
135  // KeSweepIcacheRange again.
136  //
137 
138  if ( (NextVa > BadVa) && (NextVa < EndVa) ) {
139  *Length = (ULONG) (EndVa - NextVa);
140  *BaseAddress = (PVOID)NextVa;
141  *Retry = TRUE;
142  }
143  }
144 
145  return EXCEPTION_EXECUTE_HANDLER;
146 }

155 /*++
156 
157 Routine Description:
158 
159  This function flushes the instruction cache for the specified process.
160 
161 Arguments:
162 
163  ProcessHandle - Supplies a handle to the process in which the instruction
164  cache is to be flushed. Must have PROCESS_VM_WRITE access
165  to the specified process.
166 
167  BaseAddress - Supplies an optional pointer to base of the region that
168  is flushed.
169 
170  Length - Supplies the length of the region that is flushed if the base
171  address is specified.
172 
173 Return Value:
174 
175  STATUS_SUCCESS.
176 
177 --*/
178 
179 {
180  KAPC_STATE ApcState;
181  KPROCESSOR_MODE PreviousMode;
182  PEPROCESS Process;
183  NTSTATUS Status;
184  LOGICAL Retry;
185  PVOID RangeBase;
186  SIZE_T RangeLength;
187 
188  PAGED_CODE();
189 
190  PreviousMode = KeGetPreviousMode();
191 
192  //
193  // If the base address is not specified, or the base address is specified
194  // and the length is not zero, then flush the specified instruction cache
195  // range.
196  //
197 
198  if ((ARGUMENT_PRESENT(BaseAddress) == FALSE) || (Length != 0)) {
199 
200  //
201  // If previous mode is user and the range specified falls in kernel
202  // address space, return an error.
203  //
204 
205  if ((ARGUMENT_PRESENT(BaseAddress) != FALSE) &&
206  (PreviousMode != KernelMode)) {
207  try {
208  ProbeForRead(BaseAddress, Length, sizeof(UCHAR));
209  } except(EXCEPTION_EXECUTE_HANDLER) {
210  return GetExceptionCode();
211  }
212  }
213 
214  //
215  // If the specified process is not the current process, then
216  // the process must be attached to during the flush.
217  //
218 
219  Process = NULL;
220  if (ProcessHandle != NtCurrentProcess()) {
221 
222  //
223  // Reference the specified process checking for PROCESS_VM_WRITE
224  // access.
225  //
226 
227  Status = ObReferenceObjectByHandle(ProcessHandle,
228  PROCESS_VM_WRITE,
229  PsProcessType,
230  PreviousMode,
231  (PVOID *)&Process,
232  NULL);
233 
234  if (!NT_SUCCESS(Status)) {
235  return Status;
236  }
237 
238  //
239  // Attach to the process.
240  //
241 
242  KeStackAttachProcess (&Process->Pcb, &ApcState);
243  }
244 
245  //
246  // If the base address is not specified, sweep the entire instruction
247  // cache. If the base address is specified, flush the specified range.
248  //
249 
250  if (ARGUMENT_PRESENT(BaseAddress) == FALSE) {
251  KeSweepIcache(FALSE);
252 
253  } else {
254 
255  //
256  // Parts of the specified range may be invalid. An exception
257  // handler is used to skip over those parts. Before calling
258  // KeSweepIcacheRange, we set Retry to FALSE. If an access
259  // violation occurs in KeSweepIcacheRange, the MiFlushRangeFilter
260  // exception filter is called. It updates RangeBase and
261  // RangeLength to skip over the failing page, and sets Retry to
262  // TRUE. As long as Retry is TRUE, we continue to call
263  // KeSweepIcacheRange.
264  //
265 
266  RangeBase = BaseAddress;
267  RangeLength = Length;
268 
269  do {
270  Retry = FALSE;
271 
272  try {
273 
274  KeSweepIcacheRange(FALSE, RangeBase, RangeLength);
275 
276  } except(MiFlushRangeFilter(GetExceptionInformation(),
277  &RangeBase,
278  &RangeLength,
279  &Retry)) {
280  NOTHING;
281  }
282  } while (Retry != FALSE);
283  }
284 
285  //
286  // If the specified process is not the current process, then
287  // detach from it and dereference it.
288  //
289 
290  if (Process != NULL) {
291  KeUnstackDetachProcess (&ApcState);
292  ObDereferenceObject(Process);
293  }
294  }
295 
296  return STATUS_SUCCESS;
297 }
