102 Routine Description:
103 
104  This function copies the specified address range from the specified
105  process into the specified address range of the current process.
106 
107 Arguments:
108 
109  ProcessHandle - Supplies an open handle to a process object.
110 
111  BaseAddress - Supplies the base address in the specified process
112  to be read.
113 
114  Buffer - Supplies the address of a buffer which receives the
115  contents from the specified process address space.
116 
117  BufferSize - Supplies the requested number of bytes to read from
118  the specified process.
119 
120  NumberOfBytesRead - Receives the actual number of bytes
121  transferred into the specified buffer.
122 
123 Return Value:
124 
125  NTSTATUS.
126 
127 --*/
128 
129 {
130  SIZE_T BytesCopied;
131  KPROCESSOR_MODE PreviousMode;
132  PEPROCESS Process;
133  NTSTATUS Status;
134  PETHREAD CurrentThread;
135 
136  PAGED_CODE();
137 
138  //
139  // Get the previous mode and probe output argument if necessary.
140  //
141 
142  CurrentThread = PsGetCurrentThread ();
143  PreviousMode = KeGetPreviousModeByThread(&CurrentThread->Tcb);
144  if (PreviousMode != KernelMode) {
145 
146  if (((PCHAR)BaseAddress + BufferSize < (PCHAR)BaseAddress) ||
147  ((PCHAR)Buffer + BufferSize < (PCHAR)Buffer) ||
148  ((PVOID)((PCHAR)BaseAddress + BufferSize) > MM_HIGHEST_USER_ADDRESS) ||
149  ((PVOID)((PCHAR)Buffer + BufferSize) > MM_HIGHEST_USER_ADDRESS)) {
150 
151  return STATUS_ACCESS_VIOLATION;
152  }
153

155  try {
156  ProbeForWriteUlong_ptr (NumberOfBytesRead);
157 
158  } except(EXCEPTION_EXECUTE_HANDLER) {
159  return GetExceptionCode();
160  }
161  }
162  }
163 
164  //
165  // If the buffer size is not zero, then attempt to read data from the
166  // specified process address space into the current process address
167  // space.
168  //
169 
170  BytesCopied = 0;
171  Status = STATUS_SUCCESS;
172  if (BufferSize != 0) {
173 
174  //
175  // Reference the target process.
176  //
177 
178  Status = ObReferenceObjectByHandle(ProcessHandle,
179  PROCESS_VM_READ,
180  PsProcessType,
181  PreviousMode,
182  (PVOID *)&Process,
183  NULL);
184

187  // read the specified memory either by direct mapping or copying
188  // through nonpaged pool.
189  //
190 
191  if (Status == STATUS_SUCCESS) {
192 
193  Status = MmCopyVirtualMemory (Process,
194  BaseAddress,
195  PsGetCurrentProcessByThread(CurrentThread),
196  Buffer,
197  BufferSize,
198  PreviousMode,
199  &BytesCopied);
200 
201  //
202  // Dereference the target process.
203  //
204 
205  ObDereferenceObject(Process);
206  }
207  }
208 
209  //
210  // If requested, return the number of bytes read.
211  //
212 
213  if (ARGUMENT_PRESENT(NumberOfBytesRead)) {
214  try {
215  *NumberOfBytesRead = BytesCopied;
216 
217  } except(EXCEPTION_EXECUTE_HANDLER) {
218  NOTHING;
219  }
220  }
221 
222  return Status;
223 }

360 NTSTATUS
361 MmCopyVirtualMemory(
362  IN PEPROCESS FromProcess,
363  IN CONST VOID *FromAddress,
364  IN PEPROCESS ToProcess,
365  OUT PVOID ToAddress,
366  IN SIZE_T BufferSize,
367  IN KPROCESSOR_MODE PreviousMode,
368  OUT PSIZE_T NumberOfBytesCopied
369  )
370 {
371  NTSTATUS Status;
372  PEPROCESS ProcessToLock;
373 
374  if (BufferSize == 0) {
375  ASSERT (FALSE); // No one should call with a zero size.
376  return STATUS_SUCCESS;
377  }
378 
379  ProcessToLock = FromProcess;
380  if (FromProcess == PsGetCurrentProcess()) {
381  ProcessToLock = ToProcess;
382  }
383 
384  //
385  // Make sure the process still has an address space.
386  //
387 
388  if (ExAcquireRundownProtection (&ProcessToLock->RundownProtect) == FALSE) {
389  return STATUS_PROCESS_IS_TERMINATING;
390  }
391 
392  //
393  // If the buffer size is greater than the pool move threshold,
394  // then attempt to write the memory via direct mapping.
395  //
396 
397  if (BufferSize > POOL_MOVE_THRESHOLD) {
398  Status = MiDoMappedCopy(FromProcess,
399  FromAddress,
400  ToProcess,
401  ToAddress,
402  BufferSize,
403  PreviousMode,
404  NumberOfBytesCopied);
405 
406  //
407  // If the completion status is not a working quota problem,
408  // then finish the service. Otherwise, attempt to write the
409  // memory through nonpaged pool.
410  //
411 
412  if (Status != STATUS_WORKING_SET_QUOTA) {
413  goto CompleteService;
414  }
415 
416  *NumberOfBytesCopied = 0;
417  }
418 
419  //
420  // There was not enough working set quota to write the memory via
421  // direct mapping or the size of the write was below the pool move
422  // threshold. Attempt to write the specified memory through nonpaged
423  // pool.
424  //
425 
426  Status = MiDoPoolCopy(FromProcess,
427  FromAddress,
428  ToProcess,
429  ToAddress,
430  BufferSize,
431  PreviousMode,
432  NumberOfBytesCopied);
433 
434  //
435  // Dereference the target process.
436  //
437 
438 CompleteService:
439 
440  //
441  // Indicate that the vm operation is complete.
442  //
443 
444  ExReleaseRundownProtection (&ProcessToLock->RundownProtect);
445 
446  return Status;
447 }

452  IN PEXCEPTION_POINTERS ExceptionPointers,
453  IN OUT PLOGICAL ExceptionAddressConfirmed,

457 /*++
458 
459 Routine Description:
460 
461  This routine examines a exception record and extracts the virtual
462  address of an access violation, guard page violation, or in-page error.
463 
464 Arguments:
465 
466  ExceptionPointers - Supplies a pointer to the exception record.
467 
468  ExceptionAddressConfirmed - Receives TRUE if the exception address was
469  reliably detected, FALSE if not.
470 
471  BadVa - Receives the virtual address which caused the access violation.
472 
473 Return Value:
474 
475  EXECUTE_EXCEPTION_HANDLER
476 
477 --*/
478 
479 {
480  PEXCEPTION_RECORD ExceptionRecord;
481 
482  PAGED_CODE();
483 
484  //
485  // If the exception code is an access violation, guard page violation,
486  // or an in-page read error, then return the faulting address. Otherwise.
487  // return a special address value.
488  //
489 
490  *ExceptionAddressConfirmed = FALSE;
491 
492  ExceptionRecord = ExceptionPointers->ExceptionRecord;
493 
494  if ((ExceptionRecord->ExceptionCode == STATUS_ACCESS_VIOLATION) ||
495  (ExceptionRecord->ExceptionCode == STATUS_GUARD_PAGE_VIOLATION) ||
496  (ExceptionRecord->ExceptionCode == STATUS_IN_PAGE_ERROR)) {
497 
498  //
499  // The virtual address which caused the exception is the 2nd
500  // parameter in the exception information array.
501  //
502  // The number of parameters will be zero if an exception handler
503  // above us (like the one in MmProbeAndLockPages) caught the
504  // original exception and subsequently just raised status.
505  // This means the number of bytes copied is zero.
506  //
507 
508  if (ExceptionRecord->NumberParameters > 1) {
509  *ExceptionAddressConfirmed = TRUE;
510  *BadVa = ExceptionRecord->ExceptionInformation[1];
511  }
512  }
513 
514  return EXCEPTION_EXECUTE_HANDLER;
515 }

518 MiDoMappedCopy (
519  IN PEPROCESS FromProcess,
520  IN CONST VOID *FromAddress,
521  IN PEPROCESS ToProcess,
522  OUT PVOID ToAddress,
523  IN SIZE_T BufferSize,
524  IN KPROCESSOR_MODE PreviousMode,
525  OUT PSIZE_T NumberOfBytesRead
526  )
527 
528 /*++
529 
530 Routine Description:
531 
532  This function copies the specified address range from the specified
533  process into the specified address range of the current process.
534 
535 Arguments:
536 
537  FromProcess - Supplies an open handle to a process object.
538 
539  FromAddress - Supplies the base address in the specified process
540  to be read.
541 
542  ToProcess - Supplies an open handle to a process object.
543 
544  ToAddress - Supplies the address of a buffer which receives the
545  contents from the specified process address space.
546 
547  BufferSize - Supplies the requested number of bytes to read from
548  the specified process.
549 
550  PreviousMode - Supplies the previous processor mode.
551 
552  NumberOfBytesRead - Receives the actual number of bytes
553  transferred into the specified buffer.
554 
555 Return Value:
556 
557  NTSTATUS.
558 
559 --*/
560 
561 {
562  KAPC_STATE ApcState;
563  SIZE_T AmountToMove;
564  ULONG_PTR BadVa;
565  LOGICAL Moving;
566  LOGICAL Probing;
567  LOGICAL LockedMdlPages;
568  CONST VOID *InVa;
569  SIZE_T LeftToMove;
570  PSIZE_T MappedAddress;
571  SIZE_T MaximumMoved;
572  PMDL Mdl;
573  PFN_NUMBER MdlHack[(sizeof(MDL)/sizeof(PFN_NUMBER)) + (MAX_LOCK_SIZE >> PAGE_SHIFT) + 1];
574  PVOID OutVa;
575  LOGICAL MappingFailed;
576  LOGICAL ExceptionAddressConfirmed;
577 
578  PAGED_CODE();
579 
580  MappingFailed = FALSE;
581 
582  InVa = FromAddress;
583  OutVa = ToAddress;
584 
585  MaximumMoved = MAX_LOCK_SIZE;
586  if (BufferSize <= MAX_LOCK_SIZE) {
587  MaximumMoved = BufferSize;
588  }
589 
590  Mdl = (PMDL)&MdlHack[0];
591 
592  //
593  // Map the data into the system part of the address space, then copy it.
594  //
595 
596  LeftToMove = BufferSize;
597  AmountToMove = MaximumMoved;
598 
599  Probing = FALSE;
600 
601  //
602  // Initializing BadVa & ExceptionAddressConfirmed is not needed for
603  // correctness but without it the compiler cannot compile this code
604  // W4 to check for use of uninitialized variables.
605  //
606 
607  BadVa = 0;
608  ExceptionAddressConfirmed = FALSE;
609 
610  while (LeftToMove > 0) {
611 
612  if (LeftToMove < AmountToMove) {
613 
614  //
615  // Set to move the remaining bytes.
616  //
617 
618  AmountToMove = LeftToMove;
619  }
620 
621  KeStackAttachProcess (&FromProcess->Pcb, &ApcState);
622 
623  MappedAddress = NULL;
624  LockedMdlPages = FALSE;
625  Moving = FALSE;
626  ASSERT (Probing == FALSE);
627 
628  //
629  // We may be touching a user's memory which could be invalid,
630  // declare an exception handler.
631  //
632 
633  try {
634 
635  //
636  // Probe to make sure that the specified buffer is accessible in
637  // the target process.
638  //
639 
640  if ((InVa == FromAddress) && (PreviousMode != KernelMode)){
641  Probing = TRUE;
642  ProbeForRead (FromAddress, BufferSize, sizeof(CHAR));
643  Probing = FALSE;
644  }
645 
646  //
647  // Initialize MDL for request.
648  //
649 
650  MmInitializeMdl (Mdl, (PVOID)InVa, AmountToMove);
651 
652  MmProbeAndLockPages (Mdl, PreviousMode, IoReadAccess);
653 
654  LockedMdlPages = TRUE;
655 
656  MappedAddress = MmMapLockedPagesSpecifyCache (Mdl,
657  KernelMode,
658  MmCached,
659  NULL,
660  FALSE,
661  HighPagePriority);
662 
663  if (MappedAddress == NULL) {
664  MappingFailed = TRUE;
665  ExRaiseStatus(STATUS_INSUFFICIENT_RESOURCES);
666  }
667 
668  //
669  // Deattach from the FromProcess and attach to the ToProcess.
670  //
671 
672  KeUnstackDetachProcess (&ApcState);
673  KeStackAttachProcess (&ToProcess->Pcb, &ApcState);
674 
675  //
676  // Now operating in the context of the ToProcess.
677  //
678  if ((InVa == FromAddress) && (PreviousMode != KernelMode)){
679  Probing = TRUE;
680  ProbeForWrite (ToAddress, BufferSize, sizeof(CHAR));
681  Probing = FALSE;
682  }
683 
684  Moving = TRUE;
685  RtlCopyMemory (OutVa, MappedAddress, AmountToMove);
686 
687  } except (MiGetExceptionInfo (GetExceptionInformation(),
688  &ExceptionAddressConfirmed,
689  &BadVa)) {
690 
691 
692  //
693  // If an exception occurs during the move operation or probe,
694  // return the exception code as the status value.
695  //
696 
697  KeUnstackDetachProcess (&ApcState);
698 
699  if (MappedAddress != NULL) {
700  MmUnmapLockedPages (MappedAddress, Mdl);
701  }
702  if (LockedMdlPages == TRUE) {
703  MmUnlockPages (Mdl);
704  }
705 
706  if (GetExceptionCode() == STATUS_WORKING_SET_QUOTA) {
707  return STATUS_WORKING_SET_QUOTA;
708  }
709 
710  if ((Probing == TRUE) || (MappingFailed == TRUE)) {
711  return GetExceptionCode();
712 
713  }
714 
715  //
716  // If the failure occurred during the move operation, determine
717  // which move failed, and calculate the number of bytes
718  // actually moved.
719  //
720 
721  *NumberOfBytesRead = BufferSize - LeftToMove;
722 
723  if (Moving == TRUE) {
724  if (ExceptionAddressConfirmed == TRUE) {
725  *NumberOfBytesRead = (SIZE_T)((ULONG_PTR)BadVa - (ULONG_PTR)FromAddress);
726  }
727  }
728 
729  return STATUS_PARTIAL_COPY;
730  }
731 
732  KeUnstackDetachProcess (&ApcState);
733 
734  MmUnmapLockedPages (MappedAddress, Mdl);
735  MmUnlockPages (Mdl);
736 
737  LeftToMove -= AmountToMove;
738  InVa = (PVOID)((ULONG_PTR)InVa + AmountToMove);
739  OutVa = (PVOID)((ULONG_PTR)OutVa + AmountToMove);
740  }
741 
742  //
743  // Set number of bytes moved.
744  //
745 
746  *NumberOfBytesRead = BufferSize;
747  return STATUS_SUCCESS;
748 }

885  // declare an exception handler.
886  //
887 
888  try {
889

891  // Probe to make sure that the specified buffer is accessible in
892  // the target process.
893  //
894
