182 //
183 // VfIrpCallDriverPreprocess is a macro function that may do an alloca as
184 // part of it's operation. As such callers must be careful not to use
185 // variable length arrays in a scope that encompasses
186 // VfIrpCallDriverPreProcess but not VfIrpCallDriverPostProcess.
187 //
188 #define VfIrpCallDriverPreProcess(DeviceObject, IrpPointer, CallStackData, CallerAddress) \
189  if (!VfIrpReserveCallStackData(*(IrpPointer), (CallStackData))) { \
190  *(CallStackData) = alloca(sizeof(IOFCALLDRIVER_STACKDATA)); \
191  VfIrpPrepareAllocaCallStackData(*(CallStackData)); \
192  } \
193  IovpCallDriver1((DeviceObject), (IrpPointer), *(CallStackData), (CallerAddress))
194 
195 #define VfIrpCallDriverPostProcess(DeviceObject, FinalStatus, CallStackData) \
196  IovpCallDriver2(DeviceObject, FinalStatus, CallStackData); \
197  VfIrpReleaseCallStackData(CallStackData)
