118 /*++
119 
120 Routine Description:
121 
122  Initialize the Page Attribute Table (PAT) on all processors. PAT
123  is setup to provide WB, WC, STRONG_UC and WEAK_UC as the memory
124  types such that mm macros for enabling/disabling/querying caching
125  (MI_DISABLE_CACHING, MI_ENABLE_CACHING and MI_IS_CACHING_ENABLED)
126  are unaffected.
127 
128  PAT_Entry PAT Index PCD PWT Memory Type
129  0 0 0 0 WB
130  1 0 0 1 WC *
131  2 0 1 0 WEAK_UC
132  3 0 1 1 STRONG_UC
133  4 1 0 0 WB
134  5 1 0 1 WC *
135  6 1 1 0 WEAK_UC
136  7 1 1 1 STRONG_UC
137 
138  N.B. The caller must have the PAGELK code locked and ensure that the
139  PAT feature is supported.
140 
141  Arguments:
142 
143  None.
144 
145 Return Value:
146 
147  None.
148 
149 --*/
150 {
151  PAT PatAttributes;
152  KIRQL OldIrql;
153  PKPRCB Prcb;
154  NEW_PAT NewPAT;
155 #if !defined(NT_UP)
156  KIRQL NewIrql;
157  KAFFINITY TargetProcessors;
158 #endif
159 
160  ASSERT ((KeFeatureBits & KF_PAT) != 0);
161 
162  //
163  // Initialize the PAT
164  //
165 
166  PatAttributes.hw.Pat[0] = PAT_TYPE_WB;
167  PatAttributes.hw.Pat[1] = PAT_TYPE_USWC;
168  PatAttributes.hw.Pat[2] = PAT_TYPE_WEAK_UC;
169  PatAttributes.hw.Pat[3] = PAT_TYPE_STRONG_UC;
170  PatAttributes.hw.Pat[4] = PAT_TYPE_WB;
171  PatAttributes.hw.Pat[5] = PAT_TYPE_USWC;
172  PatAttributes.hw.Pat[6] = PAT_TYPE_WEAK_UC;
173  PatAttributes.hw.Pat[7] = PAT_TYPE_STRONG_UC;
174 
175  //
176  // Synchronize with other IPI functions which may stall
177  //
178  KeAcquireSpinLock (&KiReverseStallIpiLock, &OldIrql);
179 
180  Prcb = KeGetCurrentPrcb();
181 
182  NewPAT.Attributes = PatAttributes;
183  NewPAT.Synchronize.TargetCount = 0;
184  NewPAT.Synchronize.TargetPhase = &Prcb->ReverseStall;
185  NewPAT.Synchronize.Processor = Prcb->Number;
186 
187 
188 #if !defined(NT_UP)
189 
190  //
191  // Collect all the (other) processors
192  //
193 
194  TargetProcessors = KeActiveProcessors & ~Prcb->SetMember;
195  if (TargetProcessors != 0) {
196 
197  KiIpiSendSynchronousPacket (
198  Prcb,
199  TargetProcessors,
200  KiLoadPATTarget,
201  (PVOID) (&NewPAT),
202  NULL,
203  NULL
204  );
205 
206  //
207  // Wait for all processors to be collected
208  //
209 
210  KiIpiStallOnPacketTargets(TargetProcessors);
211 
212  //
213  // All processors are now waiting. Raise to high level to
214  // ensure this processor doesn't enter the debugger due to
215  // some interrupt service routine.
216  //
217 
218  KeRaiseIrql (HIGH_LEVEL, &NewIrql);
219 
220  //
221  // There's no reason for any debug events now, so signal
222  // the other processors that they can all begin the PAT update
223  //
224 
225  Prcb->ReverseStall += 1;
226  }
227 
228 #endif
229 
230  //
231  // Update PAT
232  //
233 
234  KiLoadPAT(&NewPAT);
235 
236  //
237  // Release lock and lower to initial irql
238  //
239 
240  KeReleaseSpinLock (&KiReverseStallIpiLock, OldIrql);
241  MmEnablePAT();
242  return;
243 }
