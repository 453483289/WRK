49 
50 Arguments:

68 
69  //
70  // calculate the number of tables in the map
71  //
72  Length = Hive->Storage[Stable].Length;
73  MapSlots = Length / HBLOCK_SIZE;
74  if( MapSlots > 0 ) {
75  Tables = (MapSlots-1) / HTABLE_SLOTS;
76  } else {
77  Tables = 0;
78  }
79 
80  if( Hive->Storage[Stable].Map ) {
81  //
82  // iterate through the directory
83  //

85  Tab = Hive->Storage[Stable].Map->Directory[i];
86

88 
89  //
90  // iterate through the slots in the directory
91  //

94  //
95  // BinAddress non-zero means allocated bin
96  //

98  //
99  // a bin is freed if it is a new alloc AND it resides in paged pool
100  //

102  Bin = (PHBIN)HBIN_BASE(Me->BinAddress);

104  }
105 
106  Me->BinAddress = 0;
107  }

130 /*++
131 
132 Routine Description:
133 
134  Initialize a hive.
135 
136  Core HHive fields are always initializeed.
137 
138  File calls WILL be made BEFORE this call returns.
139 
140  Caller is expected to create/open files and store file handles
141  in a way that can be derived from the hive pointer.
142 
143  Three kinds of initialization can be done, selected by OperationType:
144 
145  HINIT_CREATE
146 
147  Create a new hive from scratch. Will have 0 storage.
148  [Used to do things like create HARDWARE hive and for parts
149  of SaveKey and RestoreKey]
150 
151 
152  HINIT_MEMORY_INPLACE
153 
154  Build a hive control structure which allows read only
155  access to a contiguous in-memory image of a hive.
156  No part of the image will be copied, but a map will
157  be made.
158  [Used by osloader.]
159 
160 
161  HINIT_FLAT
162 
163  Support very limited (read-only, no checking code) operation
164  against a hive image.
165 
166 
167  HINIT_MEMORY
168 
169  Create a new hive, using a hive image already in memory,
170  at address supplied by pointer HiveData. The data will
171  be copied. Caller is expected to free HiveData.
172  [Used for SYSTEM hive]
173 
174 
175  HINIT_FILE
176 
177  Create a hive, reading its data from a file. Recovery processing
178  via log file will be done if a log is available. If a log
179  is recovered, flush and clear operation will proceed.
180 
181 
182  HINIT_MAPFILE
183 
184  Create a hive, reading its data from a file. Data reading is
185  done by mapping views of the file in the system cache.
186 
187 
188  NOTE: The HHive is not a completely opaque structure, because it
189  is really only used by a limited set of code. Do not assume
190  that only this routine sets all of these values.
191 
192 
193 Arguments:
194 
195  Hive - supplies a pointer to hive control structure to be initialized
196  to describe this hive.
197 
198  OperationType - specifies whether to create a new hive from scratch,
199  from a memory image, or by reading a file from disk.
200 
201  HiveFlags - HIVE_VOLATILE - Entire hive is to be volatile, regardless
202  of the types of cells allocated
203  HIVE_NO_LAZY_FLUSH - Data in this hive is never written
204  to disk except by an explicit FlushKey
205 
206  FileType - HFILE_TYPE_*, HFILE_TYPE_LOG set up for logging support respectively.
207 
208  HiveData - if present, supplies a pointer to an in memory image of
209  from which to init the hive. Only useful when OperationType
210  is set to HINIT_MEMORY.
211 
212  AllocateRoutine - supplies a pointer to routine called to allocate
213  memory. WILL be called before this routine returns.
214 
215  FreeRoutine - supplies a pointer to routine called to free memory.
216  CAN be called before this routine returns.
217 
218  FileSetSizeRoutine - supplies a pointer to a routine used to set the
219  size of a file. CAN be called before this
220  routine returns.
221 
222  FileWriteRoutine - supplies a pointer to routine called to write memory
223  to a file.
224 
225  FileReadRoutine - supplies a pointer to routine called to read from
226  a file into memory. CAN be called before this
227  routine returns.
228 
229  FileFlushRoutine - supplies a pointer to routine called to flush a file.
230 
231  Cluster - clustering factor in HSECTOR_SIZE units. (i.e. Size of
232  physical sector in media / HSECTOR_SIZE. 1 for 512 byte
233  physical sectors (or smaller), 2 for 1024, 4 for 2048, etc.
234  (Numbers greater than 8 won't work.)

236  FileName - some path like "...\system32\config\system", last
237  32 or so characters will be copied into baseblock
