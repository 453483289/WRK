386 NTSTATUS
387 ExpAllocateUuids (
388  OUT PLARGE_INTEGER Time,
389  OUT PULONG Range,
390  OUT PULONG Sequence
391  )
392 
393 /*++
394 
395 Routine Description:
396 
397  Allocates a sequence number and a range of times for a set of UUIDs.
398  The caller can use this together with the network address to
399  generate complete UUIDs.
400 
401  This routine assumes that the current thread has exclusive access
402  to the ExpUuid* values.
403 
404 Arguments:
405 
406  Time - Supplies the address of a variable that will receive the
407  start time (SYSTEMTIME format) of the range of time reserved.
408 
409  Range - Supplies the address of a variable that will receive the
410  number of ticks (100ns) reserved after the value in Time.
411  The range reserved is *Time to (*Time+*Range-1).
412 
413  Sequence - Supplies the address of a variable that will receive
414  the time sequence number. This value is used with the associated
415  range of time to prevent problems with clocks going backwards.
416 
417 Return Value:
418 
419  STATUS_SUCCESS is returned if the service is successfully executed.
420 
421  STATUS_RETRY is returned if we're unable to reserve a range of
422  UUIDs. This will occur if system clock hasn't advanced
423  and the allocator is out of cached values.
424 
425  STATUS_UNSUCCESSFUL is returned if some other service reports
426  an error, most likly the registry.
427 
428 --*/
429 
430 {
431 
432  NTSTATUS Status;
433  LARGE_INTEGER CurrentTime;
434  LARGE_INTEGER AvailableTime;
435 
436  PAGED_CODE();
437 
438  //
439  // Make sure we have a valid sequence number. If not, make one up.
440  //
441 
442  if (ExpUuidSequenceNumberValid == FALSE) {
443 
444  Status = ExpUuidLoadSequenceNumber(&ExpUuidSequenceNumber);
445 
446  if (!NT_SUCCESS(Status)) {
447  // Unable read the sequence number, this means we should make one up.
448 
449  LARGE_INTEGER PerfCounter;
450  LARGE_INTEGER PerfFrequency;
451 
452  // This should only happen when we're called
453  // for the first time on a given machine. (machine, not boot)
454 
455  KdPrintEx((DPFLTR_SYSTEM_ID, DPFLTR_WARNING_LEVEL, "Uuid: Generating first sequence number.\n"));
456 
457  PerfCounter = KeQueryPerformanceCounter(&PerfFrequency);
458 
459  ExpUuidSequenceNumber ^= (ULONG)((ULONG_PTR)&Status) ^ PerfCounter.LowPart ^
460  PerfCounter.HighPart ^ (ULONG)((ULONG_PTR)Sequence);
461  }
462  else {
463  // We increment the sequence number on every boot.
464  ExpUuidSequenceNumber++;
465  }
466 
467  ExpUuidSequenceNumberValid = TRUE;
468  ExpUuidSequenceNumberNotSaved = TRUE;
469 
470  }
471 
472  //
473  // Get the current time, usually we will have plenty of available
474  // to give the caller. But we may need to deal with time going
475  // backwards and really fast machines.
476  //
477 
478  KeQuerySystemTime(&CurrentTime);
479 
480  AvailableTime.QuadPart = CurrentTime.QuadPart - ExpUuidLastTimeAllocated.QuadPart;
481 
482  if (AvailableTime.QuadPart < 0) {
483 
484  // Time has been set time backwards. This means that we must make sure
485  // that somebody increments the sequence number and saves the new
486  // sequence number in the registry.
487 
488  ExpUuidSequenceNumberNotSaved = TRUE;
489  ExpUuidSequenceNumber++;
490 
491  // The sequence number has been changed, so it's now okay to set time
492  // backwards. Since time is going backwards anyway, it's okay to set
493  // it back an extra millisecond or two.
494 
495  ExpUuidLastTimeAllocated.QuadPart = CurrentTime.QuadPart - 20000;
496  AvailableTime.QuadPart = 20000;
497  }
498 
499  if (AvailableTime.QuadPart == 0) {
500  // System time hasn't moved. The caller should yield the CPU and retry.
501  return(STATUS_RETRY);
502  }
503 
504  //
505  // Common case, time has moved forward.
506  //
507 
508  if (AvailableTime.QuadPart > 10*1000*1000) {
509  // We never want to give out really old (> 1 second) Uuids.
510  AvailableTime.QuadPart = 10*1000*1000;
511  }
512 
513  if (AvailableTime.QuadPart > 10*1000) {
514  // We've got over a millisecond to give out. We'll save some time for
515  // another caller so that we can avoid returning STATUS_RETRY very often.
516  *Range = 10*1000;
517  AvailableTime.QuadPart -= 10*1000;
518  }
519  else {
520  // Not much time avaiable, give it all away.
521  *Range = (ULONG)AvailableTime.QuadPart;
522  AvailableTime.QuadPart = 0;
523  }
524 
525  Time->QuadPart = CurrentTime.QuadPart - (*Range + AvailableTime.QuadPart);
526 
527  ExpUuidLastTimeAllocated.QuadPart = Time->QuadPart + *Range;
528 
529  // Last time allocated is just after the range we hand back to the caller
530  // this may be almost a second behind the true system time.
531 
532  *Sequence = ExpUuidSequenceNumber;
533 
534 
535  return(STATUS_SUCCESS);
536 }

540 NTSTATUS
541 NtSetUuidSeed (
542  __in PCHAR Seed
543  )
544 
545 /*++
546 
547 Routine Description:
548 
549  This routine is used to set the seed used for UUID generation. The seed
550  will be set by RPCSS at startup and each time a card is replaced.
551 
552 Arguments:
553 
554  Seed - Pointer to a six byte buffer
555 
556 Return Value:
557 
558  STATUS_SUCCESS is returned if the service is successfully executed.
559 
560  STATUS_ACCESS_DENIED If caller doesn't have the permissions to make this call.
561  You need to be logged on as Local System in order to call this API.
562 
563  STATUS_ACCESS_VIOLATION is returned if the Seed could not be read.
564 
565 --*/
566 
567 {
568 
569  NTSTATUS Status;
570  LUID AuthenticationId;
571  SECURITY_SUBJECT_CONTEXT SubjectContext;
572  LUID SystemLuid = SYSTEM_LUID;
573  BOOLEAN CapturedSubjectContext = FALSE;
574 
575  PAGED_CODE();
576 
577  ASSERT(KeGetPreviousMode() != KernelMode);
578 
579  try {
580  //
581  // Check if the caller has the appropriate permission
582  //
583  SeCaptureSubjectContext(&SubjectContext);
584  CapturedSubjectContext = TRUE;
585 
586  Status = SeQueryAuthenticationIdToken(
587  SeQuerySubjectContextToken(&SubjectContext),
588  &AuthenticationId);
589  if (!NT_SUCCESS(Status)) {
590  ExRaiseStatus(Status);
591  }
592 
593  if (RtlCompareMemory(&AuthenticationId, &SystemLuid, sizeof(LUID)) != sizeof(LUID)) {
594  ExRaiseStatus(STATUS_ACCESS_DENIED);
595  }
596 
597  //
598  // Store the UUID seed
599  //
600  ProbeForReadSmallStructure(Seed, SEED_SIZE, sizeof(CHAR));
601  RtlCopyMemory(&ExpUuidCachedValues.NodeId[0], Seed, SEED_SIZE);
602 
603  if ((Seed[0] & 0x80) == 0)
604  {
605  // If the high bit is not set the NodeId is a valid IEEE 802
606  // address and should be globally unique.
607  ExpUuidCacheValid = CACHE_VALID;
608  }
609  else
610  {
611  ExpUuidCacheValid = CACHE_LOCAL_ONLY;
612  }
613 
614  Status = STATUS_SUCCESS;
615  }
616  except (EXCEPTION_EXECUTE_HANDLER) {
617  Status = GetExceptionCode();
618  }
619 
620  if (CapturedSubjectContext) {
621  SeReleaseSubjectContext( &SubjectContext );
622  }
623 
624  return Status;
625 }

627 NTSTATUS
628 NtAllocateUuids (
629  __out PULARGE_INTEGER Time,
630  __out PULONG Range,
631  __out PULONG Sequence,
632  __out PCHAR Seed
633  )
634 
635 /*++
636 
637 Routine Description:
638 
639  This function reserves a range of time for the caller(s) to use for
640  handing out Uuids. As far a possible the same range of time and
641  sequence number will never be given out.
642 
643  (It's possible to reboot 2^14-1 times and set the clock backwards and then
644  call this allocator and get a duplicate. Since only the low 14bits of the
645  sequence number are used in a real uuid.)
646 
647 Arguments:
648 
649  Time - Supplies the address of a variable that will receive the
650  start time (SYSTEMTIME format) of the range of time reserved.
651 
652  Range - Supplies the address of a variable that will receive the
653  number of ticks (100ns) reserved after the value in Time.
654  The range reserved is *Time to (*Time + *Range - 1).
655 
656  Sequence - Supplies the address of a variable that will receive
657  the time sequence number. This value is used with the associated
658  range of time to prevent problems with clocks going backwards.
659 
660  Seed - Pointer to a 6 byte buffer. The current seed is written into this buffer.
661 
662 Return Value:
663 
664  STATUS_SUCCESS is returned if the service is successfully executed.
665 
666  STATUS_RETRY is returned if we're unable to reserve a range of
667  UUIDs. This may (?) occur if system clock hasn't advanced
668  and the allocator is out of cached values.
669 
670  STATUS_ACCESS_VIOLATION is returned if the output parameter for the
671  UUID cannot be written.
672 
673  STATUS_UNSUCCESSFUL is returned if some other service reports
674  an error, most likly the registry.
675 
676 --*/
677 
678 {
679 
680  KPROCESSOR_MODE PreviousMode;
681  NTSTATUS Status;
682 
683  LARGE_INTEGER OutputTime;
684  ULONG OutputRange;
685  ULONG OutputSequence;
686  PKTHREAD CurrentThread;
687 
688  PAGED_CODE();
689 
690  //
691  // Establish an exception handler and attempt to write the output
692  // arguments. If the write attempt fails, then return
693  // the exception code as the service status. Otherwise return success
694  // as the service status.
695  //
696 
697  try {
698 
699  //
700  // Get previous processor mode and probe arguments if necessary.
701  //
702 
703  PreviousMode = KeGetPreviousMode();
704  if (PreviousMode != KernelMode) {
705  ProbeForWriteSmallStructure((PVOID)Time, sizeof(LARGE_INTEGER), sizeof(ULONG));
706  ProbeForWriteUlongAligned32(Range);
707  ProbeForWriteUlongAligned32(Sequence);
708  ProbeForWriteSmallStructure((PVOID)Seed, SEED_SIZE, sizeof(CHAR));
709  }
710  } except (ExSystemExceptionFilter()) {
711  return GetExceptionCode();
712  }
713 
714  // Take the lock, because we're about to update the UUID cache.
715  CurrentThread = KeGetCurrentThread ();
716  KeEnterCriticalRegionThread(CurrentThread);
717  ExAcquireFastMutexUnsafe(&ExpUuidLock);
718 
719  // Get the sequence number and a range of times that can
720  // be used in UUID-generation.
721 
722  Status = ExpAllocateUuids( &OutputTime, &OutputRange, &OutputSequence );
723 
724  if( !NT_SUCCESS(Status) ) {
725  ExReleaseFastMutexUnsafe(&ExpUuidLock);
726  KeLeaveCriticalRegionThread(CurrentThread);
727  return( Status );
728  }
729 
730  // If necessary, save the sequence number. If there's an error,
731  // we'll just leave it marked as dirty, and retry on some future call.
732 
733  ExpUuidSaveSequenceNumberIf();
734 
735  // Release the lock
736  ExReleaseFastMutexUnsafe(&ExpUuidLock);
737  KeLeaveCriticalRegionThread(CurrentThread);
738 
739  //
740  // Attempt to store the result of this call into the output parameters.
741  // This is done within an exception handler in case output parameters
742  // are now invalid.
743  //
744 
745  try {
746  Time->QuadPart = OutputTime.QuadPart;
747  *Range = OutputRange;
748  *Sequence = OutputSequence;
749  RtlCopyMemory((PVOID) Seed, &ExpUuidCachedValues.NodeId[0], SEED_SIZE);
750  } except (ExSystemExceptionFilter()) {
751  return GetExceptionCode();
752  }
753 
754  return(STATUS_SUCCESS);
755 }

757 NTSTATUS
758 ExpUuidGetValues (
759  OUT UUID_CACHED_VALUES_STRUCT *Values
760  )
761 
762 /*++
763 
764 Routine Description:
765 
766  This routine allocates a block of UUIDs and stores them in
767  the caller-provided cached-values structure.
768 
769  This routine assumes that the current thread has exclusive
770  access to the ExpUuid* values.
771 
772  Note that the Time value in this cache is different than the
773  Time value returned by NtAllocateUuids (and ExpAllocateUuids).
774  As a result, the cache must be interpreted differently in
775  order to determine the valid range. The valid range from
776  these two routines is:
777 
778  NtAllocateUuids: [ Time, Time+Range )
779  ExpUuidGetValues: ( Values.Time-Values.Range, Values.Time ]
780 
781 Arguments:
782 
783  Values - Set to contain everything needed to allocate a block of uuids.
784 
785 Return Value:
786 
787  STATUS_SUCCESS is returned if the service is successfully executed.
788 
789  STATUS_RETRY is returned if we're unable to reserve a range of
790  UUIDs. This will occur if system clock hasn't advanced
791  and the allocator is out of cached values.
792 
793  STATUS_NO_MEMORY is returned if we're unable to reserve a range
794  of UUIDs, for some reason other than the clock not advancing.
795 
796 --*/
797 
798 {
799 
800  NTSTATUS Status;
801  LARGE_INTEGER Time;
802  ULONG Range;
803  ULONG Sequence;
804 
805  PAGED_CODE();
806 
807  // Allocate a range of times for use in UUIDs.
808 
809  Status = ExpAllocateUuids(&Time, &Range, &Sequence);
810 
811  if (STATUS_RETRY == Status) {
812  return(Status);
813  }
814 
815  else if (!NT_SUCCESS(Status)) {
816  return(STATUS_NO_MEMORY);
817  }
818 
819  // ExpAllocateUuids keeps time in SYSTEM_TIME format which is 100ns ticks since
820  // Jan 1, 1601. UUIDs use time in 100ns ticks since Oct 15, 1582.
821 
822  // 17 Days in Oct + 30 (Nov) + 31 (Dec) + 18 years and 5 leap days.
823 
824  Time.QuadPart += (ULONGLONG) (1000*1000*10) // seconds
825  * (ULONGLONG) (60 * 60 * 24) // days
826  * (ULONGLONG) (17+30+31+365*18+5); // # of days
827 
828  ASSERT(Range);
829 
830  Values->ClockSeqHiAndReserved =
831  UUID_RESERVED | (((UCHAR) (Sequence >> 8))
832  & (UCHAR) UUID_CLOCK_SEQ_HI_MASK);
833 
834  Values->ClockSeqLow = (UCHAR) (Sequence & 0x00FF);
835 
836

849 ExUuidCreate (
850  __out UUID *Uuid
851  )
852 
853 /*++
854 
855 Routine Description:
856 
857  This routine creates a DCE UUID and returns it in the caller's
858  buffer.
859 
860 Arguments:
861 
862  Uuid - will receive the UUID.
863 
864 Return Value:
865 
866  STATUS_SUCCESS is returned if the service is successfully executed.
867 
868  STATUS_RETRY is returned if we're unable to reserve a range of
869  UUIDs. This will occur if system clock hasn't advanced
870  and the allocator is out of cached values.
871 
872 --*/
873 
874 {
875 
876  NTSTATUS Status = STATUS_SUCCESS;
877 
878  UUID_GENERATE *UuidGen = (UUID_GENERATE *) Uuid;
879  ULONGLONG Time;
880  LONG Delta;
881  PKTHREAD CurrentThread;
882 
883  PAGED_CODE();
884 
885  //
886  // Get a value from the cache. If the cache is empty, we'll fill
887  // it and retry. The first time cache will be empty.
888  //
889 
890  CurrentThread = KeGetCurrentThread ();
891  for(;;) {
892 
893  // Get the highest value in the cache (though it may not
894  // be available).
895  Time = ExpUuidCachedValues.Time;
896 
897  // Copy the static info into the UUID. We can't do this later
898  // because the clock sequence could be updated by another thread.
899 
900  *(PULONG)&UuidGen->ClockSeqHiAndReserved =
901  *(PULONG)&ExpUuidCachedValues.ClockSeqHiAndReserved;
902  *(PULONG)&UuidGen->NodeId[2] =
903  *(PULONG)&ExpUuidCachedValues.NodeId[2];
904 
905  // See what we need to subtract from Time to get a valid GUID.
906  Delta = InterlockedDecrement(&ExpUuidCachedValues.AllocatedCount);
907 
908  if (Time != ExpUuidCachedValues.Time) {
909 
910  // If our captured time doesn't match the cache then another
911  // thread already took the lock and updated the cache. We'll
912  // just loop and try again.
913  continue;
914  }
915 
916  // If the cache hadn't already run dry, we can break out of this retry
917  // loop.
918  if (Delta >= 0) {
919  break;
920  }
921 
922  //
923  // Allocate a new block of Uuids.
924  //
925 
926  // Take the cache lock
927  KeEnterCriticalRegionThread(CurrentThread);
928  ExAcquireFastMutexUnsafe(&ExpUuidLock);
929 
930  // If the cache has already been updated, try again.
931  if (Time != ExpUuidCachedValues.Time) {
932  // Release the lock
933  ExReleaseFastMutexUnsafe(&ExpUuidLock);
934  KeLeaveCriticalRegionThread(CurrentThread);
935  continue;
936  }
937 
938  // Update the cache.
939  Status = ExpUuidGetValues( &ExpUuidCachedValues );
940 
941  if (Status != STATUS_SUCCESS) {
942  // Release the lock
943  ExReleaseFastMutexUnsafe(&ExpUuidLock);
944  KeLeaveCriticalRegionThread(CurrentThread);
945  return(Status);
946  }
947 
948  // The sequence number may have been dirtied, see if it needs
949  // to be saved. If there's an error, we'll ignore it and
950  // retry on a future call.
951 
952  ExpUuidSaveSequenceNumberIf();
953 
954  // Release the lock
955  ExReleaseFastMutexUnsafe(&ExpUuidLock);
956  KeLeaveCriticalRegionThread(CurrentThread);
957 
958  // Loop
959  }
960 
961  // Adjust the time to that of the next available UUID.
962  Time -= Delta;
963 
964  // Finish filling in the UUID.
965 
966  UuidGen->TimeLow = (ULONG) Time;
967  UuidGen->TimeMid = (USHORT) (Time >> 32);
968  UuidGen->TimeHiAndVersion = (USHORT)
969  (( (USHORT)(Time >> (32+16))
970  & UUID_TIME_HIGH_MASK) | UUID_VERSION);
971 
972  ASSERT(Status == STATUS_SUCCESS);
973 
974  if (ExpUuidCacheValid == CACHE_LOCAL_ONLY) {
975  Status = RPC_NT_UUID_LOCAL_ONLY;
976  }
977 
978  return(Status);
979 }
