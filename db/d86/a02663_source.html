80 VOID
81 RtlpCopyProcString(
82  IN OUT PWSTR *pDst,
83  OUT PUNICODE_STRING DestString,
84  IN PUNICODE_STRING SourceString,
85  IN ULONG DstAlloc OPTIONAL
86  )
87 {
88  if (!ARGUMENT_PRESENT( DstAlloc )) {
89  DstAlloc = SourceString->MaximumLength;
90  }
91 
92  ASSERT((SourceString->Length == 0) || (SourceString->Buffer != NULL));
93 
94  if (SourceString->Buffer != NULL && SourceString->Length != 0) {
95  RtlCopyMemory (*pDst,
96  SourceString->Buffer,
97  SourceString->Length);
98  }
99 
100  DestString->Buffer = *pDst;
101  DestString->Length = SourceString->Length;
102  DestString->MaximumLength = (USHORT)DstAlloc;
103 
104  if (DestString->Length < DestString->MaximumLength) {
105  RtlZeroMemory (((PUCHAR)DestString->Buffer) + DestString->Length, DestString->MaximumLength - DestString->Length);
106  }
107 
108 
109  *pDst = (PWSTR)((PCHAR)(*pDst) + ROUND_UP( DstAlloc, sizeof( ULONG ) ) );
110  return;
111 }

113 NTSTATUS
114 RtlCreateProcessParameters(
115  OUT PRTL_USER_PROCESS_PARAMETERS *pProcessParameters,
116  IN PUNICODE_STRING ImagePathName,
117  IN PUNICODE_STRING DllPath OPTIONAL,
118  IN PUNICODE_STRING CurrentDirectory OPTIONAL,
119  IN PUNICODE_STRING CommandLine OPTIONAL,
120  IN PVOID Environment OPTIONAL,
121  IN PUNICODE_STRING WindowTitle OPTIONAL,
122  IN PUNICODE_STRING DesktopInfo OPTIONAL,
123  IN PUNICODE_STRING ShellInfo OPTIONAL,
124  IN PUNICODE_STRING RuntimeData OPTIONAL
125  )
126 
127 /*++
128 
129 Routine Description:
130 
131  This function formats NT style RTL_USER_PROCESS_PARAMETERS
132  record. The record is self-contained in a single block of memory
133  allocated by this function. The allocation method is opaque and
134  thus the record must be freed by calling the
135  RtlDestroyProcessParameters function.
136 
137  The process parameters record is created in a de-normalized form,
138  thus making it suitable for passing to the RtlCreateUserProcess
139  function. It is expected that the caller will fill in additional
140  fields in the process parameters record after this function returns,
141  but prior to calling RtlCreateUserProcess.
142 
143 Arguments:
144 
145  pProcessParameters - Pointer to a variable that will receive the address
146  of the process parameter structure created by this routine. The
147  memory for the structure is allocated in an opaque manner and must
148  be freed by calling RtlDestroyProcessParameters.
149 
150  ImagePathName - Required parameter that is the fully qualified NT
151  path name of the image file that will be used to create the process
152  that will received these parameters.
153 
154  DllPath - An optional parameter that is an NT String variable pointing
155  to the search path the NT Loader is to use in the target process
156  when searching for Dll modules. If not specified, then the Dll
157  search path is filled in from the current process's Dll search
158  path.
159 
160  CurrentDirectory - An optional parameter that is an NT String variable
161  pointing to the default directory string for the target process.
162  If not specified, then the current directory string is filled in
163  from the current process's current directory string.
164 
165  CommandLine - An optional parameter that is an NT String variable that
166  will be passed to the target process as its command line. If not
167  specified, then the command line passed to the target process will
168  be a null string.
169 
170  Environment - An optional parameter that is an opaque pointer to an
171  environment variable block of the type created by
172  RtlCreateEnvironment routine. If not specified, then the target
173  process will receive a copy of the calling process's environment
174  variable block.
175 
176  WindowTitle - An optional parameter that is an NT String variable that
177  points to the title string the target process is to use for its
178  main window. If not specified, then a null string will be passed
179  to the target process as its default window title.
180 
181  DesktopInfo - An optional parameter that is an NT String variable that
182  contains uninterpreted data that is passed as is to the target
183  process. If not specified, the target process will receive a
184  pointer to an empty string.
185 
186  ShellInfo - An optional parameter that is an NT String variable that
187  contains uninterpreted data that is passed as is to the target
188  process. If not specified, the target process will receive a
189  pointer to an empty string.
190 
191  RuntimeData - An optional parameter that is an NT String variable that
192  contains uninterpreted data that is passed as is to the target
193  process. If not specified, the target process will receive a
194  pointer to an empty string.
195 
196 Return Value:
197 
198  STATUS_SUCCESS - The process parameters is De-Normalized and
199  contains entries for each of the specified argument and variable
200  strings.
201 
202  STATUS_BUFFER_TOO_SMALL - The specified process parameters buffer is
203  too small to contain the argument and environment strings. The value
204  of ProcessParameters->Length is modified to contain the buffer
205  size needed to contain the argument and variable strings.
206 
207 --*/
208 
209 {
210  PRTL_USER_PROCESS_PARAMETERS p;
211  NTSTATUS Status;
212  ULONG ByteCount;
213  PWSTR pDst;
214  PPEB Peb;
215  PRTL_USER_PROCESS_PARAMETERS ProcessParameters;
216  HANDLE CurDirHandle;
217  BOOLEAN PebLockAcquired = FALSE;
218 
219  //
220  // Acquire the Peb Lock for the duration while we copy information out
221  // of it.
222  //
223 
224  Peb = NtCurrentPeb();
225  ProcessParameters = Peb->ProcessParameters;
226 
227  Status = STATUS_SUCCESS;
228  p = NULL;
229  CurDirHandle = NULL;
230  try {
231  //
232  // Validate input parameters
233  //
234 
235 #define VALIDATE_STRING_PARAMETER(_x) \
236  do { \
237  ASSERT(ARGUMENT_PRESENT((_x))); \
238  if (!ARGUMENT_PRESENT((_x))) { \
239  Status = STATUS_INVALID_PARAMETER; \
240  leave; \
241  } \
242  if (ARGUMENT_PRESENT((_x))) { \
243  ASSERT((_x)->MaximumLength >= (_x)->Length); \
244  ASSERT(((_x)->Length == 0) || ((_x)->Buffer != NULL)); \
245  if (((_x)->MaximumLength < (_x)->Length) || \
246  (((_x)->Length != 0) && ((_x)->Buffer == NULL))) { \
247  Status = STATUS_INVALID_PARAMETER; \
248  leave; \
249  } \
250  } \
251  } while (0)
252 
253 #define VALIDATE_OPTIONAL_STRING_PARAMETER(_x) \
254  do { \
255  if (ARGUMENT_PRESENT((_x))) { \
256  ASSERT((_x)->MaximumLength >= (_x)->Length); \
257  ASSERT(((_x)->Length == 0) || ((_x)->Buffer != NULL)); \
258  if (((_x)->MaximumLength < (_x)->Length) || \

267  VALIDATE_OPTIONAL_STRING_PARAMETER (DllPath);
268  VALIDATE_OPTIONAL_STRING_PARAMETER (CurrentDirectory);
269  VALIDATE_OPTIONAL_STRING_PARAMETER (CommandLine);

301  // will be aligned on a ULONG byte boundary.
302  // We do the ones we can outside of the peb lock.
303  //

311  ByteCount += ROUND_UP (ShellInfo->MaximumLength, sizeof( ULONG ) );
312  ByteCount += ROUND_UP (RuntimeData->MaximumLength, sizeof( ULONG ) );
313 
314  PebLockAcquired = TRUE;
315  RtlAcquirePebLock ();
316 
317  //
318  // For optional pointer parameters, default them to point to their
319  // corresponding field in the current process's process parameter
320  // structure or to a null string.
321  //
322 
323  if (!ARGUMENT_PRESENT (DllPath)) {
324  DllPath = &ProcessParameters->DllPath;
325  }
326 
327  if (!ARGUMENT_PRESENT (CurrentDirectory)) {
328 
329  if (ProcessParameters->CurrentDirectory.Handle) {
330  CurDirHandle = (HANDLE)((ULONG_PTR)ProcessParameters->CurrentDirectory.Handle & ~OBJ_HANDLE_TAGBITS);
331  CurDirHandle = (HANDLE)((ULONG_PTR)CurDirHandle | RTL_USER_PROC_CURDIR_INHERIT);
332  }
333  CurrentDirectory = &ProcessParameters->CurrentDirectory.DosPath;
334  } else {
335  ASSERT(CurrentDirectory->MaximumLength >= CurrentDirectory->Length);
336  ASSERT((CurrentDirectory->Length == 0) || (CurrentDirectory->Buffer != NULL));
337 
338  if (ProcessParameters->CurrentDirectory.Handle) {
339  CurDirHandle = (HANDLE)((ULONG_PTR)ProcessParameters->CurrentDirectory.Handle & ~OBJ_HANDLE_TAGBITS);
340  CurDirHandle = (HANDLE)((ULONG_PTR)CurDirHandle | RTL_USER_PROC_CURDIR_CLOSE);
341  }
342  }
343 
344 
345  if (!ARGUMENT_PRESENT (Environment)) {
346  Environment = ProcessParameters->Environment;
347  }
348 
349  ByteCount += ROUND_UP (DllPath->MaximumLength, sizeof( ULONG ) );
350 
351  //
352  // Allocate memory for the process parameter record.
353  //
354  p = RtlAllocateHeap (RtlProcessHeap (), 0, ByteCount);
355 
356  if (p == NULL) {
357  Status = STATUS_INSUFFICIENT_RESOURCES;
358  __leave;
359  }
360 
361  RtlZeroMemory (p, sizeof (*p));
362 
363  p->MaximumLength = ByteCount;
364  p->Length = ByteCount;
365  p->Flags = RTL_USER_PROC_PARAMS_NORMALIZED;
366  p->DebugFlags = 0;
367  p->Environment = Environment;
368  p->CurrentDirectory.Handle = CurDirHandle;
369 
370  //
371  // Inherits ^C inhibit information
372  //
373 
374  p->ConsoleFlags = ProcessParameters->ConsoleFlags;
375 
376  pDst = (PWSTR)(p + 1);
377  RtlpCopyProcString (&pDst,
378  &p->CurrentDirectory.DosPath,
379  CurrentDirectory,
380  DOS_MAX_PATH_LENGTH*2);
381 
382  RtlpCopyProcString (&pDst, &p->DllPath, DllPath, 0);
383  RtlpCopyProcString (&pDst, &p->ImagePathName, ImagePathName, ImagePathName->Length + sizeof (UNICODE_NULL));
384  if (CommandLine->Length == CommandLine->MaximumLength) {
385  RtlpCopyProcString (&pDst, &p->CommandLine, CommandLine, 0);
386  } else {
387  RtlpCopyProcString (&pDst, &p->CommandLine, CommandLine, CommandLine->Length + sizeof (UNICODE_NULL));
388  }
389 
390  RtlpCopyProcString (&pDst, &p->WindowTitle, WindowTitle, 0);
391  RtlpCopyProcString (&pDst, &p->DesktopInfo, DesktopInfo, 0);
392  RtlpCopyProcString (&pDst, &p->ShellInfo, ShellInfo, 0);
393  if (RuntimeData->Length != 0) {
394  RtlpCopyProcString (&pDst, &p->RuntimeData, RuntimeData, 0);
395  }
396  *pProcessParameters = RtlDeNormalizeProcessParams (p);
397  p = NULL;
398  } finally {
399  if (PebLockAcquired) {
400  RtlReleasePebLock();
401  }
402 
403  if (AbnormalTermination ()) {
404  Status = STATUS_ACCESS_VIOLATION;
405  }
406 
407  if (p != NULL) {
408  RtlDestroyProcessParameters (p);
409  }
410 
411  }
412 
413  return Status;
414 }

581 NTSTATUS
582 RtlpCreateStack(
583  IN HANDLE Process,
584  IN SIZE_T MaximumStackSize OPTIONAL,
585  IN SIZE_T CommittedStackSize OPTIONAL,
586  IN ULONG ZeroBits OPTIONAL,
587  OUT PINITIAL_TEB InitialTeb
588  )
589 {
590  NTSTATUS Status;
591  PCH Stack;
592  SYSTEM_BASIC_INFORMATION SysInfo;
593  BOOLEAN GuardPage;
594  SIZE_T RegionSize;
595  ULONG OldProtect;
596 
597  Status = ZwQuerySystemInformation( SystemBasicInformation,
598  (PVOID)&SysInfo,
599  sizeof( SysInfo ),
600  NULL
601  );
602  if ( !NT_SUCCESS( Status ) ) {
603  return( Status );
604  }
605 
606  //
607  // if stack is in the current process, then default to
608  // the parameters from the image
609  //
610 
611  if ( Process == NtCurrentProcess() ) {
612  PPEB Peb;
613  PIMAGE_NT_HEADERS NtHeaders;
614 
615 
616  Peb = NtCurrentPeb();
617  NtHeaders = RtlImageNtHeader(Peb->ImageBaseAddress);
618 
619  if (!NtHeaders) {
620  return STATUS_INVALID_IMAGE_FORMAT;
621  }
622 
623 
624  if (!MaximumStackSize) {
625  MaximumStackSize = NtHeaders->OptionalHeader.SizeOfStackReserve;
626  }
627 
628  if (!CommittedStackSize) {
629  CommittedStackSize = NtHeaders->OptionalHeader.SizeOfStackCommit;
630  }
631 
632  }
633  else {
634 
635  if (!CommittedStackSize) {
636  CommittedStackSize = SysInfo.PageSize;
637  }
638 
639  if (!MaximumStackSize) {
640  MaximumStackSize = SysInfo.AllocationGranularity;
641  }
642 
643  }
644 
645  //
646  // Enforce a minimal stack commit if there is a PEB setting
647  // for this.
648  //
649 
650  if ( CommittedStackSize >= MaximumStackSize ) {
651  MaximumStackSize = ROUND_UP(CommittedStackSize, (1024*1024));
652  }
653 
654 
655  CommittedStackSize = ROUND_UP( CommittedStackSize, SysInfo.PageSize );
656  MaximumStackSize = ROUND_UP( MaximumStackSize,
657  SysInfo.AllocationGranularity
658  );
659 
660  Stack = NULL;
661 
662 
663  Status = ZwAllocateVirtualMemory( Process,
664  (PVOID *)&Stack,
665  ZeroBits,
666  &MaximumStackSize,
667  MEM_RESERVE,
668  PAGE_READWRITE
669  );
670 
671  if ( !NT_SUCCESS( Status ) ) {
672 #if DBG
673  DbgPrint( "NTRTL: RtlpCreateStack( %lx ) failed. Stack Reservation Status == %X\n",
674  Process,
675  Status
676  );
677 #endif // DBG
678  return( Status );
679  }
680 
681  InitialTeb->OldInitialTeb.OldStackBase = NULL;
682  InitialTeb->OldInitialTeb.OldStackLimit = NULL;
683  InitialTeb->StackAllocationBase = Stack;
684  InitialTeb->StackBase = Stack + MaximumStackSize;
685 
686  Stack += MaximumStackSize - CommittedStackSize;
687  if (MaximumStackSize > CommittedStackSize) {
688  Stack -= SysInfo.PageSize;
689  CommittedStackSize += SysInfo.PageSize;
690  GuardPage = TRUE;
691  }
692  else {
693  GuardPage = FALSE;
694  }
695  Status = ZwAllocateVirtualMemory( Process,
696  (PVOID *)&Stack,
697  0,
698  &CommittedStackSize,
699  MEM_COMMIT,
700  PAGE_READWRITE
701  );
702  InitialTeb->StackLimit = Stack;
703 
704  if ( !NT_SUCCESS( Status ) ) {
705 #if DBG
706  DbgPrint( "NTRTL: RtlpCreateStack( %lx ) failed. Stack Commit Status == %X\n",
707  Process,
708  Status
709  );
710 #endif // DBG
711  return( Status );
712  }
713 
714  //
715  // if we have space, create a guard page.
716  //
717 
718  if (GuardPage) {
719  RegionSize = SysInfo.PageSize;
720  Status = ZwProtectVirtualMemory( Process,
721  (PVOID *)&Stack,
722  &RegionSize,
723  PAGE_GUARD | PAGE_READWRITE,
724  &OldProtect);
725 
726 
727  if ( !NT_SUCCESS( Status ) ) {
728 #if DBG
729  DbgPrint( "NTRTL: RtlpCreateStack( %lx ) failed. Guard Page Creation Status == %X\n",
730  Process,
731  Status
732  );
733 #endif // DBG
734  return( Status );
735  }
736  InitialTeb->StackLimit = (PVOID)((PUCHAR)InitialTeb->StackLimit + RegionSize);
737  }
738 
739  return( STATUS_SUCCESS );
740 }
