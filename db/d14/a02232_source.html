45 //
46 // FT_PARTITION_STATE is an enumerated type that describes the state of an
47 // NTFT member.
48 // Up to one member can be 'orphaned', that is missing or out of sync with
49 // the other members. The orphaned member will remain orphaned until recovery
50 // action is commenced in DISKMAN.EXE.
51 // When READs are directed at an orphaned member, the data is returned from
52 // the redundant copy. WRITEs go to the secondary members only.
53 //
54 // Note that SyncRedundantCopy should NEVER be seen by anything other
55 // than ft itself.
56 //
57 
58 typedef enum _FT_PARTITION_STATE {
59  Healthy,
60  Orphaned,
61  Regenerating,
62  Initializing,
63  SyncRedundantCopy
64 } FT_PARTITION_STATE, *PFT_PARTITION_STATE;
