71 ObCreateObject (
72  __in KPROCESSOR_MODE ProbeMode,
73  __in POBJECT_TYPE ObjectType,
74  __in POBJECT_ATTRIBUTES ObjectAttributes,
75  __in KPROCESSOR_MODE OwnershipMode,
76  __inout_opt PVOID ParseContext,
77  __in ULONG ObjectBodySize,
78  __in ULONG PagedPoolCharge,
79  __in ULONG NonPagedPoolCharge,
80  __out PVOID *Object

83 /*++
84 
85 Routine Description:
86 
87  This functions allocates space for an NT Object from either
88  Paged or NonPaged pool. It captures the optional name and
89  SECURITY_DESCRIPTOR parameters for later use when the object is
90  inserted into an object table. No quota is charged at this time.
91  That occurs when the object is inserted into an object table.
92 
93 Arguments:
94 
95  ProbeMode - The processor mode to consider when doing a probe
96  of the input parameters
97 
98  ObjectType - A pointer of the type returned by ObCreateObjectType
99  that gives the type of object being created.
100 
101  ObjectAttributes - Optionally supplies the attributes of the object
102  being created (such as its name)
103 
104  OwnershipMode - The processor mode of who is going to own the object
105 
106  ParseContext - Ignored
107 
108  ObjectBodySize - Number of bytes to allocate for the object body. The
109  object body immediately follows the object header in memory and are
110  part of a single allocation.
111 
112  PagedPoolCharge - Supplies the amount of paged pool to charge for the
113  object. If zero is specified then the default charge for the object
114  type is used.
115 
116  NonPagedPoolCharge - Supplies the amount of nonpaged pool to charge for
117  the object. If zero is specified then the default charge for the
118  object type is used.
119 
120  Object - Receives a pointer to the newly created object
121 
122 Return Value:
123 
124  Following errors can occur:
125 
126  - invalid object type
127  - insufficient memory
128 
129 --*/
130 
131 {
132  UNICODE_STRING CapturedObjectName;
133  POBJECT_CREATE_INFORMATION ObjectCreateInfo;
134  POBJECT_HEADER ObjectHeader;
135  NTSTATUS Status;
136 
137  PAGED_CODE();
138 
139  UNREFERENCED_PARAMETER (ParseContext);
140 
141  //
142  // Allocate a buffer to capture the object creation information.
143  //
144 
145  ObjectCreateInfo = ObpAllocateObjectCreateInfoBuffer();
146 
147  if (ObjectCreateInfo == NULL) {
148 
149  Status = STATUS_INSUFFICIENT_RESOURCES;
150 
151  } else {
152 
153  //
154  // Capture the object attributes, quality of service, and object
155  // name, if specified. Otherwise, initialize the captured object
156  // name, the security quality of service, and the create attributes
157  // to default values.
158  //
159 
160  Status = ObpCaptureObjectCreateInformation( ObjectType,
161  ProbeMode,
162  OwnershipMode,
163  ObjectAttributes,
164  &CapturedObjectName,
165  ObjectCreateInfo,
166  FALSE );
167 
168  if (NT_SUCCESS(Status)) {
169 
170  //
171  // If the creation attributes are invalid, then return an error
172  // status.
173  //
174 
175  if (ObjectType->TypeInfo.InvalidAttributes & ObjectCreateInfo->Attributes) {
176 
177  Status = STATUS_INVALID_PARAMETER;
178 
179  } else {
180 
181  //
182  // Set the paged and nonpaged pool quota charges for the
183  // object allocation.
184  //
185 
186  if (PagedPoolCharge == 0) {
187 
188  PagedPoolCharge = ObjectType->TypeInfo.DefaultPagedPoolCharge;
189  }
190 
191  if (NonPagedPoolCharge == 0) {
192 
193  NonPagedPoolCharge = ObjectType->TypeInfo.DefaultNonPagedPoolCharge;
194  }
195 
196  ObjectCreateInfo->PagedPoolCharge = PagedPoolCharge;
197  ObjectCreateInfo->NonPagedPoolCharge = NonPagedPoolCharge;
198 
199  //
200  // Allocate and initialize the object.
201  //
202 
203  Status = ObpAllocateObject( ObjectCreateInfo,
204  OwnershipMode,
205  ObjectType,
206  &CapturedObjectName,
207  ObjectBodySize,
208  &ObjectHeader );
209 
210  if (NT_SUCCESS(Status)) {
211 
212  //
213  // If a permanent object is being created, then check if
214  // the caller has the appropriate privilege.
215  //
216 
217  *Object = &ObjectHeader->Body;
218 
219  if (ObjectHeader->Flags & OB_FLAG_PERMANENT_OBJECT) {
220 
221  if (!SeSinglePrivilegeCheck( SeCreatePermanentPrivilege,
222  ProbeMode)) {
223 
224  ObpFreeObject(*Object);
225 
226  Status = STATUS_PRIVILEGE_NOT_HELD;
227  }
228  }
229 
230 #ifdef POOL_TAGGING
231  if (ObpTraceEnabled && NT_SUCCESS(Status)) {
232 
233  //
234  // Register the object and push stack information for the
235  // first reference
236  //
237 
238  ObpRegisterObject( ObjectHeader );
239  ObpPushStackInfo( ObjectHeader, TRUE );
240  }
241 #endif //POOL_TAGGING
242 
243  //
244  // Here is the only successful path out of this module but
245  // this path can also return privilege not held. In the
246  // error case, all the resources have already been freed
247  // by ObpFreeObject.
248  //
249 
250  return Status;
251  }
252  }
253 
254  //
255  // An error path, free the create information.
256  //
257 
258  ObpReleaseObjectCreateInformation(ObjectCreateInfo);
259 
260  if (CapturedObjectName.Buffer != NULL) {
261 
262  ObpFreeObjectNameBuffer(&CapturedObjectName);
263  }
264  }
265 
266  //
267  // An error path, free object creation information buffer.
268  //
269 
270  ObpFreeObjectCreateInfoBuffer(ObjectCreateInfo);
271  }
272 
273  //
274  // An error path
275  //
276 
277  return Status;
278 }

281 NTSTATUS
282 ObpCaptureObjectCreateInformation (
283  IN POBJECT_TYPE ObjectType OPTIONAL,
284  IN KPROCESSOR_MODE ProbeMode,
285  IN KPROCESSOR_MODE CreatorMode,
286  IN POBJECT_ATTRIBUTES ObjectAttributes,
287  IN OUT PUNICODE_STRING CapturedObjectName,
288  IN POBJECT_CREATE_INFORMATION ObjectCreateInfo,
289  IN LOGICAL UseLookaside
290  )
291 
292 /*++
293 
294 Routine Description:
295 
296  This function captures the object creation information and stuff
297  it into the input variable ObjectCreateInfo
298 
299 Arguments:
300 
301  ObjectType - Specifies the type of object we expect to capture,
302  currently ignored.
303 
304  ProbeMode - Specifies the processor mode for doing our parameter
305  probes
306 
307  CreatorMode - Specifies the mode the object is being created for
308 
309  ObjectAttributes - Supplies the object attributes we are trying
310  to capture
311 
312  CapturedObjectName - Receives the name of the object being created
313 
314  ObjectCreateInfo - Receives the create information for the object
315  like its root, attributes, and security information
316 
317  UseLookaside - Specifies if we are to allocate the captured name
318  buffer from the lookaside list or from straight pool.
319 
320 Return Value:
321 
322  An appropriate status value
323 
324 --*/
325 
326 {
327  PUNICODE_STRING ObjectName;
328  PSECURITY_DESCRIPTOR SecurityDescriptor;
329  PSECURITY_QUALITY_OF_SERVICE SecurityQos;
330  NTSTATUS Status;
331  ULONG Size;
332 
333  PAGED_CODE();
334 
335  UNREFERENCED_PARAMETER (ObjectType);
336 
337  //
338  // Capture the object attributes, the security quality of service, if
339  // specified, and object name, if specified.
340  //
341 
342  Status = STATUS_SUCCESS;
343 
344  RtlZeroMemory(ObjectCreateInfo, sizeof(OBJECT_CREATE_INFORMATION));
345 
346  try {
347 
348  if (ARGUMENT_PRESENT(ObjectAttributes)) {
349 
350  //
351  // Probe the object attributes if necessary.
352  //
353 
354  if (ProbeMode != KernelMode) {
355 
356  ProbeForReadSmallStructure( ObjectAttributes,
357  sizeof(OBJECT_ATTRIBUTES),
358  sizeof(ULONG_PTR) );
359  }
360 
361  if (ObjectAttributes->Length != sizeof(OBJECT_ATTRIBUTES) ||
362  (ObjectAttributes->Attributes & ~OBJ_ALL_VALID_ATTRIBUTES)) {
363 
364  Status = STATUS_INVALID_PARAMETER;
365 
366  goto failureExit;
367  }
368 
369  //
370  // Capture the object attributes.
371  //
372 
373  ObjectCreateInfo->RootDirectory = ObjectAttributes->RootDirectory;
374  ObjectCreateInfo->Attributes = ObjectAttributes->Attributes & OBJ_ALL_VALID_ATTRIBUTES;
375  //
376  // Remove privileged option if passed in from user mode
377  //
378  if (CreatorMode != KernelMode) {
379  ObjectCreateInfo->Attributes &= ~OBJ_KERNEL_HANDLE;
380  } else if (ObWatchHandles) {
381  if ((ObjectCreateInfo->Attributes&OBJ_KERNEL_HANDLE) == 0 &&
382  PsGetCurrentProcess() != PsInitialSystemProcess) {
383  DbgBreakPoint ();
384  }
385  }
386  ObjectName = ObjectAttributes->ObjectName;
387  SecurityDescriptor = ObjectAttributes->SecurityDescriptor;
388  SecurityQos = ObjectAttributes->SecurityQualityOfService;
389 
390  if (ARGUMENT_PRESENT(SecurityDescriptor)) {
391 
392  Status = SeCaptureSecurityDescriptor( SecurityDescriptor,
393  ProbeMode,
394  PagedPool,
395  TRUE,
396  &ObjectCreateInfo->SecurityDescriptor );
397 
398  if (!NT_SUCCESS(Status)) {
399 
400  KdPrint(( "OB: Failed to capture security descriptor at %08x - Status == %08x\n",
401  SecurityDescriptor,
402  Status) );
403 
404  //
405  // The cleanup routine depends on this being NULL if it isn't
406  // allocated. SeCaptureSecurityDescriptor may modify this
407  // parameter even if it fails.
408  //
409 
410  ObjectCreateInfo->SecurityDescriptor = NULL;
411 
412  goto failureExit;
413  }
414 
415  SeComputeQuotaInformationSize( ObjectCreateInfo->SecurityDescriptor,
416  &Size );
417 
418  ObjectCreateInfo->SecurityDescriptorCharge = SeComputeSecurityQuota( Size );
419  ObjectCreateInfo->ProbeMode = ProbeMode;
420  }
421 
422  if (ARGUMENT_PRESENT(SecurityQos)) {
423 
424  if (ProbeMode != KernelMode) {
425 
426  ProbeForReadSmallStructure( SecurityQos, sizeof(*SecurityQos), sizeof(ULONG));
427  }
428 
429  ObjectCreateInfo->SecurityQualityOfService = *SecurityQos;
430  ObjectCreateInfo->SecurityQos = &ObjectCreateInfo->SecurityQualityOfService;
431  }
432 
433  } else {
434 
435  ObjectName = NULL;
436  }
437 
438  } except (ExSystemExceptionFilter()) {
439 
440  Status = GetExceptionCode();
441 
442  goto failureExit;
443  }
444 
445  //
446  // If an object name is specified, then capture the object name.
447  // Otherwise, initialize the object name descriptor and check for
448  // an incorrectly specified root directory.
449  //
450 
451  if (ARGUMENT_PRESENT(ObjectName)) {
452 
453  Status = ObpCaptureObjectName( ProbeMode,
454  ObjectName,
455  CapturedObjectName,
456  UseLookaside );
457 
458  } else {
459 
460  CapturedObjectName->Buffer = NULL;
461  CapturedObjectName->Length = 0;
462  CapturedObjectName->MaximumLength = 0;
463 
464  if (ARGUMENT_PRESENT(ObjectCreateInfo->RootDirectory)) {
465 
466  Status = STATUS_OBJECT_NAME_INVALID;
467  }
468  }
469 
470  //
471  // If the completion status is not successful, and a security quality
472  // of service parameter was specified, then free the security quality
473  // of service memory.
474  //
475 
476 failureExit:
477 
478  if (!NT_SUCCESS(Status)) {
479 
480  ObpReleaseObjectCreateInformation(ObjectCreateInfo);
481  }
482 
483  return Status;
484 }

487 NTSTATUS
488 ObpCaptureObjectName (
489  IN KPROCESSOR_MODE ProbeMode,
490  IN PUNICODE_STRING ObjectName,
491  IN OUT PUNICODE_STRING CapturedObjectName,
492  IN LOGICAL UseLookaside
493  )
494 
495 /*++
496 
497 Routine Description:
498 
499  This function captures the object name but first verifies that
500  it is at least properly sized.
501 
502 Arguments:
503 
504  ProbeMode - Supplies the processor mode to use when probing
505  the object name
506 
507  ObjectName - Supplies the caller's version of the object name
508 
509  CapturedObjectName - Receives the captured verified version
510  of the object name
511 
512  UseLookaside - Indicates if the captured name buffer should be
513  allocated from the lookaside list or from straight pool
514 
515 Return Value:
516 
517  An appropriate status value
518 
519 --*/
520 
521 {
522  PWCH FreeBuffer;
523  UNICODE_STRING InputObjectName;
524  ULONG Length;
525  NTSTATUS Status;
526 
527  PAGED_CODE();
528 
529  //
530  // Initialize the object name descriptor and capture the specified name
531  // string.
532  //
533 
534  CapturedObjectName->Buffer = NULL;
535  CapturedObjectName->Length = 0;
536  CapturedObjectName->MaximumLength = 0;
537 
538  Status = STATUS_SUCCESS;
539 
540  //
541  // Probe and capture the name string descriptor and probe the
542  // name string, if necessary.
543  //
544 
545  FreeBuffer = NULL;
546 
547  try {
548 
549  if (ProbeMode != KernelMode) {
550 
551  ProbeAndReadUnicodeStringEx(&InputObjectName, ObjectName);
552 
553  ProbeForRead( InputObjectName.Buffer,
554  InputObjectName.Length,
555  sizeof(WCHAR) );
556 
557  } else {
558 
559  InputObjectName = *ObjectName;
560  }
561 
562  //
563  // If the length of the string is not zero, then capture the string.
564  //
565 
566  if (InputObjectName.Length != 0) {
567 
568  //
569  // If the length of the string is not an even multiple of the
570  // size of a UNICODE character or cannot be zero terminated,
571  // then return an error.
572  //
573 
574  Length = InputObjectName.Length;
575 
576  if (((Length & (sizeof(WCHAR) - 1)) != 0) ||
577  (Length == (MAXUSHORT - sizeof(WCHAR) + 1))) {
578 
579  Status = STATUS_OBJECT_NAME_INVALID;
580 
581  } else {
582 
583  //
584  // Allocate a buffer for the specified name string.
585  //
586  // N.B. The name buffer allocation routine adds one
587  // UNICODE character to the length and initializes
588  // the string descriptor.
589  //
590 
591  FreeBuffer = ObpAllocateObjectNameBuffer( Length,
592  UseLookaside,
593  CapturedObjectName );
594 
595  if (FreeBuffer == NULL) {
596 
597  Status = STATUS_INSUFFICIENT_RESOURCES;
598 
599  } else {
600 
601  //
602  // Copy the specified name string to the destination
603  // buffer.
604  //
605 
606  RtlCopyMemory(FreeBuffer, InputObjectName.Buffer, Length);
607 
608  //
609  // Zero terminate the name string and initialize the
610  // string descriptor.
611  //
612 
613  FreeBuffer[Length / sizeof(WCHAR)] = UNICODE_NULL;
614  }
615  }
616  }
617 
618  } except(ExSystemExceptionFilter()) {
619 
620  Status = GetExceptionCode();
621 
622  if (FreeBuffer != NULL) {
623 
624  ExFreePool(FreeBuffer);
625  }
626  }
627 
628  return Status;
629 }

632 PWCHAR
633 ObpAllocateObjectNameBuffer (
634  IN ULONG Length,
635  IN LOGICAL UseLookaside,
636  IN OUT PUNICODE_STRING ObjectName
637  )
638 
639 /*++
640 
641 Routine Description:
642 
643  This function allocates an object name buffer.
644 
645  N.B. This function is nonpageable.
646 
647 Arguments:
648 
649  Length - Supplies the length of the required buffer in bytes.
650 
651  UseLookaside - Supplies a logical variable that determines whether an
652  attempt is made to allocate the name buffer from the lookaside list.
653 
654  ObjectName - Supplies a pointer to a name buffer string descriptor.
655 
656 Return Value:
657 
658  If the allocation is successful, then name buffer string descriptor
659  is initialized and the address of the name buffer is returned as the
660  function value. Otherwise, a value of NULL is returned.
661 
662 --*/
663 
664 {
665  PVOID Buffer;
666  ULONG Maximum;
667 
668  //
669  // If allocation from the lookaside lists is specified and the buffer
670  // size is less than the size of lookaside list entries, then attempt
671  // to allocate the name buffer from the lookaside lists. Otherwise,
672  // attempt to allocate the name buffer from nonpaged pool.
673  //
674 
675  Maximum = Length + sizeof(WCHAR);
676 
677  if ((UseLookaside == FALSE) || (Maximum > OBJECT_NAME_BUFFER_SIZE)) {
678 
679  //
680  // Attempt to allocate the buffer from nonpaged pool.
681  //
682 
683  Buffer = ExAllocatePoolWithTag( OB_NAMESPACE_POOL_TYPE , Maximum, 'mNbO' );
684 
685  } else {
686 
687  //
688  // Attempt to allocate the name buffer from the lookaside list. If
689  // the allocation attempt fails, then attempt to allocate the name
690  // buffer from pool.
691  //
692 
693  Maximum = OBJECT_NAME_BUFFER_SIZE;
694  Buffer = ExAllocateFromPPLookasideList(LookasideNameBufferList);
695  }
696 
697  //
698  // Initialize the string descriptor and return the buffer address.
699  //
700 
701  ObjectName->Length = (USHORT)Length;
702  ObjectName->MaximumLength = (USHORT)Maximum;
703  ObjectName->Buffer = Buffer;
704 
705  return (PWCHAR)Buffer;
706 }

814 /*++
815 
816 Routine Description:
817 
818  This routine allocates a new object including the object header
819  and body from pool and fill in the appropriate fields.
820 
821 Arguments:
822 
823  ObjectCreateInfo - Supplies the create information for the new object
824 
825  OwnershipMode - Supplies the processor mode of who is going to own
826  the object
827 
828  ObjectType - Optionally supplies the object type of the object being
829  created. If the object create info not null then this field must
830  be supplied.
831 
832  ObjectName - Supplies the name of the object being created
833 
834  ObjectBodySize - Specifies the size, in bytes, of the body of the object
835  being created
836 
837  ReturnedObjectHeader - Receives a pointer to the object header for the
838  newly created objet.
839 
840 Return Value:
841 
842  An appropriate status value.
843 
844 --*/
845 
846 {
847  ULONG HeaderSize;
848  POBJECT_HEADER ObjectHeader;
849  ULONG QuotaInfoSize;
850  ULONG HandleInfoSize;
851  ULONG NameInfoSize;
852  ULONG CreatorInfoSize;
853  POBJECT_HEADER_QUOTA_INFO QuotaInfo;
854  POBJECT_HEADER_HANDLE_INFO HandleInfo;
855  POBJECT_HEADER_NAME_INFO NameInfo;
856  POBJECT_HEADER_CREATOR_INFO CreatorInfo;
857  POOL_TYPE PoolType;
858 
859  PAGED_CODE();
860 
861 #if DBG
862  ObpObjectsCreated += 1;
863 #endif // DBG
864 
865  //
866  // Compute the sizes of the optional object header components.
867  //
868 
869  if (ObjectCreateInfo == NULL) {
870 
871  QuotaInfoSize = 0;
872  HandleInfoSize = 0;
873  NameInfoSize = sizeof( OBJECT_HEADER_NAME_INFO );
874  CreatorInfoSize = sizeof( OBJECT_HEADER_CREATOR_INFO );
875 
876  } else {
877

879  // The caller specified some additional object create info
880  //
881  // First check to see if we need to set the quota
882  //
883 
884  if (((ObjectCreateInfo->PagedPoolCharge != ObjectType->TypeInfo.DefaultPagedPoolCharge ||
885  ObjectCreateInfo->NonPagedPoolCharge != ObjectType->TypeInfo.DefaultNonPagedPoolCharge ||
886  ObjectCreateInfo->SecurityDescriptorCharge > SE_DEFAULT_SECURITY_QUOTA) &&
887  PsGetCurrentProcess() != PsInitialSystemProcess) ||
888  (ObjectCreateInfo->Attributes & OBJ_EXCLUSIVE)) {
889 
890  QuotaInfoSize = sizeof( OBJECT_HEADER_QUOTA_INFO );
891 #if DBG
892  ObpObjectsWithPoolQuota += 1;
893 #endif // DBG
894 
895  } else {
896 
897  QuotaInfoSize = 0;
898  }
899 
900  //
901  // Check if we are to allocate space to maintain handle counts
902  //
903 
904  if (ObjectType->TypeInfo.MaintainHandleCount) {
905 
906  HandleInfoSize = sizeof( OBJECT_HEADER_HANDLE_INFO );
907 #if DBG
908  ObpObjectsWithHandleDB += 1;
909 #endif // DBG
910 
911  } else {
