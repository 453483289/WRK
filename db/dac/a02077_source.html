31 CmpGetValueListFromCache(
32  IN PCM_KEY_CONTROL_BLOCK KeyControlBlock,
33  OUT PCELL_DATA *List,
34  OUT BOOLEAN *IndexCached,

37 /*++
38 
39 Routine Description:
40 
41  Get the Valve Index Array. Check if it is already cached, if not, cache it and return
42  the cached entry.
43 
44 Arguments:
45 
46  Hive - pointer to hive control structure for hive of interest
47 
48  ChildList - pointer/index to the Value Index array
49 
50  IndexCached - Indicating whether Value Index list is cached or not.
51 
52 Return Value:
53 
54  Pointer to the Valve Index Array.
55 
56  NULL when we could not map view
57 
58 --*/

60  HCELL_INDEX CellToRelease;
61  PCACHED_CHILD_LIST ChildList;
62  PHHIVE Hive;
63 #ifndef _WIN64
64  ULONG AllocSize;
65  PCM_CACHED_VALUE_INDEX CachedValueIndex;
66  ULONG i;
67 #endif
68  ASSERT_KCB_LOCKED(KeyControlBlock);
69 
70  Hive = KeyControlBlock->KeyHive;
71  ChildList = &(KeyControlBlock->ValueCache);
72  *ValueListToRelease = HCELL_NIL;
73 
74 #ifndef _WIN64
75  *IndexCached = TRUE;
76  if (CMP_IS_CELL_CACHED(ChildList->ValueList)) {
77  //
78  // The entry is already cached.
79  //
80  *List = CMP_GET_CACHED_CELLDATA(ChildList->ValueList);
81  } else {
82  //
83  // The entry is not cached. The element contains the hive index.
84  // ensure exclusive lock.
85  //
86  if( (CmpIsKCBLockedExclusive(KeyControlBlock) == FALSE) &&
87  (CmpTryConvertKCBLockSharedToExclusive(KeyControlBlock) == FALSE) ) {
88  //
89  // need to upgrade lock to exclusive
90  //
91  return SearchNeedExclusiveLock;
92  }
93 
94  CellToRelease = CMP_GET_CACHED_CELL_INDEX(ChildList->ValueList);
95  *List = (PCELL_DATA) HvGetCell(Hive, CellToRelease);
96  if( *List == NULL ) {
97  //
98  // we couldn't map a view for this cell
99  //
100  *IndexCached = FALSE;
101  return SearchFail;
102  }
103 
104  //
105  // Allocate a PagedPool to cache the value index cell.
106  //
107 
108  AllocSize = ChildList->Count * sizeof(ULONG_PTR) + FIELD_OFFSET(CM_CACHED_VALUE_INDEX, Data);
109  CachedValueIndex = (PCM_CACHED_VALUE_INDEX) ExAllocatePoolWithTag(PagedPool, AllocSize, CM_CACHE_VALUE_INDEX_TAG);
110 
111  if (CachedValueIndex) {
112 
113  CachedValueIndex->CellIndex = CMP_GET_CACHED_CELL_INDEX(ChildList->ValueList);
114 #pragma prefast(suppress:12009, "no overflow")
115  for (i=0; i<ChildList->Count; i++) {
116  CachedValueIndex->Data.List[i] = (ULONG_PTR) (*List)->u.KeyList[i];
117  }
118

120 
121  // Trying to catch the BAD guy who writes over our pool.
122  CmpMakeSpecialPoolReadOnly( CachedValueIndex );
123 
124  //
125  // Now we have the stuff cached, use the cache data.
126  //
127  *List = CMP_GET_CACHED_CELLDATA(ChildList->ValueList);
128  } else {
129  //
130  // If the allocation fails, just do not cache it. continue.
131  //
132  *IndexCached = FALSE;
133  }
134  *ValueListToRelease = CellToRelease;
135  }
136 #else
137  CellToRelease = CMP_GET_CACHED_CELL_INDEX(ChildList->ValueList);
138  *List = (PCELL_DATA) HvGetCell(Hive, CellToRelease);
139  *IndexCached = FALSE;
140  if( *List == NULL ) {
141  //
142  // we couldn't map a view for this cell
143  // OBS: we can drop this as we return List anyway; just for clarity
144  //

154 CmpGetValueKeyFromCache(
155  IN PCM_KEY_CONTROL_BLOCK KeyControlBlock,
156  IN PCELL_DATA List,
157  IN ULONG Index,
158  OUT PPCM_CACHED_VALUE *ContainingList,
159  OUT PCM_KEY_VALUE *Value,
160  IN BOOLEAN IndexCached,
161  OUT BOOLEAN *ValueCached,
162  OUT PHCELL_INDEX CellToRelease

164 /*++
165 
166 Routine Description:
167 
168  Get the Valve Node. Check if it is already cached, if not but the index is cached,
169  cache and return the value node.
170 
171 Arguments:
172 
173  Hive - pointer to hive control structure for hive of interest.
174 
175  List - pointer to the Value Index Array (of ULONG_PTR if cached and ULONG if non-cached)
176 
177  Index - Index in the Value index array
178 
179  ContainlingList - The address of the entry that will receive the found cached value.
180 
181  IndexCached - Indicate if the index list is cached. If not, everything is from the
182  original registry data.
183 
184  ValueCached - Indicating whether Value is cached or not.
185 
186 Return Value:
187 
188  Pointer to the Value Node.
189 
190  NULL when we couldn't map a view
191 --*/
192 {
193  PULONG_PTR CachedList;
194  ULONG AllocSize;
195  ULONG CopySize;
196  PCM_CACHED_VALUE CachedValue;
197  PHHIVE Hive;
198 
199  *CellToRelease = HCELL_NIL;
200  Hive = KeyControlBlock->KeyHive;
201  *Value = NULL;
202  if (IndexCached) {
203  //
204  // The index array is cached, so List is pointing to an array of ULONG_PTR.
205  // Use CachedList.
206  //
207  CachedList = (PULONG_PTR) List;
208  *ValueCached = TRUE;
209  if (CMP_IS_CELL_CACHED(CachedList[Index])) {
210  *Value = CMP_GET_CACHED_KEYVALUE(CachedList[Index]);
211  *ContainingList = &((PCM_CACHED_VALUE) CachedList[Index]);
212  } else {
213  //
214  // ensure exclusive lock.
215  //
216  if( (CmpIsKCBLockedExclusive(KeyControlBlock) == FALSE) &&
217  (CmpTryConvertKCBLockSharedToExclusive(KeyControlBlock) == FALSE) ) {
218  //
219  // need to upgrade lock to exclusive
220  //
221  return SearchNeedExclusiveLock;
222  }
223  *Value = (PCM_KEY_VALUE) HvGetCell(Hive, List->u.KeyList[Index]);
224  if( *Value == NULL ) {
225  //
226  // we couldn't map a view for this cell
227  // just return NULL; the caller must handle it gracefully
228  //
229  return SearchFail;
230  }
231  *CellToRelease = List->u.KeyList[Index];
232 
233  //
234  // Allocate a PagedPool to cache the value node.
235  //
236  CopySize = (ULONG) HvGetCellSize(Hive, *Value);
237  AllocSize = CopySize + FIELD_OFFSET(CM_CACHED_VALUE, KeyValue);
238 
239  CachedValue = (PCM_CACHED_VALUE) ExAllocatePoolWithTag(PagedPool, AllocSize, CM_CACHE_VALUE_TAG);
240 
241  if (CachedValue) {
242  //
243  // Set the information for later use if we need to cache data as well.
244  //
245  if ((*Value)->Flags & VALUE_COMP_NAME) {
246  CachedValue->HashKey = CmpComputeHashKeyForCompressedName(0,(*Value)->Name,(*Value)->NameLength);
247  } else {
248  UNICODE_STRING TmpStr;
249  TmpStr.Length = (*Value)->NameLength;
250  TmpStr.Buffer = (*Value)->Name;
251  CachedValue->HashKey = CmpComputeHashKey(0,&TmpStr
252 #if DBG
253  , TRUE
254 #endif
255  );
256  }
257  CachedValue->DataCacheType = CM_CACHE_DATA_NOT_CACHED;
258  CachedValue->ValueKeySize = (USHORT) CopySize;
259 
260  RtlCopyMemory((PVOID)&(CachedValue->KeyValue), *Value, CopySize);
261 
262 
263  // Trying to catch the BAD guy who writes over our pool.
264  CmpMakeSpecialPoolReadWrite( CMP_GET_CACHED_ADDRESS(CachedList) );
265 
266  CachedList[Index] = CMP_MARK_CELL_CACHED(CachedValue);
267 
268  // Trying to catch the BAD guy who writes over our pool.
269  CmpMakeSpecialPoolReadOnly( CMP_GET_CACHED_ADDRESS(CachedList) );
270 
271 
272  // Trying to catch the BAD guy who writes over our pool.
273  CmpMakeSpecialPoolReadOnly(CachedValue);
274 
275  *ContainingList = &((PCM_CACHED_VALUE) CachedList[Index]);
276  //
277  // Now we have the stuff cached, use the cache data.
278  //
279  (*Value) = CMP_GET_CACHED_KEYVALUE(CachedValue);
280  } else {
281  //
282  // If the allocation fails, just do not cache it. continue.
283  //
284  *ValueCached = FALSE;
285  }
286  }
287  } else {
288  //
289  // The Valve Index Array is from the registry hive, just get the cell and move on.
290  //
291  (*Value) = (PCM_KEY_VALUE) HvGetCell(Hive, List->u.KeyList[Index]);
292  *ValueCached = FALSE;
293  if( *Value == NULL ) {
294  //
295  // we couldn't map a view for this cell
296  // just return NULL; the caller must handle it gracefully
297  // OBS: we may remove this as we return pchild anyway; just for clarity
298  //
299  return SearchFail;
300  }
301  *CellToRelease = List->u.KeyList[Index];
302  }
303  return SearchSuccess;
304 }

316 /*++
317 
318 Routine Description:
319 
320  Find a value node given a value list array and a value name. It sequentially walk
321  through each value node to look for a match. If the array and
322  value nodes touched are not already cached, cache them.
323 
324 Arguments:
325 
326  Hive - pointer to hive control structure for hive of interest
327 
328  ChildList - pointer/index to the Value Index array
329 
330  Name - name of value to find
331 
332  ContainlingList - The address of the entry that will receive the found cached value.
333 
334  Index - pointer to variable to receive index for child
335 
336  ValueCached - Indicate if the value node is cached.
337 
338 Return Value:
339 
340  HCELL_INDEX for the found cell
341  HCELL_NIL if not found
342 
343 
344 Notes:
345 
346  New hives (Minor >= 4) have ValueList sorted; this implies ValueCache is sorted too;
347  So, we can do a binary search here!
348 
349 --*/
350 {
351  UNICODE_STRING Candidate;
352  LONG Result;
353  PCELL_DATA List;
354  BOOLEAN IndexCached;
355  ULONG Current;
356  HCELL_INDEX ValueListToRelease = HCELL_NIL;
357  ULONG HashKey = 0;
358  PHHIVE Hive = KeyControlBlock->KeyHive;
359  PCACHED_CHILD_LIST ChildList = &(KeyControlBlock->ValueCache);
360  VALUE_SEARCH_RETURN_TYPE ret = SearchFail;
361 
362  ASSERT_KCB_LOCKED(KeyControlBlock);
363 
364  *CellToRelease = HCELL_NIL;
365  *Value = NULL;
366 
367  if (ChildList->Count != 0) {
368  ret = CmpGetValueListFromCache(KeyControlBlock,&List, &IndexCached,&ValueListToRelease);
369  if( ret != SearchSuccess ) {
370  //
371  // retry with exclusive lock, since we need to update the cache
372  // or fail altogether
373  //
374  ASSERT( (ret == SearchFail) || (CmpIsKCBLockedExclusive(KeyControlBlock) == FALSE) );
375  ASSERT( ValueListToRelease == HCELL_NIL );
376  return ret;
377  }
378 
379  if( IndexCached ) {
380  try {
381  HashKey = CmpComputeHashKey(0,Name
382 #if DBG
383  , TRUE
384 #endif
385  );
386  } except (EXCEPTION_EXECUTE_HANDLER) {
387  ret = SearchFail;
388  goto Exit;
389  }
390  }
391  //
392  // old plain hive; simulate a for
393  //
394  Current = 0;
395 
396  while( TRUE ) {
397  if( *CellToRelease != HCELL_NIL ) {
398  HvReleaseCell(Hive,*CellToRelease);
399  *CellToRelease = HCELL_NIL;
400  }
401  ret = CmpGetValueKeyFromCache(KeyControlBlock, List, Current, ContainingList, Value, IndexCached, ValueCached, CellToRelease);
402  if( ret != SearchSuccess ) {
403  //
404  // retry with exclusive lock, since we need to update the cache
405  // or fail altogether
406  //
407  ASSERT( (ret == SearchFail) || (CmpIsKCBLockedExclusive(KeyControlBlock) == FALSE) );
408  ASSERT( *CellToRelease == HCELL_NIL );
409  return ret;
410  }
411 
412  //
413  // only compare names when hash matches.
414  //
415  if( IndexCached && (*ValueCached) && (HashKey != ((PCM_CACHED_VALUE)((CMP_GET_CACHED_ADDRESS(**ContainingList))))->HashKey) ) {
416  //
417  // no hash match; skip it
418  //
419 #if DBG
420  try {
421  //
422  // Name has user-mode buffer.
423  //
424 
425  if ((*Value)->Flags & VALUE_COMP_NAME) {
426  Result = CmpCompareCompressedName( Name,
427  (*Value)->Name,
428  (*Value)->NameLength,
429  0);
430  } else {
431  Candidate.Length = (*Value)->NameLength;
432  Candidate.MaximumLength = Candidate.Length;
433  Candidate.Buffer = (*Value)->Name;
434  Result = RtlCompareUnicodeString( Name,
435  &Candidate,
436  TRUE);
437  }
438 
439 
440  } except (EXCEPTION_EXECUTE_HANDLER) {
441  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_EXCEPTION,"CmpFindValueByNameFromCache: code:%08lx\n", GetExceptionCode()));
442  //
443  // the caller will bail out. Some ,alicious caller altered the Name buffer since we probed it.
444  //
445  *Value = NULL;
446  ret = SearchFail;
447  goto Exit;
448  }
449  ASSERT( Result != 0 );
450 #endif
451  Result = 1;
452  } else {
453  try {
454  //
455  // Name has user-mode buffer.
456  //
457 
458  if( (*Value)->Flags & VALUE_COMP_NAME) {
459  Result = CmpCompareCompressedName( Name,
460  (*Value)->Name,
461  (*Value)->NameLength,
462  0);
463  } else {
464  Candidate.Length = (*Value)->NameLength;
465  Candidate.MaximumLength = Candidate.Length;
466  Candidate.Buffer = (*Value)->Name;
467  Result = RtlCompareUnicodeString( Name,
468  &Candidate,
469  TRUE);
470  }
471 
472 
473  } except (EXCEPTION_EXECUTE_HANDLER) {
474  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_EXCEPTION,"CmpFindValueByNameFromCache: code:%08lx\n", GetExceptionCode()));
475  //
476  // the caller will bail out. Some ,alicious caller altered the Name buffer since we probed it.
477  //
478  *Value = NULL;
479  ret = SearchFail;
480  goto Exit;
481  }
482  }
483  if (Result == 0) {
484  //
485  // Success, fill the index, return data to caller and exit
486  //
487  *Index = Current;
488  ret = SearchSuccess;
489  goto Exit;
490  }
491 
492  //
493  // compute the next index to try: old'n plain hive; go on
494  //
495  Current++;
496  if( Current == ChildList->Count ) {
497  //
498  // we've reached the end of the list; nicely return
499  //
500  (*Value) = NULL;
501  ret = SearchFail;
502  goto Exit;
503  }
504 
505  } // while(TRUE)
506  }
507 
508  //
509  // in the new design we shouldn't get here; we should exit the while loop with return
510  //
511  ASSERT( ChildList->Count == 0 );
512 
513 Exit:
514  if( ValueListToRelease != HCELL_NIL ) {
515  HvReleaseCell(Hive,ValueListToRelease);
516  }
517  return ret;
518 }

520 VALUE_SEARCH_RETURN_TYPE
521 CmpGetValueDataFromCache(
522  IN PCM_KEY_CONTROL_BLOCK KeyControlBlock,
523  IN PPCM_CACHED_VALUE ContainingList,
524  IN PCELL_DATA ValueKey,
525  IN BOOLEAN ValueCached,
526  OUT PUCHAR *DataPointer,
527  OUT PBOOLEAN Allocated,
528  OUT PHCELL_INDEX CellToRelease
529 )
530 /*++
531 
532 Routine Description:
533 
534  Get the cached Value Data given a value node.
535 
536 Arguments:
537 
538  Hive - pointer to hive control structure for hive of interest
539 
540  ContainingList - Address that stores the allocation address of the value node.
541  We need to update this when we do a re-allocate to cache
542  both value key and value data.
543 
544  ValueKey - pointer to the Value Key
545 
546  ValueCached - Indicating whether Value key is cached or not.
547 
548  DataPointer - out param to receive a pointer to the data
549 
550  Allocated - out param telling if the caller should free the DataPointer
551 
552 Return Value:
553 
554  TRUE - data was retrieved
555  FALSE - some error (STATUS_INSUFFICIENT_RESOURCES) occured
556 
557 Note:
558 
559  The caller is responsible for freeing the DataPointer when we signal it to him
560  by setting Allocated on TRUE;
561 
562  Also we must be sure that MAXIMUM_CACHED_DATA is smaller than CM_KEY_VALUE_BIG
563 --*/
564 {
565  //
566  // Cache the data if needed.
567  //
568  PCM_CACHED_VALUE OldEntry;
569  PCM_CACHED_VALUE NewEntry;
570  PUCHAR Cacheddatapointer;
571  ULONG AllocSize;
572  ULONG CopySize;
573  ULONG DataSize;
574  PHHIVE Hive;
575 
576  ASSERT( MAXIMUM_CACHED_DATA < CM_KEY_VALUE_BIG );
577 
578  //
579  // this routine should not be called for small data
580  //
581  ASSERT( (ValueKey->u.KeyValue.DataLength & CM_KEY_VALUE_SPECIAL_SIZE) == 0 );
582 
583  ASSERT_KCB_LOCKED(KeyControlBlock);
584 
585  Hive = KeyControlBlock->KeyHive;
586  //
587  // init out params
588  //
589  *DataPointer = NULL;
590  *Allocated = FALSE;
591  *CellToRelease = HCELL_NIL;
592 
593  if (ValueCached) {
594  OldEntry = (PCM_CACHED_VALUE) CMP_GET_CACHED_ADDRESS(*ContainingList);
595  if (OldEntry->DataCacheType == CM_CACHE_DATA_CACHED) {
596  //
597  // Data is already cached, use it.
598  //
599  *DataPointer = (PUCHAR) ((ULONG_PTR) ValueKey + OldEntry->ValueKeySize);
600  } else {
601  //
602  // ensure exclusive lock.
603  //
604  if( (CmpIsKCBLockedExclusive(KeyControlBlock) == FALSE) &&
605  (CmpTryConvertKCBLockSharedToExclusive(KeyControlBlock) == FALSE) ) {
606  //
607  // need to upgrade lock to exclusive
608  //
609  return SearchNeedExclusiveLock;
610  }
611  if ((OldEntry->DataCacheType == CM_CACHE_DATA_TOO_BIG) ||
612  (ValueKey->u.KeyValue.DataLength > MAXIMUM_CACHED_DATA )
613  ){
614  //
615  // Mark the type and do not cache it.
616  //
617  OldEntry->DataCacheType = CM_CACHE_DATA_TOO_BIG;
618 
619  //
620  // Data is too big to warrant caching, get it from the registry;
621  // - regardless of the size; we might be forced to allocate a buffer
622  //
623  if( CmpGetValueData(Hive,&(ValueKey->u.KeyValue),&DataSize,DataPointer,Allocated,CellToRelease) == FALSE ) {
624  //
625  // insufficient resources; return NULL
626  //
627  ASSERT( *Allocated == FALSE );
628  ASSERT( *DataPointer == NULL );
629  return SearchFail;
630  }
631 
632  } else {
633  //
634  // consistency check
635  //
636  ASSERT(OldEntry->DataCacheType == CM_CACHE_DATA_NOT_CACHED);
637 
638  //
639  // Value data is not cached.
640  // Check the size of value data, if it is smaller than MAXIMUM_CACHED_DATA, cache it.
641  //
642  // Anyway, the data is for sure not stored in a big data cell (see test above)
643  //
644  //
645  *DataPointer = (PUCHAR)HvGetCell(Hive, ValueKey->u.KeyValue.Data);
646  if( *DataPointer == NULL ) {
647  //
648  // we couldn't map this cell
649  // the caller must handle this gracefully !
650  //
651  return SearchFail;
652  }
653  //
654  // inform the caller it has to release this cell
655  //
656  *CellToRelease = ValueKey->u.KeyValue.Data;
657 
658  //
659  // copy only valid data; cell might be bigger
660  //
661  DataSize = (ULONG)ValueKey->u.KeyValue.DataLength;
662 
663  //
664  // consistency check
665  //
666  ASSERT(DataSize <= MAXIMUM_CACHED_DATA);
667 
668  //
669  // Data is not cached and now we are going to do it.
670  // Reallocate a new cached entry for both value key and value data.
671  //
672  CopySize = DataSize + OldEntry->ValueKeySize;
673  AllocSize = CopySize + FIELD_OFFSET(CM_CACHED_VALUE, KeyValue);
674 
675  NewEntry = (PCM_CACHED_VALUE) ExAllocatePoolWithTag(PagedPool, AllocSize, CM_CACHE_VALUE_DATA_TAG);
676 
677  if (NewEntry) {
678  //
679  // Now fill the data to the new cached entry
680  //
681  NewEntry->HashKey = OldEntry->HashKey;
682  NewEntry->DataCacheType = CM_CACHE_DATA_CACHED;
683  NewEntry->ValueKeySize = OldEntry->ValueKeySize;
684 
685  RtlCopyMemory((PVOID)&(NewEntry->KeyValue),
686  (PVOID)&(OldEntry->KeyValue),
687  NewEntry->ValueKeySize);
688 
689  Cacheddatapointer = (PUCHAR) ((ULONG_PTR) &(NewEntry->KeyValue) + OldEntry->ValueKeySize);
690  RtlCopyMemory(Cacheddatapointer, *DataPointer, DataSize);
691 
692  // Trying to catch the BAD guy who writes over our pool.
693  CmpMakeSpecialPoolReadWrite( OldEntry );
694 
695  *ContainingList = (PCM_CACHED_VALUE) CMP_MARK_CELL_CACHED(NewEntry);
696 
697  // Trying to catch the BAD guy who writes over our pool.
698  CmpMakeSpecialPoolReadOnly( NewEntry );
699 
700  //
701  // Free the old entry
702  //
703  ExFreePool(OldEntry);
704 
705  }
706  }
707  }
708  } else {
709  if( CmpGetValueData(Hive,&(ValueKey->u.KeyValue),&DataSize,DataPointer,Allocated,CellToRelease) == FALSE ) {
710  //
711  // insufficient resources; return NULL
712  //
713  ASSERT( *Allocated == FALSE );
714  ASSERT( *DataPointer == NULL );
715  return SearchFail;
716  }
717  }
718 
719  return SearchSuccess;
720 }
