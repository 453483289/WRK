47 
48 Routine Description:
49 
50  This function locates a VirtualAddress within the image header
51  of a file that is mapped as a file and returns a pointer to the
52  section table entry for that virtual address
53 
54 Arguments:
55 
56  NtHeaders - Supplies the pointer to the image or data file.
57 
58  Base - Supplies the base of the image or data file.
59 
60  Address - Supplies the virtual address to locate.
61 
62 Return Value:
63 
64  NULL - The file does not contain data for the specified directory entry.
65 
66  NON-NULL - Returns the pointer of the section entry containing the data.
67 
68 --*/
69 
70 {
71  ULONG i;
72  PIMAGE_SECTION_HEADER NtSection;
73 
74  NtSection = IMAGE_FIRST_SECTION( NtHeaders );
75  for (i=0; i<NtHeaders->FileHeader.NumberOfSections; i++) {
76  if ((ULONG)Address >= NtSection->VirtualAddress &&
77  (ULONG)Address < NtSection->VirtualAddress + NtSection->SizeOfRawData
78  ) {
79  return NtSection;
80  }
81  ++NtSection;
82  }
83 
84  return NULL;
85 }

211 /*++
212 
213 Routine Description:
214 
215  This function locates a Directory Entry within the image header
216  and returns either the virtual address or seek address of the
217  data the Directory describes.
218 
219 Arguments:
220 
221  Base - Supplies the base of the image or data file.
222 
223  MappedAsImage - FALSE if the file is mapped as a data file.
224  - TRUE if the file is mapped as an image.
225 
226  DirectoryEntry - Supplies the directory entry to locate.
227 
228  Size - Return the size of the directory.
229 
230 Return Value:
231 
232  NULL - The file does not contain data for the specified directory entry.
233 
234  NON-NULL - Returns the address of the raw data the directory describes.
235 
236 --*/
237 
238 {
239  PIMAGE_NT_HEADERS NtHeaders;
240 
241  if (LDR_IS_DATAFILE(Base)) {
242  Base = LDR_DATAFILE_TO_VIEW(Base);
243  MappedAsImage = FALSE;
244  }
245 
246  NtHeaders = RtlImageNtHeader(Base);
247 
248  if (!NtHeaders)
249  return NULL;
250 
251  if (NtHeaders->OptionalHeader.Magic == IMAGE_NT_OPTIONAL_HDR32_MAGIC) {
252  return (RtlpImageDirectoryEntryToData32(Base,
253  MappedAsImage,
254  DirectoryEntry,
255  Size,
256  (PIMAGE_NT_HEADERS32)NtHeaders));
257  } else if (NtHeaders->OptionalHeader.Magic == IMAGE_NT_OPTIONAL_HDR64_MAGIC) {
258  return (RtlpImageDirectoryEntryToData64(Base,
259  MappedAsImage,
260  DirectoryEntry,
261  Size,
262  (PIMAGE_NT_HEADERS64)NtHeaders));
263  } else {
264  return (NULL);

269 VOID
270 RtlMakeStackTraceDataPresent(
271  VOID
272  )
273 
274 /*++
275 
276 Routine Description:
277 
278  This function walks the loaded user-mode images and makes present the
279  portions of the data necessary to support a kernel-debugger stack
280  dump of the user-mode stack.
281 
282  N.B. The necessary pieces are merely touched to be brought in. In a low
283  memory situation, some of the previously-touched pages may be paged
284  out in order for the new ones to be brought in. This routine is
285  not guaranteed to do anything useful, however it does more often than
286  not.
287 
288 Arguments:
289 
290  None.
291 
292 Return value:
293 
294  None.
295 
296 --*/
297 
298 {
299  PPEB peb;
300  PLIST_ENTRY head;
301  PLIST_ENTRY next;
302  PVOID imageBase;
303  ULONG imageCount;
304  LDR_DATA_TABLE_ENTRY UNALIGNED *ldrDataTableEntry;
305 
306  RTL_PAGED_CODE();
307 
308  //
309  // The image list is in user mode and is not to be trusted. The
310  // surrounding try/except block will guard against most forms of
311  // list corruption. imageCount is used to bail in finite time
312  // in the event of a cyclic image list.
313  //
314 
315  imageCount = 0;
316  try {
317 
318  peb = NtCurrentPeb();
319  head = &peb->Ldr->InLoadOrderModuleList;
320 
321  ProbeForReadSmallStructure( head,
322  sizeof(LIST_ENTRY),
323  PROBE_ALIGNMENT(LIST_ENTRY) );
324 
325  next = head;
326  while (imageCount < 1000) {
327 
328  next = next->Flink;
329  if (next == head) {
330  break;
331  }
332  imageCount += 1;
333 
334  //
335  // Locate the base address of the image
336  //
337 
338  ldrDataTableEntry = CONTAINING_RECORD(next,
339  LDR_DATA_TABLE_ENTRY,
340  InLoadOrderLinks);
341 
342  ProbeForReadSmallStructure( ldrDataTableEntry,
343  sizeof(LDR_DATA_TABLE_ENTRY),
344  PROBE_ALIGNMENT(LDR_DATA_TABLE_ENTRY) );
345 
346  imageBase = ldrDataTableEntry->DllBase;
347  ProbeForReadSmallStructure (imageBase, sizeof (IMAGE_DOS_HEADER), sizeof (UCHAR));
348 
349  //
350  // Make the stack trace data present for this image. Use a
351  // separate try/except block here so that subsequent images
352  // will be processed in the event of a failure.
353  //
354 
355  try {
356  RtlpMakeStackTraceDataPresentForImage(imageBase);
357  } except (EXCEPTION_EXECUTE_HANDLER) {
358  NOTHING;
359  }
360  }
361 
362  } except (EXCEPTION_EXECUTE_HANDLER) {
363  NOTHING;
364  }
365 }
