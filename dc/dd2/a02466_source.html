40 Routine Description:

42  This function initializes a kernel semaphore object. The initial

44 
45 Arguments:

55

67  //
68 
69  Semaphore->Header.Type = SemaphoreObject;
70  Semaphore->Header.Size = sizeof(KSEMAPHORE) / sizeof(LONG);

118 /*++
119 
120 Routine Description:
121 
122  This function releases a semaphore by adding the specified adjustment
123  value to the current semaphore count and attempts to satisfy as many
124  Waits as possible. The previous signal state of the semaphore object
125  is returned as the function value.
126 
127 Arguments:
128 
129  Semaphore - Supplies a pointer to a dispatcher object of type
130  semaphore.
131 
132  Increment - Supplies the priority increment that is to be applied
133  if releasing the semaphore causes a Wait to be satisfied.
134 
135  Adjustment - Supplies value that is to be added to the current
136  semaphore count.
137 
138  Wait - Supplies a boolean value that signifies whether the call to
139  KeReleaseSemaphore will be immediately followed by a call to one
140  of the kernel Wait functions.
141 
142 Return Value:
143 
144  The previous signal state of the semaphore object.
145 
146 --*/
147 
148 {
149 
150  LONG NewState;
151  KIRQL OldIrql;
152  LONG OldState;
153  PRKTHREAD Thread;
154 
155  ASSERT_SEMAPHORE( Semaphore );
156  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
157 
158  //
159  // Raise IRQL to dispatcher level and lock dispatcher database.
160  //
161 
162  KiLockDispatcherDatabase(&OldIrql);
163 
164  //
165  // Capture the current signal state of the semaphore object and
166  // compute the new count value.
167  //
168 
169  OldState = ReadForWriteAccess(&Semaphore->Header.SignalState);
170  NewState = OldState + Adjustment;
171 
172  //
173  // If the new state value is greater than the limit or a carry occurs,
174  // then unlock the dispatcher database, and raise an exception.
175  //
176 
177  if ((NewState > Semaphore->Limit) || (NewState < OldState)) {
178  KiUnlockDispatcherDatabase(OldIrql);
179  ExRaiseStatus(STATUS_SEMAPHORE_LIMIT_EXCEEDED);
180  }
181 
182  //
183  // Set the new signal state of the semaphore object and set the wait
184  // next value. If the previous signal state was Not-Signaled (i.e.
185  // the count was zero), and the wait queue is not empty, then attempt
186  // to satisfy as many Waits as possible.
187  //
188 
189  Semaphore->Header.SignalState = NewState;
190  if ((OldState == 0) && (IsListEmpty(&Semaphore->Header.WaitListHead) == FALSE)) {
191  KiWaitTest(Semaphore, Increment);
192  }
193 
194  //
195  // If the value of the Wait argument is TRUE, then return to the
196  // caller with IRQL raised and the dispatcher database locked. Else
197  // release the dispatcher database lock and lower IRQL to its
198  // previous value.
199  //
200 
201  if (Wait != FALSE) {
202  Thread = KeGetCurrentThread();
203  Thread->WaitNext = Wait;
204  Thread->WaitIrql = OldIrql;
205 
206  } else {
207  KiUnlockDispatcherDatabase(OldIrql);
208  }
209 
210  //
211  // Return previous signal state of semaphore object.
212  //
213 
214  return OldState;
215 }
