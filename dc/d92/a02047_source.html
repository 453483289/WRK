80  IN PLOADER_PARAMETER_BLOCK LoaderBlock

82 /*++
83 
84 Routine Description:
85 
86  This routine creates \\Registry\Machine\Hardware node in
87  the registry and calls SetupTree routine to put the hardware
88  information to the registry.
89 
90 Arguments:
91 
92  LoaderBlock - supplies a pointer to the LoaderBlock passed in from the
93  OS Loader.
94 
95 Returns:
96 
97  NTSTATUS code for success or reason of failure.
98 
99 --*/
100 {
101  NTSTATUS Status;
102  OBJECT_ATTRIBUTES ObjectAttributes;
103  HANDLE BaseHandle;
104  PCONFIGURATION_COMPONENT_DATA ConfigurationRoot;
105  ULONG Disposition;
106 
107  ConfigurationRoot = (PCONFIGURATION_COMPONENT_DATA)LoaderBlock->ConfigurationRoot;
108 
109  //
110  // Create \\Registry\Machine\Hardware\DeviceMap
111  //
112 
113  InitializeObjectAttributes(
114  &ObjectAttributes,
115  &CmRegistryMachineHardwareDeviceMapName,
116  0,
117  (HANDLE)NULL,
118  NULL
119  );
120  ObjectAttributes.Attributes |= OBJ_CASE_INSENSITIVE;
121 
122  Status = NtCreateKey( // Paht may already exist
123  &BaseHandle,
124  KEY_READ | KEY_WRITE,
125  &ObjectAttributes,
126  TITLE_INDEX_VALUE,
127  NULL,
128  0,
129  &Disposition
130  );
131 
132  if (!NT_SUCCESS(Status)) {
133  return(Status);
134  }
135 
136  NtClose(BaseHandle);
137 
138  ASSERT(Disposition == REG_CREATED_NEW_KEY);
139 
140  //
141  // Create \\Registry\Machine\Hardware\Description and use the
142  // returned handle as the BaseHandle to build the hardware tree.
143  //
144 
145  InitializeObjectAttributes(
146  &ObjectAttributes,
147  &CmRegistryMachineHardwareDescriptionName,
148  0,
149  (HANDLE)NULL,
150  NULL
151  );
152  ObjectAttributes.Attributes |= OBJ_CASE_INSENSITIVE;
153 
154  Status = NtCreateKey( // Path may already exist
155  &BaseHandle,
156  KEY_READ | KEY_WRITE,
157  &ObjectAttributes,
158  TITLE_INDEX_VALUE,
159  NULL,
160  0,
161  &Disposition
162  );
163 
164  if (!NT_SUCCESS(Status)) {
165  return(Status);
166  }
167 
168  ASSERT(Disposition == REG_CREATED_NEW_KEY);
169 
170  //
171  // Allocate 16K bytes memory from paged pool for constructing
172  // configuration data for controller component.
173  // NOTE: The configuration Data for controller component
174  // usually takes less than 100 bytes. But on EISA machine, the

183  );

192  //
193 
194  if (ConfigurationRoot) {
195  Status = CmpSetupConfigurationTree(ConfigurationRoot,
196  BaseHandle,
197  -1,
198  (ULONG)-1);
199  } else {
200  Status = STATUS_SUCCESS;
201  }
202 
203  ExFreePool((PVOID)CmpConfigurationData);
204  NtClose(BaseHandle);

208 NTSTATUS
209 CmpSetupConfigurationTree(
210  IN PCONFIGURATION_COMPONENT_DATA CurrentEntry,
211  IN HANDLE ParentHandle,
212  IN INTERFACE_TYPE InterfaceType,
213  IN ULONG BusNumber
214  )
215 /*++
216 
217 Routine Description:
218 
219  This routine traverses loader configuration tree and register
220  the hardware information to the registry data base.
221 
222  Note to reduce the stack usage on machines with large number of PCI buses,
223  we do not recursively process the sibling nodes. We only recursively
224  process the child trees.
225 
226 Arguments:
227 
228  CurrentEntry - Supplies a pointer to a loader configuration
229  tree or subtree.
230 
231  ParentHandle - Supplies the parent handle of CurrentEntry node.
232 
233  InterfaceType - Specify the Interface type of the bus that the
234  CurrentEntry component resides.
235 
236  BusNumber - Specify the Bus Number of the bus that the CurrentEntry
237  component resides. If Bus number is -1, it means InterfaceType
238  and BusNumber are meaningless for this component.
239 
240 Returns:
241 
242  None.
243 
244 --*/
245 {
246  NTSTATUS Status;
247  HANDLE NewHandle;
248  USHORT i;
249  CONFIGURATION_COMPONENT *Component;
250  INTERFACE_TYPE LocalInterfaceType = InterfaceType;
251  ULONG LocalBusNumber = BusNumber;
252  USHORT DeviceIndexTable[NUMBER_TYPES];
253 
254  for (i = 0; i < NUMBER_TYPES; i++) {
255  DeviceIndexTable[i] = 0;
256  }
257 
258  //
259  // Process current entry and its siblings
260  //
261 
262  while (CurrentEntry) {
263 
264  //
265  // Register current entry first before going down to its children
266  //
267 
268  Component = &CurrentEntry->ComponentEntry;
269 
270  //
271  // If the current component is a bus component, we will set up
272  // its bus number and Interface type and use them to initialize
273  // its subtree.
274  //
275 
276  if (Component->Class == AdapterClass &&
277  CurrentEntry->Parent->ComponentEntry.Class == SystemClass) {
278 
279  switch (Component->Type) {
280 
281  case EisaAdapter:
282  LocalInterfaceType = Eisa;
283  LocalBusNumber = CmpTypeCount[EISA_ADAPTER_INDEX]++;
284  break;
285  case TcAdapter:
286  LocalInterfaceType = TurboChannel;
287  LocalBusNumber = CmpTypeCount[TURBOCHANNEL_ADAPTER_INDEX]++;
288  break;
289  case MultiFunctionAdapter:
290 
291  //
292  // Here we try to distinguish if the Multifunction adapter is
293  // Isa, Mca, Internal bus and assign BusNumber based on
294  // its interface type (bus type.)
295  //
296 
297  if (Component->Identifier) {
298  for (i=0; CmpMultifunctionTypes[i].AscString; i++) {
299  if (_stricmp((PCHAR)CmpMultifunctionTypes[i].AscString,
300  Component->Identifier) == 0) {
301  break;
302  }
303  }
304 
305  LocalInterfaceType = CmpMultifunctionTypes[i].InterfaceType;
306  LocalBusNumber = CmpMultifunctionTypes[i].Count++;
307  }
308  break;
309 
310  case ScsiAdapter:
311 
312  //
313  // Set the bus type to internal.
314  //
315 
316  LocalInterfaceType = Internal;
317  LocalBusNumber = CmpTypeCount[ScsiAdapter]++;
318  break;
319 
320  default:
321  LocalInterfaceType = -1;
322  LocalBusNumber = CmpUnknownBusCount++;
323  break;
324  }
325  }
326 
327  //
328  // Initialize and copy current component to hardware registry
329  //
330 
331  Status = CmpInitializeRegistryNode(
332  CurrentEntry,
333  ParentHandle,
334  &NewHandle,
335  LocalInterfaceType,
336  LocalBusNumber,
337  DeviceIndexTable
338  );
339 
340  if (!NT_SUCCESS(Status)) {
341  return(Status);
342  }
343 
344  //
345  // Once we are going one level down, we need to clear the TypeCount
346  // table for everything under the current component class ...
347  //
348 
349  if (CurrentEntry->Child) {
350 
351  //
352  // Process the child entry of current entry
353  //
354 
355  Status = CmpSetupConfigurationTree(CurrentEntry->Child,
356  NewHandle,
357  LocalInterfaceType,
358  LocalBusNumber
359  );
360  if (!NT_SUCCESS(Status)) {
361  NtClose(NewHandle);
362  return(Status);
363  }
364  }
365  NtClose(NewHandle);
366  CurrentEntry = CurrentEntry->Sibling;
367  }
368  return(STATUS_SUCCESS);
369 }

372 NTSTATUS
373 CmpInitializeRegistryNode(
374  IN PCONFIGURATION_COMPONENT_DATA CurrentEntry,
375  IN HANDLE ParentHandle,
376  OUT PHANDLE NewHandle,
377  IN INTERFACE_TYPE InterfaceType,
378  IN ULONG BusNumber,
379  IN PUSHORT DeviceIndexTable
380  )

382 /*++
383 
384 Routine Description:
385 
386  This routine creates a node for the current firmware component
387  and puts component data to the data part of the node.
388 
389 Arguments:
390 
391  CurrentEntry - Supplies a pointer to a configuration component.
392 
393  Handle - Supplies the parent handle of CurrentEntry node.
394 
395  NewHandle - Supplies a pointer to a HANDLE to receive the handle of
396  the newly created node.
397 
398  InterfaceType - Specify the Interface type of the bus that the
399  CurrentEntry component resides. (See BusNumber also)
400 
401  BusNumber - Specify the Bus Number of the bus that the CurrentEntry
402  component resides on. If Bus number is -1, it means InterfaceType
403  and BusNumber are meaningless for this component.
404 
405 Returns:
406 
407  None.
408 
409 --*/
410 {
411 
412  NTSTATUS Status;
413  OBJECT_ATTRIBUTES ObjectAttributes;
414  UNICODE_STRING KeyName;
415  UNICODE_STRING ValueName;
416  UNICODE_STRING ValueData;
417  HANDLE Handle;
418  HANDLE OldHandle;
419  ANSI_STRING AnsiString;
420  CHAR Buffer[12];
421  WCHAR UnicodeBuffer[12];
422  CONFIGURATION_COMPONENT *Component;
423  ULONG Disposition;
424  ULONG ConfigurationDataLength = 0;
425  PCM_FULL_RESOURCE_DESCRIPTOR NewArea;
426 
427  Component = &CurrentEntry->ComponentEntry;
428 
429  //
430  // If the component class is SystemClass, we set its Type to be
431  // ArcSystem. The reason is because the detection code sets
432  // its type to MaximumType to indicate it is NOT ARC compatible.
433  // Here, we are only interested in building a System Node. So we
434  // change its Type to ArcSystem to ease the setup.
435  //
436 
437  if (Component->Class == SystemClass) {
438  Component->Type = ArcSystem;
439  }
440 
441  //
442  // Create a new key to describe the Component.
443  //
444  // The type of the component will be used as the keyname of the
445  // registry node. The class is the class of the component.
446  //
447 
448  InitializeObjectAttributes(
449  &ObjectAttributes,
450  &(CmTypeName[Component->Type]),
451  0,
452  ParentHandle,
453  NULL
454  );
455  ObjectAttributes.Attributes |= OBJ_CASE_INSENSITIVE;
456 
457  Status = NtCreateKey( // Paht may already exist
458  &Handle,
459  KEY_READ | KEY_WRITE,
460  &ObjectAttributes,
461  0,
462  NULL,
463  0,
464  &Disposition
465  );
466 
467  if (!NT_SUCCESS(Status)) {
468  return(Status);
469  }
470 
471  //
472  // If this component is NOT a SystemClass component, we will
473  // create a subkey to identify the component's ordering.
474  //
475 
476  if (Component->Class != SystemClass) {
477 
478  RtlIntegerToChar(
479  DeviceIndexTable[Component->Type]++,
480  10,
481  12,
482  Buffer
483  );
484 
485  RtlInitAnsiString(
486  &AnsiString,
487  Buffer
488  );
489 
490  KeyName.Buffer = (PWSTR)UnicodeBuffer;
491  KeyName.Length = 0;
492  KeyName.MaximumLength = sizeof(UnicodeBuffer);
493 
494  RtlAnsiStringToUnicodeString(
495  &KeyName,
496  &AnsiString,
497  FALSE
498  );
499 
500  OldHandle = Handle;
501 
502  InitializeObjectAttributes(
503  &ObjectAttributes,
504  &KeyName,
505  0,
506  OldHandle,
507  NULL
508  );
509  ObjectAttributes.Attributes |= OBJ_CASE_INSENSITIVE;
510 
511  Status = NtCreateKey(
512  &Handle,
513  KEY_READ | KEY_WRITE,
514  &ObjectAttributes,
515  0,
516  NULL,
517  0,
518  &Disposition
519  );
520 
521  NtClose(OldHandle);
522 
523  if (!NT_SUCCESS(Status)) {
524  return(Status);
525  }
526 
527  ASSERT(Disposition == REG_CREATED_NEW_KEY);
528  }
529 
530  //
531  // Create a value which describes the following component information:
532  // Flags, Cersion, Key, AffinityMask.
533  //
534 
535  RtlInitUnicodeString(
536  &ValueName,
537  L"Component Information"
538  );
539 
540  Status = NtSetValueKey(
541  Handle,
542  &ValueName,
543  TITLE_INDEX_VALUE,
544  REG_BINARY,
545  &Component->Flags,
546  FIELD_OFFSET(CONFIGURATION_COMPONENT, ConfigurationDataLength) -
547  FIELD_OFFSET(CONFIGURATION_COMPONENT, Flags)
548  );
549 
550  if (!NT_SUCCESS(Status)) {
551  NtClose(Handle);
552  return(Status);
553  }
554 
555  //
556  // Create a value which describes the component identifier, if any.
557  //
558 
559  if (Component->IdentifierLength) {
560 
561  RtlInitUnicodeString(
562  &ValueName,
563  L"Identifier"
564  );
565 
566  RtlInitAnsiString(
567  &AnsiString,
568  Component->Identifier
569  );
570 
571  Status = RtlAnsiStringToUnicodeString(
572  &ValueData,
573  &AnsiString,
574  TRUE
575  );
576 
577  if( NT_SUCCESS(Status) ) {
578  Status = NtSetValueKey(
579  Handle,
580  &ValueName,
581  TITLE_INDEX_VALUE,
582  REG_SZ,
583  ValueData.Buffer,
584  ValueData.Length + sizeof( UNICODE_NULL )
585  );
586 
587  RtlFreeUnicodeString(&ValueData);
588  }
589 
590  if (!NT_SUCCESS(Status)) {
591  NtClose(Handle);
592  return(Status);
593  }
594  }
595 
596  //
597  // Create a value entry for component configuration data.
598  //
599 
600  RtlInitUnicodeString(
601  &ValueName,
602  L"Configuration Data"
603  );
604 
605  //
606  // Create the configuration data based on CM_FULL_RESOURCE_DESCRIPTOR.
607  //
608  // Note the configuration data in firmware tree may be in the form of
609  // CM_PARTIAL_RESOURCE_LIST or nothing. In both cases, we need to
610  // set up the registry configuration data to be in the form of
611  // CM_FULL_RESOURCE_DESCRIPTOR.
612  //
613 
614  if (CurrentEntry->ConfigurationData) {
615 
616  //
617  // This component has configuration data, we copy the data
618  // to our work area, add some more data items and copy the new
619  // configuration data to the registry.
620  //
621 
622  ConfigurationDataLength = Component->ConfigurationDataLength +
623  FIELD_OFFSET(CM_FULL_RESOURCE_DESCRIPTOR,
624  PartialResourceList);
625 
626  //
627  // Make sure our reserved area is big enough to hold the data.
628  //
629 
630  if (ConfigurationDataLength > CmpConfigurationAreaSize) {
631 
632  //
633  // If reserved area is not big enough, we resize our reserved
634  // area. If, unfortunately, the reallocation fails, we simply
635  // loss the configuration data of this particular component.
636  //
637 
638  NewArea = (PCM_FULL_RESOURCE_DESCRIPTOR)ExAllocatePool(
639  PagedPool,
640  ConfigurationDataLength
641  );
642 
643  if (NewArea) {
644  CmpConfigurationAreaSize = ConfigurationDataLength;
645  ExFreePool(CmpConfigurationData);
646  CmpConfigurationData = NewArea;
647  RtlCopyMemory(
648  (PUCHAR)&CmpConfigurationData->PartialResourceList.Version,
649  CurrentEntry->ConfigurationData,
650  Component->ConfigurationDataLength
651  );
652  } else {
653  Component->ConfigurationDataLength = 0;
654  CurrentEntry->ConfigurationData = NULL;
655  }
656  } else {
657  RtlCopyMemory(
658  (PUCHAR)&CmpConfigurationData->PartialResourceList.Version,
659  CurrentEntry->ConfigurationData,
660  Component->ConfigurationDataLength
661  );
662  }
663 
664  }
665 
666  if (CurrentEntry->ConfigurationData == NULL) {
667 
668  //
669  // This component has NO configuration data (or we can't resize
670  // our reserved area to hold the data), we simple add whatever
671  // is required to set up a CM_FULL_RESOURCE_LIST.
672  //
673 
674  CmpConfigurationData->PartialResourceList.Version = 0;
675  CmpConfigurationData->PartialResourceList.Revision = 0;
676  CmpConfigurationData->PartialResourceList.Count = 0;
677  ConfigurationDataLength = FIELD_OFFSET(CM_FULL_RESOURCE_DESCRIPTOR,
678  PartialResourceList) +
679  FIELD_OFFSET(CM_PARTIAL_RESOURCE_LIST,
680  PartialDescriptors);
681  }
682 
683  //
684  // Set up InterfaceType and BusNumber for the component.
685  //
686 
687  CmpConfigurationData->InterfaceType = InterfaceType;
688  CmpConfigurationData->BusNumber = BusNumber;
689 
690  //
691  // Write the newly constructed configuration data to the hardware registry
692  //
693 
694  Status = NtSetValueKey(
695  Handle,
696  &ValueName,
697  TITLE_INDEX_VALUE,
698  REG_FULL_RESOURCE_DESCRIPTOR,
699  CmpConfigurationData,
700  ConfigurationDataLength
701  );
702 
703  if (!NT_SUCCESS(Status)) {
704  NtClose(Handle);
705  return(Status);
706  }
707 
708  *NewHandle = Handle;
709  return(STATUS_SUCCESS);
710 }
