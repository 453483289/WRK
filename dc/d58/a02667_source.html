130 //
131 // Resource used to control access to stack trace database. We opted for
132 // this solution so that we kept change in the database structure to an
133 // absolute minimal. This way the tools that depend on this structure
134 // (at least umhd and oh) will not need a new version and will not
135 // introduce backcompatibility issues.
136 //
137 
138 KSPIN_LOCK_EX RtlpStackTraceDataBaseLock;
139 
143 
144 //
145 // The following section implements a trace database used to store
146 // stack traces captured with RtlCaptureStackBackTrace(). The database
147 // is implemented as a hash table and does not allow deletions. It is
148 // sensitive to "garbage" in the sense that spurious garbage (partially
149 // correct stacks) will hash in different buckets and will tend to fill
150 // the whole table. This is a problem only on x86 if "fuzzy" stack traces
151 // are used. The typical function used to log the trace is
152 // RtlLogStackBackTrace. One of the worst limitations of this package
153 // is that traces are referred using a ushort index which means we cannot
154 // ever store more than 65535 traces (remember we never delete traces).
155 //
156

211 NTSTATUS
212 RtlInitializeStackTraceDataBase(
213  IN PVOID CommitBase,
214  IN SIZE_T CommitSize,
215  IN SIZE_T ReserveSize
216  )
217 {
218  NTSTATUS Status;
219  PSTACK_TRACE_DATABASE DataBase;
220 
221  DataBase = (PSTACK_TRACE_DATABASE)CommitBase;
222 
223  if (CommitSize == 0) {
224 
225  //
226  // Initially commit enough pages to accommodate the increased
227  // number of hash chains (for improved performance we switched from ~100
228  // to ~1000 in the hope that the hash chains will decrease ten-fold in
229  // length).
230  //
231 
232  CommitSize = ROUND_TO_PAGES (NUMBER_OF_BUCKETS * sizeof (DataBase->Buckets[ 0 ]));
233 
234  Status = ZwAllocateVirtualMemory (NtCurrentProcess(),
235  (PVOID *)&CommitBase,
236  0,
237  &CommitSize,
238  MEM_COMMIT,
239  PAGE_READWRITE);
240 
241  if (! NT_SUCCESS(Status)) {
242 
243  KdPrint (("RTL: Unable to commit space to extend stack "
244  "trace data base - Status = %lx\n",
245  Status));
246  return Status;
247  }
248 
249  DataBase->PreCommitted = FALSE;
250  }
251  else if (CommitSize == ReserveSize) {
252 
253  RtlZeroMemory (DataBase, sizeof( *DataBase ));
254 
255  DataBase->PreCommitted = TRUE;
256  }
257  else {
258 
259  return STATUS_INVALID_PARAMETER;
260  }
261 
262  DataBase->CommitBase = CommitBase;
263  DataBase->NumberOfBuckets = NUMBER_OF_BUCKETS;
264  DataBase->NextFreeLowerMemory = (PCHAR)(&DataBase->Buckets[ DataBase->NumberOfBuckets ]);
265  DataBase->NextFreeUpperMemory = (PCHAR)CommitBase + ReserveSize;
266 
267  if (! DataBase->PreCommitted) {
268 
269  DataBase->CurrentLowerCommitLimit = (PCHAR)CommitBase + CommitSize;
270  DataBase->CurrentUpperCommitLimit = (PCHAR)CommitBase + ReserveSize;
271  }
272  else {
273 
274  RtlZeroMemory (&DataBase->Buckets[ 0 ],
275  DataBase->NumberOfBuckets * sizeof (DataBase->Buckets[ 0 ]));
276  }
277 
278  DataBase->EntryIndexArray = (PRTL_STACK_TRACE_ENTRY *)DataBase->NextFreeUpperMemory;
279 
280  //
281  // Initialize the database lock.
282  //
283 
284  DataBase->Lock = &RtlpStackTraceDataBaseLock;
285 
286  Status = INITIALIZE_DATABASE_LOCK (DataBase->Lock);
287 
288  if (! NT_SUCCESS(Status)) {
289 
290  KdPrint(("RTL: Unable to initialize stack trace database lock (status %X)\n", Status));
291  return Status;
292  }
293 
294  RtlpStackTraceDataBase = DataBase;
295 
296  return STATUS_SUCCESS;
297 }

300 PRTL_STACK_TRACE_ENTRY
301 RtlpExtendStackTraceDataBase(
302  IN PRTL_STACK_TRACE_ENTRY InitialValue,
303  IN SIZE_T Size
304  )
305 /*++
306 
307 Routine Description:
308 
309  This routine extends the stack trace database in order to accommodate
310  the new stack trace that has to be saved.
311 
312 Arguments:
313 
314  InitialValue - stack trace to be saved.
315 
316  Size - size of the stack trace in bytes. Note that this is not the
317  depth of the trace but rather `Depth * sizeof(PVOID)'.
318 
319 Return Value:
320 
321  The address of the just saved stack trace or null in case we have hit
322  the maximum size of the database or we get commit errors.
323 
324 Environment:
325 
326  User mode.
327 
328  Note. In order to make all this code work in kernel mode we have to
329  rewrite this function that relies on VirtualAlloc.
330 
331 --*/
332 
333 {
334  NTSTATUS Status;
335  PRTL_STACK_TRACE_ENTRY p, *pp;
336  SIZE_T CommitSize;
337  PSTACK_TRACE_DATABASE DataBase;
338 
339  DataBase = RtlpStackTraceDataBase;
340 
341  //
342  // We will try to find space for one stack trace entry in the
343  // upper part of the database.
344  //
345 
346  pp = (PRTL_STACK_TRACE_ENTRY *)DataBase->NextFreeUpperMemory;
347 
348  if ((! DataBase->PreCommitted) &&
349  ((PCHAR)(pp - 1) < (PCHAR)DataBase->CurrentUpperCommitLimit)) {
350 
351  //
352  // No more committed space in the upper part of the database.
353  // We need to extend it downwards.
354  //
355 
356  DataBase->CurrentUpperCommitLimit =
357  (PVOID)((PCHAR)DataBase->CurrentUpperCommitLimit - PAGE_SIZE);
358 
359  if (DataBase->CurrentUpperCommitLimit < DataBase->CurrentLowerCommitLimit) {
360 
361  //
362  // No more space at all. We have got over the lower part of the db.
363  // We failed therefore increase back the UpperCommitLimit pointer.
364  //
365 
366  DataBase->CurrentUpperCommitLimit =
367  (PVOID)((PCHAR)DataBase->CurrentUpperCommitLimit + PAGE_SIZE);
368 
369  return( NULL );
370  }
371 
372  CommitSize = PAGE_SIZE;
373  Status = ZwAllocateVirtualMemory(
374  NtCurrentProcess(),
375  (PVOID *)&DataBase->CurrentUpperCommitLimit,
376  0,
377  &CommitSize,
378  MEM_COMMIT,
379  PAGE_READWRITE
380  );
381 
382  if (!NT_SUCCESS( Status )) {
383 
384  //
385  // We tried to increase the upper part of the db by one page.
386  // We failed therefore increase back the UpperCommitLimit pointer
387  //
388 
389  DataBase->CurrentUpperCommitLimit =
390  (PVOID)((PCHAR)DataBase->CurrentUpperCommitLimit + PAGE_SIZE);
391 
392  return NULL;
393  }
394  }
395 
396  //
397  // We managed to make sure we have usable space in the upper part
398  // therefore we take out one stack trace entry address.
399  //
400 
401  DataBase->NextFreeUpperMemory -= sizeof( *pp );
402 
403  //
404  // Now we will try to find space in the lower part of the database for
405  // for the actual stack trace.
406  //
407 
408  p = (PRTL_STACK_TRACE_ENTRY)DataBase->NextFreeLowerMemory;
409 
410  if ((! DataBase->PreCommitted) &&
411  (((PCHAR)p + Size) > (PCHAR)DataBase->CurrentLowerCommitLimit)) {
412 
413  //
414  // We need to extend the lower part.
415  //
416 
417  if (DataBase->CurrentLowerCommitLimit >= DataBase->CurrentUpperCommitLimit) {
418 
419  //
420  // We have hit the maximum size of the database.
421  //
422 
423  DataBase->NextFreeUpperMemory += sizeof( *pp );
424  return( NULL );
425  }
426 
427  //
428  // Extend the lower part of the database by one page.
429  //
430 
431  CommitSize = Size;
432  Status = ZwAllocateVirtualMemory(
433  NtCurrentProcess(),
434  (PVOID *)&DataBase->CurrentLowerCommitLimit,
435  0,
436  &CommitSize,
437  MEM_COMMIT,
438  PAGE_READWRITE
439  );
440 
441  if (! NT_SUCCESS( Status )) {
442 
443  DataBase->NextFreeUpperMemory += sizeof( *pp );
444  return NULL;
445  }
446 
447  DataBase->CurrentLowerCommitLimit =
448  (PCHAR)DataBase->CurrentLowerCommitLimit + CommitSize;
449  }
450 
451  //
452  // Take out the space for the stack trace.
453  //
454 
455  DataBase->NextFreeLowerMemory += Size;
456 
457  //
458  // Deal with a precommitted database case. If the lower and upper
459  // pointers have crossed each other then rollback and return failure.
460  //
461 
462  if (DataBase->PreCommitted &&
463  DataBase->NextFreeLowerMemory >= DataBase->NextFreeUpperMemory) {
464 
465  DataBase->NextFreeUpperMemory += sizeof( *pp );
466  DataBase->NextFreeLowerMemory -= Size;
467  return( NULL );
468  }
469 
470  //
471  // Save the stack trace in the database
472  //
473 
474  RtlMoveMemory( p, InitialValue, Size );
475  p->HashChain = NULL;
476  p->TraceCount = 0;
477  p->Index = (USHORT)(++DataBase->NumberOfEntriesAdded);
478 
479  //
480  // Save the address of the new stack trace entry in the
481  // upper part of the databse.
482  //
483 
484  *--pp = p;
485 
486  //
487  // Return address of the saved stack trace entry.
488  //
489 
490  return( p );
491 }

539 
540 Routine Description:
541 
542  This routine will capture the current stacktrace (skipping the
543  present function) and will save it in the global (per process)
544  stack trace database. It should be noted that we do not save
545  duplicate traces.
546 
547 Arguments:
548 
549  FramesToSkip - no of frames that are not interesting and
550  should be skipped.
551 
552 Return Value:
553 
554  Index of the stack trace saved. The index can be used by tools
555  to access quickly the trace data. This is the reason at the end of
556  the database we save downwards a list of pointers to trace entries.
557  This index can be used to find this pointer in constant time.
558 
559  A zero index will be returned for error conditions (e.g. stack
560  trace database not initialized).
561 
562 Environment:
563 
564  User mode.
565 
566 --*/
567 
568 {
569  RTL_STACK_TRACE_ENTRY Trace;
570  USHORT TraceIndex;
571  NTSTATUS Status;
572  ULONG Hash;
573  PSTACK_TRACE_DATABASE DataBase;
574 
575  //
576  // Check the context in which we are running.
577  //
578 
579  DataBase = RtlpStackTraceDataBase;
580 
581  if (DataBase == NULL) {
582  return 0;
583  }
584 
585  if (! OKAY_TO_LOCK_DATABASE (DataBase->Lock)) {
586  return 0;
587  }
588 
589  //
590  // Capture stack trace.
591  //
592 
593  if (RtlpCaptureStackTraceForLogging (&Trace, &Hash, FramesToSkip + 1, FALSE) == FALSE) {
594  return 0;
595  }
596

599  // Return trace index.
600  //
601

603 
604  return TraceIndex;

671 LOGICAL
672 RtlpCaptureStackTraceForLogging (
673  PRTL_STACK_TRACE_ENTRY Trace,
674  PULONG Hash,
675  ULONG FramesToSkip,
676  LOGICAL UserModeStackFromKernelMode
677  )
678 {
679  if (UserModeStackFromKernelMode == FALSE) {
680 
681  //
682  // Capture stack trace. The try/except was useful
683  // in the old days when the function did not validate
684  // the stack frame chain. We keep it just to be defensive.
685  //
686 
687  try {
688 
689  Trace->Depth = RtlCaptureStackBackTrace (FramesToSkip + 1,
690  MAX_STACK_DEPTH,
691  Trace->BackTrace,
692  Hash);
693  }
694  except(EXCEPTION_EXECUTE_HANDLER) {
695 
696  Trace->Depth = 0;
697  }
698 
699  if (Trace->Depth == 0) {
700 
701  return FALSE;
702  }
703  else {
704 
705  return TRUE;
706  }
707  }
708  else {
709 
710  ULONG Index;
711 
712  //
713  // Avoid weird situations.
714  //
715 
716  if (KeAreAllApcsDisabled () == TRUE) {
717  return FALSE;
718  }
719 
720  //
721  // Capture user mode stack trace and hash value.
722  //
723 
724  Trace->Depth = (USHORT) RtlWalkFrameChain(Trace->BackTrace,
725  MAX_STACK_DEPTH,
726  1);
727  if (Trace->Depth == 0) {
728 
729  return FALSE;
730  }
731  else {
732 
733  *Hash = 0;
734 
735  for (Index = 0; Index < Trace->Depth; Index += 1) {
736  *Hash += PtrToUlong (Trace->BackTrace[Index]);
737  }
738 
739  return TRUE;
740  }
741  }
742 }

745 USHORT
746 RtlpLogCapturedStackTrace(
747  PRTL_STACK_TRACE_ENTRY Trace,
748  ULONG Hash
749  )
750 {
751  PSTACK_TRACE_DATABASE DataBase;
752  PRTL_STACK_TRACE_ENTRY p, *pp;
753  ULONG RequestedSize, DepthSize;
754  USHORT ReturnValue;
755 
756  DataBase = RtlpStackTraceDataBase;
757 
758  //
759  // Update statistics counters. Since they are used only for reference and do not
760  // control decisions we increment them without protection even if this means we may
761  // have numbers slightly out of sync.
762  //
763 
764  DataBase->NumberOfEntriesLookedUp += 1;
765 
766  //
767  // Lock the global per-process stack trace database.
768  //
769 
770  if (RtlpAcquireStackTraceDataBase() == NULL) {
771 
772  //
773  // Fail the log operation if we cannot acquire the lock.
774  // This can happen only if there is a dump in progress or we are in
775  // an invalid context (process shutdown (Umode) or DPC routine (Kmode).
776  //
777 
778  return 0;
779  }
780 
781  try {
782 
783  //
784  // We will try to find out if the trace has been saved in the past.
785  // We find the right hash chain and then traverse it.
786  //
787 
788  DepthSize = Trace->Depth * sizeof (Trace->BackTrace[0]);
789 
790  pp = &DataBase->Buckets[ Hash % DataBase->NumberOfBuckets ];
791 
792  while (p = *pp) {
793 
794  if (p->Depth == Trace->Depth) {
795 
796  if (RtlCompareMemory( &p->BackTrace[ 0 ], &Trace->BackTrace[ 0 ], DepthSize) == DepthSize) {
797 
798  break;
799  }
800  }
801 
802  pp = &p->HashChain;
803  }
804 
805  if (p == NULL) {
806 
807  //
808  // If we get here we did not find a similar trace in the database. We need
809  // to add it.
810  //
811  // We got the `*pp' value (address of last chain element) while the
812  // database lock was acquired shared so we need to take into consideration
813  // the case where another thread managed to acquire database exclusively
814  // and add a new trace at the end of the chain. Therefore if `*pp' is no longer
815  // null we continue to traverse the chain until we get to the end.
816  //
817 
818  p = NULL;
819 
820  if (*pp != NULL) {
821 
822  //
823  // Somebody added some traces at the end of the chain while we
824  // were trying to convert the lock from shared to exclusive.
825  //
826 
827  while (p = *pp) {
828 
829  if (p->Depth == Trace->Depth) {
830 
831  if (RtlCompareMemory( &p->BackTrace[ 0 ], &Trace->BackTrace[ 0 ], DepthSize) == DepthSize) {
832 
833  break;
834  }
835  }
836 
837  pp = &p->HashChain;
838  }
839  }
840 
841  if (p == NULL) {
842 
843  //
844  // Nobody added the trace and now `*pp' really points to the end
845  // of the chain either because we traversed the rest of the chain
846  // or it was at the end anyway.
847  //
848 
849  RequestedSize = FIELD_OFFSET (RTL_STACK_TRACE_ENTRY, BackTrace) + DepthSize;
850 
851  p = RtlpExtendStackTraceDataBase (Trace, RequestedSize);
852 
853  if (p != NULL) {
854 
855  //
856  // We added the trace no chain it as the last element.
857  //
858 
859  *pp = p;
860  }
861  }
862  else {
863 
864  //
865  // Some other thread managed to add the same trace to the database
866  // while we were trying to acquire the lock exclusive. `p' has the
867  // address to the stack trace entry.
868  //
869  }
870  }
871  }
872  except(EXCEPTION_EXECUTE_HANDLER) {
873 
874  //
875  // We should never get here if the algorithm is correct.
876  //
877 
878  p = NULL;
879  }
880 
881  //
882  // Release locks and return. At this stage we may return zero (failure)
883  // if we did not manage to extend the database with a new trace (e.g. due to
884  // out of memory conditions).
885  //
886
