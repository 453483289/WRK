37 IopCsqCancelRoutine(
38  IN PDEVICE_OBJECT DeviceObject,
39  IN PIRP Irp
40  )
41 /*++
42 
43 Routine Description:
44 
45  This routine removes the IRP that's associated with a context from the queue.
46  It's expected that this routine will be called from a timer or DPC or other threads which complete an
47  IO. Note that the IRP associated with this context could already have been freed.
48 
49 Arguments:
50 
51  Csq - Pointer to the cancel queue.
52  Context - Context associated with Irp.
53 
54 
55 Return Value:
56 
57  Returns the IRP associated with the context. If the value is not NULL, the IRP was successfully
58  retrieved and can be used safely. If the value is NULL, the IRP was already canceled.
59 
60 --*/
61 {
62  KIRQL irql;
63  PIO_CSQ_IRP_CONTEXT irpContext;
64  PIO_CSQ cfq;
65 
66  UNREFERENCED_PARAMETER (DeviceObject);
67 
68  IoReleaseCancelSpinLock(Irp->CancelIrql);
69 
70  irpContext = Irp->Tail.Overlay.DriverContext[3];
71 
72  if (irpContext->Type == IO_TYPE_CSQ_IRP_CONTEXT) {
73  cfq = irpContext->Csq;
74  } else if ((irpContext->Type == IO_TYPE_CSQ) ||
75  (irpContext->Type == IO_TYPE_CSQ_EX)) {
76  cfq = (PIO_CSQ)irpContext;
77  } else {
78 
79  //
80  // Bad type
81  //
82 
83  ASSERT(0);
84  return;
85  }
86 
87  ASSERT(cfq);
88 
89  cfq->ReservePointer = NULL; // Force drivers to be good citizens
90 
91  cfq->CsqAcquireLock(cfq, &irql);
92  cfq->CsqRemoveIrp(cfq, Irp);
93 
94 
95  //
96  // Break the association if necessary.
97  //
98 
99  if (irpContext != (PIO_CSQ_IRP_CONTEXT)cfq) {
100  irpContext->Irp = NULL;
101 
102  Irp->Tail.Overlay.DriverContext[3] = NULL;
103  }
104  cfq->CsqReleaseLock(cfq, irql);
105 
106  cfq->CsqCompleteCanceledIrp(cfq, Irp);
107 }

189 NTSTATUS
190 CSQLIB_DDI(IoCsqInsertIrpEx)(
191  IN PIO_CSQ Csq,
192  IN PIRP Irp,
193  IN PIO_CSQ_IRP_CONTEXT Context,
194  IN PVOID InsertContext
195  )
196 /*++
197 
198 Routine Description:
199 
200  This routine inserts the IRP into the queue and associates the context with the IRP.
201  The context has to be in non-paged pool if the context will be used in a DPC or interrupt routine.
202  The routine assumes that Irp->Tail.Overlay.DriverContext[3] is available for use by the APIs.
203  It's ok to pass a NULL context if the driver assumes that it will always use IoCsqRemoveNextIrp to
204  remove an IRP.
205 
206 Arguments:
207 
208  Csq - Pointer to the cancel queue.
209  Irp - Irp to be inserted
210  Context - Context to be associated with Irp.
211  InsertContext - Context passed to the driver's insert IRP routine.
212 
213 
214 Return Value:
215 
216  NTSTATUS - Status returned by the driver's insert IRP routine. If the driver's insert IRP
217  routine returns an error status, the IRP is not inserted and the same status is returned
218  by this API. This allows drivers to implement startIo kind of functionality.
219 
220 --*/
221 {
222  KIRQL irql;
223  PDRIVER_CANCEL cancelRoutine;
224  PVOID originalDriverContext;
225  NTSTATUS status = STATUS_SUCCESS;
226 
227  //
228  // Set the association between the context and the IRP.
229  //
230 
231  if (Context) {
232  Irp->Tail.Overlay.DriverContext[3] = Context;
233  Context->Irp = Irp;
234  Context->Csq = Csq;
235  Context->Type = IO_TYPE_CSQ_IRP_CONTEXT;
236  } else {
237  Irp->Tail.Overlay.DriverContext[3] = Csq;
238  }
239 
240 
241  Csq->ReservePointer = NULL; // Force drivers to be good citizens
242 
243  originalDriverContext = Irp->Tail.Overlay.DriverContext[3];
244 
245  Csq->CsqAcquireLock(Csq, &irql);
246 
247 
248  //
249  // If the driver wants to fail the insert then do this.
250  //
251 
252  if (Csq->Type == IO_TYPE_CSQ_EX) {
253 
254  PIO_CSQ_INSERT_IRP_EX func;
255 
256  func = (PIO_CSQ_INSERT_IRP_EX)Csq->CsqInsertIrp;
257  status = func(Csq, Irp, InsertContext);
258 
259  if (!NT_SUCCESS(status)) {
260 
261  Csq->CsqReleaseLock(Csq, irql);
262 
263  return status;
264  }
265 
266  } else {
267  Csq->CsqInsertIrp(Csq, Irp);
268  }
269 
270  IoMarkIrpPending(Irp);
271 
272  cancelRoutine = IoSetCancelRoutine(Irp, IopCsqCancelRoutine);
273 
274 
275  ASSERT(!cancelRoutine);
276 
277  if (Irp->Cancel) {
278 
279  cancelRoutine = IoSetCancelRoutine(Irp, NULL);
280 
281  if (cancelRoutine) {
282 
283  Csq->CsqRemoveIrp(Csq, Irp);
284 
285  if (Context) {
286  Context->Irp = NULL;
287  }
288 
289  Irp->Tail.Overlay.DriverContext[3] = NULL;
290 
291 
292  Csq->CsqReleaseLock(Csq, irql);
293 
294  Csq->CsqCompleteCanceledIrp(Csq, Irp);
295 
296  } else {
297 
298  //
299  // The cancel routine beat us to it.
300  //
301 
302  Csq->CsqReleaseLock(Csq, irql);
303  }
304 
305  } else {
306 
307  Csq->CsqReleaseLock(Csq, irql);
308 
309  }
310  return status;
311 }

313 PIRP
314 CSQLIB_DDI(IoCsqRemoveNextIrp)(
315  IN PIO_CSQ Csq,
316  IN PVOID PeekContext
317  )
318 /*++
319 
320 Routine Description:
321 
322  This routine removes the next IRP from the queue. This routine will enumerate the queue
323  and return an IRP that's not canceled. If an IRP in the queue is canceled it goes to the next
324  IRP. If no IRP is available it returns a NULL. The IRP returned is safe and cannot be canceled.
325 
326 Arguments:
327 
328  Csq - Pointer to the cancel queue.
329 
330 
331 Return Value:
332 
333  Returns the IRP or NULL.
334 
335 --*/
336 {
337  KIRQL irql;
338  PIO_CSQ_IRP_CONTEXT context;
339  PDRIVER_CANCEL cancelRoutine;
340  PIRP irp;
341 
342 
343  irp = NULL;
344 
345  Csq->ReservePointer = NULL; // Force drivers to be good citizens
346  Csq->CsqAcquireLock(Csq, &irql);
347 
348  irp = Csq->CsqPeekNextIrp(Csq, NULL, PeekContext);
349 
350  while (1) {
351 
352  //
353  // This routine will return a pointer to the next IRP in the queue adjacent to
354  // the irp passed as a parameter. If the irp is NULL, it returns the IRP at the head of
355  // the queue.
356  //
357 
358  if (!irp) {
359  Csq->CsqReleaseLock(Csq, irql);
360  return NULL;
361  }
362 
363  cancelRoutine = IoSetCancelRoutine(irp, NULL);
364  if (!cancelRoutine) {
365  irp = Csq->CsqPeekNextIrp(Csq, irp, PeekContext);
366  continue;
367  }
368 
369  Csq->CsqRemoveIrp(Csq, irp); // Remove this IRP from the queue
370 
371  break;
372  }
373 
374  context = irp->Tail.Overlay.DriverContext[3];
375  if (context->Type == IO_TYPE_CSQ_IRP_CONTEXT) {
376  context->Irp = NULL;
377  ASSERT(context->Csq == Csq);
378  }
379 
380  irp->Tail.Overlay.DriverContext[3] = NULL;
381 
382 
383  Csq->CsqReleaseLock(Csq, irql);
384 
385  return irp;
386 }

389 CSQLIB_DDI(IoCsqRemoveIrp)(
390  IN PIO_CSQ Csq,
391  IN PIO_CSQ_IRP_CONTEXT Context
392  )
393 /*++
394 
395 Routine Description:
396 
397  This routine removes the IRP that's associated with a context from the queue.
398  It's expected that this routine will be called from a timer or DPC or other threads which complete an
399  IO. Note that the IRP associated with this context could already have been freed.
400 
401 Arguments:
402 
403  Csq - Pointer to the cancel queue.
404  Context - Context associated with Irp.
405 
406 
407 Return Value:
408 
409  Returns the IRP associated with the context. If the value is not NULL, the IRP was successfully
410  retrieved and can be used safely. If the value is NULL, the IRP was already canceled.
411 
412 --*/
413 {
414  KIRQL irql;
415  PIRP irp;
416  PDRIVER_CANCEL cancelRoutine;
417 
418  Csq->ReservePointer = NULL; // Force drivers to be good citizens
419 
420  Csq->CsqAcquireLock(Csq, &irql);
421 
422  if (Context->Irp ) {
423 
424  ASSERT(Context->Csq == Csq);
425 
426  irp = Context->Irp;
427 
428 
429  cancelRoutine = IoSetCancelRoutine(irp, NULL);
430  if (!cancelRoutine) {
431  Csq->CsqReleaseLock(Csq, irql);
432  return NULL;
433  }
434 
435  ASSERT(Context == irp->Tail.Overlay.DriverContext[3]);
436 
437  Csq->CsqRemoveIrp(Csq, irp);
438 
439  //
440  // Break the association.
441  //
442 
443  Context->Irp = NULL;
444  irp->Tail.Overlay.DriverContext[3] = NULL;
445 
446  ASSERT(Context->Csq == Csq);
447 
448  Csq->CsqReleaseLock(Csq, irql);
449 
450  return irp;
451 
452  } else {
453 
454  Csq->CsqReleaseLock(Csq, irql);
455  return NULL;
456  }
457 }
