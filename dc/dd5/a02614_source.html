77 BOOLEAN
78 RtlIsValidHandler (
79  IN PEXCEPTION_ROUTINE Handler
80  )
81 {
82  PULONG FunctionTable;
83  ULONG FunctionTableLength;
84  PVOID Base;
85 
86  FunctionTable = RtlLookupFunctionTable(Handler, &Base, &FunctionTableLength);
87 
88  if (FunctionTable && FunctionTableLength) {
89  PEXCEPTION_ROUTINE FunctionEntry;
90  LONG High, Middle, Low;
91 
92  if ((FunctionTable == LongToPtr(-1)) && (FunctionTableLength == (ULONG)-1)) {
93  // Address is in an image that shouldn't have any handlers (like a resource only dll).
94  RtlInvalidHandlerDetected((PVOID)((ULONG)Handler+(ULONG)Base), LongToPtr(-1), -1);
95  return FALSE;
96  }
97 
98  // Bias the handler value down by the image base and see if the result
99  // is in the table
100 
101  (ULONG)Handler -= (ULONG)Base;
102  Low = 0;
103  High = FunctionTableLength;
104  while (High >= Low) {
105  Middle = (Low + High) >> 1;
106  FunctionEntry = (PEXCEPTION_ROUTINE)FunctionTable[Middle];
107  if (Handler < FunctionEntry) {
108  High = Middle - 1;
109  } else if (Handler > FunctionEntry) {
110  Low = Middle + 1;
111  } else {
112  // found it
113  return TRUE;
114  }
115  }
116  // Didn't find it
117  RtlInvalidHandlerDetected((PVOID)((ULONG)Handler+(ULONG)Base), FunctionTable, FunctionTableLength);
118 
119  return FALSE;
120  }
121 
122  // Can't verify
123  return TRUE;
124 }

127 RtlDispatchException (
128  IN PEXCEPTION_RECORD ExceptionRecord,
129  IN PCONTEXT ContextRecord
130  )

132 /*++
133 
134 Routine Description:
135 
136  This function attempts to dispatch an exception to a call frame based
137  handler by searching backwards through the stack based call frames. The
138  search begins with the frame specified in the context record and continues
139  backward until either a handler is found that handles the exception, the
140  stack is found to be invalid (i.e., out of limits or unaligned), or the end
141  of the call hierarchy is reached.
142 
143 Arguments:
144 
145  ExceptionRecord - Supplies a pointer to an exception record.
146 
147  ContextRecord - Supplies a pointer to a context record.
148 
149 Return Value:
150 
151  If the exception is handled by one of the frame based handlers, then
152  a value of TRUE is returned. Otherwise a value of FALSE is returned.
153 
154 --*/
155 
156 {
157 
158  BOOLEAN Completion = FALSE;
159  DISPATCHER_CONTEXT DispatcherContext;
160  EXCEPTION_DISPOSITION Disposition;
161  PEXCEPTION_REGISTRATION_RECORD RegistrationPointer;
162  PEXCEPTION_REGISTRATION_RECORD NestedRegistration;
163  ULONG HighAddress;
164  ULONG HighLimit;
165  ULONG LowLimit;
166  EXCEPTION_RECORD ExceptionRecord1;
167  ULONG Index;
168 
169  //
170  // Get current stack limits.
171  //
172 
173  RtlpGetStackLimits(&LowLimit, &HighLimit);
174 
175  //
176  // Start with the frame specified by the context record and search
177  // backwards through the call frame hierarchy attempting to find an
178  // exception handler that will handler the exception.
179  //
180 
181  RegistrationPointer = RtlpGetRegistrationHead();
182  NestedRegistration = 0;
183 
184  while (RegistrationPointer != EXCEPTION_CHAIN_END) {
185 
186  //
187  // If the call frame is not within the specified stack limits or the
188  // call frame is unaligned, then set the stack invalid flag in the
189  // exception record and return FALSE. Else check to determine if the
190  // frame has an exception handler.
191  //
192 
193  HighAddress = (ULONG)RegistrationPointer +
194  sizeof(EXCEPTION_REGISTRATION_RECORD);
195 
196  if ( ((ULONG)RegistrationPointer < LowLimit) ||
197  (HighAddress > HighLimit) ||
198  (((ULONG)RegistrationPointer & 0x3) != 0)
199  ) {
200 
201  //
202  // Allow for the possibility that the problem occured on the
203  // DPC stack.
204  //
205 
206  ULONG TestAddress = (ULONG)RegistrationPointer;
207 
208  if (((TestAddress & 0x3) == 0) &&
209  KeGetCurrentIrql() >= DISPATCH_LEVEL) {
210 
211  PKPRCB Prcb = KeGetCurrentPrcb();
212  ULONG DpcStack = (ULONG)Prcb->DpcStack;
213 
214  if ((Prcb->DpcRoutineActive) &&
215  (HighAddress <= DpcStack) &&
216  (TestAddress >= DpcStack - KERNEL_STACK_SIZE)) {
217 
218  //
219  // This error occured on the DPC stack, switch
220  // stack limits to the DPC stack and restart
221  // the loop.
222  //
223 
224  HighLimit = DpcStack;
225  LowLimit = DpcStack - KERNEL_STACK_SIZE;
226  continue;
227  }
228  }
229 
230  ExceptionRecord->ExceptionFlags |= EXCEPTION_STACK_INVALID;
231  goto DispatchExit;
232  }
233 
234  // See if the handler is reasonable
235 
236  if (!RtlIsValidHandler(RegistrationPointer->Handler)) {
237  ExceptionRecord->ExceptionFlags |= EXCEPTION_STACK_INVALID;
238  goto DispatchExit;
239  }
240 
241  //
242  // The handler must be executed by calling another routine
243  // that is written in assembler. This is required because
244  // up level addressing of the handler information is required
245  // when a nested exception is encountered.
246  //
247 
248  if (NtGlobalFlag & FLG_ENABLE_EXCEPTION_LOGGING) {
249  Index = RtlpLogExceptionHandler(
250  ExceptionRecord,
251  ContextRecord,
252  0,
253  (PULONG)RegistrationPointer,
254  4 * sizeof(ULONG));
255  // can't use sizeof(EXCEPTION_REGISTRATION_RECORD
256  // because we need the 2 dwords above it.
257  }
258 
259  Disposition = RtlpExecuteHandlerForException(
260  ExceptionRecord,
261  (PVOID)RegistrationPointer,
262  ContextRecord,
263  (PVOID)&DispatcherContext,
264  (PEXCEPTION_ROUTINE)RegistrationPointer->Handler);
265 
266  if (NtGlobalFlag & FLG_ENABLE_EXCEPTION_LOGGING) {
267  RtlpLogLastExceptionDisposition(Index, Disposition);
268  }
269 
270  //
271  // If the current scan is within a nested context and the frame
272  // just examined is the end of the context region, then clear
273  // the nested context frame and the nested exception in the
274  // exception flags.
275  //
276 
277  if (NestedRegistration == RegistrationPointer) {
278  ExceptionRecord->ExceptionFlags &= (~EXCEPTION_NESTED_CALL);
279  NestedRegistration = 0;
280  }
281 
282  //
283  // Case on the handler disposition.
284  //
285 
286  switch (Disposition) {
287 
288  //
289  // The disposition is to continue execution. If the
290  // exception is not continuable, then raise the exception
291  // STATUS_NONCONTINUABLE_EXCEPTION. Otherwise return
292  // TRUE.
293  //
294 
295  case ExceptionContinueExecution :
296  if ((ExceptionRecord->ExceptionFlags &
297  EXCEPTION_NONCONTINUABLE) != 0) {
298  ExceptionRecord1.ExceptionCode =
299  STATUS_NONCONTINUABLE_EXCEPTION;
300  ExceptionRecord1.ExceptionFlags = EXCEPTION_NONCONTINUABLE;
301  ExceptionRecord1.ExceptionRecord = ExceptionRecord;
302  ExceptionRecord1.NumberParameters = 0;
303  RtlRaiseException(&ExceptionRecord1);
304 
305  } else {
306  Completion = TRUE;
307  goto DispatchExit;
308  }
309 
310  //
311  // The disposition is to continue the search. If the frame isn't
312  // suspect/corrupt, get next frame address and continue the search
313  //
314 
315  case ExceptionContinueSearch :
316  if (ExceptionRecord->ExceptionFlags & EXCEPTION_STACK_INVALID)
317  goto DispatchExit;
318 
319  break;
320 
321  //
322  // The disposition is nested exception. Set the nested
323  // context frame to the establisher frame address and set
324  // nested exception in the exception flags.
325  //
326 
327  case ExceptionNestedException :
328  ExceptionRecord->ExceptionFlags |= EXCEPTION_NESTED_CALL;
329  if (DispatcherContext.RegistrationPointer > NestedRegistration) {
330  NestedRegistration = DispatcherContext.RegistrationPointer;
331  }
332 
333  break;
334 
335  //
336  // All other disposition values are invalid. Raise
337  // invalid disposition exception.
338  //
339 
340  default :
341  ExceptionRecord1.ExceptionCode = STATUS_INVALID_DISPOSITION;
342  ExceptionRecord1.ExceptionFlags = EXCEPTION_NONCONTINUABLE;
343  ExceptionRecord1.ExceptionRecord = ExceptionRecord;
344  ExceptionRecord1.NumberParameters = 0;
345  RtlRaiseException(&ExceptionRecord1);
346  break;
347  }
348 
349  //
350  // If chain goes in wrong direction or loops, report an
351  // invalid exception stack, otherwise go on to the next one.
352  //
353 
354  RegistrationPointer = RegistrationPointer->Next;
355  }
356 
357  //
358  // Call vectored continue handlers.
359  //
360 
361 DispatchExit:
362 
363  return Completion;
364 }

370 VOID
371 RtlUnwind (
372  IN PVOID TargetFrame OPTIONAL,
373  IN PVOID TargetIp OPTIONAL,
374  IN PEXCEPTION_RECORD ExceptionRecord OPTIONAL,
375  IN PVOID ReturnValue
376  )
377 
378 /*++
379 
380 Routine Description:
381 
382  This function initiates an unwind of procedure call frames. The machine
383  state at the time of the call to unwind is captured in a context record
384  and the unwinding flag is set in the exception flags of the exception
385  record. If the TargetFrame parameter is not specified, then the exit unwind
386  flag is also set in the exception flags of the exception record. A backward
387  walk through the procedure call frames is then performed to find the target
388  of the unwind operation.
389 
390  N.B. The captured context passed to unwinding handlers will not be
391  a completely accurate context set for the 386. This is because
392  there isn't a standard stack frame in which registers are stored.
393 
394  Only the integer registers are affected. The segment and
395  control registers (ebp, esp) will have correct values for
396  the flat 32 bit environment.
397 
398  N.B. If you change the number of arguments, make sure you change the
399  adjustment of ESP after the call to RtlpCaptureContext (for
400  STDCALL calling convention)
401 
402 Arguments:
403 
404  TargetFrame - Supplies an optional pointer to the call frame that is the
405  target of the unwind. If this parameter is not specified, then an exit
406  unwind is performed.
407 
408  TargetIp - Supplies an optional instruction address that specifies the
409  continuation address of the unwind. This address is ignored if the
410  target frame parameter is not specified.
411 
412  ExceptionRecord - Supplies an optional pointer to an exception record.
413 
414  ReturnValue - Supplies a value that is to be placed in the integer
415  function return register just before continuing execution.
416 
417 Return Value:
418 
419  None.
420 
421 --*/
422 
423 {
424  PCONTEXT ContextRecord;
425  CONTEXT ContextRecord1;
426  DISPATCHER_CONTEXT DispatcherContext;
427  EXCEPTION_DISPOSITION Disposition;
428  PEXCEPTION_REGISTRATION_RECORD RegistrationPointer;
429  PEXCEPTION_REGISTRATION_RECORD PriorPointer;
430  ULONG HighAddress;
431  ULONG HighLimit;
432  ULONG LowLimit;
433  EXCEPTION_RECORD ExceptionRecord1;
434  EXCEPTION_RECORD ExceptionRecord2;
435 
436  //
437  // Get current stack limits.
438  //
439 
440  RtlpGetStackLimits(&LowLimit, &HighLimit);
441 
442  //
443  // If an exception record is not specified, then build a local exception
444  // record for use in calling exception handlers during the unwind operation.
445  //
446 
447  if (ARGUMENT_PRESENT(ExceptionRecord) == FALSE) {
448  ExceptionRecord = &ExceptionRecord1;
449  ExceptionRecord1.ExceptionCode = STATUS_UNWIND;
450  ExceptionRecord1.ExceptionFlags = 0;
451  ExceptionRecord1.ExceptionRecord = NULL;
452  ExceptionRecord1.ExceptionAddress = _ReturnAddress();
453  ExceptionRecord1.NumberParameters = 0;
454  }
455 
456  //
457  // If the target frame of the unwind is specified, then set EXCEPTION_UNWINDING
458  // flag in the exception flags. Otherwise set both EXCEPTION_EXIT_UNWIND and
459  // EXCEPTION_UNWINDING flags in the exception flags.
460  //
461 
462  if (ARGUMENT_PRESENT(TargetFrame) == TRUE) {
463  ExceptionRecord->ExceptionFlags |= EXCEPTION_UNWINDING;
464  } else {
465  ExceptionRecord->ExceptionFlags |= (EXCEPTION_UNWINDING |
466  EXCEPTION_EXIT_UNWIND);
467  }
468 
469  //
470  // Capture the context.
471  //
472 
473  ContextRecord = &ContextRecord1;
474  ContextRecord1.ContextFlags = CONTEXT_INTEGER | CONTEXT_CONTROL | CONTEXT_SEGMENTS;
475  RtlpCaptureContext(ContextRecord);
476 
477  //
478  // Adjust captured context to pop our arguments off the stack
479  //
480  ContextRecord->Esp += sizeof(TargetFrame) +
481  sizeof(TargetIp) +
482  sizeof(ExceptionRecord) +
483  sizeof(ReturnValue);
484  ContextRecord->Eax = (ULONG)ReturnValue;
485 
486  //
487  // Scan backward through the call frame hierarchy, calling exception
488  // handlers as they are encountered, until the target frame of the unwind
489  // is reached.
490  //
491 
492  RegistrationPointer = RtlpGetRegistrationHead();
493  while (RegistrationPointer != EXCEPTION_CHAIN_END) {
494 
495  //
496  // If this is the target of the unwind, then continue execution
497  // by calling the continue system service.
498  //
499 
500  if ((ULONG)RegistrationPointer == (ULONG)TargetFrame) {
501  ZwContinue(ContextRecord, FALSE);
502 
503  //
504  // If the target frame is lower in the stack than the current frame,
505  // then raise STATUS_INVALID_UNWIND exception.
506  //
507 
508  } else if ( (ARGUMENT_PRESENT(TargetFrame) == TRUE) &&
509  ((ULONG)TargetFrame < (ULONG)RegistrationPointer) ) {
510  ExceptionRecord2.ExceptionCode = STATUS_INVALID_UNWIND_TARGET;
511  ExceptionRecord2.ExceptionFlags = EXCEPTION_NONCONTINUABLE;
512  ExceptionRecord2.ExceptionRecord = ExceptionRecord;
513  ExceptionRecord2.NumberParameters = 0;
514  RtlRaiseException(&ExceptionRecord2);
515  }
516 
517  //
518  // If the call frame is not within the specified stack limits or the
519  // call frame is unaligned, then raise the exception STATUS_BAD_STACK.
520  // Else restore the state from the specified frame to the context
521  // record.
522  //
523 
524  HighAddress = (ULONG)RegistrationPointer +
525  sizeof(EXCEPTION_REGISTRATION_RECORD);
526 
527  if ( ((ULONG)RegistrationPointer < LowLimit) ||
528  (HighAddress > HighLimit) ||
529  (((ULONG)RegistrationPointer & 0x3) != 0)
530  ) {
531 
532  //
533  // Allow for the possibility that the problem occured on the
534  // DPC stack.
535  //
536 
537  ULONG TestAddress = (ULONG)RegistrationPointer;
538 
539  if (((TestAddress & 0x3) == 0) &&
540  KeGetCurrentIrql() >= DISPATCH_LEVEL) {
541 
542  PKPRCB Prcb = KeGetCurrentPrcb();
543  ULONG DpcStack = (ULONG)Prcb->DpcStack;
544 
545  if ((Prcb->DpcRoutineActive) &&
546  (HighAddress <= DpcStack) &&
547  (TestAddress >= DpcStack - KERNEL_STACK_SIZE)) {
548 
549  //
550  // This error occured on the DPC stack, switch
551  // stack limits to the DPC stack and restart
552  // the loop.
553  //
554 
555  HighLimit = DpcStack;
556  LowLimit = DpcStack - KERNEL_STACK_SIZE;
557  continue;
558  }
559  }
560 
561  ExceptionRecord2.ExceptionCode = STATUS_BAD_STACK;
562  ExceptionRecord2.ExceptionFlags = EXCEPTION_NONCONTINUABLE;
563  ExceptionRecord2.ExceptionRecord = ExceptionRecord;
564  ExceptionRecord2.NumberParameters = 0;
565  RtlRaiseException(&ExceptionRecord2);
566  } else {
567 
568  //
569  // The handler must be executed by calling another routine
570  // that is written in assembler. This is required because
571  // up level addressing of the handler information is required
572  // when a collided unwind is encountered.
573  //
574 
575  Disposition = RtlpExecuteHandlerForUnwind(
576  ExceptionRecord,
577  (PVOID)RegistrationPointer,
578  ContextRecord,
579  (PVOID)&DispatcherContext,
580  RegistrationPointer->Handler);
581 
582  //
583  // Case on the handler disposition.
584  //
585 
586  switch (Disposition) {
587 
588  //
589  // The disposition is to continue the search. Get next
590  // frame address and continue the search.
591  //
592 
593  case ExceptionContinueSearch :
594  break;
595 
596  //
597  // The disposition is colided unwind. Maximize the target
598  // of the unwind and change the context record pointer.
599  //
600 
601  case ExceptionCollidedUnwind :
602 
603  //
604  // Pick up the registration pointer that was active at
605  // the time of the unwind, and simply continue.
606  //
607 
608  RegistrationPointer = DispatcherContext.RegistrationPointer;
609  break;
610 
611 
612  //
613  // All other disposition values are invalid. Raise
614  // invalid disposition exception.
615  //
616 
617  default :
618  ExceptionRecord2.ExceptionCode = STATUS_INVALID_DISPOSITION;
619  ExceptionRecord2.ExceptionFlags = EXCEPTION_NONCONTINUABLE;
620  ExceptionRecord2.ExceptionRecord = ExceptionRecord;
621  ExceptionRecord2.NumberParameters = 0;
622  RtlRaiseException(&ExceptionRecord2);
623  break;
624  }
625 
626  //
627  // Step to next registration record
628  //
629 
630  PriorPointer = RegistrationPointer;
631  RegistrationPointer = RegistrationPointer->Next;
632 
633  //
634  // Unlink the unwind handler, since it's been called.
635  //
636 
637  RtlpUnlinkHandler(PriorPointer);
638 
639  //
640  // If chain goes in wrong direction or loops, raise an
641  // exception.
642  //
643 
644  }
645  }
646 
647  if (TargetFrame == EXCEPTION_CHAIN_END) {
648 
649  //
650  // Caller simply wants to unwind all exception records.
651  // This differs from an exit_unwind in that no "exit" is desired.
652  // Do a normal continue, since we've effectively found the
653  // "target" the caller wanted.
654  //
655 
656  ZwContinue(ContextRecord, FALSE);
657 
658  } else {
659 
660  //
661  // Either (1) a real exit unwind was performed, or (2) the
662  // specified TargetFrame is not present in the exception handler
663  // list. In either case, give debugger and subsystem a chance
664  // to see the unwind.
665  //
666 
667  ZwRaiseException(ExceptionRecord, ContextRecord, FALSE);
668 
669  }
670  return;
671 }
