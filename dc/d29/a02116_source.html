318 ExpInterlockedExchange (
319  IN OUT PULONG Index,
320  IN ULONG FirstIndex,
321  IN PHANDLE_TABLE_ENTRY Entry
322  )
323 /*++
324 
325 Routine Description:
326 
327  This performs the following steps:
328  1. Set Entry->NextFreeTableEntry = *Index
329  2. Loops until *Index == (the value of *Index when we entered the function)
330  When they're equal, we set *Index = FirstIndex
331 
332 
333 Arguments:
334 
335  Index - Points to the ULONG we want to set.
336 
337  FirstIndex - New value to set Index to.
338 
339  Entry - TableEntry that will get the initial value of *Index before it's
340  updated.
341 
342 Return Value:
343 
344  New value of *Index (i.e. FirstIndex).
345 
346 --*/
347 {
348  ULONG OldIndex, NewIndex;
349 
350  EXASSERT (Entry->Object == NULL);
351 
352  //
353  // Load new value and generate the sequence number on pushes
354  //
355 
356  NewIndex = FirstIndex + GetNextSeq();
357 
358  while (1) {
359 
360  //
361  // remember original value and
362  // archive it in NextFreeTableEntry.
363  //
364 
365  OldIndex = ReadForWriteAccess (Index);
366  Entry->NextFreeTableEntry = OldIndex;
367 
368 
369  //
370  // Swap in the new value, and if the swap occurs
371  // successfully, we're done.
372  //
373  if (OldIndex == (ULONG) InterlockedCompareExchange ((PLONG)Index,
374  NewIndex,
375  OldIndex)) {
376  return OldIndex;
377  }
378  }
379 }

381 ULONG
382 ExpMoveFreeHandles (
383  IN PHANDLE_TABLE HandleTable
384  )
385 {
386  ULONG OldValue, NewValue;
387  ULONG Index, OldIndex, NewIndex, FreeSize;
388  PHANDLE_TABLE_ENTRY Entry, FirstEntry;
389  EXHANDLE Handle;
390  ULONG Idx;
391  BOOLEAN StrictFIFO;
392 
393  //
394  // First remove all the handles from the free list so we can add them to the
395  // list we use for allocates.
396  //
397 
398  OldValue = InterlockedExchange ((PLONG)&HandleTable->LastFree,
399  0);
400  Index = OldValue;
401  if (Index == 0) {
402  //
403  // There are no free handles. Nothing to do.
404  //
405  return OldValue;
406  }
407 
408 
409  //
410  // We are pushing old entries onto the free list.
411  // We have the A-B-A problem here as these items may have been moved here because
412  // another thread was using them in the pop code.
413  //
414  for (Idx = 1; Idx < HANDLE_TABLE_LOCKS; Idx++) {
415  ExAcquireReleasePushLockExclusive (&HandleTable->HandleTableLock[Idx]);
416  }
417  StrictFIFO = HandleTable->StrictFIFO;
418 
419  //
420  // If we are strict FIFO then reverse the list to make handle reuse rare.
421  //
422  if (!StrictFIFO) {
423  //
424  // We have a complete chain here. If there is no existing chain we
425  // can just push this one without any hassles. If we can't then
426  // we can just fall into the reversing code anyway as we need
427  // to find the end of the chain to continue it.
428  //
429 
430  //
431  // This is a push so create a new sequence number
432  //
433 
434  if (InterlockedCompareExchange ((PLONG)&HandleTable->FirstFree,
435  OldValue + GetNextSeq(),
436  0) == 0) {
437  return OldValue;
438  }
439  }
440 
441  //
442  // Loop over all the entries and reverse the chain.
443  //
444  FreeSize = OldIndex = 0;
445  FirstEntry = NULL;
446  while (1) {
447  FreeSize++;
448  Handle.Value = Index;
449  Entry = ExpLookupHandleTableEntry (HandleTable, Handle);
450 
451  EXASSERT (Entry->Object == NULL);
452 
453  NewIndex = Entry->NextFreeTableEntry;
454  Entry->NextFreeTableEntry = OldIndex;
455  if (OldIndex == 0) {
456  FirstEntry = Entry;
457  }
458  OldIndex = Index;
459  if (NewIndex == 0) {
460  break;
461  }
462  Index = NewIndex;
463  }
464 
465  NewValue = ExpInterlockedExchange (&HandleTable->FirstFree,
466  OldIndex,
467  FirstEntry);
468 
469  //
470  // If we haven't got a pool of a few handles then force
471  // table expansion to keep the free handle size high
472  //
473  if (FreeSize < 100 && StrictFIFO) {
474  OldValue = 0;
475  }
476  return OldValue;
477 }

479 PHANDLE_TABLE_ENTRY
480 ExpAllocateHandleTableEntry (
481  IN PHANDLE_TABLE HandleTable,
482  OUT PEXHANDLE pHandle
483  )
484 /*++
485 
486 Routine Description:
487 
488  This routine does a fast allocate of a free handle. It's lock free if
489  possible.
490 
491  Only the rare case of handle table expansion is covered by the handle
492  table lock.
493 
494 Arguments:
495 
496  HandleTable - Supplies the handle table being allocated from.
497 
498  pHandle - Handle returned
499 
500 Return Value:
501 
502  PHANDLE_TABLE_ENTRY - The allocated handle table entry pointer or NULL
503  on failure.
504 
505 --*/
506 {
507  PKTHREAD CurrentThread;
508  ULONG OldValue, NewValue, NewValue1;
509  PHANDLE_TABLE_ENTRY Entry;
510  EXHANDLE Handle;
511  BOOLEAN RetVal;
512  ULONG Idx;
513 
514 
515  CurrentThread = KeGetCurrentThread ();
516  while (1) {
517 
518  OldValue = HandleTable->FirstFree;
519 
520 
521  while (OldValue == 0) {
522  //
523  // Lock the handle table for exclusive access as we will be
524  // allocating a new table level.
525  //
526  ExpLockHandleTableExclusive (HandleTable, CurrentThread);
527 
528  //
529  // If we have multiple threads trying to expand the table at
530  // the same time then by just acquiring the table lock we
531  // force those threads to complete their allocations and
532  // populate the free list. We must check the free list here
533  // so we don't expand the list twice without needing to.
534  //
535 
536  OldValue = HandleTable->FirstFree;
537  if (OldValue != 0) {
538  ExpUnlockHandleTableExclusive (HandleTable, CurrentThread);
539  break;
540  }
541 
542  //
543  // See if we have any handles on the alternate free list
544  // These handles need some locking to move them over.
545  //
546  OldValue = ExpMoveFreeHandles (HandleTable);
547  if (OldValue != 0) {
548  ExpUnlockHandleTableExclusive (HandleTable, CurrentThread);
549  break;
550  }
551 
552  //
553  // This must be the first thread attempting expansion or all the
554  // free handles allocated by another thread got used up in the gap.
555  //
556 
557  RetVal = ExpAllocateHandleTableEntrySlow (HandleTable, TRUE);
558 
559  ExpUnlockHandleTableExclusive (HandleTable, CurrentThread);
560 
561 
562  OldValue = HandleTable->FirstFree;
563 
564  //
565  // If ExpAllocateHandleTableEntrySlow had a failed allocation
566  // then we want to fail the call. We check for free entries
567  // before we exit just in case they got allocated or freed by
568  // somebody else in the gap.
569  //
570 
571  if (!RetVal) {
572  if (OldValue == 0) {
573  pHandle->GenericHandleOverlay = NULL;
574  return NULL;
575  }
576  }
577  }
578 
579 
580  Handle.Value = (OldValue & FREE_HANDLE_MASK);
581 
582  Entry = ExpLookupHandleTableEntry (HandleTable, Handle);
583 
584  Idx = ((OldValue & FREE_HANDLE_MASK)>>2) % HANDLE_TABLE_LOCKS;
585  ExpLockHandleTableShared (HandleTable, CurrentThread, Idx);
586 
587  if (OldValue != *(volatile ULONG *) &HandleTable->FirstFree) {
588  ExpUnlockHandleTableShared (HandleTable, CurrentThread, Idx);
589  continue;
590  }
591 
592  KeMemoryBarrier ();
593 
594  NewValue = *(volatile ULONG *) &Entry->NextFreeTableEntry;
595 
596  NewValue1 = InterlockedCompareExchange ((PLONG)&HandleTable->FirstFree,
597  NewValue,
598  OldValue);
599 
600  ExpUnlockHandleTableShared (HandleTable, CurrentThread, Idx);
601 
602  if (NewValue1 == OldValue) {
603  EXASSERT ((NewValue & FREE_HANDLE_MASK) < HandleTable->NextHandleNeedingPool);
604  break;
605  } else {
606  //
607  // We should have eliminated the A-B-A problem so if only the sequence number has
608  // changed we are broken.
609  //
610  EXASSERT ((NewValue1 & FREE_HANDLE_MASK) != (OldValue & FREE_HANDLE_MASK));
611  }
612  }
613  InterlockedIncrement (&HandleTable->HandleCount);
614 
615  *pHandle = Handle;
616 
617  return Entry;
618 }

647 ExpLockHandleTableEntry (
648  PHANDLE_TABLE HandleTable,
649  PHANDLE_TABLE_ENTRY HandleTableEntry
650  )
651 
652 /*++
653 
654 Routine Description:
655 
656  This routine locks the specified handle table entry. After the entry is
657  locked the sign bit will be set.
658 
659 Arguments:
660 
661  HandleTable - Supplies the handle table containing the entry being locked.
662 
663  HandleTableEntry - Supplies the handle table entry being locked.
664 
665 Return Value:
666 
667  TRUE if the entry is valid and locked, and FALSE if the entry is
668  marked free.
669 
670 --*/
671 
672 {
673  LONG_PTR NewValue;
674  LONG_PTR CurrentValue;
675 
676  //
677  // We are about to take a lock. Make sure we are protected.
678  //
679  ASSERT ((KeGetCurrentThread()->CombinedApcDisable != 0) || (KeGetCurrentIrql() == APC_LEVEL));
680 
681  //
682  // We'll keep on looping reading in the value, making sure it is not null,
683  // and if it is not currently locked we'll try for the lock and return
684  // true if we get it. Otherwise we'll pause a bit and then try again.
685  //
686 
687 
688  while (TRUE) {
689 
690  CurrentValue = ReadForWriteAccess(((volatile LONG_PTR *)&HandleTableEntry->Object));
691 
692  //
693  // If the handle value is greater than zero then it is not currently
694  // locked and we should try for the lock, by setting the lock bit and
695  // doing an interlocked exchange.
696  //
697 
698  if (CurrentValue & EXHANDLE_TABLE_ENTRY_LOCK_BIT) {
699 
700  //
701  // Remove the
702  //
703  NewValue = CurrentValue - EXHANDLE_TABLE_ENTRY_LOCK_BIT;
704 
705  if ((LONG_PTR)(InterlockedCompareExchangePointer (&HandleTableEntry->Object,
706  (PVOID)NewValue,
707  (PVOID)CurrentValue)) == CurrentValue) {
708 
709  return TRUE;
710  }
711  } else {
712  //
713  // Make sure the handle table entry is not freed
714  //
715 
716  if (CurrentValue == 0) {
717 
718  return FALSE;
719  }
720  }
721  ExpBlockOnLockedHandleEntry (HandleTable, HandleTableEntry);
722  }
723 }
