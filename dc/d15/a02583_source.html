441 NTSTATUS
442 PsSetProcessWin32Process(
443  __in PEPROCESS Process,
444  __in PVOID Win32Process,
445  __in PVOID PrevWin32Process
446  )
447 {
448  NTSTATUS Status;
449  PETHREAD CurrentThread;
450 
451  Status = STATUS_SUCCESS;
452 
453  CurrentThread = _PsGetCurrentThread();
454 
455  PspLockProcessExclusive (Process, CurrentThread);
456 
457  if (Win32Process != NULL) {
458  if ((Process->Flags&PS_PROCESS_FLAGS_PROCESS_DELETE) == 0 && Process->Win32Process == NULL) {
459  Process->Win32Process = Win32Process;
460  } else {
461  Status = STATUS_PROCESS_IS_TERMINATING;
462  }
463  } else {
464  if (Process->Win32Process == PrevWin32Process) {
465  Process->Win32Process = NULL;
466  } else {
467  Status = STATUS_UNSUCCESSFUL;
468  }
469  }
470 
471  PspUnlockProcessExclusive (Process, CurrentThread);
472 
473  return Status;
474 }

562 NTSTATUS
563 PsReferenceProcessFilePointer (
564  IN PEPROCESS Process,
565  OUT PVOID *OutFileObject
566  )
567 
568 /*++
569 
570 Routine Description:
571 
572  This routine returns a referenced pointer to the FilePointer of Process.
573  This is a rundown protected wrapper around MmGetFileObjectForSection.
574 
575 Arguments:
576 
577  Process - Supplies the process to query.
578 
579  OutFileObject - Returns the file object backing the requested section if
580  success is returned.
581 
582 Return Value:
583 
584  NTSTATUS.
585 
586 Environment:
587 
588  Kernel mode, PASSIVE_LEVEL.
589 
590 --*/
591 
592 {
593  PFILE_OBJECT FileObject;
594 
595  PAGED_CODE();
596 
597  if (!ExAcquireRundownProtection (&Process->RundownProtect)) {
598  return STATUS_UNSUCCESSFUL;
599  }
600 
601  if (Process->SectionObject == NULL) {
602  ExReleaseRundownProtection (&Process->RundownProtect);
603  return STATUS_UNSUCCESSFUL;
604  }
605 
606  FileObject = MmGetFileObjectForSection ((PVOID)Process->SectionObject);
607 
608  *OutFileObject = FileObject;
609 
610  ObReferenceObject (FileObject);
611 
612  ExReleaseRundownProtection (&Process->RundownProtect);
613 
614  return STATUS_SUCCESS;
615 }
