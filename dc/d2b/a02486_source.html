52 /*++
53 
54 Routine Description:
55 
56  This routine is used to initialize the message queue for a port object.
57 
58 Arguments:
59 
60  Port - Supplies the port object being initialized
61 
62 Return Value:
63 
64  NTSTATUS - An appropriate status value
65 
66 --*/
67 
68 {
69  PLPCP_NONPAGED_PORT_QUEUE NonPagedPortQueue;
70 
71  PAGED_CODE();
72 
73  //
74  // Allocate space for the port queue
75  //
76 
77  NonPagedPortQueue = ExAllocatePoolWithTag( NonPagedPool,
78  sizeof(LPCP_NONPAGED_PORT_QUEUE),
79  'troP' );
80 
81  if (NonPagedPortQueue == NULL) {
82 
83  return STATUS_INSUFFICIENT_RESOURCES;
84  }
85 
86  //
87  // Initialize the fields in the non paged port queue
88  //
89 
90  KeInitializeSemaphore( &NonPagedPortQueue->Semaphore, 0, 0x7FFFFFFF );
91 
92  NonPagedPortQueue->BackPointer = Port;
93 
94  //
95  // Have the port msg queue point to the non nonpaged port queue
96  //
97 
98  Port->MsgQueue.Semaphore = &NonPagedPortQueue->Semaphore;
99 
100  //
101  // Initialize the port msg queue to be empty
102  //
103 
104  InitializeListHead( &Port->MsgQueue.ReceiveHead );
105 
106  //
107  // And return to our caller
108  //
109 
110  return STATUS_SUCCESS;

114 VOID
115 LpcpDestroyPortQueue (
116  IN PLPCP_PORT_OBJECT Port,
117  IN BOOLEAN CleanupAndDestroy
118  )
119 
120 /*++
121 
122 Routine Description:
123 
124  This routine is used to teardown the message queue of a port object.
125  After running this message will either be empty (like it was just
126  initialized) or completely gone (needs to be initialized)
127 
128 Arguments:
129 
130  Port - Supplies the port containing the message queue being modified
131 
132  CleanupAndDestroy - Specifies if the message queue should be set back
133  to the freshly initialized state (value of FALSE) or completely
134  torn down (value of TRUE)
135 
136 Return Value:
137 
138  None.
139 
140 --*/
141 
142 {
143  PLIST_ENTRY Next, Head;
144  PETHREAD ThreadWaitingForReply;
145  PLPCP_MESSAGE Msg;
146  PLPCP_PORT_OBJECT ConnectionPort = NULL;
147 
148  PAGED_CODE();
149 
150  //
151  // If this port is connected to another port, then disconnect it.
152  // Protect this with a lock in case the other side is going away
153  // at the same time.
154  //
155 
156  LpcpAcquireLpcpLock();
157 
158  if ( ((Port->Flags & PORT_TYPE) != UNCONNECTED_COMMUNICATION_PORT)
159  &&
160  (Port->ConnectedPort != NULL) ) {
161 
162  Port->ConnectedPort->ConnectedPort = NULL;
163 
164  //
165  // Disconnect the connection port
166  //
167 
168  if (Port->ConnectedPort->ConnectionPort) {
169 
170  ConnectionPort = Port->ConnectedPort->ConnectionPort;
171 
172  Port->ConnectedPort->ConnectionPort = NULL;
173  }
174  }
175 
176  //
177  // If connection port, then mark name as deleted
178  //
179 
180  if ((Port->Flags & PORT_TYPE) == SERVER_CONNECTION_PORT) {
181 
182  Port->Flags |= PORT_NAME_DELETED;
183  }
184 
185  //
186  // Walk list of threads waiting for a reply to a message sent to this
187  // port. Signal each thread's LpcReplySemaphore to wake them up. They
188  // will notice that there was no reply and return
189  // STATUS_PORT_DISCONNECTED
190  //
191 
192  Head = &Port->LpcReplyChainHead;
193  Next = Head->Flink;
194 
195  while ((Next != NULL) && (Next != Head)) {
196 
197  ThreadWaitingForReply = CONTAINING_RECORD( Next, ETHREAD, LpcReplyChain );
198 
199  //
200  // If the thread is exiting, in the location of LpcReplyChain is stored the ExitTime
201  // We'll stop to search through the list.
202 
203  if ( ThreadWaitingForReply->LpcExitThreadCalled ) {
204 
205  break;
206  }
207 
208  Next = Next->Flink;
209 
210  RemoveEntryList( &ThreadWaitingForReply->LpcReplyChain );
211 
212  InitializeListHead( &ThreadWaitingForReply->LpcReplyChain );
213 
214  if (!KeReadStateSemaphore( &ThreadWaitingForReply->LpcReplySemaphore )) {
215 
216  //
217  // Thread is waiting on a message. Signal the semaphore and free
218  // the message
219  //
220 
221  Msg = LpcpGetThreadMessage(ThreadWaitingForReply);
222 
223  if ( Msg ) {
224 
225  //
226  // If the message is a connection request and has a section object
227  // attached, then dereference that section object
228  //
229 
230  if ((Msg->Request.u2.s2.Type & ~LPC_KERNELMODE_MESSAGE) == LPC_CONNECTION_REQUEST) {
231 
232  PLPCP_CONNECTION_MESSAGE ConnectMsg;
233 
234  ConnectMsg = (PLPCP_CONNECTION_MESSAGE)(Msg + 1);
235 
236  if ( ConnectMsg->SectionToMap != NULL ) {
237 
238  ObDereferenceObject( ConnectMsg->SectionToMap );
239  }
240  }
241 
242  ThreadWaitingForReply->LpcReplyMessage = NULL;
243 
244  LpcpFreeToPortZone( Msg, LPCP_MUTEX_OWNED );
245  Next = Port->LpcReplyChainHead.Flink; // Lock has been dropped
246  }
247 
248  ThreadWaitingForReply->LpcReplyMessageId = 0;
249 
250  KeReleaseSemaphore( &ThreadWaitingForReply->LpcReplySemaphore,
251  0,
252  1L,
253  FALSE );
254  }
255  }
256 
257  InitializeListHead( &Port->LpcReplyChainHead );
258 
259  //
260  // Walk list of messages queued to this port. Remove each message from
261  // the list and free it.
262  //
263 
264  while (Port->MsgQueue.ReceiveHead.Flink && !IsListEmpty (&Port->MsgQueue.ReceiveHead)) {
265 
266  Msg = CONTAINING_RECORD( Port->MsgQueue.ReceiveHead.Flink, LPCP_MESSAGE, Entry );
267 
268  RemoveEntryList (&Msg->Entry);
269 
270  InitializeListHead( &Msg->Entry );
271 
272  LpcpFreeToPortZone( Msg, LPCP_MUTEX_OWNED );
273 
274  }
275 
276  LpcpReleaseLpcpLock();
277 
278  if ( ConnectionPort ) {
279 
280  ObDereferenceObject( ConnectionPort );
281  }
282 
283  //
284  // Check if the caller wants it all to go away
285  //
286 
287  if ( CleanupAndDestroy ) {
288 
289  //
290  // Free semaphore associated with the queue.
291  //
292 
293  if (Port->MsgQueue.Semaphore != NULL) {
294 
295  ExFreePool( CONTAINING_RECORD( Port->MsgQueue.Semaphore,
296  LPCP_NONPAGED_PORT_QUEUE,
297  Semaphore ));
298  }
299  }
300 
301  //
302  // And return to our caller
303  //
304 
305  return;
306 }

356 LpcpFreeToPortZone (
357  IN PLPCP_MESSAGE Msg,
358  IN ULONG MutexFlags
359  )
360 {
361  PLPCP_CONNECTION_MESSAGE ConnectMsg;
362  PETHREAD RepliedToThread = NULL;
363  PLPCP_PORT_OBJECT ClientPort = NULL;
364 
365  PAGED_CODE();
366 
367  //
368  // Acquire the global lock if necessary
369  //
370 
371  if ((MutexFlags & LPCP_MUTEX_OWNED) == 0) {
372 
373  LpcpAcquireLpcpLock();
374  }
375 
376  //
377  // A entry field connects the message to the message queue of the
378  // owning port object. If not already removed then remove this
379  // message
380  //
381 
382  if (!IsListEmpty( &Msg->Entry )) {
383  RemoveEntryList( &Msg->Entry );
384  InitializeListHead( &Msg->Entry );
385  }
386 
387  //
388  // If the replied to thread is not null then we have a reference
389  // to the thread that we should now remove
390  //
391 
392  if (Msg->RepliedToThread != NULL) {
393  RepliedToThread = Msg->RepliedToThread;
394  Msg->RepliedToThread = NULL;
395  }
396 
397  //
398  // If the msg was for a connection request then we know that
399  // right after the lpcp message is a connection message whose
400  // client port field might need to be dereferenced
401  //
402 
403  if ((Msg->Request.u2.s2.Type & ~LPC_KERNELMODE_MESSAGE) == LPC_CONNECTION_REQUEST) {
404 
405  ConnectMsg = (PLPCP_CONNECTION_MESSAGE)(Msg + 1);
406 
407  if (ConnectMsg->ClientPort) {
408 
409  //
410  // Capture a pointer to the client port then null it
411  // out so that no one else can use it, then release
412  // lpcp lock before we dereference the client port
413  //
414 
415  ClientPort = ConnectMsg->ClientPort;
416 
417  ConnectMsg->ClientPort = NULL;
418  }
419  }
420 
421  LpcpReleaseLpcpLock();
422 
423  if ( ClientPort ) {
424 
425  ObDereferenceObject( ClientPort );
426  }
427 
428  if ( RepliedToThread ) {
429 
430  ObDereferenceObject( RepliedToThread );
431  }
432 
433  ExFreeToPagedLookasideList(&LpcpMessagesLookaside, Msg);
434 
435  if ((MutexFlags & LPCP_MUTEX_OWNED) &&
436  ((MutexFlags & LPCP_MUTEX_RELEASE_ON_RETURN) == 0)) {
437 
438  LpcpAcquireLpcpLock();
439  }
440 
441 }

448  IN PLPCP_MESSAGE Msg,

451 
452 /*++
453 
454 Routine Description:
455 
456  This routine is used in place of freeing a message and instead saves the
457  message off a separate queue from the port.
458 
459 Arguments:
460 
461  Port - Specifies the port object under which to save this message
462 
463  Msg - Supplies the message being saved
464 
465  MutexFlags - Supplies whether the mutex is owned.
466 
467 Return Value:
468 
469  None.
470 
471 --*/
472 
473 {
474  PAGED_CODE();
475 
476  //
477  // Take out the global lock if our caller didn't already.
478  //
479 
480  if ((MutexFlags & LPCP_MUTEX_OWNED) == 0) {
481  LpcpAcquireLpcpLock();
482  }
483 
484  //
485  // Make sure we get to the connection port object of this port
486  //
487 
488  if ((Port->Flags & PORT_TYPE) > UNCONNECTED_COMMUNICATION_PORT) {
489 
490  Port = Port->ConnectionPort;
491 
492  if (Port == NULL) {
493 
494  if ((MutexFlags & LPCP_MUTEX_OWNED) == 0) {
495  LpcpReleaseLpcpLock();
496  }
497 
498  return;
499  }
500  }
501 
502  LpcpTrace(( "%s Saving DataInfo Message %lx (%u.%u) Port: %lx\n",
503  PsGetCurrentProcess()->ImageFileName,
504  Msg,
505  Msg->Request.MessageId,
506  Msg->Request.CallbackId,
507  Port ));
508 
509  //
510  // Enqueue this message onto the data info chain for the port
511  //
512 
513  InsertTailList( &Port->LpcDataInfoChainHead, &Msg->Entry );
514 
515  //
516  // Free the global lock
517  //
518 
519  if ((MutexFlags & LPCP_MUTEX_OWNED) == 0) {
520  LpcpReleaseLpcpLock();
521  }
522 
523  //
524  // And return to our caller
525  //
526 
527  return;
528 }
