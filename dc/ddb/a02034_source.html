331 /*++
332 
333 Routine Description:
334 
335  This routine is intended to be called by File Systems only. It
336  initializes the cache maps for data caching. It should be called
337  every time a file is opened or created, and NO_INTERMEDIATE_BUFFERING
338  was specified as FALSE.
339

341 
342  FileObject - A pointer to the newly-created file object.

346  valid data length tracking and callbacks are not desired.
347

349  access, or TRUE if file will be used for Pin access.
350  (Files for Pin access are not limited in size as the caller
351  must access multiple areas of the file at once.)
352 
353  Callbacks - Structure of callbacks used by the Lazy Writer
354 
355  LazyWriteContext - Parameter to be passed in to above routine.
356

358 
359  None. If an error occurs, this routine will Raise the status.

377  DebugTrace( 0, me, " FileSizes = %08lx\n", FileSizes );

381  // the spin lock.
382  //
383

387  // If no FileSize was given, set to one byte before maximizing below.
388  //
389

391  LocalSizes.AllocationSize.LowPart += 1;
392  }
393

396  // size to next create modulo. (***Temp*** there may be too many
397  // apps that end up allowing shared write, thanks to our Dos heritage,
398  // to keep that part of the check in.)
399  //

402 
403  LocalSizes.AllocationSize.QuadPart = LocalSizes.AllocationSize.QuadPart + (LONGLONG)(DEFAULT_CREATE_MODULO - 1);
404  LocalSizes.AllocationSize.LowPart &= ~(DEFAULT_CREATE_MODULO - 1);
405 
406  } else {
407 
408  LocalSizes.AllocationSize.QuadPart = LocalSizes.AllocationSize.QuadPart + (LONGLONG)(VACB_MAPPING_GRANULARITY - 1);
409  LocalSizes.AllocationSize.LowPart &= ~(VACB_MAPPING_GRANULARITY - 1);
410  }
411 
412  //
413  // Do the allocate of the SharedCacheMap, based on an unsafe test,
414  // while not holding a spinlock. If the allocation fails, it's ok
415  // to fail the request even though the test was unsafe.
416  //
417 
418  if (FileObject->SectionObjectPointer->SharedCacheMap == NULL) {
419 
420 restart:
421 
422  ASSERT (CacheMapToFree == NULL);
423 
424  SharedCacheMap = ExAllocatePoolWithTag( NonPagedPool, sizeof(SHARED_CACHE_MAP), 'cScC' );
425 
426  if (SharedCacheMap == NULL) {
427  DebugTrace( 0, 0, "Failed to allocate SharedCacheMap\n", 0 );
428  ExRaiseStatus( STATUS_INSUFFICIENT_RESOURCES );
429  }
430 
431  //
432  // Stash a copy of it so we can free it in the error path below.
433  //
434 
435  CacheMapToFree = SharedCacheMap;
436 
437  //
438  // Zero the SharedCacheMap and fill in the nonzero portions later.
439  //
440 
441  RtlZeroMemory( SharedCacheMap, sizeof(SHARED_CACHE_MAP) );
442 
443  //
444  // Now initialize the Shared Cache Map.
445  //
446 
447  SharedCacheMap->NodeTypeCode = CACHE_NTC_SHARED_CACHE_MAP;
448  SharedCacheMap->NodeByteSize = sizeof(SHARED_CACHE_MAP);
449  SharedCacheMap->FileObject = FileObject;
450  SharedCacheMap->FileSize = LocalSizes.FileSize;
451  SharedCacheMap->ValidDataLength = LocalSizes.ValidDataLength;
452  SharedCacheMap->ValidDataGoal = LocalSizes.ValidDataLength;
453  // SharedCacheMap->Section set below
454

456  // Initialize the spin locks.
457  //
458

461 
462  ExInitializePushLock( &SharedCacheMap->VacbPushLock );
463 
464  if (PinAccess) {
465  SetFlag(SharedCacheMap->Flags, PIN_ACCESS);
466  }
467 
468  //
469  // If this file has FO_SEQUENTIAL_ONLY set, then remember that
470  // in the SharedCacheMap.
471  //
472 
473  if (FlagOn(FileObject->Flags, FO_SEQUENTIAL_ONLY)) {
474  SetFlag(SharedCacheMap->Flags, ONLY_SEQUENTIAL_ONLY_SEEN);
475  }
476 
477  //
478  // Do the round-robin allocation of the spinlock for the shared
479  // cache map. Note the manipulation of the next
480  // counter is safe, since we have the CcMasterSpinLock
481  // exclusive.
482  //
483

485  SharedCacheMap->Callbacks = Callbacks;
486  SharedCacheMap->LazyWriteContext = LazyWriteContext;
487 
488  //

490  //
491

493  }

495  //

497  //

501  CcAcquireMasterLock( &OldIrql );
502 
503  //
504  // Check for second initialization of same file object
505  //
506 
507  if (FileObject->PrivateCacheMap != NULL) {

510  CcReleaseMasterLock( OldIrql );
511  if (CacheMapToFree != NULL) {
512  ExFreePool(CacheMapToFree);

521  // as an Fcb.)
522  //
523

525 
526  //
527  // If there is no SharedCacheMap, then we must create a section and
528  // the SharedCacheMap structure.
529  //
530 
531  if (SharedCacheMap == NULL) {
532 
533  //
534  // Insert the new SharedCacheMap.
535  //
536 
537  if (CacheMapToFree == NULL) {
538  CcReleaseMasterLock( OldIrql );
539  SharedListOwned = FALSE;
540  goto restart;
541  }
542 
543  SharedCacheMap = CacheMapToFree;
544  CacheMapToFree = NULL;
545 
546  //
547  // Insert the new Shared Cache Map in the global list
548  //
549 
550  //
551  // Note: We do NOT use the common CcInsertIntoCleanSharedCacheMapList
552  // routine here because this shared cache map does not meet the
553  // validation conditions we check for in that routine since it is
554  // not finished being initialized.
555  //
556 
557  InsertTailList( &CcCleanSharedCacheMapList,
558  &SharedCacheMap->SharedCacheMapLinks );
559 
560  WeCreated = TRUE;
561 
562  //
563  // Finally, store the pointer to the Shared Cache Map back
564  // via the indirect pointer in the File Object.
565  //
566 
567  FileObject->SectionObjectPointer->SharedCacheMap = SharedCacheMap;
568 
569  //
570  // We must reference this file object so that it cannot go away

572  // find or rely on the FileObject that Memory Management has,
573  // although normally it will be this same one anyway.
574  //

577 
578  } else {
579 
580  //
581  // If this file has FO_SEQUENTIAL_ONLY clear, then remember that
582  // in the SharedCacheMap.
583  //
584 
585  if (!FlagOn(FileObject->Flags, FO_SEQUENTIAL_ONLY)) {
586  ClearFlag(SharedCacheMap->Flags, ONLY_SEQUENTIAL_ONLY_SEEN);
587  }
588  }
589

592  // the SharedCacheMap.
593  //
594 
595  if (FlagOn(FileObject->Flags, FO_RANDOM_ACCESS)) {
596  SetFlag(SharedCacheMap->Flags, RANDOM_ACCESS_SEEN);

614  //

616  //
617

630  //

633

698  }

703  // If this is a stream file object, then no user can map it,
704  // and we should keep the modified page writer out of it.
705  //
706

708  FSRTL_FLAG2_DO_MODIFIED_WRITE) &&
709  (FileObject->FsContext2 == NULL)) {
710

712  CcAcquireMasterLock( &OldIrql );

719  //

724  }
725  }

730  //

739  LocalSizes.AllocationSize.LowPart,
740  LocalSizes.AllocationSize.HighPart );

744  TRUE );
745

747

752  STATUS_UNEXPECTED_MM_EXTEND_ERR );
753  goto exitfinally;
754  }
755  }

758  // Extend the Vacb array.
759  //
760

763  goto exitfinally;
764  }
765  }

769  //

775  }

1100 Routine Description:
1101 
1102  This routine uninitializes the previously initialized Shared and Private
1103  Cache Maps. This routine is only intended to be called by File Systems.
1104  It should be called when the File System receives a cleanup call on the
1105  File Object.
1106 
1107  A File System which supports data caching must always call this routine
1108  whenever it closes a file, whether the caller opened the file with
1109  NO_INTERMEDIATE_BUFFERING as FALSE or not. This is because the final
1110  cleanup of a file related to truncation or deletion of the file, can
1111  only occur on the last close, whether the last closer cached the file
1112  or not. When CcUnitializeCacheMap is called on a file object for which
1113  CcInitializeCacheMap was never called, the call has a benign effect
1114  iff no one has truncated or deleted the file; otherwise the necessary
1115  cleanup relating to the truncate or close is performed.
1116 
1117  In summary, CcUnitializeCacheMap does the following:
1118 
1119  If the caller had Write or Delete access, the cache is flushed.
1120  (This could change with lazy writing.)
1121 
1122  If a Cache Map was initialized on this File Object, it is
1123  unitialized (unmap any views, delete section, and delete
1124  Cache Map structures).
1125 
1126  On the last Cleanup, if the file has been deleted, the
1127  Section is forced closed. If the file has been truncated, then
1128  the truncated pages are purged from the cache.
1129

1131

1133  CcInitializeCacheMap.
1134 
1135  TruncateSize - If specified, the file was truncated to the specified
1136  size, and the cache should be purged accordingly.
1137 
1138  UninitializeEvent - If specified, then the provided event will be set
1139  to the signaled state when the actual flush is
1140  completed. This is only of interest to file systems
1141  that require that they be notified when a cache flush
1142  operation has completed. Due to network protocol
1143  restrictions, it is critical that network file

1145  completes, by specifying this event, they can be
1146  notified when the cache section is finally purged
1147  if the section is "lazy-deleted".

1318  //

1321  !ARGUMENT_PRESENT(UninitializeEvent))
1322 
1323  ||
1324

1326 
1327  ||

1330

1332

1334

1338  //

1341  RemoveEntryList( &SharedCacheMap->SharedCacheMapLinks );
