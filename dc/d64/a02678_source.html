39 /*++
40 
41 Routine Description:
42 
43  Examines the audit queue high and low water mark values and performs
44  a general sanity check on them.
45 
46 Arguments:
47 
48  Upper - High water mark.
49 
50  Lower - Low water mark.
51 
52 Return Value:
53 
54  TRUE - values are acceptable.
55 
56  FALSE - values are unacceptable.
57 
58 
59 --*/
60 
61 {
62  PAGED_CODE();
63 
64  if ( Lower >= Upper ) {
65  return( FALSE );
66  }
67 
68  if ( Lower < 16 ) {
69  return( FALSE );
70  }
71 
72  if ( (Upper - Lower) < 16 ) {
73  return( FALSE );
74  }
75 
76  return( TRUE );
77 }

85 /*++
86 
87 Routine Description:
88 
89  Queries the registry for the high and low water mark values for the

92 
93 Arguments:
94 
95  None.
96 
97 Return Value:

99  None.
100 
101 --*/

103 {
104  NTSTATUS Status;
105  PSEP_AUDIT_BOUNDS AuditBounds;
106  UCHAR Buffer[8];
107 
108 
109  PAGED_CODE();
110 
111  Status = SepRegQueryHelper(
112  L"\\Registry\\Machine\\System\\CurrentControlSet\\Control\\Lsa",
113  L"Bounds",
114  REG_BINARY,
115  8, // 8 bytes
116  Buffer,
117  NULL
118  );
119 
120  if (!NT_SUCCESS( Status )) {
121 
122  //
123  // Didn't work, take the defaults
124  //
125 
126  return;
127  }
128 
129  AuditBounds = (PSEP_AUDIT_BOUNDS) Buffer;
130 
131  //
132  // Sanity check what we got back
133  //

135  if(SepAdtValidateAuditBounds( AuditBounds->UpperBound,
136  AuditBounds->LowerBound ))

138  //
139  // Take what we got from the registry.
140  //

158  Reads the registry to see if the user has told us to crash if an audit fails.

177

179  // Check the value of the CrashOnAudit flag in the registry.
180  //
181 
182  Status = SepRegQueryDwordValue(
183  L"\\Registry\\Machine\\System\\CurrentControlSet\\Control\\Lsa",
184  CRASH_ON_AUDIT_FAIL_VALUE,

186  );
187 
188  //
189  // If the key isn't there, don't turn on CrashOnFail.
190  //
191 
192  if (Status == STATUS_OBJECT_NAME_NOT_FOUND) {
193  return( STATUS_SUCCESS );
194  }
195 
196 
197  if (NT_SUCCESS( Status )) {
198 
199  if ( CrashOnAuditFail == LSAP_CRASH_ON_AUDIT_FAIL) {
200  SepCrashOnAuditFail = TRUE;
201  }
202  }
203 
204  return( STATUS_SUCCESS );

213 /*++
214 
215 Routine Description:
216 
217  Checks to see if there is an entry in the registry telling us to do full privilege auditing
218  (which currently means audit everything we normall audit, plus backup and restore privileges).
219 
220 Arguments:
221 
222  None
223 
224 Return Value:
225 
226  BOOLEAN - TRUE if Auditing has been initialized correctly, else FALSE.
227 
228 --*/
229 
230 {
231  HANDLE KeyHandle;
232  NTSTATUS Status;
233  NTSTATUS TmpStatus;
234  OBJECT_ATTRIBUTES Obja;
235  ULONG ResultLength;
236  UNICODE_STRING KeyName;
237  UNICODE_STRING ValueName;
238  CHAR KeyInfo[sizeof(KEY_VALUE_PARTIAL_INFORMATION) + sizeof(BOOLEAN)];
239  PKEY_VALUE_PARTIAL_INFORMATION pKeyInfo;
240  BOOLEAN Verbose;
241 
242  PAGED_CODE();
243 
244  //
245  // Query the registry to set up the privilege auditing filter.
246  //
247 
248  RtlInitUnicodeString( &KeyName, L"\\Registry\\Machine\\System\\CurrentControlSet\\Control\\Lsa");
249 
250  InitializeObjectAttributes( &Obja,
251  &KeyName,
252  OBJ_CASE_INSENSITIVE,
253  NULL,
254  NULL
255  );
256 
257  Status = NtOpenKey(
258  &KeyHandle,
259  KEY_QUERY_VALUE | KEY_SET_VALUE,
260  &Obja
261  );
262 
263 
264  if (!NT_SUCCESS( Status )) {
265 
266  if (Status == STATUS_OBJECT_NAME_NOT_FOUND) {
267 
268  return ( SepInitializePrivilegeFilter( FALSE ));
269 
270  } else {
271 
272  return( FALSE );
273  }
274  }
275 
276  RtlInitUnicodeString( &ValueName, FULL_PRIVILEGE_AUDITING );
277 
278  Status = NtQueryValueKey(
279  KeyHandle,
280  &ValueName,
281  KeyValuePartialInformation,
282  KeyInfo,
283  sizeof(KeyInfo),
284  &ResultLength
285  );
286 
287  TmpStatus = NtClose(KeyHandle);
288  ASSERT(NT_SUCCESS(TmpStatus));
289 
290  if (!NT_SUCCESS( Status )) {
291 
292  Verbose = FALSE;
293 
294  } else {
295 
296  pKeyInfo = (PKEY_VALUE_PARTIAL_INFORMATION)KeyInfo;
297  Verbose = (BOOLEAN) *(pKeyInfo->Data);
298  }
299 
300  return ( SepInitializePrivilegeFilter( Verbose ));
301 }
