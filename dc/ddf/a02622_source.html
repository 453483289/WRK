26 ; EXCEPTION_DISPOSITION
27 ; RtlpExceptionHandler (
28 ; IN PEXCEPTION_RECORD ExceptionRecord,
29 ; IN PVOID EstablisherFrame,
30 ; IN OUT PCONTEXT ContextRecord,
31 ; IN OUT PVOID DispatcherContext
32 ; )
33 ;
34 ; Routine Description:
35 ;
36 ; This function is called when a nested exception occurs. Its function
37 ; is to retrieve the establisher frame pointer from its establisher's
38 ; call frame, store this information in the dispatcher context record,
39 ; and return a disposition value of nested exception.
40 ;
41 ; Arguments:
42 ;
43 ; ExceptionRecord (rcx) - Supplies a pointer to an exception record.
44 ;
45 ; EstablisherFrame (rdx) - Supplies the frame pointer of the establisher
46 ; of this exception handler.
47 ;
48 ; ContextRecord (r8) - Supplies a pointer to a context record.
49 ;
50 ; DispatcherContext (r9) - Supplies a pointer to the dispatcher context
51 ; record.
52 ;
53 ; Return Value:
54 ;
55 ; A disposition value nested exception is returned if an unwind is not in
56 ; progress. Otherwise a value of continue search is returned.
57 ;
58 ;--
59 
60 EhFrame struct
61  P1Home dq ? ; parameter home addresses for
62  P2Home dq ? ; called functions
63  P3Home dq ? ;
64  P4Home dq ? ;
65  Context dq ? ; saved dispatcher context address
66 EhFrame ends

68  LEAF_ENTRY RtlpExceptionHandler, _TEXT$00
69 
70  mov eax, ExceptionContinueSearch ; assume unwind in progress
71  test dword ptr ErExceptionFlags[rcx], EXCEPTION_UNWIND ; check for unwind
72  jnz short Eh10 ; if nz, unwind in progress
73  mov rax, EhFrame.Context[rdx] ; get establisher context address
74  mov rax, DcEstablisherFrame[rax] ; copy the establisher frame
75  mov DcEstablisherFrame[r9], rax ; to current dispatcher context
76  mov eax, ExceptionNestedException ; set nested exception disposition
77 eH10: ret ; return
78 
79  LEAF_END RtlpExceptionHandler, _TEXT$00

108 ;
109 ; DispatcherContext (r9) - Supplies a pointer to the dispatcher context
110 ; record.
111 ;

115 ; returned as the function value.
116 ;
117 ;--
