48 /*++
49 
50 Routine Description:
51 
52  This encapsulates extracting and translating the mode bits from
53  the passed file object, to be returned from a query information call.
54 
55 Arguments:
56 
57  FileObject - Specifies the file object for which to return Mode info.
58 
59 Return Value:
60 
61  The translated mode information is returned.
62 
63 --*/
64 
65 {
66  ULONG mode = 0;
67 
68  if (FileObject->Flags & FO_WRITE_THROUGH) {
69  mode = FILE_WRITE_THROUGH;
70  }
71  if (FileObject->Flags & FO_SEQUENTIAL_ONLY) {
72  mode |= FILE_SEQUENTIAL_ONLY;
73  }
74  if (FileObject->Flags & FO_NO_INTERMEDIATE_BUFFERING) {
75  mode |= FILE_NO_INTERMEDIATE_BUFFERING;
76  }
77  if (FileObject->Flags & FO_SYNCHRONOUS_IO) {
78  if (FileObject->Flags & FO_ALERTABLE_IO) {
79  mode |= FILE_SYNCHRONOUS_IO_ALERT;
80  } else {
81  mode |= FILE_SYNCHRONOUS_IO_NONALERT;
82  }
83  }
84  if (FileObject->Flags & FO_DELETE_ON_CLOSE) {
85  mode |= FILE_DELETE_ON_CLOSE;
86  }

99 /*++
100 
101 Routine Description:
102 
103  This service returns the requested information about a specified file.
104  The information returned is determined by the FileInformationClass that
105  is specified, and it is placed into the caller's FileInformation buffer.
106 
107 Arguments:
108 
109  FileHandle - Supplies a handle to the file about which the requested
110  information should be returned.
111 
112  IoStatusBlock - Address of the caller's I/O status block.
113 
114  FileInformation - Supplies a buffer to receive the requested information
115  returned about the file.
116 
117  Length - Supplies the length, in bytes, of the FileInformation buffer.
118 
119  FileInformationClass - Specifies the type of information which should be
120  returned about the file.
121 
122 Return Value:
123 
124  The status returned is the final completion status of the operation.

147  //

153

156  // information about the file.
157  //

167  // to be performed.
168  //

171  return STATUS_INFO_LENGTH_MISMATCH;
172  }
173

175  // The caller's access mode is not kernel so probe each of the arguments
176  // and capture them as necessary. If any failures occur, the condition
177  // handler will be invoked to handle them. It will simply cleanup and
178  // return an access violation status code back to the system service
179  // dispatcher.
180  //

192  //

196 #elif defined(_WIN64)
197

199  // If we are a wow64 process, follow the X86 rules
200  //
201 
202  if (PsGetCurrentProcessByThread(CurrentThread)->Wow64Process) {
203  ProbeForWrite( FileInformation, Length, sizeof( ULONG ) );
204  } else {
205  ProbeForWrite( FileInformation,
206  Length,
207  IopQuerySetAlignmentRequirement[FileInformationClass] );
208  }
209 #else
210  ProbeForWrite( FileInformation,
211  Length,
212  IopQuerySetAlignmentRequirement[FileInformationClass] );
213 #endif
214 
215  } except(EXCEPTION_EXECUTE_HANDLER) {
216 
217  //
218  // An exception was incurred while probing the caller's
219  // parameters. Simply return an appropriate error status
220  // code.
221  //
222 
223 
224  return GetExceptionCode();
225  }
226 
227 #if DBG
228 
229  } else {
230 
231  //
232  // The caller's mode is kernel. Ensure that at least the information
233  // class and lengths are appropriate.
234  //
235 
236  if ((ULONG) FileInformationClass >= FileMaximumInformation ||
237  !IopQueryOperationLength[FileInformationClass]) {
238  return STATUS_INVALID_INFO_CLASS;
239  }
240 
241  if (Length < (ULONG) IopQueryOperationLength[FileInformationClass]) {
242  return STATUS_INFO_LENGTH_MISMATCH;
243  }
244 
245 #endif // DBG
246 
247  }
248 
249  //
250  // There were no blatant errors so far, so reference the file object so
251  // the target device object can be found. Note that if the handle does
252  // not refer to a file object, or if the caller does not have the required
253  // access to the file, then it will fail.
254  //
255 
256  status = ObReferenceObjectByHandle( FileHandle,
257  IopQueryOperationAccess[FileInformationClass],
258  IoFileObjectType,
259  requestorMode,
260  (PVOID *) &fileObject,
261  &handleInformation);
262 
263  if (!NT_SUCCESS( status )) {
264  return status;
265  }
266 
267  //
268  // Get the address of the target device object. If this file represents
269  // a device that was opened directly, then simply use the device or its
270  // attached device(s) directly. Also get the address of the Fast Io
271  // dispatch structure.
272  //
273 
274  if (!(fileObject->Flags & FO_DIRECT_DEVICE_OPEN)) {
275  deviceObject = IoGetRelatedDeviceObject( fileObject );
276  } else {
277  deviceObject = IoGetAttachedDevice( fileObject->DeviceObject );
278  }
279  fastIoDispatch = deviceObject->DriverObject->FastIoDispatch;
280 
281  //
282  // Make a special check here to determine whether this is a synchronous
283  // I/O operation. If it is, then wait here until the file is owned by
284  // the current thread. If this is not a (serialized) synchronous I/O
285  // operation, then allocate and initialize the local event.
286  //
287 
288  if (fileObject->Flags & FO_SYNCHRONOUS_IO) {
289 
290  BOOLEAN interrupted;
291 
292  if (!IopAcquireFastLock( fileObject )) {
293  status = IopAcquireFileObjectLock( fileObject,
294  requestorMode,
295  (BOOLEAN) ((fileObject->Flags & FO_ALERTABLE_IO) != 0),
296  &interrupted );
297  if (interrupted) {
298  ObDereferenceObject( fileObject );
299  return status;
300  }
301  }
302 
303  //
304  // Make a special check here to determine whether or not the caller
305  // is attempting to query the file position pointer. If so, then
306  // return it immediately and get out.
307  //
308 
309  if (FileInformationClass == FilePositionInformation) {
310 
311  //
312  // The caller has requested the current file position context
313  // information. This is a relatively frequent call, so it is
314  // optimized here to cut through the normal IRP path.
315  //
316  // Begin by establishing a condition handler and attempting to
317  // return both the file position information as well as the I/O
318  // status block. If writing the output buffer fails, then return
319  // an appropriate error status code. If writing the I/O status
320  // block fails, then ignore the error. This is what would
321  // normally happen were everything to go through normal special
322  // kernel APC processing.
323  //
324 
325  BOOLEAN writingBuffer = TRUE;
326  PFILE_POSITION_INFORMATION fileInformation = FileInformation;
327 
328  try {
329 
330  //
331  // Return the current position information.
332  //
333 
334  fileInformation->CurrentByteOffset = fileObject->CurrentByteOffset;
335  writingBuffer = FALSE;
336 
337  //
338  // Write the I/O status block.
339  //
340 
341  IoStatusBlock->Status = STATUS_SUCCESS;
342  IoStatusBlock->Information = sizeof( FILE_POSITION_INFORMATION );
343 
344  } except( EXCEPTION_EXECUTE_HANDLER ) {
345 
346  //
347  // One of writing the caller's buffer or writing the I/O
348  // status block failed. Set the final status appropriately.
349  //
350 
351  if (writingBuffer) {
352  status = GetExceptionCode();
353  }
354 
355  }
356 
357  //
358  // Note that the state of the event in the file object has not yet
359  // been reset, so it need not be set either. Therefore, simply
360  // cleanup and return.
361  //
362 
363  IopReleaseFileObjectLock( fileObject );
364  ObDereferenceObject( fileObject );
365  return status;
366 
367  //
368  // Also do a special check if the caller it doing a query for basic or
369  // standard information and if so then try the fast query calls if they
370  // exist.
371  //
372 
373  } else if (fastIoDispatch &&
374  (((FileInformationClass == FileBasicInformation) &&
375  fastIoDispatch->FastIoQueryBasicInfo) ||
376  ((FileInformationClass == FileStandardInformation) &&
377  fastIoDispatch->FastIoQueryStandardInfo))) {
378 
379  IO_STATUS_BLOCK localIoStatus;
380  BOOLEAN queryResult = FALSE;
381  BOOLEAN writingStatus = FALSE;
382 
383  //
384  // Do the query and setting of the IoStatusBlock inside an exception
385  // handler. Note that if an exception occurs, other than writing
386  // the status back, then the IRP route will be taken. If an error
387  // occurs attempting to write the status back to the caller's buffer
388  // then it will be ignored, just as it would be on the long path.
389  //
390 
391  try {
392 
393  if (FileInformationClass == FileBasicInformation) {
394  queryResult = fastIoDispatch->FastIoQueryBasicInfo( fileObject,
395  TRUE,
396  FileInformation,
397  &localIoStatus,
398  deviceObject );
399  } else {
400  queryResult = fastIoDispatch->FastIoQueryStandardInfo( fileObject,
401  TRUE,
402  FileInformation,
403  &localIoStatus,
404  deviceObject );
405  }
406 
407  if (queryResult) {
408  status = localIoStatus.Status;
409  writingStatus = TRUE;
410  *IoStatusBlock = localIoStatus;
411  }
412 
413  } except( EXCEPTION_EXECUTE_HANDLER ) {
414 
415  //
416  // If the result of the preceding block is an exception that
417  // occurred after the Fast I/O path itself, then the query
418  // actually succeeded so everything is done already, but the
419  // user's I/O status buffer is not writable. This case is
420  // ignored to be consistent w/the long path.
421  //
422 
423  if (!writingStatus) {
424  status = GetExceptionCode();
425  }
426  }
427 
428  //
429  // If the results of the preceding statement block is true, then
430  // the fast query call succeeded, so simply cleanup and return.
431  //
432 
433  if (queryResult) {
434 
435  //
436  // Note that once again, the event in the file object has not
437  // yet been set reset, so it need not be set to the Signaled
438  // state, so simply cleanup and return.
439  //
440 
441  IopReleaseFileObjectLock( fileObject );
442  ObDereferenceObject( fileObject );
443  return status;
444  }
445  }
446  synchronousIo = TRUE;
447  } else {
448 
449  //
450  // This is a synchronous API being invoked for a file that is opened
451  // for asynchronous I/O. This means that this system service is
452  // to synchronize the completion of the operation before returning
453  // to the caller. A local event is used to do this.
454  //
455 
456  event = ExAllocatePool( NonPagedPool, sizeof( KEVENT ) );
457  if (event == NULL) {
458  ObDereferenceObject( fileObject );
459  return STATUS_INSUFFICIENT_RESOURCES;
460  }
461  KeInitializeEvent( event, SynchronizationEvent, FALSE );
462  synchronousIo = FALSE;
463  }
464 
465  //
466  // Set the file object to the Not-Signaled state.
467  //
468 
469  KeClearEvent( &fileObject->Event );
470 
471  //
472  // Allocate and initialize the I/O Request Packet (IRP) for this operation.
473  // The allocation is performed with an exception handler in case the
474  // caller does not have enough quota to allocate the packet.
475  //
476 
477  irp = IoAllocateIrp( deviceObject->StackSize, FALSE );
478  if (!irp) {
479 
480  //
481  // An IRP could not be allocated. Cleanup and return an appropriate
482  // error status code.
483  //
484 
485  if (!(fileObject->Flags & FO_SYNCHRONOUS_IO)) {
486  ExFreePool( event );
487  }
488 
489  IopAllocateIrpCleanup( fileObject, (PKEVENT) NULL );
490 
491  return STATUS_INSUFFICIENT_RESOURCES;
492  }
493  irp->Tail.Overlay.OriginalFileObject = fileObject;
494  irp->Tail.Overlay.Thread = CurrentThread;
495  irp->RequestorMode = requestorMode;
496 
497  //
498  // Fill in the service independent parameters in the IRP.
499  //
500 
501  if (synchronousIo) {
502  irp->UserEvent = (PKEVENT) NULL;
503  irp->UserIosb = IoStatusBlock;
504  } else {
505  irp->UserEvent = event;
506  irp->UserIosb = &localIoStatus;
507  irp->Flags = IRP_SYNCHRONOUS_API;
508  }
509  irp->Overlay.AsynchronousParameters.UserApcRoutine = (PIO_APC_ROUTINE) NULL;
510 
511  //
512  // Get a pointer to the stack location for the first driver. This will be
513  // used to pass the original function codes and parameters.
514  //
515 
516  irpSp = IoGetNextIrpStackLocation( irp );
517  irpSp->MajorFunction = IRP_MJ_QUERY_INFORMATION;
518  irpSp->FileObject = fileObject;
519 
520  //
521  // Allocate a buffer which should be used to put the information into by
522  // the driver. This will be copied back to the caller's buffer when the
523  // service completes. This is done by setting the flag which says that
524  // this is an input operation.
525  //
526 
527  irp->UserBuffer = FileInformation;
528  irp->AssociatedIrp.SystemBuffer = (PVOID) NULL;
529  irp->MdlAddress = (PMDL) NULL;

531  try {
532 
533  //
534  // Allocate the system buffer using an exception handler so that
535  // errors can be caught and handled.
536  //
537 
538  irp->AssociatedIrp.SystemBuffer = ExAllocatePoolWithQuota( NonPagedPool,
539  Length );
540  } except(EXCEPTION_EXECUTE_HANDLER) {
541 
542  //
543  // An exception was incurred by attempting to allocate the intermediary
544  // system buffer. Cleanup everything and return an appropriate error
545  // status code.
546  //
547 
548  IopExceptionCleanup( fileObject,
549  irp,
550  (PKEVENT) NULL,
551  event );
552 
553  return GetExceptionCode();
554  }

556  irp->Flags |= IRP_BUFFERED_IO |
557  IRP_DEALLOCATE_BUFFER |
558  IRP_INPUT_OPERATION |
559  IRP_DEFER_IO_COMPLETION;

561  //
562  // Copy the caller's parameters to the service-specific portion of the
563  // IRP.
564  //
565

570  // Insert the packet at the head of the IRP list for the thread.
571  //
572

575  //
576  // Update the operation count statistic for the current process for
577  // operations other than read and write.
578  //

583  // Everything is now set to invoke the device driver with this request.
584  // However, it is possible that the information that the caller wants
585  // is device independent. If this is the case, then the request can
586  // be satisfied here without having to have all of the drivers implement

588  // the I/O completion code requires it.

599  //

605  //

622  //

632  // Return the alignment information for this file.
633  //

636 
637  //
638  // Complete the I/O operation.
639  //
640

642  skipDriver = TRUE;
643 
644  } else if (FileInformationClass == FileAllInformation) {
645 
646  PFILE_ALL_INFORMATION allInformation = irp->AssociatedIrp.SystemBuffer;
647 
648  //
649  // The caller has requested all of the information about the file.
650  // This request is handled specially because the service will fill
651  // in the Access and Mode and Alignment information in the buffer
652  // and then pass the buffer to the driver to fill in the remainder.
653  //
654  // Begin by returning the Access information for the file.
655  //
656

658  handleInformation.GrantedAccess;
659 
660  //
661  // Return the mode information for this file.
662  //
663 
664  allInformation->ModeInformation.Mode =
665  IopGetModeInformation( fileObject );
666 
667  //
668  // Return the alignment information for this file.
669  //
670 
671  allInformation->AlignmentInformation.AlignmentRequirement =
672  deviceObject->AlignmentRequirement;
673 
674  //
675  // Finally, set the information field of the IoStatus block in the IRP
676  // to account for the amount information already filled in and invoke
677  // the driver to fill in the remainder.
678  //
679 
680  irp->IoStatus.Information = sizeof( FILE_ACCESS_INFORMATION ) +
681  sizeof( FILE_MODE_INFORMATION ) +
682  sizeof( FILE_ALIGNMENT_INFORMATION );
683  }
684 
685  if (skipDriver) {
686 
687  //
688  // The requested operation has already been performed. Simply
689  // set the final status in the packet and the return state.
690  //
691 
692  status = STATUS_SUCCESS;
693  irp->IoStatus.Status = STATUS_SUCCESS;
694 
695  } else {
696 
697  //
698  // This is not a request that can be [completely] performed here, so
699  // invoke the driver at its appropriate dispatch entry with the IRP.
700  //
701 
702  status = IoCallDriver( deviceObject, irp );
703  }
704 
705  //
706  // If this operation was a synchronous I/O operation, check the return
707  // status to determine whether or not to wait on the file object. If
708  // the file object is to be waited on, wait for the operation to complete
709  // and obtain the final status from the file object itself.
710  //
711 
712  if (status == STATUS_PENDING) {
713 
714  if (synchronousIo) {
715 
716  status = KeWaitForSingleObject( &fileObject->Event,
717  Executive,
718  requestorMode,
719  (BOOLEAN) ((fileObject->Flags & FO_ALERTABLE_IO) != 0),
720  (PLARGE_INTEGER) NULL );
721 
722  if (status == STATUS_ALERTED || status == STATUS_USER_APC) {
723 
724  //
725  // The wait request has ended either because the thread was
726  // alerted or an APC was queued to this thread, because of
727  // thread rundown or CTRL/C processing. In either case, try
728  // to bail out of this I/O request carefully so that the IRP
729  // completes before this routine exists so that synchronization
730  // with the file object will remain intact.
731  //
732 
733  IopCancelAlertedRequest( &fileObject->Event, irp );
734 
735  }
736 
737  status = fileObject->FinalStatus;
738 
739  IopReleaseFileObjectLock( fileObject );
740 
741  } else {
742 
743  //
744  // This is a normal synchronous I/O operation, as opposed to a
745  // serialized synchronous I/O operation. For this case, wait for
746  // the local event and copy the final status information back to
747  // the caller.
748  //
749 
750  status = KeWaitForSingleObject( event,
751  Executive,
752  requestorMode,
753  FALSE,
754  (PLARGE_INTEGER) NULL );
755 
756  if (status == STATUS_ALERTED || status == STATUS_USER_APC) {
757 
758  //
759  // The wait request has ended either because the thread was
760  // alerted or an APC was queued to this thread, because of
761  // thread rundown or CTRL/C processing. In either case, try
762  // to bail out of this I/O request carefully so that the IRP
763  // completes before this routine exists or the event will not
764  // be around to set to the Signaled state.
765  //
766 
767  IopCancelAlertedRequest( event, irp );
768 
769  }
770 
771  status = localIoStatus.Status;
772 
773  try {
774 
775  *IoStatusBlock = localIoStatus;
776 
777  } except(EXCEPTION_EXECUTE_HANDLER) {
778 
779  //
780  // An exception occurred attempting to write the caller's I/O
781  // status block. Simply change the final status of the operation
782  // to the exception code.
783  //
784 
785  status = GetExceptionCode();
786  }
787 
788  ExFreePool( event );
789 
790  }
791 
792  } else {
793 
794  //
795  // The I/O operation finished without return a status of pending.
796  // This means that the operation has not been through I/O completion,
797  // so it must be done here.
798  //
799 
800  PKNORMAL_ROUTINE normalRoutine;
801  PVOID normalContext;
802  KIRQL irql;
803 
804  if (!synchronousIo) {
805 
806  //
807  // This is not a synchronous I/O operation, it is a synchronous
808  // I/O API to a file opened for asynchronous I/O. Since this
809  // code path need never wait on the allocated and supplied event,
810  // get rid of it so that it doesn't have to be set to the
811  // Signaled state by the I/O completion code.
812  //
813 
814  irp->UserEvent = (PKEVENT) NULL;
815  ExFreePool( event );
816  }
817 
818  irp->UserIosb = IoStatusBlock;
819  KeRaiseIrql( APC_LEVEL, &irql );
