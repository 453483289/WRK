856  *NextVaToQuery = (PVOID)((PCHAR)Va + PAGE_SIZE);
857 
858  do {
859 
860  if (!MiDoesPxeExistAndMakeValid (PointerPxe,
861  TargetProcess,
862  MM_NOIRQL,
863  &Waited)) {
864 
865 #if (_MI_PAGING_LEVELS >= 4)
866  NextVa = MiGetVirtualAddressMappedByPte (PointerPxe + 1);
867  NextVa = MiGetVirtualAddressMappedByPte (NextVa);
868  NextVa = MiGetVirtualAddressMappedByPte (NextVa);
869  *NextVaToQuery = MiGetVirtualAddressMappedByPte (NextVa);
870 #endif
871  break;
872  }
873 
874 #if (_MI_PAGING_LEVELS >= 4)
875  Waited = 0;
876 #endif
877 
878  if (!MiDoesPpeExistAndMakeValid (PointerPpe,
879  TargetProcess,
880  MM_NOIRQL,
881  &Waited)) {
882 #if (_MI_PAGING_LEVELS >= 3)
883  NextVa = MiGetVirtualAddressMappedByPte (PointerPpe + 1);
884  NextVa = MiGetVirtualAddressMappedByPte (NextVa);
885  *NextVaToQuery = MiGetVirtualAddressMappedByPte (NextVa);
886 #endif
887  break;
888  }
889 
890 #if (_MI_PAGING_LEVELS < 4)
891  Waited = 0;
892 #endif
893 
894  if (!MiDoesPdeExistAndMakeValid (PointerPde,
895  TargetProcess,
896  MM_NOIRQL,
897  &Waited)) {
898  NextVa = MiGetVirtualAddressMappedByPte (PointerPde + 1);
899  *NextVaToQuery = MiGetVirtualAddressMappedByPte (NextVa);
900  break;
901  }
902 
903  if (Waited == 0) {
904  PteDetected = TRUE;
905  }
906 
907  } while (Waited != 0);
908 
909  if (PteDetected == TRUE) {
910 
911  //
912  // A PTE exists at this address, see if it is zero.
913  //
914 
915  if (MI_PDE_MAPS_LARGE_PAGE (PointerPde)) {
916 
917  *ReturnedProtect = MI_CONVERT_FROM_PTE_PROTECTION (Vad->u.VadFlags.Protection);
918  NextVa = MiGetVirtualAddressMappedByPte (PointerPde + 1);
919  *NextVaToQuery = MiGetVirtualAddressMappedByPte (NextVa);
920  return MEM_COMMIT;
921  }
922 
923  PteContents = *PointerPte;
924 
925  if (PteContents.u.Long != 0) {
926 
927  PteIsZero = FALSE;
928 
929  //
930  // There is a non-zero PTE at this address, use
931  // it to build the information block.
932  //
933 
934  if (MiIsPteDecommittedPage (&PteContents)) {
935  ASSERT (Protect == 0);
936  ASSERT (State == MEM_RESERVE);
937  }
938  else {
939  State = MEM_COMMIT;
940  if (Vad->u.VadFlags.VadType == VadDevicePhysicalMemory) {
941 
942  //
943  // Physical mapping, there is no corresponding
944  // PFN element to get the page protection from.
945  //
946 
947  Protect = MI_CONVERT_FROM_PTE_PROTECTION (
948  Vad->u.VadFlags.Protection);
949  }
950  else if ((Vad->u.VadFlags.VadType == VadRotatePhysical) &&
951  (PteContents.u.Hard.Valid)) {
952 
953  ProtoPte = MiGetProtoPteAddress (Vad, MI_VA_TO_VPN (Va));
954 
955  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (&PteContents);
956  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
957 
958  if ((!MI_IS_PFN (PageFrameIndex)) ||
959  (Pfn1->PteAddress != ProtoPte)) {
960 
961  //
962  // This address in the view is currently pointing at
963  // the frame buffer (video RAM or AGP), protection is
964  // in the prototype (which may itself be paged out - so
965  // release the process working set pushlock before
966  // accessing it).
967  //
968 
969  Thread = PsGetCurrentThread ();
970 
971  UNLOCK_WS_SHARED (Thread, TargetProcess);
972 
973  CapturedProtoPte = *ProtoPte;
974 
975  ASSERT (CapturedProtoPte.u.Hard.Valid == 0);
976  ASSERT (CapturedProtoPte.u.Soft.Prototype == 0);
977 
978  //
979  // PTE is either demand zero, pagefile or transition,
980  // in all cases protection is in the prototype PTE.
981  //
982 
983  Protect = MI_CONVERT_FROM_PTE_PROTECTION (CapturedProtoPte.u.Soft.Protection);
984 
985  LOCK_WS_SHARED (Thread, TargetProcess);
986  }
987  else {
988  ASSERT (Pfn1->u3.e1.PrototypePte == 1);
989  Protect = MI_CONVERT_FROM_PTE_PROTECTION (
990  Pfn1->OriginalPte.u.Soft.Protection);
991  }
992  }
993  else if (Vad->u.VadFlags.VadType == VadAwe) {
994 
995  //
996  // This is an AWE frame - the original PTE field in the PFN
997  // actually contains the AweReferenceCount. These pages are
998  // either noaccess, readonly or readwrite.
999  //
1000 
1001  if (PteContents.u.Hard.Owner == MI_PTE_OWNER_KERNEL) {
1002  Protect = PAGE_NOACCESS;
1003  }
1004  else if (PteContents.u.Hard.Write == 1) {
1005  Protect = PAGE_READWRITE;
1006  }
1007  else {
1008  Protect = PAGE_READONLY;
1009  }
1010  State = MEM_COMMIT;
1011  }
1012  else {
1013 
1014  Protect = MiGetPageProtection (PointerPte);
1015 
1016  //
1017  // The PointerPte contents may have changed if
1018  // MiGetPageProtection had to release the working set
1019  // pushlock, thus we snapped it prior and can continue
1020  // to use the old copy regardless. Note also that the
1021  // page table page containing PointerPte may now have
1022  // been paged out as well.
1023  //
1024 
1025  if ((PteContents.u.Soft.Valid == 0) &&
1026  (PteContents.u.Soft.Prototype == 1) &&
1027  (Vad->u.VadFlags.PrivateMemory == 0) &&
1028  (Vad->ControlArea != NULL)) {
1029 
1030  //
1031  // Make sure the prototype PTE is committed. Note
1032  // that the prototype PTE itself may be paged out,
1033  // thus the process working set pushlock must be
1034  // released before accessing it.
1035  //
1036 
1037  ProtoPte = MiGetProtoPteAddress (Vad,
1038  MI_VA_TO_VPN (Va));
1039 
1040  if (ProtoPte != NULL) {
1041  Thread = PsGetCurrentThread ();
1042  UNLOCK_WS_SHARED (Thread, TargetProcess);
1043  CapturedProtoPte = *ProtoPte;
1044  LOCK_WS_SHARED (Thread, TargetProcess);
1045  }
1046  else {
1047  CapturedProtoPte.u.Long = 0;
1048  }
1049 
1050  if (CapturedProtoPte.u.Long == 0) {
1051  State = MEM_RESERVE;
1052  Protect = 0;
1053  }
1054  }
1055  }
1056  }
1057  }
1058  }
1059 
1060  if (PteIsZero) {
1061 
1062  //
1063  // There is no PDE at this address, the template from
1064  // the VAD supplies the information unless the VAD is
1065  // for an image file. For image files the individual
1066  // protection is on the prototype PTE.
1067  //
1068 
1069  //
1070  // Get the default protection information.
1071  //
1072 
1073  State = MEM_RESERVE;
1074  Protect = 0;
1075 
1076  if (Vad->u.VadFlags.VadType == VadAwe) {
1077  NOTHING;
1078  }
1079  else if (Vad->u.VadFlags.VadType == VadDevicePhysicalMemory) {
1080 
1081  //
1082  // Must be banked memory, just return reserved.
1083  //
1084 
1085  NOTHING;
1086  }
1087  else if ((Vad->u.VadFlags.PrivateMemory == 0) &&
1088  (Vad->ControlArea != NULL)) {
1089 
1090  //
1091  // This VAD refers to a section. Even though the PTE is
1092  // zero, the actual page may be committed in the section.
1093  //
1094 
1095  *NextVaToQuery = (PVOID)((PCHAR)Va + PAGE_SIZE);
1096 
1097  ProtoPte = MiGetProtoPteAddress (Vad, MI_VA_TO_VPN (Va));
1098 
1099  if (ProtoPte != NULL) {
1100 
1101  Thread = PsGetCurrentThread ();
1102 
1103  UNLOCK_WS_SHARED (Thread, TargetProcess);
1104 
1105  CapturedProtoPte = *ProtoPte;
1106 
1107  if (CapturedProtoPte.u.Long != 0) {
1108  State = MEM_COMMIT;
1109 
1110  if (Vad->u.VadFlags.VadType != VadImageMap) {
1111  Protect = MI_CONVERT_FROM_PTE_PROTECTION (
1112  Vad->u.VadFlags.Protection);
1113  }
1114  else {
1115 
1116  //
1117  // This is an image file, the protection is in the
1118  // prototype PTE.
1119  //
1120 
1121  if (CapturedProtoPte.u.Hard.Valid == 0) {
1122  Protect = MI_CONVERT_FROM_PTE_PROTECTION (CapturedProtoPte.u.Soft.Protection);
1123  }
1124  else {
1125 
1126  //
1127  // The prototype PTE is valid but not in this
1128  // process (at least not before we released the
1129  // working set pushlock above). So the PFN's
1130  // OriginalPte field contains the correct
1131  // protection value, but it may get trimmed at any
1132  // time. Acquire the PFN lock so we can examine
1133  // it safely.
1134  //
1135 
1136  PointerPde = MiGetPteAddress (ProtoPte);
1137 
1138  LOCK_PFN (OldIrql);
1139 
1140  if (PointerPde->u.Hard.Valid == 0) {
1141  MiMakeSystemAddressValidPfn (ProtoPte, OldIrql);
1142  }
1143 
1144  CapturedProtoPte = *ProtoPte;
1145 
1146  ASSERT (CapturedProtoPte.u.Long != 0);
1147 
1148  if (CapturedProtoPte.u.Hard.Valid) {
1149  Pfn1 = MI_PFN_ELEMENT (CapturedProtoPte.u.Hard.PageFrameNumber);
1150 
1151  Protect = MI_CONVERT_FROM_PTE_PROTECTION (
1152  Pfn1->OriginalPte.u.Soft.Protection);
1153  }
1154  else {
1155  Protect = MI_CONVERT_FROM_PTE_PROTECTION (CapturedProtoPte.u.Soft.Protection);
1156  }
1157  UNLOCK_PFN (OldIrql);
1158  }
1159  }
1160  }
1161 
1162  LOCK_WS_SHARED (Thread, TargetProcess);
1163  }
1164  }
1165  else {
1166 
1167  //
1168  // Get the protection from the corresponding VAD.
1169  //
1170 
1171  if (Vad->u.VadFlags.MemCommit) {
1172  State = MEM_COMMIT;
1173  Protect = MI_CONVERT_FROM_PTE_PROTECTION (
1174  Vad->u.VadFlags.Protection);
1175  }
1176  }
1177  }
1178 
1179  *ReturnedProtect = Protect;
1180  return State;
1181 }

1318  Entry->Protection = MI_GET_PROTECTION_FROM_SOFT_PTE(&Pfn1->OriginalPte);
1319  }
1320  else {
1321  Entry->Protection = Wsle->u1.e1.Protection;
1322  }
1323  }

1325  }

1327 #if DBG

1329 #endif

1333

1335  KeUnstackDetachProcess (&ApcState);
1336  }
