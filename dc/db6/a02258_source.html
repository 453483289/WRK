203 //
204 // VOID
205 // RtlFreeBuffer(
206 // IN OUT PRTL_BUFFER Buffer,
207 // );
208 //
209 //
210 // Routine Description:
211 //
212 // Free any heap allocated storage associated with Buffer.
213 // Notes:
214 // - RtlFreeBuffer returns a buffer to the state it was in after
215 // calling RtlInitBuffer, so you may reuse it.
216 // - If you want to shrink the buffer without freeing it, just poke Buffer->Size down.
217 // This is safe regardless of if the buffer has gone heap allocated or not.
218 // - You may RtlFreeBuffer an RTL_BUFFER that is all zeros. You do not need to track if you
219 // called RtlInitBuffer if you know you filled it with zeros.
220 // - You may RtlFreeBuffer an RTL_BUFFER repeatedly.
221 //
222 // Arguments:
223 //
224 // Buffer -
225 //
226 // Return Value:
227 //
228 // none, unconditional success
229 //
230 //--
231 #define RtlFreeBuffer(Buff) \
232  do { \
233  if ((Buff) != NULL && (Buff)->Buffer != NULL) { \
234  if (RTLP_BUFFER_IS_HEAP_ALLOCATED(Buff)) { \
235  UNICODE_STRING UnicodeString; \
236  UnicodeString.Buffer = (PWSTR)(PVOID)(Buff)->Buffer; \
237  RtlFreeUnicodeString(&UnicodeString); \
238  } \
239  (Buff)->Buffer = (Buff)->StaticBuffer; \
240  (Buff)->Size = (Buff)->StaticSize; \
241  } \
242  } while (0)

255 //
256 // MAX_UNICODE_STRING_MAXLENGTH is the maximum allowed value for UNICODE_STRING::MaximumLength.
257 // MAX_UNICODE_STRING_LENGTH is the maximum allowed value for UNICODE_STRING::Length, allowing
258 // room for a terminal nul.
259 //
260 // Explanation of MAX_UNICODE_STRING_MAXLENGTH implementation
261 // ~0 is all bits set, maximum value in two's complement arithmetic, which C guarantees for unsigned types
262 // << shifts out the number of bits that fit in UNICODE_STRING::Length
263 // ~ and now we have all bits set that fit in UNICODE_STRING::Length,
264 // like if UNICODE_STRING::Length is 16 bits, we have 0xFFFF
265 // then mask so it is even multiple of whatever UNICODE_STRING::Buffer points to.
266 // If Length is changed to ULONG or SIZE_T, this macro is still correct.
267 // If Buffer pointed to CHAR or "WIDER_CHAR" or something else, this macro is still correct.
268 //
269 #define MAX_UNICODE_STRING_MAXLENGTH ((~((~(SIZE_T)0) << (RTL_FIELD_SIZE(UNICODE_STRING, Length) * CHAR_BIT))) & ~(sizeof(((PCUNICODE_STRING)0)->Buffer[0]) - 1))
270 #define MAX_UNICODE_STRING_LENGTH (MAX_UNICODE_STRING_MAXLENGTH - sizeof(((PCUNICODE_STRING)0)->Buffer[0]))

272 //++
273 //
274 // NTSTATUS
275 // RtlInitUnicodeStringBuffer(
276 // OUT PRTL_UNICODE_STRING_BUFFER Buffer,
277 // IN PUCHAR StaticBuffer,
278 // IN SIZE_T StaticSize
279 // );
280 //
281 // Routine Description:
282 //
283 //
284 // Arguments:
285 //
286 // Buffer -
287 // StaticBuffer - can be NULL, but generally is not
288 // StaticSize - should be at least sizeof(WCHAR), but can be zero
289 // ought to be an even multiple of sizeof(WCHAR)
290 // gets rounded to down to an even multiple of sizeof(WCHAR)
291 // gets clamped to MAX_UNICODE_STRING_MAXLENGTH (64k - 2)
292 //
293 // RTL_UNICODE_STRING_BUFFER contains room for the terminal nul for the
294 // case of StaticBuffer == NULL or StaticSize < sizeof(WCHAR), or, more likely,
295 // for RtlTakeRemainingStaticBuffer leaving it with no static buffer.
296 //
297 // Return Value:
298 //
299 // STATUS_SUCCESS
300 //
301 //--
302 #define RtlInitUnicodeStringBuffer(Buff, StatBuff, StatSize) \
303  do { \
304  SIZE_T TempStaticSize = (StatSize); \
305  PUCHAR TempStaticBuff = (StatBuff); \
306  TempStaticSize &= ~(sizeof((Buff)->String.Buffer[0]) - 1); \
307  if (TempStaticSize > UNICODE_STRING_MAX_BYTES) { \
308  TempStaticSize = UNICODE_STRING_MAX_BYTES; \
309  } \
310  if (TempStaticSize < sizeof(WCHAR)) { \
311  TempStaticBuff = (Buff)->MinimumStaticBufferForTerminalNul; \
312  TempStaticSize = sizeof(WCHAR); \
313  } \
314  RtlInitBuffer(&(Buff)->ByteBuffer, TempStaticBuff, TempStaticSize); \
315  (Buff)->String.Buffer = (WCHAR*)TempStaticBuff; \
316  if ((Buff)->String.Buffer != NULL) \
317  (Buff)->String.Buffer[0] = 0; \
318  (Buff)->String.Length = 0; \
319  (Buff)->String.MaximumLength = (RTL_STRING_LENGTH_TYPE)TempStaticSize; \
320  } while (0)

546 //++
547 //
548 // NTSTATUS
549 // RtlTakeRemainingStaticBuffer(
550 // IN OUT PRTL_BUFFER Buffer,
551 // OUT PUCHAR* RemainingStaticBuffer,
552 // OUT SIZE_T* RemainingStaticSize
553 // );
554 //
555 // Routine Description:
556 //
557 // This function makes it easy to share a static buffer among
558 // multiple buffers, as long as the buffers are actually initialized
559 // and "sealed" linearly/independently/one after another with
560 // no "overlap" (overlap in control and dataflow, not in actual addresses).
561 //
562 // Note that if a buffer is using exactly all of its static buffer, you
563 // will get back 0 and STATUS_SUCCESS. This is not an error condition.
564 // This is why RtlInitUnicodeStringBuffer can now accept zero sized static buffers.
565 //
566 // Note that even if you violate the conditions that make this function most
567 // useful, your code will still work, just less quickly.
568 //
569 // A pattern that should work is allocating one static buffer and moving it "through"
570 // multiple buffers. Even if you run out of static space, it should work to
571 // move the remaining zero size static buffer forward. Of course, you'll heap allocate.
572 //
573 // Arguments:
574 //
575 // Buffer -
576 // RemainingStaticBuffer -
577 // RemainingStaticSize -
578 //
579 // Return Value:
580 //
581 // STATUS_SUCCESS
582 //
583 //--
584 #define RtlTakeRemainingStaticBuffer(Buff, OutBuff, OutSize) \
585  (((Buff)->Buffer != (Buff)->StaticBuffer) \
586  ? ( /* take the whole thing */ \
587  ( *(OutBuff) = (Buff)->StaticBuffer ), \
588  ( *(OutSize) = (Buff)->StaticSize ), \
589  /* leave the buffer with nothing */ \
590  ( (Buff)->StaticBuffer = NULL ), \
591  ( (Buff)->StaticSize = 0 ), \
592  ( STATUS_SUCCESS ) \
593  ) \
594  : ( /* only take what isn't being used */ \
595  ( *(OutBuff) = &(Buff)->StaticBuffer[(Buff)->Size] ), \
596  ( *(OutSize) = ((Buff)->StaticSize - (Buff)->Size) ), \
597  /* leave the buffer with just what it is using */ \
598  ( (Buff)->StaticSize = (Buff)->Size ), \
599  ( STATUS_SUCCESS ) \
600  ))
