63 
64 Routine Description:
65 
66  This function initializes executive structures implemented by the
67  kernel.
68 
69  N.B. This function is only called during phase 1 initialization.
70 
71 Arguments:
72 
73  None.
74 
75 Return Value:
76 
77  A value of TRUE is returned if initialization is successful. Otherwise,
78  a value of FALSE is returned.
79 
80 --*/
81 
82 {
83 
84  HANDLE Handle;
85  ULONG Index;
86  ULONG Limit;
87  OBJECT_ATTRIBUTES ObjectAttributes;
88  PKPRCB Prcb;
89  NTSTATUS Status;
90 
91  //
92  // If threaded DPCs are enabled for the host system, then create a DPC
93  // thread for each processor.
94  //
95 
96  if (KeThreadDpcEnable != FALSE) {
97  Index = 0;
98  Limit = (ULONG)KeNumberProcessors;

415 ULARGE_INTEGER
416 KeComputeReciprocal (
417  IN LONG Divisor,
418  OUT PCCHAR Shift
419  )
420 
421 /*++
422 
423 Routine Description:
424 
425  This function computes the 64-bit reciprocal of the specified value.
426 
427 Arguments:
428 
429  Divisor - Supplies the value for which the large integer reciprocal is
430  computed.
431 
432  Shift - Supplies a pointer to a variable that receives the computed
433  shift count.
434 
435 Return Value:
436 
437  The 64-bit reciprocal is returned as the function value.
438 
439 --*/
440 
441 {
442 
443  LARGE_INTEGER Fraction;
444  LONG NumberBits;
445  LONG Remainder;
446 
447  //
448  // Compute the large integer reciprocal of the specified value.
449  //
450 
451  NumberBits = 0;
452  Remainder = 1;
453  Fraction.LowPart = 0;
454  Fraction.HighPart = 0;
455  while (Fraction.HighPart >= 0) {
456  NumberBits += 1;
457  Fraction.HighPart = (Fraction.HighPart << 1) | (Fraction.LowPart >> 31);
458  Fraction.LowPart <<= 1;
459  Remainder <<= 1;
460  if (Remainder >= Divisor) {
461  Remainder -= Divisor;
462  Fraction.LowPart |= 1;
463  }
464  }
465 
466  if (Remainder != 0) {
467  if ((Fraction.LowPart == 0xffffffff) && (Fraction.HighPart == 0xffffffff)) {
468  Fraction.LowPart = 0;
469  Fraction.HighPart = 0x80000000;
470  NumberBits -= 1;
471 
472  } else {
473  if (Fraction.LowPart == 0xffffffff) {
474  Fraction.LowPart = 0;
475  Fraction.HighPart += 1;
476 
477  } else {
478  Fraction.LowPart += 1;
479  }
480  }
481  }
482 
483  //
484  // Compute the shift count value and return the reciprocal fraction.
485  //
486 
487  *Shift = (CCHAR)(NumberBits - 64);
488  return *((PULARGE_INTEGER)&Fraction);
489 }

497 /*++
498 
499 Routine Description:
500 
501  This function computes the 32-bit reciprocal of the specified value.
502 
503 Arguments:
504 
505  Divisor - Supplies the value for which the large integer reciprocal is
506  computed.
507 
508  Shift - Supplies a pointer to a variable that receives the computed
509  shift count.
510

512

514

518

521  LONG Remainder;
522 
523  //
524  // Compute the 32-bit reciprocal of the specified value.
525  //
526 
527  NumberBits = 0;
528  Remainder = 1;
529  Fraction = 0;
530  while (Fraction >= 0) {
531  NumberBits += 1;
532  Fraction <<= 1;
533  Remainder <<= 1;
534  if (Remainder >= Divisor) {
535  Remainder -= Divisor;
536  Fraction |= 1;

538  }
539

542  Fraction = 0x80000000;
543  NumberBits -= 1;
544 
545  } else {
546  Fraction += 1;
547  }
548  }
549 
550  //
551  // Compute the shift count value and return the reciprocal fraction.
552  //
553 
554  *Shift = (CCHAR)(NumberBits - 32);
555  return Fraction;
