41 /*++
42 
43 Routine Description:
44 
45  This routine is called to register the correct set of data providers
46  with the Executive part of the kernel.
47 
48  Currently, the SMBIOS raw provider is supported on all platforms.
49  The Firmware provider is only supported on X86 and X64.
50 
51 Arguments:
52 
53  None
54 
55 Return Value:
56 
57  None
58 
59 --*/ {
60  SYSTEM_FIRMWARE_TABLE_HANDLER TableHandler;
61 
62  //
63  // Register the SMBIOS raw provider.
64  //
65  TableHandler.ProviderSignature = 'RSMB'; // (Raw SMBIOS)
66  TableHandler.Register = TRUE;
67  TableHandler.FirmwareTableHandler = &WmipRawSMBiosTableHandler;
68  TableHandler.DriverObject = IoPnpDriverObject;
69  NtSetSystemInformation(
70  SystemRegisterFirmwareTableInformationHandler,
71  (PVOID) &TableHandler,
72  sizeof(SYSTEM_FIRMWARE_TABLE_HANDLER)
73  );
74 
75 #if defined(_X86_) || defined(_AMD64_)
76  //
77  // Register the Firmware provider.
78  //
79  TableHandler.ProviderSignature = 'FIRM'; // (Firmware)
80  TableHandler.Register = TRUE;
81  TableHandler.FirmwareTableHandler = &WmipFirmwareTableHandler;
82  TableHandler.DriverObject = IoPnpDriverObject;
83  NtSetSystemInformation(
84  SystemRegisterFirmwareTableInformationHandler,
85  (PVOID) &TableHandler,
86  sizeof(SYSTEM_FIRMWARE_TABLE_HANDLER)
87  );
88 #endif
89 }

91 //
92 // This array defines the legal ranges of memory that can be returned by
93 // the FirmwareTableHandler. The format is starting address and length.
94 //
95 // Addresses over 4 gigabytes are specifically not included.
96 //
97 ULONG WmipFirmwareTableArray[] = {
98  0xC0000, 0x20000, // Option ROM => C:0000 - D:FFFF
99  0xE0000, 0x20000, // System ROM => E:0000 - F:FFFF
100 };

104 WmipFirmwareTableHandler(
105  PSYSTEM_FIRMWARE_TABLE_INFORMATION TableInfo
106  )
107 /*++
108 
109 Routine Description:
110 
111  This routine handles request for firmware information.
112 
113  Currently, the only requested range is the E0000-FFFFF range, but
114  we also support handing back the C0000-DFFFF range.
115 
116 Arguments:
117 
118  TableInfo - Information about the request
119 
120 Return Value:
121 
122  NTSTATUS
123 
124 --*/ {
125  PULONG Buffer;
126  PHYSICAL_ADDRESS Address;
127  PCHAR Firmware;
128  ULONG TableCount;
129  ULONG TableIndex = 0;
130  ULONG TableSize = 0;
131  ULONG i,j;
132 
133  ASSERT( TableInfo != NULL );
134 
135  if (TableInfo == NULL) {
136  return STATUS_INVALID_PARAMETER_1;
137  }
138 
139  //
140  // Calculate the number of tables supported and their length.
141  //
142  TableCount = sizeof(WmipFirmwareTableArray) / (sizeof(ULONG) * 2);
143 
144  //
145  // Handle the enumeration case.
146  //
147  if (TableInfo->Action == SystemFirmwareTable_Enumerate) {
148 
149  //
150  // We only handle the enumerate of one table.
151  //
152  TableSize = TableCount * sizeof(ULONG);
153  if (TableInfo->TableBufferLength < TableSize) {
154 
155  TableInfo->TableBufferLength = TableSize;
156  return STATUS_BUFFER_TOO_SMALL;
157 
158  }
159  TableInfo->TableBufferLength = TableSize;
160 
161  //
162  // Fill in the starting address so that they can be used as "IDs".
163  //
164  Buffer = (PULONG) TableInfo->TableBuffer;
165  for (i = j = 0; i < TableCount; i++, j += 2) {
166 
167  Buffer[i] = WmipFirmwareTableArray[j];
168 
169  }
170  return STATUS_SUCCESS;
171 
172  }
173 
174  ASSERT(TableInfo->Action == SystemFirmwareTable_Get);
175  if (TableInfo->Action != SystemFirmwareTable_Get) {
176 
177  return STATUS_INVALID_PARAMETER_1;
178 
179  }
180 
181  //
182  // Make sure that the length field is correct and that the
183  // ID requested matches one of the supported starting addresses.
184  //
185  TableIndex = TableCount;
186  for (i = j = 0; i < TableCount; i++, j += 2) {
187 
188  //
189  // Determine if the starting location matches the specified ID.
190  // If it doesn't, then no match exits.
191  //
192  if (WmipFirmwareTableArray[j] != TableInfo->TableID) {
193 
194  continue;
195 
196  }
197 
198  //
199  // We have a match, so check the size.
200  //
201  TableSize = WmipFirmwareTableArray[j+1];
202  if (TableInfo->TableBufferLength < WmipFirmwareTableArray[j+1]) {
203 
204  TableInfo->TableBufferLength = TableSize;
205  return STATUS_BUFFER_TOO_SMALL;
206 
207  }
208  TableInfo->TableBufferLength = TableSize;
209 
210  //
211  // Remember what the matching index for the table is.
212  //
213  TableIndex = j;
214  break;
215 
216  }
217 
218  //
219  // If we failed to find a match, than abort.
220  //
221  if (i == TableCount) {
222 
223  return STATUS_INVALID_PARAMETER_1;
224 
225  }
226 
227  //
228  // Map in the relevant window.
229  //
230  Address.QuadPart = WmipFirmwareTableArray[TableIndex];
231  Firmware = MmMapIoSpace( Address, TableSize, MmNonCached );
232  if (Firmware == NULL) {
233 
234  return STATUS_NOT_FOUND;
235 
236  }
237 
238  //
239  // Copy the buffer.
240  //
241  RtlCopyMemory( TableInfo->TableBuffer, Firmware, TableSize );
242 
243  //
244  // Unamp the firmware pointer.
245  //
246  MmUnmapIoSpace( Firmware, TableSize );
247  return STATUS_SUCCESS;
248 }

252 WmipRawSMBiosTableHandler(
253  PSYSTEM_FIRMWARE_TABLE_INFORMATION TableInfo
254  )
255 /*++
256 
257 Routine Description:
258 
259  This routine handles request for raw SMBIOS information.
260 
261  Currently, this returns a single table.
262 
263 Arguments:
264 
265  TableInfo - Information about the request
266 
267 Return Value:
268 
269  NTSTATUS
270 
271 --*/ {
272  NTSTATUS status;
273  PUCHAR Buffer = NULL;
274  PUCHAR SMBios = NULL;
275  PSMBIOSVERSIONINFO VersionInfo = NULL;
276  PULONG TableSizePtr = NULL;
277  ULONG HeaderSize = 0;
278  ULONG SizeNeeded = 0;
279  ULONG TableSize = 0;
280  ASSERT( TableInfo != NULL );
281 
282  if (TableInfo == NULL) {
283 
284  return STATUS_INVALID_PARAMETER_1;
285 
286  }
287 
288  if (TableInfo->Action == SystemFirmwareTable_Enumerate) {
289 
290  //
291  // We only handle the enumerate of one table.
292  //
293  if (TableInfo->TableBufferLength < sizeof(ULONG)) {
294 
295  TableInfo->TableBufferLength = sizeof(ULONG);
296  return STATUS_BUFFER_TOO_SMALL;
297 
298  }
299  TableInfo->TableBufferLength = sizeof(ULONG);
300 
301  //
302  // The "TableID" of the only table is '0'. Make sure that we
303  // indicate that only one table exists.
304  //
305  TableSizePtr = (PULONG) TableInfo->TableBuffer;
306  TableSizePtr[0] = 0;
307  return STATUS_SUCCESS;
308  }
309 
310  ASSERT(TableInfo->Action == SystemFirmwareTable_Get);
311  if (TableInfo->Action != SystemFirmwareTable_Get) {
312 
313  return STATUS_INVALID_PARAMETER_1;
314 
315  }
316 
317  //
318  // Calculate the amount of space required.
319  //
320  status = WmipGetSMBiosTableData( NULL, &TableSize, NULL );
321  if (status != STATUS_BUFFER_TOO_SMALL || TableSize == 0) {
322 
323  TableInfo->TableBufferLength = 0;
324  return status;
325 
326  }
327  HeaderSize = sizeof(SMBIOSVERSIONINFO) + sizeof(ULONG);
328  SizeNeeded = HeaderSize + TableSize;
329 
330  //
331  // If the caller didn't provide enough memory, then fail the call.
332  //
333  if (TableInfo->TableBufferLength < SizeNeeded) {
334 
335  TableInfo->TableBufferLength = SizeNeeded;
336  return STATUS_BUFFER_TOO_SMALL;
337 
338  }
339 
340  //
341  // Setup all the pointers that we will need for getting a real copy
342  // of the tables.
343  //
344  Buffer = (PUCHAR) TableInfo->TableBuffer;
345  VersionInfo = (PSMBIOSVERSIONINFO) Buffer;
346  TableSizePtr = (PULONG) (Buffer + sizeof(SMBIOSVERSIONINFO));
347  SMBios = Buffer + HeaderSize;
348 
349  //
350  // Get the full copy of the tables this time around.
351  //
352  status = WmipGetSMBiosTableData( SMBios, &TableSize, VersionInfo );
353  if (NT_SUCCESS(status)) {
354 
355  *TableSizePtr = TableSize;
356 
357  }
358  TableInfo->TableBufferLength = HeaderSize + TableSize;
359  return status;
360 }
