220 
221 Routine Description:
222 
223  Given a master boot record type (MBR - the zero'th sector on the disk),
224  read the master boot record of a disk. If the MBR is found to be of that
225  type, allocate a structure whose layout is dependent upon that partition
226  type, fill with the appropriate values, and return a pointer to that buffer
227  in the output parameter.
228 
229  The best example for a use of this routine is to support Ontrack
230  systems DiskManager software. Ontrack software lays down a special
231  partition describing the entire drive. The special partition type
232  (0x54) will be recognized and a couple of longwords of data will
233  be passed back in a buffer for a disk driver to act upon.
234 
235 Arguments:
236 
237  DeviceObject - The device object describing the entire drive.
238 
239  SectorSize - The minimum number of bytes that an IO operation can
240  fetch.
241

243  in the MBR. This routine will understand
244  the semantics implied by this value.
245 
246  Buffer - Pointer to a buffer that returns data according to the
247  type of MBR searched for. If the MBR is not of the
248  type asked for, the buffer will not be allocated and this
249  pointer will be NULL. It is the responsibility of the
250  caller of HalExamineMBR to deallocate the buffer. The
251  caller should deallocate the memory ASAP.
252 
253 Return Value:
254 
255  None.

567 VOID
568 HalpCalculateChsValues(
569  IN PLARGE_INTEGER PartitionOffset,
570  IN PLARGE_INTEGER PartitionLength,
571  IN CCHAR ShiftCount,
572  IN ULONG SectorsPerTrack,
573  IN ULONG NumberOfTracks,
574  IN ULONG ConventionalCylinders,
575  OUT PPARTITION_DESCRIPTOR PartitionDescriptor
576  )
577 
578 /*++
579 
580 Routine Description:
581 
582  This routine will determine the cylinder, head, and sector (CHS) values
583  that should be placed in a partition table entry, given the partition's
584  location on the disk and its size. The values calculated are packed into
585  int13 format -- the high two bits of the sector byte contain bits 8 and 9
586  of the 10 bit cylinder value, the low 6 bits of the sector byte contain
587  the 6 bit sector value; the cylinder byte contains the low 8 bits
588  of the cylinder value; and the head byte contains the 8-bit head value.
589  Both the start and end CHS values are calculated.
590 
591 Arguments:
592 
593  PartitionOffset - Byte offset of the partition, relative to the entire
594  physical disk.
595 
596  PartitionLength - Size in bytes of the partition.
597 
598  ShiftCount - Shift count to convert from byte counts to sector counts.
599 
600  SectorsPerTrack - Number of sectors in a track on the media on which
601  the partition resides.
602 
603  NumberOfTracks - Number of tracks in a cylinder on the media on which
604  the partition resides.
605 
606  ConventionalCylinders - The "normalized" disk cylinders. We will never
607  set the cylinders greater than this.
608 
609  PartitionDescriptor - Structure to be filled in with the start and
610  end CHS values. Other fields in the structure are not referenced
611  or modified.
612 
613 Return Value:
614 
615  None.
616 
617 Note:
618 
619  The Cylinder and Head values are 0-based but the Sector value is 1-based.
620 
621  If the start or end cylinder overflows 10 bits (ie, > 1023), CHS values
622  will be set to all 1's.
623 
624  No checking is done on the SectorsPerTrack and NumberOfTrack values.
625 
626 --*/
627 
628 {
629  ULONG startSector, sectorCount, endSector;
630  ULONG sectorsPerCylinder;
631  ULONG remainder;
632  ULONG startC, startH, startS, endC, endH, endS;
633  LARGE_INTEGER tempInt;
634 
635  PAGED_CODE();
636 
637  //
638  // Calculate the number of sectors in a cylinder. This is the
639  // number of heads multiplied by the number of sectors per track.
640  //
641 
642  sectorsPerCylinder = SectorsPerTrack * NumberOfTracks;
643 
644  //
645  // Convert byte offset/count to sector offset/count.
646  //
647 
648  tempInt.QuadPart = PartitionOffset->QuadPart >> ShiftCount;
649  startSector = tempInt.LowPart;
650 
651  tempInt.QuadPart = PartitionLength->QuadPart >> ShiftCount;
652  sectorCount = tempInt.LowPart;
653 
654  endSector = startSector + sectorCount - 1;
655 
656  startC = startSector / sectorsPerCylinder;
657  endC = endSector / sectorsPerCylinder;
658 
659  if (!ConventionalCylinders) {
660 
661  ConventionalCylinders = 1024;
662 
663  }
664 
665  //
666  // Set these values so that win95 is happy.
667  //
668 
669  if (startC >= ConventionalCylinders) {
670 
671  startC = ConventionalCylinders - 1;
672 
673  }
674 
675  if (endC >= ConventionalCylinders) {
676 
677  endC = ConventionalCylinders - 1;
678 
679  }
680 
681  //
682  // Calculate the starting track and sector.
683  //
684 
685  remainder = startSector % sectorsPerCylinder;
686  startH = remainder / SectorsPerTrack;
687  startS = remainder % SectorsPerTrack;
688 
689  //
690  // Calculate the ending track and sector.
691  //
692 
693  remainder = endSector % sectorsPerCylinder;
694  endH = remainder / SectorsPerTrack;
695  endS = remainder % SectorsPerTrack;
696 
697  //
698  // Pack the result into the caller's structure.
699  //
700 
701  // low 8 bits of the cylinder => C value
702 
703  PartitionDescriptor->StartingCylinderMsb = (UCHAR) startC;
704  PartitionDescriptor->EndingCylinderMsb = (UCHAR) endC;
705 
706  // 8 bits of head value => H value
707 
708  PartitionDescriptor->StartingTrack = (UCHAR) startH;
709  PartitionDescriptor->EndingTrack = (UCHAR) endH;
710 
711  // bits 8-9 of cylinder and 6 bits of the sector => S value
712 
713  PartitionDescriptor->StartingCylinderLsb = (UCHAR) (((startS + 1) & 0x3f)
714  | ((startC >> 2) & 0xc0));
715 
716  PartitionDescriptor->EndingCylinderLsb = (UCHAR) (((endS + 1) & 0x3f)
717  | ((endC >> 2) & 0xc0));
718 }

824 
825  if (partInfo.Mbr.BootIndicator) {
826  *PartitionType = BOOTABLE_PARTITION;
827  } else {
828  *PartitionType = PRIMARY_PARTITION;
829  }
830 
831  return STATUS_SUCCESS;
832  }
833  }
834 
835  *PartitionType = LOGICAL_PARTITION;
836
