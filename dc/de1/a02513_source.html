35 /*++
36 
37 Routine Description:
38 
39  This routine creates an address space which maps the system
40  portion and contains a hyper space entry.
41 
42 Arguments:
43 
44  MinimumWorkingSetSize - Supplies the minimum working set size for
45  this address space. This value is only used
46  to ensure that ample physical pages exist
47  to create this process.
48 
49  NewProcess - Supplies a pointer to the process object being created.
50 
51  DirectoryTableBase - Returns the value of the newly created
52  address space's Page Directory (PD) page and
53  hyper space page.
54 
55 Return Value:
56 
57  Returns TRUE if an address space was successfully created, FALSE
58  if ample physical pages do not exist.
59 
60 Environment:
61 
62  Kernel mode. APCs Disabled.
63 
64 --*/
65 
66 {
67  LOGICAL FlushTbNeeded;
68  PFN_NUMBER PageDirectoryIndex;
69  PFN_NUMBER HyperSpaceIndex;
70  PFN_NUMBER PageContainingWorkingSet;
71  PFN_NUMBER VadBitMapPage;
72  MMPTE TempPte;
73  MMPTE TempPte2;
74  PEPROCESS CurrentProcess;
75  KIRQL OldIrql;
76  PMMPFN Pfn1;
77  ULONG Color;
78  PMMPTE PointerPte;
79  ULONG PdeOffset;
80  PMMPTE MappingPte;
81  PMMPTE PointerFillPte;
82  PMMPTE CurrentAddressSpacePde;
83 
84  //
85  // Charge commitment for the page directory pages, working set page table
86  // page, and working set list. If Vad bitmap lookups are enabled, then
87  // charge for a page or two for that as well.
88  //
89 
90  if (MiChargeCommitment (MM_PROCESS_COMMIT_CHARGE, NULL) == FALSE) {
91  return FALSE;
92  }
93 
94  FlushTbNeeded = FALSE;
95  CurrentProcess = PsGetCurrentProcess ();
96 
97  NewProcess->NextPageColor = (USHORT) (RtlRandom (&MmProcessColorSeed));
98  KeInitializeSpinLock (&NewProcess->HyperSpaceLock);
99 
100  //
101  // Get the PFN lock to get physical pages.
102  //
103 
104  LOCK_PFN (OldIrql);
105 
106  //
107  // Check to make sure the physical pages are available.
108  //
109 
110  if (MI_NONPAGEABLE_MEMORY_AVAILABLE() <= (SPFN_NUMBER)MinimumWorkingSetSize){
111 
112  UNLOCK_PFN (OldIrql);
113  MiReturnCommitment (MM_PROCESS_COMMIT_CHARGE);
114 
115  //
116  // Indicate no directory base was allocated.
117  //
118 
119  return FALSE;
120  }
121 
122  MM_TRACK_COMMIT (MM_DBG_COMMIT_PROCESS_CREATE, MM_PROCESS_COMMIT_CHARGE);
123 
124  MI_DECREMENT_RESIDENT_AVAILABLE (MinimumWorkingSetSize,
125  MM_RESAVAIL_ALLOCATE_CREATE_PROCESS);
126 
127  //
128  // Allocate a page directory page.
129  //
130 
131  if (MmAvailablePages < MM_HIGH_LIMIT) {
132  MiEnsureAvailablePageOrWait (NULL, OldIrql);
133  }
134 
135  Color = MI_PAGE_COLOR_PTE_PROCESS (PDE_BASE,
136  &CurrentProcess->NextPageColor);
137 
138  PageDirectoryIndex = MiRemoveZeroPageMayReleaseLocks (Color, OldIrql);
139 
140  Pfn1 = MI_PFN_ELEMENT (PageDirectoryIndex);
141 
142  if (Pfn1->u3.e1.CacheAttribute != MiCached) {
143  Pfn1->u3.e1.CacheAttribute = MiCached;
144  FlushTbNeeded = TRUE;
145  }
146 
147  //
148  // Allocate the hyper space page table page.
149  //
150 
151  if (MmAvailablePages < MM_HIGH_LIMIT) {
152  MiEnsureAvailablePageOrWait (NULL, OldIrql);
153  }
154 
155  Color = MI_PAGE_COLOR_PTE_PROCESS (MiGetPdeAddress(HYPER_SPACE),
156  &CurrentProcess->NextPageColor);
157 
158  HyperSpaceIndex = MiRemoveZeroPageMayReleaseLocks (Color, OldIrql);
159 
160  Pfn1 = MI_PFN_ELEMENT (HyperSpaceIndex);
161 
162  if (Pfn1->u3.e1.CacheAttribute != MiCached) {
163  Pfn1->u3.e1.CacheAttribute = MiCached;
164  FlushTbNeeded = TRUE;
165  }
166 
167  //
168  // Remove page(s) for the VAD bitmap.
169  //
170 
171  if (MmAvailablePages < MM_HIGH_LIMIT) {
172  MiEnsureAvailablePageOrWait (NULL, OldIrql);
173  }
174 
175  Color = MI_PAGE_COLOR_VA_PROCESS (MmWorkingSetList,
176  &CurrentProcess->NextPageColor);
177 
178  VadBitMapPage = MiRemoveZeroPageMayReleaseLocks (Color, OldIrql);
179 
180  Pfn1 = MI_PFN_ELEMENT (VadBitMapPage);
181 
182  if (Pfn1->u3.e1.CacheAttribute != MiCached) {
183  Pfn1->u3.e1.CacheAttribute = MiCached;
184  FlushTbNeeded = TRUE;
185  }
186 
187  //
188  // Remove a page for the working set list.
189  //
190 
191  if (MmAvailablePages < MM_HIGH_LIMIT) {
192  MiEnsureAvailablePageOrWait (NULL, OldIrql);
193  }
194 
195  Color = MI_PAGE_COLOR_VA_PROCESS (MmWorkingSetList,
196  &CurrentProcess->NextPageColor);
197 
198  PageContainingWorkingSet = MiRemoveZeroPageMayReleaseLocks (Color, OldIrql);
199 
200  Pfn1 = MI_PFN_ELEMENT (PageContainingWorkingSet);
201 
202  if (Pfn1->u3.e1.CacheAttribute != MiCached) {
203  Pfn1->u3.e1.CacheAttribute = MiCached;
204  FlushTbNeeded = TRUE;
205  }
206 
207  UNLOCK_PFN (OldIrql);
208 
209  if (FlushTbNeeded == TRUE) {
210  MI_FLUSH_TB_FOR_CACHED_ATTRIBUTE ();
211  }
212 
213  ASSERT (NewProcess->AddressSpaceInitialized == 0);
214  PS_SET_BITS (&NewProcess->Flags, PS_PROCESS_FLAGS_ADDRESS_SPACE1);
215  ASSERT (NewProcess->AddressSpaceInitialized == 1);
216 
217  NewProcess->Vm.MinimumWorkingSetSize = MinimumWorkingSetSize;
218 
219  NewProcess->WorkingSetPage = PageContainingWorkingSet;
220 
221  INITIALIZE_DIRECTORY_TABLE_BASE (&DirectoryTableBase[0], PageDirectoryIndex);
222 
223  INITIALIZE_DIRECTORY_TABLE_BASE (&DirectoryTableBase[1], HyperSpaceIndex);
224 
225  //
226  // Initialize the page reserved for hyper space.
227  //
228 
229  TempPte = ValidPdePde;
230  MI_SET_GLOBAL_STATE (TempPte, 0);
231 
232  MappingPte = MiReserveSystemPtes (1, SystemPteSpace);
233 
234  if (MappingPte != NULL) {
235 
236  MI_MAKE_VALID_KERNEL_PTE (TempPte2,
237  HyperSpaceIndex,
238  MM_READWRITE,
239  MappingPte);
240 
241  MI_SET_PTE_DIRTY (TempPte2);
242 
243  MI_WRITE_VALID_PTE (MappingPte, TempPte2);
244 
245  PointerPte = MiGetVirtualAddressMappedByPte (MappingPte);
246  }
247  else {
248  PointerPte = MiMapPageInHyperSpace (CurrentProcess, HyperSpaceIndex, &OldIrql);
249  }
250 
251  TempPte.u.Hard.PageFrameNumber = VadBitMapPage;
252  PointerPte[MiGetPteOffset(VAD_BITMAP_SPACE)] = TempPte;
253 
254  TempPte.u.Hard.PageFrameNumber = PageContainingWorkingSet;
255  PointerPte[MiGetPteOffset(MmWorkingSetList)] = TempPte;
256 
257  if (MappingPte != NULL) {
258  MiReleaseSystemPtes (MappingPte, 1, SystemPteSpace);
259  }
260  else {
261  MiUnmapPageInHyperSpace (CurrentProcess, PointerPte, OldIrql);
262  }
263 
264  //
265  // Set the PTE address in the PFN for the page directory page.
266  //
267 
268  Pfn1 = MI_PFN_ELEMENT (PageDirectoryIndex);
269 
270  Pfn1->PteAddress = (PMMPTE)PDE_BASE;
271 
272  TempPte = ValidPdePde;
273  TempPte.u.Hard.PageFrameNumber = HyperSpaceIndex;
274  MI_SET_GLOBAL_STATE (TempPte, 0);
275 
276  //
277  // Add the new process to our internal list prior to filling any
278  // system PDEs so if a system PDE changes (large page map or unmap)
279  // it can mark this process for a subsequent update.
280  //
281 
282  ASSERT (NewProcess->Pcb.DirectoryTableBase[0] == 0);
283 
284  LOCK_EXPANSION (OldIrql);
285 
286  InsertTailList (&MmProcessList, &NewProcess->MmProcessLinks);
287 
288  UNLOCK_EXPANSION (OldIrql);
289 
290  //
291  // Map the page directory page in hyperspace.
292  //
293 
294  MappingPte = MiReserveSystemPtes (1, SystemPteSpace);
295 
296  if (MappingPte != NULL) {
297 
298  MI_MAKE_VALID_KERNEL_PTE (TempPte2,
299  PageDirectoryIndex,
300  MM_READWRITE,
301  MappingPte);
302 
303  MI_SET_PTE_DIRTY (TempPte2);
304 
305  MI_WRITE_VALID_PTE (MappingPte, TempPte2);
306 
307  PointerPte = MiGetVirtualAddressMappedByPte (MappingPte);
308  }
309  else {
310  PointerPte = MiMapPageInHyperSpace (CurrentProcess, PageDirectoryIndex, &OldIrql);
311  }
312 
313  PdeOffset = MiGetPdeOffset (MmSystemRangeStart);
314  PointerFillPte = &PointerPte[PdeOffset];
315  CurrentAddressSpacePde = MiGetPdeAddress (MmSystemRangeStart);
316 
317  RtlCopyMemory (PointerFillPte,
318  CurrentAddressSpacePde,
319  PAGE_SIZE - PdeOffset * sizeof (MMPTE));
320 
321  //
322  // Map the working set page table page.
323  //
324 
325  PdeOffset = MiGetPdeOffset (HYPER_SPACE);
326  PointerPte[PdeOffset] = TempPte;
327 
328  //
329  // Zero the remaining page directory range used to map the working
330  // set list and its hash.
331  //
332 
333  PdeOffset += 1;
334  ASSERT (MiGetPdeOffset (MmHyperSpaceEnd) >= PdeOffset);
335 
336  MiZeroMemoryPte (&PointerPte[PdeOffset],
337  (MiGetPdeOffset (MmHyperSpaceEnd) - PdeOffset + 1));
338 
339  //
340  // Recursively map the page directory page so it points to itself.
341  //
342 
343  TempPte.u.Hard.PageFrameNumber = PageDirectoryIndex;
344  PointerPte[MiGetPdeOffset(PTE_BASE)] = TempPte;
345 
346  if (MappingPte != NULL) {
347  MiReleaseSystemPtes (MappingPte, 1, SystemPteSpace);
348  }
349  else {
350  MiUnmapPageInHyperSpace (CurrentProcess, PointerPte, OldIrql);
351  }
352 
353  InterlockedExchangeAddSizeT (&MmProcessCommit, MM_PROCESS_COMMIT_CHARGE);
354 
355  //
356  // Up the session space reference count.
357  //
358 
359  MiSessionAddProcess (NewProcess);
360 
361  return TRUE;
362 }

367 VOID
368 MiUpdateSystemPdes (
369  IN PEPROCESS Process
370  )
371 
372 /*++
373 
374 Routine Description:
375 
376  This routine updates the system PDEs, typically due to a large page
377  system PTE mapping being created or destroyed. This is rare.
378 
379  Note this is only needed for 32-bit platforms (64-bit platforms share
380  a common top level system page).
381 
382 Arguments:
383 
384  Process - Supplies a pointer to the process to update.
385 
386 Return Value:
387 
388  None.
389 
390 Environment:
391 
392  Kernel mode, expansion lock held.
393 
394  The caller acquired the expansion lock prior to clearing the update
395  bit from this process. We must update the PDEs prior to releasing
396  it so that any new updates can also be rippled.
397 
398 --*/
399 
400 {
401  ULONG i;
402  ULONG PdeOffset;
403  ULONG PdeEndOffset;
404  MMPTE TempPte;
405  PFN_NUMBER PageDirectoryIndex;
406  PFN_NUMBER TargetPageDirectoryIndex;
407  PEPROCESS CurrentProcess;
408  PMMPTE PointerPte;
409  PMMPTE PointerPde;
410  PMMPTE TargetPdePage;
411  PMMPTE TargetAddressSpacePde;
412 
413  ASSERT (KeGetCurrentIrql () == DISPATCH_LEVEL);
414 
415  CurrentProcess = PsGetCurrentProcess ();
416 
417  //
418  // Map the page directory page in hyperspace.
419  // Note for PAE, this is the high 1GB virtual only.
420  //
421 
422  ASSERT (Process->Pcb.DirectoryTableBase[0] != 0);
423  TargetPageDirectoryIndex = Process->Pcb.DirectoryTableBase[0] >> PAGE_SHIFT;
424 
425  ASSERT (PsInitialSystemProcess != NULL);
426  ASSERT (PsInitialSystemProcess->Pcb.DirectoryTableBase[0] != 0);
427  PageDirectoryIndex = PsInitialSystemProcess->Pcb.DirectoryTableBase[0] >> PAGE_SHIFT;
428 
429  TempPte = ValidKernelPte;
430  TempPte.u.Hard.PageFrameNumber = TargetPageDirectoryIndex;
431  ASSERT (MiLargePageHyperPte->u.Long == 0);
432  MI_WRITE_VALID_PTE (MiLargePageHyperPte, TempPte);
433  TargetPdePage = MiGetVirtualAddressMappedByPte (MiLargePageHyperPte);
434 
435  //
436  // Map the system process page directory as we know that's always kept
437  // up to date.
438  //
439 
440  PointerPte = MiMapPageInHyperSpaceAtDpc (CurrentProcess,
441  PageDirectoryIndex);
442 
443  //
444  // Copy all system PTE ranges.
445  //
446 
447  for (i = 0; i < MiPteRangeIndex; i += 1) {
448 
449  PdeOffset = MiGetPdeOffset (MiPteRanges[i].StartingVa);
450  PdeEndOffset = MiGetPdeOffset (MiPteRanges[i].EndingVa);
451 
452  PointerPde = &PointerPte[PdeOffset];
453  TargetAddressSpacePde = &TargetPdePage[PdeOffset];
454 
455  RtlCopyMemory (TargetAddressSpacePde,
456  PointerPde,
457  (PdeEndOffset - PdeOffset + 1) * sizeof (MMPTE));
458 
459  }
460 
461  MiUnmapPageInHyperSpaceFromDpc (CurrentProcess, PointerPte);
462 
463  //
464  // Just invalidate the mapping on the current processor as we cannot
465  // have context switched.
466  //
467 
468  MI_WRITE_ZERO_PTE (MiLargePageHyperPte);
469  MI_FLUSH_CURRENT_TB_SINGLE (TargetPdePage);
470 
471  return;
472 }
