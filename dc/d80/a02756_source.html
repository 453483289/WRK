47 WmiStartTrace(
48  IN OUT PWMI_LOGGER_INFORMATION LoggerInfo
49  )
50 /*++
51 
52 Routine Description:
53 
54  This routine is used to create and start an event tracing session.
55  NOTE: A special instance (KERNEL_LOGGER) is reserved exclusively for
56  logging kernel tracing.
57 
58  To turn on KERNEL_LOGGER, LoggerInfo->Wnode.Guid should be set to
59  SystemTraceControlGuid, and sufficient space must be provided in
60  LoggerInfo->LoggerName.
61 
62  To turn on other loggers, simply provide a name in LoggerName. The
63  logger id will be returned.
64 
65 Arguments:
66 
67  LoggerInfo a pointer to the structure for the logger's control
68  and status information
69 
70 Return Value:
71 
72  The status of performing the action requested.
73 
74 --*/
75 
76 {
77  NTSTATUS status;
78  PWCHAR LogFileName = NULL;
79  HANDLE FileHandle = NULL;
80  ULONG DelayOpen;
81 
82  PAGED_CODE();
83 
84  if (LoggerInfo == NULL)
85  return STATUS_INVALID_PARAMETER;
86 
87  if (LoggerInfo->Wnode.BufferSize < sizeof(WMI_LOGGER_INFORMATION))
88  return STATUS_INVALID_BUFFER_SIZE;
89 
90  //
91  // We assume that the caller is always kernel mode
92  // First, we try and see it is a delay create.
93  // If not, if we can even open the file
94  //
95  DelayOpen = LoggerInfo->LogFileMode & EVENT_TRACE_DELAY_OPEN_FILE_MODE;
96 
97  if (!DelayOpen) {
98  if (LoggerInfo->LogFileName.Buffer != NULL) { // && !delay_create
99  status = WmipCreateNtFileName(
100  LoggerInfo->LogFileName.Buffer,
101  &LogFileName);
102  if (!NT_SUCCESS(status))
103  return status;
104  status = WmipCreateDirectoryFile(LogFileName,
105  FALSE,
106  &FileHandle,
107  LoggerInfo->LogFileMode & EVENT_TRACE_FILE_MODE_APPEND);
108  if (LogFileName != NULL) {
109  ExFreePool(LogFileName);
110  }
111  if (!NT_SUCCESS(status)) {
112  return status;
113  }
114  ZwClose(FileHandle);
115  }
116  }
117 
118  status = WmipStartLogger(LoggerInfo);
119  if (NT_SUCCESS(status)) {
120  status = WmiFlushTrace(LoggerInfo);
121  }
122  return status;
123 }

166

293 // break;
294  }
295 */

392  // in UpdateTrace()

398  ((NewMode & EVENT_TRACE_USE_GLOBAL_SEQUENCE) &&
399  (NewMode & EVENT_TRACE_USE_LOCAL_SEQUENCE)) ||
400

402  (NewMode & EVENT_TRACE_FILE_MODE_CIRCULAR)) ||
403

405  ((NewMode & EVENT_TRACE_FILE_MODE_CIRCULAR) &&
406  (NewMode & EVENT_TRACE_FILE_MODE_APPEND))
407 
408  ) {
409  InterlockedDecrement(&LoggerContext->MutexCount);
410  TraceDebug((1, "WmiUpdateTrace: Release mutex1 %d %d\n",
411  LoggerContext->LoggerId, LoggerContext->MutexCount));
412  WmipReleaseMutex(&LoggerContext->LoggerMutex);
413 #if DBG
414  RefCount =
415 #endif

417  TraceDebug((1, "WmiUpdateTrace: Status2=%X %d %d->%d\n",
418  Status, LoggerId, RefCount+1, RefCount));
419  return STATUS_INVALID_PARAMETER;
420  }

424  //

427  LoggerMode |= EVENT_TRACE_REAL_TIME_MODE;
428  DesiredAccess |= TRACELOG_CREATE_REALTIME;
429  } else {

437  else {
438  LoggerMode &= ~EVENT_TRACE_BUFFERING_MODE;
439  }

454  WmipReleaseMutex(&LoggerContext->LoggerMutex);
455 #if DBG
456  RefCount =
457 #endif

483  //

487  LoggerInfo->LogFileName.Buffer,
488  LoggerInfo->LogFileName.Length,

515  //
