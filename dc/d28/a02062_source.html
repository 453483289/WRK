21 // //
22 // "The" POST BLOCK RULE : //
23 // //
24 // To operate on a post block (i.e. add or remove it from a list - notify,thread,slave), //
25 // you should at least: //
26 // 1. Hold the registry lock exclusively //
27 // OR //
28 // 2. Hold the registry lock shared and acquire the postblock mutex. //
29 // //
30 // //
31 // WARNING!!! //
32 // Failing to do that could arise in obscure registry deadlocks or usage of already freed memory (bugcheck) //
33 // //
35 
37 // //
38 // Other important rules to follow: //
39 // //
40 // 1. We DO NOT dereference objects in CmpPostApc ! //
41 // 2. We DO NOT dereference objects while walking the notify list! //
42 // 3. All operations with Thread PostList are done in CmpPostApc or at APC level. This should avoid two threads //
43 // operating on the same list at the same time //
44 // //

153  PCM_KEY_CONTROL_BLOCK KeyControlBlock,
154  PHHIVE Hive,
155  HCELL_INDEX Cell,
156  ULONG Filter
157  )
158 /*++
159 
160 Routine Description:
161 
162  This routine is called when a notifiable event occurs. It will
163  apply CmpReportNotifyHelper to the hive the event occured in,
164  and the master hive if different.
165 
166 Arguments:
167 
168  KeyControlBlock - KCB of the key at which the event occured.
169  For create or delete this is the created or deleted key.
170 
171  Hive - pointer to hive containing cell of Key at which event occured.
172 
173  Cell - cell of Key at which event occured
174 
175  (hive and cell correspond with name.)
176 
177  Filter - event to be reported
178 
179 Return Value:
180 
181  NONE.
182 
183 --*/
184 {
185  HCELL_INDEX CellToRelease = HCELL_NIL;
186 
187  CM_PAGED_CODE();
188  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"CmpReportNotify:\n"));
189  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"\tHive:%p Cell:%08lx Filter:%08lx\n", Hive, Cell, Filter));
190 
191  //
192  // If the operation was create or delete, treat it as a change
193  // to the parent.
194  //
195  if (Filter == REG_NOTIFY_CHANGE_NAME) {
196  PCM_KEY_NODE pcell;
197  ULONG flags;
198 
199  pcell = (PCM_KEY_NODE)HvGetCell(Hive, Cell);
200  if( pcell == NULL ) {
201  //
202  // we couldn't map the bin containing this cell
203  // Bad luck! notifications will be broken.
204  //
205  return;
206  }
207 
208  CellToRelease = Cell;
209 
210  flags = pcell->Flags;
211  Cell = pcell->Parent;
212  if (flags & KEY_HIVE_ENTRY) {
213  ASSERT( CellToRelease != HCELL_NIL );
214  HvReleaseCell(Hive,CellToRelease);
215 
216  Hive = &(CmpMasterHive->Hive);
217  pcell = (PCM_KEY_NODE)HvGetCell(Hive, Cell);
218  if( pcell == NULL ) {
219  //
220  // we couldn't map the bin containing this cell
221  // Bad luck! notifications will be broken.
222  //
223  return;
224  }
225  CellToRelease = Cell;
226  }
227 
228 
229  KeyControlBlock = KeyControlBlock->ParentKcb;
230 
231  //
232  // if we're at an exit/link node, back up the real node
233  // that MUST be it's parent.
234  //
235  if (pcell->Flags & KEY_HIVE_EXIT) {
236  Cell = pcell->Parent;
237  }
238 
239  ASSERT( CellToRelease != HCELL_NIL );
240  HvReleaseCell(Hive,CellToRelease);
241 
242  }
243 
244  //
245  // Report to notifies waiting on the event's hive
246  //
247  CmpReportNotifyHelper(KeyControlBlock, Hive, Hive, Cell, Filter);
248 
249 
250  //
251  // If containing hive is not the master hive, apply to master hive
252  //
253  if (Hive != &(CmpMasterHive->Hive)) {
254  CmpReportNotifyHelper(KeyControlBlock,
255  &(CmpMasterHive->Hive),
256  Hive,
257  Cell,
258  Filter);
259  }
260 
261  return;
262 }

340 VOID
341 CmpReportNotifyHelper(
342  PCM_KEY_CONTROL_BLOCK KeyControlBlock,
343  IN PHHIVE SearchHive,
344  IN PHHIVE Hive,
345  IN HCELL_INDEX Cell,
346  IN ULONG Filter
347  )
348 /*++
349 
350 Routine Description:
351 
352  Scan the list of active notifies for the specified hive. For
353  any with scope including KeyControlBlock and filter matching
354  Filter, and with proper security access, post the notify.
355 
356 Arguments:
357 
358  Name - canonical path name (as in a key control block) of the key
359  at which the event occured. (This is the name for
360  reporting purposes.)
361 
362  SearchHive - hive to search for matches (which notify list to check)
363 
364  Hive - Supplies hive containing node to match with.
365 
366  Cell - cell identifying the node in Hive
367 
368  Filter - type of event
369 
370 Return Value:
371 
372  NONE.
373 
374 --*/
375 {
376  PLIST_ENTRY NotifyPtr;
377  PCM_NOTIFY_BLOCK NotifyBlock;
378  PCMHIVE CmSearchHive;
379  KIRQL OldIrql;
380  LIST_ENTRY DelayedDeref;
381  PCM_KEY_NODE Node;
382 
383  CM_PAGED_CODE();
384 
385  Node = (PCM_KEY_NODE)HvGetCell(Hive,Cell);
386  if( Node == NULL ) {
387  //
388  // bad luck, we cannot map the view containing this cell
389  //
390  return;
391  }
392 
393  CmSearchHive = CONTAINING_RECORD(SearchHive, CMHIVE, Hive);
394  CmLockHive(CmSearchHive);
395 
396  KeRaiseIrql(APC_LEVEL, &OldIrql);
397 
398  NotifyPtr = &(CmSearchHive->NotifyList);
399 
400  InitializeListHead(&(DelayedDeref));
401 
402  while (NotifyPtr->Flink != NULL) {
403 
404  NotifyPtr = NotifyPtr->Flink;
405 
406  NotifyBlock = CONTAINING_RECORD(NotifyPtr, CM_NOTIFY_BLOCK, HiveList);
407  if (NotifyBlock->KeyControlBlock->TotalLevels > KeyControlBlock->TotalLevels) {
408  //
409  // list is level sorted, we're past all shorter entries
410  //
411  break;
412  } else {
413  PCM_KEY_CONTROL_BLOCK kcb;
414  ULONG LevelDiff, l;
415 
416  LevelDiff = KeyControlBlock->TotalLevels - NotifyBlock->KeyControlBlock->TotalLevels;
417 
418  kcb = KeyControlBlock;
419  for (l=0; l<LevelDiff; l++) {
420  kcb = kcb->ParentKcb;
421  }
422 
423  if (kcb == NotifyBlock->KeyControlBlock) {
424  //
425  // This Notify path is the prefix of this kcb.
426  //
427  if ((NotifyBlock->Filter & Filter)
428  &&
429  ((NotifyBlock->WatchTree == TRUE) ||
430  (Cell == kcb->KeyCell))
431  )
432  {
433  // Filter matches, this event is relevant to this notify
434  // AND
435  // Either the notify spans the whole subtree, or the cell
436  // (key) of interest is the one it applies to
437  //
438  // THEREFORE: The notify is relevant.
439  //
440 
441  //
442  // Correct scope, does caller have access?
443  //
444  if (CmpNotifyTriggerCheck(NotifyBlock,Hive,Node)) {
445  //
446  // Notify block has KEY_NOTIFY access to the node
447  // the event occured at. It is relevant. Therefore,
448  // it gets to see this event. Post and be done.
449  //
450  // we specify that we want no key body dereferenciation
451  // during the CmpPostNotify call. This is to prevent the
452  // deletion of the current notify block
453  //
454  CmpPostNotify(
455  NotifyBlock,
456  NULL,
457  Filter,
458  STATUS_NOTIFY_ENUM_DIR,
459  FALSE,
460  &DelayedDeref
461 #if DBG
462  ,(PCMHIVE)CmSearchHive
463 #endif
464  );
465 
466  } // else no KEY_NOTIFY access to node event occured at
467  } // else not relevant (wrong scope, filter, etc)
468  }
469  }
470  }
471 
472  KeLowerIrql(OldIrql);
473  CmUnlockHive(CmSearchHive);
474 
475  HvReleaseCell(Hive,Cell);
476 
477  //
478  // finish the job started in CmpPostNotify (i.e. dereference the keybodies
479  // we prevented. this may cause some notifyblocks to be freed
480  //
481  CmpDelayedDerefKeys(&DelayedDeref);
482 
483  return;
484 }

486 VOID
487 CmpPostNotify(
488  PCM_NOTIFY_BLOCK NotifyBlock,
489  PUNICODE_STRING Name OPTIONAL,
490  ULONG Filter,
491  NTSTATUS Status,
492  BOOLEAN PostListLockHeld,
493  PLIST_ENTRY ExternalKeyDeref OPTIONAL
494 #if DBG
495  ,
496  PCMHIVE CmHive
497 #endif
498  )
499 /*++
500 
501 Routine Description:
502 
503  Actually report the notify event by signaling events, enqueing
504  APCs, and so forth.
505 
506  When Status is STATUS_NOTIFY_CLEANUP:
507 
508  - if the post block is a slave one, just cancel it.
509  - if the post block is a master one, cancel all slave post blocks
510  and trigger event on the master block.
511 
512 Comments:
513 
514  This routine is using a "delayed dereferencing" technique to prevent
515  deadlocks that may appear when a keybody is dereferenced while holding
516  the post block lock. As for this, a list with keybodies that have to be
517  dereferenced is constructed while walking the list of postblocks attached
518  to the current notify block and the related (slave or master) post blocks.
519  The list is built by tricking postblocks. For all postblock about to be
520  freed the PostKeyBody member is added to the local list and then set to NULL
521  on the postblock. This will avoid the key body dereferencing in CmpFreePostBlock.
522  Instead, after the postblock lock is released, the local list is iterated and
523  the keybodies are dereferenced and the storage for associated CM_POST_KEY_BODY
524  objects is freed.
525 
526 
527 Arguments:
528 
529  NotifyBlock - pointer to structure that describes the notify
530  operation. (Where to post to)
531 
532  Name - name of key at which event occurred.
533 
534  Filter - nature of event
535 
536  Status - completion status to report
537 
538  ExternalKeyDeref - this parameter (when not NULL) specifies that the caller doesn't
539  want any keybody to be dereferenced while in this routine
540 
541 Return Value:
542 
543  NONE.
544 
545 --*/
546 {
547  PCM_POST_BLOCK PostBlock;
548  PCM_POST_BLOCK SlavePostBlock;
549  LIST_ENTRY LocalDelayedDeref;
550  KIRQL OldIrql;
551  PLIST_ENTRY DelayedDeref;
552 
553  Filter;
554  Name;
555 
556  CM_PAGED_CODE();
557  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"CmpPostNotify:\n"));
558  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"\tNotifyBlock:%p ", NotifyBlock));
559  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"\tName = %wZ\n", Name));
560  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"\tFilter:%08lx Status=%08lx\n", Filter, Status));
561  ASSERT_CM_LOCK_OWNED();
562 
563  if( ARGUMENT_PRESENT(ExternalKeyDeref) ) {
564  //
565  // The caller want to do all keybody dereferencing by himself
566  //
567  DelayedDeref = ExternalKeyDeref;
568  } else {
569  // local delayed dereferencing (the caller doesn't care!)
570  DelayedDeref = &LocalDelayedDeref;
571  InitializeListHead(DelayedDeref);
572  }
573 
574  ASSERT_HIVE_LOCK_OWNED(CmHive);
575  //
576  // Acquire exclusive access over the postlist(s)
577  //
578  if( !PostListLockHeld ) {
579  LOCK_POST_LIST();
580  }
581 
582  if (IsListEmpty(&(NotifyBlock->PostList)) == TRUE) {
583  //
584  // Nothing to post, set a mark and return
585  //
586  NotifyBlock->NotifyPending = TRUE;
587  if( !PostListLockHeld ) {
588  UNLOCK_POST_LIST();
589  }
590  return;
591  }
592  NotifyBlock->NotifyPending = FALSE;
593 
594  //
595  // IMPLEMENTATION NOTE:
596  // If we ever want to actually implement the code that returns
597  // names of things that changed, this is the place to add the
598  // name and operation type to the buffer.
599  //
600 
601  //
602  // Pull and post all the entries in the post list
603  //
604  while (IsListEmpty(&(NotifyBlock->PostList)) == FALSE) {
605 
606  //
607  // Remove from the notify block list, and enqueue the apc.
608  // The apc will remove itself from the thread list
609  //
610  PostBlock = (PCM_POST_BLOCK)RemoveHeadList(&(NotifyBlock->PostList));
611  PostBlock = CONTAINING_RECORD(PostBlock,
612  CM_POST_BLOCK,
613  NotifyList);
614 
615  // Protect for multiple deletion of the same object
616  CmpClearListEntry(&(PostBlock->NotifyList));
617 
618  if( (Status == STATUS_NOTIFY_CLEANUP) && !IsMasterPostBlock(PostBlock) ) {
619  //
620  // Cleanup notification (i.e. the key handle was closed or the key was deleted)
621  // When the post is a slave one, just cancel it. Canceling means:
622  // 1. Removing from the notify PostList (already done at this point - see above)
623  // 2. Unchaining from the Master Block CancelPostList
624  // 3. Delisting from the thread PostBlockList
625  // 4. Actually freeing the memory
626  //
627 
628  // Use Cmp variant to protect for multiple deletion of the same object
629  CmpRemoveEntryList(&(PostBlock->CancelPostList));
630  //
631  // FIX 289351
632  //
633  // Use Cmp variant to protect for multiple deletion of the same object
634  KeRaiseIrql(APC_LEVEL, &OldIrql);
635  CmpRemoveEntryList(&(PostBlock->ThreadList));
636  KeLowerIrql(OldIrql);
637 
638 #if DBG
639  if(PostBlock->TraceIntoDebugger) {
640  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"[CM]\tCmpPostNotify: PostBlock:%p is a slave block,and notify is CLEANUP==> just cleanning\n", PostBlock));
641  }
642 #endif
643  if( PostBlock->NotifyType != PostSynchronous ) {
644 
645  // add to the deref list and clean the post block
646  CmpAddToDelayedDeref(PostBlock,DelayedDeref);
647 
648  //
649  // Front-end routine will do self cleanup for synchronous notifications
650  CmpFreePostBlock(PostBlock);
651  }
652 
653  continue; //try the next one
654  }
655 
656  //
657  // Simulate that this block is the master one, so we can free the others
658  // Doing that will ensure the right memory deallocation when the master
659  // (from now on this block) will be freed.
660  //
661  if(!IsMasterPostBlock(PostBlock)) {
662  //
663  // this is not the master block, we have some more work to do
664  //
665  SlavePostBlock = PostBlock;
666  do {
667  SlavePostBlock = (PCM_POST_BLOCK)SlavePostBlock->CancelPostList.Flink;
668  SlavePostBlock = CONTAINING_RECORD(SlavePostBlock,
669  CM_POST_BLOCK,
670  CancelPostList);
671  //
672  // reset the "master flag" if set
673  //
674  ClearMasterPostBlockFlag(SlavePostBlock);
675  } while (SlavePostBlock != PostBlock);
676 
677  //
678  // Make this post block the master one
679  //
680  SetMasterPostBlockFlag(PostBlock);
681  }
682 
683 #if DBG
684  if(PostBlock->TraceIntoDebugger) {
685  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_NOTIFY,"[CM]\tCmpPostNotify: Master block switched to :%p\n", PostBlock));
686  }
687 #endif
688 
689  //
690  // Cancel all slave Post requests that may be linked to self
691  //
692 
693  if( PostBlockType(PostBlock) != PostSynchronous ) {
694  //
695  // Front-end routine will do self cleanup for synchronous notifications
696  CmpCancelSlavePost(PostBlock,DelayedDeref);
697  //
698  // Do the same for the master (in case master and slave got switched)
699  // This will avoid dereferencing the keybody from CmpPostApc
700  CmpAddToDelayedDeref(PostBlock,DelayedDeref);
701  }
702 
703  switch (PostBlockType(PostBlock)) {
704  case PostSynchronous:
705  //
706  // This is a SYNC notify call. There will be no user event,
707  // and no user apc routine. Quick exit here, just fill in
708  // the Status and poke the event.
709  //
710  // Holder of the systemevent will wake up and free the
711  // postblock. If we free it here, we get a race & bugcheck.
712  //
713  // Set the flink to NULL so that the front side can tell this
714  // has been removed if its wait aborts.
715  //
716  PostBlock->NotifyList.Flink = NULL;
717  PostBlock->u->Sync.Status = Status;
718  KeSetEvent(PostBlock->u->Sync.SystemEvent,
719  0,
720  FALSE);
721  break;
722 
723  case PostAsyncUser:
724  //
725  // Insert the APC into the queue
726  //
727  KeInsertQueueApc(PostBlock->u->AsyncUser.Apc,
728  (PVOID)ULongToPtr(Status),
729  (PVOID)PostBlock,
730  0);
731  break;
732 
733  case PostAsyncKernel:
734  //
735  // Queue the work item, then free the post block.
736  //
737  if (PostBlock->u->AsyncKernel.WorkItem != NULL) {
738  ExQueueWorkItem(PostBlock->u->AsyncKernel.WorkItem,
739  PostBlock->u->AsyncKernel.QueueType);
740  }
741 
742  //
743  // Signal Event if present, and deref it.
744  //
745  if (PostBlock->u->AsyncKernel.Event != NULL) {
746  KeSetEvent(PostBlock->u->AsyncKernel.Event,
747  0,
748  FALSE);
749  ObDereferenceObject(PostBlock->u->AsyncKernel.Event);
750  }
751 
752  //
753  // Multiple async kernel notification are not allowed
754  //
755  ASSERT(IsListEmpty(&(PostBlock->CancelPostList)) == TRUE);
756  //
757  // remove the post block from the thread list, and free it
758  //
759  // Use Cmp variant to protect for multiple deletion of the same object
760  KeRaiseIrql(APC_LEVEL, &OldIrql);
761  CmpRemoveEntryList(&(PostBlock->ThreadList));
762  KeLowerIrql(OldIrql);
763 
764  // it was already added to delayed deref.
765  CmpFreePostBlock(PostBlock);
766  break;
767  }
768  }
769 
770  if( !PostListLockHeld ) {
771  UNLOCK_POST_LIST();
772  }
773 
774  //
775  // At this point we have a list of keybody elements that have to be dereferenciated
776  // and the associated storage for the covering objects freed. The keybodies in this
777  // list have only one reference count on them (they were referenced only in
778  // NtNotifyChangeMultipleKeys), dereferencing them here should free the object
779  //

1034  finds, it will:
1035

1037  that we hold the Registry mutex.
1038

1040  that we run at APC level.
1041 
1042  3. By the time this procedure runs, user apcs are not interesting
1043  and neither are SystemEvents, so do not bother processing
1044  them.
1045 
1046  UserEvents and IoStatusBlocks could be referred to by other
1047  threads in the same process, or even a different process,
1048  so process them so those threads know what happened, use
1049  status code of STATUS_NOTIFY_CLEANUP.

1052  Else only remove this notification from the master CancelPortList
