36 /*++
37 
38 Routine Description:
39 
40  This routine inserts an object into the current processes handle table.
41 
42  The Object header includes a pointer to a SecurityDescriptor passed in
43  an object creation call. This SecurityDescriptor is not assumed to have
44  been captured. This routine is responsible for making an appropriate
45  SecurityDescriptor and removing the reference in the object header.
46 
47 Arguments:
48 
49  Object - Supplies a pointer to the new object body
50 
51  AccessState - Optionally supplies the access state for the new
52  handle
53 
54  DesiredAccess - Optionally supplies the desired access we want for the
55  new handle
56 
57  ObjectPointerBias - Supplies a bias to apply for the pointer count for the
58  object
59 
60  NewObject - Optionally receives the pointer to the new object that we've
61  created a handle for
62 
63  Handle - Receives the new handle, If NULL then no handle is created.
64  Objects that don't have handles created must be unnamed and
65  have an object bias of zero.
66 
67 Return Value:
68 
69  An appropriate NTSTATUS value.
70 
71 --*/

76  PUNICODE_STRING ObjectName;
77  POBJECT_TYPE ObjectType;
78  POBJECT_HEADER_NAME_INFO NameInfo;
79  PSECURITY_DESCRIPTOR ParentDescriptor = NULL;

98  // the object type, and the address of the object name descriptor, if
99  // specified.
100  //
101

103 
104 #if DBG
105

107

109  KdBreakPoint();

123 
124  ObjectName = NULL;
125 
126  if ((NameInfo != NULL) && (NameInfo->Name.Buffer != NULL)) {
127 
128  ObjectName = &NameInfo->Name;
129  }
130 
131  ASSERT (ARGUMENT_PRESENT (Handle) || (ObjectPointerBias == 0 && ObjectName == NULL &&
132  ObjectType->TypeInfo.SecurityRequired && NewObject == NULL));
133 
134  //
135  // If security checks are not required and an object name is not
136  // specified, insert an unnamed object, biasing the count
137  // by one, dereference the bias, and return to our caller
138  //
139 
140  PreviousMode = KeGetPreviousMode();
141

143 
144  ObjectHeader->ObjectCreateInfo = NULL;
145 
146  *Handle = NULL;
147 
148  Status = ObpCreateUnnamedHandle( Object,
149  DesiredAccess,
150  1 + ObjectPointerBias,
151  ObjectCreateInfo->Attributes,
152  PreviousMode,
153  NewObject,
154  Handle );

156  // Free the object creation information and dereference the object.
157  //
158 
159  ObpFreeObjectCreateInformation(ObjectCreateInfo);
160 
161  ObpDereferenceNameInfo( NameInfo );
162  ObDereferenceObject(Object);
163 
164  return Status;
165  }
166 
167  //
168  // The object is either named or requires full security checks. If the
169  // caller hasn't specified an access state then dummy up a local one
170  // using the requested desired access
171  //
172 
173  if (!ARGUMENT_PRESENT(AccessState)) {
174 
175  AccessState = &LocalAccessState;
176 
177  Status = SeCreateAccessState( &LocalAccessState,
178  &AuxData,
179  DesiredAccess,
180  &ObjectType->TypeInfo.GenericMapping );
181 
182  if (!NT_SUCCESS(Status)) {
183 
184  ObpDereferenceNameInfo( NameInfo );
185  ObDereferenceObject(Object);
186 
187  return Status;
188  }
189  }

192

194  // Check the desired access mask against the security descriptor
195  //
196 
197  Status = ObpValidateAccessMask( AccessState );
198 
199  if (!NT_SUCCESS( Status )) {
200 
201  if (AccessState == &LocalAccessState) {
202 
203  SeDeleteAccessState( AccessState );
204  }
205 
206  ObpDereferenceNameInfo( NameInfo );
207  ObDereferenceObject(Object);
208 
209  if (AccessState == &LocalAccessState) {
210 
211  SeDeleteAccessState( AccessState );
212  }
213 
214  return( Status );
215  }

218  // Set some local state variables
219  //
220 
221  ObpInitializeLookupContext(&LookupContext);
222 
223  InsertObject = Object;
224  OpenReason = ObCreateHandle;
225 
226  //
227  // Check if we have an object name. If so then
228  // lookup the name
229  //
230 
231  if (ObjectName != NULL) {
232 
233  Status = ObpLookupObjectName( ObjectCreateInfo->RootDirectory,
234  ObjectName,
235  ObjectCreateInfo->Attributes,
236  ObjectType,
237  (KPROCESSOR_MODE)(ObjectHeader->Flags & OB_FLAG_KERNEL_OBJECT
238  ? KernelMode : UserMode),
239  ObjectCreateInfo->ParseContext,
240  ObjectCreateInfo->SecurityQos,
241  Object,
242  AccessState,
243  &LookupContext,
244  &InsertObject );
245 
246  //
247  // We found the name and it is not the object we have as our input.
248  // So we cannot insert the object again so we'll return an
249  // appropriate status
250  //
251 
252  if (NT_SUCCESS(Status) &&
253  (InsertObject != NULL) &&
254  (InsertObject != Object)) {
255 
256  OpenReason = ObOpenHandle;
257 
258  if (ObjectCreateInfo->Attributes & OBJ_OPENIF) {
259 
260  if (ObjectType != OBJECT_TO_OBJECT_HEADER(InsertObject)->Type) {
261 
262  Status = STATUS_OBJECT_TYPE_MISMATCH;
263 
264  } else {
265 
266  Status = STATUS_OBJECT_NAME_EXISTS; // Warning only
267  }
268 
269  } else {
270 
271  if (OBJECT_TO_OBJECT_HEADER(InsertObject)->Type == ObpSymbolicLinkObjectType) {
272 
273  ObDereferenceObject(InsertObject);
274  }
275 
276  Status = STATUS_OBJECT_NAME_COLLISION;
277  }
278  }
279 
280  //
281  // We did not find the name so we'll cleanup after ourselves
282  // and return to our caller
283  //
284 
285  if (!NT_SUCCESS( Status )) {
286 
287  ObpReleaseLookupContext( &LookupContext );
288 
289  ObpDereferenceNameInfo( NameInfo );
290  ObDereferenceObject( Object );
291 
292  //
293  // Free security information if we allocated it
294  //
295 
296  if (AccessState == &LocalAccessState) {
297 
298  SeDeleteAccessState( AccessState );
299  }
300 
301  return( Status );
302 
303  } else {
304 
305  //
306  // Otherwise we did locate the object name
307  //
308  // If we just created a named symbolic link then call out to
309  // handle any Dos Device name semantics.
310  //
311 
312  if (ObjectType == ObpSymbolicLinkObjectType) {
313 
314  ObpCreateSymbolicLinkName( (POBJECT_SYMBOLIC_LINK)InsertObject );
315  }
316  }
317  }
318 
319  //
320  // If we are creating a new object, then we need assign security
321  // to it. A pointer to the captured caller-proposed security
322  // descriptor is contained in the AccessState structure. The
323  // SecurityDescriptor field in the object header must point to
324  // the final security descriptor, or to NULL if no security is
325  // to be assigned to the object.
326  //
327 
328  if (InsertObject == Object) {
329 
330  //
331  // Only the following objects have security descriptors:
332  //
333  // - Named Objects
334  // - Unnamed objects whose object-type information explicitly
335  // indicates a security descriptor is required.
336  //
337 
338  if ((ObjectName != NULL) || ObjectType->TypeInfo.SecurityRequired) {
339 
340  //
341  // Get the parent's descriptor, if there is one...
342  //
343 
344  if ((NameInfo != NULL) && (NameInfo->Directory != NULL)) {
345 
346  //
347  // This will allocate a block of memory and copy
348  // the parent's security descriptor into it, and
349  // return the pointer to the block.
350  //
351  // Call ObReleaseObjectSecurity to free up this
352  // memory.
353  //
354 
355  ObGetObjectSecurity( NameInfo->Directory,
356  &ParentDescriptor,
357  &SecurityDescriptorAllocated );
358  }
359  else {
360  SecurityDescriptorAllocated = FALSE;
361  }
362 
363  //
364  // Take the captured security descriptor in the AccessState,
365  // put it into the proper format, and call the object's
366  // security method to assign the new security descriptor to
367  // the new object.
368  //
369 
370  Status = ObAssignSecurity( AccessState,
371  ParentDescriptor,
372  Object,
373  ObjectType );
374 
375  if (ParentDescriptor != NULL) {
376 
377  ObReleaseObjectSecurity( ParentDescriptor,
378  SecurityDescriptorAllocated );
379 
380  } else if (NT_SUCCESS( Status )) {
381 
382  SeReleaseSecurityDescriptor( ObjectCreateInfo->SecurityDescriptor,
383  ObjectCreateInfo->ProbeMode,
384  TRUE );
385 
386  ObjectCreateInfo->SecurityDescriptor = NULL;
387  AccessState->SecurityDescriptor = NULL;
388  }
389  }
390 
391  if (!NT_SUCCESS( Status )) {
392 
393  //
394  // The attempt to assign the security descriptor to
395  // the object failed.
396  //
397 
398  if (LookupContext.DirectoryLocked) {
399 
400  //
401  // If ObpLookupObjectName already inserted the
402  // object into the directory we have to backup this
403  //
404 
405  //
406  // Capture the object Directory
407  //
408 
409  DirObject = NameInfo->Directory;
410 
411  ObpDeleteDirectoryEntry( &LookupContext );
412  }
413 
414  ObpReleaseLookupContext( &LookupContext );
415 
416  //
417  // If ObpLookupObjectName inserted the object into the directory
418  // it added a reference to the object and to its directory
419  // object. We should remove the extra-references
420  //
421 
422  if (DirObject) {
423 
424  ObDereferenceObject( Object );
425  ObDereferenceObject( DirObject );
426  }
427 
428  //
429  // The first backout logic used ObpDeleteNameCheck
430  // which is wrong because the security descriptor for
431  // the object is not initialized. Actually ObpDeleteNameCheck
432  // had no effect because the object was removed before from
433  // the directory
434  //
435 
436  ObpDereferenceNameInfo( NameInfo );
437  ObDereferenceObject( Object );
438 
439  //
440  // Free security information if we allocated it
441  //
442 
443  if (AccessState == &LocalAccessState) {
444 
445  SeDeleteAccessState( AccessState );
446  }
447 
448  return( Status );
449  }
450  }
451 
452  ReturnStatus = Status;
453 
454  ObjectHeader->ObjectCreateInfo = NULL;
455 
456  //
457  // Create a named handle for the object with a pointer bias
458  // This call also will unlock the directory lock is necessary
459  // on return
460  //
461 
462  if (ARGUMENT_PRESENT (Handle)) {
463 
464  Status = ObpCreateHandle( OpenReason,
465  InsertObject,
466  NULL,
467  AccessState,
468  1 + ObjectPointerBias,
469  ObjectCreateInfo->Attributes,
470  &LookupContext,
471  PreviousMode,
472  NewObject,
473  &NewHandle );
474 
475  //
476  // If the insertion failed, the following dereference will cause
477  // the newly created object to be deallocated.
478  //
479 
480  if (!NT_SUCCESS( Status )) {
481 
482  //
483  // Make the name reference go away if an error.
484  //
485 
486  if (ObjectName != NULL) {
487 
488  ObpDeleteNameCheck( Object );
489  }
490 
491  *Handle = NULL;
492 
493  ReturnStatus = Status;
494 
495  } else {
496  *Handle = NewHandle;
497  }
498 
499  ObpDereferenceNameInfo( NameInfo );
500 
501  ObDereferenceObject( Object );
502 
503 
504  } else {
505 
506  BOOLEAN IsNewObject;
507 
508  //
509  // Charge the user quota for the object.
510  //
511 
512  ObpLockObject( ObjectHeader );
513 
514  ReturnStatus = ObpChargeQuotaForObject( ObjectHeader, ObjectType, &IsNewObject );
515 
516  ObpUnlockObject( ObjectHeader );
517 
518  if (!NT_SUCCESS (ReturnStatus)) {
519  ObDereferenceObject( Object );
520  }
521 
522  //
523  // N.B. An object cannot be named if no Handle parameter is specified.
524  // The calls to ObpDereferenceNameInfo and ObpReleaseLookupContext are
525  // not necessary in this path then.
526  //
527  }
528 
529  ObpFreeObjectCreateInformation( ObjectCreateInfo );
530 
531  //
532  // Free security information if we allocated it
533  //
534 
535  if (AccessState == &LocalAccessState) {
536 
537  SeDeleteAccessState( AccessState );
538  }
539 
540  return( ReturnStatus );
