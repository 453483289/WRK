28 /*++
29 
30 Routine Description:
31 
32  This routine is called as the result of the interval timer interrupt on
33  all processors in the system. Its function is update the per processor
34  tick count, update the runtime of the current thread, update the runtime
35  of the current thread's process, perform DPC moderation, and decrement
36  the current thread quantum if a full tick has expired. This routine also
37  performs real time scheduling quantum end and next interval processing.
38 
39  N.B. This routine is executed on all processors in a multiprocessor
40  system.
41 
42 Arguments:
43 
44  TrapFrame - Supplies the address of a trap frame.
45 
46  Increment - Supplies the time increment value in 100 nanosecond units.
47 
48 Return Value:
49 
50  None.
51 
52 --*/
53 
54 {
55 
56  PKPRCB Prcb;
57  PKTHREAD Thread;
58 
59  //
60  // If the clock tick should be skipped, then clear the skip flag and
61  // return.
62  //
63 
64  Prcb = KeGetCurrentPrcb();
65 
66 #if DBG
67 
68  if (Prcb->SkipTick != FALSE) {
69  Prcb->SkipTick = FALSE;
70  return;
71  }
72 
73 #endif
74 
75  //
76  // Update the tick count offset and check if a full tick has expired.
77  //
78 
79  if ((Prcb->TickOffset -= Increment) <= 0) {
80 
81  //
82  // Update the appropriate time counter based on previous mode, IRQL,
83  // and whether there is currently a DPC active.
84  //
85 
86  Prcb->TickOffset += KeMaximumIncrement;
87  Thread = KeGetCurrentThread();
88  if ((TrapFrame->SegCs & MODE_MASK) == 0) {
89 
90  //
91  // Update the total time spent in kernel mode.
92  //
93  // If the interrupt nesting level is one and a DPC routine is
94  // active, then update the DPC time. Otherwise, if the interrupt
95  // nesting level is greater than one, then update the interrupt
96  // time. Otherwise, update the current thread kernel time.
97  //
98 
99  Prcb->KernelTime += 1;
100  if ((Prcb->NestingLevel == 1) && (Prcb->DpcRoutineActive != FALSE)) {
101  Prcb->DpcTime += 1;
102 
103  //
104  // If the kernel debugger is enabled and the time spent in
105  // the current DPC exceeds the system DPC time out limit,
106  // then print a warning message and break into the debugger.
107  //
108 
109 #if DBG
110 
111  if ((KdDebuggerEnabled != 0) &
112  ((Prcb->DebugDpcTime += 1) >= KiDPCTimeout)) {
113 
114  DbgPrint("*** DPC execution time exceeds system limit\n");

116  DbgPrint(" This is an ERROR in a DPC routine\n");
117  DbgPrint(" Perform a stack trace to find the culprit\n");

121 
122 #endif
123 
124  } else if (Prcb->NestingLevel > 1) {

126 
127  } else {

129  }
130 
131  } else {
132 
133  //

135  //
136

138  Thread->UserTime += 1;
139  }
140 
141  //
142  // Update the DPC request rate which is computed as the average of
143  // the previous rate and the current rate.
144  //
145 
146  Prcb->DpcRequestRate += (Prcb->DpcData[DPC_NORMAL].DpcCount - Prcb->DpcLastCount);
147  Prcb->DpcRequestRate >>= 1;
148  Prcb->DpcLastCount = Prcb->DpcData[DPC_NORMAL].DpcCount;
149 
150  //
151  // If the current DPC queue depth is not zero, a DPC routine is not
152  // active, and a DPC interrupt has not been requested, then request
153  // a dispatch interrupt, decrement the maximum DPC queue depth, and
154  // reset the threshold counter if appropriate. Otherwise, count down
155  // the adjustment threshold and if the count reaches zero, then
156  // increment the maximum DPC queue depth, but not above the initial
157  // value and reset the adjustment threshold value.
158  //
159 
160  if ((Prcb->DpcData[DPC_NORMAL].DpcQueueDepth != 0) &&
161  (Prcb->DpcRoutineActive == FALSE) &&
162  (Prcb->DpcInterruptRequested == FALSE)) {
163 
164  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
165  Prcb->AdjustDpcThreshold = KiAdjustDpcThreshold;
166 
167  //
168  // If the DPC request rate is less than the ideal rate and the
169  // DPC queue depth is not one, then decrement the maximum queue
170  // depth.
171  //
172 
173  if ((Prcb->DpcRequestRate < KiIdealDpcRate) &&
174  (Prcb->MaximumDpcQueueDepth != 1)) {
175 
176  Prcb->MaximumDpcQueueDepth -= 1;
177  }
178 
179  } else {
180  if ((Prcb->AdjustDpcThreshold -= 1) == 0) {
181  Prcb->AdjustDpcThreshold = KiAdjustDpcThreshold;
182  if (Prcb->MaximumDpcQueueDepth != KiMaximumDpcQueueDepth) {
183  Prcb->MaximumDpcQueueDepth += 1;
184  }
185  }
186  }
187 
188  //
189  // If the current thread is not the idle thread, then decrement the
190  // thread quantum.
191  //
192  // If the thread quantum is exhausted, then set the quantum end true
193  // and request a dispatch interrupt on the current processor.
194  //
195 
196  if ((Thread != Prcb->IdleThread) &&
197  ((Thread->Quantum -= CLOCK_QUANTUM_DECREMENT) <= 0)) {
198 
199  Prcb->QuantumEnd = TRUE;
200  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
201  }
202 
203  //
204  // If the debugger is enabled, check if a break is requested.
205  //
206  // N.B. A poll break in attempt only occurs on each processor when
207  // the poll slot matches the current processor number.
208  //
209 
210  if (KdDebuggerEnabled != FALSE) {
211  if (Prcb->PollSlot == Prcb->Number) {
212  KdCheckForDebugBreak();
213  }
214 
215  Prcb->PollSlot += 1;
216  if (Prcb->PollSlot == KeNumberProcessors) {
217  Prcb->PollSlot = 0;
218  }
219  }
220  }
221 
222  return;
223 }
