57 CmGetSystemControlValues(
58  __in PVOID SystemHiveBuffer,
59  __inout PCM_SYSTEM_CONTROL_VECTOR ControlVector
60  )
61 /*++
62 
63 Routine Description:
64 
65  Look for registry values in current control set, as specified
66  by entries in ControlVector. Report data for value entries
67  (if any) to variables ControlVector points to.
68 
69 Arguments:
70 
71  SystemHiveBuffer - pointer to flat image of the system hive
72 
73  ControlVector - pointer to structure that describes what values
74  to pull out and store
75 
76 Return Value:
77 
78  NONE.
79 
80 --*/
81 {
82  NTSTATUS status;
83  PHHIVE SystemHive;
84  HCELL_INDEX RootCell;
85  HCELL_INDEX BaseCell;
86  UNICODE_STRING Name;
87  HCELL_INDEX KeyCell;
88  HCELL_INDEX ValueCell;
89  PCM_KEY_VALUE ValueBody;
90  ULONG Length;
91  BOOLEAN AutoSelect;
92  BOOLEAN small;
93  ULONG tmplength;
94  PCM_KEY_NODE Node;
95 
96  //
97  // set up to read flat system hive image loader passes us
98  //
99  RtlZeroMemory((PVOID)&CmControlHive, sizeof(CmControlHive));
100  SystemHive = &(CmControlHive.Hive);
101  CmpInitHiveViewList((PCMHIVE)SystemHive);
102  CmpInitSecurityCache((PCMHIVE)SystemHive);
103  status = HvInitializeHive(
104  SystemHive,
105  HINIT_FLAT,
106  HIVE_VOLATILE,
107  HFILE_TYPE_PRIMARY,
108  SystemHiveBuffer,
109  NULL,
110  NULL,
111  NULL,
112  NULL,
113  NULL,
114  NULL,
115  1,
116  NULL
117  );
118  if (!NT_SUCCESS(status)) {
119  CM_BUGCHECK(BAD_SYSTEM_CONFIG_INFO,BAD_SYSTEM_CONTROL_VALUES,1,SystemHive,status);
120  }
121 
122  //
123  // don't bother locking/releasing cells
124  //
125  ASSERT( SystemHive->ReleaseCellRoutine == NULL );
126  //
127  // get hive.cell of root of current control set
128  //
129  RootCell = ((PHBASE_BLOCK)SystemHiveBuffer)->RootCell;
130  RtlInitUnicodeString(&Name, L"current");
131  BaseCell = CmpFindControlSet(
132  SystemHive,
133  RootCell,
134  &Name,
135  &AutoSelect
136  );
137  if (BaseCell == HCELL_NIL) {
138  CM_BUGCHECK(BAD_SYSTEM_CONFIG_INFO,BAD_SYSTEM_CONTROL_VALUES,2,SystemHive,&Name);
139  }
140 
141  Node = (PCM_KEY_NODE)HvGetCell(SystemHive,BaseCell);
142  if( Node == NULL ) {
143  //
144  // we couldn't map a view for the bin containing this cell
145  //
146  return;
147  }
148  RtlInitUnicodeString(&Name, L"control");
149  BaseCell = CmpFindSubKeyByName(SystemHive,
150  Node,
151  &Name);
152  if (BaseCell == HCELL_NIL) {
153  CM_BUGCHECK(BAD_SYSTEM_CONFIG_INFO,BAD_SYSTEM_CONTROL_VALUES,3,Node,&Name);
154  }
155 
156  //
157  // SystemHive.BaseCell = \registry\machine\system\currentcontrolset\control
158  //
159 
160  //
161  // step through vector, trying to fetch each value
162  //
163  while (ControlVector->KeyPath != NULL) {
164 
165  //
166  // Assume we will fail to find the key or value.
167  //
168 
169  Length = (ULONG)-1;
170 
171  KeyCell = CmpWalkPath(SystemHive, BaseCell, ControlVector->KeyPath);
172 
173  if (KeyCell != HCELL_NIL) {
174 
175  //
176  // found the key, look for the value entry
177  //
178  Node = (PCM_KEY_NODE)HvGetCell(SystemHive,KeyCell);
179  if( Node == NULL ) {
180  //
181  // we couldn't map a view for the bin containing this cell
182  //
183  return;
184  }
185  RtlInitUnicodeString(&Name, ControlVector->ValueName);
186  ValueCell = CmpFindValueByName(SystemHive,
187  Node,
188  &Name);
189  if (ValueCell != HCELL_NIL) {
190 
191  //
192  // SystemHive.ValueCell is value entry body
193  //
194 
195  if (ControlVector->BufferLength == NULL) {
196  tmplength = sizeof(ULONG);
197  } else {
198  tmplength = *(ControlVector->BufferLength);
199  }
200 
201  ValueBody = (PCM_KEY_VALUE)HvGetCell(SystemHive, ValueCell);
202  if( ValueBody == NULL ) {
203  //
204  // we couldn't map a view for the bin containing this cell
205  //
206  return;
207  }
208 
209  small = CmpIsHKeyValueSmall(Length, ValueBody->DataLength);
210 
211  if (tmplength < Length) {
212  Length = tmplength;
213  }
214 
215  if (Length > 0) {
216 
217  PCELL_DATA Buffer;
218  BOOLEAN BufferAllocated;
219  ULONG realsize;
220  HCELL_INDEX CellToRelease;
221 
222  ASSERT((small ? (Length <= CM_KEY_VALUE_SMALL) : TRUE));
223  //
224  // get the data from source, regardless of the size
225  //
226  if( CmpGetValueData(SystemHive,ValueBody,&realsize,&Buffer,&BufferAllocated,&CellToRelease) == FALSE ) {
227  //
228  // insufficient resources; return NULL
229  //
230  ASSERT( BufferAllocated == FALSE );
231  ASSERT( Buffer == NULL );
232  return;
233  }
234 
235  RtlCopyMemory(
236  ControlVector->Buffer,
237  Buffer,
238  Length
239  );
240 
241  //
242  // cleanup the temporary buffer
243  //
244  if( BufferAllocated == TRUE ) {
245  ExFreePool( Buffer );
246  }
247  if( CellToRelease != HCELL_NIL ) {
248  HvReleaseCell(SystemHive,CellToRelease);
249  }
250  }
251 
252  if (ControlVector->Type != NULL) {
253  *(ControlVector->Type) = ValueBody->Type;
254  }
255  }
256  }
257 
258  //
259  // Stash the length of result (-1 if nothing was found)
260  //
261 
262  if (ControlVector->BufferLength != NULL) {
263  *(ControlVector->BufferLength) = Length;
264  }
265 
266  ControlVector++;
267  }
268 
269  //
270  // Get the default locale ID for the system from the registry.
271  //
272 
273  if (CmDefaultLanguageIdType == REG_SZ) {
274  PsDefaultSystemLocaleId = (LCID) CmpConvertLangId(
275  CmDefaultLanguageId,
276  CmDefaultLanguageIdLength);
277  } else {
278  PsDefaultSystemLocaleId = 0x00000409;
279  }
280 
281  //
282  // Get the install (native UI) language ID for the system from the registry.
283  //
284 
285  if (CmInstallUILanguageIdType == REG_SZ) {
286  PsInstallUILanguageId = CmpConvertLangId(
287  CmInstallUILanguageId,
288  CmInstallUILanguageIdLength);
289  } else {
290  PsInstallUILanguageId = LANGIDFROMLCID(PsDefaultSystemLocaleId);
291  }
292 
293  //
294  // Set the default thread locale to the default system locale
295  // for now. This will get changed as soon as somebody logs in.
296  // Use the install (native) language id as our default UI language id.
297  // This also will get changed as soon as somebody logs in.
298  //
299 
300  PsDefaultThreadLocaleId = PsDefaultSystemLocaleId;
301  PsDefaultUILanguageId = PsInstallUILanguageId;
302 }

306 CmpWalkPath(
307  PHHIVE SystemHive,
308  HCELL_INDEX ParentCell,
309  PWSTR Path
310  )
311 /*++
312 
313 Routine Description:
314 
315  Walk the path.
316 
317 Arguments:
318 
319  SystemHive - hive
320 
321  ParentCell - where to start
322 
323  Path - string to walk
324 
325 Return Value:
326 
327  HCELL_INDEX of found key cell, or HCELL_NIL for error
328 
329 --*/
330 {
331  UNICODE_STRING PathString;
332  UNICODE_STRING NextName;
333  BOOLEAN Last;
334  HCELL_INDEX KeyCell;
335  PCM_KEY_NODE Node;
336 
337  //
338  // don't bother counting/releasing used cells
339  //
340  ASSERT( SystemHive->ReleaseCellRoutine == NULL );
341 
342  KeyCell = ParentCell;
343  RtlInitUnicodeString(&PathString, Path);
344 
345  while (TRUE) {
346 
347  CmpGetNextName(&PathString, &NextName, &Last);
348 
349  if (NextName.Length == 0) {
350  return KeyCell;
351  }
352 
353  Node = (PCM_KEY_NODE)HvGetCell(SystemHive,KeyCell);
354  if( Node == NULL ) {
355  //
356  // we couldn't map a view for the bin containing this cell
357  //
358  return HCELL_NIL;
359  }
360  KeyCell = CmpFindSubKeyByName(SystemHive,
361  Node,
362  &NextName);
363 
364  if (KeyCell == HCELL_NIL) {
365  return HCELL_NIL;
366  }
367  }
368 }

370 LANGID
371 CmpConvertLangId(
372  PWSTR LangIdString,
373  ULONG LangIdStringLength
374 )
375 {
376 
377 
378  USHORT i, Digit;
379  WCHAR c;
380  LANGID LangId;
381 
382  LangId = 0;
383  LangIdStringLength = LangIdStringLength / sizeof( WCHAR );
384  for (i=0; i < LangIdStringLength; i++) {
385  c = LangIdString[ i ];
386 
387  if (c >= L'0' && c <= L'9') {
388  Digit = c - L'0';
389 
390  } else if (c >= L'A' && c <= L'F') {
391  Digit = c - L'A' + 10;
392 
393  } else if (c >= L'a' && c <= L'f') {
394  Digit = c - L'a' + 10;
395 
396  } else {
397  break;
398  }
399 
400  if (Digit >= 16) {
401  break;
402  }
403 
404  LangId = (LangId << 4) | Digit;
405  }
406 
407  return LangId;
408 }
