9 Module Name:
10 
11  wsmanage.c
12 
13 Abstract:
14 
15  This module contains routines which manage the set of active working
16  set lists.
17 
18  Working set management is accomplished by a parallel group of actions
19 
20  1. Writing modified pages.
21 
22  2. Trimming working sets by :
23 
24  a) Aging pages by turning off access bits and incrementing age
25  counts for pages which haven't been accessed.
26  b) Estimating the number of unused pages in a working set and
27  keeping a global count of that estimate.
28  c) When getting tight on memory, replacing rather than adding
29  pages in a working set when a fault occurs in a working set
30  that has a significant proportion of unused pages.
31  d) When memory is tight, reducing (trimming) working sets which
32  are above their maximum towards their minimum. This is done
33  especially if there are a large number of available pages
34  in it.
35 
36  The metrics are set such that writing modified pages is typically
37  accomplished before trimming working sets, however, under certain cases
38  where modified pages are being generated at a very high rate, working
39  set trimming will be initiated to free up more pages.
40 
41  Once a process has had its working set raised above the minimum
42  specified, the process is put on the Working Set Expanded list and
43  is now eligible for trimming. Note that at this time the FLINK field
44  in the WorkingSetExpansionLink has an address value.
45 
46 --*/

514  //

522  // granted, don't trim it.
523  //
524

526 
527  MiDetachSession ();

529  return FALSE;
530  }

679 ULONG
680 MiComputeSystemTrimCriteria (
681  IN PMMWS_TRIM_CRITERIA Criteria
682  )
683 
684 /*++
685 
686 Routine Description:
687 
688  Decide whether to trim, age or adjust claim estimations at this time.
689 
690 Arguments:
691 
692  Criteria - Supplies a pointer to the trim criteria information. Various
693  fields in this structure are set as needed by this routine.
694 
695 Return Value:
696 
697  The actions our caller should apply to the working sets.
698 
699 Environment:
700 
701  Kernel mode. No locks held. APC level or below.
702 
703  This is called at least once per second on entry to MmWorkingSetManager.
704 
705 --*/
706 
707 {
708  KIRQL OldIrql;
709  ULONG OutFlags;
710  PFN_NUMBER Available;
711  ULONG StandbyRemoved;
712  ULONG StandbyTemp;
713  PFN_NUMBER PlentyFreePages;
714 
715  //
716  // Check the number of pages available to see if any trimming (or aging)
717  // is really required.
718  //
719 
720  Available = MmAvailablePages;
721 
722  StandbyRemoved = MmStandbyRePurposed;
723 
724  //
725  // If the counter wrapped, it's ok to just ignore it this time around.
726  //
727 
728  if (StandbyRemoved <= MiLastStandbyRePurposed) {
729  MiLastStandbyRePurposed = StandbyRemoved;
730  StandbyRemoved = 0;
731  }
732  else {
733 
734  //
735  // The value is nonzero, we need to synchronize so we get a coordinated
736  // snapshot of both values.
737  //
738 
739  LOCK_PFN (OldIrql);
740  Available = MmAvailablePages;
741  StandbyRemoved = MmStandbyRePurposed;
742  UNLOCK_PFN (OldIrql);
743 
744  if (StandbyRemoved <= MiLastStandbyRePurposed) {
745  MiLastStandbyRePurposed = StandbyRemoved;
746  StandbyRemoved = 0;
747  }
748  else {
749  StandbyTemp = StandbyRemoved;
750  StandbyRemoved -= MiLastStandbyRePurposed;
751  MiLastStandbyRePurposed = StandbyTemp;
752  }
753  }
754 
755  //
756  // MmPlentyFreePages can change dynamically, so snap it now.
757  //
758 
759  PlentyFreePages = MmPlentyFreePages;
760 
761  //
762  // If we're low on pages, or we've been replacing within a given
763  // working set, or we've been cannibalizing a large number of standby
764  // pages, then trim now.
765  //
766 
767  if ((Available <= PlentyFreePages) ||
768  (MiReplacing == TRUE) ||
769  (StandbyRemoved >= (Available >> 2))) {
770 
771  //
772  // Only pulse the event if it is not already set. This saves
773  // on dispatcher lock contention, but more importantly since
774  // KePulse always clears the event it saves us having to check
775  // whether to set it (and potentially do the setting) afterwards.
776  //
777 
778  if (MiLowMemoryEvent->Header.SignalState == 0) {
779  LOCK_PFN (OldIrql);
780  if (MiLowMemoryEvent->Header.SignalState == 0) {
781  KePulseEvent (MiLowMemoryEvent, 0, FALSE);
782  }
783  UNLOCK_PFN (OldIrql);
784  }
785 
786  //
787  // Inform our caller to start trimming since we're below
788  // plenty pages - order the list so the bigger working sets are
789  // in front so our caller trims those first.
790  //
791 
792  Criteria->NumPasses = 0;
793  Criteria->DesiredFreeGoal = PlentyFreePages + (PlentyFreePages / 2);
794  Criteria->NewTotalClaim = 0;
795  Criteria->NewTotalEstimatedAvailable = 0;
796 
797  //
798  // If more than 25% of the available pages were recycled standby
799  // pages, then trim more aggressively in an attempt to get more of the
800  // cold pages into standby for the next pass.
801  //
802 
803  if (StandbyRemoved >= (Available >> 2)) {
804  Criteria->TrimAllPasses = TRUE;
805  }
806  else {
807  Criteria->TrimAllPasses = FALSE;
808  }
809 
810 #if DBG
811  if (MmDebug & MM_DBG_WS_EXPANSION) {
812  DbgPrintEx (DPFLTR_MM_ID, DPFLTR_TRACE_LEVEL,
813  "\nMM-wsmanage: Desired = %ld, Avail %ld\n",
814  Criteria->DesiredFreeGoal, MmAvailablePages);
815  }
816 #endif
817 
818  //
819  // No need to lock synchronize the MiReplacing clearing as it
820  // gets set every time a page replacement happens anyway.
821  //
822 
823  MiReplacing = FALSE;
824 
825  //
826  // Start trimming the bigger working sets first.
827  //
828 
829  MiRearrangeWorkingSetExpansionList ();
830 
831  OutFlags = MI_TRIM_ALL_WORKING_SETS;
832  }
833  else if (Available < MM_ENORMOUS_LIMIT) {
834 
835  //
836  // Don't trim but do age unused pages and estimate
837  // the amount available in working sets.
838  //
839 
840  OutFlags = MI_AGE_ALL_WORKING_SETS;
841  }
842  else {
843 
844  //
845  // There is an overwhelming surplus of memory and this is a big
846  // server then don't even bother aging at this point (note the claim
847  // and estimated available are not cleared so they may contain stale
848  // values, but at this level it doesn't really matter).
849  //
850 
851  OutFlags = 0;
852  }
853 
854  //
855  // See if any working set requests have been queued to us.
856  //
857 
858  OutFlags |= MiWorkingSetRequestFlags;
859 
860  ASSERT ((OutFlags & ~MI_WORKING_SET_FLAGS) == 0);
861 
862  return OutFlags;
863 }

890 
891  Kernel mode. Expansion lock held. APC level or below.
892

899  (Criteria->NumPasses >= MI_MAX_TRIM_PASSES)) {
900 
901  //
902  // We have enough pages or we trimmed as many as we're going to get.
903  //

905  return TRUE;
906  }

910  //

918  // thread may have the system cache locked but has been preempted

922  //

927  FALSE,

932  //

935 
936  //
937  // Now we have enough pages so break out.
938  //
939 
940  FinishedTrimming = TRUE;
941  }
942  else {
943 
944  //
945  // We don't have enough pages so let's do another pass.
946  // Go get the next working set list which is probably the
947  // one we put back before we gave up the processor.
948  //
949 
950  FinishedTrimming = FALSE;
951 
952  if (Criteria->NumPasses == 0) {
953  MiAdjustClaimParameters (FALSE);
954  }
955 
956  Criteria->NumPasses += 1;
957  Criteria->NewTotalClaim = 0;
958  Criteria->NewTotalEstimatedAvailable = 0;
959 
960  }
961 
962  LOCK_EXPANSION (OldIrql);
963 
964  return FinishedTrimming;

1543 MiAgeWorkingSet (

1548  IN OUT PPFN_NUMBER TotalEstimatedAvailable
1549  )
1550

1552 
1553 Routine Description:
1554 
1555  Age pages (clear the access bit or if the page hasn't been
1556  accessed, increment the age) for a portion of the working
1557  set. Also, walk through a sample of the working set
1558  building a set of counts of how old the pages are.
1559 
1560  The counts are used to create a claim of the amount
1561  the system can steal from this process if memory
1562  becomes tight.
1563

1565

1567

1569  added to the availability estimation.

1572  control to prevent excessive aging on large systems with
1573  many processes.
1574 
1575  TotalClaim - Supplies a pointer to system wide claim to update.
1576 
1577  TotalEstimatedAvailable - Supplies a pointer to system wide estimate
1578  to update.
1579 
1580 Return Value:
1581 
1582  None
1583

1585 
1586  Kernel mode, APCs disabled, working set mutex. PFN lock NOT held.
1587 
1588 --*/

1592  WSLE_NUMBER LastEntry;
1593  WSLE_NUMBER StartEntry;
1594  WSLE_NUMBER FirstDynamic;
1595  WSLE_NUMBER CurrentEntry;
1596  PMMWSL WorkingSetList;
1597  PMMWSLE Wsle;
1598  PMMPTE PointerPte;

1600  WSLE_NUMBER Claim;
1601  ULONG Estimate;

1607  WSLE_NUMBER Temp;
1608  ULONG i;
1609

1611  Wsle = WorkingSetList->Wsle;
1612  AgeSize = 0;
1613 
1614  LastEntry = WorkingSetList->LastEntry;
1615  FirstDynamic = WorkingSetList->FirstDynamic;
1616 
1617  if (DoAging == TRUE) {
1618 
1619  //
1620  // Clear the used bits or increment the age of a portion of the
1621  // working set.
1622  //
1623  // Try to walk the entire working set every 2^MI_AGE_AGING_SHIFT
1624  // seconds.
1625  //
1626 
1627  if (VmSupport->WorkingSetSize > WorkingSetList->FirstDynamic) {
1628  NumberToExamine = (VmSupport->WorkingSetSize - WorkingSetList->FirstDynamic) >> MiAgingShift;
1629

1632  // terabytes so limit the absolute walk.
1633  //

1641  // have huge numbers of processes - checking the aggregate number
1642  // of working set list entries scanned prevents this situation
1643  // from triggering excessive scanning.
1644  //
