216 /*++
217 
218 Routine Description:
219 
220  This function creates a profile object.
221 
222 Arguments:
223 
224  ProfileHandle - Supplies a pointer to a variable that will receive
225  the profile object handle.
226 
227  Process - Optionally, supplies the handle to the process whose
228  address space to profile. If the value is NULL (0), then
229  all address spaces are included in the profile.
230 
231  RangeBase - Supplies the address of the first byte of the address
232  space for which profiling information is to be collected.
233 
234 
235  RangeSize - Supplies the size of the range to profile in the
236  address space. RangeBase and RangeSize are interpreted
237  such that RangeBase <= address < RangeBase+RangeSize
238  will generate a profile hit.
239 
240  BucketSize - Supplies the LOG base 2 of the size of the profiling
241  bucket. Thus, BucketSize = 2 yields four-byte
242  buckets, BucketSize = 7 yields 128-byte buckets.
243  All profile hits in a given bucket will increment
244  the corresponding counter in Buffer. Buckets
245  cannot be smaller than a ULONG. The acceptable range
246  of this value is 2 to 30 inclusive.
247 
248  Buffer - Supplies an array of ULONGs. Each ULONG is a hit counter,
249  which records the number of hits of the corresponding
250  bucket.
251 
252  BufferSize - Size in bytes of Buffer.
253 
254  ProfileSource - Supplies the source for the profile interrupt
255 
256  Affinity - Supplies the processor set for the profile interrupt
257 
258 --*/
259 
260 {
261 
262  PEPROFILE Profile;
263  HANDLE Handle;
264  KPROCESSOR_MODE PreviousMode;
265  NTSTATUS Status;
266  PEPROCESS ProcessAddress;
267  OBJECT_ATTRIBUTES ObjectAttributes;
268  BOOLEAN HasPrivilege = FALSE;
269  ULONG Segment = FALSE;
270 #ifdef i386
271  USHORT PowerOf2;
272 #endif
273 
274  //
275  // Verify that the base and size arguments are reasonable.
276  //
277 
278  if (BufferSize == 0) {
279  return STATUS_INVALID_PARAMETER_7;
280  }
281 
282  //
283  // Improper use of bucket size. If bucket size is zero, and
284  // RangeBase < 64K, then create a profile object to attach
285  // to a non-flat code segment. In this case, RangeBase is
286  // the non-flat CS for this profile object.
287  //
288 
289 #ifdef i386
290 
291  if ((BucketSize == 0) && (RangeBase < (PVOID)(64 * 1024))) {
292 
293  if (BufferSize < sizeof(ULONG)) {
294  return STATUS_INVALID_PARAMETER_7;
295  }
296 
297  Segment = (ULONG)RangeBase;
298  RangeBase = 0;
299  BucketSize = RangeSize / (BufferSize / sizeof(ULONG));
300 
301  //
302  // Convert Bucket size of log2(BucketSize)
303  //
304  PowerOf2 = 0;
305  BucketSize = BucketSize - 1;
306  while (BucketSize >>= 1) {
307  PowerOf2++;
308  }
309 
310  BucketSize = PowerOf2 + 1;
311 
312  if (BucketSize < 2) {
313  BucketSize = 2;
314  }
315  }
316 
317 #endif
318 
319  if ((BucketSize > 31) || (BucketSize < 2)) {
320  return STATUS_INVALID_PARAMETER;
321  }
322 
323  if ((RangeSize >> (BucketSize - 2)) > BufferSize) {
324  return STATUS_BUFFER_TOO_SMALL;
325  }
326 
327  if (((ULONG_PTR)RangeBase + RangeSize) < RangeSize) {
328  return STATUS_BUFFER_OVERFLOW;
329  }
330 
331  //
332  // Establish an exception handler, probe the output handle address, and
333  // attempt to create a profile object. If the probe fails, then return the
334  // exception code as the service status. Otherwise return the status value
335  // returned by the object insertion routine.
336  //
337 
338  try {
339  //
340  // Get previous processor mode and probe output handle address if
341  // necessary.
342  //
343 
344  PreviousMode = KeGetPreviousMode ();
345 
346  if (PreviousMode != KernelMode) {
347  ProbeForWriteHandle(ProfileHandle);
348 
349  ProbeForWrite(Buffer,
350  BufferSize,
351  sizeof(ULONG));
352  }
353 
354  //
355  // If an exception occurs during the probe of the output handle address,
356  // then always handle the exception and return the exception code as the
357  // status value.
358  //
359 
360  } except (EXCEPTION_EXECUTE_HANDLER) {
361  return GetExceptionCode();
362  }
363 
364  if (!ARGUMENT_PRESENT(Process)) {
365 
366  //
367  // Don't attach segmented profile objects to all processes
368  //
369 
370  if (Segment) {
371  return STATUS_INVALID_PARAMETER;
372  }
373 
374  //
375  // Profile all processes. Make sure that the specified
376  // address range is in system space, unless SeSystemProfilePrivilege.
377  //
378 
379  if (RangeBase <= MM_HIGHEST_USER_ADDRESS) {
380 
381  //
382  // Check for privilege before allowing a user to profile
383  // all processes and USER addresses.
384  //
385 
386  if (PreviousMode != KernelMode) {
387  HasPrivilege = SeSinglePrivilegeCheck(
388  SeSystemProfilePrivilege,
389  PreviousMode
390  );
391 
392  if (!HasPrivilege) {
393 #if DBG
394  DbgPrint("SeSystemProfilePrivilege needed to profile all USER addresses.\n");
395 #endif //DBG
396  return( STATUS_PRIVILEGE_NOT_HELD );
397  }
398 
399  }
400  }
401 
402  ProcessAddress = NULL;
403 
404 
405  } else {
406 
407  //
408  // Reference the specified process.
409  //
410 
411  Status = ObReferenceObjectByHandle ( Process,
412  PROCESS_QUERY_INFORMATION,
413  PsProcessType,
414  PreviousMode,
415  (PVOID *)&ProcessAddress,
416  NULL );
417 
418  if (!NT_SUCCESS(Status)) {
419  return Status;
420  }
421  }
422 
423  InitializeObjectAttributes( &ObjectAttributes,
424  NULL,
425  OBJ_EXCLUSIVE,
426  NULL,
427  NULL );
428 
429  Status = ObCreateObject( KernelMode,
430  ExProfileObjectType,
431  &ObjectAttributes,
432  PreviousMode,
433  NULL,
434  sizeof(EPROFILE),
435  0,
436  sizeof(EPROFILE) + sizeof(KPROFILE),
437  (PVOID *)&Profile);
438 
439  //
440  // If the profile object was successfully allocated, initialize
441  // the profile object.
442  //
443  if (NT_SUCCESS(Status)) {
444 
445 
446  if (ProcessAddress != NULL) {
447  Profile->Process = &ProcessAddress->Pcb;
448  } else {
449  Profile->Process = NULL;
450  }
451 
452  Profile->RangeBase = RangeBase;
453  Profile->RangeSize = RangeSize;
454  Profile->Buffer = Buffer;
455  Profile->BufferSize = BufferSize;
456  Profile->BucketSize = BucketSize;
457  Profile->LockedBufferAddress = NULL;
458  Profile->Segment = Segment;
459  Profile->ProfileSource = ProfileSource;
460  Profile->Affinity = Affinity;
461 
462  Status = ObInsertObject(Profile,
463  NULL,
464  PROFILE_CONTROL,
465  0,
466  (PVOID *)NULL,
467  &Handle);
468  //
469  // If the profile object was successfully inserted in the current
470  // process' handle table, then attempt to write the profile object
471  // handle value. If the write attempt fails, then do not report
472  // an error. When the caller attempts to access the handle value,
473  // an access violation will occur.
474  //
475  if (NT_SUCCESS(Status)) {
476  try {
477  *ProfileHandle = Handle;
478  } except(EXCEPTION_EXECUTE_HANDLER) {
479  }
480  }
481  } else {
482  //
483  // We failed, remove our reference to the process object.
484  //
485 
486  if (ProcessAddress != NULL) {
487  ObDereferenceObject (ProcessAddress);
488  }
489  }
490 
491  //
492  // Return service status.
493  //
494 
495  return Status;
496 }

503 /*++
504 
505 Routine Description:
506 
507  The NtStartProfile routine starts the collecting data for the
508  specified profile object. This involved allocating nonpaged
509  pool to lock the specified buffer in memory, creating a kernel
510  profile object and starting collecting on that profile object.
511

513 
514  ProfileHandle - Supplies the profile handle to start profiling on.
515 
516 --*/

518 {
519 
520  KPROCESSOR_MODE PreviousMode;
521  NTSTATUS Status;
522  PEPROFILE Profile;
523  PKPROFILE ProfileObject;
524  PVOID LockedVa;
525  BOOLEAN State;
526  PMDL Mdl;
527 
528  PreviousMode = KeGetPreviousMode();
529 
530  Status = ObReferenceObjectByHandle (ProfileHandle,
531  PROFILE_CONTROL,
532  ExProfileObjectType,
533  PreviousMode,
534  &Profile,
535  NULL);
536  if (!NT_SUCCESS(Status)) {
537  return Status;
538  }
539 
540  //
541  // Acquire the profile state mutex so two threads can't
542  // operate on the same profile object simultaneously.
543  //
544 
545  KeWaitForSingleObject (&ExpProfileStateMutex,
546  Executive,
547  KernelMode,
548  FALSE,
549  (PLARGE_INTEGER)NULL);
550 
551  //
552  // Make sure profiling is not already enabled.
553  //
554 
555  if (Profile->LockedBufferAddress != NULL) {
556  KeReleaseMutex (&ExpProfileStateMutex, FALSE);
557  ObDereferenceObject (Profile);
558  return STATUS_PROFILING_NOT_STOPPED;
559  }
560 
561  if (ExpCurrentProfileUsage == ACTIVE_PROFILE_LIMIT) {
562  KeReleaseMutex (&ExpProfileStateMutex, FALSE);
563  ObDereferenceObject (Profile);
564  return STATUS_PROFILING_AT_LIMIT;
565  }
566 
567  ProfileObject = ExAllocatePoolWithTag (NonPagedPool,
568  MmSizeOfMdl(Profile->Buffer,
569  Profile->BufferSize) +
570  sizeof(KPROFILE),
571  'forP');
572 
573  if (ProfileObject == NULL) {
574  KeReleaseMutex (&ExpProfileStateMutex, FALSE);
575  ObDereferenceObject (Profile);
576  return STATUS_INSUFFICIENT_RESOURCES;
577  }
578 
579  Mdl = (PMDL)(ProfileObject + 1);
580  Profile->Mdl = Mdl;
581  Profile->ProfileObject = ProfileObject;
582 
583  //
584  // Probe and lock the specified buffer.
585  //
586 
587  MmInitializeMdl(Mdl, Profile->Buffer, Profile->BufferSize);
588 
589  LockedVa = NULL;
590 
591  try {
592 
593  MmProbeAndLockPages (Mdl,
594  PreviousMode,
595  IoWriteAccess );
596 
597  } except (EXCEPTION_EXECUTE_HANDLER) {
598 
599  KeReleaseMutex (&ExpProfileStateMutex, FALSE);
600  ExFreePool (ProfileObject);
601  ObDereferenceObject (Profile);
602  return GetExceptionCode();
603  }
604 
605  //
606  // Since kernel space is specified below, this call cannot raise
607  // an exception.
608  //
609 
610  LockedVa = MmMapLockedPagesSpecifyCache (Profile->Mdl,
611  KernelMode,
612  MmCached,
613  NULL,
614  FALSE,
615  NormalPagePriority);
616 
617  if (LockedVa == NULL) {
618  KeReleaseMutex (&ExpProfileStateMutex, FALSE);
619 
620  MmUnlockPages (Mdl);
621  ExFreePool (ProfileObject);
622  ObDereferenceObject (Profile);
623  return STATUS_INSUFFICIENT_RESOURCES;
624  }
625 
626  //
627  // Initialize the profile object.
628  //
629 
630  KeInitializeProfile (ProfileObject,
631  Profile->Process,
632  Profile->RangeBase,
633  Profile->RangeSize,
634  Profile->BucketSize,
635  Profile->Segment,
636  Profile->ProfileSource,
637  Profile->Affinity);
638 
639  State = KeStartProfile (ProfileObject, LockedVa);
640  ASSERT (State != FALSE);
641 
642  Profile->LockedBufferAddress = LockedVa;
643 
644  KeReleaseMutex (&ExpProfileStateMutex, FALSE);
645  ObDereferenceObject (Profile);
646 
647  return STATUS_SUCCESS;
648 }
