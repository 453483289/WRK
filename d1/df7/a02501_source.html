33 MiSubsectionConsistent (

36 /*++
37 
38 Routine Description:
39 
40  This function checks to ensure the subsection is consistent.
41 
42 Arguments:
43 
44  Subsection - Supplies a pointer to the subsection to be checked.
45 
46 Return Value:
47 
48  None.
49 
50 --*/

52 {
53  ULONG Sectors;
54  ULONG FullPtes;
55 
56  //
57  // Compare the disk sectors (4K units) to the PTE allocation.
58  //
59 
60  Sectors = Subsection->NumberOfFullSectors;
61  if (Subsection->u.SubsectionFlags.SectorEndOffset) {
62  Sectors += 1;
63  }
64 
65  //
66  // Calculate how many PTEs are needed to map this number of sectors.
67  //
68

72  FullPtes += 1;
73  }
74 
75  if (FullPtes != Subsection->PtesInSubsection) {
76  DbgPrintEx (DPFLTR_MM_ID, DPFLTR_ERROR_LEVEL, "Mm: Subsection inconsistent (%x vs %x)\n",
77  FullPtes,

87  __in HANDLE SectionHandle,
88  __inout PLARGE_INTEGER NewSectionSize

91 /*++
92 
93 Routine Description:
94 
95  This function extends the size of the specified section. If
96  the current size of the section is greater than or equal to the
97  specified section size, the size is not updated.
98 
99 Arguments:
100 
101  SectionHandle - Supplies an open handle to a section object.
102 
103  NewSectionSize - Supplies the new size for the section object.
104 
105 Return Value:
106 
107  NTSTATUS.
108 
109 --*/
110 
111 {
112  KPROCESSOR_MODE PreviousMode;
113  PVOID Section;
114  NTSTATUS Status;
115  LARGE_INTEGER CapturedNewSectionSize;
116 
117  PAGED_CODE();
118 
119  //
120  // Check to make sure the new section size is accessible.
121  //
122 
123  PreviousMode = KeGetPreviousMode();
124 
125  if (PreviousMode != KernelMode) {
126 
127  try {
128 
129  ProbeForWriteSmallStructure (NewSectionSize,
130  sizeof(LARGE_INTEGER),
131  PROBE_ALIGNMENT (LARGE_INTEGER));
132 
133  CapturedNewSectionSize = *NewSectionSize;
134 
135  } except (EXCEPTION_EXECUTE_HANDLER) {
136 
137  //
138  // If an exception occurs during the probe or capture
139  // of the initial values, then handle the exception and
140  // return the exception code as the status value.
141  //
142 
143  return GetExceptionCode ();
144  }
145 
146  }
147  else {
148 
149  CapturedNewSectionSize = *NewSectionSize;
150  }
151 
152  //
153  // Reference the section object.
154  //
155 
156  Status = ObReferenceObjectByHandle (SectionHandle,
157  SECTION_EXTEND_SIZE,
158  MmSectionObjectType,
159  PreviousMode,
160  (PVOID *)&Section,
161  NULL);
162 
163  if (!NT_SUCCESS(Status)) {
164  return Status;
165  }
166 
167  Status = MmExtendSection (Section, &CapturedNewSectionSize, FALSE);
168 
169  ObDereferenceObject (Section);
170 
171  //
172  // Update the NewSectionSize field.
173  //
174 
175  try {
176 
177  //
178  // Return the captured section size.
179  //
180 
181  *NewSectionSize = CapturedNewSectionSize;
182 
183  } except (EXCEPTION_EXECUTE_HANDLER) {
184  NOTHING;
185  }
186 
187  return Status;
188 }

191 LOGICAL
192 MiAppendSubsectionChain (
193  IN PMSUBSECTION LastSubsection,
194  IN PMSUBSECTION ExtendedSubsectionHead
195  )
196 
197 /*++
198 
199 Routine Description:
200 
201  This non-pageable wrapper function extends the specified subsection chain.
202 
203 Arguments:
204 
205  LastSubsection - Supplies the last subsection in the existing control area.
206 
207  ExtendedSubsectionHead - Supplies an anchor pointing to the first
208  subsection in the chain to append.
209 
210 Return Value:
211 
212  TRUE if the chain was appended, FALSE if not.
213 
214 --*/
215 
216 {
217  KIRQL OldIrql;
218  PMSUBSECTION NewSubsection;
219 
220  ASSERT (ExtendedSubsectionHead->NextSubsection != NULL);
221 
222  ASSERT (ExtendedSubsectionHead->u.SubsectionFlags.SectorEndOffset == 0);
223 
224  NewSubsection = (PMSUBSECTION) ExtendedSubsectionHead->NextSubsection;
225 
226  LOCK_PFN (OldIrql);
227 
228  //
229  // This subsection may be extending a range that is already
230  // mapped by a VAD(s). There is no way to tell how many VADs
231  // already map it so if any do, just leave all the new subsections
232  // marked as not reclaimable until the control area is deleted.
233  //
234  // If however, there are no *USER* references to this control area,
235  // then the subsections can be marked as dynamic now. Note other
236  // portions of code (currently only prefetch) that issue "dereference
237  // from this subsection to the end of file" are safe because these
238  // portions create user sections first and so the first check below
239  // will be FALSE.
240  //
241 
242  if (LastSubsection->ControlArea->NumberOfUserReferences != 0) {
243 
244  //
245  // The caller has not allocated prototype PTEs and they are required.
246  // Return so the caller can allocate and retry.
247  //
248 
249  if (NewSubsection->SubsectionBase == NULL) {
250  ASSERT (NewSubsection->u.SubsectionFlags.SubsectionStatic == 0);
251  UNLOCK_PFN (OldIrql);
252  return FALSE;
253  }
254 #if DBG
255  do {
256  ASSERT (NewSubsection->u.SubsectionFlags.SubsectionStatic == 1);
257  NewSubsection = (PMSUBSECTION) NewSubsection->NextSubsection;
258  } while (NewSubsection != NULL);
259 #endif
260  }
261  else if (NewSubsection->SubsectionBase != NULL) {
262 
263  //
264  // The prototype PTEs are no longer required (user views went away)
265  // even though they were when the caller first built the subsections.
266  // Mark the subsections as dynamic now.
267  //
268 
269  do {
270  ASSERT (NewSubsection->u.SubsectionFlags.SubsectionStatic == 1);
271 
272  MI_SNAP_SUB (NewSubsection, 0x1);
273 
274  NewSubsection->u.SubsectionFlags.SubsectionStatic = 0;
275  NewSubsection->u2.SubsectionFlags2.SubsectionConverted = 1;
276  NewSubsection->NumberOfMappedViews = 1;
277 
278  MiRemoveViewsFromSection (NewSubsection,
279  NewSubsection->PtesInSubsection);
280 
281  MiExtendedSubsectionsConvertedToDynamic += 1;
282 
283  MI_SNAP_SUB (NewSubsection, 0x2);
284  NewSubsection = (PMSUBSECTION) NewSubsection->NextSubsection;
285  } while (NewSubsection != NULL);
286  }
287 
288  LastSubsection->u.SubsectionFlags.SectorEndOffset = 0;
289 
290  LastSubsection->NumberOfFullSectors = ExtendedSubsectionHead->NumberOfFullSectors;
291 
292  //
293  // A memory barrier is needed to ensure the writes initializing the
294  // subsection fields are visible prior to linking the subsection into
295  // the chain. This is because some reads from these fields are done
296  // lock free for improved performance.
297  //
298 
299  KeMemoryBarrier ();
300 
301  LastSubsection->NextSubsection = ExtendedSubsectionHead->NextSubsection;
302 
303  UNLOCK_PFN (OldIrql);
304 
305  return TRUE;
306 }

309 NTSTATUS
310 MmExtendSection (
311  IN PVOID SectionToExtend,
312  IN OUT PLARGE_INTEGER NewSectionSize,
313  IN ULONG IgnoreFileSizeChecking
314  )
315 
316 /*++
317 
318 Routine Description:
319 
320  This function extends the size of the specified section. If
321  the current size of the section is greater than or equal to the
322  specified section size, the size is not updated.
323 
324 Arguments:
325 
326  Section - Supplies a pointer to a referenced section object.
327 
328  NewSectionSize - Supplies the new size for the section object.
329 
330  IgnoreFileSizeChecking - Supplies the value TRUE is file size
331  checking should be ignored (i.e., it
332  is being called from a file system which
333  has already done the checks). FALSE
334  if the checks still need to be made.
335 
336 Return Value:
337 
338  NTSTATUS.
339 
340 --*/
341 
342 {
343  LOGICAL Appended;
344  PMMPTE ProtoPtes;
345  MMPTE TempPte;
346  PCONTROL_AREA ControlArea;
347  PSEGMENT Segment;
348  PSECTION Section;
349  PSUBSECTION LastSubsection;
350  PSUBSECTION Subsection;
351  PMSUBSECTION ExtendedSubsection;
352  MSUBSECTION ExtendedSubsectionHead;
353  PMSUBSECTION LastExtendedSubsection;
354  UINT64 RequiredPtes;
355  UINT64 NumberOfPtes;
356  UINT64 TotalNumberOfPtes;
357  ULONG PtesUsed;
358  ULONG UnusedPtes;
359  UINT64 AllocationSize;
360  UINT64 RunningSize;
361  UINT64 EndOfFile;
362  NTSTATUS Status;
363  LARGE_INTEGER NumberOf4KsForEntireFile;
364  LARGE_INTEGER Starting4K;
365  LARGE_INTEGER NextSubsection4KStart;
366  LARGE_INTEGER Last4KChunk;
367  ULONG PartialSize;
368  SIZE_T AllocationFragment;
369  PKTHREAD CurrentThread;
370 
371  PAGED_CODE();
372 
373  Section = (PSECTION)SectionToExtend;
374 
375  //
376  // Make sure the section is really extendable - physical and
377  // image sections are not.
378  //
379 
380  Segment = Section->Segment;
381  ControlArea = Segment->ControlArea;
382 
383  if ((ControlArea->u.Flags.PhysicalMemory || ControlArea->u.Flags.Image) ||
384  (ControlArea->FilePointer == NULL)) {
385  return STATUS_SECTION_NOT_EXTENDED;
386  }
387 
388  //
389  // Acquire the section extension mutex, this blocks other threads from
390  // updating the size at the same time.
391  //
392 
393  CurrentThread = KeGetCurrentThread ();
394  KeEnterCriticalRegionThread (CurrentThread);
395  ExAcquireResourceExclusiveLite (&MmSectionExtendResource, TRUE);
396 
397  //
398  // Calculate the number of prototype PTE chunks to build for this section.
399  // A subsection is also allocated for each chunk as all the prototype PTEs
400  // in any given chunk are initially encoded to point at the same subsection.
401  //
402  // The maximum total section size is 16PB (2^54). This is because the
403  // StartingSector4132 field in each subsection, ie: 2^42-1 bits of file
404  // offset where the offset is in 4K (not pagesize) units. Thus, a
405  // subsection may describe a *BYTE* file start offset of maximum
406  // 2^54 - 4K.
407  //
408  // Each subsection can span at most 16TB - 64K. This is because the
409  // NumberOfFullSectors and various other fields in the subsection are
410  // ULONGs. In reality, this is a nonissue as far as maximum section size
411  // is concerned because any number of subsections can be chained together
412  // and in fact, subsections are allocated to span less to facilitate
413  // efficient dynamic prototype PTE trimming and reconstruction.
414  //
415 
416  if (NewSectionSize->QuadPart > MI_MAXIMUM_SECTION_SIZE) {
417  Status = STATUS_SECTION_TOO_BIG;
418  goto ReleaseAndReturn;
419  }
420 
421  NumberOfPtes = (NewSectionSize->QuadPart + PAGE_SIZE - 1) >> PAGE_SHIFT;
422 
423  if (ControlArea->u.Flags.WasPurged == 0) {
424 
425  if ((UINT64)NewSectionSize->QuadPart <= (UINT64)Section->SizeOfSection.QuadPart) {
426  *NewSectionSize = Section->SizeOfSection;
427  goto ReleaseAndReturnSuccess;
428  }
429  }
430 
431  //
432  // If a file handle was specified, set the allocation size of the file.
433  //
434 
435  if (IgnoreFileSizeChecking == FALSE) {
436 
437  //
438  // Release the resource so we don't deadlock with the file
439  // system trying to extend this section at the same time.
440  //
441 
442  ExReleaseResourceLite (&MmSectionExtendResource);
443 
444  //
445  // Get a different resource to single thread query/set operations.
446  //
447 
448  ExAcquireResourceExclusiveLite (&MmSectionExtendSetResource, TRUE);
449 
450  //
451  // Query the file size to see if this file really needs extending.
452  //
453  // If the specified size is less than the current size, return
454  // the current size.
455  //
456 
457  Status = FsRtlGetFileSize (ControlArea->FilePointer,
458  (PLARGE_INTEGER)&EndOfFile);
459 
460  if (!NT_SUCCESS (Status)) {
461  ExReleaseResourceLite (&MmSectionExtendSetResource);
462  KeLeaveCriticalRegionThread (CurrentThread);
463  return Status;
464  }
465 
466  if ((UINT64)NewSectionSize->QuadPart > EndOfFile) {
467 
468  //
469  // Don't allow section extension unless the section was originally
470  // created with write access. The check couldn't be done at create
471  // time without breaking existing binaries, so the caller gets the
472  // error at this point instead.
473  //
474 
475  if (((Section->InitialPageProtection & PAGE_READWRITE) |
476  (Section->InitialPageProtection & PAGE_EXECUTE_READWRITE)) == 0) {
477  ExReleaseResourceLite (&MmSectionExtendSetResource);
478  KeLeaveCriticalRegionThread (CurrentThread);
479  return STATUS_SECTION_NOT_EXTENDED;
480  }
481 
482  //
483  // Current file is smaller, attempt to set a new end of file.
484  //
485 
486  EndOfFile = *(PUINT64)NewSectionSize;
487 
488  Status = FsRtlSetFileSize (ControlArea->FilePointer,
489  (PLARGE_INTEGER)&EndOfFile);
490 
491  if (!NT_SUCCESS (Status)) {
492  ExReleaseResourceLite (&MmSectionExtendSetResource);
493  KeLeaveCriticalRegionThread (CurrentThread);
494  return Status;
495  }
496  }
497 
498  if (Segment->ExtendInfo) {
499  KeAcquireGuardedMutex (&MmSectionBasedMutex);
500  if (Segment->ExtendInfo) {
501  Segment->ExtendInfo->CommittedSize = EndOfFile;
502  }
503  KeReleaseGuardedMutex (&MmSectionBasedMutex);
504  }
505 
506  //
507  // Release the query/set resource and reacquire the extend section
508  // resource.
509  //
510 
511  ExReleaseResourceLite (&MmSectionExtendSetResource);
512  ExAcquireResourceExclusiveLite (&MmSectionExtendResource, TRUE);
513  }
514 
515  //
516  // Find the last subsection.
517  //
518 
519  ASSERT (ControlArea->u.Flags.GlobalOnlyPerSession == 0);
520 
521  if (((PMAPPED_FILE_SEGMENT)Segment)->LastSubsectionHint != NULL) {
522  LastSubsection = (PSUBSECTION) ((PMAPPED_FILE_SEGMENT)Segment)->LastSubsectionHint;
523  }
524  else {
525  if (ControlArea->u.Flags.Rom == 1) {
526  LastSubsection = (PSUBSECTION)((PLARGE_CONTROL_AREA)ControlArea + 1);
527  }
528  else {
529  LastSubsection = (PSUBSECTION)(ControlArea + 1);
530  }
531  }
532 
533  while (LastSubsection->NextSubsection != NULL) {
534  ASSERT (LastSubsection->UnusedPtes == 0);
535  LastSubsection = LastSubsection->NextSubsection;
536  }
537 
538  MI_CHECK_SUBSECTION (LastSubsection);
539 
540  //
541  // Does the structure need extending?
542  //
543 
544  TotalNumberOfPtes = (((UINT64)Segment->SegmentFlags.TotalNumberOfPtes4132) << 32) | Segment->TotalNumberOfPtes;
545 
546  if (NumberOfPtes <= TotalNumberOfPtes) {
547 
548  //
549  // The segment is already large enough, just update
550  // the section size and return.
551  //
552 
553  Section->SizeOfSection = *NewSectionSize;
554  if (Segment->SizeOfSegment < (UINT64)NewSectionSize->QuadPart) {
555 
556  //
557  // Only update if it is really bigger.
558  //
559 
560  Segment->SizeOfSegment = *(PUINT64)NewSectionSize;
561 
562  Mi4KStartFromSubsection(&Starting4K, LastSubsection);
563 
564  Last4KChunk.QuadPart = (NewSectionSize->QuadPart >> MM4K_SHIFT) - Starting4K.QuadPart;
565 
566  ASSERT (Last4KChunk.HighPart == 0);
567 
568  LastSubsection->NumberOfFullSectors = Last4KChunk.LowPart;
569  LastSubsection->u.SubsectionFlags.SectorEndOffset =
570  NewSectionSize->LowPart & MM4K_MASK;
571  MI_CHECK_SUBSECTION (LastSubsection);
572  }
573  goto ReleaseAndReturnSuccess;
574  }
575 
576  //
577  // Add new structures to the section - locate the last subsection
578  // and add there.
579  //
580 
581  RequiredPtes = NumberOfPtes - TotalNumberOfPtes;
582  PtesUsed = 0;
583 
584  if (RequiredPtes < LastSubsection->UnusedPtes) {
585 
586  //
587  // There are ample PTEs to extend the section already allocated.
588  //
589 
590  PtesUsed = (ULONG) RequiredPtes;
591  RequiredPtes = 0;
592 
593  }
594  else {
595  PtesUsed = LastSubsection->UnusedPtes;
596  RequiredPtes -= PtesUsed;
597  }
598 
599  LastSubsection->PtesInSubsection += PtesUsed;
600  LastSubsection->UnusedPtes -= PtesUsed;
601  Segment->SizeOfSegment += (UINT64)PtesUsed * PAGE_SIZE;
602 
603  TotalNumberOfPtes += PtesUsed;
604 
605  Segment->TotalNumberOfPtes = (ULONG) TotalNumberOfPtes;
606  if (TotalNumberOfPtes >= 0x100000000) {
607  Segment->SegmentFlags.TotalNumberOfPtes4132 = (ULONG_PTR)(TotalNumberOfPtes >> 32);
608  }
609 
610  if (RequiredPtes == 0) {
611 
612  //
613  // There is no extension necessary, update the high VBN.
614  //
615 
616  Mi4KStartFromSubsection(&Starting4K, LastSubsection);
617 
618  Last4KChunk.QuadPart = (NewSectionSize->QuadPart >> MM4K_SHIFT) - Starting4K.QuadPart;
619 
620  ASSERT (Last4KChunk.HighPart == 0);
621 
622  LastSubsection->NumberOfFullSectors = Last4KChunk.LowPart;
623 
624  LastSubsection->u.SubsectionFlags.SectorEndOffset =
625  NewSectionSize->LowPart & MM4K_MASK;
626  MI_CHECK_SUBSECTION (LastSubsection);
627  }
628  else {
629 
630  //
631  // An extension is required.
632  //
633  // Allocate the subsection(s) now.
634  //
635  // If there are any user views, then also allocate the prototype
636  // PTEs now (because a user view may be for an extended VAD which
637  // would already be encompassing this new extension). If there
638  // are no user views, then don't allocate the prototype PTEs until
639  // the views are actually mapped (system views never pre-extend past
640  // the end of the file).
641  //
642 
643  NumberOf4KsForEntireFile.QuadPart = Segment->SizeOfSegment >> MM4K_SHIFT;
644  AllocationSize = MI_ROUND_TO_SIZE (RequiredPtes * sizeof(MMPTE), PAGE_SIZE);
645 
646  AllocationFragment = MmAllocationFragment;
647 
648  RunningSize = 0;
649 
650  ExtendedSubsectionHead = *(PMSUBSECTION)LastSubsection;
651 
652  LastExtendedSubsection = &ExtendedSubsectionHead;
653 
654  ASSERT (LastExtendedSubsection->NextSubsection == NULL);
655 
656  SATISFY_OVERZEALOUS_COMPILER (NextSubsection4KStart.QuadPart = 0);
657 
658  do {
659 
660  //
661  // Bound the size of each prototype PTE allocation so both :
662  // 1. it can succeed even in cases where the pool is fragmented.
663  // 2. later on last control area dereference, each subsection
664  // is converted to dynamic and can be pruned/recreated
665  // individually without losing (or requiring) contiguous pool.
666  //
667 
668  if (AllocationSize - RunningSize > AllocationFragment) {
669  PartialSize = (ULONG) AllocationFragment;
670  }
671  else {
672  PartialSize = (ULONG) (AllocationSize - RunningSize);
673  }
674 
675  //
676  // Allocate an extended subsection.
677  //
678 
679  ExtendedSubsection = (PMSUBSECTION) ExAllocatePoolWithTag (NonPagedPool,
680  sizeof(MSUBSECTION),
681  'dSmM');
682  if (ExtendedSubsection == NULL) {
683  goto ExtensionFailed;
684  }
685 
686  ExtendedSubsection->SubsectionBase = NULL;
687  ExtendedSubsection->NextSubsection = NULL;
688 
689  LastExtendedSubsection->NextSubsection = (PSUBSECTION) ExtendedSubsection;
690 
691  ASSERT (ControlArea->FilePointer != NULL);
692 
693  ExtendedSubsection->u.LongFlags = 0;
694 
695  ExtendedSubsection->ControlArea = ControlArea;
696 
697  ExtendedSubsection->PtesInSubsection = PartialSize / sizeof(MMPTE);
698  ExtendedSubsection->UnusedPtes = 0;
699 
700  RunningSize += PartialSize;
701 
702  if (RunningSize > (RequiredPtes * sizeof(MMPTE))) {
703  UnusedPtes = (ULONG)(RunningSize / sizeof(MMPTE) - RequiredPtes);
704  ExtendedSubsection->PtesInSubsection -= UnusedPtes;
705  ExtendedSubsection->UnusedPtes = UnusedPtes;
706  }
707 
708  ExtendedSubsection->u.SubsectionFlags.Protection =
709  (unsigned) Segment->SegmentPteTemplate.u.Soft.Protection;
710 
711  ExtendedSubsection->DereferenceList.Flink = NULL;
712  ExtendedSubsection->DereferenceList.Blink = NULL;
713  ExtendedSubsection->NumberOfMappedViews = 0;
714  ExtendedSubsection->u2.LongFlags2 = 0;
715 
716 
717  //
718  // Adjust the previous subsection to account for the new length.
719  // Note that since the next allocation in this loop may fail,
720  // the very first previous subsection changes are not rippled
721  // to the chained subsection until the loop completes successfully.
722  //
723 
724  if (LastExtendedSubsection == &ExtendedSubsectionHead) {
725 
726  Mi4KStartFromSubsection (&Starting4K, LastExtendedSubsection);
727 
728  Last4KChunk.QuadPart = NumberOf4KsForEntireFile.QuadPart -
729  Starting4K.QuadPart;
730 
731  if (LastExtendedSubsection->u.SubsectionFlags.SectorEndOffset) {
732  Last4KChunk.QuadPart += 1;
733  }
734 
735  ASSERT (Last4KChunk.HighPart == 0);
736 
737  LastExtendedSubsection->NumberOfFullSectors = Last4KChunk.LowPart;
738  LastExtendedSubsection->u.SubsectionFlags.SectorEndOffset = 0;
739 
740  //
741  // If the number of sectors doesn't completely fill the PTEs
742  // (only possible when the page size is not MM4K), then
743  // fill it now.
744  //
745 
746  if (LastExtendedSubsection->NumberOfFullSectors & ((1 << (PAGE_SHIFT - MM4K_SHIFT)) - 1)) {
747  LastExtendedSubsection->NumberOfFullSectors += 1;
748  }
749 
750  MI_CHECK_SUBSECTION (LastExtendedSubsection);
751 
752  Starting4K.QuadPart += LastExtendedSubsection->NumberOfFullSectors;
753  NextSubsection4KStart.QuadPart = Starting4K.QuadPart;
754  }
755  else {
756  NextSubsection4KStart.QuadPart += LastExtendedSubsection->NumberOfFullSectors;
757  }
758 
759  //
760  // Initialize the newly allocated subsection.
761  //
762 
763  Mi4KStartForSubsection (&NextSubsection4KStart, ExtendedSubsection);
764 
765  if (RunningSize < AllocationSize) {
766 
767  //
768  // Not the final subsection so all quantities are full pages.
769  //
770 
771  ExtendedSubsection->NumberOfFullSectors =
772  (PartialSize / sizeof (MMPTE)) << (PAGE_SHIFT - MM4K_SHIFT);
773  ExtendedSubsection->u.SubsectionFlags.SectorEndOffset = 0;
774  }
775  else {
776 
777  //
778  // The final subsection so quantities are not always full pages.
779  //
780 
781  Last4KChunk.QuadPart =
782  (NewSectionSize->QuadPart >> MM4K_SHIFT) - NextSubsection4KStart.QuadPart;
783 
784  ASSERT (Last4KChunk.HighPart == 0);
785 
786  ExtendedSubsection->NumberOfFullSectors = Last4KChunk.LowPart;
787 
788  ExtendedSubsection->u.SubsectionFlags.SectorEndOffset =
789  NewSectionSize->LowPart & MM4K_MASK;
790  }
791 
792  MI_CHECK_SUBSECTION (ExtendedSubsection);
