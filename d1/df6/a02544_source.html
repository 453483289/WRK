821  BaseAddress = MiGetVirtualAddressMappedByPte (PointerPte);
822  continue;
823  }
824 #endif

826  PointerPde = MiGetPdeAddress (BaseAddress);
827 
828  if (PointerPde->u.Hard.Valid == 0) {
829  PointerPde += 1;
830  PointerPte = MiGetVirtualAddressMappedByPte (PointerPde);
831  BaseAddress = MiGetVirtualAddressMappedByPte (PointerPte);
832  continue;
833  }
834  }
835 
836  //
837  // If the PTE is marked dirty (or writable) OR the BitMap says it's
838  // dirtied, then let the caller know.
839  //
840 
841  PteContents = *PointerPte;
842 
843  if ((PteContents.u.Hard.Valid == 1) &&
844  (MI_IS_PTE_DIRTY(PteContents))) {
845 
846  //
847  // For the uniprocessor x86, just the dirty bit is cleared.
848  // For all other platforms, the PTE writable bit must be
849  // disabled now so future writes trigger write watch updates.
850  //
851

853  Pfn1 = MI_PFN_ELEMENT(PageFrameIndex);
854  ASSERT (Pfn1->u3.e1.PrototypePte == 0);
855 
856  MI_MAKE_VALID_USER_PTE (TempPte,
857  PageFrameIndex,
858  Pfn1->OriginalPte.u.Soft.Protection,
859  PointerPte);
860 
861  WorkingSetIndex = MI_GET_WORKING_SET_FROM_PTE (&PteContents);
862  MI_SET_PTE_IN_WORKING_SET (&TempPte, WorkingSetIndex);
863 
864  //
865  // Flush the TB as the protection of a valid PTE is being changed.
866  //
867 
868  PreviousPte = *PointerPte;
869 
870  ASSERT (PreviousPte.u.Hard.Valid == 1);
871 
872  MI_WRITE_VALID_PTE_NEW_PROTECTION (PointerPte, TempPte);
873 
874  if (PteFlushList.Count < MM_MAXIMUM_FLUSH_COUNT) {
875  PteFlushList.FlushVa[PteFlushList.Count] = BaseAddress;
876  PteFlushList.Count += 1;
877  }
878 
879  ASSERT (PreviousPte.u.Hard.Valid == 1);
880 
881  //
882  // A page's protection is being changed, on certain
883  // hardware the dirty bit should be ORed into the
884  // modify bit in the PFN element.
885  //
886 
887  MI_CAPTURE_DIRTY_BIT_TO_PFN (&PreviousPte, Pfn1);
888  }
889 
890  PointerPte += 1;
891  BaseAddress = (PVOID)((PCHAR)BaseAddress + PAGE_SIZE);
892  }
893 
894  if (PteFlushList.Count != 0) {
895  MiFlushPteList (&PteFlushList);
896  }
897 
898  UNLOCK_PFN (OldIrql);
899 
900  UNLOCK_WS (CurrentThread, Process);
901 
902  Status = STATUS_SUCCESS;
903 
904 ErrorReturn:
905 
906  if (Attached == TRUE) {
907  KeUnstackDetachProcess (&ApcState);
908  Attached = FALSE;
909  }
910 
911  if (ProcessHandle != NtCurrentProcess()) {
912  ObDereferenceObject (Process);
913  }
914 
915  return Status;
916 }
