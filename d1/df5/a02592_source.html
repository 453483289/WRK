118

122

225 PsReferenceEffectiveToken(

229  OUT PSECURITY_IMPERSONATION_LEVEL ImpersonationLevel

232 /*++
233 
234 Routine Description:
235 
236  This function returns a pointer to the effective token of a thread. The
237  effective token of a thread is the thread's impersonation token if it has
238  one. Otherwise, it is the primary token of the thread's process.
239 
240  The reference count of the effective token is incremented to protect
241  the pointer returned.
242 
243  If the thread is impersonating a client, then the impersonation level
244  is also returned.
245 
246  Either PsDereferenceImpersonationToken() (for an impersonation token) or
247  PsDereferencePrimaryToken() (for a primary token) must be called to
248  decrement the token's reference count when the pointer is no longer
249  needed.
250 
251 
252 Arguments:
253 
254  Thread - Supplies the address of the thread whose effective token
255  is to be referenced.
256 
257  TokenType - Receives the type of the effective token. If the thread
258  is currently impersonating a client, then this will be
259  TokenImpersonation. Otherwise, it will be TokenPrimary.
260 
261  EffectiveOnly - If the token type is TokenImpersonation, then this
262  receives the value of the client thread's Thread->Client->EffectiveOnly field.
263  Otherwise, it is set to FALSE.
264 
265  ImpersonationLevel - The current value of the Thread->Client->ImpersonationLevel
266  field for an impersonation token and is not set for a primary token.
267 
268 Return Value:
269 
270  A pointer to the specified thread's effective token.
271 
272 --*/

288  //

370 NTSTATUS
371 PsOpenTokenOfThread(
372  IN HANDLE ThreadHandle,
373  IN BOOLEAN OpenAsSelf,
374  OUT PACCESS_TOKEN *Token,
375  OUT PBOOLEAN CopyOnOpen,
376  OUT PBOOLEAN EffectiveOnly,
377  OUT PSECURITY_IMPERSONATION_LEVEL ImpersonationLevel
378  )
379 
380 /*++
381 
382 Routine Description:
383 
384  This function does the thread specific processing of
385  an NtOpenThreadToken() service.
386 
387  The service validates that the handle has appropriate access
388  to reference the thread. If so, it goes on to increment
389  the reference count of the token object to prevent it from
390  going away while the rest of the NtOpenThreadToken() request
391  is processed.
392 
393  NOTE: If this call completes successfully, the caller is responsible
394  for decrementing the reference count of the target token.
395  This must be done using PsDereferenceImpersonationToken().
396 
397 Arguments:
398 
399  ThreadHandle - Supplies a handle to a thread object.
400 
401  OpenAsSelf - Is a boolean value indicating whether the access should
402  be made using the calling thread's current security context, which
403  may be that of a client (if impersonating), or using the caller's
404  process-level security context. A value of FALSE indicates the
405  caller's current context should be used un-modified. A value of
406  TRUE indicates the request should be fulfilled using the process
407  level security context.
408 
409  Token - If successful, receives a pointer to the thread's token
410  object.
411 
412  CopyOnOpen - The current value of the Thread->Client->CopyOnOpen field.
413 
414  EffectiveOnly - The current value of the Thread->Client->EffectiveOnly field.
415 
416  ImpersonationLevel - The current value of the Thread->Client->ImpersonationLevel
417  field.
418 
419 Return Value:
420 
421  STATUS_SUCCESS - Indicates the call completed successfully.
422 
423  STATUS_NO_TOKEN - Indicates the referenced thread is not currently
424  impersonating a client.
425 
426  STATUS_CANT_OPEN_ANONYMOUS - Indicates the client requested anonymous
427  impersonation level. An anonymous token can not be opened.
428 
429  status may also be any value returned by an attempt the reference
430  the thread object for THREAD_QUERY_INFORMATION access.
431 
432 --*/
433 
434 {
435 
436  NTSTATUS
437  Status;
438 
439  PETHREAD
440  Thread;
441 
442  KPROCESSOR_MODE
443  PreviousMode;
444 
445  PAGED_CODE();
446 
447  PreviousMode = KeGetPreviousMode();
448 
449 
450  UNREFERENCED_PARAMETER (OpenAsSelf);
451 
452  //
453  // Make sure the handle grants the appropriate access to the specified
454  // thread.
455  //
456 
457  Status = ObReferenceObjectByHandle (ThreadHandle,
458  THREAD_QUERY_INFORMATION,
459  PsThreadType,
460  PreviousMode,
461  &Thread,
462  NULL);
463 
464 
465 
466  if (!NT_SUCCESS (Status)) {
467  return Status;
468  }
469 
470  //
471  // Reference the impersonation token, if there is one
472  //
473 
474  (*Token) = PsReferenceImpersonationToken (Thread,
475  CopyOnOpen,
476  EffectiveOnly,
477  ImpersonationLevel);
478 
479 
480  //
481  // dereference the target thread.
482  //
483 
484  ObDereferenceObject (Thread);
485 
486  //
487  // Make sure there is a token
488  //
489 
490  if (*Token == NULL) {
491  return STATUS_NO_TOKEN;
492  }
493 
494  //
495  // Make sure the ImpersonationLevel is high enough to allow
496  // the token to be opened.
497  //
498 
499  if ((*ImpersonationLevel) <= SecurityAnonymous) {
500  PsDereferenceImpersonationToken (*Token);
501  (*Token) = NULL;
502  return STATUS_CANT_OPEN_ANONYMOUS;
503  }
504 
505 
506  return STATUS_SUCCESS;
507 
508 }

663 PsImpersonateClient(
664  __inout PETHREAD Thread,
665  __in PACCESS_TOKEN Token,
666  __in BOOLEAN CopyOnOpen,
667  __in BOOLEAN EffectiveOnly,
668  __in SECURITY_IMPERSONATION_LEVEL ImpersonationLevel
669  )
670 
671 /*++
672 
673 Routine Description:
674 
675  This routine sets up the specified thread so that it is impersonating
676  the specified client. This will result in the reference count of the
677  token representing the client being incremented to reflect the new
678  reference.
679 
680  If the thread is currently impersonating a client, that token will be
681  dereferenced.
682 
683 
684 
685 Arguments:
686 
687  Thread - points to the thread which is going to impersonate a client.
688 
689  Token - Points to the token to be assigned as the impersonation token.
690  This does NOT have to be a TokenImpersonation type token. This
691  allows direct reference of client process's primary tokens.
692 
693  CopyOnOpen - If TRUE, indicates the token is considered to be private
694  by the assigner and should be copied if opened. For example, a
695  session layer may be using a token to represent a client's context.
696  If the session is trying to synchronize the context of the client,
697  then user mode code should not be given direct access to the session
698  layer's token.
699 
700  Basically, session layers should always specify TRUE for this, while
701  tokens assigned by the server itself (handle based) should specify
702  FALSE.
703 
704 
705  EffectiveOnly - Is a boolean value to be assigned as the
706  Thread->ImpersonationInfo->EffectiveOnly field value for the
707  impersonation. A value of FALSE indicates the server is allowed
708  to enable currently disabled groups and privileges.
709 
710  ImpersonationLevel - Is the impersonation level that the server is allowed
711  to access the token with.
712 
713 
714 Return Value:
715 
716  STATUS_SUCCESS - Indicates the call completed successfully.
717 
718 
719 --*/

722

724  PACCESS_TOKEN OldToken;
725  PACCESS_TOKEN NewerToken=NULL;
726  PACCESS_TOKEN ProcessToken ;
727  NTSTATUS Status;
728  PPS_JOB_TOKEN_FILTER Filter;
729  PEPROCESS Process;
730  PETHREAD CurrentThread;
731  PEJOB Job;
732  PPS_IMPERSONATION_INFORMATION ImpersonationInfo;
733  BOOLEAN DontReference = FALSE ;
734  BOOLEAN NewTokenCreated = FALSE ;
735

737 
738  ASSERT (Thread->Tcb.Header.Type == ThreadObject);
739 
740  Process = THREAD_TO_PROCESS (Thread);
741 
742  if (!ARGUMENT_PRESENT(Token)) {
743 
744  OldToken = NULL;
745  if (PS_IS_THREAD_IMPERSONATING (Thread)) {
746 
747  //
748  // Lock the process security fields
749  //
750 
751  CurrentThread = PsGetCurrentThread ();
752 
753  PspLockThreadSecurityExclusive (Thread, CurrentThread);
754 
755  if (PS_IS_THREAD_IMPERSONATING (Thread)) {
756  //
757  // Grab impersonation info block.
758  //
759  ImpersonationInfo = Thread->ImpersonationInfo;
760 
761  //
762  // This is a request to revert to self.
763  // Clean up any client information.
764  //
765  OldToken = ImpersonationInfo->Token;
766  PS_CLEAR_BITS (&Thread->CrossThreadFlags,
767  PS_CROSS_THREAD_FLAGS_IMPERSONATING);
768  }
769  //
770  // Release the security fields
771  //
772  PspUnlockThreadSecurityExclusive (Thread, CurrentThread);
773 
774  PspWriteTebImpersonationInfo (Thread, CurrentThread);
775  }
776 
777  } else {
778

780  // Allocate and set up the Client block. We do this without holding the Process
781  // security lock so we reduce contention. Only one thread will manage to assign this.
782  // The client block once created never goes away until the last dereference of
783  // the process. We can touch this without locks
784  //
785  NewClient = Thread->ImpersonationInfo;
786  if (NewClient == NULL) {
787  NewClient = ExAllocatePoolWithTag (PagedPool,
788  sizeof (PS_IMPERSONATION_INFORMATION),
789  'mIsP'|PROTECTED_POOL);
790 
791  if (NewClient == NULL) {
792  return STATUS_NO_MEMORY;
793  }
794  FreeClient = InterlockedCompareExchangePointer (&Thread->ImpersonationInfo,
795  NewClient,
796  NULL);
797  //
798  // We got beaten by another thread. Free our context and use the new one
799  //
800  if (FreeClient != NULL) {
801  ExFreePoolWithTag (NewClient, 'mIsP'|PROTECTED_POOL);
802  NewClient = FreeClient;
803  }
804 
805  }
806

808  // Check process token for rules on impersonation
809  //
810

812 
813  if ( ProcessToken ) {
814 
815  Status = SeTokenCanImpersonate(
816  ProcessToken,
817  Token,
818  ImpersonationLevel );
819 
820  PsDereferencePrimaryTokenEx( Process, ProcessToken );
821 
822  if ( !NT_SUCCESS( Status ) ) {
823 
824  Status = SeCopyClientToken(
825  Token,
826  SecurityIdentification,
827  KernelMode,
828  &NewerToken );
829 
830  if ( !NT_SUCCESS(Status)) {
831 
832  return Status ;
833 
834  }
835

837  // We have a substitute token. Change Token to be this new
838  // one, but do not add a reference later. Right now, there
839  // is exactly one reference, so it will go away when the
840  // thread stops impersonating. Note that we still need to
841  // do the job filters below, hence the switch.
842  //
843 
844  Token = NewerToken ;
845  NewerToken = NULL ;
846  DontReference = TRUE ;
847  NewTokenCreated = TRUE ;
848  ImpersonationLevel = SecurityIdentification ;
849 
850  }
851 
852  }

856  // restrictions:
857  //

869 
870  }

873

888  // Filter installed. Need to create a restricted token
889  // dynamically.
890  //

892  Status = SeFastFilterToken (Token,
893  KernelMode,

896  Filter->CapturedGroups,
897  Filter->CapturedPrivilegeCount,
898  Filter->CapturedPrivileges,
899  Filter->CapturedSidCount,

901  Filter->CapturedSidsLength,
