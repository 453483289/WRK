56 MiMarkImageInSystem (
57  IN PCONTROL_AREA ControlArea
58  )
59 
60 /*++
61 
62 Routine Description:
63 
64  This routine marks the given image as mapped into system space.
65 
66 Arguments:
67 
68  ControlArea - Supplies the relevant control area.
69 
70 Return Value:
71 
72  TRUE on success, FALSE on failure.
73 
74 Environment:
75 
76  Kernel mode, APC_LEVEL and below, MmSystemLoadLock held.
77 
78 --*/
79 
80 {
81  LOGICAL Status;
82  KIRQL OldIrql;
83 
84  ASSERT (ControlArea->u.Flags.ImageMappedInSystemSpace == 0);
85 
86  Status = TRUE;
87 
88  //
89  // Lock synchronization is not needed for our callers as they always hold
90  // the system load mutant - but it is needed to modify this field in the
91  // control area as other threads may be modifying other bits in the flags.
92  //
93 
94  LOCK_PFN (OldIrql);
95 
96  //
97  // Before handling the relocations for this image, ensure it
98  // is not mapped in user space anywhere. Note we have 1 user
99  // reference at this point, so any beyond that are someone
100  // elses and force us to pagefile-back this image.
101  //
102 
103  if (ControlArea->NumberOfUserReferences <= 1) {
104 
105  ControlArea->u.Flags.ImageMappedInSystemSpace = 1;
106 
107  //
108  // This flag is set so when the image is removed from the loaded
109  // module list, the control area is destroyed. This is required
110  // because images mapped in session space inherit their PTE protections
111  // from the shared prototype PTEs.
112  //
113  // Consider the following scenario :
114  //
115  // If image A is loaded at its based (preferred) address, and then
116  // unloaded. Image B is not rebased properly and then loads at
117  // image A's preferred address. Image A then reloads.
118  //
119  // Now image A cannot use the original prototype PTEs which enforce
120  // readonly code, etc, because fixups will need to be done on it.
121  //
122  // Setting DeleteOnClose solves this problem by simply destroying
123  // the entire control area on last unload.
124  //
125 
126  ControlArea->u.Flags.DeleteOnClose = 1;
127  }
128  else {
129  Status = FALSE;
130  }
131 
132  UNLOCK_PFN (OldIrql);
133 
134  return Status;
135 }

138 MiShareSessionImage (
139  IN PVOID MappedBase,
140  IN PSECTION Section
141  )
142 
143 /*++
144 
145 Routine Description:
146 
147  This routine maps the given image into the current session space.
148  This allows the image to be executed backed by the image file in the
149  filesystem and allow code and read-only data to be shared.
150 
151 Arguments:
152 
153  MappedBase - Supplies the base address the image is to be mapped at.
154 
155  Section - Supplies a pointer to a section.
156 
157 Return Value:
158 
159  Returns STATUS_SUCCESS on success, various NTSTATUS codes on failure.
160 
161 Environment:
162 
163  Kernel mode, APC_LEVEL and below, MmSystemLoadLock held.
164 
165 --*/
166 
167 {
168  PSUBSECTION Subsection;
169  PCONTROL_AREA ControlArea;
170  PFN_NUMBER NumberOfPtes;
171  PMMPTE StartPte;
172 #if DBG
173  PMMPTE EndPte;
174 #endif
175  SIZE_T AllocationSize;
176  NTSTATUS Status;
177  SIZE_T CommittedPages;
178  LOGICAL Relocated;
179  PIMAGE_ENTRY_IN_SESSION DriverImage;
180 
181  PAGED_CODE();
182 
183  SYSLOAD_LOCK_OWNED_BY_ME ();
184 
185  ASSERT (MmIsAddressValid (MmSessionSpace) == TRUE);
186 
187  if (MappedBase != Section->Segment->BasedAddress) {
188  Relocated = TRUE;
189  }
190  else {
191  Relocated = FALSE;
192  }
193 
194  ASSERT (BYTE_OFFSET (MappedBase) == 0);
195 
196  //
197  // Check to see if a purge operation is in progress and if so, wait
198  // for the purge to complete. In addition, up the count of mapped
199  // views for this control area.
200  //
201 
202  ControlArea = Section->Segment->ControlArea;
203 
204  if ((ControlArea->u.Flags.GlobalOnlyPerSession == 0) &&
205  (ControlArea->u.Flags.Rom == 0)) {
206  Subsection = (PSUBSECTION)(ControlArea + 1);
207  }
208  else {
209  Subsection = (PSUBSECTION)((PLARGE_CONTROL_AREA)ControlArea + 1);
210  }
211 
212  Status = MiCheckPurgeAndUpMapCount (ControlArea, FALSE);
213 
214  if (!NT_SUCCESS (Status)) {
215  return Status;
216  }
217 
218  NumberOfPtes = Section->Segment->TotalNumberOfPtes;
219  AllocationSize = NumberOfPtes << PAGE_SHIFT;
220 
221  //
222  // Calculate the PTE ranges and amount.
223  //
224 
225  StartPte = MiGetPteAddress (MappedBase);
226 
227  //
228  // The image commitment will be the same as the number of PTEs if the
229  // image was not linked with native page alignment for the subsections.
230  //
231  // If it is linked native, then the commit will just be the number of
232  // writable pages. Note for this case, if we need to relocate it, then
233  // we need to charge for the full number of PTEs and bump the commit
234  // charge in the segment so that the return on unload is correct also.
235  //
236 
237  ASSERT (Section->Segment->u1.ImageCommitment != 0);
238  ASSERT (Section->Segment->u1.ImageCommitment <= NumberOfPtes);
239 
240  if (Relocated == TRUE) {
241  CommittedPages = NumberOfPtes;
242  }
243  else {
244  CommittedPages = Section->Segment->u1.ImageCommitment;
245  }
246 
247  if (MiChargeCommitment (CommittedPages, NULL) == FALSE) {
248  MM_BUMP_SESSION_FAILURES (MM_SESSION_FAILURE_NO_COMMIT);
249 
250  //
251  // Don't bother releasing the page tables or their commit here, another
252  // load will happen shortly or the whole session will go away. On
253  // session exit everything will be released automatically.
254  //
255 
256  MiDereferenceControlArea (ControlArea);
257  return STATUS_NO_MEMORY;
258  }
259 
260  InterlockedExchangeAddSizeT (&MmSessionSpace->CommittedPages,
261  CommittedPages);
262 
263  //
264  // Make sure we have page tables for the PTE
265  // entries we must fill in the session space structure.
266  //
267 
268  Status = MiSessionCommitPageTables (MappedBase,
269  (PVOID)((PCHAR)MappedBase + AllocationSize));
270 
271  if (!NT_SUCCESS(Status)) {
272 
273  InterlockedExchangeAddSizeT (&MmSessionSpace->CommittedPages,
274  0 - CommittedPages);
275 
276  MiDereferenceControlArea (ControlArea);
277  MiReturnCommitment (CommittedPages);
278 
279  return STATUS_NO_MEMORY;
280  }
281 
282 #if DBG
283  EndPte = StartPte + NumberOfPtes;
284  while (StartPte < EndPte) {
285  ASSERT (StartPte->u.Long == 0);
286  StartPte += 1;
287  }
288 
289  StartPte = MiGetPteAddress (MappedBase);
290 #endif
291 
292  //
293  // If the image was linked with subsection alignment of >= PAGE_SIZE,
294  // then all of the prototype PTEs were initialized to proper protections by
295  // the initial section creation. The protections in these PTEs are used
296  // to fill the actual PTEs as each address is faulted on.
297  //
298  // If the image has less than PAGE_SIZE section alignment, then
299  // section creation uses a single subsection to map the entire file and
300  // sets all the prototype PTEs to copy on write. For this case, the
301  // appropriate permissions are set by the MiWriteProtectSystemImage below.
302  //
303 
304  //
305  // Initialize the PTEs to point at the prototype PTEs.
306  //
307 
308  Status = MiAddMappedPtes (StartPte, NumberOfPtes, ControlArea);
309 
310  if (!NT_SUCCESS (Status)) {
311 
312  //
313  // Regardless of whether the PTEs were mapped, leave the control area
314  // marked as mapped in system space so user applications cannot map the
315  // file as an image as clearly the intent is to run it as a driver.
316  //
317 
318  InterlockedExchangeAddSizeT (&MmSessionSpace->CommittedPages,
319  0 - CommittedPages);
320 
321  MiDereferenceControlArea (ControlArea);
322  MiReturnCommitment (CommittedPages);
323  return Status;
324  }
325 
326  MM_TRACK_COMMIT (MM_DBG_COMMIT_SESSION_SHARED_IMAGE, CommittedPages);
327 
328  MM_BUMP_SESS_COUNTER (MM_DBG_SESSION_SYSMAPPED_PAGES_COMMITTED, (ULONG)CommittedPages);
329 
330  MM_BUMP_SESS_COUNTER (MM_DBG_SESSION_SYSMAPPED_PAGES_ALLOC, (ULONG)NumberOfPtes);
331 
332  //
333  // No session space image faults may be taken until these fields of the
334  // image entry are initialized.
335  //
336 
337  DriverImage = MiSessionLookupImage (MappedBase);
338  ASSERT (DriverImage);
339 
340  DriverImage->LastAddress = (PVOID)((PCHAR)MappedBase + AllocationSize - 1);
341  DriverImage->PrototypePtes = Subsection->SubsectionBase;
342 
343  //
344  // The loaded module list section reference protects the image from
345  // being purged while it's in use.
346  //
347 
348  MiDereferenceControlArea (ControlArea);
349 
350  return STATUS_SUCCESS;
351 }

359 /*++
360 
361 Routine Description:
362 
363  This routine allocates an image entry for the specified address in the
364  current session space.
365 
366 Arguments:
367 
368  BaseAddress - Supplies the base address for the executable image.
369 
370 Return Value:
371 
372  STATUS_SUCCESS or various NTSTATUS error codes on failure.
373 
374 Environment:
375 
376  Kernel mode, APC_LEVEL and below, MmSystemLoadLock held.
377 
378  Note both the system load resource and the session working set
379  mutex must be held to modify the list of images in this session.
380  Either may be held to safely walk the list.
381 
382 --*/
383 
384 {
385  PLIST_ENTRY NextEntry;
386  PIMAGE_ENTRY_IN_SESSION Image;
387  PIMAGE_ENTRY_IN_SESSION NewImage;
388  PETHREAD Thread;
389  PMMSUPPORT Ws;
390 
391  PAGED_CODE();
392 
393  SYSLOAD_LOCK_OWNED_BY_ME ();
394 
395  Thread = PsGetCurrentThread ();
396 
397  //
398  // Create and initialize a new image entry prior to acquiring the session
399  // space ws mutex. This is to reduce the amount of time the mutex is held.
400  // If an existing entry is found this allocation is just discarded.
401  //
402 
403  NewImage = ExAllocatePoolWithTag (NonPagedPool,
404  sizeof(IMAGE_ENTRY_IN_SESSION),
405  'iHmM');
406 
407  if (NewImage == NULL) {
408  MM_BUMP_SESSION_FAILURES (MM_SESSION_FAILURE_NO_NONPAGED_POOL);
409  return STATUS_NO_MEMORY;
410  }
411 
412  RtlZeroMemory (NewImage, sizeof(IMAGE_ENTRY_IN_SESSION));
413 
414  NewImage->Address = BaseAddress;
415  NewImage->ImageCountInThisSession = 1;
416 
417  //
418  // Check to see if the address is already loaded.
419  //
420 
421  Ws = &MmSessionSpace->GlobalVirtualAddress->Vm;
422 
423  LOCK_WORKING_SET (Thread, Ws);
424 
425  NextEntry = MmSessionSpace->ImageList.Flink;
426 
427  while (NextEntry != &MmSessionSpace->ImageList) {
428  Image = CONTAINING_RECORD (NextEntry, IMAGE_ENTRY_IN_SESSION, Link);
429 
430  if (Image->Address == BaseAddress) {
431  Image->ImageCountInThisSession += 1;
432  UNLOCK_WORKING_SET (Thread, Ws);
433  ExFreePool (NewImage);
434  return STATUS_ALREADY_COMMITTED;
435  }
436  NextEntry = NextEntry->Flink;
437  }
438 
439  //
440  // Insert the image entry into the session space structure.
441  //
442 
443  InsertTailList (&MmSessionSpace->ImageList, &NewImage->Link);
444 
445  UNLOCK_WORKING_SET (Thread, Ws);
446 
447  return STATUS_SUCCESS;
448 }

519 
520  NextEntry = NextEntry->Flink;
521  }
522 
523  UNLOCK_WORKING_SET (Thread, Ws);
524 
525  return STATUS_NOT_FOUND;

582 VOID
583 MiSessionUnloadAllImages (
584  VOID
585  )
586 
587 /*++
588 
589 Routine Description:
590 
591  This routine dereferences each image that has been loaded in the
592  current session space.
593 
594  As each image is dereferenced, checks are made:
595 
596  If this session's reference count to the image reaches zero, the VA
597  range in this session is deleted. If the reference count to the image
598  in the SESSIONWIDE list drops to zero, then the SESSIONWIDE's VA
599  reservation is removed and the address space is made available to any
600  new image.
601 
602  If this is the last systemwide reference to the driver then the driver
603  is deleted from memory.
604 
605 Arguments:
606 
607  None.
608 
609 Return Value:
610 
611  None.
612 
613 Environment:
614 
615  Kernel mode. This is called in one of two contexts:
616  1. the last thread in the last process of the current session space.
617  2. or by any thread in the SMSS process.
618 
619  Note both the system load resource and the session working set
620  mutex must be held to modify the list of images in this session.
621  Either may be held to safely walk the list.
622 
623 --*/
624 
625 {
626  NTSTATUS Status;
627  PLIST_ENTRY NextEntry;
628  PIMAGE_ENTRY_IN_SESSION Module;
629  PKLDR_DATA_TABLE_ENTRY ImageHandle;
630 
631  ASSERT (MmSessionSpace->ReferenceCount == 0);
632 
633  //
634  // The session's working set lock does not need to be acquired here since
635  // no thread can be faulting on these addresses.
636  //
637 
638  NextEntry = MmSessionSpace->ImageList.Flink;
639 
640  while (NextEntry != &MmSessionSpace->ImageList) {
641 
642  Module = CONTAINING_RECORD(NextEntry, IMAGE_ENTRY_IN_SESSION, Link);
643 
644  //
645  // Lookup the image entry in the system PsLoadedModuleList,
646  // unload the image and delete it.
647  //
648 
649  ImageHandle = MiLookupDataTableEntry (Module->Address, FALSE);
650 
651  ASSERT (ImageHandle);
652 
653  Status = MmUnloadSystemImage (ImageHandle);
654 
655  //
656  // Restart the search at the beginning since the entry has been deleted.
657  //
658 
659  ASSERT (MmSessionSpace->ReferenceCount == 0);
660 
661  NextEntry = MmSessionSpace->ImageList.Flink;
662  }
663 }

719 NTSTATUS
720 MiSessionWideReserveImageAddress (
721  IN PSECTION Section,
722  OUT PVOID *AssignedAddress,
723  OUT PSECTION *NewSectionPointer
724  )
725 
726 /*++
727 
728 Routine Description:
729 
730  This routine allocates a range of virtual address space within
731  session space. This address range is unique system-wide and in this
732  manner, code and pristine data of session drivers can be shared across
733  multiple sessions.
734 
735  This routine does not actually commit pages, but reserves the virtual
736  address region for the named image. An entry is created here and attached
737  to the current session space to track the loaded image. Thus if all
738  the references to a given range go away, the range can then be reused.
739 
740 Arguments:
741 
742  Section - Supplies the section (and thus, the preferred address that the
743  driver has been linked (rebased) at. If this address is
744  available, the driver will require no relocation. The section
745  is also used to derive the number of bytes to reserve.
746 
747  AssignedAddress - Supplies a pointer to a variable that receives the
748  allocated address if the routine succeeds.
749 
750 Return Value:
751 
752  Returns STATUS_SUCCESS on success, various NTSTATUS codes on failure.
753 
754 Environment:
755 
756  Kernel mode, APC_LEVEL and below, MmSystemLoadLock held.
757 
758 --*/
759 
760 {
761  ULONG StartPosition;
762  ULONG NumberOfPtes;
763  NTSTATUS Status;
764  PWCHAR pName;
765  PVOID NewAddress;
766  ULONG_PTR SessionSpaceEnd;
767  PVOID PreferredAddress;
768  PCONTROL_AREA ControlArea;
769  PIMAGE_ENTRY_IN_SESSION Image;
770  PSESSION_GLOBAL_SUBSECTION_INFO GlobalSubs;
771 
772  PAGED_CODE();
773 
774  SYSLOAD_LOCK_OWNED_BY_ME ();
775 
776  ASSERT (PsGetCurrentProcess()->Flags & PS_PROCESS_FLAGS_IN_SESSION);
777  ASSERT (MmIsAddressValid (MmSessionSpace) == TRUE);
778  ASSERT (Section->u.Flags.Image == 1);
779 
780  *NewSectionPointer = NULL;
781 
782  ControlArea = Section->Segment->ControlArea;
783 
784  if (ControlArea->u.Flags.ImageMappedInSystemSpace == 1) {
785 
786  //
787  // We are going to add a new entry to the loaded module list. We
788  // have a section in hand. The case which must be handled carefully is:
789  //
790  // When a session image unloads, its entry is removed from the
791  // loaded module list, but the section object itself may continue
792  // to live on due to other references on the object. For session
793  // images, the relocations and import snaps, verifier updates, etc,
794  // are done directly to the prototype PTEs (the modified ones become
795  // backed by the pagefile) at whatever address the image is loaded at.
796  //
797  // Thus, if the image's load address the second time around is different
798  // from the first time around, this presents problems. Likewise, any
799  // image it imports from is an issue, since if their load addresses
800  // change, the IAT snaps need updating. This only gets more complicated
801  // with recursive imports, imports where only some of the images have
802  // lingering object reference counts, etc.
803  //
804  // There is a lingering object reference to this image since it
805  // was last unloaded. It's ok just to fail this since it
806  // can't be a user-generated reference and any kernel/driver
807  // SEC_IMAGE reference would be extremely unusual (and short lived).
808  //
809 
810  MM_BUMP_SESSION_FAILURES (MM_SESSION_FAILURE_IMAGE_ZOMBIE);
811 
812  KeDelayExecutionThread (KernelMode, FALSE, (PLARGE_INTEGER)&MmShortTime);
813  return STATUS_CONFLICTING_ADDRESSES;
814  }
815 
816  pName = NULL;
817 
818  StartPosition = NO_BITS_FOUND;
819  PreferredAddress = Section->Segment->BasedAddress;
820  NumberOfPtes = Section->Segment->TotalNumberOfPtes;
821 
822  SessionSpaceEnd = MiSessionImageEnd;
823 
824  //
825  // Try to put the module into its requested address so it can be shared.
826  //
827  // If the requested address is not properly aligned or not in the session
828  // space region, pick an address for it. This image will not be shared.
829  //
830 
831  if ((BYTE_OFFSET (PreferredAddress) == 0) &&
832  (PreferredAddress >= (PVOID) MiSessionImageStart) &&
833  (PreferredAddress < (PVOID) MiSessionImageEnd)) {
834 
835  StartPosition = (ULONG) ((ULONG_PTR) PreferredAddress - MiSessionImageStart) >> PAGE_SHIFT;
836 
837  if (RtlAreBitsClear (&MiSessionWideVaBitMap,
838  StartPosition,
