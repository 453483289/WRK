9 Module Name:
10 
11  buildmdl.c
12 
13 Abstract:
14 
15  This module contains the Mm support routines for the cache manager to
16  prefetching groups of pages from secondary storage using logical file
17  offsets instead of virtual addresses. This saves the cache manager from
18  having to map pages unnecessarily.
19 
20  The caller builds a list of various file objects and logical block offsets,
21  passing them to MmPrefetchPagesIntoLockedMdl. The code here then examines
22  the internal pages, reading in those that are not already valid or in
23  transition. These pages are read with a single read, using a dummy page
24  to bridge gaps of pages that were valid or transition prior to the I/O
25  being issued.
26 
27  Upon conclusion of the I/O, control is returned to the calling thread.
28  All pages are referenced counted as though they were probed and locked,
29  regardless of whether they are currently valid or transition.
30 
31 --*/

101  IN SIZE_T Length,
102  OUT PMDL *MdlOut

104 
105 /*++
106 
107 Routine Description:
108 
109  This routine fills an MDL with pages described by the file object's
110  offset and length.
111 
112  This routine is for cache manager usage only.
113 
114 Arguments:
115 
116  FileObject - Supplies a pointer to the file object for a file which was
117  opened with NO_INTERMEDIATE_BUFFERING clear, i.e., for
118  which CcInitializeCacheMap was called by the file system.
119 
120  FileOffset - Supplies the byte offset in the file for the desired data.
121 
122  Length - Supplies the length of the desired data in bytes.
123 
124  MdlOut - On output it returns a pointer to an Mdl describing
125  the desired data.
126 
127 Return Value:
128 
129  NTSTATUS.
130 
131 Environment:
132 
133  Kernel mode. PASSIVE_LEVEL.
134 
135 --*/
136 
137 {
138  MI_READ_INFO MiReadInfo;
139  NTSTATUS status;
140  LOGICAL ApcNeeded;
141  PETHREAD CurrentThread;
142 
143  ASSERT (KeGetCurrentIrql() == PASSIVE_LEVEL);
144 
145  RtlZeroMemory (&MiReadInfo, sizeof(MiReadInfo));
146 
147  MiReadInfo.FileObject = FileObject;
148  MiReadInfo.FileOffset = *FileOffset;
149  MiReadInfo.LengthInBytes = Length;
150 
151  //
152  // Prepare for the impending read : allocate MDLs, inpage blocks,
153  // reference count subsections, etc.
154  //
155 
156  status = MiCcPrepareReadInfo (&MiReadInfo);
157 
158  if (!NT_SUCCESS (status)) {
159  MiCcReleasePrefetchResources (&MiReadInfo, status);
160  return status;
161  }
162 
163  ASSERT (MiReadInfo.InPageSupport != NULL);
164 
165  //
166  // APCs must be disabled once we put a page in transition. Otherwise
167  // a thread suspend will stop us from issuing the I/O - this will hang
168  // any other threads that need the same page.
169  //
170 
171  CurrentThread = PsGetCurrentThread();
172  ApcNeeded = FALSE;
173 
174  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
175 
176  //
177  // The nested fault count protects this thread from deadlocks where a
178  // special kernel APC fires and references the same user page(s) we are
179  // putting in transition.
180  //
181 
182  KeEnterGuardedRegionThread (&CurrentThread->Tcb);
183  ASSERT (CurrentThread->ActiveFaultCount == 0);
184  CurrentThread->ActiveFaultCount += 1;
185  KeLeaveGuardedRegionThread (&CurrentThread->Tcb);
186 
187  //
188  // Allocate physical memory, lock down all the pages and issue any
189  // I/O that may be needed. When MiCcPutPagesInTransition returns
190  // STATUS_SUCCESS or STATUS_ISSUE_PAGING_IO, it guarantees that the
191  // ApiMdl contains reference-counted (locked-down) pages.
192  //
193 
194  status = MiCcPutPagesInTransition (&MiReadInfo);
195 
196  if (NT_SUCCESS (status)) {
197 
198  //
199  // No I/O was issued because all the pages were already resident and
200  // have now been locked down.
201  //
202 
203  ASSERT (MiReadInfo.ApiMdl != NULL);
204  }
205  else if (status == STATUS_ISSUE_PAGING_IO) {
206 
207  //
208  // Wait for the I/O to complete. Note APCs must remain disabled.
209  //
210 
211  ASSERT (MiReadInfo.InPageSupport != NULL);
212 
213  status = MiCcCompletePrefetchIos (&MiReadInfo);
214  }
215  else {
216 
217  //
218  // Some error occurred (like insufficient memory, etc) so fail
219  // the request by falling through.
220  //
221  }
222 
223  //
224  // Release acquired resources like pool, subsections, etc.
225  //
226 
227  MiCcReleasePrefetchResources (&MiReadInfo, status);
228 
229  //
230  // Only now that the I/O have been completed (not just issued) can
231  // APCs be re-enabled. This prevents a user-issued suspend APC from
232  // keeping a shared page in transition forever.
233  //
234 
235  KeEnterGuardedRegionThread (&CurrentThread->Tcb);
236 
237  ASSERT (CurrentThread->ActiveFaultCount == 1);
238 
239  CurrentThread->ActiveFaultCount -= 1;
240 
241  if (CurrentThread->ApcNeeded == 1) {
242  ApcNeeded = TRUE;
243  CurrentThread->ApcNeeded = 0;
244  }
245 
246  KeLeaveGuardedRegionThread (&CurrentThread->Tcb);
247 
248  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
249 
250  ASSERT (KeGetCurrentIrql() == PASSIVE_LEVEL);
251  ASSERT (CurrentThread->ActiveFaultCount == 0);
252  ASSERT (CurrentThread->ApcNeeded == 0);
253 
254  if (ApcNeeded == TRUE) {
255  IoRetryIrpCompletions ();
256  }
257 
258  *MdlOut = MiReadInfo.ApiMdl;
259 
260  return status;
261 }

269 /*++
270 
271 Routine Description:
272 
273  This routine releases all resources consumed to handle a system cache
274  logical offset based prefetch.
275 
276 Environment:
277 
278  Kernel mode, PASSIVE_LEVEL.
279 
280 --*/
281 
282 {

285

288  //

333 MiCcPrepareReadInfo (
334  IN PMI_READ_INFO MiReadInfo
335  )
336 
337 /*++
338 
339 Routine Description:
340 
341  This routine constructs MDLs that describe the pages in the argument
342  read-list. The caller will then issue the I/O on return.
343 
344 Arguments:
345 
346  MiReadInfo - Supplies a pointer to the read-list.
347 
348 Return Value:
349 
350  Various NTSTATUS codes.
351 
352 Environment:
353 
354  Kernel mode, PASSIVE_LEVEL.
355 
356 --*/
357 
358 {
359  UINT64 PteOffset;
360  NTSTATUS Status;
361  PMMPTE ProtoPte;
362  PMMPTE LastProto;
363  PMMPTE *ProtoPteArray;
364  PCONTROL_AREA ControlArea;
365  PSUBSECTION Subsection;
366  PMMINPAGE_SUPPORT InPageSupport;
367  PMDL Mdl;
368  PMDL IoMdl;
369  PMDL ApiMdl;
370  ULONG i;
371  PFN_NUMBER NumberOfPages;
372 
373  ASSERT (KeGetCurrentIrql() == PASSIVE_LEVEL);
374 
375  NumberOfPages = ADDRESS_AND_SIZE_TO_SPAN_PAGES (MiReadInfo->FileOffset.LowPart, MiReadInfo->LengthInBytes);
376 
377  //
378  // Translate the section object into the relevant control area.
379  //
380 
381  ControlArea = (PCONTROL_AREA)MiReadInfo->FileObject->SectionObjectPointer->DataSectionObject;
382 
383  //
384  // If the section is backed by a ROM, then there's no need to prefetch
385  // anything as it would waste RAM.
386  //
387 
388  if (ControlArea->u.Flags.Rom == 1) {
389  return STATUS_NOT_SUPPORTED;
390  }
391 
392  //
393  // Initialize the internal Mi readlist.
394  //
395 
396  MiReadInfo->ControlArea = ControlArea;
397 
398  //
399  // Allocate and initialize an inpage support block for this run.
400  //
401 
402  InPageSupport = MiGetInPageSupportBlock (MM_NOIRQL, &Status);
403 
404  if (InPageSupport == NULL) {
405  ASSERT (!NT_SUCCESS (Status));
406  return Status;
407  }
408 
409  MiReadInfo->InPageSupport = InPageSupport;
410 
411  //
412  // Allocate and initialize an MDL to return to our caller. The actual
413  // frame numbers are filled in when all the pages are reference counted.
414  //
415 
416  ApiMdl = MmCreateMdl (NULL, NULL, NumberOfPages << PAGE_SHIFT);
417 
418  if (ApiMdl == NULL) {
419  return STATUS_INSUFFICIENT_RESOURCES;
420  }
421 
422  ApiMdl->MdlFlags |= MDL_PAGES_LOCKED;
423 
424  MiReadInfo->ApiMdl = ApiMdl;
425 
426  //
427  // Allocate and initialize an MDL to use for the actual transfer (if any).
428  //
429 
430  IoMdl = MmCreateMdl (NULL, NULL, NumberOfPages << PAGE_SHIFT);
431 
432  if (IoMdl == NULL) {
433  return STATUS_INSUFFICIENT_RESOURCES;
434  }
435 
436  MiReadInfo->IoMdl = IoMdl;
437  Mdl = IoMdl;
438 
439  //
440  // Make sure the section is really prefetchable - physical and
441  // pagefile-backed sections are not.
442  //
443 
444  if ((ControlArea->u.Flags.PhysicalMemory) ||
445  (ControlArea->u.Flags.Image == 1) ||
446  (ControlArea->FilePointer == NULL)) {
447 
448  return STATUS_INVALID_PARAMETER_1;
449  }
450 
451  //
452  // Start the read at the proper file offset.
453  //
454 
455  InPageSupport->ReadOffset = MiReadInfo->FileOffset;
456  ASSERT (BYTE_OFFSET (InPageSupport->ReadOffset.LowPart) == 0);
457  InPageSupport->FilePointer = MiReadInfo->FileObject;
458 
459  //
460  // Stash a pointer to the start of the prototype PTE array (the values
461  // in the array are not contiguous as they may cross subsections)
462  // in the inpage block so we can walk it quickly later when the pages
463  // are put into transition.
464  //
465 
466  ProtoPteArray = (PMMPTE *)(Mdl + 1);
467 
468  InPageSupport->BasePte = (PMMPTE) ProtoPteArray;
469 
470  //
471  // Data (but not image) reads use the whole page and the filesystems
472  // zero fill any remainder beyond valid data length so we don't
473  // bother to handle this here. It is important to specify the
474  // entire page where possible so the filesystem won't post this
475  // which will hurt perf. LWFIX: must use CcZero to make this true.
476  //
477 
478  ASSERT (((ULONG_PTR)Mdl & (sizeof(QUAD) - 1)) == 0);
479  InPageSupport->u1.e1.PrefetchMdlHighBits = ((ULONG_PTR)Mdl >> 3);
480 
481  //
482  // Initialize the prototype PTE pointers.
483  //
484 
485  ASSERT (ControlArea->u.Flags.GlobalOnlyPerSession == 0);
486 
487  if (ControlArea->u.Flags.Rom == 0) {
488  Subsection = (PSUBSECTION)(ControlArea + 1);
489  }
490  else {
491  Subsection = (PSUBSECTION)((PLARGE_CONTROL_AREA)ControlArea + 1);
492  }
493 
494 #if DBG
495  if (MiCcDebug & MI_CC_FORCE_PREFETCH) {
496  MiRemoveUserPages ();
497  }
498 #endif
499 
500  //
501  // Calculate the first prototype PTE address.
502  //
503 
504  PteOffset = (UINT64)(MiReadInfo->FileOffset.QuadPart >> PAGE_SHIFT);
505 
506  //
507  // Make sure the PTEs are not in the extended part of the segment.
508  //
509 
510  while (TRUE) {
511 
512  //
513  // A memory barrier is needed to read the subsection chains
514  // in order to ensure the writes to the actual individual
515  // subsection data structure fields are visible in correct
516  // order. This avoids the need to acquire any stronger
517  // synchronization (ie: PFN lock), thus yielding better
518  // performance and pageability.
519  //
520 
521  KeMemoryBarrier ();
522 
523  if (PteOffset < (UINT64) Subsection->PtesInSubsection) {
524  break;
525  }
526 
527  PteOffset -= Subsection->PtesInSubsection;
528  Subsection = Subsection->NextSubsection;
529  }
530 
531  Status = MiAddViewsForSectionWithPfn ((PMSUBSECTION) Subsection,
532  Subsection->PtesInSubsection);
533 
534  if (!NT_SUCCESS (Status)) {
535  return Status;
536  }
537 
538  MiReadInfo->FirstReferencedSubsection = Subsection;
539  MiReadInfo->LastReferencedSubsection = Subsection;
540 
541  ProtoPte = &Subsection->SubsectionBase[PteOffset];
542  LastProto = &Subsection->SubsectionBase[Subsection->PtesInSubsection];
543 
544  for (i = 0; i < NumberOfPages; i += 1) {
545 
546  //
547  // Calculate which PTE maps the given logical block offset.
548  //
549  // Always look forwards (as an optimization) in the subsection chain.
550  //
551  // A quick check is made first to avoid recalculations and loops where
552  // possible.
553  //
554 
555  if (ProtoPte >= LastProto) {
556 
557  //
558  // Handle extended subsections. Increment the view count for
559  // every subsection spanned by this request, creating prototype
560  // PTEs if needed.
561  //
562 
563  ASSERT (i != 0);
564 
565  Subsection = Subsection->NextSubsection;
566 
567  Status = MiAddViewsForSectionWithPfn ((PMSUBSECTION) Subsection,
568  Subsection->PtesInSubsection);
569 
570  if (!NT_SUCCESS (Status)) {
571  return Status;
572  }
573 
574  MiReadInfo->LastReferencedSubsection = Subsection;
575 
576  ProtoPte = Subsection->SubsectionBase;
577 
578  LastProto = &Subsection->SubsectionBase[Subsection->PtesInSubsection];
579  }
580 
581  *ProtoPteArray = ProtoPte;
582  ProtoPteArray += 1;
583 
584  ProtoPte += 1;
585  }
586 
587  return STATUS_SUCCESS;
588 }

591 MiCcPutPagesInTransition (
592  IN PMI_READ_INFO MiReadInfo
593  )
594 
595 /*++
596 
597 Routine Description:
598 
599  This routine allocates physical memory for the specified read-list and
600  puts all the pages in transition (so collided faults from other threads
601  for these same pages remain coherent). I/O for any pages not already
602  resident are issued here. The caller must wait for their completion.
603 
604 Arguments:
605 
606  MiReadInfo - Supplies a pointer to the read-list.
607 
608 Return Value:
609 
610  STATUS_SUCCESS - all the pages were already resident, reference counts
611  have been applied and no I/O needs to be waited for.
612 
613  STATUS_ISSUE_PAGING_IO - the I/O has been issued and the caller must wait.
614 
615  Various other failure status values indicate the operation failed.
616 
617 Environment:
618 
619  Kernel mode. PASSIVE_LEVEL.
620 
621 --*/
622 
623 {
624  NTSTATUS status;
625  PMMPTE LocalPrototypePte;
626  PVOID StartingVa;
627  PFN_NUMBER MdlPages;
628  KIRQL OldIrql;
629  MMPTE PteContents;
630  PFN_NUMBER PageFrameIndex;
631  PFN_NUMBER ResidentAvailableCharge;
632  PPFN_NUMBER IoPage;
633  PPFN_NUMBER ApiPage;
634  PPFN_NUMBER Page;
635  PPFN_NUMBER DestinationPage;
636  ULONG PageColor;
637  PMMPTE PointerPte;
638  PMMPTE *ProtoPteArray;
639  PMMPTE *EndProtoPteArray;
640  PFN_NUMBER DummyPage;
641  PMDL Mdl;
642  PMDL FreeMdl;
643  PMMPFN PfnProto;
644  PMMPFN Pfn1;
645  PMMPFN DummyPfn1;
646  ULONG i;
647  PFN_NUMBER DummyTrim;
648  ULONG NumberOfPagesNeedingIo;
649  MMPTE TempPte;
650  PMMPTE PointerPde;
651  PEPROCESS CurrentProcess;
652  PMMINPAGE_SUPPORT InPageSupport;
653  PKPRCB Prcb;
654 
655  ASSERT (KeGetCurrentIrql() == PASSIVE_LEVEL);
656 
657  MiReadInfo->DummyPagePfn = NULL;
658 
659  FreeMdl = NULL;
660  CurrentProcess = PsGetCurrentProcess();
661 
662  PfnProto = NULL;
663  PointerPde = NULL;
664 
665  InPageSupport = MiReadInfo->InPageSupport;
666 
667  Mdl = MI_EXTRACT_PREFETCH_MDL (InPageSupport);
668  ASSERT (Mdl == MiReadInfo->IoMdl);
669 
670  IoPage = (PPFN_NUMBER)(Mdl + 1);
671  ApiPage = (PPFN_NUMBER)(MiReadInfo->ApiMdl + 1);
672 
673  StartingVa = (PVOID)((PCHAR)Mdl->StartVa + Mdl->ByteOffset);
674 
675  MdlPages = ADDRESS_AND_SIZE_TO_SPAN_PAGES (StartingVa,
676  Mdl->ByteCount);
677 
678  if (MdlPages + 1 > MAXUSHORT) {
679 
680  //
681  // The PFN ReferenceCount for the dummy page could wrap, refuse the
682  // request.
683  //
684 
685  return STATUS_INSUFFICIENT_RESOURCES;
686  }
687 
688  NumberOfPagesNeedingIo = 0;
689 
690  ProtoPteArray = (PMMPTE *)InPageSupport->BasePte;
691  EndProtoPteArray = ProtoPteArray + MdlPages;
692 
693  ASSERT (*ProtoPteArray != NULL);
694 
695  LOCK_PFN (OldIrql);
696 
697  //
698  // Ensure sufficient pages exist for the transfer plus the dummy page.
699  //
700 
701  if (((SPFN_NUMBER)MdlPages > (SPFN_NUMBER)(MmAvailablePages - MM_HIGH_LIMIT)) ||
702  (MI_NONPAGEABLE_MEMORY_AVAILABLE() <= (SPFN_NUMBER)MdlPages)) {
703 
704  UNLOCK_PFN (OldIrql);
705 
706  return STATUS_INSUFFICIENT_RESOURCES;
707  }
708 
709  //
710  // Charge resident available immediately as the PFN lock may get released
711  // and reacquired below before all the pages have been locked down.
712  // Note the dummy page is immediately charged separately.
713  //
714 
715  MI_DECREMENT_RESIDENT_AVAILABLE (MdlPages, MM_RESAVAIL_ALLOCATE_BUILDMDL);
716 
717  ResidentAvailableCharge = MdlPages;
718 
719  //
720  // Allocate a dummy page to map discarded pages that aren't skipped.
721  //
722 
723  DummyPage = MiRemoveAnyPage (0);
724  Pfn1 = MI_PFN_ELEMENT (DummyPage);
725 
726  ASSERT (Pfn1->u2.ShareCount == 0);
727  ASSERT (Pfn1->u3.e2.ReferenceCount == 0);
728 
729  MiInitializePfnForOtherProcess (DummyPage, MI_PF_DUMMY_PAGE_PTE, 0);
730 
731  //
732  // Give the page a containing frame so MiIdentifyPfn won't crash.
733  //
734 
735  Pfn1->u4.PteFrame = PsInitialSystemProcess->Pcb.DirectoryTableBase[0] >> PAGE_SHIFT;
736 
737  //
738  // Always bias the reference count by 1 and charge for this locked page
739  // up front so the myriad increments and decrements don't get slowed
740  // down with needless checking.
741  //
742 
743  Pfn1->u3.e1.PrototypePte = 0;
744 
745  MI_ADD_LOCKED_PAGE_CHARGE (Pfn1);
746 
747  Pfn1->u3.e1.ReadInProgress = 1;
748 
749  MiReadInfo->DummyPagePfn = Pfn1;
750 
751  DummyPfn1 = Pfn1;
752 
753  DummyPfn1->u3.e2.ReferenceCount =
754  (USHORT)(DummyPfn1->u3.e2.ReferenceCount + MdlPages);
755 
756  //
757  // Properly initialize the inpage support block fields we overloaded.
758  //
759 
760  InPageSupport->BasePte = *ProtoPteArray;
761 
762  //
763  // Build the proper InPageSupport and MDL to describe this run.
764  //
765 
766  for (; ProtoPteArray < EndProtoPteArray; ProtoPteArray += 1, IoPage += 1, ApiPage += 1) {
767 
768  //
769  // Fill the MDL entry for this RLE.
770  //
771 
772  PointerPte = *ProtoPteArray;
773 
774  ASSERT (PointerPte != NULL);
775 
776  //
777  // The PointerPte better be inside a prototype PTE allocation
778  // so that subsequent page trims update the correct PTEs.
779  //
780 
781  ASSERT (((PointerPte >= (PMMPTE)MmPagedPoolStart) &&
782  (PointerPte <= (PMMPTE)MmPagedPoolEnd)) ||
783  ((PointerPte >= (PMMPTE)MmSpecialPoolStart) && (PointerPte <= (PMMPTE)MmSpecialPoolEnd)));
784 
785  //
786  // Check the state of this prototype PTE now that the PFN lock is held.
787  // If the page is not resident, the PTE must be put in transition with
788  // read in progress before the PFN lock is released.
789  //
790 
791  //
792  // Lock page containing prototype PTEs in memory by
793  // incrementing the reference count for the page.
794  // Unlock any page locked earlier containing prototype PTEs if
795  // the containing page is not the same for both.
796  //
797 
798  if (PfnProto != NULL) {
799 
800  if (PointerPde != MiGetPteAddress (PointerPte)) {
801 
802  ASSERT (PfnProto->u3.e2.ReferenceCount > 1);
803  MI_REMOVE_LOCKED_PAGE_CHARGE_AND_DECREF (PfnProto);
804  PfnProto = NULL;
805  }
806  }
