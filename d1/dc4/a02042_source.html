32 /*++
33 
34 Routine Description:
35 
36  Delete a registry key, clean up Notify block.
37 
38 Arguments:
39 
40  KeyBody - pointer to key handle object
41 
42 Return Value:
43 
44  NTSTATUS
45 
46 --*/
47 {
48  NTSTATUS status;
49  PCM_KEY_NODE ptarget;
50  PHHIVE Hive;
51  HCELL_INDEX Cell;
52  HCELL_INDEX Parent;
53  PCM_KEY_CONTROL_BLOCK KeyControlBlock;
54  LARGE_INTEGER TimeStamp;
55 
56  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_CM,"CmDeleteKey\n"));
57 
58  CmpLockRegistry();
59 
60  //
61  // If already marked for deletion, storage is gone, so
62  // do nothing and return success.
63  //
64  KeyControlBlock = KeyBody->KeyControlBlock;
65 
66  if( KeyControlBlock->ParentKcb == NULL ) {
67  //
68  // cannot delete \Registry
69  //
70  CmpUnlockRegistry();
71  return STATUS_CANNOT_DELETE;
72  }
73 
74  PERFINFO_REG_DELETE_KEY(KeyControlBlock);
75  //
76  // serialize access to this key. and its parent
77  // need to lock them both atomically
78  //
79  CmpLockTwoHashEntriesExclusive(KeyControlBlock->ConvKey,KeyControlBlock->ParentKcb->ConvKey);
80 
81  if (KeyControlBlock->Delete == TRUE) {
82  status = STATUS_SUCCESS;
83  goto Exit;
84  }
85  Hive = KeyControlBlock->KeyHive;
86  Cell = KeyControlBlock->KeyCell;
87  //
88  // no flush from this point on
89  //
90  CmpLockHiveFlusherShared((PCMHIVE)KeyControlBlock->KeyHive);
91 
92  // Mark the hive as read only
93  CmpMarkAllBinsReadOnly(KeyControlBlock->KeyHive);
94 
95  ptarget = (PCM_KEY_NODE)HvGetCell(KeyControlBlock->KeyHive, KeyControlBlock->KeyCell);
96  if( ptarget == NULL ) {
97  //
98  // we couldn't map a view for the bin containing this cell
99  //
100 
101  status = STATUS_INSUFFICIENT_RESOURCES;
102  goto Exit2;
103  }
104 
105  ASSERT( ptarget->Flags == KeyControlBlock->Flags );
106 
107  if ( ((ptarget->SubKeyCounts[Stable] + ptarget->SubKeyCounts[Volatile]) == 0) &&
108  ((ptarget->Flags & KEY_NO_DELETE) == 0))
109  {
110  //
111  // Cell is NOT marked NO_DELETE and does NOT have children
112  // Send Notification while key still present, if delete fails,
113  // we'll have sent a spurious notify, that doesn't matter
114  // Delete the actual storage
115  //
116  Parent = ptarget->Parent;
117 
118  CmpReportNotify(
119  KeyControlBlock,
120  Hive,
121  Cell,
122  REG_NOTIFY_CHANGE_NAME
123  );
124 
125  status = CmpFreeKeyByCell(Hive, Cell, TRUE);
126 
127  if (NT_SUCCESS(status)) {
128  //
129  // post any waiting notifies
130  //
131  CmpFlushNotifiesOnKeyBodyList(KeyControlBlock,FALSE);
132 
133  //
134  // Remove kcb out of cache, but do NOT
135  // free its storage, CmDelete will do that when
136  // the RefCount becomes zero.
137  //
138  // There are two things that can hold the RefCount non-zero.
139  //
140  // 1. open handles for this key
141  // 2. Fake subKeys that are still in DelayClose.
142  //
143  // At this point, we have no way of deleting the fake subkeys from cache
144  // unless we do a search for the whole cache, which is too expensive.
145  // Thus, we decide to either let the fake keys age out of cache or when
146  // someone is doing the lookup for the fake key, then we delete it at that point.
147  // See routine CmpCacheLookup in cmparse.c for more details.
148  //
149  // If the parent has the subkey info or hint cached, free it.
150  // Again, registry is locked exclusively, no need to lock KCB.
151  //
152  CmpCleanUpSubKeyInfo(KeyControlBlock->ParentKcb);
153  ptarget = (PCM_KEY_NODE)HvGetCell(Hive, Parent);
154  if( ptarget != NULL ) {
155 
156  //
157  // this should always be true as CmpFreeKeyByCell always marks the parent dirty on success
158  //
159  KeyControlBlock->ParentKcb->KcbMaxNameLen = (USHORT)ptarget->MaxNameLen;
160  // sanity
161  ASSERT_CELL_DIRTY(Hive,Parent);
162  //
163  // update the LastWriteTime on parent and kcb too
164  //
165  KeQuerySystemTime(&TimeStamp);
166  ptarget->LastWriteTime = TimeStamp;
167  KeyBody->KeyControlBlock->ParentKcb->KcbLastWriteTime = TimeStamp;
168  HvReleaseCell(Hive, Parent);
169  }
170 
171  KeyControlBlock->Delete = TRUE;
172  CmpRemoveKeyControlBlock(KeyControlBlock);
173  KeyControlBlock->KeyCell = HCELL_NIL;
174  }
175 
176  } else {
177 
178  status = STATUS_CANNOT_DELETE;
179 
180  }
181 
182 Exit2:
183  HvReleaseCell(Hive, Cell);
184  CmpUnlockHiveFlusher((PCMHIVE)KeyControlBlock->KeyHive);
185 
186 Exit:
187  CmpUnlockTwoHashEntries(KeyControlBlock->ConvKey,KeyControlBlock->ParentKcb->ConvKey);
188 
189  CmpUnlockRegistry();
190 
191  // Mark the hive as read only
192  CmpMarkAllBinsReadOnly(KeyControlBlock->KeyHive);
193 
194  return status;
195 }
