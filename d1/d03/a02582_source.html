39 PsEnumProcesses (
40  IN PROCESS_ENUM_ROUTINE CallBack,
41  IN PVOID Context
42  )
43 /*++
44 
45 Routine Description:
46 
47  This function calls the callback routine for each active process in the system.
48  Process objects in the process of being deleted are skipped.
49  Returning anything but a success code from the callback routine terminates the enumeration at that point.
50  Processes may be referenced and used later safely.
51 
52 Arguments:
53 
54  CallBack - Routine to be called with its first parameter the enumerated process
55 
56 Return Value:
57 
58  NTSTATUS - Status of call
59 
60 --*/
61 {
62  PLIST_ENTRY ListEntry;
63  PEPROCESS Process, NewProcess;
64  PETHREAD CurrentThread;
65  NTSTATUS Status;
66 
67  Process = NULL;
68 
69  CurrentThread = PsGetCurrentThread ();
70 
71  PspLockProcessList (CurrentThread);
72 
73  for (ListEntry = PsActiveProcessHead.Flink;
74  ListEntry != &PsActiveProcessHead;
75  ListEntry = ListEntry->Flink) {
76 
77  NewProcess = CONTAINING_RECORD (ListEntry, EPROCESS, ActiveProcessLinks);
78  if (ObReferenceObjectSafe (NewProcess)) {
79 
80  PspUnlockProcessList (CurrentThread);
81 
82  if (Process != NULL) {
83  ObDereferenceObject (Process);
84  }
85 
86  Process = NewProcess;
87 
88  Status = CallBack (Process, Context);
89 
90  if (!NT_SUCCESS (Status)) {
91  ObDereferenceObject (Process);
92  return Status;
93  }
94 
95  PspLockProcessList (CurrentThread);
96 
97  }
98  }
99 
100  PspUnlockProcessList (CurrentThread);
101 
102  if (Process != NULL) {
103  ObDereferenceObject (Process);
104  }
105 
106  return STATUS_SUCCESS;
107 }

110 PsGetNextProcess (
111  IN PEPROCESS Process
112  )
113 /*++
114 
115 Routine Description:
116 
117  This function allows code to enumerate all the active processes in the system.
118  The first process (if Process is NULL) or subsequent process (if process not NULL) is returned on
119  each call.
120  If process is not NULL then this process must have previously been obtained by a call to PsGetNextProcess.
121  Enumeration may be terminated early by calling PsQuitNextProcess on the last non-NULL process
122  returned by PsGetNextProcess.
123 
124  Processes may be referenced and used later safely.
125 
126  For example, to enumerate all system processes in a loop use this code fragment:
127 
128  for (Process = PsGetNextProcess (NULL);
129  Process != NULL;
130  Process = PsGetNextProcess (Process)) {
131  ...
132  ...
133  //
134  // Early terminating conditions are handled like this:
135  //
136  if (NeedToBreakOutEarly) {
137  PsQuitNextProcess (Process);
138  break;
139  }
140  }
141 
142 
143 Arguments:
144 
145  Process - Process to get the next process from or NULL for the first process
146 
147 Return Value:
148 
149  PEPROCESS - Next process or NULL if no more processes available
150 
151 --*/
152 {
153  PEPROCESS NewProcess = NULL;
154  PETHREAD CurrentThread;
155  PLIST_ENTRY ListEntry;
156 
157  CurrentThread = PsGetCurrentThread ();
158 
159  PspLockProcessList (CurrentThread);
160 
161  for (ListEntry = (Process == NULL) ? PsActiveProcessHead.Flink : Process->ActiveProcessLinks.Flink;
162  ListEntry != &PsActiveProcessHead;
163  ListEntry = ListEntry->Flink) {
164 
165  NewProcess = CONTAINING_RECORD (ListEntry, EPROCESS, ActiveProcessLinks);
166 
167  //
168  // Processes are removed from this list during process objected deletion (object reference count goes
169  // to zero). To prevent double deletion of the process we need to do a safe reference here.
170  //
171  if (ObReferenceObjectSafe (NewProcess)) {
172  break;
173  }
174  NewProcess = NULL;
175  }
176  PspUnlockProcessList (CurrentThread);
177 
178  if (Process != NULL) {
179  ObDereferenceObject (Process);
180  }
181 
182  return NewProcess;
183 }

246  if (ListEntry != &Process->ThreadListHead) {
247  NewThread = CONTAINING_RECORD (ListEntry, ETHREAD, ThreadListEntry);

595 NtGetNextProcess (
596  __in HANDLE ProcessHandle,
597  __in ACCESS_MASK DesiredAccess,
598  __in ULONG HandleAttributes,
599  __in ULONG Flags,
600  __out PHANDLE NewProcessHandle
601  )
602 /*++
603 
604 Routine Description:
605 
606  This function gets the next for first process in the list of all processes in the system
607 
608 Arguments:
609 
610  ProcessHandle - Process obtained from a previous call to NtGetNextProcess or NULL for the first process
611  DesiredAccess - Access requested for process handle
612  HandleAttributes - Handle attributes requested.
613  Flags - Flags for the operation
614  NewProcessHandle - Pointer to a handle value that is returned on success
615 
616 Return Value:
617 
618  NTSTATUS - Status of operation.
619 
620 --*/
621 {
622  KPROCESSOR_MODE PreviousMode;
623  PEPROCESS Process, NewProcess;
624  NTSTATUS Status;
625  ACCESS_STATE AccessState;
626  AUX_ACCESS_DATA AuxData;
627  HANDLE Handle;
628 
629  PAGED_CODE ();
630 
631  PreviousMode = KeGetPreviousMode ();
632 
633  //
634  // Sanitize handle attributes
635  //
636  HandleAttributes = ObSanitizeHandleAttributes (HandleAttributes, PreviousMode);
637 
638  //
639  // Validate pointer arguments
640  //
641 
642  try {
643  if (PreviousMode != KernelMode) {
644  ProbeForWriteHandle (NewProcessHandle);
645  }
646  *NewProcessHandle = NULL;
647  } except (ExSystemExceptionFilter ()) {
648  return GetExceptionCode ();
649  }
650 
651  //
652  // Check for inclusion of reserved flags and reject the call if present
653  //
654 
655  if (Flags != 0) {
656  return STATUS_INVALID_PARAMETER;
657  }
658 
659  if (ProcessHandle == NULL) {
660  Process = NULL;
661  } else {
662  Status = ObReferenceObjectByHandle (ProcessHandle,
663  0,
664  PsProcessType,
665  PreviousMode,
666  &Process,
667  NULL);
668 
669  if (!NT_SUCCESS (Status)) {
670  return Status;
671  }
672  }
673 
674  NewProcess = PsGetNextProcess (Process);
675 
676  if (NewProcess == NULL) {
677  return STATUS_NO_MORE_ENTRIES;
678  }
679 
680  Status = SeCreateAccessState (&AccessState,
681  &AuxData,
682  DesiredAccess,
683  &PsProcessType->TypeInfo.GenericMapping);
684 
685  if (!NT_SUCCESS (Status)) {
686  ObDereferenceObject (NewProcess);
687  return Status;
688  }
689 
690  if (SeSinglePrivilegeCheck (SeDebugPrivilege, PreviousMode)) {
691  if (AccessState.RemainingDesiredAccess & MAXIMUM_ALLOWED) {
692  AccessState.PreviouslyGrantedAccess |= PROCESS_ALL_ACCESS;
693  } else {
694  AccessState.PreviouslyGrantedAccess |= AccessState.RemainingDesiredAccess;
695  }
696  AccessState.RemainingDesiredAccess = 0;
697  }
698 
699 
700  while (1) {
701  if (NewProcess->GrantedAccess != 0) {
702 
703  Status = ObOpenObjectByPointer (NewProcess,
704  HandleAttributes,
705  &AccessState,
706  0,
707  PsProcessType,
708  PreviousMode,
709  &Handle);
710  if (NT_SUCCESS (Status)) {
711  try {
712  *NewProcessHandle = Handle;
713  } except (ExSystemExceptionFilter ()) {
714  Status = GetExceptionCode ();
715  }
716  break;
717  }
718 
719  if (Status != STATUS_ACCESS_DENIED) {
720  break;
721  }
722  }
723 
724  NewProcess = PsGetNextProcess (NewProcess);
725 
726  if (NewProcess == NULL) {
727  Status = STATUS_NO_MORE_ENTRIES;
728  break;
729  }
730  }
731 
732  SeDeleteAccessState (&AccessState);
733 
734  if (NewProcess != NULL) {
735  ObDereferenceObject (NewProcess);
736  }
737 
738  return Status;
739 }
