150 ObpGetTraceIndex (
151  POBJECT_REF_TRACE Trace
152  )
153 
154 /*++
155 
156 Routine Description:
157 
158  This routine returns the index of 'Trace' in the stack
159  trace hash table (ObpStackTable). If Trace does not exist
160  in the table, it is added, and the new index is returned.
161 
162 Arguments:
163 
164  Trace - Pointer to a stack trace to find in the table
165 
166 Return Value:
167 
168  The index of Trace in ObpStackTable
169 
170 --*/
171 
172 {
173  ULONG_PTR Value = 0;
174  ULONG Index;
175  PUSHORT Key;
176  ULONG Hash;
177 
178  //
179  // Determine the hash value for the stack trace
180  //
181 
182  Key = (PUSHORT)Trace->StackTrace;
183  for (Index = 0; Index < sizeof(Trace->StackTrace) / sizeof(*Key); Index += 2) {
184 
185  Value += Key[Index] ^ Key[Index + 1];
186  }
187 
188  Hash = ((ULONG)Value) % OBTRACE_STACKS;
189 
190  //
191  // Look up the trace at that index (linear probing)
192  //
193 
194  while (ObpStackTable[Hash].StackTrace[0] != NULL &&
195  RtlCompareMemory(&ObpStackTable[Hash], Trace, sizeof(OBJECT_REF_TRACE)) != sizeof(OBJECT_REF_TRACE)) {
196 
197  Hash = (Hash + 1) % OBTRACE_STACKS;
198  if (Hash == ((ULONG)Value) % OBTRACE_STACKS) {
199 
200  return OBTRACE_STACKS;
201  }
202  }
203 
204  //
205  // If the trace doesn't already exist in the table, add it.
206  //
207 
208  if (ObpStackTable[Hash].StackTrace[0] == NULL) {
209 
210  RtlCopyMemory(&ObpStackTable[Hash], Trace, sizeof(OBJECT_REF_TRACE));
211  ObpNumStackTraces++;
212  }
213 
214  return Hash;
215 }

503 
504 Routine Description:
505

508  so, adds the necessary trace to the object reference info.
509

511 
512  ObjectHeader - The object to trace.
513  IsRef - TRUE if this is a ref, FALSE if a deref

516 
517 --*/

569

602 /*++
603 
604 Routine Description:
605 
606 
607  This function opens an object with full access validation and auditing.
608  Soon after entering we capture the SubjectContext for the caller. This
609  context must remain captured until auditing is complete, and passed to
610  any routine that may have to do access checking or auditing.
611 
612 Arguments:
613 
614  ObjectAttributes - Supplies a pointer to the object attributes.
615 
616  ObjectType - Supplies an optional pointer to the object type descriptor.
617 
618  AccessMode - Supplies the processor mode of the access.
619 
620  AccessState - Supplies an optional pointer to the current access status
621  describing already granted access types, the privileges used to get
622  them, and any access types yet to be granted.
623 
624  DesiredAcess - Supplies the desired access to the object.
625 
626  ParseContext - Supplies an optional pointer to parse context.
627 
628  Handle - Supplies a pointer to a variable that receives the handle value.
629 
630 Return Value:
631 
632  If the object is successfully opened, then a handle for the object is
633  created and a success status is returned. Otherwise, an error status is
634  returned.
635 
636 --*/

639  NTSTATUS Status;
640  NTSTATUS HandleStatus;
641  PVOID ExistingObject;
642  HANDLE NewHandle;
643  OB_OPEN_REASON OpenReason;
644  POBJECT_HEADER ObjectHeader;
645  UNICODE_STRING CapturedObjectName;
646  PGENERIC_MAPPING GenericMapping;
647 
648  PAGED_CODE();
649 
650  ObpValidateIrql("ObOpenObjectByName");
651 
652  //
653  // If the object attributes are not specified, then return an error.
654  //
655 
656  *Handle = NULL;
657 
658  if (!ARGUMENT_PRESENT(ObjectAttributes)) {
659 
660  Status = STATUS_INVALID_PARAMETER;
661 
662  } else {
663 
664  POB_TEMP_BUFFER TempBuffer;
665 
666 #if defined(_AMD64_)
667 
668  OB_TEMP_BUFFER ObpTempBuffer;
669 
670  TempBuffer = &ObpTempBuffer;

672 #else
673 
674  TempBuffer = ExAllocatePoolWithTag( NonPagedPool,
675  sizeof(OB_TEMP_BUFFER),
676  'tSbO'

678

680

682  }
683 
684 #endif
685

687  // Capture the object creation information.
688  //
689 
690  Status = ObpCaptureObjectCreateInformation( ObjectType,
691  AccessMode,
692  AccessMode,
693  ObjectAttributes,
694  &CapturedObjectName,
695  &TempBuffer->ObjectCreateInfo,
696  TRUE );
697 
698  //
699  // If the object creation information is successfully captured,
700  // then generate the access state.
701  //
702 
703  if (NT_SUCCESS(Status)) {
704

709  // associated generic mapping. Otherwise, use no generic

716

718  }

736  //

739

741  }

745  //

751  // name.
752  //
753

757  &CapturedObjectName,
758  TempBuffer->ObjectCreateInfo.Attributes,
759  ObjectType,
760  AccessMode,

763  NULL,

765  &TempBuffer->LookupContext,
766  &ExistingObject );

771  //

776

779  // must be a open-if operation. Otherwise, a handle to
780  // an object is being opened.
781  //
782

784

786

788

790  ObjectHeader->ObjectCreateInfo = NULL;
791  }
792 
793  } else {
794 
795  OpenReason = ObOpenHandle;
796  }
797 
798  //
799  // If any of the object attributes are invalid, then
800  // return an error status.
801  //
802 
803  if (ObjectHeader->Type->TypeInfo.InvalidAttributes & TempBuffer->ObjectCreateInfo.Attributes) {
804 
805  Status = STATUS_INVALID_PARAMETER;
806 
807  ObpReleaseLookupContext( &TempBuffer->LookupContext );
808 
809  ObDereferenceObject(ExistingObject);
810 
811  } else {
812 
813  //
814  // The status returned by the object lookup routine
815  // must be returned if the creation of a handle is
816  // successful. Otherwise, the handle creation status
817  // is returned.
818  //
819 
820  HandleStatus = ObpCreateHandle( OpenReason,
821  ExistingObject,
822  ObjectType,
823  AccessState,
824  0,
825  TempBuffer->ObjectCreateInfo.Attributes,
826  &TempBuffer->LookupContext,
827  AccessMode,
828  (PVOID *)NULL,
829  &NewHandle );
830 
831  if (!NT_SUCCESS(HandleStatus)) {
832 
833  ObDereferenceObject(ExistingObject);
834 
835  Status = HandleStatus;
836 
837  } else {
838 
839  *Handle = NewHandle;
840  }
841  }
842 
843  } else {
844 
845  ObpReleaseLookupContext( &TempBuffer->LookupContext );
846  }
847  }
848 
849  //
850  // If the access state was generated, then delete the access
851  // state.
852  //
853 
854  if (AccessState == &TempBuffer->LocalAccessState) {
855 
856  SeDeleteAccessState(AccessState);
857  }
858 
859  //
860  // Free the create information.
861  //
862 
863  FreeCreateInfo:
864 
865  ObpReleaseObjectCreateInformation(&TempBuffer->ObjectCreateInfo);
866 
867  if (CapturedObjectName.Buffer != NULL) {
868 
869  ObpFreeObjectNameBuffer(&CapturedObjectName);
870  }
871  }
872 
873 #if !defined(_AMD64_)
874 
875  ExFreePool(TempBuffer);
876 
877 #endif
878 
879  }
880 
881  return Status;
882 }
