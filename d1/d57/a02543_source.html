29 /*++
30 
31 Routine Description:
32 
33  This routine performs a copy on write operation for the specified
34  virtual address.
35 
36 Arguments:
37 
38  FaultingAddress - Supplies the virtual address which caused the fault.
39 
40  PointerPte - Supplies the pointer to the PTE which caused the page fault.
41 
42 Return Value:
43 
44  Returns TRUE if the page was actually split, FALSE if not.
45 
46 Environment:
47 
48  Kernel mode, APCs disabled, working set mutex held.
49 
50 --*/
51 
52 {
53  MMPTE TempPte;
54  MMPTE TempPte2;
55  PMMPTE MappingPte;
56  PFN_NUMBER PageFrameIndex;
57  PFN_NUMBER NewPageIndex;
58  PVOID CopyTo;
59  PVOID CopyFrom;
60  KIRQL OldIrql;
61  PMMPFN Pfn1;
62  PEPROCESS CurrentProcess;
63  PMMCLONE_BLOCK CloneBlock;
64  PMMCLONE_DESCRIPTOR CloneDescriptor;
65  WSLE_NUMBER WorkingSetIndex;
66  LOGICAL FakeCopyOnWrite;
67  PMMWSL WorkingSetList;
68  PVOID SessionSpace;
69  PLIST_ENTRY NextEntry;
70  PIMAGE_ENTRY_IN_SESSION Image;
71 
72  //
73  // This is called from MmAccessFault, the PointerPte is valid
74  // and the working set mutex ensures it cannot change state.
75  //
76  // Capture the PTE contents to TempPte.
77  //
78 
79  TempPte = *PointerPte;
80  ASSERT (TempPte.u.Hard.Valid == 1);
81 
82  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (&TempPte);
83  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
84 
85  //
86  // Check to see if this is a prototype PTE with copy on write enabled.
87  //
88 
89  FakeCopyOnWrite = FALSE;
90  CurrentProcess = PsGetCurrentProcess ();
91  CloneBlock = NULL;
92 
93  if (FaultingAddress >= (PVOID) MmSessionBase) {
94 
95  WorkingSetList = MmSessionSpace->Vm.VmWorkingSetList;
96  ASSERT (Pfn1->u3.e1.PrototypePte == 1);
97  SessionSpace = (PVOID) MmSessionSpace;
98 
99  MM_SESSION_SPACE_WS_LOCK_ASSERT ();
100 
101  if (MmSessionSpace->ImageLoadingCount != 0) {
102 
103  NextEntry = MmSessionSpace->ImageList.Flink;
104 
105  while (NextEntry != &MmSessionSpace->ImageList) {
106 
107  Image = CONTAINING_RECORD (NextEntry, IMAGE_ENTRY_IN_SESSION, Link);
108 
109  if ((FaultingAddress >= Image->Address) &&
110  (FaultingAddress <= Image->LastAddress)) {
111 
112  if (Image->ImageLoading) {
113 
114  ASSERT (Pfn1->u3.e1.PrototypePte == 1);
115 
116  TempPte.u.Hard.CopyOnWrite = 0;
117  TempPte.u.Hard.Write = 1;
118 
119  //
120  // The page is no longer copy on write, update the PTE
121  // setting both the dirty bit and the accessed bit.
122  //
123  // Even though the page's current backing is the image
124  // file, the modified writer will convert it to
125  // pagefile backing when it notices the change later.
126  //
127 
128  MI_SET_PTE_DIRTY (TempPte);
129  MI_SET_ACCESSED_IN_PTE (&TempPte, 1);
130 
131  MI_WRITE_VALID_PTE_NEW_PROTECTION (PointerPte, TempPte);
132 
133  //
134  // The TB entry must be flushed as the valid PTE with
135  // the dirty bit clear has been fetched into the TB. If
136  // it isn't flushed, another fault is generated as the
137  // dirty bit is not set in the cached TB entry.
138  //
139 
140  MI_FLUSH_SINGLE_TB (FaultingAddress, TRUE);
141 
142  return FALSE;
143  }
144  break;
145  }
146 
147  NextEntry = NextEntry->Flink;
148  }
149  }
150  }
151  else {
152  WorkingSetList = MmWorkingSetList;
153  SessionSpace = NULL;
154 
155  //
156  // If a fork operation is in progress, block until the fork is
157  // completed, then retry the whole operation as the state of
158  // everything may have changed between when the mutexes were
159  // released and reacquired.
160  //
161 
162  if (CurrentProcess->ForkInProgress != NULL) {
163  if (MiWaitForForkToComplete (CurrentProcess) == TRUE) {
164  return FALSE;
165  }
166  }
167 
168  if (TempPte.u.Hard.CopyOnWrite == 0) {
169 
170  //
171  // This is a fork page which is being made private in order
172  // to change the protection of the page.
173  // Do not make the page writable.
174  //
175 
176  FakeCopyOnWrite = TRUE;
177  }
178  }
179 
180  WorkingSetIndex = MiLocateWsle (FaultingAddress,
181  WorkingSetList,
182  Pfn1->u1.WsIndex,
183  FALSE);
184 
185  //
186  // The page must be copied into a new page.
187  //
188 
189  LOCK_PFN (OldIrql);
190 
191  if ((MmAvailablePages < MM_HIGH_LIMIT) &&
192  (MiEnsureAvailablePageOrWait (SessionSpace != NULL ? HYDRA_PROCESS : CurrentProcess, OldIrql))) {
193 
194  //
195  // A wait operation was performed to obtain an available
196  // page and the working set mutex and PFN lock have
197  // been released and various things may have changed for
198  // the worse. Rather than examine all the conditions again,
199  // return and if things are still proper, the fault will
200  // be taken again.
201  //
202 
203  UNLOCK_PFN (OldIrql);
204  return FALSE;
205  }
206 
207  //
208  // This must be a prototype PTE. Perform the copy on write.
209  //
210 
211  ASSERT (Pfn1->u3.e1.PrototypePte == 1);
212 
213  //
214  // A page is being copied and made private, the global state of
215  // the shared page needs to be updated at this point on certain
216  // hardware. This is done by ORing the dirty bit into the modify bit in
217  // the PFN element.
218  //
219  // Note that a session page cannot be dirty (no POSIX-style forking is
220  // supported for these drivers).
221  //
222 
223  if (SessionSpace != NULL) {
224  ASSERT ((TempPte.u.Hard.Valid == 1) && (TempPte.u.Hard.Write == 0));
225  ASSERT (!MI_IS_PTE_DIRTY (TempPte));
226 
227  NewPageIndex = MiRemoveAnyPage (MI_GET_PAGE_COLOR_FROM_SESSION(MmSessionSpace));
228  }
229  else {
230  MI_CAPTURE_DIRTY_BIT_TO_PFN (PointerPte, Pfn1);
231  CloneBlock = (PMMCLONE_BLOCK) Pfn1->PteAddress;
232 
233  //
234  // Get a new page with the same color as this page.
235  //
236 
237  NewPageIndex = MiRemoveAnyPage (
238  MI_PAGE_COLOR_PTE_PROCESS(PageFrameIndex,
239  &CurrentProcess->NextPageColor));
240  }
241 
242  MiInitializeCopyOnWritePfn (NewPageIndex,
243  PointerPte,
244  WorkingSetIndex,
245  WorkingSetList);
246 
247  UNLOCK_PFN (OldIrql);
248 
249  InterlockedIncrement (&KeGetCurrentPrcb ()->MmCopyOnWriteCount);
250 
251  CopyFrom = PAGE_ALIGN (FaultingAddress);
252 
253  MappingPte = MiReserveSystemPtes (1, SystemPteSpace);
254 
255  if (MappingPte != NULL) {
256 
257  MI_MAKE_VALID_KERNEL_PTE (TempPte2,
258  NewPageIndex,
259  MM_READWRITE,
260  MappingPte);
261 
262  MI_SET_PTE_DIRTY (TempPte2);
263 
264  if (Pfn1->u3.e1.CacheAttribute == MiNonCached) {
265  MI_DISABLE_CACHING (TempPte2);
266  }
267  else if (Pfn1->u3.e1.CacheAttribute == MiWriteCombined) {
268  MI_SET_PTE_WRITE_COMBINE (TempPte2);
269  }
270 
271  MI_WRITE_VALID_PTE (MappingPte, TempPte2);
272 
273  CopyTo = MiGetVirtualAddressMappedByPte (MappingPte);
274  }
275  else {
276 
277  CopyTo = MiMapPageInHyperSpace (CurrentProcess,
278  NewPageIndex,
279  &OldIrql);
280  }
281 
282  KeCopyPage (CopyTo, CopyFrom);
283 
284  if (MappingPte != NULL) {
285  MiReleaseSystemPtes (MappingPte, 1, SystemPteSpace);
286  }
287  else {
288  MiUnmapPageInHyperSpace (CurrentProcess, CopyTo, OldIrql);
289  }
290 
291  if (!FakeCopyOnWrite) {
292 
293  //
294  // If the page was really a copy on write page, make it
295  // accessed, dirty and writable. Also, clear the copy-on-write
296  // bit in the PTE.
297  //
298 
299  MI_SET_PTE_DIRTY (TempPte);
300  TempPte.u.Hard.Write = 1;
301  MI_SET_ACCESSED_IN_PTE (&TempPte, 1);
302  TempPte.u.Hard.CopyOnWrite = 0;
303  }
304 
305  //
306  // Regardless of whether the page was really a copy on write,
307  // the frame field of the PTE must be updated.
308  //
309 
310  TempPte.u.Hard.PageFrameNumber = NewPageIndex;
311 
312  //
313  // If the modify bit is set in the PFN database for the
314  // page, the data cache must be flushed. This is due to the
315  // fact that this process may have been cloned and the cache
316  // still contains stale data destined for the page we are
317  // going to remove.
318  //
319 
320  ASSERT (TempPte.u.Hard.Valid == 1);
321 
322  MI_WRITE_VALID_PTE_NEW_PAGE (PointerPte, TempPte);
323 
324  //
325  // Flush the TB entry for this page.
326  //
327 
328  if (SessionSpace == NULL) {
329 
330  MI_FLUSH_SINGLE_TB (FaultingAddress, FALSE);
331 
332  //
333  // Increment the number of private pages.
334  //
335 
336  CurrentProcess->NumberOfPrivatePages += 1;
337  }
338  else {
339 
340  MI_FLUSH_SINGLE_TB (FaultingAddress, TRUE);
341 
342  ASSERT (Pfn1->u3.e1.PrototypePte == 1);
343  }
344 
345  //
346  // Decrement the share count for the page which was copied
347  // as this PTE no longer refers to it.
348  //
349 
350  LOCK_PFN (OldIrql);
351 
352  MiDecrementShareCount (Pfn1, PageFrameIndex);
353 
354  if (SessionSpace == NULL) {
355 
356  CloneDescriptor = MiLocateCloneAddress (CurrentProcess,
357  (PVOID)CloneBlock);
358 
359  if (CloneDescriptor != NULL) {
360 
361  //
362  // Decrement the reference count for the clone block,
363  // note that this could release and reacquire the mutexes.
364  //
365 
366  MiDecrementCloneBlockReference (CloneDescriptor,
367  CloneBlock,
368  CurrentProcess,
369  NULL,
370  OldIrql);
371  }
372  }
373 
374  UNLOCK_PFN (OldIrql);
375  return TRUE;
376 }

388 /*++
389 
390 Routine Description:
391 
392  This routine sets dirty in the specified PTE and the modify bit in the
393  corresponding PFN element. If any page file space is allocated, it
394  is deallocated.
395 
396 Arguments:
397 
398  FaultingAddress - Supplies the faulting address.
399 
400  PointerPte - Supplies a pointer to the corresponding valid PTE.
401 
402  PfnHeld - Supplies TRUE if the PFN lock is already held.
403 
404 Return Value:
405 
406  TRUE if action was taken, FALSE if not.
407 
408 Environment:
409 
410  Kernel mode, APCs disabled, working set pushlock held.
411 
412 --*/
413 
414 {
415  MMPTE TempPte;
416  PFN_NUMBER PageFrameIndex;
417  PMMPFN Pfn1;
418 
419  //
420  // The page is NOT copy on write, update the PTE setting both the
421  // dirty bit and the accessed bit. Note, that as this PTE is in
422  // the TB, the TB must be flushed.
423  //
424 
425  TempPte = *PointerPte;
426 
427  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (&TempPte);
428 
429  //
430  // This may be a PTE from a rotate physical frame so there may be no
431  // corresponding PFN for it.
432  //
433 
434  if (!MI_IS_PFN (PageFrameIndex)) {
435  return FALSE;
436  }
437 
438  MI_SET_PTE_DIRTY (TempPte);
439  MI_SET_ACCESSED_IN_PTE (&TempPte, 1);
440 
441  MI_WRITE_VALID_PTE_NEW_PROTECTION (PointerPte, TempPte);
442 
443  //
444  // Check state of PFN lock and if not held, don't update PFN database.
445  //
446 
447  if (PfnHeld) {
448 
449  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
450 
451  //
452  // Set the modified field in the PFN database, also, if the physical
453  // page is currently in a paging file, free up the page file space
454  // as the contents are now worthless.
455  //
456 
457  if ((Pfn1->OriginalPte.u.Soft.Prototype == 0) &&
458  (Pfn1->u3.e1.WriteInProgress == 0)) {
459 
460  //
461  // This page is in page file format, deallocate the page file space.
462  //
463 
464  MiReleasePageFileSpace (Pfn1->OriginalPte);
465 
466  //
467  // Change original PTE to indicate no page file space is reserved,
468  // otherwise the space will be deallocated when the PTE is
469  // deleted.
470  //
471 
472  Pfn1->OriginalPte.u.Soft.PageFileHigh = 0;
473  }
474 
475  MI_SET_MODIFIED (Pfn1, 1, 0x17);
476  }
477 
478  //
479  // The TB entry must be flushed as the valid PTE with the dirty bit clear
480  // has been fetched into the TB. If it isn't flushed, another fault
481  // is generated as the dirty bit is not set in the cached TB entry.
482  //
483 
484  KeFillEntryTb (FaultingAddress);
485  return TRUE;
486 }
