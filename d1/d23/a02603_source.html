66 /*++
67 
68 Routine Description:
69 
70  This routine implements the NtQueryVolumeInformation API call.
71 
72 Arguments:
73 
74  Vcb - Supplies the volume being queried.
75 
76  Irp - Supplies the Irp being processed.
77 
78  IrpSp - Supplies parameters describing the read
79 
80 Return Value:
81 
82  NTSTATUS - The status for the Irp.
83 
84 --*/
85 
86 {
87  NTSTATUS Status;
88 
89  ULONG Length;
90  FS_INFORMATION_CLASS FsInformationClass;
91  PVOID Buffer;
92 
93  PAGED_CODE();
94 
95  //
96  // Reference our input parameters to make things easier
97  //
98 
99  Length = IrpSp->Parameters.QueryVolume.Length;
100  FsInformationClass = IrpSp->Parameters.QueryVolume.FsInformationClass;
101  Buffer = Irp->AssociatedIrp.SystemBuffer;
102 
103  //
104  // Based on the information class we'll do different actions. Each
105  // of the procedures that we're calling fills up the output buffer
106  // if possible and returns true if it successfully filled the buffer
107  // and false if it couldn't wait for any I/O to complete.
108  //
109 
110  switch (FsInformationClass) {
111 
112  case FileFsVolumeInformation:
113 
114  Status = RawQueryFsVolumeInfo( Vcb, Buffer, &Length );
115  break;
116 
117  case FileFsSizeInformation:
118 
119  Status = RawQueryFsSizeInfo( Vcb, Buffer, &Length );
120  break;
121 
122  case FileFsDeviceInformation:
123 
124  Status = RawQueryFsDeviceInfo( Vcb, Buffer, &Length );
125  break;
126 
127  case FileFsAttributeInformation:
128 
129  Status = RawQueryFsAttributeInfo( Vcb, Buffer, &Length );
130  break;
131 
132  default:
133 
134  Status = STATUS_INVALID_PARAMETER;
135  break;
136  }
137 
138  //
139  // Set the information field to the number of bytes actually filled in,
140  // and complete the request.
141  //
142 
143  Irp->IoStatus.Information = IrpSp->Parameters.QueryVolume.Length - Length;
144 
145  RawCompleteRequest( Irp, Status );
146 
147  return Status;
148 }

220 /*++
221 
222 Routine Description:
223 
224  This routine implements the query volume size call
225 
226 Arguments:
227 
228  Vcb - Supplies the Vcb being queried
229 
230  Buffer - Supplies a pointer to the output buffer where the information
231  is to be returned
232 
233  Length - Supplies the length of the buffer in byte. This variable
234  upon return receives the remaining bytes free in the buffer
235 
236 Return Value:
237 
238  Status - Returns the status for the query
239 
240 --*/
241 
242 {
243  PIRP Irp;
244  KEVENT Event;
245  NTSTATUS Status;
246  IO_STATUS_BLOCK Iosb;
247  PDEVICE_OBJECT RealDevice;
248 
249  DISK_GEOMETRY DiskGeometry;
250  PARTITION_INFORMATION PartitionInformation;
251  GET_LENGTH_INFORMATION GetLengthInformation;
252 
253  BOOLEAN DriveIsPartitioned;
254 
255  PAGED_CODE();
256 
257  //
258  // Make sure the buffer is large enough
259  //
260 
261  if (*Length < sizeof(FILE_FS_SIZE_INFORMATION)) {
262 
263  return STATUS_BUFFER_OVERFLOW;
264  }
265 
266  RtlZeroMemory( Buffer, sizeof(FILE_FS_SIZE_INFORMATION) );
267 
268  //
269  // Prepare for our device control below. The device drivers only
270  // have to copy geometry and partition info from in-memory structures,
271  // so it is OK to make these calls even when we can't wait.
272  //
273 
274  KeInitializeEvent( &Event, NotificationEvent, FALSE );
275  RealDevice = Vcb->Vpb->RealDevice;
276 
277  //
278  // Query the disk geometry
279  //
280 
281  Irp = IoBuildDeviceIoControlRequest( IOCTL_DISK_GET_DRIVE_GEOMETRY,
282  RealDevice,
283  NULL,
284  0,
285  &DiskGeometry,
286  sizeof(DISK_GEOMETRY),
287  FALSE,
288  &Event,
289  &Iosb );
290 
291  if ( Irp == NULL ) {
292  return STATUS_INSUFFICIENT_RESOURCES;
293  }
294 
295  if ( (Status = IoCallDriver( RealDevice, Irp )) == STATUS_PENDING ) {
296 
297  (VOID) KeWaitForSingleObject( &Event,
298  Executive,
299  KernelMode,
300  FALSE,
301  (PLARGE_INTEGER)NULL );
302 
303  Status = Iosb.Status;
304  }
305 
306  //
307  // If this call didn't succeed, the drive hasn't even been low-level
308  // formatted, and thus geometry information is undefined.
309  //
310 
311  if (!NT_SUCCESS( Status )) {
312 
313  *Length = 0;
314  return Status;
315  }
316 
317  //
318  // See if we have to check the partition information (floppy disks are
319  // the only type that can't have partitions )
320  //
321 
322  if ( FlagOn( RealDevice->Characteristics, FILE_FLOPPY_DISKETTE )) {
323 
324  DriveIsPartitioned = FALSE;
325  PartitionInformation.PartitionLength.QuadPart = 0;
326 
327  } else {
328 
329  //
330  // Query the length info.
331  //
332 
333  KeResetEvent( &Event );
334 
335  Irp = IoBuildDeviceIoControlRequest( IOCTL_DISK_GET_LENGTH_INFO,
336  RealDevice,
337  NULL,
338  0,
339  &GetLengthInformation,
340  sizeof(GET_LENGTH_INFORMATION),
341  FALSE,
342  &Event,
343  &Iosb );
344 
345  if ( Irp == NULL ) {
346  return STATUS_INSUFFICIENT_RESOURCES;
347  }
348 
349  if ( (Status = IoCallDriver( RealDevice, Irp )) == STATUS_PENDING ) {
350 
351  (VOID) KeWaitForSingleObject( &Event,
352  Executive,
353  KernelMode,
354  FALSE,
355  (PLARGE_INTEGER)NULL );
356 
357  Status = Iosb.Status;
358  }
359 
360  PartitionInformation.PartitionLength = GetLengthInformation.Length;
361 
362  if ( !NT_SUCCESS (Status) ) {
363 
364  //
365  // Query the partition table
366  //
367 
368  KeResetEvent( &Event );
369 
370  Irp = IoBuildDeviceIoControlRequest( IOCTL_DISK_GET_PARTITION_INFO,
371  RealDevice,
372  NULL,
373  0,
374  &PartitionInformation,
375  sizeof(PARTITION_INFORMATION),
376  FALSE,
377  &Event,
378  &Iosb );
379 
380  if ( Irp == NULL ) {
381  return STATUS_INSUFFICIENT_RESOURCES;
382  }
383 
384  if ( (Status = IoCallDriver( RealDevice, Irp )) == STATUS_PENDING ) {
385 
386  (VOID) KeWaitForSingleObject( &Event,
387  Executive,
388  KernelMode,
389  FALSE,
390  (PLARGE_INTEGER)NULL );
391 
392  Status = Iosb.Status;
393  }
394 
395  //
396  // If we get back invalid device request, the disk is not partitioned
397  //
398 
399  if ( !NT_SUCCESS (Status) ) {
400 
401  DriveIsPartitioned = FALSE;
402 
403  } else {
404 
405  DriveIsPartitioned = TRUE;
406  }
407 
408  } else {
409 
410  DriveIsPartitioned = TRUE;
411  }
412  }
413 
414  //
415  // Set the output buffer
416  //
417 
418  Buffer->BytesPerSector = DiskGeometry.BytesPerSector;
419 
420  Buffer->SectorsPerAllocationUnit = 1;
421 
422  //
423  // Now, based on whether the disk is partitioned, compute the
424  // total number of sectors on this disk.
425  //
426 
427  Buffer->TotalAllocationUnits =
428  Buffer->AvailableAllocationUnits = ( DriveIsPartitioned == TRUE ) ?
429 
430  RtlExtendedLargeIntegerDivide( PartitionInformation.PartitionLength,
431  DiskGeometry.BytesPerSector,
432  NULL )
433 
434  :
435 
436  RtlExtendedIntegerMultiply( DiskGeometry.Cylinders,
437  DiskGeometry.TracksPerCylinder *
438  DiskGeometry.SectorsPerTrack );
439 
440  //
441  // Adjust the length variable
442  //
443 
444  *Length -= sizeof(FILE_FS_SIZE_INFORMATION);
445 
446  //
447  // And return success to our caller
448  //
449 
450  return STATUS_SUCCESS;
451 }
