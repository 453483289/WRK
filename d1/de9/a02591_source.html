37 PsSuspendThread (
38  IN PETHREAD Thread,
39  OUT PULONG PreviousSuspendCount OPTIONAL
40  )
41 /*++
42 
43 Routine Description:
44 
45  This function suspends the target thread, and optionally
46  returns the previous suspend count.
47 
48 Arguments:
49 
50  ThreadHandle - Supplies a handle to the thread object to suspend.
51 
52  PreviousSuspendCount - An optional parameter, that if specified
53  points to a variable that receives the thread's previous suspend
54  count.
55 
56 Return Value:
57 
58  NTSTATUS - Status of call
59 
60 --*/
61 {
62  NTSTATUS Status;
63  ULONG LocalPreviousSuspendCount = 0;
64 
65  PAGED_CODE();
66 
67  if (Thread == PsGetCurrentThread ()) {
68  try {
69  LocalPreviousSuspendCount = (ULONG) KeSuspendThread (&Thread->Tcb);
70  Status = STATUS_SUCCESS;
71  } except ((GetExceptionCode () == STATUS_SUSPEND_COUNT_EXCEEDED)?
72  EXCEPTION_EXECUTE_HANDLER :
73  EXCEPTION_CONTINUE_SEARCH) {
74  Status = GetExceptionCode();
75  }
76  } else {
77  //
78  // Protect the remote thread from being rundown.
79  //
80  if (ExAcquireRundownProtection (&Thread->RundownProtect)) {
81 
82  //
83  // Don't allow suspend if we are being deleted
84  //
85  if (Thread->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_TERMINATED) {
86  Status = STATUS_THREAD_IS_TERMINATING;
87  } else {
88  try {
89  LocalPreviousSuspendCount = (ULONG) KeSuspendThread (&Thread->Tcb);
90  Status = STATUS_SUCCESS;
91  } except ((GetExceptionCode () == STATUS_SUSPEND_COUNT_EXCEEDED)?
92  EXCEPTION_EXECUTE_HANDLER :
93  EXCEPTION_CONTINUE_SEARCH) {
94  Status = GetExceptionCode();
95  }
96  //
97  // If deletion was started after we suspended then wake up the thread
98  //
99  if (Thread->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_TERMINATED) {
100  KeForceResumeThread (&Thread->Tcb);
101  LocalPreviousSuspendCount = 0;
102  Status = STATUS_THREAD_IS_TERMINATING;
103  }
104  }
105  ExReleaseRundownProtection (&Thread->RundownProtect);
106  } else {
107  Status = STATUS_THREAD_IS_TERMINATING;
108  }
109  }
110 
111  if (ARGUMENT_PRESENT (PreviousSuspendCount)) {
112  *PreviousSuspendCount = LocalPreviousSuspendCount;
113  }
114  return Status;
115 }
