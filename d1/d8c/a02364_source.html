41 ; VOID
42 ; KeUpdateSystemTime (
43 ; IN PKTRAP_FRAME TrapFrame,
44 ; IN ULONG64 Increment
45 ; )
46 ;
47 ; Routine Description:
48 ;
49 ; This routine is called as the result of an interrupt generated by the
50 ; interval timer. Its function is to update the interrupt time, update the
51 ; system time, and check to determine if a timer has expired.
52 ;
53 ; N.B. This routine is executed on a single processor in a multiprocess
54 ; system. The remainder of the processors only execute the quantum end
55 ; and runtime update code.
56 ;
57 ; Arguments:
58 ;
59 ; TrapFrame (rcx) - Supplies the address of a trap frame.
60 ;
61 ; Increment (rdx) - Supplies the time increment value in 100 nanosecond
62 ; units.
63 ;
64 ; Return Value:
65 ;
66 ; None.
67 ;
68 ;--

72  P2Home dq ? ;
73  P3Home dq ? ;
74  P4Home dq ?

76 UsFrame ends

103 ;
104 ; N.B. Interrupt time is aligned 0 mod 8.

114  mov UsInterruptTime[rcx], r8 ; save updated interrupt time

116 
117 ifndef NT_UP

121 else

129  add gs:[PcMasterOffset], eax ; add maximum time to residue

131 ;
132 ; Update system time.
133 ;
134 ; N.B. System time is aligned 4 mod 8, however, this data does not cross
135 ; a cache line and is, therefore, updated atomically,
136 ;

160 ;
161 
162  .errnz (TIMER_ENTRY_SIZE - 24)
163 
164  mov rcx, r10 ; copy tick count value
165  and ecx, TIMER_TABLE_SIZE - 1 ; isolate current hand value
166  lea rcx, [rcx + rcx * 2] ; multiply by 3
167  cmp r8, TtTime[r11 + rcx * 8] ; compare due time
168  jae short KiUS30 ; if ae, timer has expired

171 ;
172 ; Check to determine if a timer has expired.
173 ;
174 
175 KiUS20: mov rcx, r10 ; copy tick count value
176  and ecx, TIMER_TABLE_SIZE - 1 ; isolate current hand value
177  lea rcx, [rcx + rcx * 2] ; multiply by 3
178  cmp r8, TtTime[r11 + rcx * 8] ; compare due time
179  jb short KiUS40 ; if b, timer has not expired
180 
181 ;
182 ; A timer has expired.
183 ;
184 ; Set the timer hand value in the current processor block if it is not already
185 ; set.
186 ;
187 
188 KiUS30: mov rdx, gs:[PcCurrentPrcb] ; get current processor block address
189  cmp qword ptr PbTimerRequest[rdx], 0 ; check if expiration active
190  jne short KiUS40 ; if ne, expiration already active
191  mov PbTimerHand[rdx], r10 ; set timer hand value
192  mov byte ptr PbInterruptRequest[rdx], TRUE ; set interrupt request
193

196 ;
197

199  mov rdx, KiTimeIncrement ; set time increment value
200  call KeUpdateRunTime ; update runtime
201

203 
204 KiUS50: mov byte ptr gs:[PcSkipTick], 0 ; clear skip tick indicator
205 
206 endif

209  add rsp, (sizeof UsFrame) ; deallocate stack frame
210  ret ; return
