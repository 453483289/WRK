53 SepCreateImpersonationTokenDacl(
54  __in PTOKEN Token,
55  __in PACCESS_TOKEN PrimaryToken,
56  __deref_out PACL *Acl
57  )
58 /*++
59 
60 Routine Description:
61 
62  This routine modifies the DACL protecting the passed token to allow
63  the current user (described by the PrimaryToken parameter) full access.
64  This permits callers of NtOpenThreadToken to call with OpenAsSelf==TRUE
65  and succeed.
66 
67  The new DACL placed on the token is as follows:
68 
69  ACE 0 - Server gets TOKEN_ALL_ACCESS
70 
71  ACE 1 - Client gets TOKEN_ALL_ACCESS
72 
73  ACE 2 - Admins gets TOKEN_ALL_ACCESS
74 
75  ACE 3 - System gets TOKEN_ALL_ACCESS
76 
77  ACE 4 - Restricted gets TOKEN_ALL_ACCESS
78 
79 
80 Arguments:
81 
82  Token - The token whose protection is to be modified.
83 
84  PrimaryToken - Token representing the subject to be granted access.
85 
86  Acl - Returns the modified ACL, allocated out of PagedPool.
87 
88 
89 Return Value:
90 
91 
92 --*/
93 
94 {
95  PSID ServerUserSid;
96  PSID ClientUserSid;
97  NTSTATUS Status = STATUS_SUCCESS;
98  ULONG AclLength;
99  PACL NewDacl;
100  PSECURITY_DESCRIPTOR OldDescriptor;
101  BOOLEAN MemoryAllocated;
102  PACL OldDacl;
103  BOOLEAN DaclPresent;
104  BOOLEAN DaclDefaulted;
105 
106  PAGED_CODE();
107 
108  ServerUserSid = ((PTOKEN)PrimaryToken)->UserAndGroups[0].Sid;
109 
110  ClientUserSid = Token->UserAndGroups[0].Sid;
111 
112  //
113  // Compute how much space we'll need for the new DACL.
114  //
115 
116  AclLength = 5 * sizeof( ACCESS_ALLOWED_ACE ) - 5 * sizeof( ULONG ) +
117  SeLengthSid( ServerUserSid ) + SeLengthSid( SeLocalSystemSid ) +
118  SeLengthSid( ClientUserSid ) + SeLengthSid( SeAliasAdminsSid ) +
119  SeLengthSid( SeRestrictedSid ) + sizeof( ACL );
120 
121  NewDacl = ExAllocatePool( PagedPool, AclLength );
122 
123  if (NewDacl == NULL) {
124 
125  *Acl = NULL;
126  return STATUS_INSUFFICIENT_RESOURCES;
127  }
128 
129  Status = RtlCreateAcl( NewDacl, AclLength, ACL_REVISION2 );
130  ASSERT(NT_SUCCESS( Status ));
131 
132  Status = RtlAddAccessAllowedAce (
133  NewDacl,
134  ACL_REVISION2,
135  TOKEN_ALL_ACCESS,
136  ServerUserSid
137  );
138  ASSERT( NT_SUCCESS( Status ));
139 
140  Status = RtlAddAccessAllowedAce (
141  NewDacl,
142  ACL_REVISION2,
143  TOKEN_ALL_ACCESS,
144  ClientUserSid
145  );
146  ASSERT( NT_SUCCESS( Status ));
147 
148  Status = RtlAddAccessAllowedAce (
149  NewDacl,
150  ACL_REVISION2,
151  TOKEN_ALL_ACCESS,
152  SeAliasAdminsSid
153  );
154  ASSERT( NT_SUCCESS( Status ));
155 
156  Status = RtlAddAccessAllowedAce (
157  NewDacl,
158  ACL_REVISION2,
159  TOKEN_ALL_ACCESS,
160  SeLocalSystemSid
161  );
162  ASSERT( NT_SUCCESS( Status ));
163 
164  if(ARGUMENT_PRESENT(((PTOKEN)PrimaryToken)->RestrictedSids) ||
165  ARGUMENT_PRESENT(Token->RestrictedSids)) {
166  Status = RtlAddAccessAllowedAce (
167  NewDacl,
168  ACL_REVISION2,
169  TOKEN_ALL_ACCESS,
170  SeRestrictedSid
171  );
172  ASSERT( NT_SUCCESS( Status ));
173  }
174 
175  *Acl = NewDacl;
176  return STATUS_SUCCESS;
177 }

219 NTSTATUS
220 NtOpenProcessTokenEx(
221  __in HANDLE ProcessHandle,
222  __in ACCESS_MASK DesiredAccess,
223  __in ULONG HandleAttributes,
224  __out PHANDLE TokenHandle
225  )
226 
227 /*++
228 
229 Routine Description:
230 
231  Open a token object associated with a process and return a handle
232  that may be used to access that token.
233 
234 Arguments:
235 
236  ProcessHandle - Specifies the process whose token is to be
237  opened.
238 
239  DesiredAccess - Is an access mask indicating which access types
240  are desired to the token. These access types are reconciled
241  with the Discretionary Access Control list of the token to
242  determine whether the accesses will be granted or denied.
243 
244  HandleAttributes - Attributes for the created handle. Only OBJ_KERNEL_HANDLE at present.
245 
246  TokenHandle - Receives the handle of the newly opened token.
247 
248 Return Value:
249 
250  STATUS_SUCCESS - Indicates the operation was successful.
251 
252 --*/
253 {
254 
255  PVOID Token;
256  KPROCESSOR_MODE PreviousMode;
257  NTSTATUS Status;
258 
259  HANDLE LocalHandle;
260 
261  PAGED_CODE();
262 
263  PreviousMode = KeGetPreviousMode();
264 
265  //
266  // Sanitize the handle attribute flags
267  //
268  HandleAttributes = ObSanitizeHandleAttributes (HandleAttributes, PreviousMode);
269 
270  //
271  // Probe parameters
272  //
273 
274  if (PreviousMode != KernelMode) {
275 
276  try {
277 
278  ProbeForWriteHandle(TokenHandle);
279 
280  } except(EXCEPTION_EXECUTE_HANDLER) {
281  return GetExceptionCode();
282  } // end_try
283 
284  } //end_if
285 
286 
287  //
288  // Validate access to the process and obtain a pointer to the
289  // process's token. If successful, this will cause the token's
290  // reference count to be incremented.
291  //
292 
293  Status = PsOpenTokenOfProcess( ProcessHandle, ((PACCESS_TOKEN *)&Token));
294 
295  if (!NT_SUCCESS(Status)) {
296  return Status;
297  }
298 
299  //
300  // Now try to open the token for the specified desired access
301  //
302 
303  Status = ObOpenObjectByPointer(
304  (PVOID)Token, // Object
305  HandleAttributes, // HandleAttributes
306  NULL, // AccessState
307  DesiredAccess, // DesiredAccess
308  SeTokenObjectType, // ObjectType
309  PreviousMode, // AccessMode
310  &LocalHandle // Handle
311  );
312 
313  //
314  // And decrement the reference count of the token to counter
315  // the action performed by PsOpenTokenOfProcess(). If the open
316  // was successful, the handle will have caused the token's
317  // reference count to have been incremented.
318  //
319 
320  ObDereferenceObject( Token );
321 
322  //
323  // Return the new handle
324  //
325 
326  if (NT_SUCCESS(Status)) {
327 
328  try {
329 
330  *TokenHandle = LocalHandle;
331 
332  } except(EXCEPTION_EXECUTE_HANDLER) {
333 
334  return GetExceptionCode();
335 
336  }
337  }
338 
339  return Status;
340 
341 }

343 NTSTATUS
344 SepOpenTokenOfThread(
345  __in HANDLE ThreadHandle,
346  __in BOOLEAN OpenAsSelf,
347  __deref_out PACCESS_TOKEN *Token,
348  __deref_out PETHREAD *Thread,
349  __out PBOOLEAN CopyOnOpen,
350  __out PBOOLEAN EffectiveOnly,
351  __out PSECURITY_IMPERSONATION_LEVEL ImpersonationLevel
352  )
353 
354 /*++
355 
356 Routine Description:
357 
358  This function does the thread specific processing of
359  an NtOpenThreadToken() service.
360 
361  The service validates that the handle has appropriate access
362  to reference the thread. If so, it goes on to increment
363  the reference count of the token object to prevent it from
364  going away while the rest of the NtOpenThreadToken() request
365  is processed.
366 
367  NOTE: If this call completes successfully, the caller is responsible
368  for decrementing the reference count of the target token.
369  This must be done using PsDereferenceImpersonationToken().
370 
371 Arguments:
372 
373  ThreadHandle - Supplies a handle to a thread object.
374 
375  OpenAsSelf - Is a boolean value indicating whether the access should
376  be made using the calling thread's current security context, which
377  may be that of a client (if impersonating), or using the caller's
378  process-level security context. A value of FALSE indicates the
379  caller's current context should be used un-modified. A value of
380  TRUE indicates the request should be fulfilled using the process
381  level security context.
382 
383  Token - If successful, receives a pointer to the thread's token
384  object.
385 
386  CopyOnOpen - The current value of the Thread->Client->CopyOnOpen field.
387 
388  EffectiveOnly - The current value of the Thread->Client->EffectiveOnly field.
389 
390  ImpersonationLevel - The current value of the Thread->Client->ImpersonationLevel
391  field.
392 
393 Return Value:
394 
395  STATUS_SUCCESS - Indicates the call completed successfully.
396 
397  STATUS_NO_TOKEN - Indicates the referenced thread is not currently
398  impersonating a client.
399 
400  STATUS_CANT_OPEN_ANONYMOUS - Indicates the client requested anonymous
401  impersonation level. An anonymous token can not be opened.
402 
403  status may also be any value returned by an attempt the reference
404  the thread object for THREAD_QUERY_INFORMATION access.
405 
406 --*/
407 
408 {
409 
410  NTSTATUS
411  Status;
412 
413  KPROCESSOR_MODE
414  PreviousMode;
415 
416  SE_IMPERSONATION_STATE
417  DisabledImpersonationState;
418 
419  BOOLEAN
420  RestoreImpersonationState = FALSE;
421 
422  PAGED_CODE();
423 
424  PreviousMode = KeGetPreviousMode();
425 
426 
427  //
428  // Make sure the handle grants the appropriate access to the specified
429  // thread.
430  //
431 
432  Status = ObReferenceObjectByHandle(
433  ThreadHandle,
434  THREAD_QUERY_INFORMATION,
435  PsThreadType,
436  PreviousMode,
437  (PVOID *)Thread,
438  NULL
439  );
440 
441  if (!NT_SUCCESS(Status)) {
442  return Status;
443  }
444 
445  //
446  // Reference the impersonation token, if there is one
447  //
448 
449  (*Token) = PsReferenceImpersonationToken( *Thread,
450  CopyOnOpen,
451  EffectiveOnly,
452  ImpersonationLevel
453  );
454 
455 
456 
457 
458  //
459  // Make sure there is a token
460  //
461 
462  if ((*Token) == NULL) {
463  ObDereferenceObject( *Thread );
464  (*Thread) = NULL;
465  return STATUS_NO_TOKEN;
466  }
467 
468 
469  //
470  // Make sure the ImpersonationLevel is high enough to allow
471  // the token to be opened.
472  //
473 
474  if ((*ImpersonationLevel) <= SecurityAnonymous) {
475  PsDereferenceImpersonationToken( (*Token) );
476  ObDereferenceObject( *Thread );
477  (*Thread) = NULL;
478  (*Token) = NULL;
479  return STATUS_CANT_OPEN_ANONYMOUS;
480  }
481 
482 
483  return STATUS_SUCCESS;
484 
485 }

550 NTSTATUS
551 NtOpenThreadTokenEx(
552  __in HANDLE ThreadHandle,
553  __in ACCESS_MASK DesiredAccess,
554  __in BOOLEAN OpenAsSelf,
555  __in ULONG HandleAttributes,
556  __out PHANDLE TokenHandle
557  )
558 
559 /*++
560 
561 
562 Routine Description:
563 
564 Open a token object associated with a thread and return a handle that
565 may be used to access that token.
566 
567 Arguments:
568 
569  ThreadHandle - Specifies the thread whose token is to be opened.
570 
571  DesiredAccess - Is an access mask indicating which access types
572  are desired to the token. These access types are reconciled
573  with the Discretionary Access Control list of the token to
574  determine whether the accesses will be granted or denied.
575 
576  OpenAsSelf - Is a boolean value indicating whether the access should
577  be made using the calling thread's current security context, which
578  may be that of a client if impersonating, or using the caller's
579  process-level security context. A value of FALSE indicates the
580  caller's current context should be used un-modified. A value of
581  TRUE indicates the request should be fulfilled using the process
582  level security context.
583 
584  This parameter is necessary to allow a server process to open
585  a client's token when the client specified IDENTIFICATION level
586  impersonation. In this case, the caller would not be able to
587  open the client's token using the client's context (because you
588  can't create executive level objects using IDENTIFICATION level
589  impersonation).
590 
591  HandleAttributes - Attributes applied to the handle OBJ_KERNEL_HANDLE
592 
593  TokenHandle - Receives the handle of the newly opened token.
594 
595 Return Value:
596 
597  STATUS_SUCCESS - Indicates the operation was successful.
598 
599  STATUS_NO_TOKEN - Indicates an attempt has been made to open a
600  token associated with a thread that is not currently
601  impersonating a client.
602 
603  STATUS_CANT_OPEN_ANONYMOUS - Indicates the client requested anonymous
604  impersonation level. An anonymous token can not be opened.
605 
606 --*/
607 {
608 
609  KPROCESSOR_MODE PreviousMode;
610  NTSTATUS Status;
611 
612  PVOID Token;
613  PTOKEN NewToken = NULL;
614  BOOLEAN CopyOnOpen;
615  BOOLEAN EffectiveOnly;
616  SECURITY_IMPERSONATION_LEVEL ImpersonationLevel;
617  SE_IMPERSONATION_STATE DisabledImpersonationState;
618  BOOLEAN RestoreImpersonationState = FALSE;
619 
620  HANDLE LocalHandle = NULL;
621  SECURITY_DESCRIPTOR SecurityDescriptor;
622  OBJECT_ATTRIBUTES ObjectAttributes;
623  PACL NewAcl = NULL;
624  PETHREAD Thread = NULL;
625  PETHREAD OriginalThread = NULL;
626  PACCESS_TOKEN PrimaryToken;
627  SECURITY_SUBJECT_CONTEXT SubjectSecurityContext;
628 
629  PAGED_CODE();
630 
631  PreviousMode = KeGetPreviousMode();
632 
633  //
634  // Sanitize the handle attribute flags
635  //
636 
637  HandleAttributes = ObSanitizeHandleAttributes (HandleAttributes, PreviousMode);
638 
639  //
640  // Probe parameters
641  //
642 
643  if (PreviousMode != KernelMode) {
644 
645  try {
646 
647  ProbeForWriteHandle(TokenHandle);
648 
649  } except(EXCEPTION_EXECUTE_HANDLER) {
650  return GetExceptionCode();
651  } // end_try
652 
653  } //end_if
654 
655  //
656  // Validiate access to the thread and obtain a pointer to the
657  // thread's token (if there is one). If successful, this will
658  // cause the token's reference count to be incremented.
659  //
660  // This routine disabled impersonation as necessary to properly
661  // honor the OpenAsSelf flag.
662  //
663 
664  Status = SepOpenTokenOfThread( ThreadHandle,
665  OpenAsSelf,
666  ((PACCESS_TOKEN *)&Token),
667  &OriginalThread,
668  &CopyOnOpen,
669  &EffectiveOnly,
670  &ImpersonationLevel
671  );
672 
673  if (!NT_SUCCESS(Status)) {
674  return Status;
675  }
676 
677 
678  //
679  // The token was successfully referenced.
680  //
681 
682  //
683  // We need to create and/or open a token object, so disable impersonation
684  // if necessary.
685  //
686 
687  if (OpenAsSelf) {
688  RestoreImpersonationState = PsDisableImpersonation(
689  PsGetCurrentThread(),
690  &DisabledImpersonationState
691  );
692  }
693 
694  //
695  // If the CopyOnOpen flag is not set, then the token can be
696  // opened directly. Otherwise, the token must be duplicated,
697  // and a handle to the duplicate returned.
698  //
699 
700  if (CopyOnOpen) {
701 
702  //
703  // Create the new security descriptor for the token.
704  //
705  // We must obtain the correct SID to put into the Dacl. Do this
706  // by finding the process associated with the passed thread
707  // and grabbing the User SID out of that process's token.
708  // If we just use the current SubjectContext, we'll get the
709  // SID of whoever is calling us, which isn't what we want.
710  //
711 
712  Status = ObReferenceObjectByHandle(
713  ThreadHandle,
714  THREAD_ALL_ACCESS,
715  PsThreadType,
716  KernelMode,
717  (PVOID)&Thread,
718  NULL
719  );
720 
721  //
722  // Verify that the handle is still pointer to the same thread
723  //
724 
725  if (NT_SUCCESS(Status) && (Thread != OriginalThread)) {
726  Status = STATUS_OBJECT_TYPE_MISMATCH;
727  }
728 
729  if (NT_SUCCESS(Status)) {
730  PEPROCESS Process = THREAD_TO_PROCESS(Thread);
731 
732  PrimaryToken = PsReferencePrimaryToken(Process);
733 
734  Status = SepCreateImpersonationTokenDacl(
735  (PTOKEN)Token,
736  PrimaryToken,
737  &NewAcl
738  );
739 
740  PsDereferencePrimaryTokenEx( Process, PrimaryToken );
741 
742  if (NT_SUCCESS( Status )) {
743 
744  if (NewAcl != NULL) {
745 
746  //
747  // There exist tokens that either do not have security descriptors at all,
748  // or have security descriptors, but do not have DACLs. In either case, do
749  // nothing.
750  //
751 
752  Status = RtlCreateSecurityDescriptor ( &SecurityDescriptor, SECURITY_DESCRIPTOR_REVISION );
753  ASSERT( NT_SUCCESS( Status ));
754 
755  Status = RtlSetDaclSecurityDescriptor (
756  &SecurityDescriptor,
757  TRUE,
758  NewAcl,
759  FALSE
760  );
761 
762  ASSERT( NT_SUCCESS( Status ));
763  }
764 
765  InitializeObjectAttributes(
766  &ObjectAttributes,
767  NULL,
768  HandleAttributes,
769  NULL,
770  NewAcl == NULL ? NULL : &SecurityDescriptor
771  );
772 
773  //
774  // Open a copy of the token
775  //
776 
777  Status = SepDuplicateToken(
778  (PTOKEN)Token, // ExistingToken
779  &ObjectAttributes, // ObjectAttributes
780  EffectiveOnly, // EffectiveOnly
781  TokenImpersonation, // TokenType
782  ImpersonationLevel, // ImpersonationLevel
783  KernelMode, // RequestorMode must be kernel mode
784  &NewToken
785  );
786 
787  if (NT_SUCCESS( Status )) {
788 
789  //
790  // Reference the token so it doesn't go away
791  //
792 
793  ObReferenceObject(NewToken);
794 
795  //
796  // Insert the new token
797  //
798 
799  Status = ObInsertObject( NewToken,
800  NULL,
801  DesiredAccess,
802  0,
803  (PVOID *)NULL,
804  &LocalHandle
805  );
806  }
807  }
808  }
809 
810 
811  } else {
812 
813  //
814  // We do not have to modify the security on the token in the static case,
815  // because in all the places in the system where impersonation takes place
816  // over a secure transport (e.g., LPC), CopyOnOpen is set. The only reason
817  // we'be be here is if the impersonation is taking place because someone did
818  // an NtSetInformationThread and passed in a token.
819  //
820  // In that case, we absolutely do not want to give the caller guaranteed
821  // access, because that would allow anyone who has access to a thread to
822  // impersonate any of that thread's clients for any access.
823  //
824 
825  //
826  // Open the existing token
827  //
828 
829  Status = ObOpenObjectByPointer(
830  (PVOID)Token, // Object
831  HandleAttributes, // HandleAttributes
832  NULL, // AccessState
833  DesiredAccess, // DesiredAccess
834  SeTokenObjectType, // ObjectType
835  PreviousMode, // AccessMode
836  &LocalHandle // Handle
837  );
838  }
839 
840  if (NewAcl != NULL) {
841  ExFreePool( NewAcl );
842  }
843 
844  if (RestoreImpersonationState) {
845  PsRestoreImpersonation(
846  PsGetCurrentThread(),
847  &DisabledImpersonationState
848  );
849  }
850 
851  //
852  // And decrement the reference count of the existing token to counter
853  // the action performed by PsOpenTokenOfThread. If the open
854  // was successful, the handle will have caused the token's
855  // reference count to have been incremented.
856  //
857 
858  ObDereferenceObject( Token );
859 
860  if (NT_SUCCESS( Status ) && CopyOnOpen) {
861 
862  //
863  // Assign the newly duplicated token to the thread.
864  //
865 
866  PsImpersonateClient( Thread,
867  NewToken,
868  FALSE, // turn off CopyOnOpen flag
869  EffectiveOnly,
870  ImpersonationLevel
871  );
872 
873  }
874 
875  //
876  // We've impersonated the token so let go of our reference
877  //
878 
879  if (NewToken != NULL) {
880  ObDereferenceObject( NewToken );
