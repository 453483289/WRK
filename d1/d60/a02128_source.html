9 Module Name:
10 
11  pushlock.c
12 
13 Abstract:
14 
15  This module houses routines that do push locking.
16 
17  Push locks are capable of being acquired in both shared and exclusive mode.
18 
19  Properties include:
20 
21  They cannot be acquired recursively.
22  They are small (the size of a pointer) and can be used when embedded in pageable data.
23  Acquire and release is done lock free. On lock contention the waiters are chained
24  through the lock and local stack space.
25 
26  This is the structure of a push lock:
27 
28 
29  L == Lock bit, This bit is set when the lock is held exclusive or shared
30  W == Waiters present
31  K == Waking threads or optimizing chains. This bit can't be set unless the waiters bit is set.
32  M == Multiple shared owners present. When they release these threads must walk the waitblock list
33  SC == Share count
34  P == Pointer to wait block
35 
36  +--------------+---+---+---+---+
37  | SC | M | K | W | L | L, W, K and M are single bits W == 0
38  +--------------+---+---+---+---+
39 
40  +--------------+---+---+---+---+
41  | P | M | K | W | L | W == 1. Pointer is the address of a chain of stack local wait blocks
42  +--------------+---+---+---+---+
43 
44  The non-contented acquires and releases are trivial. Interlocked operations make the following
45  transformations.
46 
47  Uncontended exclusive acquire. Notice that we can acquire even if there are waiters.
48 
49  (SC=0,L=0,W=w,K=k,M=0) === Exclusive acquire ===> (SC=0,L=1,W=w,K=k,M=0)
50 
51  There are two cases for shared acquire. Notice that we can only acquire
52  in the absence of waiters. The first and later acquires:
53 
54  (SC=0,L=0,W=0,K=0,M=0) === First shared acquire ===> (SC=1, L=1,W=0,K=0,M=0)
55  (SC=n,L=1,W=0,K=0,M=0) === later shared acquire ===> (SC=n+1,L=1,W=0,K=0,M=0)
56 
57  Uncontended exclusive release:
58 
59  (SC=0,L=1,W=0,K=0,M=0) === Exclusive release ===> (SC=0,L=0,W=0,K=0,M=0)
60 
61  There are two cases for uncontended shared release. Last and non-last:
62 
63  (SC=n,L=1,W=0,K=0,M=0) === Non-last Shared release ===> (SC=n-1,L=1,W=0,K=0,M=0) n > 0
64  (SC=1,L=1,W=0,K=0,M=0) === last Shared release ===> (SC=0, L=0,W=0,K=0,M=0)
65 
66  Contention causes the acquiring thread to produce a local stack based wait block and to
67  enqueue it to the front of the list.
68 
69  Notice that after a contended acquire takes place the K bit is always set.
70  It was either set by another thread or this thread. The thread that sets this bit has
71  the right to walk the waiters chain and optimize it (place pointers to ending blocks etc).
72 
73  Owners of the K bit must either clear the K bit if the lock is still locked or clear the K bit and
74  wake at least one thread (if the lock is not locked).
75 
76  (SC=n,L=1,W=0,K=0,M=0) === Exclusive acquire ===> (P=LWB(SSC=n,E=e),W=1,K=1,M=m)
77  LWB = local wait block,
78  SSC = Saved share count,
79  E = Exclusive waiter flag
80  m = Multiple shared owners. m=(n>1)?1:0
81 
82  (SC=0,L=1,W=0,K=0,M=0) === Shared acquire ===> (P=LWB(SSC=0,E=0),W=1,K=1,M=0)
83  LWB = local wait block,
84  SSC = Saved share count.
85  E = Exclusive waiter flag
86 
87  After contention has caused one or more threads to queue up wait blocks releases are more
88  complicated. The K bit arbitrates who is allowed to manipulate the waiters list but it does
89  not guaranteed complete freedom. The following are the rights to manipulate the list:
90 
91  1) New waiters at any time may push themselves onto the list. If they are not the first
92  waiter they will also attempt to set the K bit to perform list optimization.
93  2) Shared releasers are allowed to walk the list (hopefully optimized by the waiters) to
94  find the end of the chain where the share count lives. If they decrement the shared count
95  to zero they behave like exclusive releasers and try to set the K bit.
96  3) K bit owners, either waiters that pend and manage to set the bit or releasers who manage
97  to set the bit have the right to walk the chain and break it and wake. They can't reorder
98  without excluding shared walkers (we don't attempt to reorder just put in end pointers).
99 
100  The K bit is not a lock in the sense that we don't need to spin on it. Only one owner is needed
101  at any time and failure to obtain K bit ownership means some other thread will wake etc.
102 
103  The M bit is needed for lock releasers who don't know if they own the lock shared. It also has the
104  benefit that only when there are more than one shared owner do we need to traverse the wait list.
105 
106 
107 Revision History:
108 
109  Shifted to not pass on ownership in contended cases to reduce lock hold times.
110 
111 --*/

209 ExfWakePushLock (
210  IN PEX_PUSH_LOCK PushLock,
211  IN EX_PUSH_LOCK TopValue
212  )
213 /*++
214 
215 Routine Description:
216 
217  Walks the pushlock waiting list and wakes waiters if the lock is still unacquired.
218 
219 Arguments:
220 
221  PushLock - Push lock to be walked
222 
223  TopValue - Start of the chain (*PushLock)
224 
225 Return Value:
226 
227  None
228 
229 --*/
230 {
231  EX_PUSH_LOCK OldValue, NewValue;
232  PEX_PUSH_LOCK_WAIT_BLOCK WaitBlock, NextWaitBlock, FirstWaitBlock, PreviousWaitBlock;
233  KIRQL OldIrql;
234 
235  OldValue = TopValue;
236 
237  while (1) {
238 
239  //
240  // Nobody should be walking the list while we manipulate it.
241  //
242 
243  ASSERT (!OldValue.MultipleShared);
244 
245  //
246  // No point waking somebody to find a locked lock. Just clear the waking bit
247  //
248 
249  while (OldValue.Locked) {
250  NewValue.Value = OldValue.Value - EX_PUSH_LOCK_WAKING;
251  ASSERT (!NewValue.Waking);
252  ASSERT (NewValue.Locked);
253  ASSERT (NewValue.Waiting);
254  if ((NewValue.Ptr = InterlockedCompareExchangePointer (&PushLock->Ptr,
255  NewValue.Ptr,
256  OldValue.Ptr)) == OldValue.Ptr) {
257  return;
258  }
259  OldValue = NewValue;
260  }
261 
262  WaitBlock = (PEX_PUSH_LOCK_WAIT_BLOCK)
263  (OldValue.Value & ~(ULONG_PTR)EX_PUSH_LOCK_PTR_BITS);
264 
265  FirstWaitBlock = WaitBlock;
266 
267  while (1) {
268 
269  NextWaitBlock = WaitBlock->Last;
270  if (NextWaitBlock != NULL) {
271  WaitBlock = NextWaitBlock;
272  break;
273  }
274 
275  PreviousWaitBlock = WaitBlock;
276  WaitBlock = WaitBlock->Next;
277  WaitBlock->Previous = PreviousWaitBlock;
278  }
279 
280  if (WaitBlock->Flags&EX_PUSH_LOCK_FLAGS_EXCLUSIVE &&
281  (PreviousWaitBlock = WaitBlock->Previous) != NULL) {
282 
283  FirstWaitBlock->Last = PreviousWaitBlock;
284 
285  WaitBlock->Previous = NULL;
286 
287  ASSERT (FirstWaitBlock != WaitBlock);
288 
289  ASSERT (PushLock->Waiting);
290 
291 #if defined (_WIN64)
292  InterlockedAnd64 ((LONG64 *)&PushLock->Value, ~EX_PUSH_LOCK_WAKING);
293 #else
294  InterlockedAnd ((LONG *)&PushLock->Value, ~EX_PUSH_LOCK_WAKING);
295 #endif
296 
297  break;
298  } else {
299  NewValue.Value = 0;
300  ASSERT (!NewValue.Waking);
301  if ((NewValue.Ptr = InterlockedCompareExchangePointer (&PushLock->Ptr,
302  NewValue.Ptr,
303  OldValue.Ptr)) == OldValue.Ptr) {
304  break;
305  }
306  OldValue = NewValue;
307  }
308  }
309 
310  //
311  // If we are waking more than one thread then raise to DPC level to prevent us
312  // getting rescheduled part way through the operation
313  //
314 
315  OldIrql = DISPATCH_LEVEL;
316  if (WaitBlock->Previous != NULL) {
317  KeRaiseIrql (DISPATCH_LEVEL, &OldIrql);
318  }
319 
320  while (1) {
321 
322  NextWaitBlock = WaitBlock->Previous;
323 #if DBG
324  ASSERT (!WaitBlock->Signaled);
325  WaitBlock->Signaled = TRUE;
326 #endif
327 
328  if (!InterlockedBitTestAndReset (&WaitBlock->Flags, EX_PUSH_LOCK_FLAGS_SPINNING_V)) {
329  KeSignalGateBoostPriority (&WaitBlock->WakeGate);
330  }
331 
332  WaitBlock = NextWaitBlock;
333  if (WaitBlock == NULL) {
334  break;
335  }
336  }
337 
338  if (OldIrql != DISPATCH_LEVEL) {
339  KeLowerIrql (OldIrql);
340  }
341 }

388 /*++
389 
390 Routine Description:
391 
392  Walks the pushlock waiting list during contention and optimizes it
393 
394 Arguments:
395 
396  PushLock - Push lock to be walked
397 
398  TopValue - Start of the chain (*PushLock)
399 
400 Return Value:
401 
402  None
403 
404 --*/
405 {
406  EX_PUSH_LOCK OldValue, NewValue;
407  PEX_PUSH_LOCK_WAIT_BLOCK WaitBlock, PreviousWaitBlock, FirstWaitBlock, NextWaitBlock;
408 
409  OldValue = TopValue;
410  while (1) {
411  if (!OldValue.Locked) {
412  ExfWakePushLock (PushLock, OldValue);
413  break;
414  }
415 
416  WaitBlock = (PEX_PUSH_LOCK_WAIT_BLOCK)(OldValue.Value & ~(EX_PUSH_LOCK_PTR_BITS));
417 
418  FirstWaitBlock = WaitBlock;
419 
420  while (1) {
421 
422  NextWaitBlock = WaitBlock->Last;
423  if (NextWaitBlock != NULL) {
424  FirstWaitBlock->Last = NextWaitBlock;
425  break;
426  }
427 
428  PreviousWaitBlock = WaitBlock;
429  WaitBlock = WaitBlock->Next;
430  WaitBlock->Previous = PreviousWaitBlock;
431  }
432 
433  NewValue.Value = OldValue.Value - EX_PUSH_LOCK_WAKING;
434  ASSERT (NewValue.Locked);
435  ASSERT (!NewValue.Waking);
436  if ((NewValue.Ptr = InterlockedCompareExchangePointer (&PushLock->Ptr,
437  NewValue.Ptr,
438  OldValue.Ptr)) == OldValue.Ptr) {
439  break;
440  }
441  OldValue = NewValue;
442  }

453 /*++
454 
455 Routine Description:
456 
457  Acquire a push lock exclusively
458 
459 Arguments:
460 
461  PushLock - Push lock to be acquired
462 
463 Return Value:
464 
465  None
466 
467 --*/
468 {
469  EX_PUSH_LOCK OldValue, NewValue, TopValue;
470  EX_PUSH_LOCK_WAIT_BLOCK WaitBlock;
471  BOOLEAN Optimize;
472 #if defined (USE_EXP_BACKOFF)
473  RTL_BACKOFF Backoff = {0};
474 #endif
475 
476  OldValue = ReadForWriteAccess (PushLock);
477 
478  while (1) {
479  //
480  // If the lock is already held exclusively/shared or there are waiters then
481  // we need to wait.
482  //
483  if ((OldValue.Value&EX_PUSH_LOCK_LOCK) == 0) {
484  NewValue.Value = OldValue.Value + EX_PUSH_LOCK_LOCK;
485 
486  ASSERT (NewValue.Locked);
487 
488  NewValue.Ptr = InterlockedCompareExchangePointer (&PushLock->Ptr,
489  NewValue.Ptr,
490  OldValue.Ptr);
491  if (NewValue.Ptr == OldValue.Ptr) {
492  break;
493  }
494 
495 #if defined (USE_EXP_BACKOFF)
496 
497  //
498  // Use backof to limit memory bandwidth
499  //
500  RtlBackoff (&Backoff);
501 
502  NewValue = *PushLock;
503 
504 #endif
505  } else {
506  WaitBlock.Flags = EX_PUSH_LOCK_FLAGS_EXCLUSIVE | EX_PUSH_LOCK_FLAGS_SPINNING;
507  WaitBlock.Previous = NULL;
508  Optimize = FALSE;
509 
510  //
511  // Move the sharecount to our wait block if need be.
512  //
513  if (OldValue.Waiting) {
514  WaitBlock.Last = NULL;
515  WaitBlock.Next = (PEX_PUSH_LOCK_WAIT_BLOCK)
516  (OldValue.Value & ~EX_PUSH_LOCK_PTR_BITS);
517  WaitBlock.ShareCount = 0;
518  NewValue.Ptr = (PVOID)(((ULONG_PTR) &WaitBlock) |
519  (OldValue.Value & EX_PUSH_LOCK_PTR_BITS) |
520  EX_PUSH_LOCK_WAITING | EX_PUSH_LOCK_WAKING |
521  EX_PUSH_LOCK_LOCK);
522  if (!OldValue.Waking) {
523  Optimize = TRUE;
524  }
525  } else {
526  WaitBlock.Last = &WaitBlock;
527  WaitBlock.ShareCount = (ULONG) OldValue.Shared;
528  if (WaitBlock.ShareCount > 1) {
529  NewValue.Ptr = (PVOID)(((ULONG_PTR) &WaitBlock) | EX_PUSH_LOCK_WAITING |
530  EX_PUSH_LOCK_LOCK |
531  EX_PUSH_LOCK_MULTIPLE_SHARED);
532  } else {
533  WaitBlock.ShareCount = 0;
534  NewValue.Ptr = (PVOID)(((ULONG_PTR) &WaitBlock) | EX_PUSH_LOCK_WAITING |
535  EX_PUSH_LOCK_LOCK);
536  }
537  }
538 
539 #if DBG
540  WaitBlock.Signaled = FALSE;
541  WaitBlock.OldValue = OldValue.Ptr;
542  WaitBlock.NewValue = NewValue.Ptr;
543  WaitBlock.PushLock = PushLock;
544 #endif
545  ASSERT (NewValue.Waiting);
546  ASSERT (OldValue.Locked);
547  ASSERT (NewValue.Locked);
548 
549  TopValue = NewValue;
550  NewValue.Ptr = InterlockedCompareExchangePointer (&PushLock->Ptr,
551  NewValue.Ptr,
552  OldValue.Ptr);
553  if (NewValue.Ptr == OldValue.Ptr) {
554  ULONG i;
555 
556  //
557  // If we set the waiting bit then optimize the list
558  //
559  if (Optimize) {
560  ExpOptimizePushLockList (PushLock, TopValue);
561  }
562 
563  KeInitializeGate (&WaitBlock.WakeGate);
564 
565  for (i = ExPushLockSpinCount; i > 0; i--) {
566  if (((*(volatile LONG *)&WaitBlock.Flags)&EX_PUSH_LOCK_FLAGS_SPINNING) == 0) {
567  break;
568  }
569  KeYieldProcessor ();
570  }
571 
572  if (InterlockedBitTestAndReset (&WaitBlock.Flags, EX_PUSH_LOCK_FLAGS_SPINNING_V)) {
573 
574  KeWaitForGate (&WaitBlock.WakeGate, WrPushLock, KernelMode);
575 #if DBG
576  ASSERT (WaitBlock.Signaled);
577 #endif
578 
579  }
580  ASSERT ((WaitBlock.ShareCount == 0));
581  } else {
582 
583 #if defined (USE_EXP_BACKOFF)
584  //
585  // Use backof to limit memory bandwidth
586  //
587  RtlBackoff (&Backoff);
588 
589  NewValue = *PushLock;
590 #endif
591  }
592  }
593  OldValue = NewValue;
594  }

646  break;
647  }
648

650 
651  //
652  // Use backof to limit memory bandwidth
653  //
654  RtlBackoff (&Backoff);
655 
656  NewValue = *PushLock;
657 
658 #endif
659 
660  } else {
661  WaitBlock.Flags = EX_PUSH_LOCK_FLAGS_SPINNING;
662  WaitBlock.ShareCount = 0;
663  Optimize = FALSE;
664  WaitBlock.Previous = NULL;
665 
666  //
667  // Move the sharecount to our wait block if need be.
668  //

670  WaitBlock.Last = NULL;
671  WaitBlock.Next = (PEX_PUSH_LOCK_WAIT_BLOCK)

674  (OldValue.Value & (EX_PUSH_LOCK_LOCK | EX_PUSH_LOCK_MULTIPLE_SHARED)) |
675  EX_PUSH_LOCK_WAITING | EX_PUSH_LOCK_WAKING);
676  if (!OldValue.Waking) {
677  Optimize = TRUE;
678  }
679  } else {
680  WaitBlock.Last = &WaitBlock;
681  NewValue.Ptr = (PVOID)(((ULONG_PTR) &WaitBlock) |
682  (OldValue.Value & EX_PUSH_LOCK_PTR_BITS) |
683  EX_PUSH_LOCK_WAITING);
684  }
685 
686  ASSERT (NewValue.Waiting);
687 
688 #if DBG
689  WaitBlock.Signaled = FALSE;
690  WaitBlock.OldValue = OldValue.Ptr;
691  WaitBlock.NewValue = NewValue.Ptr;
692  WaitBlock.PushLock = PushLock;
693 #endif
694  TopValue = NewValue;
695  NewValue.Ptr = InterlockedCompareExchangePointer (&PushLock->Ptr,
696  NewValue.Ptr,
697  OldValue.Ptr);
698 
699  if (NewValue.Ptr == OldValue.Ptr) {

701

703  // If we set the waiting bit then optimize the list
704  //
705  if (Optimize) {

711  // a gate signal to always follow gate initialization.
712  //

714

727 #endif

730

732

737  //

761
