314 /*++
315 
316 Routine Description:
317 
318  This function is called as part of phase1 init, after the object
319  manager has been initialized, but before IoInit. It's purpose is to
320  set up basic registry object operations, and transform data
321  captured during boot into registry format (whether it was read
322  from the SYSTEM hive file by the osloader or computed by recognizers.)
323  After this call, Nt*Key calls work, but only part of the name
324  space is available and any changes written must be held in
325  memory.
326 
327  CmpMachineHiveList entries marked CM_PHASE_1 are available
328  after return from this call, but writes must be held in memory.
329 
330  This function will:
331 
332  1. Create the registry worker/lazy-write thread
333  2. Create the registry key object type
334  4. Create the master hive
335  5. Create the \REGISTRY node
336  6. Create a KEY object that refers to \REGISTRY
337  7. Create \REGISTRY\MACHINE node
338  8. Create the SYSTEM hive, fill in with data from loader
339  9. Create the HARDWARE hive, fill in with data from loader
340  10. Create:
341  \REGISTRY\MACHINE\SYSTEM
342  \REGISTRY\MACHINE\HARDWARE
343  Both of which will be link nodes in the master hive.
344 
345  NOTE: We do NOT free allocated pool in failure case. This is because
346  our caller is going to bugcheck anyway, and having the memory
347  object to look at is useful.
348 
349 Arguments:
350 
351  LoaderBlock - supplies the LoaderBlock passed in from the OSLoader.
352  By looking through the memory descriptor list we can find the
353  SYSTEM hive which the OSLoader has placed in memory for us.
354 
355 Return Value:
356 
357  TRUE if all operations were successful, false if any failed.
358 
359  Bugchecks when something went wrong (CONFIG_INITIALIZATION_FAILED,INIT_SYSTEM1,.....)
360 
361 --*/

371  // environment
372  //

375 
376  //
377  // On Remote boot client share the system hives
378  //
379  // NOTE : We can't assume exclusive access to WinPE
380  // remote boot clients. We don't flush anything to
381  // system hives in WinPE. All the system hives are
382  // loaded in memory in scratch mode
383  //

673  }

1397 
1398 Routine Description:
1399 
1400  This routine guards calling the InterlockedFunction in the
1401  passed RegistryValueKey.
1402

1404  call to the InterlockedFunction. If the system crashes
1405  during this call then ValueKey will be left in a state
1406  where the InterlockedFunction will not be called on subsequent
1407  attempts.
1408

1410 
1411  RegistryValueKey - ValueKey name for Control\Session Manager
1412  InterlockedFunction - Function to call
1413

1415 
1416  STATUS_SUCCESS - The interlocked function was successfully called
1417 
1418 
1419 --*/
1420 {
1421  OBJECT_ATTRIBUTES objectAttributes;
1422  HANDLE hControl, hSession;
1423  UNICODE_STRING Name;
1424  UCHAR Buffer [sizeof(KEY_VALUE_PARTIAL_INFORMATION)+sizeof(ULONG)];
1425  ULONG length, Value;
1426  NTSTATUS status;
1427 
1428  CM_PAGED_CODE();
1429 
1430  //
1431  // Open CurrentControlSet
1432  //
1433 
1434  InitializeObjectAttributes (
1435  &objectAttributes,
1436  &CmRegistryMachineSystemCurrentControlSet,
1437  OBJ_CASE_INSENSITIVE,
1438  NULL,
1439  NULL
1440  );
1441 
1442  status = NtOpenKey (&hControl, KEY_READ | KEY_WRITE, &objectAttributes);
1443  if (!NT_SUCCESS(status)) {
1444  return status;
1445  }
1446 
1447  //
1448  // Open Control\Session Manager
1449  //
1450 
1451  RtlInitUnicodeString (&Name, CmpControlSessionManager);
1452  InitializeObjectAttributes (
1453  &objectAttributes,
1454  &Name,
1455  OBJ_CASE_INSENSITIVE,
1456  hControl,
1457  NULL
1458  );
1459 
1460  status = NtOpenKey (&hSession, KEY_READ | KEY_WRITE, &objectAttributes );
1461  NtClose (hControl);
1462  if (!NT_SUCCESS(status)) {
1463  return status;
1464  }
1465 
1466  //
1467  // Read ValueKey to interlock operation with
1468  //
1469 
1470  RtlInitUnicodeString (&Name, RegistryValueKey);
1471  status = NtQueryValueKey (hSession,
1472  &Name,
1473  KeyValuePartialInformation,
1474  Buffer,
1475  sizeof (Buffer),
1476  &length );
1477 
1478  Value = 0;
1479  if (NT_SUCCESS(status)) {
1480  Value = ((PKEY_VALUE_PARTIAL_INFORMATION)Buffer)->Data[0];
1481  }
1482 
1483  //
1484  // Value 0 - Before InterlockedFunction
1485  // 1 - In the middle of InterlockedFunction
1486  // 2 - After InterlockedFunction

1489  // interlocked function, set the value to a 1 and try it.
1490  //

1496  //

1499 
1500  if (Value != 2) {

1503  // a 1 to this value so we can detect if we crash during
1504  // this call.
1505  //

1509  NtFlushKey (hSession); // wait until it's on the disk
1510  }

1515  //
1516  // The worker function didn't crash - update the value for
1517  // this interlocked function to 2.
1518  //
1519
