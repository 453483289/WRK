189 void WmipDSCleanup(
190  PCHUNKINFO ChunkInfo,
191  PENTRYHEADER Entry
192  )
193 /*++
194 
195 Routine Description:
196 
197  Cleans up data source structure and any other structures or handles
198  associated with it.
199 
200 Arguments:
201 
202  Data source structure to free
203 
204 Return Value:
205 
206 --*/
207 {
208  PBDATASOURCE DataSource = (PBDATASOURCE)Entry;
209  PBINSTANCESET InstanceSet;
210  PLIST_ENTRY InstanceSetList;
211  ULONG i;
212 
213  UNREFERENCED_PARAMETER (ChunkInfo);
214 
215  WmipAssert(DataSource != NULL);
216  WmipAssert(DataSource->Flags & FLAG_ENTRY_INVALID);
217 
218  WmipEnterSMCritSection();
219 
220  InstanceSetList = DataSource->ISHead.Flink;
221  while (InstanceSetList != &DataSource->ISHead)
222  {
223  InstanceSet = CONTAINING_RECORD(InstanceSetList,
224  INSTANCESET,
225  DSISList);
226 
227  if (InstanceSet->GuidISList.Flink != NULL)
228  {
229  RemoveEntryList(&InstanceSet->GuidISList);
230  InstanceSet->DataSource = NULL;
231  InstanceSet->GuidEntry->ISCount--;
232  }
233 
234  if ((InstanceSet->GuidEntry != NULL) &&
235  (! (InstanceSet->Flags & IS_NEWLY_REGISTERED)))
236  {
237 
238  if (IsEqualGUID(&InstanceSet->GuidEntry->Guid,
239  &WmipBinaryMofGuid))
240  {
241  WmipLeaveSMCritSection();
242 
243  WmipGenerateBinaryMofNotification(InstanceSet,
244  &GUID_MOF_RESOURCE_REMOVED_NOTIFICATION);
245 
246  WmipEnterSMCritSection();
247  }
248 
249  WmipUnreferenceGE(InstanceSet->GuidEntry);
250  }
251  InstanceSet->GuidEntry = NULL;
252 
253  InstanceSetList = InstanceSetList->Flink;
254 
255  WmipUnreferenceIS(InstanceSet);
256  }
257 
258  WmipLeaveSMCritSection();
259 
260  for (i = 0; i < DataSource->MofResourceCount; i++)
261  {
262  if (DataSource->MofResources[i] != NULL)
263  {
264  WmipUnreferenceMR(DataSource->MofResources[i]);
265  }
266  }
267 
268  if (DataSource->MofResources != DataSource->StaticMofResources)
269  {
270  WmipFree(DataSource->MofResources);
271  DataSource->MofResources = NULL;
272  }
273 
274  if (DataSource->RegistryPath != NULL)
275  {
276  WmipFree(DataSource->RegistryPath);
277  DataSource->RegistryPath = NULL;
278  }

661  while (InstanceSetList != &GuidEntry->ISHead)
662  {
663  InstanceSet = CONTAINING_RECORD(InstanceSetList,
664  INSTANCESET,
665  GuidISList);

686  WmipAssert(InstanceSet->Signature == IS_SIGNATURE);
687  WmipLeaveSMCritSection();
