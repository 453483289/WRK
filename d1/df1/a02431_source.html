33 /*++
34 
35 Routine Description:
36 
37  This routine returns the contents of an entry in the GDT. If the
38  entry is thread specific, the entry for the specified thread is
39  created and returned (KGDT_LDT, and KGDT_R3_TEB). If the selector
40  is processor dependent, the entry for the current processor is
41  returned (KGDT_R0_PCR).
42 
43 Arguments:
44 
45  Thread -- Supplies a pointer to the thread to return the entry for.
46 
47  Offset -- Supplies the offset in the Gdt. This value must be 0
48  mod 8.
49 
50  Descriptor -- Returns the contents of the Gdt descriptor
51 
52 Return Value:
53 
54  None.
55 
56 --*/
57 
58 {
59  PKGDTENTRY Gdt;
60  PKPROCESS Process;
61 
62  //
63  // If the entry is out of range, don't return anything
64  //
65 
66  if (Offset >= KGDT_NUMBER * sizeof(KGDTENTRY)) {
67  return ;
68  }
69 
70  if (Offset == KGDT_LDT) {
71 
72  //
73  // Materialize Ldt selector
74  //
75 
76  Process = Thread->Process;
77  RtlCopyMemory( Descriptor,
78  &(Process->LdtDescriptor),
79  sizeof(KGDTENTRY)
80  );
81 
82  } else {
83 
84  //
85  // Copy Selector from Ldt
86  //
87  // N.B. We will change the base later, if it is KGDT_R3_TEB
88  //
89 
90 
91  Gdt = KiPcr()->GDT;
92 
93  RtlCopyMemory(Descriptor, (PCHAR)Gdt + Offset, sizeof(KGDTENTRY));
94 
95  //
96  // if it is the TEB selector, fix the base
97  //
98 
99  if (Offset == KGDT_R3_TEB) {
100  Descriptor->BaseLow = (USHORT)((ULONG)(Thread->Teb) & 0xFFFF);
101  Descriptor->HighWord.Bytes.BaseMid =
102  (UCHAR) ( ( (ULONG)(Thread->Teb) & 0xFF0000L) >> 16);
103  Descriptor->HighWord.Bytes.BaseHi =
104  (CHAR) ( ( (ULONG)(Thread->Teb) & 0xFF000000L) >> 24);
105  }
106  }
107 
108  return ;
109 }
