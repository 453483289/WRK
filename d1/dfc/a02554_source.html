111 /*++
112 
113 Routine Description:
114 
115  This function allows safe cross process handle table references. Table deletion
116  at process exit waits for all outstanding references to finish. Once the table
117  is marked deleted further references are denied byt this function.
118 
119 Arguments:
120 
121  SourceProcesss - Process whose handle table is to be referenced
122 
123 Return Value:
124

127 
128 --*/
129 
130 {
131 
132  PHANDLE_TABLE ObjectTable;
133 
134  ObjectTable = NULL;
135 
136  if (ExAcquireRundownProtection (&SourceProcess->RundownProtect)) {
137 
138  ObjectTable = SourceProcess->ObjectTable;
139 
140  if (ObjectTable == NULL) {
141 
142  ExReleaseRundownProtection (&SourceProcess->RundownProtect);
143  }
144  }
145

220 /*++
221 
222 Routine Description:
223 
224  This function creates a handle that is a duplicate of the specified
225  source handle. The source handle is evaluated in the context of the
226  specified source process. The calling process must have
227  PROCESS_DUP_HANDLE access to the source process. The duplicate
228  handle is created with the specified attributes and desired access.
229  The duplicate handle is created in the handle table of the specified
230  target process. The calling process must have PROCESS_DUP_HANDLE
231  access to the target process.
232 
233 Arguments:
234 
235  SourceProcess - Supplies a pointer to the source process for the
236  handle being duplicated
237 
238  SourceHandle - Supplies the handle being duplicated
239 
240  TargetProcess - Optionally supplies a pointer to the target process
241  that is to receive the new handle
242 
243  TargetHandle - Optionally returns a the new duplicated handle
244 
245  DesiredAccess - Desired access for the new handle
246 
247  HandleAttributes - Desired attributes for the new handle
248 
249  Options - Duplication options that control things like close source,
250  same access, and same attributes.
251 
252  PreviousMode - Mode of caller
253 
254 Return Value:
255 
256  NTSTATUS - Status pf call
257 
258 --*/

260  NTSTATUS Status;
261  PVOID SourceObject;
262  POBJECT_HEADER ObjectHeader;
263  POBJECT_TYPE ObjectType;
264  BOOLEAN Attached;

283  // If the caller is not asking for the same access then
284  // validate the access they are requesting doesn't contain
285  // any bad bits
286  //

289 
290  Status = ObpValidateDesiredAccess( DesiredAccess );
291 
292  if (!NT_SUCCESS( Status )) {
293 
294  return( Status );
295  }

309  //

314  //

317

322  // The the input source handle get a pointer to the
323  // source object itself, then detach from the process
324  // if necessary and check if we were given a good
325  // source handle.
326  //
327

329  SourceProcess,
330  SourceObjectTable,
331  PreviousMode,
332  &SourceObject,
333  &HandleInformation,
334  &AuditMask);
335 
336  if (NT_SUCCESS( Status )) {
337 
338  if ((HandleInformation.HandleAttributes & OBJ_AUDIT_OBJECT_CLOSE) == 0) {
339  AuditMask = 0;
340  }
341  }
342 
343 
344  if (!NT_SUCCESS( Status )) {
345 
346  ObDereferenceProcessHandleTable (SourceProcess);
347 
348  return( Status );
349  }
350 
351  //
352  // We are all done if no target process handle was specified.
353  // This is practically a noop because the only really end result
354  // could be that we've closed the source handle.
355  //
356 
357  if (!ARGUMENT_PRESENT( TargetProcess )) {
358 
359  //
360  // If no TargetProcessHandle, then only possible option is to close
361  // the source handle in the context of the source process.
362  //

381 
382  return( Status );

386 
387  //
388  // Make sure the target process has not exited
389  //

392  if ( TargetObjectTable == NULL ) {
393 
394  if (Options & DUPLICATE_CLOSE_SOURCE) {
395 
396  KeStackAttachProcess( &SourceProcess->Pcb, &ApcState );
397 
398  NtClose( SourceHandle );
399

404

406 
407  return STATUS_PROCESS_IS_TERMINATING;

411  // If the specified target process is not the current process, attach
412  // to the specified target process.
413  //
414

416 
417  KeStackAttachProcess( &TargetProcess->Pcb, &ApcState );
418 
419  Attached = TRUE;
420  }
421 
422  //
423  // Construct the proper desired access and attributes for the new handle
424  //
425 
426  if (Options & DUPLICATE_SAME_ACCESS) {
427 
428  DesiredAccess = SourceAccess;
429  }
430 
431  if (Options & DUPLICATE_SAME_ATTRIBUTES) {
432

434 
435  } else {
436 
437  //
438  // Always propagate auditing information.
439  //

443

445  // Get the object header for the source object
446  //
447 
448  ObjectHeader = OBJECT_TO_OBJECT_HEADER( SourceObject );
449  ObjectType = ObjectHeader->Type;
450 
451  RtlZeroMemory( &ObjectTableEntry, sizeof( HANDLE_TABLE_ENTRY ) );

453  ObjectTableEntry.Object = ObjectHeader;
454 
455  ObjectTableEntry.ObAttributes |= (HandleAttributes & OBJ_HANDLE_ATTRIBUTES);
456 
457  //
458  // The following will be reset below if AVR is performed.
459  //
460 
461  ObjectInfo.AuditMask = AuditMask;
462 
463  //
464  // If any of the generic access bits are specified then map those to more
465  // specific access bits
466  //
467 
468  if ((DesiredAccess & GENERIC_ACCESS) != 0) {
469 
470  RtlMapGenericMask( &DesiredAccess,
471  &ObjectType->TypeInfo.GenericMapping );
472  }
473 
474  //
475  // Make sure to preserve ACCESS_SYSTEM_SECURITY, which most likely is not
476  // found in the ValidAccessMask
477  //
478 
479  TargetAccess = DesiredAccess &
480  (ObjectType->TypeInfo.ValidAccessMask | ACCESS_SYSTEM_SECURITY);
481 
482  //
483  // If the access requested for the target is a superset of the
484  // access allowed in the source, perform full AVR. If it is a
485  // subset or equal, do not perform any access validation.
486  //
487  // Do not allow superset access if object type has a private security
488  // method, as there is no means to call them in this case to do the
489  // access check.
490  //
491  // If the AccessState is not passed to ObpIncrementHandleCount
492  // there will be no AVR.
493  //
494 
495  if (TargetAccess & ~SourceAccess) {
496 
497  if (ObjectType->TypeInfo.SecurityProcedure == SeDefaultObjectMethod) {
498 
499  Status = SeCreateAccessState( &AccessState,
500  &AuxData,
501  TargetAccess, // DesiredAccess
502  &ObjectType->TypeInfo.GenericMapping );
503 
504  PassedAccessState = &AccessState;
505 
506  } else {
507 
508  Status = STATUS_ACCESS_DENIED;
509  }
510 
511  } else {
512 
513  //
514  // Do not perform AVR
515  //
516 
517  PassedAccessState = NULL;
518 
519  Status = STATUS_SUCCESS;
520  }
521 
522  //
523  // Increment the new handle count and get a pointer to
524  // the target processes object table
525  //
526 
527  if ( NT_SUCCESS( Status )) {
528 
529  Status = ObpIncrementHandleCount( ObDuplicateHandle,
530  PsGetCurrentProcess(), // this is already the target process
531  SourceObject,
532  ObjectType,
533  PassedAccessState,
534  PreviousMode,
535  HandleAttributes );
536 
537  }
538 
539  if (Attached) {
540 
541  KeUnstackDetachProcess (&ApcState);
542 
543  Attached = FALSE;
544  }
545 
546  if (Options & DUPLICATE_CLOSE_SOURCE) {
547 
548  KeStackAttachProcess( &SourceProcess->Pcb, &ApcState );
549 
550  NtClose( SourceHandle );
551 
552  KeUnstackDetachProcess(&ApcState);
553  }
554 
555  if (!NT_SUCCESS( Status )) {
556 
557  if (PassedAccessState != NULL) {
558 
559  SeDeleteAccessState( PassedAccessState );
560  }
561 
562  ObDereferenceProcessHandleTable (SourceProcess);
563  ObDereferenceProcessHandleTable (TargetProcess);
564 
565  ObDereferenceObject( SourceObject );
566 
567  return( Status );
568  }
569

571 
572  //
573  // If we performed AVR opening the handle, then mark the handle as needing
574  // auditing when it's closed. Also save away the maximum audit mask
575  // if one was created.
576  //
577 
578  ObjectTableEntry.ObAttributes |= OBJ_AUDIT_OBJECT_CLOSE;

581

587

602  //
603

605

607 
608  } else {
609 
610  ObjectTableEntry.CreatorBackTraceIndex = 0;
611  }
612 
613  } else {
614

616  }
617

621 
622 #endif // i386

627  //
628

630

632

634 
635  if ( ObjectInfo.AuditMask != 0 ) {
636 
637  //
638  // Make sure it's the same object before setting the handle information.
639  // The user may have closed it immediately after the ExCreateHandle call,
640  // so at this time it may either be invalid or a completely different object.

642

645

727 NtDuplicateObject (
728  __in HANDLE SourceProcessHandle,
729  __in HANDLE SourceHandle,

737 /*++
738 
739 Routine Description:
740 
741  This function creates a handle that is a duplicate of the specified
742  source handle. The source handle is evaluated in the context of the
743  specified source process. The calling process must have
744  PROCESS_DUP_HANDLE access to the source process. The duplicate
745  handle is created with the specified attributes and desired access.
746  The duplicate handle is created in the handle table of the specified
747  target process. The calling process must have PROCESS_DUP_HANDLE
748  access to the target process.
749 
750 Arguments:
751 
752  SourceProcessHandle - Supplies a handle to the source process for the
753  handle being duplicated
754 
755  SourceHandle - Supplies the handle being duplicated
756 
757  TargetProcessHandle - Optionally supplies a handle to the target process
758  that is to receive the new handle
759 
760  TargetHandle - Optionally returns a the new duplicated handle
761 
762  DesiredAccess - Desired access for the new handle
763 
764  HandleAttributes - Desired attributes for the new handle
765 
766  Options - Duplication options that control things like close source,
767  same access, and same attributes.
768 
769 --*/
770 
771 {
772  KPROCESSOR_MODE PreviousMode;
773  NTSTATUS Status, Status1;
774  PEPROCESS SourceProcess;
775  PEPROCESS TargetProcess;
776  HANDLE NewHandle = NULL;
777 
778  //
779  // Get previous processor mode and probe output arguments if necessary.
780  //
781 
782  PreviousMode = KeGetPreviousMode();
783 
784  if (ARGUMENT_PRESENT( TargetHandle ) && (PreviousMode != KernelMode)) {
785 
786  try {
787 
788  ProbeForWriteHandle( TargetHandle );
789  *TargetHandle = NULL;
790 
791  } except( EXCEPTION_EXECUTE_HANDLER ) {
792 
793  return( GetExceptionCode() );
794  }
795  }
796 
797 
798  //
799  // Given the input source process handle get a pointer
800  // to the source process object
801  //
802 
803  Status = ObReferenceObjectByHandle( SourceProcessHandle,
804  PROCESS_DUP_HANDLE,
805  PsProcessType,
806  PreviousMode,
807  (PVOID *)&SourceProcess,
808  NULL );
809 
810  if (!NT_SUCCESS( Status )) {
811 
812  return Status;
813  }
814 
815  //
816  // We are all done if no target process handle was specified.
817  // This is practically a noop because the only really end result
818  // could be that we've closed the source handle.
819  //
820 
821  if (!ARGUMENT_PRESENT( TargetProcessHandle )) {
822 
823  //
824  // If no TargetProcessHandle, then only possible option is to close
825  // the source handle in the context of the source process.
826  //
827 
828  TargetProcess = NULL;
829  Status1 = STATUS_SUCCESS;
830  } else {
831  //
832  // At this point the caller did specify for a target process
833  // So from the target process handle get a pointer to the
834  // target process object.
835  //
836 
837  Status1 = ObReferenceObjectByHandle( TargetProcessHandle,
838  PROCESS_DUP_HANDLE,
839  PsProcessType,
840  PreviousMode,
841  (PVOID *)&TargetProcess,
842  NULL );
843  //
844  // The original structure of Duplicate cased us to still close the input handle if the output process handle
845  // was bad. We preserve that functionality.
846  //
847  if (!NT_SUCCESS (Status1)) {
848  TargetProcess = NULL;
849  }
850  }
851 
852  Status = ObDuplicateObject (SourceProcess,
853  SourceHandle,
854  TargetProcess,
855  &NewHandle,
856  DesiredAccess,
857  HandleAttributes,
858  Options,
859  PreviousMode );
860 
861  if (ARGUMENT_PRESENT( TargetHandle )) {
862 
863  try {
864 
865  *TargetHandle = NewHandle;
866 
867  } except( EXCEPTION_EXECUTE_HANDLER ) {
868 
869  //
870  // Fall through, since we cannot undo what we have done.
871  //
872  }
873  }
874 
875  ObDereferenceObject( SourceProcess );
876  if ( TargetProcess != NULL) {
877  ObDereferenceObject( TargetProcess );
878  }
879 
880  if (!NT_SUCCESS (Status1)) {
881  Status = Status1;
882  }
883  return( Status );
884 }

1221

1224  necessary it will allocate new and free old handle databases.
1225

1227  space in the handle table.
1228

1230

1232

1237

1247  ULONG NewSize;

1251

1253  // Check if the object has any handle information
1254  //
1255 
1256  HandleInfo = OBJECT_HEADER_TO_HANDLE_INFO(ObjectHeader);
1257 
1258  if (HandleInfo == NULL) {
1259

1261  }
1262

1264  // The object does have some handle information. If it has
1265  // a single handle entry then we'll construct a local dummy
1266  // handle count database and come up with a new data base for
1267  // storing two entries.
1268  //

1280

1282  ((CountEntries - 1) * sizeof( OBJECT_HANDLE_COUNT_ENTRY ));
1283 
1284  } else {
1285

1288  // current handle database, and compute a new size bumped by four
1289  //
1290

1292 
1293  CountEntries = OldHandleCountDataBase->CountEntries;
1294 
1295  OldSize = sizeof(OBJECT_HANDLE_COUNT_DATABASE) +

1297 
1298  CountEntries += 4;
1299 
1300  NewSize = sizeof(OBJECT_HANDLE_COUNT_DATABASE) +
1301  ((CountEntries - 1) * sizeof( OBJECT_HANDLE_COUNT_ENTRY));
1302  }

1306  //
1307

1311 
1312  return NULL;
1313  }

1318  //

1321

1325  //
1326

1328 
1329  ObjectHeader->Flags &= ~OB_FLAG_SINGLE_HANDLE_ENTRY;
1330 
1331  } else {
1332

1334  }

1338  // the memory
1339  //

1349  //

1357  //
1358
