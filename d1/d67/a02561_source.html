214 
215 /*++
216 
217 Routine Description:
218 
219  Hashes a buffer into a 32 bit value
220 
221 Arguments:
222 
223  Data - Buffer containing the data to be hashed.
224 
225  Length - The length in bytes of the buffer
226 
227 
228 Return Value:
229 
230  ULONG - a 32 bit hash value.
231 
232 --*/
233 
234 {
235  PULONG Buffer, BufferEnd;
236  PUCHAR Bufferp, BufferEndp;
237 
238  ULONG Result = 0;
239 
240  //
241  // Calculate buffer bounds as byte pointers
242  //
243  Bufferp = Data;
244  BufferEndp = Bufferp + Length;
245 
246  //
247  // Calculate buffer bounds as rounded down ULONG pointers
248  //
249  Buffer = Data;
250  BufferEnd = (PULONG)(Bufferp + (Length&~(sizeof (ULONG) - 1)));
251 
252  //
253  // Loop over a whole number of ULONGs
254  //
255  while (Buffer < BufferEnd) {
256  Result ^= *Buffer++;
257  Result = _rotl (Result, 3);
258  }
259 
260  //
261  // Pull in the remaining bytes
262  //
263  Bufferp = (PUCHAR) Buffer;
264  while (Bufferp < BufferEndp) {
265  Result ^= *Bufferp++;
266  Result = _rotl (Result, 3);
267  }
268 
269 
270 
271  return Result;
272 }

276 ObLogSecurityDescriptor (
277  __in PSECURITY_DESCRIPTOR InputSecurityDescriptor,
278  __out PSECURITY_DESCRIPTOR *OutputSecurityDescriptor,
279  __in ULONG RefBias
280  )
281 
282 /*++
283 
284 Routine Description:
285 
286  Takes a passed security descriptor and registers it into the
287  security descriptor database.
288 
289 Arguments:
290 
291  InputSecurityDescriptor - The new security descriptor to be logged into
292  the database. On a successful return this memory will have been
293  freed back to pool.
294 
295  OutputSecurityDescriptor - Output security descriptor to be used by the
296  caller.
297 
298  RefBias - Amount to bias the security descriptor reference count by.
299  Typically either 1 or ExFastRefGetAdditionalReferenceCount () + 1,
300 
301 Return Value:
302 
303  An appropriate status value
304 
305 --*/
306 
307 {
308  ULONG FullHash;
309  ULONG Slot;
310  PSECURITY_DESCRIPTOR_HEADER NewDescriptor;
311  PLIST_ENTRY Front;
312  PSECURITY_DESCRIPTOR_HEADER Header = NULL;
313  BOOLEAN Match;
314  POB_SD_CACHE_LIST Chain;
315  PETHREAD CurrentThread;
316  ULONG Length;
317 
318  Length = RtlLengthSecurityDescriptor (InputSecurityDescriptor);
319 
320  FullHash = ObpHashSecurityDescriptor (InputSecurityDescriptor, Length);
321 
322  Slot = FullHash % SECURITY_DESCRIPTOR_CACHE_ENTRIES;
323 
324  NewDescriptor = NULL;
325 
326  //
327  // First lock the table for read access. We will change this to write if we have to insert later
328  //
329  Chain = &ObsSecurityDescriptorCache[Slot];
330 
331  CurrentThread = PsGetCurrentThread ();
332  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
333  ExAcquirePushLockShared (&Chain->PushLock);
334 
335  do {
336  //
337  // See if the list for this slot is in use.
338  // Lock the table first, unlock if if we don't need it.
339  //
340  Match = FALSE;
341 
342  //
343  // Zoom down the hash bucket looking for a full hash match
344  //
345 
346  for (Front = Chain->Head.Flink;
347  Front != &Chain->Head;
348  Front = Front->Flink) {
349 
350  Header = LINK_TO_SD_HEADER (Front);
351 
352  //
353  // The list is ordered by full hash value and is maintained this way by virtue
354  // of the fact that we use the 'Back' variable for the insert.
355  //
356 
357  if (Header->FullHash > FullHash) {
358  break;
359  }
360 
361  if (Header->FullHash == FullHash) {
362 
363  Match = ObpCompareSecurityDescriptors (InputSecurityDescriptor,
364  Length,
365  &Header->SecurityDescriptor);
366 
367  if (Match) {
368 
369  break;
370  }
371 
372  ObPrint (SHOW_COLLISIONS, ("Got a collision on %d, no match\n", Slot));
373  }
374  }
375 
376  //
377  // If we have a match then we'll get the caller to use the old
378  // cached descriptor, but bumping its ref count, freeing what
379  // the caller supplied and returning the old one to our caller
380  //
381 
382  if (Match) {
383 
384  InterlockedExchangeAdd ((PLONG)&Header->RefCount, RefBias);
385 
386  ObPrint (SHOW_REFERENCES, ("Reference Hash = 0x%lX, New RefCount = %d\n", Header->FullHash, Header->RefCount));
387 
388  ExReleasePushLock (&Chain->PushLock);
389  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
390 
391  *OutputSecurityDescriptor = &Header->SecurityDescriptor;
392 
393  if (NewDescriptor != NULL) {
394  ExFreePool (NewDescriptor);
395  }
396 
397  return STATUS_SUCCESS;
398  }
399 
400 
401  if (NewDescriptor == NULL) {
402  ExReleasePushLockShared (&Chain->PushLock);
403  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
404 
405  //
406  // Can't use an existing one, create a new entry
407  // and insert it into the list.
408  //
409 
410  NewDescriptor = ObpCreateCacheEntry (InputSecurityDescriptor,
411  Length,
412  FullHash,
413  RefBias);
414 
415  if (NewDescriptor == NULL) {
416  return STATUS_INSUFFICIENT_RESOURCES;
417  }
418  //
419  // Reacquire the lock in write mode. We will probably have to insert now
420  //
421  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
422  ExAcquirePushLockExclusive (&Chain->PushLock);
423  } else {
424  break;
425  }
426  } while (1);
427 
428 #if OB_DIAGNOSTICS_ENABLED
429 
430  InterlockedIncrement (&ObsTotalCacheEntries);
431 
432 #endif
433 
434  ObPrint (SHOW_STATISTICS, ("ObsTotalCacheEntries = %d \n", ObsTotalCacheEntries));
435  ObPrint (SHOW_COLLISIONS, ("Adding new entry for index #%d \n", Slot));
436 
437 
438  //
439  // Insert the entry before the 'Front' entry. If there is no 'Front' entry then this
440  // is just inserting at the head
441  //
442 
443  InsertTailList (Front, &NewDescriptor->Link);
444 
445  ExReleasePushLockExclusive (&Chain->PushLock);
446  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
447 
448  //
449  // Set the output security descriptor and return to our caller
450  //
451 
452  *OutputSecurityDescriptor = &NewDescriptor->SecurityDescriptor;
453 
454  return( STATUS_SUCCESS );
455 }

459 ObpCreateCacheEntry (
460  PSECURITY_DESCRIPTOR InputSecurityDescriptor,
461  ULONG SecurityDescriptorLength,
462  ULONG FullHash,
463  ULONG RefBias
464  )
465 
466 /*++
467 
468 Routine Description:
469 
470  Allocates and initializes a new cache entry.
471 
472 Arguments:
473 
474  InputSecurityDescriptor - The security descriptor to be cached.
475 
476  Length - Length of security descriptor
477 
478  FullHash - Full 32 bit hash of the security descriptor.
479 
480  RefBias - Amount to bias the security descriptor reference count by.
481  Typically either 1 or ExFastRefGetAdditionalReferenceCount () + 1,
482 
483 Return Value:
484 
485  A pointer to the newly allocated cache entry, or NULL
486 
487 --*/
488 
489 {
490  ULONG CacheEntrySize;
491  PSECURITY_DESCRIPTOR_HEADER NewDescriptor;
492 
493  //
494  // Compute the size that we'll need to allocate. We need space for
495  // the security descriptor cache minus the funny quad at the end and the
496  // security descriptor itself.
497  //
498 
499  ASSERT (SecurityDescriptorLength == RtlLengthSecurityDescriptor (InputSecurityDescriptor));
500  CacheEntrySize = SecurityDescriptorLength + (sizeof (SECURITY_DESCRIPTOR_HEADER) - sizeof(QUAD));
501 
502  //
503  // Now allocate space for the cached entry
504  //
505 
506  NewDescriptor = ExAllocatePoolWithTag (PagedPool, CacheEntrySize, 'cSbO');
507 
508  if (NewDescriptor == NULL) {
509 
510  return NULL;
511  }
512 
513  //
514  // Fill the header, copy over the descriptor data, and return to our
515  // caller
516  //
517 
518  NewDescriptor->RefCount = RefBias;
519  NewDescriptor->FullHash = FullHash;
520 
521  RtlCopyMemory (&NewDescriptor->SecurityDescriptor,
522  InputSecurityDescriptor,
523  SecurityDescriptorLength);
524 
525  return NewDescriptor;
526 }

563 PSECURITY_DESCRIPTOR
564 ObpReferenceSecurityDescriptor (
565  POBJECT_HEADER ObjectHeader
566  )
567 
568 /*++
569 
570 Routine Description:
571 
572  References the security descriptor of the passed object.
573 
574 Arguments:
575 
576  Object - Object being access validated.
577 
578 Return Value:
579 
580  The security descriptor of the object.
581 
582 --*/
583 
584 {
585  PSECURITY_DESCRIPTOR_HEADER SecurityDescriptorHeader;
586  PSECURITY_DESCRIPTOR SecurityDescriptor;
587  PEX_FAST_REF FastRef;
588  EX_FAST_REF OldRef;
589  ULONG RefsToAdd, Unused;
590 
591  //
592  // Attempt the fast reference
593  //
594  FastRef = (PEX_FAST_REF) &ObjectHeader->SecurityDescriptor;
595 
596  OldRef = ExFastReference (FastRef);
597 
598  SecurityDescriptor = ExFastRefGetObject (OldRef);
599 
600  //
601  // See if we can fast reference this security descriptor. Return NULL if there wasn't one
602  // and go the slow way if there are no more cached references left.
603  //
604  Unused = ExFastRefGetUnusedReferences (OldRef);
605 
606  if (Unused >= 1 || SecurityDescriptor == NULL) {
607  if (Unused == 1) {
608  //
609  // If we took the counter to zero then attempt to make life easier for
610  // the next referencer by resetting the counter to its max. Since we now
611  // have a reference to the security descriptor we can do this.
612  //
613  RefsToAdd = ExFastRefGetAdditionalReferenceCount ();
614  SecurityDescriptorHeader = SD_TO_SD_HEADER( SecurityDescriptor );
615  InterlockedExchangeAdd ((PLONG)&SecurityDescriptorHeader->RefCount, RefsToAdd);
616 
617  //
618  // Try to add the added references to the cache. If we fail then just
619  // release them. This dereference can not take the reference count to zero.
620  //
621  if (!ExFastRefAddAdditionalReferenceCounts (FastRef, SecurityDescriptor, RefsToAdd)) {
622  InterlockedExchangeAdd ((PLONG)&SecurityDescriptorHeader->RefCount, -(LONG)RefsToAdd);
623  }
624  }
625  return SecurityDescriptor;
626  }
627 
628  ObpLockObjectShared( ObjectHeader );
629 
630  SecurityDescriptor = ExFastRefGetObject (*FastRef);
631 
632  IF_OB_GLOBAL( STOP_INVALID_DESCRIPTOR ) {
633 
634  if(!RtlValidSecurityDescriptor ( SecurityDescriptor )) {
635 
636  DbgBreakPoint();
637  }
638  }
639 
640  //
641  // The objects security descriptor is not allowed to go fron NON-NULL to NULL.
642  //
643  SecurityDescriptorHeader = SD_TO_SD_HEADER( SecurityDescriptor );
644  ObPrint( SHOW_REFERENCES, ("Referencing Hash %lX, Refcount = %d \n",SecurityDescriptorHeader->FullHash,
645  SecurityDescriptorHeader->RefCount));
646 
647  //
648  // Increment the reference count
649  //
650  InterlockedIncrement ((PLONG) &SecurityDescriptorHeader->RefCount);
651 
652  ObpUnlockObject( ObjectHeader );
653 
654 
655  return( SecurityDescriptor );
656 }

703  __in PSECURITY_DESCRIPTOR SecurityDescriptor,
704  __in ULONG Count
705  )
706 
707 /*++
708 
709 Routine Description:
710 
711  Decrements the refcount of a cached security descriptor
712 
713 Arguments:
714 
715  SecurityDescriptor - Points to a cached security descriptor
716 
717 Return Value:
718 
719  None.
720 
721 --*/
722 
723 {
724  PSECURITY_DESCRIPTOR_HEADER SecurityDescriptorHeader;
725  PVOID PoolToFree;
726  LONG OldValue, NewValue;
727  POB_SD_CACHE_LIST Chain;
728  PETHREAD CurrentThread;
729  ULONG Slot;
730 
731  SecurityDescriptorHeader = SD_TO_SD_HEADER( SecurityDescriptor );
732 
733  //
734  // First see if its possible to do a non-zero transition lock free.
735  //
736  OldValue = SecurityDescriptorHeader->RefCount;
737 
738  //
739  // If the old value is equal to the decrement then we will be the deleter of this block. We need the lock for that
740  //
741  while (OldValue != (LONG) Count) {
742 
743  NewValue = InterlockedCompareExchange ((PLONG)&SecurityDescriptorHeader->RefCount, OldValue - Count, OldValue);
744  if (NewValue == OldValue) {
745  return;
746  }
747  OldValue = NewValue;
748  }
749 
750  //
751  // Lock the security descriptor cache and get a pointer
752  // to the security descriptor header
753  //
754  Slot = SecurityDescriptorHeader->FullHash % SECURITY_DESCRIPTOR_CACHE_ENTRIES;
755 
756  Chain = &ObsSecurityDescriptorCache[Slot];
757 
758  CurrentThread = PsGetCurrentThread ();
759  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
760  ExAcquirePushLockExclusive (&Chain->PushLock);
761 
762  //
763  // Do some debug work
764  //
765 
766  ObPrint( SHOW_REFERENCES, ("Dereferencing SecurityDescriptor %x, hash %lx, refcount = %d \n", SecurityDescriptor,
767  SecurityDescriptorHeader->FullHash,
768  SecurityDescriptorHeader->RefCount));
769 
770  ASSERT(SecurityDescriptorHeader->RefCount != 0);
771 
772  //
773  // Decrement the ref count and if it is now zero then
774  // we can completely remove this entry from the cache
775  //
776 
777  if (InterlockedExchangeAdd ((PLONG)&SecurityDescriptorHeader->RefCount, -(LONG)Count) == (LONG)Count) {
778 
779  PoolToFree = ObpDestroySecurityDescriptorHeader (SecurityDescriptorHeader);
780  //
781  // Unlock the security descriptor cache and free the pool
782  //
783 
784  ExReleasePushLockExclusive (&Chain->PushLock);
785  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
786 
787  ExFreePool (PoolToFree);
788  } else {
789 
790  //
791  // Unlock the security descriptor cache and return to our caller
792  //
793 
794  ExReleasePushLockExclusive (&Chain->PushLock);
795  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
796  }
797 
798 }
