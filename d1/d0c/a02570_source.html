97 NTSTATUS
98 PerfInfoStartLog (
99  PERFINFO_GROUPMASK *PGroupMask,
100  PERFINFO_START_LOG_LOCATION StartLogLocation
101  )
102 
103 /*++
104 
105 Routine Description:
106 
107  This routine is called by WMI as part of kernel logger initialization.
108 
109 Arguments:
110 
111  GroupMask - Masks for what to log. This pointer point to an allocated
112  area in WMI LoggerContext.
113  StartLogLocation - Indication of whether we're starting logging
114  at boot time or while the system is running. If
115  we're starting at boot time, we don't need to snapshot
116  the open files because there aren't any and we would
117  crash if we tried to find the list.
118 
119 Return Value:
120 
121 --*/
122 
123 {
124  NTSTATUS Status = STATUS_SUCCESS;
125  BOOLEAN ProfileInitialized = FALSE;
126  BOOLEAN ContextSwapStarted = FALSE;
127  BOOLEAN IoPerfInitialized = FALSE;
128 
129  PERFINFO_CLEAR_GROUPMASK(&PerfGlobalGroupMask);
130 
131  //
132  // Enable logging.
133  //
134 
135  PPerfGlobalGroupMask = &PerfGlobalGroupMask;
136  PerfSetLogging(PPerfGlobalGroupMask);
137 
138  if (PerfIsGroupOnInGroupMask(PERF_MEMORY, PGroupMask) ||
139  PerfIsGroupOnInGroupMask(PERF_FILENAME, PGroupMask) ||
140  PerfIsGroupOnInGroupMask(PERF_DRIVERS, PGroupMask)) {
141  PERFINFO_OR_GROUP_WITH_GROUPMASK(PERF_FILENAME_ALL, PGroupMask);
142  }
143 
144 
145  if (StartLogLocation == PERFINFO_START_LOG_FROM_GLOBAL_LOGGER) {
146  //
147  // From the wmi global logger, need to do Rundown in kernel mode
148  //
149  if (PerfIsGroupOnInGroupMask(PERF_PROC_THREAD, PGroupMask)) {
150  Status = PerfInfoProcessRunDown();
151  if (!NT_SUCCESS(Status)) {
152  goto Finish;
153  }
154  }
155 
156  if (PerfIsGroupOnInGroupMask(PERF_PROC_THREAD, PGroupMask)) {
157  Status = PerfInfoSysModuleRunDown();
158  if (!NT_SUCCESS(Status)) {
159  goto Finish;
160  }
161  }
162  }
163 
164  //
165  // File Name Rundown code
166  //
167  if ((StartLogLocation != PERFINFO_START_LOG_AT_BOOT) &&
168  PerfIsGroupOnInGroupMask(PERF_FILENAME_ALL, PGroupMask)) {
169  PERFINFO_OR_GROUP_WITH_GROUPMASK(PERF_FILENAME_ALL, PPerfGlobalGroupMask);
170  Status = PerfInfoFileNameRunDown();
171  if (!NT_SUCCESS(Status)) {
172  goto Finish;
173  }
174  }
175 
176  //
177  // Initialize Perf Driver hooks
178  //
179  if (PerfIsGroupOnInGroupMask(PERF_DRIVERS, PGroupMask)) {
180  Status = IoPerfInit();
181  if (NT_SUCCESS(Status)) {
182  IoPerfInitialized = TRUE;
183  } else {
184  goto Finish;
185  }
186  }
187 
188  //
189  // Enable context swap tracing
190  //
191  if ( PerfIsGroupOnInGroupMask(PERF_CONTEXT_SWITCH, PGroupMask) ) {
192  WmiStartContextSwapTrace();
193  ContextSwapStarted = TRUE;
194  }
195 
196  //
197  // Sampled Profile
198  //
199  if (PerfIsGroupOnInGroupMask(PERF_PROFILE, PGroupMask)) {
200  if ((KeGetPreviousMode() == KernelMode) ||
201  (SeSinglePrivilegeCheck(SeSystemProfilePrivilege, UserMode))) {
202  PerfInfoProfileInit();
203  ProfileInitialized = TRUE;
204  } else {
205  Status = STATUS_NO_SUCH_PRIVILEGE;
206  goto Finish;
207  }
208  }
209 
210  //
211  // See if we need to empty the working set to start
212  //
213  if (PerfIsGroupOnInGroupMask(PERF_FOOTPRINT, PGroupMask) ||
214  PerfIsGroupOnInGroupMask(PERF_BIGFOOT, PGroupMask)) {
215  MmEmptyAllWorkingSets ();
216  }
217 
218 Finish:
219 
220  if (!NT_SUCCESS(Status)) {
221  //
222  // Failed to turn on trace, clean up now.
223  //
224 
225  if (ContextSwapStarted) {
226  WmiStopContextSwapTrace();
227  }
228 
229  if (ProfileInitialized) {
230  PerfInfoProfileUninit();
231  }
232 
233  if (IoPerfInitialized) {
234  IoPerfReset();
235  }
236 
237  //
238  // Disable logging.
239  //
240 
241  PPerfGlobalGroupMask = NULL;
242  PerfSetLogging(NULL);
243 
244  PERFINFO_CLEAR_GROUPMASK(&PerfGlobalGroupMask);
245  } else {
246  *PPerfGlobalGroupMask = *PGroupMask;
247  }
248 
249  return Status;
250 }

253 NTSTATUS
254 PerfInfoStopLog (
255  )
256 
257 /*++
258 
259 Routine Description:
260 
261  This routine turn off the PerfInfo trace hooks.
262 
263 Arguments:
264 
265  None.
266 
267 Return Value:
268 
269 --*/
270 
271 {
272  NTSTATUS Status = STATUS_SUCCESS;
273  BOOLEAN DisableContextSwaps=FALSE;
274 
275  if (PPerfGlobalGroupMask == NULL) {
276  return Status;
277  }
278 
279  if (PERFINFO_IS_GROUP_ON(PERF_MEMORY)) {
280  MmIdentifyPhysicalMemory();
281  }
282 
283  if (PERFINFO_IS_GROUP_ON(PERF_PROFILE)) {
284  PerfInfoProfileUninit();
285  }
286 
287  if (PERFINFO_IS_GROUP_ON(PERF_DRIVERS)) {
288  IoPerfReset();
289  }
290 
291  if ( PERFINFO_IS_GROUP_ON(PERF_CONTEXT_SWITCH) ) {
292  DisableContextSwaps = TRUE;
293  }
294 
295  //
296  // Reset the PPerfGlobalGroupMask.
297  //
298 
299  PERFINFO_CLEAR_GROUPMASK(PPerfGlobalGroupMask);
300 
301  //
302  // Disable logging.
303  //
304 
305  PPerfGlobalGroupMask = NULL;
306  PerfSetLogging(NULL);
307 
308  //
309  // Disable context swap tracing.
310  // IMPORTANT: This must be done AFTER the global flag is set to NULL!!!
311  //
312  if( DisableContextSwaps ) {
313 
314  WmiStopContextSwapTrace();
315  }
316 
317  return (Status);
318 
319 }
