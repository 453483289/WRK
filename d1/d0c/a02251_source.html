120 //
121 // Define the custom notification for the u-mode
122 // recipient of ReportTargetDeviceChange.
123 // The following structure header is used for all other (i.e., 3rd-party)
124 // target device change events. The structure accommodates both a
125 // variable-length binary data buffer, and a variable-length unicode text
126 // buffer. The header must indicate where the text buffer begins, so that
127 // the data can be delivered in the appropriate format (ANSI or Unicode)
128 // to user-mode recipients (i.e., that have registered for handle-based
129 // notification via RegisterDeviceNotification).
130 //
131 typedef struct _PLUGPLAY_CUSTOM_NOTIFICATION {
132  PLUGPLAY_NOTIFY_HDR HeaderInfo;
133  //
134  // Event-specific data
135  //
136  PVOID FileObject; // This field must be set to NULL by callers of
137  // IoReportTargetDeviceChange. Clients that
138  // have registered for target device change
139  // notification on the affected PDO will be
140  // called with this field set to the file object
141  // they specified during registration.
142  //
143  LONG NameBufferOffset; // offset (in bytes) from beginning of
144  // CustomDataBuffer where text begins (-1 if none)
145  //
146  UCHAR CustomDataBuffer[1]; // variable-length buffer, containing (optionally)
147  // a binary data at the start of the buffer,
148  // followed by an optional unicode text buffer
149  // (word-aligned).
150  //
151 
152 } PLUGPLAY_CUSTOM_NOTIFICATION, *PPLUGPLAY_CUSTOM_NOTIFICATION;
