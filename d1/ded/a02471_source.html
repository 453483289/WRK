29 /*++
30 
31 Routine Description:
32 
33  This function completes a timer that could not be inserted in the timer
34  table because its due time has already passed.
35 
36  N.B. This function must be called with the corresponding timer table
37  entry locked at raised IRQL.
38 
39  N.B. This function returns with no locks held at raised IRQL.
40 
41 Arguments:
42 
43  Timer - Supplies a pointer to a dispatcher object of type timer.
44 
45  LockQueue - Supplies a pointer to a timer table lock queue entry.
46 
47 Return Value:
48 
49  None.
50 
51 -*/
52 
53 {
54 
55  LIST_ENTRY ListHead;
56  BOOLEAN RequestInterrupt;
57 
58  //
59  // Remove the timer from the specified timer table list and insert
60  // the timer in a dummy list in case the timer is cancelled while
61  // releasing the timer table lock and acquiring the dispatcher lock.
62  //
63 
64  KiRemoveEntryTimer(Timer);
65  ListHead.Flink = &Timer->TimerListEntry;
66  ListHead.Blink = &Timer->TimerListEntry;
67  Timer->TimerListEntry.Flink = &ListHead;
68  Timer->TimerListEntry.Blink = &ListHead;
69  KiReleaseTimerTableLock(LockQueue);
70  RequestInterrupt = FALSE;
71  KiLockDispatcherDatabaseAtSynchLevel();
72 
73  //
74  // If the timer has not been removed from the dummy list, then signal
75  // the timer.
76  //
77 
78  if (ListHead.Flink != &ListHead) {
79  RequestInterrupt = KiSignalTimer(Timer);
80  }
81 
82  //
83  // Release the dispatcher lock and request a DPC interrupt if required.
84  //
85 
86  KiUnlockDispatcherDatabaseFromSynchLevel();
87  if (RequestInterrupt == TRUE) {
88  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
89  }
90 
91  return;
92 }

101 /*++
102 
103 Routine Description:
104 
105  This function inserts a timer object in the timer table.
106 
107  N.B. This routine assumes that the timer table lock has been acquired.
108 
109 Arguments:
110 
111  Timer - Supplies a pointer to a dispatcher object of type timer.
112 
113  Hand - supplies the timer table hand value.
114 
115 Return Value:
116 
117  If the timer has expired, than a value of TRUE is returned. Otherwise, a
118  value of FALSE is returned.
119 
120 --*/
121 
122 {
123 
124  ULONG64 DueTime;
125  LOGICAL Expired;
126  ULARGE_INTEGER InterruptTime;
127  PLIST_ENTRY ListHead;
128  PLIST_ENTRY NextEntry;
129  PKTIMER NextTimer;
130 
131  //
132  // Set the signal state to FALSE if the period is zero.
133  //
134  // N.B. The timer state is set to inserted.
135  //
136 
137  Expired = FALSE;
138  if (Timer->Period == 0) {
139  Timer->Header.SignalState = FALSE;
140  }
141 
142  //
143  // If the timer is due before the first entry in the computed list
144  // or the computed list is empty, then insert the timer at the front
145  // of the list and check if the timer has already expired. Otherwise,
146  // insert then timer in the sorted order of the list searching from
147  // the back of the list forward.
148  //
149  // N.B. The sequence of operations below is critical to avoid the race
150  // condition that exists between this code and the clock interrupt
151  // code that examines the timer table lists to detemine when timers
152  // expire.
153  //
154 
155  DueTime = Timer->DueTime.QuadPart;
156 
157  ASSERT(Hand == KiComputeTimerTableIndex(DueTime));
158 
159  ListHead = &KiTimerTableListHead[Hand].Entry;
160  NextEntry = ListHead->Blink;
161  while (NextEntry != ListHead) {
162 
163  //
164  // Compute the maximum search count.
165  //
166 
167  NextTimer = CONTAINING_RECORD(NextEntry, KTIMER, TimerListEntry);
168  if (DueTime >= (ULONG64)NextTimer->DueTime.QuadPart) {
169  break;
170  }
171 
172  NextEntry = NextEntry->Blink;
173  }
174 
175  InsertHeadList(NextEntry, &Timer->TimerListEntry);
176  if (NextEntry == ListHead) {
177 
178  //
179  // The computed list is empty or the timer is due to expire before
180  // the first entry in the list.
181  //
182  // Make sure the writes for the update of the due time table are done
183  // before reading the interrupt time.
184  //
185 
186  KiTimerTableListHead[Hand].Time.QuadPart = DueTime;
187  KeMemoryBarrier();
188  KiQueryInterruptTime((PLARGE_INTEGER)&InterruptTime);
189  if (DueTime <= (ULONG64)InterruptTime.QuadPart) {
190  Expired = TRUE;
191  }
192  }
193 
194  return Expired;
195 }

203 /*++
204 
205 Routine Description:
206 
207  This function signals a timer that could not be inserted in the timer
208  table because its due time has already passed.
209 
210  N.B. This function must be called with the dispatcher database locked at
211  at raised IRQL.
212 
213 Arguments:
214 
215  Timer - Supplies a pointer to a dispatcher object of type timer.
216 
217 Return Value:
218 
219  If a software interrupt should be requested, then TRUE is returned.
220  Otherwise, a value of FALSE is returned.
221 
222 -*/
223 
224 {
225 
226  PKDPC Dpc;
227  LARGE_INTEGER Interval;
228  LONG Period;
229  BOOLEAN RequestInterrupt;
230  LARGE_INTEGER SystemTime;
231 
232  //
233  // Set time timer state to not inserted and the signal state to signaled.
234  //
235 
236  RequestInterrupt = FALSE;
237  Timer->Header.Inserted = FALSE;
238  Timer->Header.SignalState = 1;
239 
240  //
241  // Capture the DPC and period fields from the timer object. Once wait
242  // test is called, the timer must not be touched again unless it is
243  // periodic. The reason for this is that a thread may allocate a timer
244  // on its local stack and wait on it. Wait test can cause that thread
245  // to immediately start running on another processor on an MP system.
246  // If the thread returns, then the timer will be corrupted.
247  //
248 
249  Dpc = Timer->Dpc;
250  Period = Timer->Period;
251  if (IsListEmpty(&Timer->Header.WaitListHead) == FALSE) {
252  if (Timer->Header.Type == TimerNotificationObject) {
253  KiWaitTestWithoutSideEffects(Timer, TIMER_EXPIRE_INCREMENT);
254 
255  } else {
256  KiWaitTestSynchronizationObject(Timer, TIMER_EXPIRE_INCREMENT);
257  }
258  }
259 
260  //
261  // If the timer is periodic, then compute the next interval time and
262  // reinsert the timer in the timer tree.
263  //
264  // N.B. Even though the timer insertion is relative, it can still fail
265  // if the period of the timer elapses in between computing the time
266  // and inserting the timer. If this happens, then the insertion is
267  // retried.
268  //
269 
270  if (Period != 0) {
271  Interval.QuadPart = Int32x32To64(Period, - 10 * 1000);
272  do {
273  } while (KiInsertTreeTimer(Timer, Interval) == FALSE);
274  }
275 
276  //
277  // If a DPC is specified, then insert it in the target processor's DPC
278  // queue or capture the parameters in the DPC table for subsequent
279  // execution on the current processor.
280  //
281  // N.B. A dispatch interrupt must be forced on the current processor,
282  // however, this will occur very infrequently.
283  //
284 
285  if (Dpc != NULL) {
286  KiQuerySystemTime(&SystemTime);
287  KeInsertQueueDpc(Dpc,
288  ULongToPtr(SystemTime.LowPart),
289  ULongToPtr(SystemTime.HighPart));
290 
291  RequestInterrupt = TRUE;
292  }
293 
294  return RequestInterrupt;
295 }
