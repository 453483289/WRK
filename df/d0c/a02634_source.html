69 /*++
70 
71 Routine Description:
72 
73  Records the dispatching of exceptions to frame-based handlers.
74  The debugger may inspect the table later and interpret the data
75  to discover the address of the filters and handlers.
76 
77 Arguments:
78 
79  ExceptionRecord - Supplies an exception record
80 
81  ContextRecord - Supplies the context at the exception
82 
83  ControlPc - Supplies the PC where control left the frame being
84  dispatched to.
85 
86  HandlerData - Supplies a pointer to the host-dependent exception
87  data. On the RISC machines this is a RUNTIME_FUNCTION record;
88  on x86 it is the registration record from the stack frame.
89 
90  Size - Supplies the size of HandlerData
91 
92 Returns:
93 
94  The index to the log entry used, so that if the handler returns
95  a disposition it may be recorded.
96 
97 --*/
98 {
99  ULONG LogIndex;
100 
101  if (!RtlpExceptionLog) {
102  return 0;
103  }
104 
105  ASSERT(Size <= MAX_EXCEPTION_LOG_DATA_SIZE * sizeof(ULONG));
106 
107  do {
108  LogIndex = RtlpExceptionLogCount;
109  } while (LogIndex != (ULONG)InterlockedCompareExchange(
110  (PLONG)&RtlpExceptionLogCount,
111  ((LogIndex + 1) % MAX_EXCEPTION_LOG),
112  LogIndex));
113 
114  //
115  // the debugger will have to interpret the exception handler
116  // data, because it cannot be done safely here.
117  //
118 
119  RtlCopyMemory(RtlpExceptionLog[LogIndex].HandlerData,
120  HandlerData,
121  Size);
122  RtlpExceptionLog[LogIndex].ExceptionRecord = *ExceptionRecord;
123  RtlpExceptionLog[LogIndex].ContextRecord = *ContextRecord;
124  RtlpExceptionLog[LogIndex].Disposition = -1;
125 
126  return LogIndex;
127 }
