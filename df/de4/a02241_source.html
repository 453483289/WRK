70 //
71 // Define per processor lock queue structure.
72 //
73 // N.B. The lock field of the spin lock queue structure contains the address
74 // of the associated kernel spin lock, an owner bit, and a lock bit. Bit
75 // 0 of the spin lock address is the wait bit and bit 1 is the owner bit.
76 // The use of this field is such that the bits can be set and cleared
77 // noninterlocked, however, the back pointer must be preserved.
78 //
79 // The lock wait bit is set when a processor enqueues itself on the lock
80 // queue and it is not the only entry in the queue. The processor will
81 // spin on this bit waiting for the lock to be granted.
82 //
83 // The owner bit is set when the processor owns the respective lock.
84 //
85 // The next field of the spin lock queue structure is used to line the
86 // queued lock structures together in fifo order. It also can set set and
87 // cleared noninterlocked.
88 //
89 
90 #define LOCK_QUEUE_WAIT 1
91 #define LOCK_QUEUE_WAIT_BIT 0
92 
93 #define LOCK_QUEUE_OWNER 2
94 #define LOCK_QUEUE_OWNER_BIT 1
