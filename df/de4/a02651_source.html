113  OUT PVOID *Address OPTIONAL,
114  OUT PULONG Size OPTIONAL

116 
117 /*++
118 
119 Routine Description:
120 
121  This function returns the data necessary to actually examine the
122  contents of a particular resource, without allowing for the .mui
123  feature. It used to be the tail of LdrpAccessResourceData, from
124  which it is now called.
125 
126 Arguments:
127 
128  DllHandle - Supplies a handle to the image file that the resource is
129  contained in.
130 
131  ResourceDataEntry - Supplies a pointer to the resource data entry in
132  the resource data directory of the image file specified by the
133  DllHandle parameter. This pointer should have been one returned
134  by the LdrFindResource function.
135 
136  Address - Optional pointer to a variable that will receive the
137  address of the resource specified by the first two parameters.
138 
139  Size - Optional pointer to a variable that will receive the size of
140  the resource specified by the first two parameters.
141 
142 --*/
143 
144 {
145  PIMAGE_RESOURCE_DIRECTORY ResourceDirectory;
146  ULONG ResourceSize;
147  PIMAGE_NT_HEADERS NtHeaders;
148  ULONG_PTR VirtualAddressOffset;
149  PIMAGE_SECTION_HEADER NtSection;
150  NTSTATUS Status = STATUS_SUCCESS;
151 
152  RTL_PAGED_CODE();
153 
154  try {
155  ResourceDirectory = (PIMAGE_RESOURCE_DIRECTORY)
156  RtlImageDirectoryEntryToData(DllHandle,
157  TRUE,
158  IMAGE_DIRECTORY_ENTRY_RESOURCE,
159  &ResourceSize
160  );
161  if (!ResourceDirectory) {
162  return STATUS_RESOURCE_DATA_NOT_FOUND;
163  }
164 
165  if (LDR_IS_DATAFILE(DllHandle)) {
166  ULONG ResourceRVA;
167  DllHandle = LDR_DATAFILE_TO_VIEW(DllHandle);
168  NtHeaders = RtlImageNtHeader( DllHandle );
169  if (NtHeaders->OptionalHeader.Magic == IMAGE_NT_OPTIONAL_HDR32_MAGIC) {
170  ResourceRVA=((PIMAGE_NT_HEADERS32)NtHeaders)->OptionalHeader.DataDirectory[ IMAGE_DIRECTORY_ENTRY_RESOURCE ].VirtualAddress;
171  } else if (NtHeaders->OptionalHeader.Magic == IMAGE_NT_OPTIONAL_HDR64_MAGIC) {
172  ResourceRVA=((PIMAGE_NT_HEADERS64)NtHeaders)->OptionalHeader.DataDirectory[ IMAGE_DIRECTORY_ENTRY_RESOURCE ].VirtualAddress;
173  } else {
174  ResourceRVA = 0;
175  }
176 
177  if (!ResourceRVA) {
178  return STATUS_RESOURCE_DATA_NOT_FOUND;
179  }
180 
181  VirtualAddressOffset = (ULONG_PTR)DllHandle + ResourceRVA - (ULONG_PTR)ResourceDirectory;
182 
183  //
184  // Now, we must check to see if the resource is not in the
185  // same section as the resource table. If it's in .rsrc1,
186  // we've got to adjust the RVA in the ResourceDataEntry
187  // to point to the correct place in the non-VA data file.
188  //
189  NtSection = RtlSectionTableFromVirtualAddress( NtHeaders, DllHandle, ResourceRVA);
190 
191  if (!NtSection) {
192  return STATUS_RESOURCE_DATA_NOT_FOUND;
193  }
194 
195  if ( ResourceDataEntry->OffsetToData > NtSection->Misc.VirtualSize ) {
196  ULONG rva;
197 
198  rva = NtSection->VirtualAddress;
199  NtSection = RtlSectionTableFromVirtualAddress(NtHeaders,
200  DllHandle,
201  ResourceDataEntry->OffsetToData
202  );
203  if (!NtSection) {
204  return STATUS_RESOURCE_DATA_NOT_FOUND;
205  }
206  VirtualAddressOffset +=
207  ((ULONG_PTR)NtSection->VirtualAddress - rva) -
208  ((ULONG_PTR)RtlAddressInSectionTable ( NtHeaders, DllHandle, NtSection->VirtualAddress ) - (ULONG_PTR)ResourceDirectory);
209  }
210  } else {
211  VirtualAddressOffset = 0;
212  }
213 
214  if (ARGUMENT_PRESENT( Address )) {
215  *Address = (PVOID)( (PCHAR)DllHandle +
216  (ResourceDataEntry->OffsetToData - VirtualAddressOffset)
217  );
218  }
219 
220  if (ARGUMENT_PRESENT( Size )) {
221  *Size = ResourceDataEntry->Size;
222  }
223 
224  } except (EXCEPTION_EXECUTE_HANDLER) {
225  Status = GetExceptionCode();
226  }
227 
228 #if DBG
229  if (!NT_SUCCESS(Status)) {
230  KdPrintEx((DPFLTR_LDR_ID, DPFLTR_LEVEL_STATUS(Status), "LDR: %s() exiting 0x%08lx\n", __FUNCTION__, Status));
231  }
232 #endif
233  return Status;
234 }

294 LdrFindEntryForAddress(
295  IN PVOID Address,
296  OUT PLDR_DATA_TABLE_ENTRY *TableEntry
297  )
298 /*++
299 
300 Routine Description:
301 
302  This function returns the load data table entry that describes the virtual
303  address range that contains the passed virtual address.
304 
305 Arguments:
306 
307  Address - Supplies a 32-bit virtual address.
308 
309  TableEntry - Supplies a pointer to the variable that will receive the
310  address of the loader data table entry.
311 
312 
313 Return Value:
314 
315  Status
316 
317 --*/
318 {
319  PPEB_LDR_DATA Ldr;
320  PLIST_ENTRY Head, Next;
321  PLDR_DATA_TABLE_ENTRY Entry;
322  PIMAGE_NT_HEADERS NtHeaders;
323  PVOID ImageBase;
324  PVOID EndOfImage;
325  NTSTATUS Status;
326 
327  Ldr = NtCurrentPeb()->Ldr;
328  if (Ldr == NULL) {
329  Status = STATUS_NO_MORE_ENTRIES;
330  goto Exit;
331  }
332 
333  Entry = (PLDR_DATA_TABLE_ENTRY) Ldr->EntryInProgress;
334  if (Entry != NULL) {
335  NtHeaders = RtlImageNtHeader( Entry->DllBase );
336  if (NtHeaders != NULL) {
337  ImageBase = (PVOID)Entry->DllBase;
338 
339  EndOfImage = (PVOID)
340  ((ULONG_PTR)ImageBase + NtHeaders->OptionalHeader.SizeOfImage);
341 
342  if ((ULONG_PTR)Address >= (ULONG_PTR)ImageBase && (ULONG_PTR)Address < (ULONG_PTR)EndOfImage) {
343  *TableEntry = Entry;
344  Status = STATUS_SUCCESS;
345  goto Exit;
346  }
347  }
348  }
349 
350  Head = &Ldr->InMemoryOrderModuleList;
351  Next = Head->Flink;
352  while ( Next != Head ) {
353  Entry = CONTAINING_RECORD( Next, LDR_DATA_TABLE_ENTRY, InMemoryOrderLinks );
354 
355  NtHeaders = RtlImageNtHeader( Entry->DllBase );
356  if (NtHeaders != NULL) {
357  ImageBase = (PVOID)Entry->DllBase;
358 
359  EndOfImage = (PVOID)
360  ((ULONG_PTR)ImageBase + NtHeaders->OptionalHeader.SizeOfImage);
361 
362  if ((ULONG_PTR)Address >= (ULONG_PTR)ImageBase && (ULONG_PTR)Address < (ULONG_PTR)EndOfImage) {
363  *TableEntry = Entry;
364  Status = STATUS_SUCCESS;
365  goto Exit;
366  }
367  }
368 
369  Next = Next->Flink;
370  }
371 
372  Status = STATUS_NO_MORE_ENTRIES;
373 Exit:
374  if (!NT_SUCCESS(Status)) {
375  KdPrintEx((DPFLTR_LDR_ID, DPFLTR_LEVEL_STATUS(Status), "LDR: %s() exiting 0x%08lx\n", __FUNCTION__, Status));
376  }
377  return( Status );
378 }

543 LdrpCompareResourceNames_U(
544  IN ULONG_PTR ResourceName,
545  IN const IMAGE_RESOURCE_DIRECTORY* ResourceDirectory,
546  IN const IMAGE_RESOURCE_DIRECTORY_ENTRY* ResourceDirectoryEntry
547  )
548 {
549  LONG li;
550  PIMAGE_RESOURCE_DIR_STRING_U ResourceNameString;
551 
552  if (ResourceName & LDR_RESOURCE_ID_NAME_MASK) {
553  if (!ResourceDirectoryEntry->NameIsString) {
554  return( -1 );
555  }
556 
557  ResourceNameString = (PIMAGE_RESOURCE_DIR_STRING_U)
558  ((PCHAR)ResourceDirectory + ResourceDirectoryEntry->NameOffset);
559 
560  li = wcsncmp( (LPWSTR)ResourceName,
561  ResourceNameString->NameString,
562  ResourceNameString->Length
563  );
564 
565  if (!li && wcslen((PWSTR)ResourceName) != ResourceNameString->Length) {
566  return( 1 );
567  }
568 
569  return(li);
570  }
571  else {
572  if (ResourceDirectoryEntry->NameIsString) {
573  return( 1 );
574  }
575 
576  return( (ULONG)(ResourceName - ResourceDirectoryEntry->Name) );
577  }
578 }

581 // set above 16 should be ok, and this value only has
582 // to fit in a ULONG_PTR. 0x10000 should be sufficient.
583 // The value used is actually 0xFFFF regardless of 32bit or 64bit,
584 // I guess assuming this is not an actual langid, which it isn't,
585 // due to the relatively small number of languages, around 70.
586 #define USE_FIRSTAVAILABLE_LANGID (0xFFFFFFFF & ~LDR_RESOURCE_ID_NAME_MASK)

588 NTSTATUS
589 LdrpSearchResourceSection_U(
590  IN PVOID DllHandle,
591  IN const ULONG_PTR* ResourceIdPath,
592  IN ULONG ResourceIdPathLength,
593  IN ULONG Flags,
594  OUT PVOID *ResourceDirectoryOrData
595  )
596 
597 /*++
598 
599 Routine Description:
600 
601  This function locates the address of the specified resource in the
602  specified DLL and returns its address.
603 
604 Arguments:
605 
606  DllHandle - Supplies a handle to the image file that the resource is
607  contained in.
608 
609  ResourceIdPath - Supplies a pointer to an array of 32-bit resource
610  identifiers. Each identifier is either an integer or a pointer
611  to a null terminated string (PSZ) that specifies a resource
612  name. The array is used to traverse the directory structure
613  contained in the resource section in the image file specified by
614  the DllHandle parameter.
615 
616  ResourceIdPathLength - Supplies the number of elements in the
617  ResourceIdPath array.
618 
619  Flags -
620  LDRP_FIND_RESOURCE_DIRECTORY
621  searching for a resource directory, otherwise the caller is
622  searching for a resource data entry.
623 
624  LDR_FIND_RESOURCE_LANGUAGE_EXACT
625  searching for a resource with, and only with, the language id
626  specified in ResourceIdPath, otherwise the caller wants the routine
627  to come up with default when specified langid is not found.
628 
629  LDR_FIND_RESOURCE_LANGUAGE_REDIRECT_VERSION
630  searching for a resource version in main and alternative
631  modules paths
632 
633  FindDirectoryEntry - Supplies a boolean that is TRUE if caller is
634  searching for a resource directory, otherwise the caller is
635  searching for a resource data entry.
636 
637  ExactLangMatchOnly - Supplies a boolean that is TRUE if caller is
638  searching for a resource with, and only with, the language id
639  specified in ResourceIdPath, otherwise the caller wants the routine
640  to come up with default when specified langid is not found.
641 
642  ResourceDirectoryOrData - Supplies a pointer to a variable that will
643  receive the address of the resource directory or data entry in
644  the resource data section of the image file specified by the
645  DllHandle parameter.
646 --*/
647 
648 {
649  NTSTATUS Status;
650  PIMAGE_RESOURCE_DIRECTORY LanguageResourceDirectory, ResourceDirectory, TopResourceDirectory;
651  PIMAGE_RESOURCE_DIRECTORY_ENTRY ResourceDirEntLow;
652  PIMAGE_RESOURCE_DIRECTORY_ENTRY ResourceDirEntMiddle;
653  PIMAGE_RESOURCE_DIRECTORY_ENTRY ResourceDirEntHigh;
654  PIMAGE_RESOURCE_DATA_ENTRY ResourceEntry;
655  USHORT n, half;
656  LONG dir;
657  ULONG size;
658  ULONG_PTR ResourceIdRetry;
659  ULONG RetryCount;
660  LANGID NewLangId;
661  const ULONG_PTR* IdPath = ResourceIdPath;
662  ULONG IdPathLength = ResourceIdPathLength;
663  BOOLEAN fIsNeutral = FALSE;
664  LANGID GivenLanguage;
665 
666  RTL_PAGED_CODE();
667 
668  try {
669  TopResourceDirectory = (PIMAGE_RESOURCE_DIRECTORY)
670  RtlImageDirectoryEntryToData(DllHandle,
671  TRUE,
672  IMAGE_DIRECTORY_ENTRY_RESOURCE,
673  &size
674  );
675  if (!TopResourceDirectory) {
676  return( STATUS_RESOURCE_DATA_NOT_FOUND );
677  }
678 
679  ResourceDirectory = TopResourceDirectory;
680  ResourceIdRetry = USE_FIRSTAVAILABLE_LANGID;
681  RetryCount = 0;
682  ResourceEntry = NULL;
683  LanguageResourceDirectory = NULL;
684  while (ResourceDirectory != NULL && ResourceIdPathLength--) {
685  //
686  // If search path includes a language id, then attempt to
687  // match the following language ids in this order:
688  //
689  // (0) use given language id
690  // (1) use primary language of given language id
691  // (2) use id 0 (neutral resource)
692  // (4) use user UI language
693  //
694  // If the PRIMARY language id is ZERO, then ALSO attempt to
695  // match the following language ids in this order:
696  //
697  // (3) use thread language id for console app
698  // (4) use user UI language
699  // (5) use lang id of TEB for windows app if it is different from user locale
700  // (6) use UI lang from exe resource
701  // (7) use primary UI lang from exe resource
702  // (8) use Install Language
703  // (9) use lang id from user's locale id
704  // (10) use primary language of user's locale id
705  // (11) use lang id from system default locale id
706  // (12) use lang id of system default locale id
707  // (13) use primary language of system default locale id
708  // (14) use US English lang id
709  // (15) use any lang id that matches requested info
710  //
711  if (ResourceIdPathLength == 0 && IdPathLength == 3) {
712  LanguageResourceDirectory = ResourceDirectory;
713  }
714 
715  if (LanguageResourceDirectory != NULL) {
716  GivenLanguage = (LANGID)IdPath[ 2 ];
717  fIsNeutral = (PRIMARYLANGID( GivenLanguage ) == LANG_NEUTRAL);
718 TryNextLangId:
719  switch( RetryCount++ ) {
720  case 0: // Use given language id
721  NewLangId = GivenLanguage;
722  break;
723 
724  case 1: // Use primary language of given language id
725  NewLangId = PRIMARYLANGID( GivenLanguage );
726  break;
727 
728  case 2: // Use id 0 (neutral resource)
729  NewLangId = 0;
730  break;
731 
732  case 3: // Use user's default UI language
733  NewLangId = (LANGID)ResourceIdRetry;
734  break;
735 
736  case 4: // Use native UI language
737  if ( !fIsNeutral ) {
738  // Stop looking - Not in the neutral case
739  goto ReturnFailure;
740  break;
741  }
742  NewLangId = PsInstallUILanguageId;
743  break;
744 
745  case 5: // Use default system locale
746  NewLangId = LANGIDFROMLCID(PsDefaultSystemLocaleId);
747  break;
748 
749  case 6:
750  // Use US English language
751  NewLangId = MAKELANGID( LANG_ENGLISH, SUBLANG_ENGLISH_US );
752  break;
753
