39 /*++
40 
41 Routine Description:
42 
43  This function initializes a kernel mutant object.
44 
45 Arguments:
46 
47  Mutant - Supplies a pointer to a dispatcher object of type mutant.
48 
49  InitialOwner - Supplies a boolean value that determines whether the
50  current thread is to be the initial owner of the mutant object.
51 
52 Return Value:
53 
54  None.
55 
56 --*/
57 
58 {
59 
60  PLIST_ENTRY ListEntry;
61  KIRQL OldIrql;
62  PRKTHREAD Thread;
63 
64  //
65  // Initialize standard dispatcher object header, set the owner thread to
66  // NULL, set the abandoned state to FALSE, and set the APC disable count
67  // to zero (this is the only thing that distinguishes a mutex from a mutant).
68  //
69 
70  Mutant->Header.Type = MutantObject;
71  Mutant->Header.Size = sizeof(KMUTANT) / sizeof(LONG);
72  if (InitialOwner == TRUE) {
73  Thread = KeGetCurrentThread();
74  Mutant->Header.SignalState = 0;
75  Mutant->OwnerThread = Thread;
76  KiLockDispatcherDatabase(&OldIrql);
77  ListEntry = Thread->MutantListHead.Blink;
78  InsertHeadList(ListEntry, &Mutant->MutantListEntry);
79  KiUnlockDispatcherDatabase(OldIrql);
80 
81  } else {
82  Mutant->Header.SignalState = 1;
83  Mutant->OwnerThread = (PKTHREAD)NULL;
84  }
85 
86  InitializeListHead(&Mutant->Header.WaitListHead);
87  Mutant->Abandoned = FALSE;
88  Mutant->ApcDisable = 0;
89  return;
90 }

179 /*++
180 
181 Routine Description:
182 
183  This function releases a mutant object by incrementing the mutant
184  count. If the resultant value is one, then an attempt is made to
185  satisfy as many Waits as possible. The previous signal state of
186  the mutant is returned as the function value. If the Abandoned
187  parameter is TRUE, then the mutant object is released by settings
188  the signal state to one.
189 
190 Arguments:
191 
192  Mutant - Supplies a pointer to a dispatcher object of type mutant.
193 
194  Increment - Supplies the priority increment that is to be applied
195  if setting the event causes a Wait to be satisfied.
196 
197  Abandoned - Supplies a boolean value that signifies whether the
198  mutant object is being abandoned.
199 
200  Wait - Supplies a boolean value that signifies whether the call to
201  KeReleaseMutant will be immediately followed by a call to one
202  of the kernel Wait functions.
203 
204 Return Value:
205 
206  The previous signal state of the mutant object.
207 
208 --*/
209 
210 {
211 
212  KIRQL OldIrql;
213  LONG OldState;
214  PRKTHREAD Thread;
215  ULONG LeaveCriticalRegion;
216 
217  ASSERT_MUTANT(Mutant);
218  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
219  ASSERT((Mutant->ApcDisable == 0) || (Mutant->ApcDisable == 1));
220 
221  //
222  // Raise IRQL to dispatcher level and lock dispatcher database.
223  //
224 
225  LeaveCriticalRegion = 0;
226  Thread = KeGetCurrentThread();
227  KiLockDispatcherDatabase(&OldIrql);
228 
229  //
230  // Capture the current signal state of the mutant object.
231  //
232 
233  OldState = Mutant->Header.SignalState;
234 
235  //
236  // If the Abandoned parameter is TRUE, then force the release of the
237  // mutant object by setting its ownership count to one and setting its
238  // abandoned state to TRUE. Otherwise increment mutant ownership count.
239  // If the result count is one, then remove the mutant object from the
240  // thread's owned mutant list, set the owner thread to NULL, and attempt
241  // to satisfy a Wait for the mutant object if the mutant object wait
242  // list is not empty.
243  //
244 
245  if (Abandoned != FALSE) {
246  Mutant->Header.SignalState = 1;
247  Mutant->Abandoned = TRUE;
248 
249  } else {
250 
251  //
252  // If the Mutant object is not owned by the current thread, then
253  // unlock the dispatcher data base and raise an exception. Otherwise
254  // increment the ownership count.
255  //
256 
257  if (Mutant->OwnerThread != Thread) {
258  KiUnlockDispatcherDatabase(OldIrql);
259  ExRaiseStatus(Mutant->Abandoned ?
260  STATUS_ABANDONED : STATUS_MUTANT_NOT_OWNED);
261  }
262 
263  Mutant->Header.SignalState += 1;
264  }
265 
266  if (Mutant->Header.SignalState == 1) {
267  if (OldState <= 0) {
268  RemoveEntryList(&Mutant->MutantListEntry);
269  LeaveCriticalRegion = Mutant->ApcDisable;
270  }
271 
272  Mutant->OwnerThread = (PKTHREAD)NULL;
273  if (IsListEmpty(&Mutant->Header.WaitListHead) == FALSE) {
274  KiWaitTest(Mutant, Increment);
275  }
276  }
277 
278  //
279  // If the value of the Wait argument is TRUE, then return to
280  // caller with IRQL raised and the dispatcher database locked.
281  // Else release the dispatcher database lock and lower IRQL to
282  // its previous value.
283  //
284 
285  if (Wait != FALSE) {
286  Thread->WaitNext = Wait;
287  Thread->WaitIrql = OldIrql;
288 
289  } else {
290  KiUnlockDispatcherDatabase(OldIrql);
291  }
292 
293  //
294  // If kernel APC disable was biased when the mutant was acquired (mutex
295  // variant), then leave critical region.
296  //
297 
298  if (LeaveCriticalRegion != 0) {
299  KeLeaveCriticalRegionThread(Thread);
300  }
301 
302  //
303  // Return previous signal state of mutant object.
304  //
305 
306  return OldState;
307 }
