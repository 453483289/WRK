11  Oplock.c
12 
13 Abstract:
14 
15  The OPLOCK routines provide support to filesystems which implement
16  opportunistic locks. The specific actions needed are based on
17  the current oplocked state of the file (maintained in the OPLOCK
18  structure) and the Irp the Io system provides to the file systems.
19  Rather than define separate entry points for each oplock operation
20  a single generic entry point is defined.
21 
22  The file systems will maintain a variable of type OPLOCK for
23  each open file in the system. This variable is initialized
24  when an unopened file is opened. It is uninitialized when the
25  last reference to the file is cleared when the Io system calls
26  the file system with a close call.
27 
28  The following routines are provided by this package:
29 
30  o FsRtlInitializeOplock - Initialize a new OPLOCK structure. There
31  should be one OPLOCK for every opened file. Each OPLOCK structure
32  must be initialized before it can be used by the system.
33 
34  o FsRtlUninitializeOplock - Uninitialize an OPLOCK structure. This
35  call is used to cleanup any ancillary structures allocated and
36  maintained by the OPLOCK. After being uninitialized the OPLOCK
37  must again be initialized before it can be used by the system.
38 
39 --*/

409 VOID
410 FsRtlUninitializeOplock (
411  __inout POPLOCK Oplock
412  )
413 
414 /*++
415 
416 Routine Description:
417 
418  This routine uninitializes an OPLOCK structure. After calling this
419  routine, the OPLOCK structure must be reinitialized before being
420  used again.
421 
422 Arguments:
423 
424  Oplock - Supplies the address of an opaque OPLOCK structure.
425 
426 Return Value:
427 
428  None.
429 
430 --*/
431 
432 
433 {
434  PNONOPAQUE_OPLOCK ThisOplock;
435 
436  DebugTrace(+1, Dbg, "FsRtlUninitializeOplock: Oplock -> %08lx\n", *Oplock );
437 
438  //
439  // If the Oplock structure has not been allocated, there is no action
440  // to take.
441  //
442 
443  if (*Oplock != NULL) {
444 
445  //
446  // Remove this from the user's structure.
447  //
448 
449  ThisOplock = (PNONOPAQUE_OPLOCK) *Oplock;
450 
451  *Oplock = NULL;
452 
453  //
454  // Grab the waiting lock queue mutex to exclude anyone from messing
455  // with the queue while we're using it
456  //
457 
458  ExAcquireFastMutexUnsafe( ThisOplock->FastMutex );
459 
460  try {
461 
462  PIRP Irp;
463 
464  //
465  // Release any waiting Irps held.
466  //
467 
468  while (!IsListEmpty( &ThisOplock->WaitingIrps )) {
469 
470  PWAITING_IRP WaitingIrp;
471  PIRP ThisIrp;
472 
473  WaitingIrp = CONTAINING_RECORD( ThisOplock->WaitingIrps.Flink,
474  WAITING_IRP,
475  Links );
476 
477  RemoveHeadList( &ThisOplock->WaitingIrps );
478 
479  ThisIrp = WaitingIrp->Irp;
480 
481  IoAcquireCancelSpinLock( &ThisIrp->CancelIrql );
482 
483  IoSetCancelRoutine( ThisIrp, NULL );
484  IoReleaseCancelSpinLock( ThisIrp->CancelIrql );
485 
486  ThisIrp->IoStatus.Information = 0;
487 
488  //
489  // Call the completion routine in the Waiting Irp.
490  //
491 
492  WaitingIrp->CompletionRoutine( WaitingIrp->Context,
493  WaitingIrp->Irp );
494 
495  ExFreePool( WaitingIrp );
496  }
497 
498  //
499  // Release any oplock II irps held.
500  //
501 
502  while (!IsListEmpty( &ThisOplock->IrpOplocksII )) {
503 
504  Irp = CONTAINING_RECORD( ThisOplock->IrpOplocksII.Flink,
505  IRP,
506  Tail.Overlay.ListEntry );
507 
508  RemoveHeadList( &ThisOplock->IrpOplocksII );
509 
510  IoAcquireCancelSpinLock( &Irp->CancelIrql );
511 
512  IoSetCancelRoutine( Irp, NULL );
513  IoReleaseCancelSpinLock( Irp->CancelIrql );
514 
515  //
516  // Complete the oplock II Irp.
517  //
518 
519  ObDereferenceObject( IoGetCurrentIrpStackLocation( Irp )->FileObject );
520 
521  Irp->IoStatus.Information = FILE_OPLOCK_BROKEN_TO_NONE;
522  FsRtlCompleteRequest( Irp, STATUS_SUCCESS );
523  }
524 
525  //
526  // Release any exclusive oplock held.
527  //
528 
529  if (ThisOplock->IrpExclusiveOplock != NULL) {
530 
531  Irp = ThisOplock->IrpExclusiveOplock;
532 
533  IoAcquireCancelSpinLock( &Irp->CancelIrql );
534 
535  IoSetCancelRoutine( Irp, NULL );
536  IoReleaseCancelSpinLock( Irp->CancelIrql );
537 
538  Irp->IoStatus.Information = FILE_OPLOCK_BROKEN_TO_NONE;
539  FsRtlCompleteRequest( Irp, STATUS_SUCCESS );
540 
541  ThisOplock->IrpExclusiveOplock = NULL;
542 
543  if (ThisOplock->FileObject != NULL) {
544 
545  ObDereferenceObject( ThisOplock->FileObject );
546  }
547  }
548 
549  } finally {
550 
551  //
552  // No matter how we complete the preceding statements we will
553  // now release the waiting lock queue mutex
554  //
555 
556  ExReleaseFastMutexUnsafe( ThisOplock->FastMutex );
557  }
558 
559  //
560  // Deallocate the mutex.
561  //
562 
563  ExFreePool( ThisOplock->FastMutex );
564 
565  //
566  // Deallocate the Oplock structure.
567  //
568 
569  ExFreePool( ThisOplock );
570  }
571 
572  DebugTrace( -1, Dbg, "FsRtlUninitializeOplock: Exit\n", 0 );
573  return;
574 }

577 NTSTATUS
578 FsRtlOplockFsctrl (
579  __in POPLOCK Oplock,
580  __in PIRP Irp,
581  __in ULONG OpenCount
582  )
583 
584 /*++
585 
586 Routine Description:
587 
588  This is the interface with the filesystems for Fsctl calls, it handles
589  oplock requests, break acknowledgement and break notify.
590 
591 Arguments:
592 
593  Oplock - Supplies the address of the opaque OPLOCK structure.
594 
595  Irp - Supplies a pointer to the Irp which declares the requested
596  operation.
597 
598  OpenCount - This is the number of user handles on the file if we are requesting
599  an exclusive oplock. A non-zero value for a level II request indicates
600  that there are locks on the file.
601 
602 Return Value:
603 
604  NTSTATUS - Returns the result of this operation. If this is an Oplock
605  request which is granted, then STATUS_PENDING is returned.
606  If the Oplock isn't granted then STATUS_OPLOCK_NOT_GRANTED
607  is returned. If this is an Oplock I break to no oplock,
608  then STATUS_SUCCESS. If this is an Oplock I break to
609  Oplock II then STATUS_PENDING is returned. Other
610  error codes returned depend on the nature of the error.
611 
612  STATUS_CANCELLED is returned if the Irp is cancelled during
613  this operation.
614 
615  STATUS_SUCCESS is returned if this is a create asking for
616  a filter oplock.
617 
618 --*/
619 
620 {
621  NTSTATUS Status;
622  PIO_STACK_LOCATION IrpSp;
623  OPLOCK_STATE OplockState;
624 
625  PAGED_CODE();
626 
627  //
628  // Get the current IRP stack location
629  //
630 
631  IrpSp = IoGetCurrentIrpStackLocation( Irp );
632 
633  DebugTrace(+1, Dbg, "FsRtlOplockFsctrl: Entered\n", 0);
634  DebugTrace( 0, Dbg, "FsRtlOplockFsctrl: Oplock -> %08lx\n", *Oplock );
635  DebugTrace( 0, Dbg, "FsRtlOplockFsctrl: Irp -> %08lx\n", Irp );
636 
637  //
638  // Check if this is the create case where the user is requesting a pending
639  // filter oplock.
640  //
641 
642  if (IrpSp->MajorFunction == IRP_MJ_CREATE) {
643 
644  //
645  // Check that all the conditions hold to grant this oplock.
646  // The conditions that must hold are:
647  //
648  // - This is the only opener of the file.
649  // - Desired Access must be exactly FILE_READ_ATTRIBUTES.
650  // This will ensure an asynch open since the SYNCHRONIZE
651  // flag can't be set.
652  // - Share access is precisely
653  // (FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE)
654  //
655 
656  if ((OpenCount != 1) ||
657  (FlagOn( IrpSp->Parameters.Create.SecurityContext->DesiredAccess,
658  ~(FILE_READ_ATTRIBUTES))) ||
659  ((IrpSp->Parameters.Create.ShareAccess &
660  (FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE)) !=
661  (FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE))) {
662 
663  Status = STATUS_OPLOCK_NOT_GRANTED;
664 
665  } else {
666 
667  Status = FsRtlRequestExclusiveOplock( (PNONOPAQUE_OPLOCK *) Oplock,
668  IrpSp,
669  NULL,
670  OpFilterReqPending );
671  }
672 
673  //
674  // Case on the FsControlFile code control code.
675  //
676 
677  } else {
678 
679  //
680  // Assume this is an OplockLevel I.
681  //
682  // NOTE - This code depends on the defined bits for these oplock types.
683  // FILTER_OPLOCK = 4 * LEVEL_I_OPLOCK
684  // BATCH_OPLOCK = 2 * LEVEL_I_OPLOCK
685  //
686 
687  OplockState = LEVEL_I_OPLOCK;
688 
689  switch (IrpSp->Parameters.FileSystemControl.FsControlCode) {
690 
691  case FSCTL_REQUEST_FILTER_OPLOCK :
692 
693  OplockState *= 2;
694 
695  case FSCTL_REQUEST_BATCH_OPLOCK :
696 
697  OplockState *= 2;
698 
699  case FSCTL_REQUEST_OPLOCK_LEVEL_1 :
700 
701  //
702  // Set the other flags for an exclusive oplock.
703  //
704 
705  SetFlag( OplockState, EXCLUSIVE );
706 
707  //
708  // We short circuit the request if this request is treated
709  // synchronously or the open count is not 1. Otherwise the Io system
710  // will hold the return code until the Irp is completed.
711  //
712  // Also fail this if the flag is set which indicates that
713  // the IO system should copy data back to a user's buffer.
714  //
715  // If cleanup has occurred on this file, then we refuse
716  // the oplock request.
717  //
718 
719  if ((OpenCount != 1) ||
720  IoIsOperationSynchronous( Irp ) ||
721  FlagOn( Irp->Flags, IRP_INPUT_OPERATION ) ||
722  FlagOn( IrpSp->FileObject->Flags, FO_CLEANUP_COMPLETE )) {
723 
724  FsRtlCompleteRequest( Irp, STATUS_OPLOCK_NOT_GRANTED );
725  Status = STATUS_OPLOCK_NOT_GRANTED;
726 
727  } else {
728 
729  Status = FsRtlRequestExclusiveOplock( (PNONOPAQUE_OPLOCK *) Oplock,
730  IrpSp,
731  Irp,
732  OplockState );
733  }
734 
735  break;
736 
737  case FSCTL_REQUEST_OPLOCK_LEVEL_2 :
738 
739  //
740  // We short circuit the request if this request is treated
741  // synchronously. Otherwise the Io system will hold the return
742  // code until the Irp is completed.
743  //
744  // If cleanup has occurred on this file, then we refuse
745  // the oplock request.
746  //
747  // Also fail this if the flag is set which indicates that
748  // the IO system should copy data back to a user's buffer.
749  //
750  // A non-zero open count in this case indicates that there are
751  // file locks on the file. We will also fail the request in
752  // this case.
753  //
754 
755  if ((OpenCount != 0) ||
756  IoIsOperationSynchronous( Irp ) ||
757  FlagOn( Irp->Flags, IRP_INPUT_OPERATION ) ||
758  FlagOn( IrpSp->FileObject->Flags, FO_CLEANUP_COMPLETE )) {
759 
760  FsRtlCompleteRequest( Irp, STATUS_OPLOCK_NOT_GRANTED );
761  Status = STATUS_OPLOCK_NOT_GRANTED;
762 
763  } else {
764 
765  Status = FsRtlRequestOplockII( (PNONOPAQUE_OPLOCK *) Oplock,
766  IrpSp,
767  Irp );
768  }
769 
770  break;
771 
772  case FSCTL_OPLOCK_BREAK_ACKNOWLEDGE :
773 
774  Status = FsRtlAcknowledgeOplockBreak( (PNONOPAQUE_OPLOCK) *Oplock,
775  IrpSp,

825  __in_opt POPLOCK_FS_PREPOST_IRP PostIrpRoutine
826  )
827 
828 /*++
829 
830 Routine Description:
831 
832  This routine is called as a support routine from a file system.
833  It is used to synchronize I/O requests with the current Oplock
834  state of a file. If the I/O operation will cause the Oplock to
835  break, that action is initiated. If the operation cannot continue
836  until the Oplock break is complete, STATUS_PENDING is returned and
837  the caller supplied routine is called.
838

840 
841  Oplock - Supplies a pointer to the non-opaque oplock structure for
842  this file.
843 
844  Irp - Supplies a pointer to the Irp which declares the requested
845  operation.
846 
847  Context - This value is passed as a parameter to the completion routine.
848 
849  CompletionRoutine - This is the routine which is called if this
850  Irp must wait for an Oplock to break. This
851  is a synchronous operation if not specified
852  and we block in this thread waiting on
853  an event.
854 
855  PostIrpRoutine - This is the routine to call before we put anything
856  on our waiting Irp queue.
857 
858 Return Value:
859 
860  STATUS_SUCCESS if we can complete the operation on exiting this thread.
861  STATUS_PENDING if we return here but hold the Irp.
862  STATUS_CANCELLED if the Irp is cancelled before we return.
863 
864 --*/
865 
866 {
867  NTSTATUS Status = STATUS_SUCCESS;
868  PNONOPAQUE_OPLOCK ThisOplock = *Oplock;
869

871 
872  DebugTrace( +1, Dbg, "FsRtlCheckOplock: Entered\n", 0 );
873  DebugTrace( 0, Dbg, "Oplock -> %08lx\n", Oplock );
874  DebugTrace( 0, Dbg, "Irp -> %08lx\n", Irp );
875

877  // If there is no oplock structure or this is system I/O, we allow
878  // the operation to continue. Otherwise we check the major function code.
879  //
880

882  !FlagOn( Irp->Flags, IRP_PAGING_IO )) {
883 
884  OPLOCK_STATE OplockState;
885  PFILE_OBJECT OplockFileObject;
886 
887  BOOLEAN BreakToII;
888  BOOLEAN BreakToNone;
889 
890  ULONG CreateDisposition;
891 
892  //
893  // Capture the file object first and then the oplock state to perform
894  // the unsafe checks below. We capture the file object first in case
895  // there is an exclusive oplock break in progress. Otherwise the oplock
896  // state may indicate break in progress but it could complete by
897  // the time we snap the file object.
898  //

906  //

917  //

920

923  //
924  // If we are opening for attribute access only, we
925  // return status success. Always break the oplock if this caller
926  // wants a filter oplock.
927  //
928

930  ~(FILE_READ_ATTRIBUTES | FILE_WRITE_ATTRIBUTES | SYNCHRONIZE) ) &&
931  !FlagOn( IrpSp->Parameters.Create.Options, FILE_RESERVE_OPFILTER )) {
932 
933  break;

937  // If there is a filter oplock granted and this create iS reading
938  // the file then don't break the oplock as long as we share
939  // for reads.
940  //

943  !FlagOn( IrpSp->Parameters.Create.SecurityContext->DesiredAccess,
944  ~FILTER_OPLOCK_VALID_FLAGS ) &&
945  FlagOn( IrpSp->Parameters.Create.ShareAccess, FILE_SHARE_READ )) {
946 
947  break;
948  }
949 
950  //
951  // We we are superseding or overwriting, then break to none.
952  //

954  CreateDisposition = (IrpSp->Parameters.Create.Options >> 24) & 0x000000ff;
955 
956  if ((CreateDisposition == FILE_SUPERSEDE) ||
957  (CreateDisposition == FILE_OVERWRITE) ||
958  (CreateDisposition == FILE_OVERWRITE_IF) ||
959  FlagOn( IrpSp->Parameters.Create.Options, FILE_RESERVE_OPFILTER )) {
960

962 
963  } else {

966  }
967

974  // We will assume the oplock will have been broken

976  //
977

979

993  IrpSp );
994 
995  break;
996 
997  case IRP_MJ_LOCK_CONTROL :
998 
999  //
1000  // If a filter oplock has been granted then do nothing.
1001  // We will assume the oplock will have been broken
1002  // if this create needed to do that.
1003  //
1004 
1005  if (FlagOn( OplockState, FILTER_OPLOCK )) {
1006 
1007  break;
1008  }
1009 
1010  case IRP_MJ_WRITE :
1011 
1012  BreakToNone = TRUE;
1013  break;
1014 
1015  case IRP_MJ_SET_INFORMATION :
1016 
1017  //
1018  // We are only interested in calls that shrink the file size
1019  // or breaking batch oplocks for the rename case.
1020  //
1021

1023 
1024  case FileEndOfFileInformation :
1025 
1026  //
1027  // Break immediately if this is the lazy writer callback.
1028  //
1029 
1030  if (IrpSp->Parameters.SetFile.AdvanceOnly) {
1031 
1032  break;
1033  }
1034 
1035  case FileAllocationInformation :
1036 
1037  BreakToNone = TRUE;
1038  break;
1039 
1040  case FileRenameInformation :
1041  case FileLinkInformation :
1042  case FileShortNameInformation :
1043 
1044  if (FlagOn( OplockState, BATCH_OPLOCK | FILTER_OPLOCK )) {
1045 
1046  BreakToNone = TRUE;
1047  }
1048 
1049  break;
1050  }
1051 
1052  case IRP_MJ_FILE_SYSTEM_CONTROL :
1053 
1054  //
1055  // We need to break to none if this is a zeroing operation.
1056  //
1057 
1058  if (IrpSp->Parameters.FileSystemControl.FsControlCode == FSCTL_SET_ZERO_DATA) {
1059 
1060  BreakToNone = TRUE;
1061  }
1062  }
1063 
1064  if (BreakToII) {
1065 
1066  //
1067  // If there are no outstanding oplocks or level II oplocks are held,
1068  // we can return immediately. If the first two tests fail then there
1069  // is an exclusive oplock. If the file objects match we allow the
1070  // operation to continue.
1071  //
1072 
1073  if ((OplockState != OplockIIGranted) &&
1074  (OplockFileObject != IrpSp->FileObject)) {
1075 
1076  Status = FsRtlOplockBreakToII( (PNONOPAQUE_OPLOCK) *Oplock,
1077  IrpSp,
1078  Irp,
1079  Context,
1080  CompletionRoutine,
1081  PostIrpRoutine );
1082  }
1083 
1084  } else if (BreakToNone) {
1085 
1086  //
1087  // If there are no oplocks, we can return immediately.
1088  // Otherwise if there is no level 2 oplock and this file
1089  // object matches the owning file object then this write is
1090  // on behalf of the owner of the oplock.
1091  //
1092 
1093  if ((OplockState == OplockIIGranted) ||
1094  (OplockFileObject != IrpSp->FileObject)) {
1095 
1096  Status = FsRtlOplockBreakToNone( (PNONOPAQUE_OPLOCK) *Oplock,
1097  IrpSp,
1098  Irp,
1099  Context,
1100  CompletionRoutine,
1101  PostIrpRoutine );
1102  }
1103  }
1104  }
1105  }
1106 
1107  DebugTrace( -1, Dbg, "FsRtlCheckOplock: Exit -> %08lx\n", Status );
1108
