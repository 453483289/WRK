31 /*++
32 
33 Routine Description:
34 
35  This function initializes the machine dependent context of a thread
36  object.
37 
38  N.B. This function does not check if context record is accessible.
39  It is assumed the the caller of this routine is either prepared
40  to handle access violations or has probed and copied the context
41  record as appropriate.
42 
43 Arguments:
44 
45  Thread - Supplies a pointer to a dispatcher object of type thread.
46 
47  SystemRoutine - Supplies a pointer to the system function that is to be
48  called when the thread is first scheduled for execution.
49 
50  StartRoutine - Supplies an optional pointer to a function that is to be
51  called after the system has finished initializing the thread. This
52  parameter is specified if the thread is a system thread and will
53  execute totally in kernel mode.
54 
55  StartContext - Supplies an optional pointer to a data structure that
56  will be passed to the StartRoutine as a parameter. This parameter
57  is specified if the thread is a system thread and will execute
58  totally in kernel mode.
59 
60  ContextRecord - Supplies an optional pointer a context record which
61  contains the initial user mode state of the thread. This parameter
62  is specified if the thread will execute in user mode.
63 
64 Return Value:
65 
66  None.
67 
68 --*/
69 
70 {
71 
72  CONTEXT ContextFrame;
73  PKEXCEPTION_FRAME ExFrame;
74  ULONG64 InitialStack;
75  PXMM_SAVE_AREA32 NpxFrame;
76  PKSTART_FRAME SfFrame;
77  PKERNEL_STACK_CONTROL StackControl;
78  PKSWITCH_FRAME SwFrame;
79  PKTRAP_FRAME TrFrame;
80 
81  //
82  // Allocate a legacy floating point save area at the base of the thread
83  // stack and record the initial stack as this address. All threads have
84  // a legacy floating point save are to avoid special cases in the context
85  // switch code.
86  //
87 
88  InitialStack = (ULONG64)Thread->InitialStack;
89  NpxFrame = (PXMM_SAVE_AREA32)(InitialStack - KERNEL_STACK_CONTROL_LENGTH);
90  RtlZeroMemory(NpxFrame, KERNEL_STACK_CONTROL_LENGTH);
91 
92  //
93  // Initialize the current kernel stack segment descriptor in the kernel
94  // stack control area. This descriptor is used to control kernel stack
95  // expansion from drivers.
96  //
97  // N.B. The previous stack segment descriptor is zeroed.
98  //
99 
100  StackControl = (PKERNEL_STACK_CONTROL)NpxFrame;
101  StackControl->Current.StackBase = InitialStack;
102  StackControl->Current.ActualLimit = InitialStack - KERNEL_STACK_SIZE;
103 
104  //
105  // If a context record is specified, then initialize a trap frame, and
106  // an exception frame with the specified user mode context.
107  //
108  // N.B. The initial context of a thread cannot set the debug or floating
109  // state.
110  //
111 
112  if (ARGUMENT_PRESENT(ContextRecord)) {
113  ContextFrame = *ContextRecord;
114  ContextRecord = &ContextFrame;
115  ContextRecord->ContextFlags &= ~(CONTEXT_DEBUG_REGISTERS | CONTEXT_FLOATING_POINT);
116  ContextRecord->ContextFlags |= CONTEXT_CONTROL;
117 
118  //
119  // Allocate a trap frame, an exception frame, and a context switch
120  // frame.
121  //
122 
123  TrFrame = (PKTRAP_FRAME)(((ULONG64)NpxFrame - KTRAP_FRAME_LENGTH));
124  ExFrame = (PKEXCEPTION_FRAME)(((ULONG64)TrFrame - KEXCEPTION_FRAME_LENGTH));
125  SwFrame = (PKSWITCH_FRAME)(((ULONG64)ExFrame - KSWITCH_FRAME_LENGTH));
126 
127  //
128  // Set CS and SS for user mode 64-bit execution in the machine frame.
129  //
130 
131  ContextRecord->SegCs = KGDT64_R3_CODE | RPL_MASK;
132  ContextRecord->SegSs = KGDT64_R3_DATA | RPL_MASK;
133 
134  //
135  // The main entry point for the user thread will be jumped to via a
136  // continue operation from the user APC dispatcher. Therefore, the
137  // user stack must be initialized to an 8 mod 16 boundary.
138  //
139  // In addition, we must have room for the home addresses for the
140  // first four parameters.
141  //
142 
143  ContextRecord->Rsp =
144  (ContextRecord->Rsp & ~STACK_ROUND) - ((4 * 8) + 8);
145 
146  //
147  // Zero the exception and trap frames and copy information from the
148  // specified context frame to the trap and exception frames.
149  //
150  // N.B. The function that performs the context to kernel frames does
151  // not initialize the legacy floating context.
152  //
153 
154  RtlZeroMemory(ExFrame, sizeof(KEXCEPTION_FRAME));
155  RtlZeroMemory(TrFrame, sizeof(KTRAP_FRAME));
156  KxContextToKframes(TrFrame,
157  ExFrame,
158  ContextRecord,
159  ContextRecord->ContextFlags,
160  UserMode);
161 
162  //
163  // Initialize user thread startup information.
164  //
165 
166  ExFrame->P1Home = (ULONG64)StartContext;
167  ExFrame->P2Home = (ULONG64)StartRoutine;
168  ExFrame->P3Home = (ULONG64)SystemRoutine;
169  ExFrame->Return = (ULONG64)KiStartUserThreadReturn;
170 
171  //
172  // Initialize start address.
173  //
174 
175  SwFrame->Return = (ULONG64)KiStartUserThread;
176 
177  //
178  // Set the initial legacy floating point control/tag word state and
179  // the XMM control/status state.
180  //
181 
182  NpxFrame->ControlWord = INITIAL_FPCSR;
183  NpxFrame->MxCsr = INITIAL_MXCSR;
184  TrFrame->MxCsr = INITIAL_MXCSR;
185 
186  //
187  // Set legacy floating point state to switch.
188  //
189 
190  Thread->NpxState = LEGACY_STATE_SWITCH;
191 
192  //
193  // Set the saved previous processor mode in the thread object.
194  //
195 
196  Thread->PreviousMode = UserMode;
197 
198  } else {
199 
200  //
201  // Allocate an exception frame and a context switch frame.
202  //
203 
204  TrFrame = NULL;
205  SfFrame = (PKSTART_FRAME)(((ULONG64)NpxFrame - KSTART_FRAME_LENGTH));
206  SwFrame = (PKSWITCH_FRAME)(((ULONG64)SfFrame - KSWITCH_FRAME_LENGTH));
207 
208  //
209  // Initialize the system thread start frame.
210  //
211 
212  SfFrame->P1Home = (ULONG64)StartContext;
213  SfFrame->P2Home = (ULONG64)StartRoutine;
214  SfFrame->P3Home = (ULONG64)SystemRoutine;
215  SfFrame->Return = 0;
216 
217  //
218  // Initialize start address.
219  //
220 
221  SwFrame->Return = (ULONG64)KiStartSystemThread;
222 
223  //
224  // Set legacy floating point state to unused.
225  //
226 
227  Thread->NpxState = LEGACY_STATE_UNUSED;
228 
229  //
230  // Set the previous mode in thread object to kernel.
231  //
232 
233  Thread->PreviousMode = KernelMode;
234  }
235 
236  //
237  // Initialize context switch frame and set thread start up parameters.
238  //
239 
240  SwFrame->ApcBypass = APC_LEVEL;
241  SwFrame->Rbp = (ULONG64)TrFrame + 128;
242 
243  //
244  // Set the initial kernel stack pointer.
245  //
246 
247  Thread->InitialStack = (PVOID)NpxFrame;
248  Thread->KernelStack = SwFrame;
249  return;
250 }
