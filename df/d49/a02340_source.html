59 /*++
60 
61 Routine Description:
62 
63  This routine is invoked whenever a handle to a file is deleted. If the
64  handle being deleted is the last handle to the file (the ProcessHandleCount
65  parameter is one), then all locks for the file owned by the specified
66  process must be released.
67 
68  Likewise, if the SystemHandleCount is one then this is the last handle
69  for this for file object across all processes. For this case, the file
70  system is notified so that it can perform any necessary cleanup on the
71  file.
72 
73 Arguments:
74 
75  Process - A pointer to the process that closed the handle.
76 
77  Object - A pointer to the file object that the handle referenced.
78 
79  GrantedAccess - Access that was granted to the object through the handle.
80 
81  ProcessHandleCount - Count of handles outstanding to the object for the
82  process specified by the Process argument. If the count is one
83  then this is the last handle to this file by that process.
84 
85  SystemHandleCount - Count of handles outstanding to the object for the
86  entire system. If the count is one then this is the last handle
87  to this file in the system.
88 
89 Return Value:
90 
91  None.
92 
93 --*/

434 VOID
435 IopDeleteFile(
436  IN PVOID Object
437  )
438 
439 /*++
440 
441 Routine Description:
442 
443  This routine is invoked when the last handle to a specific file handle is
444  being closed and the file object is going away. It is the responsibility
445  of this routine to perform the following functions:
446 
447  o Notify the device driver that the file object is open on that the
448  file is being closed.
449 
450  o Dereference the user's error port for the file object, if there
451  is one associated with the file object.
452 
453  o Decrement the device object reference count.
454 
455 Arguments:
456 
457  Object - Pointer to the file object being deleted.
458 
459 Return Value:
460 
461  None.
462 
463 --*/
464 
465 {
466  PIRP irp;
467  PIO_STACK_LOCATION irpSp;
468  PDEVICE_OBJECT deviceObject;
469  PDEVICE_OBJECT fsDevice = NULL;
470  IO_STATUS_BLOCK ioStatusBlock;
471  KIRQL irql;
472  NTSTATUS status;
473  PFILE_OBJECT fileObject;
474  KEVENT event;
475  PVPB vpb;
476  BOOLEAN referenceCountDecremented;
477 
478  //
479  // Obtain a pointer to the file object.
480  //
481 
482  fileObject = (PFILE_OBJECT) Object;
483 
484  //
485  // Get a pointer to the first device driver which should be notified that
486  // this file is going away. If the device driver field is NULL, then this
487  // file is being shut down due to an error attempting to get it open in the
488  // first place, so do not do any further processing.
489  //
490 
491  if (fileObject->DeviceObject) {
492  if (!(fileObject->Flags & FO_DIRECT_DEVICE_OPEN)) {
493  deviceObject = IoGetRelatedDeviceObject( fileObject );
494  } else {
495  deviceObject = IoGetAttachedDevice( fileObject->DeviceObject );
496  }
497 
498  //
499  // On IopDeleteFile path the lock should always be ours as there should be
500  // no one else using this object.
501  //
502 
503  ASSERT (!(fileObject->Flags & FO_SYNCHRONOUS_IO) ||
504  (InterlockedExchange( (PLONG) &fileObject->Busy, (ULONG) TRUE ) == FALSE ));
505 
506  //
507  // If this file has never had a file handle created for it, and yet
508  // it exists, invoke the close file procedure so that the file system
509  // gets the cleanup IRP it is expecting before sending the close IRP.
510  //
511 
512  if (!(fileObject->Flags & (FO_HANDLE_CREATED | FO_FILE_OPEN_CANCELLED))) {
513  IopCloseFile( (PEPROCESS) NULL,
514  Object,
515  0,
516  1,
517  1 );
518  }
519 
520  //
521  // Reset a local event that can be used to wait for the device driver
522  // to close the file.
523  //
524 
525  KeInitializeEvent( &event, SynchronizationEvent, FALSE );
526 
527  //
528  // Reset the event in the file object.
529  //
530 
531  KeClearEvent( &fileObject->Event );
532 
533  //
534  // Allocate an I/O Request Packet (IRP) to be used in communicating with
535  // the appropriate device driver that the file is being closed. Notice
536  // that the allocation of this packet is done without charging quota so
537  // that the operation will not fail. This is done because there is no
538  // way to return an error to the caller at this point.
539  //
540 
541  irp = IoAllocateIrp( deviceObject->StackSize, FALSE );
542  if (!irp) {
543  irp = IopAllocateIrpMustSucceed( deviceObject->StackSize );
544  }
545 
546  //
547  // Get a pointer to the stack location for the first driver. This is
548  // where the function codes and parameters are placed.
549  //
550 
551  irpSp = IoGetNextIrpStackLocation( irp );
552 
553  //
554  // Fill in the IRP, indicating that this file object is being deleted.
555  //
556 
557  irpSp->MajorFunction = IRP_MJ_CLOSE;
558  irpSp->FileObject = fileObject;
559  irp->UserIosb = &ioStatusBlock;
560  irp->UserEvent = &event;
561  irp->Tail.Overlay.OriginalFileObject = fileObject;
562  irp->Tail.Overlay.Thread = PsGetCurrentThread();
563  irp->AssociatedIrp.SystemBuffer = (PVOID) NULL;
564  irp->Flags = IRP_CLOSE_OPERATION | IRP_SYNCHRONOUS_API;
565 
566  //
567  // Place this packet in the thread's I/O pending queue.
568  //
569 
570  IopQueueThreadIrp( irp );
571 
572  //
573  // Decrement the reference count on the VPB, if necessary. We
574  // have to do this BEFORE handing the Irp to the file system
575  // because of a trick the file systems play with close, and
576  // believe me, you really don't want to know what it is.
577  //
578  // Since there is not a error path here (close cannot fail),
579  // and the file system is the only ome who can actually synchronize
580  // with the actual completion of close processing, the file system
581  // is the one responsible for Vpb deletion.
582  //
583 
584  vpb = fileObject->Vpb;
585 
586 
587  if (vpb && !(fileObject->Flags & FO_DIRECT_DEVICE_OPEN)) {
588  IopInterlockedDecrementUlong( LockQueueIoVpbLock,
589  (PLONG) &vpb->ReferenceCount );
590 
591  //
592  // Bump the handle count of the filesystem volume device object.
593  // This will prevent the filesystem filter stack from being torn down
594  // until after the close IRP completes.
595  //
596 
597  fsDevice = vpb->DeviceObject;
598  if (fsDevice) {
599  IopInterlockedIncrementUlong( LockQueueIoDatabaseLock,
600  &fsDevice->ReferenceCount );
601  }
602  }
603 
604  //
605  // If this device object has stated for a fact that it knows it will
606  // never have the final non-zero reference count among the other
607  // device objects associated with our driver object, then decrement
608  // our reference count here BEFORE calling the file system. This
609  // is required because for a special class of device objects, the
610  // file system may delete them.
611  //
612 
613  if (fileObject->DeviceObject->Flags & DO_NEVER_LAST_DEVICE) {
614  IopInterlockedDecrementUlong( LockQueueIoDatabaseLock,
615  &fileObject->DeviceObject->ReferenceCount );
616 
617  referenceCountDecremented = TRUE;
618  } else {
619  referenceCountDecremented = FALSE;
620  }
621 
622  //
623  // Give the device driver the packet. If this request does not work,
624  // there is nothing that can be done about it. This is unfortunate
625  // because the driver may have had problems that it was about to
626  // report about other operations (e.g., write behind failures, etc.)
627  // that it can no longer report. The reason is that this routine
628  // is really initially invoked by NtClose, which has already closed
629  // the caller's handle, and that's what the return status from close
630  // indicates: the handle has successfully been closed.
631  //
632 
633  status = IoCallDriver( deviceObject, irp );
634 
635  if (status == STATUS_PENDING) {
636  (VOID) KeWaitForSingleObject( &event,
637  Executive,
638  KernelMode,
639  FALSE,
640  (PLARGE_INTEGER) NULL );
641  }
642 
643  //
644  // Perform any completion operations that need to be performed on
645  // the IRP that was used for this request. This is done here as
646  // as opposed to in normal completion code because there is a race
647  // condition between when this routine executes if it was invoked
648  // from a special kernel APC (e.g., some IRP was just completed and
649  // dereferenced this file object for the last time), and when the
650  // special kernel APC because of this packet's completion executing.
651  //
652  // This problem is solved by not having to queue a special kernel
653  // APC routine for completion of this packet. Rather, it is treated
654  // much like a synchronous paging I/O operation, except that the
655  // packet is not even freed during I/O completion. This is because
656  // the packet is still in this thread's queue, and there is no way
657  // to get it out except at APC_LEVEL. Unfortunately, the part of
658  // I/O completion that needs to dequeue the packet is running at
659  // DISPATCH_LEVEL.
660  //
661  // Hence, the packet must be removed from the queue (synchronized,
662  // of course), and then it must be freed.
663  //
664 
665  KeRaiseIrql( APC_LEVEL, &irql );
666  IopDequeueThreadIrp( irp );
667  KeLowerIrql( irql );
668 
669  IoFreeIrp( irp );
670 
671  //
672  // Free the file name string buffer if there was one.
673  //
674 
675  if (fileObject->FileName.Length != 0) {
676  ExFreePool( fileObject->FileName.Buffer );
677  }
678 
679  //
680  // If there was an completion port associated w/this file object, dereference
681  // it now, and deallocate the completion context pool.
682  //
683 
684  if (fileObject->CompletionContext) {
685  ObDereferenceObject( fileObject->CompletionContext->Port );
686  ExFreePool( fileObject->CompletionContext );
687  }
688 
689  //
690  // Free the file context control structure if it exists.
691  //
692 
693  if (fileObject->Flags & FO_FILE_OBJECT_HAS_EXTENSION) {
694  FsRtlPTeardownPerFileObjectContexts(fileObject);
695  }
696 
697  //
698  // Get a pointer to the real device object so its reference count
699  // can be decremented.
700  //
701 
702  deviceObject = fileObject->DeviceObject;
703 
704  //
705  // Decrement the reference count on the device object. Note that
706  // if the driver has been marked for an unload operation, and the
707  // reference count goes to zero, then the driver may need to be
708  // unloaded at this point.
709  //
710  // Note: only do this if the reference count was not already done
711  // above. The device object may be gone in this case.
712  //
713 
714  if (!referenceCountDecremented) {
715 
716  IopDecrementDeviceObjectRef(
717  deviceObject,
718  FALSE,
719  !ObIsObjectDeletionInline(Object)
720  );
721  }
722 
723  //
724  // Decrement the filesystem's volume device object handle count
725  // so that deletes can proceed.
726  //
727 
728  if (fsDevice && vpb) {
729  IopDecrementDeviceObjectRef(fsDevice,
730  FALSE,
731  !ObIsObjectDeletionInline(Object)
732  );
733  }
734  }
735 }

743 
744 Routine Description:
745 
746  This routine is invoked when the reference count for a driver object
747  becomes zero. That is, the last reference for the driver has gone away.
748  This routine ensures that the object is cleaned up and the driver
749  unloaded.

752

754  to zero.

758  None.
759

764  PIO_CLIENT_EXTENSION extension;
765  PIO_CLIENT_EXTENSION nextExtension;

924  IN PDEVICE_OBJECT DeviceObject,
925  IN PSECURITY_INFORMATION SecurityInformation,

927  IN POOL_TYPE PoolType,
928  IN PGENERIC_MAPPING GenericMapping

930 /*++
931 
932 Routine Description:
933 
934  This routine sets the security descriptor on a single device object
935 
936 Arguments:
937 
938  DeviceObject - pointer to base device object
939 
940  SecurityInformation - Fields of SD to change
941  SecurityDescriptor - New security descriptor
942  PoolType - Pool type for allocations
943  GenericMapping - Generic mapping for this object
944 
945 ReturnValue:
946 
947  success, or error while setting the descriptor for the device object.
948 
949 --*/
950 {
951 
952  PSECURITY_DESCRIPTOR OldDescriptor;
953  PSECURITY_DESCRIPTOR NewDescriptor;
954  PSECURITY_DESCRIPTOR CachedDescriptor;
955  NTSTATUS Status;
956  PKTHREAD CurrentThread;
957 
958  PAGED_CODE();
959 
960  //
961  // In order to preserve some protected fields in the SD (like the SACL) we need to make sure that only one
962  // thread updates it at any one time. If we didn't do this another modification could wipe out a SACL
963  // an administrator was adding.
964  //
965  CurrentThread = KeGetCurrentThread ();
966  while (1) {
967 
968  //
969  // Reference the security descriptor
970  //
971 
972  KeEnterCriticalRegionThread(CurrentThread);
973  ExAcquireResourceSharedLite( &IopSecurityResource, TRUE );
974 
975  OldDescriptor = DeviceObject->SecurityDescriptor;
976  if (OldDescriptor != NULL) {
977  ObReferenceSecurityDescriptor( OldDescriptor, 1 );
978  }
979 
980  ExReleaseResourceLite( &IopSecurityResource );
981  KeLeaveCriticalRegionThread(CurrentThread);
982 
983  NewDescriptor = OldDescriptor;
984 
985  Status = SeSetSecurityDescriptorInfo( NULL,
986  SecurityInformation,
987  SecurityDescriptor,
988  &NewDescriptor,
989  PoolType,
990  GenericMapping );
991  //
992  // If we successfully set the new security descriptor then we
993  // need to log it in our database and get yet another pointer
994  // to the finally security descriptor
995  //
996  if ( NT_SUCCESS( Status )) {
997  Status = ObLogSecurityDescriptor( NewDescriptor,
998  &CachedDescriptor,
999  1 );
1000  ExFreePool( NewDescriptor );
1001  if ( NT_SUCCESS( Status )) {
1002  //
1003  // Now we need to see if anyone else update this security descriptor inside the
1004  // gap where we didn't hold the lock. If they did then we just try it all again.
1005  //
1006  KeEnterCriticalRegionThread(CurrentThread);
1007  ExAcquireResourceExclusiveLite( &IopSecurityResource, TRUE );
1008 
1009  if (DeviceObject->SecurityDescriptor == OldDescriptor) {
1010  //
1011  // Do the swap
1012  //
1013  DeviceObject->SecurityDescriptor = CachedDescriptor;
1014 
1015  ExReleaseResourceLite( &IopSecurityResource );
1016  KeLeaveCriticalRegionThread(CurrentThread);
1017 
1018  //
1019  // If there was an original object then we need to work out how many
1020  // cached references there were (if any) and return them.
1021  //
1022  ObDereferenceSecurityDescriptor( OldDescriptor, 2 );
1023  break;
1024  } else {
1025 
1026  ExReleaseResourceLite( &IopSecurityResource );
1027  KeLeaveCriticalRegionThread(CurrentThread);
1028 
1029  ObDereferenceSecurityDescriptor( OldDescriptor, 1 );
1030  ObDereferenceSecurityDescriptor( CachedDescriptor, 1);
1031  }
1032 
1033  } else {
1034 
1035  //
1036  // Dereference old SecurityDescriptor
1037  //
1038 
1039  ObDereferenceSecurityDescriptor( OldDescriptor, 1 );
1040  break;
1041  }
1042  } else {
1043 
1044  //
1045  // Dereference old SecurityDescriptor
1046  //
1047  if (OldDescriptor != NULL) {
1048  ObDereferenceSecurityDescriptor( OldDescriptor, 1 );
1049  }
1050  break;
1051  }
1052  }
1053 
1054  //
1055  // And return to our caller
1056  //
1057 
1058  return( Status );
1059 }

1063 IopSetDeviceSecurityDescriptors(

1072 
1073 Routine Description:
1074 
1075  This routine sets the security descriptor on all the devices on the
1076  device stack for a PNP device. Ideally when the object manager asks the
1077  IO manager to set the security descriptor of a device object the IO manager
1078  should set the descriptor only on that device object. This is the classical
1079  behavior.
1080  Unfortunately for PNP devices there may be multiple devices on a device
1081  stack with names.
1082  If the descriptor is applied to only one of the devices on the stack its
1083  opens up a security hole as there may be other devices with name on the
1084  stack which can be opened by a random program.
1085  To protect against this the descriptor is applied to all device objects on
1086  the stack.
1087  Its important that to protect compatibility we need to return the same
1088  status as what would have been returned if only the requested device
1089  object's descriptor was set.
1090 
1091 Arguments:
1092 
1093  OriginalDeviceObject - Pointer to the device object passed by the object manager
1094  DeviceObject - pointer to base device object (first one to set)
1095  SecurityInformation )_ passed directly from IopGetSetSecurityObject
1096  SecurityDescriptor )
1097  PoolType )
1098  GenericMapping )

1101 
1102  success, or error while setting the descriptor for the original device object.
1103 
1104 --*/

1106  PDEVICE_OBJECT NewDeviceObject = NULL;
1107  NTSTATUS status;
1108  NTSTATUS returnStatus = STATUS_SUCCESS;
1109 
1110  ASSERT(DeviceObject);
1111 
1112  PAGED_CODE();

1114  //
1115  // pre-reference this object to match the dereference later
1116  //
1117 
1118  ObReferenceObject( DeviceObject );
1119

1121 
1122  //
1123  // Reference the existing security descriptor so it can't be reused.
1124  // We will only do the final security change if nobody else changes
1125  // the security while we don't hold the lock. Doing this prevents
1126  // privileged information being lost like the SACL.
1127  //
1128 
1129  //
1130  // Save away and return the device status only for the main device object
1131  // For example if OldSecurityDescriptor is NULL the IO manager should
1132  // return STATUS_NO_SECURITY_ON_OBJECT.
1133  //
1134

1137  SecurityDescriptor,
1138  PoolType,

1143  returnStatus = status;
1144  }

1148  // We don't need to acquire the database lock because
1149  // we have a handle to this device stack.
1150  //
1151

1153  if ( NewDeviceObject != NULL ) {
1154  ObReferenceObject( NewDeviceObject );
1155  }

1158  DeviceObject = NewDeviceObject;
1159 
1160  } while (NewDeviceObject);
