45 NtCreateIoCompletion (

47  __in ACCESS_MASK DesiredAccess,
48  __in_opt POBJECT_ATTRIBUTES ObjectAttributes,
49  __in ULONG Count OPTIONAL

52 /*++
53 
54 Routine Description:
55 
56  This function creates an I/O completion object, sets the maximum
57  target concurrent thread count to the specified value, and opens
58  a handle to the object with the specified desired access.
59 
60 Arguments:
61 
62  IoCompletionHandle - Supplies a pointer to a variable that receives
63  the I/O completion object handle.
64 
65  DesiredAccess - Supplies the desired types of access for the I/O
66  completion object.
67 
68  ObjectAttributes - Supplies a pointer to an object attributes structure.
69 
70  Count - Supplies the target maximum number of threads that should
71  be concurrently active. If this parameter is not specified, then
72  the number of processors is used.
73 
74 Return Value:
75 
76  STATUS_SUCCESS is returned if the function is success. Otherwise, an
77  error status is returned.
78 
79 --*/
80 
81 {
82 
83  HANDLE Handle;
84  KPROCESSOR_MODE PreviousMode;
85  PVOID IoCompletion;
86  NTSTATUS Status;
87 
88  //
89  // Establish an exception handler, probe the output handle address, and
90  // attempt to create an I/O completion object. If the probe fails, then
91  // return the exception code as the service status. Otherwise, return the
92  // status value returned by the object insertion routine.
93  //
94 
95  try {
96 
97  //
98  // Get previous processor mode and probe output handle address if
99  // necessary.
100  //
101 
102  PreviousMode = KeGetPreviousMode();
103  if (PreviousMode != KernelMode) {
104  ProbeForWriteHandle(IoCompletionHandle);
105  }
106 
107  //
108  // Allocate I/O completion object.
109  //
110 
111  Status = ObCreateObject(PreviousMode,
112  IoCompletionObjectType,
113  ObjectAttributes,
114  PreviousMode,
115  NULL,
116  sizeof(KQUEUE),
117  0,
118  0,
119  (PVOID *)&IoCompletion);
120 
121  //
122  // If the I/O completion object was successfully allocated, then
123  // initialize the object and attempt to insert it in the handle
124  // table of the current process.
125  //
126 
127  if (NT_SUCCESS(Status)) {
128  KeInitializeQueue((PKQUEUE)IoCompletion, Count);
129  Status = ObInsertObject(IoCompletion,
130  NULL,
131  DesiredAccess,
132  0,
133  (PVOID *)NULL,
134  &Handle);
135 
136  //
137  // If the I/O completion object was successfully inserted in
138  // the handle table of the current process, then attempt to
139  // write the handle value. If the write attempt fails, then
140  // do not report an error. When the caller attempts to access
141  // the handle value, an access violation will occur.
142  //
143 
144  if (NT_SUCCESS(Status)) {
145  try {
146  *IoCompletionHandle = Handle;
147 
148  } except(ExSystemExceptionFilter()) {
149  NOTHING;
150  }
151  }
152  }
153 
154  //
155  // If an exception occurs during the probe of the output handle address,
156  // then always handle the exception and return the exception code as the
157  // status value.
158  //
159 
160  } except(ExSystemExceptionFilter()) {
161  Status = GetExceptionCode();
162  }
163 
164  //
165  // Return service status.
166  //
167 
168  return Status;
169 }

275 NtQueryIoCompletion (
276  __in HANDLE IoCompletionHandle,
277  __in IO_COMPLETION_INFORMATION_CLASS IoCompletionInformationClass,
278  __out_bcount(IoCompletionInformationLength) PVOID IoCompletionInformation,
279  __in ULONG IoCompletionInformationLength,
280  __out_opt PULONG ReturnLength
281  )
282 
283 /*++
284 
285 Routine Description:
286 
287  This function queries the state of an I/O completion object and returns
288  the requested information in the specified record structure.
289 
290 Arguments:
291 
292  IoCompletionHandle - Supplies a handle to an I/O completion object.
293 
294  IoCompletionInformationClass - Supplies the class of information being
295  requested.
296 
297  IoCompletionInformation - Supplies a pointer to a record that receives
298  the requested information.
299 
300  IoCompletionInformationLength - Supplies the length of the record that
301  receives the requested information.
302 
303  ReturnLength - Supplies an optional pointer to a variable that receives
304  the actual length of the information that is returned.
305 
306 Return Value:
307 
308  STATUS_SUCCESS is returned if the function is success. Otherwise, an
309  error status is returned.
310 
311 --*/
312 
313 {
314 
315  PVOID IoCompletion;
316  LONG Depth;
317  KPROCESSOR_MODE PreviousMode;
318  NTSTATUS Status;
319 
320  //
321  // Establish an exception handler, probe the output arguments, reference
322  // the I/O completion object, and return the specified information. If
323  // the probe fails, then return the exception code as the service status.
324  // Otherwise return the status value returned by the reference object by
325  // handle routine.
326  //
327 
328  try {
329 
330  //
331  // Get previous processor mode and probe output arguments if necessary.
332  //
333 
334  PreviousMode = KeGetPreviousMode();
335  if (PreviousMode != KernelMode) {
336 
337 C_ASSERT(sizeof(IO_COMPLETION_BASIC_INFORMATION) == sizeof(ULONG));
338 
339  ProbeForWriteUlongAligned32((PULONG)IoCompletionInformation);
340 
341  if (ARGUMENT_PRESENT(ReturnLength)) {
342  ProbeForWriteUlong(ReturnLength);
343  }
344  }
345 
346  //
347  // Check argument validity.
348  //
349 
350  if (IoCompletionInformationClass != IoCompletionBasicInformation) {
351  return STATUS_INVALID_INFO_CLASS;
352  }
353 
354  if (IoCompletionInformationLength != sizeof(IO_COMPLETION_BASIC_INFORMATION)) {
355  return STATUS_INFO_LENGTH_MISMATCH;
356  }
357 
358  //
359  // Reference the I/O completion object by handle.
360  //
361 
362  Status = ObReferenceObjectByHandle(IoCompletionHandle,
363  IO_COMPLETION_QUERY_STATE,
364  IoCompletionObjectType,
365  PreviousMode,
366  &IoCompletion,
367  NULL);
368 
369  //
370  // If the reference was successful, then read the current state of
371  // the I/O completion object, dereference the I/O completion object,
372  // fill in the information structure, and return the structure length
373  // if specified. If the write of the I/O completion information or
374  // the return length fails, then do not report an error. When the
375  // caller accesses the information structure or length an access
376  // violation will occur.
377  //
378 
379  if (NT_SUCCESS(Status)) {
380  Depth = KeReadStateQueue((PKQUEUE)IoCompletion);
381  ObDereferenceObject(IoCompletion);
382  try {
383  ((PIO_COMPLETION_BASIC_INFORMATION)IoCompletionInformation)->Depth = Depth;
384  if (ARGUMENT_PRESENT(ReturnLength)) {
385  *ReturnLength = sizeof(IO_COMPLETION_BASIC_INFORMATION);
386  }
387 
388  } except(ExSystemExceptionFilter()) {
389  NOTHING;
390  }
391  }
392 
393  //
394  // If an exception occurs during the probe of the output arguments, then
395  // always handle the exception and return the exception code as the status
396  // value.
397  //
398 
399  } except(ExSystemExceptionFilter()) {
400  Status = GetExceptionCode();
401  }
402 
403  //
404  // Return service status.
405  //
406 
407  return Status;
408 }

478 NTSTATUS
479 NtRemoveIoCompletion (
480  __in HANDLE IoCompletionHandle,
481  __out PVOID *KeyContext,
482  __out PVOID *ApcContext,
483  __out PIO_STATUS_BLOCK IoStatusBlock,
484  __in_opt PLARGE_INTEGER Timeout
485  )
486 
487 /*++
488 
489 Routine Description:
490 
491  This function removes an entry from an I/O completion object. If there
492  are currently no entries available, then the calling thread waits for
493  an entry.
494 
495 Arguments:
496 
497  Completion - Supplies a handle to an I/O completion object.
498 
499  KeyContext - Supplies a pointer to a variable that receives the key
500  context that was specified when the I/O completion object was
501  associated with a file object.
502 
503  ApcContext - Supplies a pointer to a variable that receives the
504  context that was specified when the I/O operation was issued.
505 
506  IoStatus - Supplies a pointer to a variable that receives the
507  I/O completion status.
508 
509  Timeout - Supplies a pointer to an optional time out value.
510 
511 Return Value:
512 
513  STATUS_SUCCESS is returned if the function is success. Otherwise, an
514  error status is returned.
515 
516 --*/
517 
518 {
519 
520  PLARGE_INTEGER CapturedTimeout;
521  PLIST_ENTRY Entry;
522  PVOID IoCompletion;
523  PIRP Irp;
524  KPROCESSOR_MODE PreviousMode;
525  NTSTATUS Status;
526  LARGE_INTEGER TimeoutValue;
527  PVOID LocalApcContext;
528  PVOID LocalKeyContext;
529  IO_STATUS_BLOCK LocalIoStatusBlock;
530  PIOP_MINI_COMPLETION_PACKET MiniPacket;
531 
532  //
533  // Establish an exception handler, probe the I/O context, the I/O
534  // status, and the optional timeout value if specified, reference
535  // the I/O completion object, and attempt to remove an entry from
536  // the I/O completion object. If the probe fails, then return the
537  // exception code as the service status. Otherwise, return a value
538  // dependent on the outcome of the queue removal.
539  //
540 
541  try {
542 
543  //
544  // Get previous processor mode and probe the I/O context, status,
545  // and timeout if necessary.
546  //
547 
548  CapturedTimeout = NULL;
549  PreviousMode = KeGetPreviousMode();
550  if (PreviousMode != KernelMode) {
551  ProbeForWriteUlong_ptr((PULONG_PTR)ApcContext);
552  ProbeForWriteUlong_ptr((PULONG_PTR)KeyContext);
553  ProbeForWriteIoStatus(IoStatusBlock);
554  if (ARGUMENT_PRESENT(Timeout)) {
555  CapturedTimeout = &TimeoutValue;
556  TimeoutValue = ProbeAndReadLargeInteger(Timeout);
557  }
558 
559  } else{
560  if (ARGUMENT_PRESENT(Timeout)) {
561  CapturedTimeout = Timeout;
562  }
563  }
564 
565  //
566  // Reference the I/O completion object by handle.
567  //
568 
569  Status = ObReferenceObjectByHandle(IoCompletionHandle,
570  IO_COMPLETION_MODIFY_STATE,
571  IoCompletionObjectType,
572  PreviousMode,
573  &IoCompletion,
574  NULL);
575 
576  //
577  // If the reference was successful, then attempt to remove an entry
578  // from the I/O completion object. If an entry is removed from the
579  // I/O completion object, then capture the completion information,
580  // release the associated IRP, and attempt to write the completion
581  // information. If the write of the completion information fails,
582  // then do not report an error. When the caller attempts to access
583  // the completion information, an access violation will occur.
584  //
585 
586  if (NT_SUCCESS(Status)) {
587  Entry = KeRemoveQueue((PKQUEUE)IoCompletion,
588  PreviousMode,
589  CapturedTimeout);
590 
591  //
592  // N.B. The entry value returned can be the address of a list
593  // entry, STATUS_USER_APC, or STATUS_TIMEOUT.
594  //
595 
596  if (((LONG_PTR)Entry == STATUS_TIMEOUT) ||
597  ((LONG_PTR)Entry == STATUS_USER_APC)) {
598  Status = (NTSTATUS)((LONG_PTR)Entry);
599 
600  } else {
601 
602  //
603  // Set the completion status, capture the completion
604  // information, deallocate the associated IRP, and
605  // attempt to write the completion information.
606  //
607 
608  Status = STATUS_SUCCESS;
609  try {
610  MiniPacket = CONTAINING_RECORD(Entry,
611  IOP_MINI_COMPLETION_PACKET,
612  ListEntry);
613 
614  if ( MiniPacket->PacketType == IopCompletionPacketIrp ) {
615  Irp = CONTAINING_RECORD(Entry, IRP, Tail.Overlay.ListEntry);
616  LocalApcContext = Irp->Overlay.AsynchronousParameters.UserApcContext;
617  LocalKeyContext = (PVOID)Irp->Tail.CompletionKey;
618  LocalIoStatusBlock = Irp->IoStatus;
619  IoFreeIrp(Irp);
620 
621  } else {
622 
623  LocalApcContext = MiniPacket->ApcContext;
624  LocalKeyContext = (PVOID)MiniPacket->KeyContext;
625  LocalIoStatusBlock.Status = MiniPacket->IoStatus;
626  LocalIoStatusBlock.Information = MiniPacket->IoStatusInformation;
627  IopFreeMiniPacket(MiniPacket);
628  }
629 
630  *ApcContext = LocalApcContext;
631  *KeyContext = LocalKeyContext;
632  *IoStatusBlock = LocalIoStatusBlock;
633 
634  } except(ExSystemExceptionFilter()) {
635  NOTHING;
636  }
637  }
638 
639  //
640  // Deference I/O completion object.
641  //
642 
643  ObDereferenceObject(IoCompletion);
644  }
645 
646  //
647  // If an exception occurs during the probe of the previous count, then
648  // always handle the exception and return the exception code as the status
649  // value.
650  //
651 
652  } except(ExSystemExceptionFilter()) {
653  Status = GetExceptionCode();
654  }
655 
656  //
657  // Return service status.
658  //
659 
660  return Status;
661 }

665 IoSetIoCompletion (
666  IN PVOID IoCompletion,
667  IN PVOID KeyContext,
668  IN PVOID ApcContext,
669  IN NTSTATUS IoStatus,
670  IN ULONG_PTR IoStatusInformation,
671  IN BOOLEAN Quota
672  )
673 /*++
674 
675 Routine Description:
676 
677  This function allows the caller to queue an Irp to an I/O completion
678  port and specify all of the information that is returned out the other
679  end using NtRemoveIoCompletion.
680 
681 Arguments:
682 
683  IoCompletion - Supplies a a pointer to the completion port that the caller
684  intends to queue a completion packet to.
685 
686  KeyContext - Supplies the key context that is returned during a call
687  to NtRemoveIoCompletion.
688 
689  ApcContext - Supplies the apc context that is returned during a call
690  to NtRemoveIoCompletion.
691 
692  IoStatus - Supplies the IoStatus->Status data that is returned during
693  a call to NtRemoveIoCompletion.
694 
695  IoStatusInformation - Supplies the IoStatus->Information data that
696  is returned during a call to NtRemoveIoCompletion.
697 
698 Return Value:
699 
700  STATUS_SUCCESS is returned if the function is success. Otherwise, an
701  error status is returned.
702 
703 --*/
704 
705 {
706 
707  PGENERAL_LOOKASIDE Lookaside;
708  PIOP_MINI_COMPLETION_PACKET MiniPacket;
709  ULONG PacketType;
710  PKPRCB Prcb;
711  NTSTATUS Status = STATUS_SUCCESS;
712 
713  PAGED_CODE();
714 
715  //
716  // Attempt to allocate the minpacket from the per processor lookaside list.
717  //
718 
719  PacketType = IopCompletionPacketMini;
720  Prcb = KeGetCurrentPrcb();
721  Lookaside = Prcb->PPLookasideList[LookasideCompletionList].P;
722  Lookaside->TotalAllocates += 1;
723  MiniPacket = (PVOID)InterlockedPopEntrySList(&Lookaside->ListHead);
724 
725  //
726  // If the per processor lookaside list allocation failed, then attempt to
727  // allocate from the system lookaside list.
728  //
729 
730  if (MiniPacket == NULL) {
731  Lookaside->AllocateMisses += 1;
732  Lookaside = Prcb->PPLookasideList[LookasideCompletionList].L;
733  Lookaside->TotalAllocates += 1;
734  MiniPacket = (PVOID)InterlockedPopEntrySList(&Lookaside->ListHead);
735  }
736 
737  //
738  // If both lookaside allocation attempts failed, then attempt to allocate
739  // from pool.
740  //
741 
742  if (MiniPacket == NULL) {
743  Lookaside->AllocateMisses += 1;
744 
745  //
746  // If quota is specified, then allocate pool with quota charged.
747  // Otherwise, allocate pool without quota.
748  //
749 
750  if (Quota != FALSE) {
751  PacketType = IopCompletionPacketQuota;
752  try {
753  MiniPacket = ExAllocatePoolWithQuotaTag(NonPagedPool,
754  sizeof(*MiniPacket),
755  ' pcI');
756 
757  } except(EXCEPTION_EXECUTE_HANDLER) {
758  NOTHING;
759  }
760 
761  } else {
762  MiniPacket = ExAllocatePoolWithTagPriority(NonPagedPool,
763  sizeof(*MiniPacket),
764  ' pcI',
765  LowPoolPriority);
766  }
767  }
768 
769  //
770  // If a minipacket was successfully allocated, then initialize and
771  // queue the packet to the specified I/O completion queue.
772  //
773 
774  if (MiniPacket != NULL) {
775  MiniPacket->PacketType = PacketType;
776  MiniPacket->KeyContext = KeyContext;
777  MiniPacket->ApcContext = ApcContext;
778  MiniPacket->IoStatus = IoStatus;
779  MiniPacket->IoStatusInformation = IoStatusInformation;
780  KeInsertQueue((PKQUEUE)IoCompletion, &MiniPacket->ListEntry);
781 
782  } else {
783  Status = STATUS_INSUFFICIENT_RESOURCES;
784  }
785 
786  return Status;
787 }

789 VOID
790 IopFreeMiniPacket (
791  PIOP_MINI_COMPLETION_PACKET MiniPacket
792  )
793 
794 /*++
795 
796 Routine Description:
797 
798  This function free the specified I/O completion packet.
799 
800 Arguments:
801 
802  MiniPacket - Supplies a pointer to an I/O completion minipacket.
803 
804 Return Value:
805 
806  None.
807 
808 --*/
809 
810 {
811 
812  PGENERAL_LOOKASIDE Lookaside;
813  PKPRCB Prcb;
814 
815  //
816  // If the minipacket cannot be returned to either the per processor or
817  // system lookaside list, then free the minipacket to pool. Otherwise,
818  // release the quota if quota was allocated and push the entry onto
819  // one of the lookaside lists.
820  //
821 
822  Prcb = KeGetCurrentPrcb();
823  Lookaside = Prcb->PPLookasideList[LookasideCompletionList].P;
824  Lookaside->TotalFrees += 1;
825  if (ExQueryDepthSList(&Lookaside->ListHead) >= Lookaside->Depth) {
826  Lookaside->FreeMisses += 1;
827  Lookaside = Prcb->PPLookasideList[LookasideCompletionList].L;
828  Lookaside->TotalFrees += 1;
829  if (ExQueryDepthSList(&Lookaside->ListHead) >= Lookaside->Depth) {
830  Lookaside->FreeMisses += 1;
831  ExFreePool(MiniPacket);
832 
833  } else {
834  if (MiniPacket->PacketType == IopCompletionPacketQuota) {
835  ExReturnPoolQuota(MiniPacket);
836  }
837 
838  InterlockedPushEntrySList(&Lookaside->ListHead,
839  (PSLIST_ENTRY)MiniPacket);
840  }
841 
842  } else {
843  if (MiniPacket->PacketType == IopCompletionPacketQuota) {
844  ExReturnPoolQuota(MiniPacket);
845  }
846 
847  InterlockedPushEntrySList(&Lookaside->ListHead,
848  (PSLIST_ENTRY)MiniPacket);
849  }
850 
851  return;
852 }

860 
861 Routine Description:
