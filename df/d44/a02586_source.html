73 NtCreateJobObject (
74  __out PHANDLE JobHandle,
75  __in ACCESS_MASK DesiredAccess,
76  __in_opt POBJECT_ATTRIBUTES ObjectAttributes
77  )
78 {
79 
80  PEJOB Job;
81  HANDLE Handle;
82  KPROCESSOR_MODE PreviousMode;
83  NTSTATUS Status;
84  PETHREAD CurrentThread;
85 
86  PAGED_CODE();
87 
88  //
89  // Establish an exception handler, probe the output handle address, and
90  // attempt to create a job object. If the probe fails, then return the
91  // exception code as the service status. Otherwise return the status value
92  // returned by the object insertion routine.
93  //
94 
95  CurrentThread = PsGetCurrentThread ();
96  PreviousMode = KeGetPreviousModeByThread (&CurrentThread->Tcb);
97  try {
98 
99  //
100  // Probe output handle address if
101  // necessary.
102  //
103 
104  if (PreviousMode != KernelMode) {
105  ProbeForWriteHandle (JobHandle);
106  }
107  *JobHandle = NULL;
108 
109  } except (ExSystemExceptionFilter ()) {
110  return GetExceptionCode();
111  }
112 
113  //
114  // Allocate job object.
115  //
116 
117  Status = ObCreateObject (PreviousMode,
118  PsJobType,
119  ObjectAttributes,
120  PreviousMode,
121  NULL,
122  sizeof (EJOB),
123  0,
124  0,
125  &Job);
126 
127  //
128  // If the job object was successfully allocated, then initialize it
129  // and attempt to insert the job object in the current
130  // process' handle table.
131  //
132 
133  if (NT_SUCCESS(Status)) {
134 
135  RtlZeroMemory (Job, sizeof (EJOB));
136  InitializeListHead (&Job->ProcessListHead);
137  InitializeListHead (&Job->JobSetLinks);
138  KeInitializeEvent (&Job->Event, NotificationEvent, FALSE);
139  PspInitializeJobLimitsLock (Job);
140 
141  //
142  // Job Object gets the SessionId of the Process creating the Job
143  // We will use this sessionid to restrict the processes that can
144  // be added to a job.
145  //
146  Job->SessionId = MmGetSessionId (PsGetCurrentProcessByThread (CurrentThread));
147 
148  //
149  // Initialize the scheduling class for the Job
150  //
151  Job->SchedulingClass = PSP_DEFAULT_SCHEDULING_CLASSES;
152 
153  ExInitializeResourceLite (&Job->JobLock);
154 
155  PspLockJobListExclusive (CurrentThread);
156 
157  InsertTailList (&PspJobList, &Job->JobLinks);
158 
159  PspUnlockJobListExclusive (CurrentThread);
160 
161 
162  Status = ObInsertObject (Job,
163  NULL,
164  DesiredAccess,
165  0,
166  NULL,
167  &Handle);
168 
169  //
170  // If the job object was successfully inserted in the current
171  // process' handle table, then attempt to write the job object
172  // handle value.
173  //
174  if (NT_SUCCESS (Status)) {
175  try {
176  *JobHandle = Handle;
177  } except (ExSystemExceptionFilter ()) {
178  Status = GetExceptionCode ();
179  }
180  }
181  }
182  //
183  // Return service status.
184  //
185 
186  return Status;
187 }

191 NtOpenJobObject(
192  __out PHANDLE JobHandle,
193  __in ACCESS_MASK DesiredAccess,
194  __in POBJECT_ATTRIBUTES ObjectAttributes
195  )
196 {
197  HANDLE Handle;
198  KPROCESSOR_MODE PreviousMode;
199  NTSTATUS Status;
200 
201  PAGED_CODE();
202 
203  //
204  // Establish an exception handler, probe the output handle address, and
205  // attempt to open the job object. If the probe fails, then return the
206  // exception code as the service status. Otherwise return the status value
207  // returned by the object open routine.
208  //
209 
210  PreviousMode = KeGetPreviousMode ();
211 
212  if (PreviousMode != KernelMode) {
213  try {
214 
215  //
216  // Probe output handle address
217  // if necessary.
218  //
219 
220  ProbeForWriteHandle (JobHandle);
221 
222  } except (EXCEPTION_EXECUTE_HANDLER) {
223 
224  //
225  // If an exception occurs during the probe of the output job handle,
226  // then always handle the exception and return the exception code as the
227  // status value.
228  //
229 
230  return GetExceptionCode ();
231  }
232  }
233 
234 
235  //
236  // Open handle to the event object with the specified desired access.
237  //
238 
239  Status = ObOpenObjectByName (ObjectAttributes,
240  PsJobType,
241  PreviousMode,
242  NULL,
243  DesiredAccess,
244  NULL,
245  &Handle);
246 
247  //
248  // If the open was successful, then attempt to write the job object
249  // handle value. If the write attempt fails then just report an error.
250  // When the caller attempts to access the handle value, an
251  // access violation will occur.
252  //
253 
254  if (NT_SUCCESS (Status)) {
255  try {
256  *JobHandle = Handle;
257  } except(ExSystemExceptionFilter ()) {
258  return GetExceptionCode ();
259  }
260  }
261 
262  return Status;
263 }

267 NtAssignProcessToJobObject(
268  __in HANDLE JobHandle,
269  __in HANDLE ProcessHandle
270  )
271 {
272  PEJOB Job;
273  PEPROCESS Process;
274  PETHREAD CurrentThread;
275  NTSTATUS Status, Status1;
276  KPROCESSOR_MODE PreviousMode;
277  BOOLEAN IsAdmin;
278  PACCESS_TOKEN JobToken, NewToken = NULL;
279  ULONG SessionId;
280 
281  PAGED_CODE();
282 
283  CurrentThread = PsGetCurrentThread ();
284 
285  PreviousMode = KeGetPreviousModeByThread (&CurrentThread->Tcb);
286 
287  //
288  // Now reference the job object. Then we need to lock the process and check again
289  //
290 
291  Status = ObReferenceObjectByHandle (JobHandle,
292  JOB_OBJECT_ASSIGN_PROCESS,
293  PsJobType,
294  PreviousMode,
295  &Job,
296  NULL);
297  if (!NT_SUCCESS (Status)) {
298  return Status;
299  }
300 
301  JobToken = Job->Token;
302 
303  //
304  // Reference the process object, lock the process, test for already been assigned
305  //
306 
307  Status = ObReferenceObjectByHandle (ProcessHandle,
308  PROCESS_SET_QUOTA | PROCESS_TERMINATE |
309  ((JobToken != NULL)?PROCESS_SET_INFORMATION:0),
310  PsProcessType,
311  PreviousMode,
312  &Process,
313  NULL);
314  if (!NT_SUCCESS (Status)) {
315  ObDereferenceObject (Job);
316  return Status;
317  }
318 
319  //
320  // Quick Check for prior assignment
321  //
322 
323  if (Process->Job) {
324  Status = STATUS_ACCESS_DENIED;
325  goto deref_and_return_status;
326  }
327 
328  //
329  // We only allow a process that is running in the Job creator's hydra session
330  // to be assigned to the job.
331  //
332 
333  SessionId = MmGetSessionId (Process);
334  if (SessionId != Job->SessionId) {
335  Status = STATUS_ACCESS_DENIED;
336  goto deref_and_return_status;
337  }
338 
339  //
340  // Security Rules: If the job has no-admin set, and it is running
341  // as admin, that's not allowed
342  //
343 
344  if (Job->SecurityLimitFlags & JOB_OBJECT_SECURITY_NO_ADMIN) {
345  PACCESS_TOKEN Token;
346 
347  Token = PsReferencePrimaryToken (Process);
348 
349  IsAdmin = SeTokenIsAdmin (Token);
350 
351  PsDereferencePrimaryTokenEx (Process, Token);
352 
353  if (IsAdmin) {
354  Status = STATUS_ACCESS_DENIED;
355  goto deref_and_return_status;
356  }
357  }
358 
359  //
360  // Duplicate the primary token so we can assign it to the process.
361  //
362  if (JobToken != NULL) {
363  Status = SeSubProcessToken (JobToken,
364  &NewToken,
365  FALSE,
366  SessionId);
367 
368  if (!NT_SUCCESS (Status)) {
369  goto deref_and_return_status;
370  }
371  }
372 
373  if (!ExAcquireRundownProtection (&Process->RundownProtect)) {
374  Status = STATUS_PROCESS_IS_TERMINATING;
375  if (JobToken != NULL) {
376  ObDereferenceObject (NewToken);
377  }
378  goto deref_and_return_status;
379  }
380 
381 
382  //
383  // ref the job for the process
384  //
385 
386  ObReferenceObject (Job);
387 
388  if (InterlockedCompareExchangePointer (&Process->Job, Job, NULL) != NULL) {
389  ExReleaseRundownProtection (&Process->RundownProtect);
390  ObDereferenceObject (Process);
391  ObDereferenceObjectEx (Job, 2);
392  if (JobToken != NULL) {
393  ObDereferenceObject (NewToken);
394  }
395  return STATUS_ACCESS_DENIED;
396  }
397  //
398  // If the job has a token filter established,
399  // use it to filter the
400  //
401  ExReleaseRundownProtection (&Process->RundownProtect);
402 
403  Status = PspAddProcessToJob (Job, Process);
404  if (!NT_SUCCESS (Status)) {
405 
406  Status1 = PspTerminateProcess (Process, ERROR_NOT_ENOUGH_QUOTA);
407  if (NT_SUCCESS (Status1)) {
408 
409  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
410  ExAcquireResourceExclusiveLite (&Job->JobLock, TRUE);
411 
412  Job->TotalTerminatedProcesses++;
413 
414  ExReleaseResourceLite (&Job->JobLock);
415  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
416  }
417  }
418 
419  //
420  // If the job has UI restrictions and this is a GUI process, call ntuser
421  //
422  if ((Job->UIRestrictionsClass != JOB_OBJECT_UILIMIT_NONE) &&
423  (Process->Win32Process != NULL)) {
424  WIN32_JOBCALLOUT_PARAMETERS Parms;
425 
426  Parms.Job = Job;
427  Parms.CalloutType = PsW32JobCalloutAddProcess;
428  Parms.Data = Process->Win32Process;
429 
430  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
431  ExAcquireResourceExclusiveLite(&Job->JobLock, TRUE);
432 
433  PspWin32SessionCallout(PspW32JobCallout, &Parms, Job->SessionId);
434 
435  ExReleaseResourceLite (&Job->JobLock);
436  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
437  }
438 
439  if (JobToken != NULL) {
440  Status1 = PspSetPrimaryToken (NULL, Process, NULL, NewToken, TRUE);
441  ObDereferenceObject (NewToken);
442  //
443  // Only bad callers should fail here.
444  //
445  ASSERT (NT_SUCCESS (Status1));
446  }
447 
448 deref_and_return_status:
449 
450  ObDereferenceObject (Process);
451  ObDereferenceObject (Job);
452 
453  return Status;
454 }

456 NTSTATUS
457 PspAddProcessToJob(
458  PEJOB Job,
459  PEPROCESS Process
460  )
461 {
462 
463  NTSTATUS Status;
464  PETHREAD CurrentThread;
465  SIZE_T MinWs,MaxWs;
466  KAPC_STATE ApcState;
467  ULONG SetLimit;
468 
469  PAGED_CODE();
470 
471 
472  CurrentThread = PsGetCurrentThread ();
473 
474  Status = STATUS_SUCCESS;
475 
476 
477  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
478  ExAcquireResourceExclusiveLite (&Job->JobLock, TRUE);
479 
480  InsertTailList (&Job->ProcessListHead, &Process->JobLinks);
481 
482  //
483  // Update relevant ADD accounting info.
484  //
485 
486  Job->TotalProcesses++;
487  Job->ActiveProcesses++;
488 
489  //
490  // Test for active process count exceeding limit
491  //
492 
493  if ((Job->LimitFlags & JOB_OBJECT_LIMIT_ACTIVE_PROCESS) &&
494  Job->ActiveProcesses > Job->ActiveProcessLimit) {
495 
496  PS_SET_CLEAR_BITS (&Process->JobStatus,
497  PS_JOB_STATUS_NOT_REALLY_ACTIVE | PS_JOB_STATUS_ACCOUNTING_FOLDED,
498  PS_JOB_STATUS_LAST_REPORT_MEMORY);
499 
500  if (Job->CompletionPort != NULL) {
501  IoSetIoCompletion (Job->CompletionPort,
502  Job->CompletionKey,
503  NULL,
504  STATUS_SUCCESS,
505  JOB_OBJECT_MSG_ACTIVE_PROCESS_LIMIT,
506  TRUE);
507  }
508 
509  Status = STATUS_QUOTA_EXCEEDED;
510  }
511 
512  if ((Job->LimitFlags & JOB_OBJECT_LIMIT_JOB_TIME) && KeReadStateEvent (&Job->Event)) {
513  PS_SET_BITS (&Process->JobStatus, PS_JOB_STATUS_NOT_REALLY_ACTIVE | PS_JOB_STATUS_ACCOUNTING_FOLDED);
514 
515  Status = STATUS_QUOTA_EXCEEDED;
516  }
517 
518  //
519  // If the last handle to the job has been closed and the kill on close option is set
520  // we don't let new processes enter the job. This is to make cleanup solid.
521  //
522 
523  if (PS_TEST_ALL_BITS_SET (Job->JobFlags, PS_JOB_FLAGS_CLOSE_DONE|JOB_OBJECT_LIMIT_KILL_ON_JOB_CLOSE)) {
524  PS_SET_BITS (&Process->JobStatus, PS_JOB_STATUS_NOT_REALLY_ACTIVE | PS_JOB_STATUS_ACCOUNTING_FOLDED);
525  Status = STATUS_INVALID_PARAMETER;
526  }
527 
528  if (NT_SUCCESS (Status)) {
529 
530  PspApplyJobLimitsToProcess (Job, Process);
531 
532  if (Job->CompletionPort != NULL &&
533  Process->UniqueProcessId &&
534  !(Process->JobStatus & PS_JOB_STATUS_NOT_REALLY_ACTIVE) &&
535  !(Process->JobStatus & PS_JOB_STATUS_NEW_PROCESS_REPORTED)) {
536 
537  PS_SET_CLEAR_BITS (&Process->JobStatus,
538  PS_JOB_STATUS_NEW_PROCESS_REPORTED,
539  PS_JOB_STATUS_LAST_REPORT_MEMORY);
540 
541  IoSetIoCompletion (Job->CompletionPort,
542  Job->CompletionKey,
543  (PVOID)Process->UniqueProcessId,
544  STATUS_SUCCESS,
545  JOB_OBJECT_MSG_NEW_PROCESS,
546  FALSE);
547  }
548 
549  } else {
550  Job->ActiveProcesses--;
551  }
552 
553  ExReleaseResourceLite (&Job->JobLock);
554 
555  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
556 
557  if (NT_SUCCESS (Status)) {
558 
559  //
560  // We can't attach with APCs diabled so we have to unlock attach and then check
561  // working set limits.
562  //
563 
564  KeStackAttachProcess (&Process->Pcb, &ApcState);
565 
566  KeEnterGuardedRegionThread (&CurrentThread->Tcb);
567 
568  ExAcquireResourceExclusiveLite (&Job->JobLock, TRUE);
569 
570  MinWs = Job->MinimumWorkingSetSize;
571  MaxWs = Job->MaximumWorkingSetSize;
572  if (Job->LimitFlags & JOB_OBJECT_LIMIT_WORKINGSET) {
573  SetLimit = MM_WORKING_SET_MAX_HARD_ENABLE;
574  } else {
575  SetLimit = MM_WORKING_SET_MAX_HARD_DISABLE;
576  }
577 
578  PspLockWorkingSetChangeExclusiveUnsafe ();
579 
580  ExReleaseResourceLite (&Job->JobLock);
581 
582  //
583  // See if we need to apply WS limits
584  //
585 
586  if (SetLimit != MM_WORKING_SET_MAX_HARD_DISABLE) {
587 
588  MmAdjustWorkingSetSize (MinWs,
589  MaxWs,
590  FALSE,
591  TRUE);
592 
593  }
594  MmEnforceWorkingSetLimit (Process, SetLimit);
595 
596  PspUnlockWorkingSetChangeExclusiveUnsafe ();
597 
598  KeLeaveGuardedRegionThread (&CurrentThread->Tcb);
599 
600  KeUnstackDetachProcess (&ApcState);
601 
602 
603  if (!MmAssignProcessToJob (Process)) {
604  Status = STATUS_QUOTA_EXCEEDED;
605  }
606 
607  }
608 
609  return Status;
610 }

682  IN PVOID Object
683  )
684 {
685  PEJOB Job, tJob;
686  WIN32_JOBCALLOUT_PARAMETERS Parms;
687  PPS_JOB_TOKEN_FILTER Filter;
688  PETHREAD CurrentThread;
689 
690  PAGED_CODE();
691 
692  Job = (PEJOB) Object;
693 
694  //
695  // call ntuser to delete its job structure
696  //
697 
698  Parms.Job = Job;
699  Parms.CalloutType = PsW32JobCalloutTerminate;
700  Parms.Data = NULL;
701 
702  CurrentThread = PsGetCurrentThread ();
703 
704  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
705  ExAcquireResourceExclusiveLite(&Job->JobLock, TRUE);
706 
707  PspWin32SessionCallout(PspW32JobCallout, &Parms, Job->SessionId);
708 
709  ExReleaseResourceLite(&Job->JobLock);
710  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
711 
712  Job->LimitFlags = 0;
713 
714  if (Job->CompletionPort != NULL) {
715  ObDereferenceObject (Job->CompletionPort);
716  Job->CompletionPort = NULL;
717  }
718 
719 
720  //
721  // Remove Job on Job List and job set
722  //
723 
724  tJob = NULL;
725 
726  PspLockJobListExclusive (CurrentThread);
727 
728  RemoveEntryList (&Job->JobLinks);
729 
730  //
731  // If we are part of a jobset then we must be the pinning job. We must pass on the pin to the next
732  // job in the chain.
733  //
734  if (!IsListEmpty (&Job->JobSetLinks)) {
735  tJob = CONTAINING_RECORD (Job->JobSetLinks.Flink, EJOB, JobSetLinks);
736  RemoveEntryList (&Job->JobSetLinks);
737  }
738 
739  PspUnlockJobListExclusive (CurrentThread);
740 
741  //
742  // Removing the pin from the job set can cause a cascade of deletes that would cause a stack overflow
743  // as we recursed at this point. We break recursion by forcing the deferred delete path here.
744  //
745  if (tJob != NULL) {
746  ObDereferenceObjectDeferDelete (tJob);
747  }
748 
749  //
750  // Free Security clutter:
751  //
752 
753  if (Job->Token != NULL) {
754  ObDereferenceObject (Job->Token);
755  Job->Token = NULL;
756  }
757 
758  Filter = Job->Filter;
759  if (Filter != NULL) {
760  if (Filter->CapturedSids != NULL) {
761  ExFreePool (Filter->CapturedSids);
762  }
763 
764  if (Filter->CapturedPrivileges != NULL) {
765  ExFreePool (Filter->CapturedPrivileges);
766  }
767 
768  if (Filter->CapturedGroups != NULL) {
769  ExFreePool (Filter->CapturedGroups);
770  }
771 
772  ExFreePool (Filter);
773 
774  }
775 
776  ExDeleteResourceLite (&Job->JobLock);
777 }

786  )

789 Routine Description:
790

792 
793 Arguments:
794 
795  Process - Process doing the close
796  Object - Job object being closed
797  GrantedAccess - Access ranted for this handle

799  SystemHandleCount - Current handle count for this object
800 
801 Return Value:
802 
803  None.
804 
805 --*/
806 {
807  PEJOB Job = Object;
808  PVOID Port;
809  PETHREAD CurrentThread;
810 
811  PAGED_CODE ();
812 
813  UNREFERENCED_PARAMETER (Process);
814  UNREFERENCED_PARAMETER (GrantedAccess);
815  UNREFERENCED_PARAMETER (ProcessHandleCount);
816  //
817  // If this isn't the last handle then do nothing.
818  //
819  if (SystemHandleCount > 1) {
820  return;
821  }
822 
823  CurrentThread = PsGetCurrentThread ();
824 
825 
826
