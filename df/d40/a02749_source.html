230 
231 Routine Description:
232 
233  Registry query values callback routine for reading SDs for guids
234 
235 Arguments:
236 
237  ValueName - the name of the value
238 
239  ValueType - the type of the value
240 
241  ValueData - the data in the value (unicode string data)
242 
243  ValueLength - the number of bytes in the value data
244 
245  Context - Not used
246 
247  EntryContext - Pointer to PSECURITY_DESCRIPTOR to store a pointer to

284 NTSTATUS
285 WmipAddLogHeader(
286  IN PWMI_LOGGER_CONTEXT LoggerContext,
287  IN OUT PWMI_BUFFER_HEADER Buffer
288  )
289 /*++
290 
291 Routine Description:
292 
293  Add a standard logfile header in kernel mode.
294  To make sure the first buffer of the log file contains the file header,
295  we pop a buffer from the free list, write the header, and flush the buffer
296  right away.
297 
298 Arguments:
299 
300  LoggerContext - The logger context
301 
302 Return Value:
303 
304  NT Status code
305 
306 --*/
307 {
308  PTRACE_LOGFILE_HEADER LogfileHeader;
309  USHORT HeaderSize;
310  PSYSTEM_TRACE_HEADER EventTrace;
311  PETHREAD Thread;
312  NTSTATUS Status = STATUS_SUCCESS;
313  ULONG BufferProvided = (Buffer != NULL);
314  ULONG LocalBuffer = FALSE;
315 
316  if (LoggerContext == NULL) {
317  return STATUS_INVALID_PARAMETER;
318  }
319  // if this is a non-kernel logger started under wow64,
320  // we need to shrink the logfileheader
321  if (LoggerContext->Wow && !LoggerContext->KernelTraceOn) {
322  HeaderSize = sizeof(TRACE_LOGFILE_HEADER)
323  + LoggerContext->LoggerName.Length + sizeof(UNICODE_NULL)
324  + LoggerContext->LogFileName.Length + sizeof(UNICODE_NULL)
325  - 8;
326  }
327  else {
328  HeaderSize = sizeof(TRACE_LOGFILE_HEADER)
329  + LoggerContext->LoggerName.Length + sizeof(UNICODE_NULL)
330  + LoggerContext->LogFileName.Length + sizeof(UNICODE_NULL);
331  }
332  if (LoggerContext->BufferSize <
333  (HeaderSize + sizeof(WMI_BUFFER_HEADER) + sizeof(SYSTEM_TRACE_HEADER))) {
334  return STATUS_NO_MEMORY;
335  }
336 
337  //
338  // Pop a buffer from Free List
339  //
340 
341  if (!BufferProvided) {
342  Buffer = WmipGetFreeBuffer(LoggerContext);
343 
344  if (Buffer == NULL) {
345  Buffer = ExAllocatePoolWithTag(NonPagedPool,
346  LoggerContext->BufferSize, TRACEPOOLTAG);
347  if (Buffer == NULL) {
348 
349  //
350  // No buffer available.
351  //
352  return STATUS_NO_MEMORY;
353  }
354  LocalBuffer = TRUE;
355 
356  Buffer->Flags = 1;
357 
358  WmipResetBufferHeader(LoggerContext, Buffer, WMI_BUFFER_TYPE_RUNDOWN);
359 
360  KeQuerySystemTime(&Buffer->TimeStamp);
361  }
362  }
363 
364  //
365  // Fill in the Header Info.
366  //
367  Thread = PsGetCurrentThread();
368  EventTrace = (PSYSTEM_TRACE_HEADER) (Buffer+1);
369  EventTrace->Packet.Group = (UCHAR) (EVENT_TRACE_GROUP_HEADER >> 8);
370  EventTrace->Packet.Type = EVENT_TRACE_TYPE_INFO;
371  EventTrace->Packet.Size = HeaderSize + sizeof(SYSTEM_TRACE_HEADER);
372  EventTrace->Marker = SYSTEM_TRACE_MARKER;
373  EventTrace->ThreadId = HandleToUlong(Thread->Cid.UniqueThread);
374  EventTrace->KernelTime = Thread->Tcb.KernelTime;
375  EventTrace->UserTime = Thread->Tcb.UserTime;
376  EventTrace->SystemTime = LoggerContext->ReferenceTimeStamp;
377 
378 
379  LogfileHeader = (PTRACE_LOGFILE_HEADER) (EventTrace+1);
380  RtlZeroMemory(LogfileHeader, HeaderSize);
381 
382  LogfileHeader->BufferSize = LoggerContext->BufferSize;
383  LogfileHeader->VersionDetail.MajorVersion = (UCHAR) NtMajorVersion;
384  LogfileHeader->VersionDetail.MinorVersion = (UCHAR) NtMinorVersion;
385  LogfileHeader->VersionDetail.SubVersion = TRACE_VERSION_MAJOR;
386  LogfileHeader->VersionDetail.SubMinorVersion = TRACE_VERSION_MINOR;
387  LogfileHeader->ProviderVersion = (NtBuildNumber & 0xffff);
388 
389  LogfileHeader->StartBuffers = 1;
390  LogfileHeader->LogFileMode =
391  LoggerContext->LoggerMode & (~(EVENT_TRACE_REAL_TIME_MODE));
392  LogfileHeader->NumberOfProcessors = KeNumberProcessors;
393  LogfileHeader->MaximumFileSize = LoggerContext->MaximumFileSize;
394  LogfileHeader->TimerResolution = KeMaximumIncrement;
395 
396  if (LoggerContext->Wow && !LoggerContext->KernelTraceOn) {
397  // We need to shrink a log file header for a non-kernel WOW64 logger.
398  PUCHAR LoggerNamePtr64, LogFileNamePtr64;
399  KeQueryPerformanceCounter((PLARGE_INTEGER)((PUCHAR)(&LogfileHeader->PerfFreq) - 8));
400  *((PLARGE_INTEGER)((PUCHAR)(&LogfileHeader->StartTime) - 8))
401  = LoggerContext->ReferenceSystemTime;
402  *((PLARGE_INTEGER)((PUCHAR)(&LogfileHeader->BootTime) - 8))
403  = KeBootTime;
404  LogfileHeader->PointerSize = sizeof(ULONG);
405 
406  //
407  // ReservedFlags to indicate using Perf Clock
408  //
409  *((PULONG)((PUCHAR)(&LogfileHeader->ReservedFlags) - 8))
410  = LoggerContext->UsePerfClock;
411 
412  LoggerNamePtr64 = (PUCHAR) LogfileHeader
413  + sizeof(TRACE_LOGFILE_HEADER)
414  - 8;
415  *((PULONG)(&LogfileHeader->LoggerName)) = PtrToUlong(LoggerNamePtr64);
416 
417  LogFileNamePtr64 = (PUCHAR)LogfileHeader
418  + sizeof(TRACE_LOGFILE_HEADER)
419  - 8
420  + LoggerContext->LoggerName.Length
421  + sizeof(UNICODE_NULL);
422  *((PULONG)((PUCHAR)(&LogfileHeader->LogFileName) - 4)) = PtrToUlong(LogFileNamePtr64);
423 
424  RtlCopyMemory(LoggerNamePtr64,
425  LoggerContext->LoggerName.Buffer,
426  LoggerContext->LoggerName.Length + sizeof(UNICODE_NULL));
427  RtlCopyMemory(LogFileNamePtr64,
428  LoggerContext->LogFileName.Buffer,
429  LoggerContext->LogFileName.Length + sizeof(UNICODE_NULL));
430  RtlQueryTimeZoneInformation((PRTL_TIME_ZONE_INFORMATION)((PUCHAR)(&LogfileHeader->TimeZone) - 8));
431  }
432  else {
433  KeQueryPerformanceCounter(&LogfileHeader->PerfFreq);
434  LogfileHeader->StartTime = LoggerContext->ReferenceSystemTime;
435  LogfileHeader->BootTime = KeBootTime;
436  LogfileHeader->PointerSize = sizeof(PVOID);
437 
438  //
439  // ReservedFlags to indicate using Perf Clock
440  //
441  LogfileHeader->ReservedFlags = LoggerContext->UsePerfClock;
442 
443  LogfileHeader->LoggerName = (PWCHAR) ( (PUCHAR) LogfileHeader
444  + sizeof(TRACE_LOGFILE_HEADER) );
445  LogfileHeader->LogFileName = (PWCHAR) ( (PUCHAR)LogfileHeader->LoggerName
446  + LoggerContext->LoggerName.Length
447  + sizeof(UNICODE_NULL));
448 
449  RtlCopyMemory(LogfileHeader->LoggerName,
450  LoggerContext->LoggerName.Buffer,
451  LoggerContext->LoggerName.Length + sizeof(UNICODE_NULL));
452  RtlCopyMemory(LogfileHeader->LogFileName,
453  LoggerContext->LogFileName.Buffer,
454  LoggerContext->LogFileName.Length + sizeof(UNICODE_NULL));
455  RtlQueryTimeZoneInformation(&LogfileHeader->TimeZone);
456  }
457 
458  //
459  // Adjust the Offset;
460  //
461  Buffer->CurrentOffset += ALIGN_TO_POWER2(sizeof(SYSTEM_TRACE_HEADER) + HeaderSize,
462  WmiTraceAlignment);
463 
464  //
465  // Log the Group Masks if it is from the kernel logger
466  //
467  if(LoggerContext == WmipLoggerContext[WmipKernelLogger]) {
468  PPERFINFO_GROUPMASK PGroupMask;
469 
470  HeaderSize = sizeof(PERFINFO_GROUPMASK) + sizeof(SYSTEM_TRACE_HEADER);
471  EventTrace = (PSYSTEM_TRACE_HEADER) ((PCHAR) Buffer + Buffer->CurrentOffset);
472  EventTrace->Packet.Group = (UCHAR) (EVENT_TRACE_GROUP_HEADER >> 8);
473  EventTrace->Packet.Type = EVENT_TRACE_TYPE_EXTENSION;
474  EventTrace->Packet.Size = HeaderSize;
475  EventTrace->Marker = SYSTEM_TRACE_MARKER;
476  EventTrace->ThreadId = HandleToUlong(Thread->Cid.UniqueThread);
477  EventTrace->KernelTime = Thread->Tcb.KernelTime;
478  EventTrace->UserTime = Thread->Tcb.UserTime;
479  EventTrace->SystemTime = LoggerContext->ReferenceTimeStamp;
480 
481  PGroupMask = (PPERFINFO_GROUPMASK) (EventTrace+1);
482  ASSERT(PPerfGlobalGroupMask != NULL);
483  RtlCopyMemory(PGroupMask, PPerfGlobalGroupMask, sizeof(PERFINFO_GROUPMASK));
484 
485  Buffer->CurrentOffset += ALIGN_TO_POWER2( HeaderSize, WmiTraceAlignment);
486  }
487 
488 
489 
490 
491  if (BufferProvided)
492  return Status;
493  //
494  // The buffer is prepared properly. Flush it so it can be written out to disk.
495  //
496  Status = WmipFlushBuffer(LoggerContext, Buffer, WMI_BUFFER_FLAG_NORMAL);
497 
498  if (LocalBuffer && (Buffer != NULL)) {
499  ExFreePool(Buffer);
500  return Status;
501  }
502 
503  //
504  // Reference count is overwritten during the flush,
505  // Set it back before push the buffer into free list.
506  //
507  Buffer->ReferenceCount = 0;
508 
509  //
510  // Set the buffer flags to "free" state
511  //
512  Buffer->State.Flush = 0;
513  Buffer->State.InUse = 0;
514  Buffer->State.Free = 1;
515 
516  //
517  // Push into free list
518  //
519  InterlockedPushEntrySList(&LoggerContext->FreeList,
520  (PSLIST_ENTRY) &Buffer->SlistEntry);
521  InterlockedIncrement(&LoggerContext->BuffersAvailable);
522  InterlockedDecrement(&LoggerContext->BuffersInUse);
523 
524  TraceDebug((2, "WmipAddLogHeader: Boot %I64u Current %I64u Difference %I64u\n",
525  KeBootTime, EventTrace->SystemTime,
526  EventTrace->SystemTime.QuadPart - KeBootTime.QuadPart));
527 
528  return Status;
529 }

531 NTSTATUS
532 WmipDelayCreate(
533  OUT PHANDLE FileHandle,
534  IN OUT PUNICODE_STRING FileName,
535  IN ULONG Append
536  )
537 /*++
538 
539 Routine Description:
540 
541  This is called by the global logger to actually open the logfile
542  when the first buffer needs to flush (instead of when the logger started)
543 
544 Arguments:
545 
546  LoggerHandle The handle to the logfile to be returned
547  FileName The logfile name. If the default was chosen, we will
548  returned the actual pathname in %systemroot%
549 
550 Return Value:
551 
552  NT Status code
553 
554 --*/
555 {
556  PWCHAR Buffer;
557  PWCHAR strBuffer = NULL;
558  ULONG DefaultFile, Length;
559  UNICODE_STRING LogFileName;
560  NTSTATUS Status;
561 
562  if (FileName == NULL)
563  return STATUS_INVALID_PARAMETER;
564 
565  RtlInitUnicodeString(&LogFileName, DEFAULT_GLOBAL_LOGFILE_ROOT);
566  DefaultFile = (RtlCompareUnicodeString(FileName, &LogFileName, TRUE) == 0);
567 
568  if (DefaultFile) {
569  HRESULT hr;
570  //
571  // Try creating the file first
572  //
573  Length = (ULONG) ( NtSystemRoot.Length
574  + sizeof(WCHAR) * (wcslen(DEFAULT_GLOBAL_DIRECTORY) +
575  wcslen(DEFAULT_GLOBAL_LOGFILE) + 1)
576  + sizeof(UNICODE_NULL));
577  strBuffer = (PWCHAR) ExAllocatePoolWithTag(
578  PagedPool, Length, TRACEPOOLTAG);
579  if (strBuffer == NULL)
580  return STATUS_NO_MEMORY;
581 
582  hr = StringCbPrintfW(strBuffer,
583  Length,
584  L"%ws%ws\\%ws",
585  NtSystemRoot.Buffer,
586  DEFAULT_GLOBAL_DIRECTORY,
587  DEFAULT_GLOBAL_LOGFILE);
588  if (FAILED(hr)) {
589  ExFreePool(strBuffer);
590  return STATUS_INVALID_PARAMETER;
591  }
592 
593  Status = WmipCreateNtFileName(strBuffer, &Buffer);
594  if (!NT_SUCCESS(Status)) {
595  ExFreePool(strBuffer);
596  return Status;
597  }
598 
599  Status = WmipCreateDirectoryFile(Buffer, FALSE, FileHandle, Append);
600  if (!NT_SUCCESS(Status)) {
601  ULONG DirLen;
602  //
603  // Probably directory does not exist, so try and create it
604  //
605  DirLen = (ULONG)
606  (wcslen(Buffer)-wcslen(DEFAULT_GLOBAL_LOGFILE)) - 5;
607  Buffer[DirLen] = UNICODE_NULL;
608  Status = WmipCreateDirectoryFile(Buffer, TRUE, NULL, Append);
609  if (NT_SUCCESS(Status)) {
610  Buffer[DirLen] = L'\\';
611  DirLen += 4;
612  Buffer[DirLen] = UNICODE_NULL;
613  Status = WmipCreateDirectoryFile(Buffer, TRUE, NULL, Append);
614  Buffer[DirLen] = L'\\';
615  }
616  if (NT_SUCCESS(Status)) {
617  Status = WmipCreateDirectoryFile(Buffer, FALSE, FileHandle, Append);
618  }
619  }
620  // Make sure that directory is there first
621 
622  if (NT_SUCCESS(Status)) {
623  if (FileName->Buffer != NULL) {
624  RtlFreeUnicodeString(FileName);
625  }
626  RtlInitUnicodeString(FileName, strBuffer);
627  if (FileName->MaximumLength < Length)
628  FileName->MaximumLength = (USHORT) Length;
629  }
630  }
631  else {
632  Status = WmipCreateNtFileName(FileName->Buffer, &Buffer);
633  if (NT_SUCCESS(Status)) {
634  Status = WmipCreateDirectoryFile(Buffer, FALSE, FileHandle, Append);
635  }
636  }
637 
638  if (Buffer != NULL) {
639  ExFreePool(Buffer);
640  }
641  return Status;
642 }
