86 /*++
87 
88 Routine Description:
89 
90  This function is the special APC routine that is called to remove
91  a timer from the current thread's active timer list.
92 
93 Arguments:
94 
95  Apc - Supplies a pointer to the APC object used to invoke this routine.
96 
97  NormalRoutine - Supplies a pointer to a pointer to the normal routine
98  function that was specified when the APC was initialized.
99 
100  NormalContext - Supplies a pointer to a pointer to an arbitrary data
101  structure that was specified when the APC was initialized.
102 
103  SystemArgument1, SystemArgument2 - Supplies a set of two pointers to
104  two arguments that contain untyped data.
105 
106 Return Value:
107 
108  None.
109 
110 --*/
111 
112 {
113 
114  ULONG DereferenceCount;
115  PETHREAD ExThread;
116  PETIMER ExTimer;
117  KIRQL OldIrql1;
118 
119  UNREFERENCED_PARAMETER(NormalContext);
120  UNREFERENCED_PARAMETER(SystemArgument1);
121  UNREFERENCED_PARAMETER(SystemArgument2);
122 
123  //
124  // Get address of executive timer object and the current thread object.
125  //
126 
127  ExThread = PsGetCurrentThread();
128  ExTimer = CONTAINING_RECORD(Apc, ETIMER, TimerApc);
129 
130  //
131  // If the timer is still in the current thread's active timer list, then
132  // remove it if it is not a periodic timer and set APC associated FALSE.
133  // It is possible for the timer not to be in the current thread's active
134  // timer list since the APC could have been delivered, and then another
135  // thread could have set the timer again with another APC. This would
136  // have caused the timer to be removed from the current thread's active
137  // timer list.
138  //
139  // N. B. The spin locks for the timer and the active timer list must be
140  // acquired in the order: 1) timer lock, 2) thread list lock.
141  //
142 
143  DereferenceCount = 1;
144  ExAcquireSpinLock(&ExTimer->Lock, &OldIrql1);
145  ExAcquireSpinLockAtDpcLevel(&ExThread->ActiveTimerListLock);
146  if ((ExTimer->ApcAssociated) && (&ExThread->Tcb == ExTimer->TimerApc.Thread)) {
147  if (ExTimer->Period == 0) {
148  RemoveEntryList(&ExTimer->ActiveTimerListEntry);
149  ExTimer->ApcAssociated = FALSE;
150  DereferenceCount += 1;
151  }
152 
153  } else {
154  *NormalRoutine = (PKNORMAL_ROUTINE)NULL;
155  }
156 
157  ExReleaseSpinLockFromDpcLevel(&ExThread->ActiveTimerListLock);
158  ExReleaseSpinLock(&ExTimer->Lock, OldIrql1);
159  ObDereferenceObjectEx(ExTimer, DereferenceCount);
160  return;
161 }

171 /*++
172 
173 Routine Description:
174 
175  This function is the DPC routine that is called when a timer expires that
176  has an associated APC routine. Its function is to insert the associated
177  APC into the target thread's APC queue.
178 
179 Arguments:
180 
181  Dpc - Supplies a pointer to a control object of type DPC.
182 
183  DeferredContext - Supplies a pointer to the executive timer that contains
184  the DPC that caused this routine to be executed.
185 
186  SystemArgument1, SystemArgument2 - Supplies values that are not used by
187  this routine.
188 
189 Return Value:
190 
191  None.
192 
193 --*/
194 
195 {
196 
197  PETIMER ExTimer;
198  BOOLEAN Inserted;
199  PKTIMER KeTimer;
200 
201  //
202  // Reference the timer so the APC is free to manipulate it.
203  //
204  // If the object is being deleted the delete routine will flush all
205  // pending DPC's so the object won't be completely deleted until after
206  // the following code completes.
207  //
208 
209  ExTimer = (PETIMER)DeferredContext;
210 
211 #if defined(_AMD64_)
212 
213  try {
214 
215 #else
216 
217  UNREFERENCED_PARAMETER(Dpc);
218 
219 #endif
220 
221  if (ObReferenceObjectSafe(ExTimer) == FALSE) {
222  return;
223  }
224 
225 #if defined(_AMD64_)
226 
227  } except(KiKernelDpcFilter(Dpc, GetExceptionInformation())) {
228  return;
229  }
230 
231 #endif
232 
233  //
234  // If there is still an APC associated with the timer, then insert the APC
235  // in target thread's APC queue. It is possible that the timer does not
236  // have an associated APC. This can happen when the timer is set to expire
237  // by a thread running on another processor just after the DPC has been
238  // removed from the DPC queue, but before it has acquired the timer related
239  // spin lock.
240  //
241 
242  KeTimer = &ExTimer->KeTimer;
243  Inserted = FALSE;
244  ExAcquireSpinLockAtDpcLevel(&ExTimer->Lock);
245  if (ExTimer->ApcAssociated) {
246  Inserted = KeInsertQueueApc(&ExTimer->TimerApc,
247  SystemArgument1,
248  SystemArgument2,
249  TIMER_APC_INCREMENT);
250  }
251 
252  ExReleaseSpinLockFromDpcLevel(&ExTimer->Lock);
253 
254  //
255  // If the timer APC wasn't inserted in the APC list, then release the
256  // reference associated with it.
257  //
258 
259  if (Inserted == FALSE) {
260  ObDereferenceObject(ExTimer);
261  }
262 
263  return;
264 }

393  VOID

396 /*++
397 
398 Routine Description:
399 
400  This function is called when a thread is about to be terminated to
401  process the active timer list. It is assumed that APC's have been
402  disabled for the subject thread, thus this code cannot be interrupted
403  to execute an APC for the current thread.
404 
405 Arguments:
406 
407  None.
408 
409 Return Value:
410 
411  None.
412 
413 --*/
414 
415 {
416 
417  LONG DereferenceCount;
418  PETHREAD ExThread;
419  PETIMER ExTimer;
420  PLIST_ENTRY NextEntry;
421  KIRQL OldIrql1;
422 
423  //
424  // Process each entry in the active timer list.
425  //
426 
427  ExThread = PsGetCurrentThread();
428  ExAcquireSpinLock(&ExThread->ActiveTimerListLock, &OldIrql1);
429  NextEntry = ExThread->ActiveTimerListHead.Flink;
430  while (NextEntry != &ExThread->ActiveTimerListHead) {
431  ExTimer = CONTAINING_RECORD(NextEntry, ETIMER, ActiveTimerListEntry);
432 
433  //
434  // Increment the reference count on the object so that it cannot be
435  // deleted, and then drop the active timer list lock.
436  //
437  // N. B. The object reference cannot fail and will acquire no mutexes.
438  //
439 
440  ObReferenceObject(ExTimer);
441  ExReleaseSpinLock(&ExThread->ActiveTimerListLock, OldIrql1);
442  DereferenceCount = 1;
443 
444  //
445  // Acquire the timer spin lock and reacquire the active time list spin
446  // lock. If the timer is still in the current thread's active timer
447  // list, then cancel the timer, remove the timer's DPC from the DPC
448  // queue, remove the timer's APC from the APC queue, remove the timer
449  // from the thread's active timer list, and set the associate APC
450  // flag FALSE.
451  //
452  // N. B. The spin locks for the timer and the active timer list must be
453  // acquired in the order: 1) timer lock, 2) thread list lock.
454  //
455 
456  ExAcquireSpinLock(&ExTimer->Lock, &OldIrql1);
457  ExAcquireSpinLockAtDpcLevel(&ExThread->ActiveTimerListLock);
458  if ((ExTimer->ApcAssociated) && (&ExThread->Tcb == ExTimer->TimerApc.Thread)) {
459  RemoveEntryList(&ExTimer->ActiveTimerListEntry);
460  ExTimer->ApcAssociated = FALSE;
461  KeCancelTimer(&ExTimer->KeTimer);
462  KeRemoveQueueDpc(&ExTimer->TimerDpc);
463  if (KeRemoveQueueApc(&ExTimer->TimerApc)) {
464  DereferenceCount += 1;
465  }
466 
467  DereferenceCount += 1;
468  }
469 
470  ExReleaseSpinLockFromDpcLevel(&ExThread->ActiveTimerListLock);
471  ExReleaseSpinLock(&ExTimer->Lock, OldIrql1);
472  ObDereferenceObjectEx(ExTimer, DereferenceCount);
473 
474  //
475  // Raise IRQL to DISPATCH_LEVEL and reacquire active timer list
476  // spin lock.
477  //
478 
479  ExAcquireSpinLock(&ExThread->ActiveTimerListLock, &OldIrql1);
480  NextEntry = ExThread->ActiveTimerListHead.Flink;
481  }
482 
483  ExReleaseSpinLock(&ExThread->ActiveTimerListLock, OldIrql1);
484  return;
485 }

495 /*++
496 
497 Routine Description:
498 
499  This function creates an timer object and opens a handle to the object with
500  the specified desired access.
501 
502 Arguments:
503 
504  TimerHandle - Supplies a pointer to a variable that will receive the
505  timer object handle.
506 
507  DesiredAccess - Supplies the desired types of access for the timer object.
508 
509  ObjectAttributes - Supplies a pointer to an object attributes structure.
510 
511  TimerType - Supplies the type of the timer which can be synchronization
512  or notification.
513 
514 Return Value:
515 
516  NTSTATUS.
517 
518 --*/
519 
520 {
521 
522  PETIMER ExTimer;
523  HANDLE Handle;
524  KPROCESSOR_MODE PreviousMode;
525  NTSTATUS Status;
526 
527  //
528  // Check argument validity.
529  //
530 
531  if ((TimerType != NotificationTimer) &&
532  (TimerType != SynchronizationTimer)) {
533 
534  return STATUS_INVALID_PARAMETER_4;
535  }
536 
537  //
538  // Get previous processor mode and probe output handle address if
539  // necessary.
540  //
541 
542  PreviousMode = KeGetPreviousMode();
543  if (PreviousMode != KernelMode) {
544  try {
545  ProbeForWriteHandle(TimerHandle);
546 
547  } except(EXCEPTION_EXECUTE_HANDLER) {
548  return GetExceptionCode();
549  }
550  }
551 
552  //
553  // Allocate timer object.
554  //
555 
556  Status = ObCreateObject(PreviousMode,
557  ExTimerObjectType,
558  ObjectAttributes,
559  PreviousMode,
560  NULL,
561  sizeof(ETIMER),
562  0,
563  0,
564  &ExTimer);
565 
566  //
567  // If the timer object was successfully allocated, then initialize the
568  // timer object and attempt to insert the time object in the current
569  // process' handle table.
570  //
571 
572  if (NT_SUCCESS(Status)) {
573  KeInitializeDpc(&ExTimer->TimerDpc,
574  ExpTimerDpcRoutine,
575  (PVOID)ExTimer);
576 
577  KeInitializeTimerEx(&ExTimer->KeTimer, TimerType);
578  KeInitializeSpinLock(&ExTimer->Lock);
579  ExTimer->ApcAssociated = FALSE;
580  ExTimer->WakeTimer = FALSE;
581  ExTimer->WakeTimerListEntry.Flink = NULL;

583  NULL,
584  DesiredAccess,
585  0,
586  NULL,
587  &Handle);
588

591  // process' handle table, then attempt to write the timer object
592  // handle value. If the write attempt fails, then do not report
593  // an error. When the caller attempts to access the handle value,
594  // an access violation will occur.
595  //
596 
597  if (NT_SUCCESS(Status)) {
598  if (PreviousMode != KernelMode) {
599  try {
600  *TimerHandle = Handle;
601 
602  } except(EXCEPTION_EXECUTE_HANDLER) {
603  NOTHING;
604  }
605 
606  } else {
607  *TimerHandle = Handle;
608  }
609  }
610  }
611 
612  //
613  // Return service status.
614  //
615 
616  return Status;
617 }

709 NTSTATUS
710 NtCancelTimer (
711  __in HANDLE TimerHandle,
712  __out_opt PBOOLEAN CurrentState
713  )
714 
715 /*++
716 
717 Routine Description:
718 
719  This function cancels a timer object.
720 
721 Arguments:
722 
723  TimerHandle - Supplies a handle to an timer object.
724 
725  CurrentState - Supplies an optional pointer to a variable that will
726  receive the current state of the timer object.
727 
728 Return Value:
729 
730  NTSTATUS.
731 
732 --*/
733 
734 {
735 
736  ULONG DereferenceCount;
737  PETHREAD ExThread;
738  PETIMER ExTimer;
739  KIRQL OldIrql1;
740  KPROCESSOR_MODE PreviousMode;
741  BOOLEAN State;
742  NTSTATUS Status;
743 
744  //
745  // Get previous processor mode and probe current state address if
746  // necessary.
747  //
748 
749  PreviousMode = KeGetPreviousMode();
750  if (ARGUMENT_PRESENT(CurrentState) && (PreviousMode != KernelMode)) {
751  try {
752  ProbeForWriteBoolean(CurrentState);
753 
754  } except(EXCEPTION_EXECUTE_HANDLER) {
755  return GetExceptionCode();
756  }
757  }
758 
759  //
760  // Reference timer object by handle.
761  //
762 
763  Status = ObReferenceObjectByHandle(TimerHandle,
764  TIMER_MODIFY_STATE,
765  ExTimerObjectType,
766  PreviousMode,
767  &ExTimer,
768  NULL);
769 
770  //
771  // If the reference was successful, then cancel the timer object,
772  // dereference the timer object, and write the current state value
773  // if specified. If the write attempt fails, then do not report an
774  // error. When the caller attempts to access the current state value,
775  // an access violation will occur.
776  //
777 
778  if (NT_SUCCESS(Status)) {
779  DereferenceCount = 1;
780  ExAcquireSpinLock(&ExTimer->Lock, &OldIrql1);
781  if (ExTimer->ApcAssociated) {
782  ExThread = CONTAINING_RECORD(ExTimer->TimerApc.Thread, ETHREAD, Tcb);
783  ExAcquireSpinLockAtDpcLevel(&ExThread->ActiveTimerListLock);
784  RemoveEntryList(&ExTimer->ActiveTimerListEntry);
785  ExTimer->ApcAssociated = FALSE;
786  ExReleaseSpinLockFromDpcLevel(&ExThread->ActiveTimerListLock);
787  KeCancelTimer(&ExTimer->KeTimer);
788  KeRemoveQueueDpc(&ExTimer->TimerDpc);
789  if (KeRemoveQueueApc(&ExTimer->TimerApc)) {
790  DereferenceCount += 1;
791  }
792 
793  DereferenceCount += 1;
794 
795  } else {
796  KeCancelTimer(&ExTimer->KeTimer);
797  }
798 
799  if (ExTimer->WakeTimerListEntry.Flink) {
800  ExAcquireSpinLockAtDpcLevel(&ExpWakeTimerListLock);
801 
802  //
803  // Check again since wait next time might have removed the timer.
804  //
805 
806  if (ExTimer->WakeTimerListEntry.Flink) {
807  RemoveEntryList(&ExTimer->WakeTimerListEntry);
808  ExTimer->WakeTimerListEntry.Flink = NULL;
809  }
810 
811  ExReleaseSpinLockFromDpcLevel(&ExpWakeTimerListLock);
812  }
813 
814  ExReleaseSpinLock(&ExTimer->Lock, OldIrql1);
815 
816  //
817  // Read current state of timer, dereference timer object, and set
818  // current state.
819  //
820 
821  State = KeReadStateTimer(&ExTimer->KeTimer);
822  ObDereferenceObjectEx(ExTimer, DereferenceCount);
823  if (ARGUMENT_PRESENT(CurrentState)) {
824  if (PreviousMode != KernelMode) {
825  try {
826  *CurrentState = State;
827 
828  } except(EXCEPTION_EXECUTE_HANDLER) {
829  NOTHING;
830  }
831 
832  } else {
833  *CurrentState = State;
834  }
835  }
836  }
837 
838  //
839  // Return service status.
840  //
841 
842  return Status;
843 }
