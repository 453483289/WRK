140 BOOLEAN WmipFindSMBiosEPSHeader(
141  PUCHAR SMBiosVirtualAddress,
142  ULONG BiosSize,
143  PSMBIOS_EPS_HEADER EPSHeader
144  )
145 /*++
146 
147 Routine Description:
148 
149  Search for the SMBIOS 2.1 EPS structure and copy it.
150 
151 Arguments:
152 
153  SMBiosVirtualAddress is the beginning virtual address to start searching
154  for the SMBIOS 2.1 EPS anchor string.
155 
156  BiosSize is the number of bytes to search for the anchor string
157 
158  EPSHeader is the memory into which the EPS header is copied
159 Return Value:
160 
161  Pointer to SMBIOS 2.1 EPS or NULL if EPS not found
162 
163 --*/
164 {
165  PUCHAR SearchEnd;
166  UCHAR CheckSum;
167  PSMBIOS_EPS_HEADER SMBiosEPSHeader;
168  PDMIBIOS_EPS_HEADER DMIBiosEPSHeader;
169  ULONG i;
170  ULONG CheckLength;
171 
172  PAGED_CODE();
173 
174  UNREFERENCED_PARAMETER (BiosSize);
175 
176  RtlZeroMemory(EPSHeader, sizeof(SMBIOS_EPS_HEADER));
177 
178  //
179  // Scan the bios for the two anchor strings that that signal the SMBIOS
180  // table.
181  SearchEnd = SMBiosVirtualAddress + SMBIOS_EPS_SEARCH_SIZE -
182  2 * SMBIOS_EPS_SEARCH_INCREMENT;
183 
184  while (SMBiosVirtualAddress < SearchEnd)
185  {
186  SMBiosEPSHeader = (PSMBIOS_EPS_HEADER)SMBiosVirtualAddress;
187  DMIBiosEPSHeader = (PDMIBIOS_EPS_HEADER)SMBiosVirtualAddress;
188 
189  //
190  // First check for _DMI_ anchor string
191  if ((*((PULONG)DMIBiosEPSHeader->Signature2) == DMI_EPS_SIGNATURE) &&
192  (DMIBiosEPSHeader->Signature2[4] == '_'))
193  {
194  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Found possible DMIBIOS EPS Header at %x\n", SMBiosEPSHeader));
195  CheckLength = sizeof(DMIBIOS_EPS_HEADER);
196  }
197 
198  //
199  // Then check for full _SM_ anchor string
200  else if ((*((PULONG)SMBiosEPSHeader->Signature) == SMBIOS_EPS_SIGNATURE) &&
201  (SMBiosEPSHeader->Length >= sizeof(SMBIOS_EPS_HEADER)) &&
202  (*((PULONG)SMBiosEPSHeader->Signature2) == DMI_EPS_SIGNATURE) &&
203  (SMBiosEPSHeader->Signature2[4] == '_' ))
204  {
205  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Found possible SMBIOS EPS Header at %p\n", SMBiosEPSHeader));
206  CheckLength = SMBiosEPSHeader->Length;
207  } else {
208  //
209  // Did not find anchor string, go search next paragraph
210  SMBiosVirtualAddress += SMBIOS_EPS_SEARCH_INCREMENT;
211  continue;
212  }
213 
214  //
215  // Verify anchor string with checksum
216  CheckSum = 0;
217  for (i = 0; i < CheckLength ; i++)
218  {
219  CheckSum = (UCHAR)(CheckSum + SMBiosVirtualAddress[i]);
220  }
221 
222  if (CheckSum == 0)
223  {
224  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Found SMBIOS EPS Header at %p\n", SMBiosEPSHeader));
225  if (CheckLength == sizeof(DMIBIOS_EPS_HEADER))
226  {
227  //
228  // We only had got a DMI header so copy that
229  //
230  RtlCopyMemory(&EPSHeader->Signature2[0],
231  DMIBiosEPSHeader,
232  sizeof(DMIBIOS_EPS_HEADER));
233  } else {
234  //
235  // We got the full SMBIOS header so copy that
236  //
237  RtlCopyMemory(EPSHeader,
238  SMBiosEPSHeader,
239  sizeof(SMBIOS_EPS_HEADER));
240  }
241  return(TRUE);
242  }
243  SMBiosVirtualAddress += SMBIOS_EPS_SEARCH_INCREMENT;
244 
245  }
246 
247  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: SMBIOS EPS Header not found\n"));
248  return(FALSE);
249 }

318 NTSTATUS WmipSMBiosDataRegQueryRoutine(
319  IN PWSTR ValueName,
320  IN ULONG ValueType,
321  IN PVOID ValueData,
322  IN ULONG ValueLength,
323  IN PVOID Context,
324  IN PVOID EntryContext
325  )
326 /*++
327 
328 Routine Description:
329 
330  Registry query values callback routine for reading SMBIOS data from
331  registry.
332 
333 Arguments:
334 
335  ValueName - the name of the value
336 
337  ValueType - the type of the value
338 
339  ValueData - the data in the value (unicode string data)
340 
341  ValueLength - the number of bytes in the value data
342 
343  Context - Not used
344 
345  EntryContext - Pointer to PUCHAR to store a pointer to
346  store the SMBIOS data read from the registry value. If this is NULL
347  then the caller is not interested in the SMBIOS data
348 
349 Return Value:
350 
351  NT Status code -
352  STATUS_SUCCESS - SMBIOS data is present in the value
353  STATUS_INSUFFICIENT_RESOURCES - Not enough memory to keep SMBIOS data
354  STATUS_UNSUCCESSFUL - SMBios data is not present in the value
355 
356 --*/
357 {
358  NTSTATUS Status;
359  PCM_PARTIAL_RESOURCE_LIST PartialResourceList;
360  PUCHAR Buffer;
361  PCM_PARTIAL_RESOURCE_DESCRIPTOR PartialDescriptor;
362  ULONG BufferSize;
363  PREGQUERYBUFFERXFER RegQueryBufferXfer;
364 
365  PAGED_CODE();
366 
367  UNREFERENCED_PARAMETER (ValueName);
368  UNREFERENCED_PARAMETER (ValueLength);
369  UNREFERENCED_PARAMETER (Context);
370 
371  WmipAssert(EntryContext != NULL);
372 
373  if ((ValueType == REG_FULL_RESOURCE_DESCRIPTOR) &&
374  (ValueData != NULL))
375  {
376  //
377  // On x86 get the actual SMBIOS data out of the registry and
378  // place it into a buffer
379  //
380  RegQueryBufferXfer = (PREGQUERYBUFFERXFER)EntryContext;
381 
382  PartialResourceList = &(((PCM_FULL_RESOURCE_DESCRIPTOR)ValueData)->PartialResourceList);
383  if (PartialResourceList->Count > 1)
384  {
385  //
386  // Second partial resource descriptor contains SMBIOS data. There
387  // should ALWAYS be a second partial resource descriptor and it
388  // may have 0 bytes in the case that SMBIOS data was not collected
389  // by NTDETECT.
390 
391  PartialDescriptor = &PartialResourceList->PartialDescriptors[0];
392  Buffer = (PUCHAR)PartialDescriptor +
393  sizeof(CM_PARTIAL_RESOURCE_DESCRIPTOR) +
394  PartialDescriptor->u.DeviceSpecificData.DataSize;
395  PartialDescriptor = (PCM_PARTIAL_RESOURCE_DESCRIPTOR)Buffer;
396  BufferSize = PartialDescriptor->u.DeviceSpecificData.DataSize;
397  RegQueryBufferXfer->BufferSize = BufferSize;
398  Status = STATUS_SUCCESS;
399  if (BufferSize > 0)
400  {
401  RegQueryBufferXfer->Buffer = (PUCHAR)ExAllocatePoolWithTag(
402  PagedPool,
403  BufferSize,
404  WMIPOOLTAG);
405  if (RegQueryBufferXfer->Buffer != NULL)
406  {
407  Buffer += sizeof(CM_PARTIAL_RESOURCE_DESCRIPTOR);
408  RtlCopyMemory(RegQueryBufferXfer->Buffer,
409  Buffer,
410  BufferSize);
411  } else {
412  Status = STATUS_INSUFFICIENT_RESOURCES;
413  }
414  }
415  } else {
416  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Old NTDETECT.COM - No SMBIOS partial resource descriptor\n"));
417  Status = STATUS_SUCCESS;
418  RegQueryBufferXfer->BufferSize = 0;
419  }
420  } else {
421  Status = STATUS_UNSUCCESSFUL;
422  }
423  return(Status);
424 }

488 BOOLEAN WmipFindSMBiosTable(
489  PPHYSICAL_ADDRESS SMBiosTablePhysicalAddress,
490  PUCHAR *SMBiosTableVirtualAddress,
491  PULONG SMBiosTableLength,
492  PSMBIOSVERSIONINFO SMBiosVersionInfo
493  )
494 /*++
495 
496 Routine Description:
497 
498  Determines if the SMBIOS data is available
499 
500 Arguments:
501 
502  SMBiosTablePhysicalAddress points to a variable to return the physical
503  address of the SMBIOS 2.1 table. If table is not available then
504  it returns with 0.
505 
506  SMBiosTableVirtualAddress points to a variable to return the virtual
507  address of the SMBIOS 2.0 table as collected by NTDETECT. If the
508  SMBIOS 2.0 data was not collected by NTDETECT it returns with 0.
509 
510  SMBiosTableLength points to a variable to return the length of the
511  SMBIOS table data.
512 
513  SMBiosVersionInfo returns with the version information for SMBIOS
514 
515 Return Value:
516 
517  TRUE if SMBIOS data is available, else FALSE
518 
519 --*/
520 {
521  PHYSICAL_ADDRESS BiosPhysicalAddress;
522  PUCHAR BiosVirtualAddress;
523  PDMIBIOS_EPS_HEADER DMIBiosEPSHeader;
524 
525  NTSTATUS Status;
526  UNICODE_STRING BaseKeyName;
527  HANDLE KeyHandle;
528  OBJECT_ATTRIBUTES ObjectAttributes;
529  ULONG KeyIndex;
530  ULONG KeyInformationLength;
531 
532  union {
533  KEY_BASIC_INFORMATION Info;
534  UCHAR Buffer[sizeof(KEY_BASIC_INFORMATION) + (MAXSMBIOSKEYNAMESIZE * sizeof(WCHAR))];
535  } KeyBasic;
536 
537  SMBIOS_EPS_HEADER SMBiosEPSHeader;
538  BOOLEAN HaveEPSHeader = FALSE;
539  BOOLEAN SearchForHeader = TRUE;
540 
541  PAGED_CODE();
542 
543  SMBiosTablePhysicalAddress->QuadPart = 0;
544  *SMBiosTableVirtualAddress = NULL;
545  *SMBiosTableLength = 0;
546 
547  //
548  // First check registry to see if we captured SMBIOS 2.0 data in
549  // NTDETECT. Search the keys under
550  // MultiFunctionAdapter for the one
551  // with the "PnP Bios" (x86)
552  //
553  RtlInitUnicodeString(&BaseKeyName,
554  SMBIOSPARENTKEYNAME);
555 
556  InitializeObjectAttributes(&ObjectAttributes,
557  &BaseKeyName,
558  OBJ_CASE_INSENSITIVE,
559  NULL,
560  NULL);
561 
562  Status = ZwOpenKey(&KeyHandle,
563  KEY_READ,
564  &ObjectAttributes);
565 
566  if (NT_SUCCESS(Status))
567  {
568 
569  KeyIndex = 0;
570 
571  while (NT_SUCCESS(Status))
572  {
573 
574  Status = ZwEnumerateKey(KeyHandle,
575  KeyIndex++,
576  KeyBasicInformation,
577  &KeyBasic.Info,
578  sizeof(KeyBasic.Buffer) - sizeof(WCHAR),
579  &KeyInformationLength);
580  if (NT_SUCCESS(Status))
581  {
582  KeyBasic.Info.Name[KeyBasic.Info.NameLength / sizeof(WCHAR)] = UNICODE_NULL;
583  if (WmipIsSMBiosKey(KeyHandle,
584  KeyBasic.Info.Name,
585  SMBiosTableVirtualAddress,
586  SMBiosTableLength))
587  {
588  if (*SMBiosTableLength != 0)
589  {
590  SMBiosVersionInfo->Used20CallingMethod = TRUE;
591  SearchForHeader = FALSE;
592  }
593  break;
594  }
595  } else {
596  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Status %x enum H\\D\\S\\MultiFunctionAdapter key, index %d\n",
597  Status, KeyIndex-1));
598  }
599  }
600  ZwClose(KeyHandle);
601  } else {
602  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Status %x opening H\\D\\S\\MultiFunctionAdapter key\n",
603  Status));
604  }
605 
606  if (SearchForHeader)
607  {
608  //
609  // If not in registry then check for EPS in the BIOS
610  BiosPhysicalAddress.QuadPart = SMBIOS_EPS_SEARCH_START;
611  BiosVirtualAddress = MmMapIoSpace(BiosPhysicalAddress,
612  SMBIOS_EPS_SEARCH_SIZE,
613  MmNonCached);
614 
615  if (BiosVirtualAddress != NULL)
616  {
617  HaveEPSHeader = WmipFindSMBiosEPSHeader(BiosVirtualAddress,
618  SMBIOS_EPS_SEARCH_SIZE,
619  &SMBiosEPSHeader);
620  MmUnmapIoSpace(BiosVirtualAddress, SMBIOS_EPS_SEARCH_SIZE);
621  }
622  } else {
623  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: SMBIOS data recovered from loader\n"));
624  }
625 
626  if (HaveEPSHeader)
627  {
628  //
629  // We found the EPS so just extract the physical
630  // address of the table
631  //
632  DMIBiosEPSHeader = (PDMIBIOS_EPS_HEADER)&SMBiosEPSHeader.Signature2[0];
633 
634  //
635  // Ignore tables with invalid length sizes
636  //
637  if (DMIBiosEPSHeader->StructureTableLength)
638  {
639  SMBiosVersionInfo->Used20CallingMethod = FALSE;
640 
641  SMBiosTablePhysicalAddress->HighPart = 0;
642  SMBiosTablePhysicalAddress->LowPart = DMIBiosEPSHeader->StructureTableAddress;
643 
644  *SMBiosTableLength = DMIBiosEPSHeader->StructureTableLength;
645 
646  SMBiosVersionInfo->SMBiosMajorVersion = SMBiosEPSHeader.MajorVersion;
647  SMBiosVersionInfo->SMBiosMinorVersion = SMBiosEPSHeader.MinorVersion;
648 
649  SMBiosVersionInfo->DMIBiosRevision = DMIBiosEPSHeader->Revision;
650  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: SMBIOS 2.1 data at (%x%x) size %x \n",
651  SMBiosTablePhysicalAddress->HighPart,
652  SMBiosTablePhysicalAddress->LowPart,
653  *SMBiosTableLength));
654  }
655  }
656 
657  return(*SMBiosTableLength > 0 ? TRUE : FALSE);
658 }

660 NTSTATUS WmipGetSMBiosTableData(
661  OUT PUCHAR Buffer,
662  IN OUT PULONG BufferSize,
663  OUT PSMBIOSVERSIONINFO SMBiosVersionInfo
664  )
665 /*++
666 
667 Routine Description:
668 
669  Registry query values callback routine for reading SMBIOS data from
670  registry.
671 
672 Arguments:
673 
674  Buffer is a pointer to a buffer in which to write the SMBIOS data
675 
676  *BufferSize has the maximum number of bytes available to write into
677  Buffer. On return it has the actual size of the SMBIOS data.
678 
679 Return Value:
680 
681  NT Status code -
682  STATUS_SUCCESS - Buffer filled with SMBIOS data
683  STATUS_BUFFER_TOO_SMALL - Buffer not filled with SMBIOS data,
684  *BufferSize returns with buffer size neeeded
685 
686 --*/
687 {
688  NTSTATUS status;
689  PUCHAR SMBiosDataVirtualAddress;
690 
691  PAGED_CODE();
692 
693  WmipEnterSMCritSection();
694  if (! WmipSMBiosChecked)
695  {
696  //
697  // See if there is any SMBIOS information and if so register
698  WmipFindSMBiosTable(&WmipSMBiosTablePhysicalAddress,
699  &WmipSMBiosTableVirtualAddress,
700  &WmipSMBiosTableLength,
701  &WmipSMBiosVersionInfo);
702  WmipSMBiosChecked = TRUE;
703  }
704  WmipLeaveSMCritSection();
705 
706  if (SMBiosVersionInfo != NULL)
707  {
708  *SMBiosVersionInfo = WmipSMBiosVersionInfo;
709  }
710  if (*BufferSize >= WmipSMBiosTableLength)
711  {
712  if (WmipSMBiosTableLength == 0)
713  {
714  status = STATUS_INVALID_DEVICE_REQUEST;
715  }
716  else if (WmipSMBiosTablePhysicalAddress.QuadPart != 0)
717  {
