950 NTSTATUS
951 CmpRefreshHive(
952  IN PCM_KEY_CONTROL_BLOCK KeyControlBlock
953  )
954 /*++
955 
956 Routine Description:
957 
958  Backs out all changes to a hives since it was last flushed.
959  Used as a transaction abort by the security system.
960 
961  Caller must have SeTcbPrivilege.
962 
963  The target hive must have HIVE_NOLAZYFLUSH set.
964 
965  KeyControlBlock must refer to the root of the hive (HIVE_ENTRY must
966  be set in the key.)
967 
968  Any kcbs that point into this hive (and thus any handles open
969  against it) will be force to DELETED state. (If we do any work.)
970 
971  All notifies pending against the hive will be flushed.
972 
973  When we're done, only the tombstone kcbs, handles, and attached
974  notify blocks will be left.
975 
976  WARNNOTE: Once reads have begun, if the operation fails, the hive
977  will be corrupt, so we will bugcheck.
978 
979 Arguments:
980 
981  KeyControlBlock - provides a reference to the root of the hive
982  we wish to refresh.
983 
984 Return Value:
985 
986  NTSTATUS
987 
988 --*/
989 {
990  PHHIVE Hive;
991  PLIST_ENTRY ptr;
992  PCM_NOTIFY_BLOCK node;
993 
994  CM_PAGED_CODE();
995  //
996  // Check to see if the caller has the privilege to make this call.
997  //
998  if (!SeSinglePrivilegeCheck(SeTcbPrivilege, KeGetPreviousMode())) {
999  return STATUS_PRIVILEGE_NOT_HELD;
1000  }
1001 
1002  if (KeyControlBlock->Delete) {
1003  return(STATUS_KEY_DELETED);
1004  }
1005  CmpLockRegistryExclusive();
1006  Hive = KeyControlBlock->KeyHive;
1007 
1008  if( IsHiveFrozen(((PCMHIVE)Hive)) ) {
1009  //
1010  // deny attempts to clobber with a frozen hive
1011  //
1012  CmpUnlockRegistry();
1013  return STATUS_TOO_LATE;
1014  }
1015 
1016  //
1017  // check to see if hive is of proper type
1018  //
1019  if ( ! (Hive->HiveFlags & HIVE_NOLAZYFLUSH)) {
1020  CmpUnlockRegistry();
1021  return STATUS_INVALID_PARAMETER;
1022  }
1023 
1024  //
1025  // punt if any volatile storage has been allocated
1026  //
1027  if (Hive->Storage[Volatile].Length != 0) {
1028  CmpUnlockRegistry();
1029  return STATUS_UNSUCCESSFUL;
1030  }
1031 
1032  if ( ! (KeyControlBlock->Flags & KEY_HIVE_ENTRY)) {
1033  CmpUnlockRegistry();
1034  return STATUS_INVALID_PARAMETER;
1035  }
1036 
1037  //
1038  // Flush all NotifyBlocks attached to this hive
1039  //
1040  while (TRUE) {
1041 
1042  //
1043  // flush below will edit list, so restart at beginning each time
1044  //
1045  ptr = &(((PCMHIVE)Hive)->NotifyList);
1046  if (ptr->Flink == NULL) {
1047  break;
1048  }
1049 
1050  ptr = ptr->Flink;
1051  node = CONTAINING_RECORD(ptr, CM_NOTIFY_BLOCK, HiveList);
1052  ASSERT((node->KeyBody)->NotifyBlock == node);
1053  CmpFlushNotify(node->KeyBody,TRUE);
1054  }
1055 
1056  //
1057  // Force all kcbs that refer to this hive to the deleted state.
1058  //
1059  CmpSearchKeyControlBlockTree(
1060  CmpRefreshWorkerRoutine,
1061  (PVOID)Hive,
1062  NULL
1063  );
1064 
1065  //
1066  // Call the worker to do the refresh
1067  //
1068  HvRefreshHive(Hive);
1069 
1070  CmpUnlockRegistry();
1071  //
1072  // we're back (rather than bugchecked) so it worked
1073  //
1074  return STATUS_SUCCESS;
1075 }
