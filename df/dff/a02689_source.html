55 
56 /*++
57 
58 Routine Description:
59 
60  This function is called by Phase 1 System Initialization to initialize
61  the Security Reference Monitor. Note that initialization of the
62  Reference Monitor Global State has already been performed in Phase 0
63  initialization to allow access validation routines to operate without
64  having to check that Reference Monitor Initialization is complete.
65 
66  The steps listed below are performed in this routine. The remainder
67  of Reference Monitor initialization requires the LSA subsystem to have run,
68  so that initialization is performed in a separate thread (the RM Command
69  Server Thread, see below), so that the present thread can create the
70  Session Manager which execs the LSA.
71 
72  o Create the Reference Monitor Command LPC port. The LSA subsystem sends
73  commands (e.g. turn on auditing) which change the Reference Monitor
74  Global State.
75  o Create an Event for use in synchronizing with the LSA subsystem. The
76  LSA will signal the event when the portion of LSA initialization upon
77  with the Reference Monitor depends is complete. The Reference Monitor
78  uses another LPC port, called the LSA Command Port to send commands
79  to the LSA, so the RM must know that this port has been created before
80  trying to connect to it.
81  o Create the Reference Monitor Command Server Thread. This thread is
82  a permanent thread of the System Init process that fields the Reference
83  Monitor State Change commands described above.
84 
85 
86 Arguments:
87 
88  None.
89 
90 Return Value:
91 
92  BOOLEAN - TRUE if Rm Initialization (Phase 1) succeeded, else FALSE
93 
94 --*/
95 
96 {
97  NTSTATUS Status;
98  UNICODE_STRING RmCommandPortName;
99  OBJECT_ATTRIBUTES ObjectAttributes;
100  UNICODE_STRING LsaInitEventName;
101  OBJECT_ATTRIBUTES LsaInitEventObjectAttributes;
102  SECURITY_DESCRIPTOR LsaInitEventSecurityDescriptor;
103  ULONG AclSize;
104 
105  PAGED_CODE();
106 
107  //
108  // Create an LPC port called the Reference Monitor Command Port.
109  // This will be used by the LSA to send commands to the Reference
110  // Monitor to update its state data.
111  //
112 
113  RtlInitUnicodeString( &RmCommandPortName, L"\\SeRmCommandPort" );
114 
115  InitializeObjectAttributes(
116  &ObjectAttributes,
117  &RmCommandPortName,
118  0,
119  NULL,
120  NULL
121  );
122 
123  Status = ZwCreatePort(
124  &SepRmState.RmCommandServerPortHandle,
125  &ObjectAttributes,
126  sizeof(SEP_RM_CONNECT_INFO),
127  sizeof(RM_COMMAND_MESSAGE),
128  sizeof(RM_COMMAND_MESSAGE) * 32
129  );
130 
131  if( !NT_SUCCESS(Status) ) {
132 
133  KdPrint(("Security: Rm Create Command Port failed 0x%lx\n", Status));
134  return FALSE;
135  }
136 
137  //
138  // Prepare to create an event for synchronizing with the LSA.
139  // First, build the Security Descriptor for the Init Event Object
140  //
141 
142  Status = RtlCreateSecurityDescriptor(
143  &LsaInitEventSecurityDescriptor,
144  SECURITY_DESCRIPTOR_REVISION
145  );
146 
147  if (!NT_SUCCESS(Status)) {
148 
149  KdPrint(("Security: Creating Lsa Init Event Desc failed 0x%lx\n",
150  Status));
151  return FALSE;
152  }
153 
154  //
155  // Allocate a temporary buffer from the paged pool. It is a fatal
156  // system error if the allocation fails since security cannot be
157  // enabled.
158  //
159 
160  AclSize = sizeof(ACL) +
161  sizeof(ACCESS_ALLOWED_ACE) +
162  SeLengthSid(SeLocalSystemSid);
163  LsaInitEventSecurityDescriptor.Dacl =
164  ExAllocatePoolWithTag(PagedPool, AclSize, 'cAeS');
165 
166  if (LsaInitEventSecurityDescriptor.Dacl == NULL) {
167 
168  KdPrint(("Security LSA: Insufficient resources to initialize\n"));
169  return FALSE;
170  }
171 
172  //
173  // Now create the Discretionary ACL within the Security Descriptor
174  //
175 
176  Status = RtlCreateAcl(
177  LsaInitEventSecurityDescriptor.Dacl,
178  AclSize,
179  ACL_REVISION2
180  );
181 
182  if (!NT_SUCCESS(Status)) {
183 
184  KdPrint(("Security: Creating Lsa Init Event Dacl failed 0x%lx\n",
185  Status));
186  return FALSE;
187  }
188 
189  //
190  // Now add an ACE giving GENERIC_ALL access to the User ID
191  //
192 
193  Status = RtlAddAccessAllowedAce(
194  LsaInitEventSecurityDescriptor.Dacl,
195  ACL_REVISION2,
196  GENERIC_ALL,
197  SeLocalSystemSid
198  );
199 
200  if (!NT_SUCCESS(Status)) {
201 
202  KdPrint(("Security: Adding Lsa Init Event ACE failed 0x%lx\n",
203  Status));
204  return FALSE;
205  }
206 
207  //
208  // Set up the Object Attributes for the Lsa Initialization Event
209  //
210 
211  RtlInitUnicodeString( &LsaInitEventName, L"\\SeLsaInitEvent" );
212 
213  InitializeObjectAttributes(
214  &LsaInitEventObjectAttributes,
215  &LsaInitEventName,
216  0,
217  NULL,
218  &LsaInitEventSecurityDescriptor
219  );
220 
221  //
222  // Create an event for use in synchronizing with the LSA. The LSA will
223  // signal this event when LSA initialization has reached the point
224  // where the LSA's Reference Monitor Server Port has been created.
225  //
226 
227  Status = ZwCreateEvent(
228  &(SepRmState.LsaInitEventHandle),
229  EVENT_MODIFY_STATE,
230  &LsaInitEventObjectAttributes,
231  NotificationEvent,
232  FALSE);
233 
234  if (!NT_SUCCESS(Status)) {
235 
236  KdPrint(("Security: LSA init event creation failed.0x%xl\n",
237  Status));
238  return FALSE;
239  }
240 
241  //
242  // Deallocate the pool memory used for the Init Event DACL
243  //
244 
245  ExFreePool( LsaInitEventSecurityDescriptor.Dacl );
246 
247  //
248  // Create a permanent thread of the Sysinit Process, called the
249  // Reference Monitor Server Thread. This thread is dedicated to
250  // receiving Reference Monitor commands and dispatching them.
251  //
252 
253  Status = PsCreateSystemThread(
254  &SepRmState.SepRmThreadHandle,
255  THREAD_GET_CONTEXT |
256  THREAD_SET_CONTEXT |
257  THREAD_SET_INFORMATION,
258  NULL,
259  NULL,
260  NULL,
261  SepRmCommandServerThread,
262  NULL
263  );
264 
265  if (!NT_SUCCESS(Status)) {
266 
267  KdPrint(("Security: Rm Server Thread creation failed 0x%lx\n", Status));
268  return FALSE;
269  }
270 
271  //
272  // Initialize data from the registry. This must go here because all other
273  // Se initialization takes place before the registry is initialized.
274  //
275 
276  SepAdtInitializeCrashOnFail();
277  SepAdtInitializePrivilegeAuditing();
278  SepAdtInitializeAuditingOptions();
279 
280  //
281  // Reference Monitor initialization is successful if we get to here.
282  //
283 
284  ZwClose( SepRmState.SepRmThreadHandle );
285  SepRmState.SepRmThreadHandle = NULL;
286  return TRUE;
287 }

290 VOID
291 SepRmCommandServerThread(
292  IN PVOID StartContext
293 )
294 
295 /*++
296 
297 Routine Description:
298 
299  This function is executed indefinitely by a dedicated permanent thread
300  of the Sysinit Process, called the Reference Monitor Server Thread.
301  This thread updates Reference Monitor Global State Data by dispatching
302  commands sent from the LSA through the the Reference Monitor LPC Command
303  Port. The following steps are repeated indefinitely:
304 
305  o Initialize RM Command receive and reply buffer headers
306  o Perform remaining Reference Monitor initialization involving LSA
307  o Wait for RM command sent from LSA, send reply to previous command
308  (if any)
309  o Validate command
310  o Dispatch to command worker routine to execute command.
311 
312 Arguments:
313 
314  None.
315 
316 Return Value:
317 
318  None.
319 
320 --*/
321 
322 {
323  NTSTATUS Status;
324  PRM_REPLY_MESSAGE Reply;
325  RM_COMMAND_MESSAGE CommandMessage;
326  RM_REPLY_MESSAGE ReplyMessage;
327 
328  PAGED_CODE();
329 
330  //
331  // Perform the rest of the Reference Monitor initialization, involving
332  // synchronization with the LSA or dependency on the LSA having run.
333  //
334 
335  if (!SepRmCommandServerThreadInit()) {
336 
337  KdPrint(("Security: Terminating Rm Command Server Thread\n"));
338  return;
339  }
340 
341  Status = PoRequestShutdownEvent (NULL);
342  if (!NT_SUCCESS (Status)) {
343  ZwClose (SepRmState.RmCommandPortHandle);
344  ZwClose (SepRmState.RmCommandServerPortHandle);
345  ZwClose (SepRmState.LsaCommandPortHandle);
346  ZwClose (SepLsaHandle);
347  SepRmState.RmCommandPortHandle = NULL;
348  SepRmState.RmCommandServerPortHandle = NULL;
349  SepRmState.LsaCommandPortHandle = NULL;
350  SepLsaHandle = NULL;
351  return;
352  }
353 
354  //
355  // Initialize LPC port message header type and length fields for the
356  // received command message.
357  //
358 
359  CommandMessage.MessageHeader.u2.ZeroInit = 0;
360  CommandMessage.MessageHeader.u1.s1.TotalLength =
361  (CSHORT) sizeof(RM_COMMAND_MESSAGE);
362  CommandMessage.MessageHeader.u1.s1.DataLength =
363  CommandMessage.MessageHeader.u1.s1.TotalLength -
364  (CSHORT) sizeof(PORT_MESSAGE);
365 
366  //
367  // Initialize the LPC port message header type and data sizes for
368  // for the reply message.
369  //
370 
371  ReplyMessage.MessageHeader.u2.ZeroInit = 0;
372  ReplyMessage.MessageHeader.u1.s1.TotalLength =
373  (CSHORT) sizeof(RM_COMMAND_MESSAGE);
374  ReplyMessage.MessageHeader.u1.s1.DataLength =
375  ReplyMessage.MessageHeader.u1.s1.TotalLength -
376  (CSHORT) sizeof(PORT_MESSAGE);
377 
378  //
379  // First time through, there is no reply.
380  //
381 
382  Reply = NULL;
383 
384  //
385  // Now loop indefinitely, processing incoming Rm commands from the LSA.
386  //
387 
388  for(;;) {
389 
390  //
391  // Wait for Command, send reply to previous command (if any)
392  //
393 
394  Status = ZwReplyWaitReceivePort(
395  SepRmState.RmCommandPortHandle,
396  NULL,
397  (PPORT_MESSAGE) Reply,
398  (PPORT_MESSAGE) &CommandMessage
399  );
400 
401  if (!NT_SUCCESS(Status)) {
402 
403  //
404  // malicious user apps can try to connect to this port. We will
405  // fail later, but if their thread vanishes, we'll get a failure
406  // here. Ignore it:
407  //
408 
409  if (Status == STATUS_UNSUCCESSFUL ||
410  Status == STATUS_INVALID_CID ||
411  Status == STATUS_REPLY_MESSAGE_MISMATCH)
412  {
413  //
414  // skip it:
415  //
416 
417  Reply = NULL ;
418  continue;
419  }
420 
421  KdPrint(("Security: RM message receive from Lsa failed %lx\n",
422  Status));
423 
424  }
425 
426  //
427  // Now dispatch to a routine to handle the command. Allow
428  // command errors to occur without bringing system down just now.
429  //
430 
431  CommandMessage.MessageHeader.u2.s2.Type &= ~LPC_KERNELMODE_MESSAGE;
432 
433  if ( CommandMessage.MessageHeader.u2.s2.Type == LPC_REQUEST ) {
434 
435  if ( (CommandMessage.CommandNumber >= RmAuditSetCommand) &&
436  (CommandMessage.CommandNumber <= RmDeleteLogonSession) ) {
437 
438  (*(SepRmCommandDispatch[CommandMessage.CommandNumber]))
439  (&CommandMessage, &ReplyMessage);
440 
441  //
442  // Initialize the client thread info and message id for the
443  // reply message. First time through, the reply message structure
444  // is not used.
445  //
446 
447  ReplyMessage.MessageHeader.ClientId =
448  CommandMessage.MessageHeader.ClientId;
449  ReplyMessage.MessageHeader.MessageId =
450  CommandMessage.MessageHeader.MessageId;
451 
452  Reply = &ReplyMessage;
453 
454  } else {
455 
456  ASSERT( (CommandMessage.CommandNumber >= RmAuditSetCommand) &&
457  (CommandMessage.CommandNumber <= RmDeleteLogonSession) );
458  Reply = NULL;
459  }
460 
461  } else if (CommandMessage.MessageHeader.u2.s2.Type == LPC_PORT_CLOSED ) {
462  KEVENT Event;
463  BOOLEAN Wait;
464 
465  KeInitializeEvent (&Event, NotificationEvent, FALSE);
466 
467  SepLockLsaQueue();
468 
469  SepAdtLsaDeadEvent = &Event;
470 
471  Wait = !SepWorkListEmpty ();
472 
473  SepUnlockLsaQueue();
474 
475  if (Wait) {
476  KeWaitForSingleObject (&Event,
477  Executive,
478  KernelMode,
479  FALSE,
480  NULL);
481  }
482  //
483  // Our only client closed its handle. Tidy up and exit.
484  //
485  ZwClose (SepRmState.LsaCommandPortHandle);
486  ZwClose (SepRmState.RmCommandPortHandle);
487  ZwClose (SepRmState.RmCommandServerPortHandle);
488  ZwClose (SepLsaHandle);
489  SepRmState.LsaCommandPortHandle = NULL;
490  SepRmState.RmCommandPortHandle = NULL;
491  SepRmState.RmCommandServerPortHandle = NULL;
492  SepLsaHandle = NULL;
493  break;
494  } else if (CommandMessage.MessageHeader.u2.s2.Type == LPC_CONNECTION_REQUEST) {
495  HANDLE tmp;
496  //
497  // Reject extra connection attempts
498  //
499  Status = ZwAcceptConnectPort(&tmp,
500  NULL,
501  (PPORT_MESSAGE) &CommandMessage,
502  FALSE,
503  NULL,
504  NULL);
505  } else {
506 
507  Reply = NULL;
508  }
509  } // end_for
510 
511  UNREFERENCED_PARAMETER( StartContext );
512 
513 }

517 SepRmCommandServerThreadInit(

521 /*++
522 
523 Routine Description:
524 
525  This function performs initialization of the Reference Monitor Server
526  thread. The following steps are performed.
527 
528  o Wait on the LSA signaling the event. When the event is signaled,
529  the LSA has already created the LSA Command Server LPC Port
530  o Close the LSA Init Event Handle. The event is not used again.
531  o Listen for the LSA to connect to the Port
532  o Accept the connection.
533  o Connect to the LSA Command Server LPC Port
534 
535 Arguments:
536 
537  None.
538 
539 Return Value:
540 
541 --*/
542 
543 {
544  NTSTATUS Status;
545  UNICODE_STRING LsaCommandPortName;
546  PORT_MESSAGE ConnectionRequest;
547  SECURITY_QUALITY_OF_SERVICE DynamicQos;
548  OBJECT_ATTRIBUTES ObjectAttributes;
549  PORT_VIEW ClientView;
550  REMOTE_PORT_VIEW LsaClientView;
551  BOOLEAN BooleanStatus = TRUE;
552 
553  PAGED_CODE();
554 
555  //
556  // Save a pointer to our process so we can get back into this process
557  // to send commands to the LSA (using a handle to an LPC port created
558  // below).
559  //
560 
561  SepRmLsaCallProcess = PsGetCurrentProcess();
562 
563  ObReferenceObject(SepRmLsaCallProcess);
564 
565  //
566  // Wait on the LSA signaling the event. This means that the LSA
567  // has created its command port, not that LSA initialization is
568  // complete.
569  //
570 
571  Status = ZwWaitForSingleObject(
572  SepRmState.LsaInitEventHandle,
573  FALSE,
574  NULL);
575 
576  if ( !NT_SUCCESS(Status) ) {
577 
578  KdPrint(("Security Rm Init: Waiting for LSA Init Event failed 0x%lx\n", Status));
579  goto RmCommandServerThreadInitError;
580  }
581 
582  //
583  // Close the LSA Init Event Handle. The event is not used again.
584  //
585 
586  ZwClose(SepRmState.LsaInitEventHandle);
587 
588  //
589  // Listen for a connection to be made by the LSA to the Reference Monitor
590  // Command Port. This connection will be made by the LSA process.
591  //
592 
593  ConnectionRequest.u1.s1.TotalLength = sizeof(ConnectionRequest);
594  ConnectionRequest.u1.s1.DataLength = (CSHORT)0;
595  Status = ZwListenPort(
596  SepRmState.RmCommandServerPortHandle,
597  &ConnectionRequest
598  );
599 
600  if (!NT_SUCCESS(Status)) {
601 
602  KdPrint(("Security Rm Init: Listen to Command Port failed 0x%lx\n",
603  Status));
604  goto RmCommandServerThreadInitError;
605  }
606 
607  //
608  // Obtain a handle to the LSA process for use when auditing.
609  //
610 
611  InitializeObjectAttributes( &ObjectAttributes, NULL, 0, NULL, NULL );
612 
613  Status = ZwOpenProcess(
614  &SepLsaHandle,
615  PROCESS_VM_OPERATION | PROCESS_VM_WRITE,
616  &ObjectAttributes,
617  &ConnectionRequest.ClientId
618  );
619 
620  if (!NT_SUCCESS(Status)) {
621 
622  KdPrint(("Security Rm Init: Open Listen to Command Port failed 0x%lx\n",
623  Status));
624  goto RmCommandServerThreadInitError;
625  }
626 
627  //
628  // Accept the connection made by the LSA process.
629  //
630 
631  LsaClientView.Length = sizeof(LsaClientView);
632 
633 
634  Status = ZwAcceptConnectPort(
635  &SepRmState.RmCommandPortHandle,
636  NULL,
637  &ConnectionRequest,
638  TRUE,
639  NULL,
640  &LsaClientView
641  );
642 
643  if (!NT_SUCCESS(Status)) {
644 
645  KdPrint(("Security Rm Init: Accept Connect to Command Port failed 0x%lx\n",
646  Status));
647 
648  goto RmCommandServerThreadInitError;
649  }
650 
651  //
652  // Complete the connection.
653  //
654 
655  Status = ZwCompleteConnectPort(SepRmState.RmCommandPortHandle);
656 
657  if (!NT_SUCCESS(Status)) {
658 
659  KdPrint(("Security Rm Init: Complete Connect to Command Port failed 0x%lx\n",
660  Status));
661  goto RmCommandServerThreadInitError;
662  }
663 
664  //
665  // Set up the security quality of service parameters to use over the
666  // Lsa Command LPC port. Use the most efficient (least overhead) - which
667  // is dynamic rather than static tracking.
668  //
669 
670  DynamicQos.ImpersonationLevel = SecurityImpersonation;
671  DynamicQos.ContextTrackingMode = SECURITY_DYNAMIC_TRACKING;
672  DynamicQos.EffectiveOnly = TRUE;
673 
674  //
675  // Create the section to be used as unnamed shared memory for
676  // communication between the RM and LSA.
677  //
678 
679  SepRmState.LsaCommandPortSectionSize.LowPart = PAGE_SIZE;
680  SepRmState.LsaCommandPortSectionSize.HighPart = 0;
681 
682  Status = ZwCreateSection(
683  &SepRmState.LsaCommandPortSectionHandle,
684  SECTION_ALL_ACCESS,
685  NULL, // ObjectAttributes
686  &SepRmState.LsaCommandPortSectionSize,
687  PAGE_READWRITE,
688  SEC_COMMIT,
689  NULL // FileHandle
690  );
691 
692  if (!NT_SUCCESS(Status)) {
693 
694  KdPrint(("Security Rm Init: Create Memory Section for LSA port failed: %X\n", Status));
695  goto RmCommandServerThreadInitError;
696  }
697 
698  //
699  // Set up for a call to NtConnectPort and connect to the LSA port.
700  // This setup includes a description of the port memory section so that
701  // the LPC connection logic can make the section visible to both the
702  // client and server processes.
703  //
704 
705  ClientView.Length = sizeof(ClientView);
706  ClientView.SectionHandle = SepRmState.LsaCommandPortSectionHandle;
707  ClientView.SectionOffset = 0;
708  ClientView.ViewSize = SepRmState.LsaCommandPortSectionSize.LowPart;
709  ClientView.ViewBase = 0;
710  ClientView.ViewRemoteBase = 0;
711 
712  //
713  // Set up the security quality of service parameters to use over the
714  // port. Use dynamic tracking so that XACTSRV will impersonate the
715  // user that we are impersonating when we call NtRequestWaitReplyPort.
716  // If we used static tracking, XACTSRV would impersonate the context
717  // when the connection is made.
718  //
719 
720  DynamicQos.ImpersonationLevel = SecurityImpersonation;
721  DynamicQos.ContextTrackingMode = SECURITY_DYNAMIC_TRACKING;
722  DynamicQos.EffectiveOnly = TRUE;
723 
724  //
725  // Connect to the Lsa Command LPC Port. This port is used to send
726  // commands from the RM to the LSA.
727  //
728 
729  RtlInitUnicodeString( &LsaCommandPortName, L"\\SeLsaCommandPort" );
730 
731  Status = ZwConnectPort(
732  &SepRmState.LsaCommandPortHandle,
733  &LsaCommandPortName,
734  &DynamicQos,
735  &ClientView,
736  NULL, // ServerView
737  NULL, // MaxMessageLength
738  NULL, // ConnectionInformation
739  NULL // ConnectionInformationLength
740  );
741 
742  if (!NT_SUCCESS(Status)) {
743 
744  KdPrint(("Security Rm Init: Connect to LSA Port failed 0x%lx\n", Status));
745  goto RmCommandServerThreadInitError;
746  }
747 
748  //
749  // Store information about the section so that we can create pointers
750  // meaningful to LSA.
751  //
752 
753  SepRmState.RmViewPortMemory = ClientView.ViewBase;
754  SepRmState.LsaCommandPortMemoryDelta =
755  (LONG)((ULONG_PTR)ClientView.ViewRemoteBase - (ULONG_PTR) ClientView.ViewBase );
756  SepRmState.LsaViewPortMemory = ClientView.ViewRemoteBase;
757 
758 
759 RmCommandServerThreadInitFinish:
760 
761  //
762  // Dont need this section handle any more, even if returning
763  // success.
764  //
765 
766  if ( SepRmState.LsaCommandPortSectionHandle != NULL ) {
767 
768  NtClose( SepRmState.LsaCommandPortSectionHandle );
769  SepRmState.LsaCommandPortSectionHandle = NULL;
770  }
771 
772  //
773  // The Reference Monitor Thread has successfully initialized.
774  //
775 
776  return BooleanStatus;
777 
778 RmCommandServerThreadInitError:
779 
780  if ( SepRmState.LsaCommandPortHandle != NULL ) {
781 
782  NtClose( SepRmState.LsaCommandPortHandle );
783  SepRmState.LsaCommandPortHandle = NULL;
784  }
785 
786  BooleanStatus = FALSE;
787  goto RmCommandServerThreadInitFinish;
788 }

793 NTSTATUS
794 SepRmCallLsa(
795  PSEP_WORK_ITEM SepWorkItem
796  )
797 /*++
798 
799 Routine Description:
800 
801  This function sends a command to the LSA via the LSA Reference Monitor
802  Server Command LPC Port. If the command has parameters, they will be
803  copied directly into a message structure and sent via LPC, therefore,
804  the supplied parameters may not contain any absolute pointers. A caller
805  must remove pointers by "marshalling" them into the buffer CommandParams.
806 
807  This function will create a queue of requests. This is in order to allow
808  greater throughput for the majority if its callers. If a thread enters
809  this routine and finds the queue empty, it is the responsibility of that
810  thread to service all requests that come in while it is working until the
811  queue is empty again. Other threads that enter will simply hook their work
812  item onto the queue and exit.
813 
814 
815  To implement a new LSA command, do the following:
816  ================================================
817 
818  (1) If the command takes no parameters, just call this routine directly
819  and provide an LSA worker routine called Lsap<command>Wrkr. See
