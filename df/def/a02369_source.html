442 BOOLEAN
443 KiPreprocessFault (
444  IN OUT PEXCEPTION_RECORD ExceptionRecord,
445  IN PKTRAP_FRAME TrapFrame,
446  IN OUT PCONTEXT ContextRecord,
447  IN KPROCESSOR_MODE PreviousMode
448  )
449 
450 /*++
451 
452 Routine Description:
453 
454  A wide variety of exception conditions can result in a general
455  protection fault.
456 
457  This routine attempts to further determine the cause of the fault
458  and take appropriate action, which can include
459 
460  - Updating the context record with a more appropriate status
461 
462  - Emulating the instruction or otherwise modifying the caller's
463  context and/or istream and returning
464 
465 Arguments:
466 
467  ExceptionRecord - Supplies a pointer to an exception record.
468 
469  ContextRecord - Supplies a pointer to a context record.
470 
471  TrapFrame - Supplies a pointer to the trap frame.
472 
473  PreviousMode - Supplies the execution mode at the time of the exception.
474 
475 Return Value:
476 
477  TRUE - the caller's context should be updated and control returned
478  to the caller.
479 
480  FALSE - the exception should be raised with (perhaps) an updated
481  context record.
482 
483 --*/
484 
485 {
486  DECODE_CONTEXT decodeContext;
487  PDECODE_ENTRY decodeEntry;
488  ULONG faultType;
489  BOOLEAN retry;
490  NTSTATUS status;
491 
492  //
493  // Determine whether this is an internal exception and if so replace
494  // it with the canonical one and continue processing.
495  //
496 
497  switch (ExceptionRecord->ExceptionCode) {
498 
499  case KI_EXCEPTION_GP_FAULT:
500 
501  faultType = OPF_GP;
502  ExceptionRecord->ExceptionCode = STATUS_ACCESS_VIOLATION;
503  break;
504 
505  case KI_EXCEPTION_INVALID_OP:
506  faultType = OPF_INVAL;
507  ExceptionRecord->ExceptionCode = STATUS_ILLEGAL_INSTRUCTION;
508  break;
509 
510  case KI_EXCEPTION_INTEGER_DIVIDE_BY_ZERO:
511  faultType = OPF_DIV;
512  ExceptionRecord->ExceptionCode = STATUS_INTEGER_DIVIDE_BY_ZERO;
513  break;
514 
515  case KI_EXCEPTION_ACCESS_VIOLATION:
516  retry = KiOpPreprocessAccessViolation(ExceptionRecord,
517  ContextRecord);
518  if (retry != FALSE) {
519  return TRUE;
520  }
521  faultType = OPF_AV;
522  ExceptionRecord->ExceptionCode = STATUS_ACCESS_VIOLATION;
523  break;
524 
525  default:
526  return FALSE;
527  }
528 
529  //
530  // Decode the offending instruction. This tells us the length of the
531  // instruction.
532  //
533 
534  status = KiOpDecode(ExceptionRecord,
535  ContextRecord,
536  PreviousMode,
537  &decodeContext);
538 
539  //
540  // Now that the instruction has been decoded, determine whether
541  // any code patches have taken place between the time of the exception
542  // and now.
543  //
544  // If so, re-execute the code that generated the exception. If it was
545  // not this code that was patched, then another exception will fire and
546  // we'll try again.
547  //
548 
549  if (TrapFrame->CodePatchCycle != KiCodePatchCycle) {
550  return TRUE;
551  }
552 
553  if (NT_SUCCESS(status)) {
554 
555  //
556  // If this is an interesting instruction (i.e. it appears in the
557  // decode tables) then call its handler.
558  //
559 
560  decodeEntry = decodeContext.DecodeEntry;
561  if (decodeEntry != NULL) {
562  status = decodeEntry->Handler(&decodeContext);
563  }
564  } else {
565  decodeEntry = NULL;
566  }
567 
568  //
569  // Indicate whether the operation should be retried, and return.
570  //
571 
572  if (NT_SUCCESS(status) && (decodeEntry != NULL)) {
573 
574  retry = decodeContext.Retry;
575 
576  } else {
577 
578  if (faultType == OPF_GP) {
579 
580  //
581  // If a specific action was not carried out for a general
582  // protection fault then the default action is to raise an
583  // access violation.
584  //
585 
586  //
587  // Set two parameters to "read", "unknown faulting VA"
588  //
589 
590  ExceptionRecord->NumberParameters = 2;
591  ExceptionRecord->ExceptionInformation[0] = 0;
592  ExceptionRecord->ExceptionInformation[1] = (ULONG_PTR)-1;
593  }
594 
595  retry = FALSE;
596  }
597 
598  if ((retry == FALSE) && (faultType == OPF_AV)) {
599 
600  //
601  // Check for an ATL thunk
602  //
603 
604  if (NT_SUCCESS(KiCheckForAtlThunk(&decodeContext))) {
605  retry = decodeContext.Retry;
606  }
607  }
608 
609  return retry;
610 }

612 NTSTATUS
613 KiOpDecode (
614  IN OUT PEXCEPTION_RECORD ExceptionRecord,
615  IN OUT PCONTEXT ContextRecord,
616  IN KPROCESSOR_MODE PreviousMode,
617  OUT PDECODE_CONTEXT DecodeContext
618  )
619 
620 /*++
621 
622 Routine Description:
623 
624  This routine decodes the instruction stream sufficiently to positively
625  determine whether this is an opcode we are interested in and, if so,
626  to positively identify it.
627 
628 Arguments:
629 
630  ExceptionRecord - Supplies a pointer to an exception record.
631 
632  ContextRecord - Supplies a pointer to a context record.
633 
634  PreviousMode - Supplies the execution mode at the time of the exception.
635 
636  DecodeContext - Supplies a pointer to an uninitialized DECODE_CONTEXT.
637 
638 
639 Return Value:
640 
641  NTSTATUS - Status of the operation.
642 
643 --*/
644 
645 {
646  CHAR byteVal;
647  PDECODE_ENTRY decodeEntry;
648  BOOLEAN isPrefix;
649  LONG longVal;
650  UCHAR opByte;
651  ULONG prefixLength;
652  PUCHAR rip;
653  NTSTATUS status;
654  SHORT shortVal;
655 
656  //
657  // Initialize the decode context according to the parameters supplied.
658  //
659  // Also, make sure the instruction pointer refers to the user mode
660  // data area.
661  //
662 
663  RtlZeroMemory(DecodeContext,sizeof(*DecodeContext));
664 
665  rip = (PUCHAR)ContextRecord->Rip;
666 
667  //
668  // Make sure the faulting address is valid. Length of the region
669  // is 1, as we don't yet know how the true length of the opcode.
670  //
671 
672  if (PreviousMode == UserMode) {
673  try {
674  ProbeAndReadUchar(rip);
675  } except (EXCEPTION_EXECUTE_HANDLER) {
676  return GetExceptionCode();
677  }
678  }
679 
680  DecodeContext->Start = rip;
681  DecodeContext->Next = rip;
682  DecodeContext->ExceptionRecord = ExceptionRecord;
683  DecodeContext->ContextRecord = ContextRecord;
684  DecodeContext->PreviousMode = PreviousMode;
685 
686  //
687  // Record whether the fault occured in a compatibility mode segment.
688  //
689 
690  if (ContextRecord->SegCs == (KGDT64_R3_CMCODE | RPL_MASK)) {
691 
692  ASSERT( PreviousMode == UserMode );
693  DecodeContext->CompatibilityMode = TRUE;
694  }
695 
696  //
697  // Accumulate all of the prefix bytes
698  //
699 
700  prefixLength = 0;
701  do {
702 
703  status = KiOpFetchNextByte(DecodeContext,&opByte);
704  if (!NT_SUCCESS(status)) {
705  return status;
706  }
707 
708  status = KiOpIsPrefix(DecodeContext,opByte,&isPrefix);
709  if (!NT_SUCCESS(status)) {
710  return status;
711  }
712 
713  } while (isPrefix != FALSE);
714 
715  //
716  // opByte contains the first base byte of the opcode. If this is
717  // a two byte opcode, retrieve the second byte.
718  //
719 
720  if (opByte == OP_TWOBYTE) {
721 
722  DecodeContext->TwoByte = TRUE;
723 
724  status = KiOpFetchNextByte(DecodeContext,&opByte);
725  if (!NT_SUCCESS(status)) {
726  return status;
727  }
728  }
729 
730  DecodeContext->OpCodeLocation = DecodeContext->Next - 1;
731  DecodeContext->OpCode = opByte;
732  status = KiOpLocateDecodeEntry(DecodeContext);
733  if (!NT_SUCCESS(status)) {
734  return status;
735  }
736 
737  //
738  // If there was no decode entry then we are not interested in handling
739  // this instruction. This is not an error condition, so return
740  // success.
741  //
742 
743  decodeEntry = DecodeContext->DecodeEntry;
744  if (decodeEntry == NULL) {
745  return STATUS_SUCCESS;
746  }
747 
748  //
749  // Decode any address portion of the instruction
750  //
751 
752  if ((decodeEntry->Flags & OPF_MODRM) != 0) {
753  status = KiOpDecodeModRM(DecodeContext);
754  if (!NT_SUCCESS(status)) {
755  return status;
756  }
757  }
758 
759  //
760  // Followed by a possible 8-, 16- or 32-bit immediate
761  //
762 
763  if ((decodeEntry->Flags & OPF_IMM8) != 0) {
764 
765  //
766  // 8-bit displacement
767  //
768 
769  status = KiOpFetchBytes(DecodeContext,
770  sizeof(byteVal),
771  &byteVal);
772  if (!NT_SUCCESS(status)) {
773  return status;
774  }
775  DecodeContext->Immediate = byteVal;
776 
777  } else if ((DecodeContext->DecodeEntry->Flags & OPF_IMM32) != 0) {
778 
779  if ((DecodeContext->PrefixMask & OP_PREFIX_OPERAND) != 0) {
780 
781  //
782  // 16-bit immediate
783  //
784 
785  status = KiOpFetchBytes(DecodeContext,
786  sizeof(shortVal),
787  &shortVal);
788  if (!NT_SUCCESS(status)) {
789  return status;
790  }
791  DecodeContext->Immediate = shortVal;
792 
793  } else {
794 
795  //
796  // 32-bit immediate
797  //
798 
799  status = KiOpFetchBytes(DecodeContext,
800  sizeof(longVal),
801  &longVal);
802  if (!NT_SUCCESS(status)) {
803  return status;
804  }
805  DecodeContext->Immediate = longVal;
806  }
