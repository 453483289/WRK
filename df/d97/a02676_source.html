142 SeCaptureObjectTypeList (
143  __in_ecount_opt(ObjectTypeListLength) POBJECT_TYPE_LIST ObjectTypeList,
144  __in ULONG ObjectTypeListLength,

146  __out PIOBJECT_TYPE_LIST *CapturedObjectTypeList
147 )
148 /*++
149 
150 Routine Description:
151 
152  This routine probes and captures a copy of any object type list
153  that might have been provided via the ObjectTypeList argument.
154 
155  The object type list is converted to the internal form that explicitly
156  specifies the hierarchical relationship between the entries.
157 
158  The object types list is validated to ensure a valid hierarchical
159  relationship is represented.
160 
161 Arguments:
162 
163  ObjectTypeList - The object type list from which the type list
164  information is to be retrieved.
165 
166  ObjectTypeListLength - Number of elements in ObjectTypeList
167 
168  RequestorMode - Indicates the processor mode by which the access
169  is being requested.
170 
171  CapturedObjectTypeList - Receives the captured type list which
172  must be freed using SeFreeCapturedObjectTypeList().
173 
174 Return Value:
175 
176  STATUS_SUCCESS indicates no exceptions were encountered.
177 
178  Any access violations encountered will be returned.
179 
180 --*/
181 
182 {
183  NTSTATUS Status = STATUS_SUCCESS;
184  ULONG i;
185  PIOBJECT_TYPE_LIST LocalTypeList = NULL;
186 
187  ULONG Levels[ACCESS_MAX_LEVEL+1];
188 
189  PAGED_CODE();
190 
191  //
192  // Set default return
193  //
194 
195  *CapturedObjectTypeList = NULL;
196 
197  if (RequestorMode != UserMode) {
198  return STATUS_NOT_IMPLEMENTED;
199  }
200 
201  try {
202 
203  if ( ObjectTypeListLength == 0 ) {
204 
205  // Drop through
206 
207  } else if ( !ARGUMENT_PRESENT(ObjectTypeList) ) {
208 
209  Status = STATUS_INVALID_PARAMETER;
210 
211  } else {
212 
213  if ( !IsValidElementCount( ObjectTypeListLength, IOBJECT_TYPE_LIST ) )
214  {
215  Status = STATUS_INVALID_PARAMETER ;
216 
217  //
218  // No more to do, get out of the try statement:
219  //
220 
221  leave ;
222  }
223 
224  ProbeForRead( ObjectTypeList,
225  sizeof(OBJECT_TYPE_LIST) * ObjectTypeListLength,
226  sizeof(ULONG)
227  );
228 
229  //
230  // Allocate a buffer to copy into.
231  //
232 
233  LocalTypeList = ExAllocatePoolWithTag( PagedPool, sizeof(IOBJECT_TYPE_LIST) * ObjectTypeListLength, 'tOeS' );
234 
235  if ( LocalTypeList == NULL ) {
236  Status = STATUS_INSUFFICIENT_RESOURCES;
237 
238  //
239  // Copy the callers structure to the local structure.
240  //
241 
242  } else {
243  GUID * CapturedObjectType;
244  for ( i=0; i<ObjectTypeListLength; i++ ) {
245  USHORT CurrentLevel;
246 
247  //
248  // Limit ourselves
249  //
250  CurrentLevel = ObjectTypeList[i].Level;
251  if ( CurrentLevel > ACCESS_MAX_LEVEL ) {
252  Status = STATUS_INVALID_PARAMETER;
253  break;
254  }
255 
256  //
257  // Copy data the caller passed in
258  //
259  LocalTypeList[i].Level = CurrentLevel;
260  LocalTypeList[i].Flags = 0;
261  CapturedObjectType = ObjectTypeList[i].ObjectType;
262  ProbeForReadSmallStructure(
263  CapturedObjectType,
264  sizeof(GUID),
265  sizeof(ULONG)
266  );
267  LocalTypeList[i].ObjectType = *CapturedObjectType;
268  LocalTypeList[i].Remaining = 0;
269  LocalTypeList[i].CurrentGranted = 0;
270  LocalTypeList[i].CurrentDenied = 0;
271 
272  //
273  // Ensure that the level number is consistent with the
274  // level number of the previous entry.
275  //
276 
277  if ( i == 0 ) {
278  if ( CurrentLevel != 0 ) {
279  Status = STATUS_INVALID_PARAMETER;
280  break;
281  }
282 
283  } else {
284 
285  //
286  // The previous entry is either:
287  // my immediate parent,
288  // my sibling, or
289  // the child (or grandchild, etc.) of my sibling.
290  //
291  if ( CurrentLevel > LocalTypeList[i-1].Level + 1 ) {
292  Status = STATUS_INVALID_PARAMETER;
293  break;
294  }
295 
296  //
297  // Don't support two roots.
298  //
299  if ( CurrentLevel == 0 ) {
300  Status = STATUS_INVALID_PARAMETER;
301  break;
302  }
303 
304  }
305 
306  //
307  // If the above rules are maintained,
308  // then my parent object is the last object seen that
309  // has a level one less than mine.
310  //
311 
312  if ( CurrentLevel == 0 ) {
313  LocalTypeList[i].ParentIndex = -1;
314  } else {
315  LocalTypeList[i].ParentIndex = Levels[CurrentLevel-1];
316  }
317 
318  //
319  // Save this object as the last object seen at this level.
320  //
321 
322  Levels[CurrentLevel] = i;
323 
324  }
325 
326  }
327 
328  } // end_if
329 
330  } except(EXCEPTION_EXECUTE_HANDLER) {
331 
332  Status = GetExceptionCode();
333 
334  } // end_try
335 
336 
337  if ( NT_SUCCESS( Status ) ) {
338 
339  *CapturedObjectTypeList = LocalTypeList;
340 
341  } else {
342 
343  //
344  // If we captured any proxy data, we need to free it now.
345  //
346 
347  if ( LocalTypeList != NULL ) {
348  ExFreePool( LocalTypeList );
349  }
350  }
351 
352  return Status;
353 }

444 VOID
445 SepUpdateParentTypeList (
446  IN PIOBJECT_TYPE_LIST ObjectTypeList,
447  IN ULONG ObjectTypeListLength,
448  IN ULONG StartIndex
449 )
450 /*++
451 
452 Routine Description:
453 
454  Update the Access fields of the parent object of the specified object.
455 
456 
457  The "remaining" field of a parent object is the logical OR of
458  the remaining field of all of its children.
459 
460  The CurrentGranted field of the parent is the collection of bits
461  granted to every one of its children..
462 
463  The CurrentDenied fields of the parent is the logical OR of
464  the bits denied to any of its children.
465 
466  This routine takes an index to one of the children and updates the
467  remaining field of the parent (and grandparents recursively).
468 
469 Arguments:
470 
471  ObjectTypeList - The object type list to update.
472 
473  ObjectTypeListLength - Number of elements in ObjectTypeList
474 
475  StartIndex - Index to the "child" element whose parents are to be updated.
476 
477 Return Value:
478 
479  None.
480 
481 
482 --*/
483 
484 {
485  ULONG Index;
486  ULONG ParentIndex;
487  ULONG Level;
488  ACCESS_MASK NewRemaining = 0;
489  ACCESS_MASK NewCurrentGranted = 0xFFFFFFFF;
490  ACCESS_MASK NewCurrentDenied = 0;
491 
492  PAGED_CODE();
493 
494  //
495  // If the target node is at the root,
496  // we're all done.
497  //
498 
499  if ( ObjectTypeList[StartIndex].ParentIndex == -1 ) {
500  return;
501  }
502 
503  //
504  // Get the index to the parent that needs updating and the level of
505  // the siblings.
506  //
507 
508  ParentIndex = ObjectTypeList[StartIndex].ParentIndex;
509  Level = ObjectTypeList[StartIndex].Level;
510 
511  //
512  // Loop through all the children.
513  //
514 
515  for ( Index=ParentIndex+1; Index<ObjectTypeListLength; Index++ ) {
516 
517  //
518  // By definition, the children of an object are all those entries
519  // immediately following the target. The list of children (or
520  // grandchildren) stops as soon as we reach an entry the has the
521  // same level as the target (a sibling) or lower than the target
522  // (an uncle).
523  //
524 
525  if ( ObjectTypeList[Index].Level <= ObjectTypeList[ParentIndex].Level ) {
526  break;
527  }
528 
529  //
530  // Only handle direct children of the parent.
531  //
532 
533  if ( ObjectTypeList[Index].Level != Level ) {
534  continue;
535  }
536 
537  //
538  // Compute the new bits for the parent.
539  //
540 
541  NewRemaining |= ObjectTypeList[Index].Remaining;
542  NewCurrentGranted &= ObjectTypeList[Index].CurrentGranted;
543  NewCurrentDenied |= ObjectTypeList[Index].CurrentDenied;
544 
545  }
546 
547  //
548  // If we've not changed the access to the parent,
549  // we're done.
550  //
551 
552  if ( NewRemaining == ObjectTypeList[ParentIndex].Remaining &&
553  NewCurrentGranted == ObjectTypeList[ParentIndex].CurrentGranted &&
554  NewCurrentDenied == ObjectTypeList[ParentIndex].CurrentDenied ) {
555  return;
556  }
557 
558 
559  //
560  // Change the parent.
561  //
562 
563  ObjectTypeList[ParentIndex].Remaining = NewRemaining;
564  ObjectTypeList[ParentIndex].CurrentGranted = NewCurrentGranted;
565  ObjectTypeList[ParentIndex].CurrentDenied = NewCurrentDenied;
566 
567  //
568  // Go update the grand parents.
569  //
570 
571  SepUpdateParentTypeList( ObjectTypeList,
572  ObjectTypeListLength,
573  ParentIndex );
574 }

577 VOID
578 SepAddAccessTypeList (
579  IN PIOBJECT_TYPE_LIST ObjectTypeList,
580  IN ULONG ObjectTypeListLength,
581  IN ULONG StartIndex,
582  IN ACCESS_MASK AccessMask,
583  IN ACCESS_MASK_FIELD_TO_UPDATE FieldToUpdate
584 )
585 /*++
586 
587 Routine Description:
588 
589  This routine grants the specified AccessMask to all of the objects that
590  are descendents of the object specified by StartIndex.
591 
592  The Access fields of the parent objects are also recomputed as needed.
593 
594  For example, if an ACE granting access to a Property Set is found,
595  that access is granted to all the Properties in the Property Set.
596 
597 Arguments:
598 
599  ObjectTypeList - The object type list to update.
600 
601  ObjectTypeListLength - Number of elements in ObjectTypeList
602 
603  StartIndex - Index to the target element to update.
604 
605  AccessMask - Mask of access to grant to the target element and
606  all of its descendents
607 
608  FieldToUpdate - Indicate which fields to update in object type list
609 
610 Return Value:
611 
612  None.
613 
614 --*/
615 
616 {
617  ULONG Index;
618  ACCESS_MASK OldRemaining;
619  ACCESS_MASK OldCurrentGranted;
620  ACCESS_MASK OldCurrentDenied;
621  BOOLEAN AvoidParent = FALSE;
622 
623  PAGED_CODE();
624 
625  //
626  // Update the requested field.
627  //
628  // Always handle the target entry.
629  //
630  // If we've not actually changed the bits,
631  // early out.
632  //
633 
634  switch (FieldToUpdate ) {
635  case UpdateRemaining:
636 
637  OldRemaining = ObjectTypeList[StartIndex].Remaining;
638  ObjectTypeList[StartIndex].Remaining = OldRemaining & ~AccessMask;
639 
640  if ( OldRemaining == ObjectTypeList[StartIndex].Remaining ) {
641  return;
642  }
643  break;
644 
645  case UpdateCurrentGranted:
646 
647  OldCurrentGranted = ObjectTypeList[StartIndex].CurrentGranted;
648  ObjectTypeList[StartIndex].CurrentGranted |=
649  AccessMask & ~ObjectTypeList[StartIndex].CurrentDenied;
650 
651  if ( OldCurrentGranted == ObjectTypeList[StartIndex].CurrentGranted ) {
652  //
653  // We can't simply return here.
654  // We have to visit our children. Consider the case where there
655  // was a previous deny ACE on a child. That deny would have
656  // propagated up the tree to this entry. However, this allow ACE
657  // needs to be added all of the children that haven't been
658  // explictly denied.
659  //
660  AvoidParent = TRUE;
661  }
662  break;
663 
664  case UpdateCurrentDenied:
665 
666  OldCurrentDenied = ObjectTypeList[StartIndex].CurrentDenied;
667  ObjectTypeList[StartIndex].CurrentDenied |=
668  AccessMask & ~ObjectTypeList[StartIndex].CurrentGranted;
669 
670  if ( OldCurrentDenied == ObjectTypeList[StartIndex].CurrentDenied ) {
671 
672  //
673  // We must visit the children - consider the case where we
674  // are applying an ACCESS_DENIED Ace to the entire tree.
675  // Suppose a node of the tree was already DENIED the same bits
676  // via an ACCESS_DENIED_OBJECT_ACE. Those deny bits would have
677  // propagated up the tree. If we are denying the bits to this
678  // node then we must visit the children.
679  //
680 
681  AvoidParent = TRUE;
682  }
683  break;
684 
685  default:
686  return;
687  }
688 
689 
690  //
691  // Go update parent of the target.
692  //
693 
694  if ( !AvoidParent ) {
695  SepUpdateParentTypeList( ObjectTypeList,
696  ObjectTypeListLength,
697  StartIndex );
698  }
699 
700  //
701  // Loop handling all children of the target.
702  //
703 
704  for ( Index=StartIndex+1; Index<ObjectTypeListLength; Index++ ) {
705 
706  //
707  // By definition, the children of an object are all those entries
708  // immediately following the target. The list of children (or
709  // grandchildren) stops as soon as we reach an entry the has the
710  // same level as the target (a sibling) or lower than the target
711  // (an uncle).
712  //
713 
714  if ( ObjectTypeList[Index].Level <= ObjectTypeList[StartIndex].Level ) {
715  break;
716  }
717 
718  //
719  // Grant access to the children
720  //
721 
722  switch (FieldToUpdate) {
723  case UpdateRemaining:
724 
725  ObjectTypeList[Index].Remaining &= ~AccessMask;
726  break;
727 
728  case UpdateCurrentGranted:
729 
730  ObjectTypeList[Index].CurrentGranted |=
731  AccessMask & ~ObjectTypeList[Index].CurrentDenied;
732  break;
733 
734  case UpdateCurrentDenied:
735 
736  ObjectTypeList[Index].CurrentDenied |=
737  AccessMask & ~ObjectTypeList[Index].CurrentGranted;
738  break;
739 
740  default:
741  return;
742  }
743  }
744 }

747 SepSidInToken (
748  IN PACCESS_TOKEN AToken,
749  IN PSID PrincipalSelfSid,
750  IN PSID Sid,
751  IN BOOLEAN DenyAce
752  )
753 
754 /*++
755 
756 Routine Description:
757 
758  Checks to see if a given SID is in the given token.
759 
760  N.B. The code to compute the length of a SID and test for equality
761  is duplicated from the security runtime since this is such a
762  frequently used routine.
763 
764 Arguments:
765 
766  Token - Pointer to the token to be examined
767 
768  PrincipalSelfSid - If the object being access checked is an object which
769  represents a principal (e.g., a user object), this parameter should
770  be the SID of the object. Any ACE containing the constant
771  PRINCIPAL_SELF_SID is replaced by this SID.
772 
773  The parameter should be NULL if the object does not represent a principal.
774 
775 
776  Sid - Pointer to the SID of interest
777 
778 Return Value:
779 
780  A value of TRUE indicates that the SID is in the token, FALSE
781  otherwise.
782 
783 --*/
784 
785 {
786 
787  ULONG i;
788  PISID MatchSid;
789  ULONG SidLength;
790  PTOKEN Token;
791  PSID_AND_ATTRIBUTES TokenSid;
792  ULONG UserAndGroupCount;
793  USHORT TargetShort;
794 
795  C_ASSERT (FIELD_OFFSET (SID, Revision) + sizeof (((SID *)Sid)->Revision) == FIELD_OFFSET (SID, SubAuthorityCount));
796  C_ASSERT (sizeof (((SID *)Sid)->Revision) + sizeof (((SID *)Sid)->SubAuthorityCount) == sizeof (USHORT));
797 
798 
799  PAGED_CODE();
800 
801 #if DBG
802 
803  SepDumpTokenInfo(AToken);
804 
805 #endif
806 
807  //
808  // If Sid is the constant PrincipalSelfSid,
809  // replace it with the passed in PrincipalSelfSid.
810  //
811 
812  if ( PrincipalSelfSid != NULL &&
813  RtlEqualSid( SePrincipalSelfSid, Sid ) ) {
814  Sid = PrincipalSelfSid;
815  }
816 
817  //
818  // Get the length of the source SID since this only needs to be computed
819  // once.
820  //
821  SidLength = 8 + (4 * ((PISID)Sid)->SubAuthorityCount);
822 
823  //
824  // To speed up processing we compare the sub authority count and the revision at the same time.
825  //
826  TargetShort = *(USHORT *)&((PISID)Sid)->Revision;
827 
828  //
829  // Get address of user/group array and number of user/groups.
830  //
831 
832  Token = (PTOKEN)AToken;
833  TokenSid = Token->UserAndGroups;
834  UserAndGroupCount = Token->UserAndGroupCount;
835 
836  //
837  // Scan through the user/groups and attempt to find a match with the
838  // specified SID.
839  //
840 
841  for (i = 0 ; i < UserAndGroupCount ; i += 1) {
842  MatchSid = (PISID)TokenSid->Sid;
843 
844  //
845  // If revision and sub authority count matches, then compare the SIDs
846  // for equality.
847  //
848 
849  if (*(USHORT *) &MatchSid->Revision == TargetShort) {
850  if (RtlEqualMemory(Sid, MatchSid, SidLength)) {
851 
852  //
853  // If this is the first one in the list, then it is the User,
854  // and return success immediately.
855  //
856  // If this is not the first one, then it represents a group,
857  // and we must make sure the group is currently enabled before
858  // we can say that the group is "in" the token.
859  //
860 
861  if ((i == 0) || (TokenSid->Attributes & SE_GROUP_ENABLED) ||
862  (DenyAce && (TokenSid->Attributes & SE_GROUP_USE_FOR_DENY_ONLY))) {
863  return TRUE;
864 
865  } else {
866  return FALSE;
867  }
868  }
869  }
870 
871  TokenSid += 1;
872  }
873 
874  return FALSE;
875 }
