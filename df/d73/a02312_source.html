480 
481 FORCEINLINE
482 PVOID
483 Wow64SetFilesystemRedirectorEx (
484  __in PVOID NewValue
485  )
486 /*++
487 
488 Routine Description:
489 
490  This routine allows a thread running inside Wow64 to disable file-system
491  redirection for all calls happening in the context of this thread.
492 
493 
494  NOTE: This routine should only called from a wow64 process, and is only available
495  when running on .NET server platforms and beyond. If you component will
496  run on downlevel platforms (XP 2600 for example), you shouldn't use WOW64_FILE_SYSTEM_DISABLE_REDIRECT (see below).
497 
498 Example (Enumerating files under c:\windows\system32):
499 
500  {
501  HANDLE File;
502  WIN32_FIND_DATA FindData;
503 #ifndef _WIN64
504  BOOL bWow64Process = FALSE;
505  PVOID Wow64RedirectionOld;
506 #endif
507 
508  //
509  // Disable Wow64 file system redirection
510  //
511 #ifndef _WIN64
512  IsWow64Process (GetCurrentProcess (), &bWow64Process);
513  if (bWow64Process == TRUE) {
514  Wow64RedirectionOld = Wow64SetFilesystemRedirectorEx (WOW64_FILE_SYSTEM_DISABLE_REDIRECT);
515  }
516 #endif
517  File = FindFirstFileA ("c:\\windows\\system32\\*.*", &FindData);
518 
519  do {
520  .
521  .
522  } while (FindNextFileA (File, &FindData) != 0);
523 
524  FindClose (File);
525 
526  //
527  // Enable Wow64 file-system redirection
528  //
529 #ifndef _WIN64
530  if (bWow64Process == TRUE) {
531  Wow64SetFilesystemRedirectorEx (Wow64RedirectionOld);
532  }
533 #endif
534  }
535 
536 
537 Arguments:
538 
539  NewValue - New Wow64 file-system redirection value. This can either be:
540  a- pointer to a unicode string with a fully-qualified path name (e.g. L"c:\\windows\\notepad.exe").
541  b- any of the following predefined values :
542  * WOW64_FILE_SYSTEM_ENABLE_REDIRECT : Enables file-system redirection (default)
543  * WOW64_FILE_SYSTEM_DISABLE_REDIRECT : Disables file-system redirection on all
544  file I/O operations happening within the context of the current thread.
545  * WOW64_FILE_SYSTEM_DISABLE_REDIRECT_LEGACY: Use this only if you want to run on
546  download level platforms (for example XP 2600), as it will have no effect
547  and prevents your program from malfunctioning.
548 
549 Return:
550 
551  Old Wow64 file-system redirection value
552 
553 --*/
554 {
555  PVOID OldValue;
556  OldValue = (PVOID)(ULONG_PTR)NtCurrentTeb64()->TlsSlots[WOW64_TLS_FILESYSREDIR];
557  NtCurrentTeb64()->TlsSlots[WOW64_TLS_FILESYSREDIR] = (ULONGLONG)PtrToUlong(NewValue);
558  return OldValue;
559 }

585 Wow64KiWrapApcProc (
586  __inout PVOID *ApcContext,
587  __inout PVOID *ApcRoutine)
588 
589 /*++
590 
591 Routine Description:
592 
593  This routine is used by kernel mode callers to queue APCs to a thread
594  running inside a Wow64 process. It wraps the original APC routine
595  with a jacket routine inside Wow64. The target Apc routine must be
596  inside 32-bit code. This routine must be executed in the context of the
597  Wow64 thread where the target APC is going to run in.
598 
599  This routine, as it accesses a TEB-relative offset, has to be executed
600  within a try/except block.
601 
602 Environment:
603 
604  Kernel mode only.
605 
606 Arguments:
607 
608  ApcContext - Pointer to the original ApcContext parameter.
609 
610  ApcRoutine - Pointer to the original ApcRoutine that is targeted to run
611  32-bit code.
612 
613 Return:
614 
615  None.
616 
617 --*/
618 {
619  *ApcContext = (PVOID)((ULONG_PTR)*ApcContext | ((ULONG_PTR)*ApcRoutine << 32));
620  *ApcRoutine = Wow64TlsGetValue (WOW64_TLS_APC_WRAPPER);
621 }
