311 KeFlushMultipleTb (
312  IN ULONG Number,
313  IN PVOID *Virtual,
314  IN BOOLEAN AllProcessors
315  )
316 
317 /*++
318 
319 Routine Description:
320 
321  This function flushes multiple entries from the translation buffer
322  on all processors that are currently running threads which are
323  children of the current process or flushes a multiple entries from
324  the translation buffer on all processors in the host configuration.
325 
326 Arguments:
327 
328  Number - Supplies the number of TB entries to flush.
329 
330  Virtual - Supplies a pointer to an array of virtual addresses that
331  are within the pages whose translation buffer entries are to be
332  flushed.
333 
334  AllProcessors - Supplies a boolean value that determines which
335  translation buffers are to be flushed.
336 
337 Return Value:
338 
339  The previous contents of the specified page table entry is returned
340  as the function value.
341 
342 --*/
343 
344 {
345 
346  PVOID *End;
347  KIRQL OldIrql;
348  PKPRCB Prcb;
349  PKPROCESS Process;
350  KAFFINITY TargetProcessors;
351 
352  ASSERT(Number != 0);
353 
354  End = Virtual + Number;
355 
356  //
357  // Compute target set of processors.
358  //
359 
360  OldIrql = KeRaiseIrqlToSynchLevel();
361  Prcb = KeGetCurrentPrcb();
362  if (AllProcessors != FALSE) {
363  TargetProcessors = KeActiveProcessors;
364 
365  } else {
366  Process = Prcb->CurrentThread->ApcState.Process;
367  TargetProcessors = Process->ActiveProcessors;
368  }
369 
370  TargetProcessors &= ~Prcb->SetMember;
371 
372  //
373  // If any target processors are specified, then send a flush multiple
374  // packet to the target set of processors.
375  //
376 
377  if (TargetProcessors != 0) {
378  KiIpiSendPacket(TargetProcessors,
379  KiFlushTargetMultipleTb,
380  NULL,
381  (PVOID)End,
382  (PVOID)Virtual);
383 
384  IPI_INSTRUMENT_COUNT (Prcb->Number, FlushMultipleTb);
385  }
386 
387  //
388  // Flush the specified entries from the TB on the current processor.
389  //
390 
391  do {
392  KiFlushSingleTb(*Virtual);
393  Virtual += 1;
394  } while (Virtual < End);
395 
396  //
397  // Wait until all target processors have finished and complete packet.
398  //
399 
400  if (TargetProcessors != 0) {
401  KiIpiStallOnPacketTargets(TargetProcessors);
402  }
403 
404  //
405  // Lower IRQL to its previous level.
406  //
407 
408  KeLowerIrql(OldIrql);
409  return;
410 }
