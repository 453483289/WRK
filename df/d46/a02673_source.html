13 Abstract:
14 
15  This module implements the absolute time conversion routines for NT.
16 
17  Absolute LARGE_INTEGER in NT is represented by a 64-bit large integer accurate
18  to 100ns resolution. The smallest time resolution used by this package
19  is One millisecond. The basis for NT time is the start of 1601 which
20  was chosen because it is the start of a new quadricentury. Some facts
21  to note are:
22 
23  o At 100ns resolution 32 bits is good for about 429 seconds (or 7 minutes)
24 
25  o At 100ns resolution a large integer (i.e., 63 bits) is good for
26  about 29,227 years, or around 10,675,199 days
27 
28  o At 1 second resolution 31 bits is good for about 68 years
29 
30  o At 1 second resolution 32 bits is good for about 136 years
31 
32  o 100ns Time (ignoring time less than a millisecond) can be expressed
33  as two values, Days and Milliseconds. Where Days is the number of
34  whole days and Milliseconds is the number of milliseconds for the
35  partial day. Both of these values are ULONG.
36 
37  Given these facts most of the conversions are done by first splitting
38  LARGE_INTEGER into Days and Milliseconds.
39 
40 --*/

228 ULONG
229 ElapsedDaysToYears (
230  IN ULONG ElapsedDays
231  )
232 
233 /*++
234 
235 Routine Description:
236 
237  This routine computes the number of total years contained in the indicated
238  number of elapsed days. The computation is to first compute the number of
239  400 years and subtract that it, then do the 100 years and subtract that out,
240  then do the number of 4 years and subtract that out. Then what we have left
241  is the number of days with in a normalized 4 year block. Normalized being that
242  the first three years are not leap years.
243 
244 Arguments:
245 
246  ElapsedDays - Supplies the number of days to use
247 
248 Return Value:
249 
250  ULONG - Returns the number of whole years contained within the input number
251  of days.
252 
253 --*/
254 
255 {
256  ULONG NumberOf400s;
257  ULONG NumberOf100s;
258  ULONG NumberOf4s;
259  ULONG Years;
260 
261  //
262  // A 400 year time block is 365*400 + 400/4 - 400/100 + 400/400 = 146097 days
263  // long. So we simply compute the number of whole 400 year block and the
264  // the number days contained in those whole blocks, and subtract if from the
265  // elapsed day total
266  //
267 
268  NumberOf400s = ElapsedDays / 146097;
269  ElapsedDays -= NumberOf400s * 146097;
270 
271  //
272  // A 100 year time block is 365*100 + 100/4 - 100/100 = 36524 days long.
273  // The computation for the number of 100 year blocks is biased by 3/4 days per
274  // 100 years to account for the extra leap day thrown in on the last year
275  // of each 400 year block.
276  //
277 
278  NumberOf100s = (ElapsedDays * 100 + 75) / 3652425;
279  ElapsedDays -= NumberOf100s * 36524;
280 
281  //
282  // A 4 year time block is 365*4 + 4/4 = 1461 days long.
283  //
284 
285  NumberOf4s = ElapsedDays / 1461;
286  ElapsedDays -= NumberOf4s * 1461;
287 
288  //
289  // Now the number of whole years is the number of 400 year blocks times 400,
290  // 100 year blocks time 100, 4 year blocks times 4, and the number of elapsed
291  // whole years, taking into account the 3/4 day per year needed to handle the
292  // leap year.
293  //
294 
295  Years = (NumberOf400s * 400) +
296  (NumberOf100s * 100) +
297  (NumberOf4s * 4) +
298  (ElapsedDays * 100 + 75) / 36525;
299 
300  return Years;
301 }

304 //
305 // ULONG
306 // NumberOfLeapYears (
307 // IN ULONG ElapsedYears
308 // );
309 //
310 // The number of leap years is simply the number of years divided by 4
311 // minus years divided by 100 plus years divided by 400. This says
312 // that every four years is a leap year except centuries, and the
313 // exception to the exception is the quadricenturies
314 //
315 
316 #define NumberOfLeapYears(YEARS) ( \
317  ((YEARS) / 4) - ((YEARS) / 100) + ((YEARS) / 400) \
318  )
319 
320 //
321 // ULONG
322 // ElapsedYearsToDays (
323 // IN ULONG ElapsedYears
324 // );
325 //
326 // The number of days contained in elapsed years is simply the number
327 // of years times 365 (because every year has at least 365 days) plus
328 // the number of leap years there are (i.e., the number of 366 days years)
329 //
330 
331 #define ElapsedYearsToDays(YEARS) ( \
332  ((YEARS) * 365) + NumberOfLeapYears(YEARS) \
333  )
334 
335 //
336 // BOOLEAN
337 // IsLeapYear (
338 // IN ULONG ElapsedYears
339 // );

341 // If it is an even 400 or a non century leapyear then the
342 // answer is true otherwise it's false
343 //
344

346  (((YEARS) % 400 == 0) || \
347  ((YEARS) % 100 != 0) && ((YEARS) % 4 == 0)) ? \

382 TimeToDaysAndFraction (
383  IN PLARGE_INTEGER Time,
384  OUT PULONG ElapsedDays,
385  OUT PULONG Milliseconds
386  )
387 
388 /*++
389 
390 Routine Description:
391 
392  This routine converts an input 64-bit time value to the number
393  of total elapsed days and the number of milliseconds in the
394  partial day.
395 
396 Arguments:
397 
398  Time - Supplies the input time to convert from
399 
400  ElapsedDays - Receives the number of elapsed days
401 
402  Milliseconds - Receives the number of milliseconds in the partial day
403 
404 Return Value:
405 
406  None
407 
408 --*/
409 
410 {
411  LARGE_INTEGER TotalMilliseconds;
412  LARGE_INTEGER Temp;
413 
414  //
415  // Convert the input time to total milliseconds
416  //
417 
418  TotalMilliseconds = Convert100nsToMilliseconds( *(PLARGE_INTEGER)Time );
419 
420  //
421  // Convert milliseconds to total days
422  //
423 
424  Temp = ConvertMillisecondsToDays( TotalMilliseconds );
425 
426  //
427  // Set the elapsed days from temp, we've divided it enough so that
428  // the high part must be zero.
429  //
430 
431  *ElapsedDays = Temp.LowPart;
432 
433  //
434  // Calculate the exact number of milliseconds in the elapsed days
435  // and subtract that from the total milliseconds to figure out
436  // the number of milliseconds left in the partial day
437  //
438 
439  Temp.QuadPart = ConvertDaysToMilliseconds( *ElapsedDays );
440 
441  Temp.QuadPart = TotalMilliseconds.QuadPart - Temp.QuadPart;
442 
443  //
444  // Set the fraction part from temp, the total number of milliseconds in
445  // a day guarantees that the high part must be zero.
446  //
447 
448  *Milliseconds = Temp.LowPart;
449 
450  //
451  // And return to our caller
452  //
453 
454  return;
455 }

470 /*++
471 
472 Routine Description:
473 
474  This routine converts an input elapsed day count and partial time
475  in milliseconds to a 64-bit time value.
476 
477 Arguments:
478 
479  ElapsedDays - Supplies the number of elapsed days
480 
481  Milliseconds - Supplies the number of milliseconds in the partial day
482 
483  Time - Receives the output time to value

487  None
488 
489 --*/

491 {
492  LARGE_INTEGER Temp;
493  LARGE_INTEGER Temp2;
494 
495  //
496  // Calculate the exact number of milliseconds in the elapsed days.
497  //
498 
499  Temp.QuadPart = ConvertDaysToMilliseconds( ElapsedDays );
500 
501  //
502  // Convert milliseconds to a large integer
503  //
504 
505  Temp2.LowPart = Milliseconds;
506  Temp2.HighPart = 0;
507 
508  //
509  // add milliseconds to the whole day milliseconds
510  //
511 
512  Temp.QuadPart = Temp.QuadPart + Temp2.QuadPart;
513 
514  //
515  // Finally convert the milliseconds to 100ns resolution
516  //
517 
518  *(PLARGE_INTEGER)Time = ConvertMillisecondsTo100ns( Temp );
519 
520  //
521  // and return to our caller
522  //

528 VOID
529 RtlTimeToTimeFields (
530  IN PLARGE_INTEGER Time,
531  OUT PTIME_FIELDS TimeFields
532  )
533 
534 /*++
535 
536 Routine Description:
537 
538  This routine converts an input 64-bit LARGE_INTEGER variable to its corresponding
539  time field record. It will tell the caller the year, month, day, hour,
540  minute, second, millisecond, and weekday corresponding to the input time
541  variable.
542 
543 Arguments:
544 
545  Time - Supplies the time value to interpret
546 
547  TimeFields - Receives a value corresponding to Time
548 
549 Return Value:
550 
551  None
552 
553 --*/
554 
555 {
556  ULONG Years;
557  ULONG Month;
558  ULONG Days;
559 
560  ULONG Hours;
561  ULONG Minutes;
562  ULONG Seconds;
563  ULONG Milliseconds;
564 
565  //
566  // First divide the input time 64 bit time variable into
567  // the number of whole days and part days (in milliseconds)
568  //
569 
570  TimeToDaysAndFraction( Time, &Days, &Milliseconds );
571 
572  //
573  // Compute which weekday it is and save it away now in the output
574  // variable. We add the weekday of the base day to bias our computation
575  // which means that if one day has elapsed then we the weekday we want
576  // is the Jan 2nd, 1601.
577  //
578 
579  TimeFields->Weekday = (CSHORT)((Days + WEEKDAY_OF_1601) % 7);
580 
581  //
582  // Calculate the number of whole years contained in the elapsed days
583  // For example if Days = 500 then Years = 1
584  //
585 
586  Years = ElapsedDaysToYears( Days );
587 
588  //
589  // And subtract the number of whole years from our elapsed days
590  // For example if Days = 500, Years = 1, and the new days is equal
591  // to 500 - 365 (normal year).
592  //
593 
594  Days = Days - ElapsedYearsToDays( Years );
595 
596  //
597  // Now test whether the year we are working on (i.e., The year
598  // after the total number of elapsed years) is a leap year
599  // or not.
600  //
601 
602  if (IsLeapYear( Years + 1 )) {
603 
604  //
605  // The current year is a leap year, so figure out what month
606  // it is, and then subtract the number of days preceding the
607  // month from the days to figure out what day of the month it is
608  //
609 
610  Month = LeapYearDayToMonth[Days];
611  Days = Days - LeapYearDaysPrecedingMonth[Month];
612 
613  } else {
614 
615  //
616  // The current year is a normal year, so figure out the month
617  // and days as described above for the leap year case
618  //
619 
620  Month = NormalYearDayToMonth[Days];
621  Days = Days - NormalYearDaysPrecedingMonth[Month];
622 
623  }
624 
625  //
626  // Now we need to compute the elapsed hour, minute, second, milliseconds
627  // from the millisecond variable. This variable currently contains
628  // the number of milliseconds in our input time variable that did not
629  // fit into a whole day. To compute the hour, minute, second part
630  // we will actually do the arithmetic backwards computing milliseconds
631  // seconds, minutes, and then hours. We start by computing the
632  // number of whole seconds left in the day, and then computing
633  // the millisecond remainder.
634  //
635 
636  Seconds = Milliseconds / 1000;
637  Milliseconds = Milliseconds % 1000;
638 
639  //
640  // Now we compute the number of whole minutes left in the day
641  // and the number of remainder seconds
642  //
643 
644  Minutes = Seconds / 60;
645  Seconds = Seconds % 60;
646 
647  //
648  // Now compute the number of whole hours left in the day
649  // and the number of remainder minutes
650  //
651 
652  Hours = Minutes / 60;
653  Minutes = Minutes % 60;
654 
655  //
656  // As our final step we put everything into the time fields
657  // output variable
658  //
659 
660  TimeFields->Year = (CSHORT)(Years + 1601);
661  TimeFields->Month = (CSHORT)(Month + 1);
662  TimeFields->Day = (CSHORT)(Days + 1);
663  TimeFields->Hour = (CSHORT)Hours;
