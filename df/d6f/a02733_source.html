79  //
80  VERIFIER_OPTION_FORCE_PENDING,

87  VERIFIER_OPTION_ROTATE_STATUS,

89  //
90  // If this option is enabled, the verifier will undo the effects of

92  // (Exempting the case where an IRP was forwarded to another stack)
93  //
94  VERIFIER_OPTION_CONSUME_ALWAYS,
95 
96  //
97  // If this option is enabled, the verifier will update SRB's to handle
98  // surrogate IRPs. Some SCSI IRPs can't be surrogated unless the
99  // SRB->OriginalRequest pointer is updated. This is due to a busted SRB
100  // architecture. Note that the technique used to identify an SRB IRP is
101  // "fuzzy", and could in theory touch an IRP it shouldn't have!
102  //
103  VERIFIER_OPTION_SMASH_SRBS,
104 
105  //
106  // If this option is enabled, the verifier will replace original IRPs with
107  // surrogates when traveling down the stack. The surrogates are allocated
108  // from special pool, and get freed immediately upon completion. This lets
109  // the verifier catch drivers that touch IRPs after they're completed.
110  //
111  VERIFIER_OPTION_SURROGATE_IRPS,
112 
113  //
114  // If this option is enabled, the verifier buffers all direct I/O. It does
115  // this by allocating an alternate MDL and copying the MDL contents back
116  // to user mode only after IRP completion. This allows overruns, underruns,
117  // and late accesses to be detected.
118  //
119  VERIFIER_OPTION_BUFFER_DIRECT_IO,
120 
121  //
122  // If this option is enabled, the verifier delays completion of all IRPs
123  // via timer. VERIFIER_OPTION_FORCE_PENDING is set by inference.
124  //
125  VERIFIER_OPTION_DEFER_COMPLETION,
126 
127  //
128  // If this option is enabled, the verifier completes every IRP at

132  VERIFIER_OPTION_COMPLETE_AT_PASSIVE,

157  //

164  //
165  VERIFIER_OPTION_VERIFY_DO_FLAGS,

175  //
176  VERIFIER_OPTION_DETECT_DEADLOCKS,

180  // validated.
181  //
182  VERIFIER_OPTION_VERIFY_DMA,

184  //
185  // This option double buffers all dma and erects guard pages on each side
186  // of all common buffers and mapped buffers. Is memory-intensive but can

188  // buffers.
189  //
190  VERIFIER_OPTION_DOUBLE_BUFFER_DMA,

202  //
