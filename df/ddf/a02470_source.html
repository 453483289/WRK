161 /*++
162 
163 Routine Description:
164 
165  This function cancels a timer that was previously set to expire at
166  a specified time. If the timer is not currently set, then no operation
167  is performed. Canceling a timer does not set the state of the timer to
168  Signaled.
169 
170 Arguments:
171 
172  Timer - Supplies a pointer to a dispatcher object of type timer.
173 
174 Return Value:
175 
176  A boolean value of TRUE is returned if the the specified timer was
177  currently set. Else a value of FALSE is returned.
178 
179 --*/
180 
181 {
182 
183  BOOLEAN Inserted;
184  KIRQL OldIrql;
185 
186  ASSERT_TIMER(Timer);
187  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
188 
189  //
190  // Raise IRQL to dispatcher level, lock the dispatcher database, and
191  // capture the timer inserted status. If the timer is currently set,
192  // then remove it from the timer list.
193  //
194 
195  KiLockDispatcherDatabase(&OldIrql);
196  Inserted = Timer->Header.Inserted;
197  if (Inserted != FALSE) {
198  KiRemoveTreeTimer(Timer);
199  }
200 
201  //
202  // Unlock the dispatcher database, lower IRQL to its previous value, and
203  // return boolean value that signifies whether the timer was set of not.
204  //
205 
206  KiUnlockDispatcherDatabase(OldIrql);
207  return Inserted;
208 }

292 /*++
293 
294 Routine Description:
295 
296  This function sets a timer to expire at a specified time. If the timer is
297  already set, then it is implicitly canceled before it is set to expire at
298  the specified time. Setting a timer causes its due time to be computed,
299  its state to be set to Not-Signaled, and the timer object itself to be
300  inserted in the timer list.
301 
302 Arguments:
303 
304  Timer - Supplies a pointer to a dispatcher object of type timer.
305 
306  DueTime - Supplies an absolute or relative time at which the timer
307  is to expire.
308 
309  Period - Supplies an optional period for the timer in milliseconds.
310 
311  Dpc - Supplies an optional pointer to a control object of type DPC.
312 
313 Return Value:
314 
315  A boolean value of TRUE is returned if the the specified timer was
316  currently set. Otherwise, a value of FALSE is returned.
317 
318 --*/
319 
320 {
321 
322  ULONG Hand;
323  BOOLEAN Inserted;
324  KIRQL OldIrql;
325  BOOLEAN RequestInterrupt;
326 
327  ASSERT_TIMER(Timer);
328 
329  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
330 
331  //
332  // Raise IRQL to dispatcher level and lock dispatcher database.
333  //
334  // Capture the timer inserted status and if the timer is currently
335  // set, then remove it from the timer list.
336  //
337 
338  KiLockDispatcherDatabase(&OldIrql);
339  Inserted = Timer->Header.Inserted;
340  if (Inserted != FALSE) {
341  KiRemoveTreeTimer(Timer);
342  }
343 
344  //
345  // Set the DPC address, set the period, and compute the timer due time.
346  // If the timer has already expired, then signal the timer. Otherwise,
347  // set the signal state to false and attempt to insert the timer in the
348  // timer table.
349  //
350  // N.B. The signal state must be cleared before it is inserted in the
351  // timer table in case the period is not zero.
352  //
353 
354  Timer->Dpc = Dpc;
355  Timer->Period = Period;
356  if (KiComputeDueTime(Timer, DueTime, &Hand) == FALSE) {
357  RequestInterrupt = KiSignalTimer(Timer);
358  KiUnlockDispatcherDatabaseFromSynchLevel();
359  if (RequestInterrupt == TRUE) {
360  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
361  }
362 
363  } else {
364  Timer->Header.SignalState = FALSE;
365  KiInsertOrSignalTimer(Timer, Hand);
366  }
367 
368  KiExitDispatcher(OldIrql);
369 
370  //
371  // Return boolean value that signifies whether the timer was set or
372  // not.
373  //
374 
375  return Inserted;
376 }

434 KeCheckForTimer (
435  __in_bcount(BlockSize) PVOID BlockStart,
436  __in SIZE_T BlockSize
437  )

439 /*++
440 
441 Routine Description:
442 
443  This function checks to detemine if the specified block of memory
444  overlaps the range of an active timer.
445 
446 Arguments:
447 
448  BlockStart - Supplies a pointer to the block of memory to check.
449 
450  BlockSize - Supplies the size, in bytes, of the block of memory to check.
451 
452 Return Value:
453 
454  None.
455 
456 --*/
457 
458 {
459 
460  PUCHAR Address;
461  PUCHAR End;
462  ULONG Index;
463  PLIST_ENTRY ListHead;
464  PKSPIN_LOCK_QUEUE LockQueue;
465  PLIST_ENTRY NextEntry;
466  KIRQL OldIrql;
467  PKTIMER Timer;
468  PUCHAR Start;
469 
470  //
471  // Make sure timer checks are enabled before proceeding.
472  //
473 
474  if ((KeTimerCheckFlags & KE_TIMER_CHECK_FREES) == 0) {
475  return;
476  }
477 
478  //
479  // Compute the ending memory location.
480  //
481 
482  Start = (PUCHAR)BlockStart;
483  End = Start + BlockSize;
484 
485  //
486  // Raise IRQL to dispatcher level and lock dispatcher database.
487  //
488  // Scan the timer database and check for any timers in the specified
489  // memory block.
490  //
491 
492  KiLockDispatcherDatabase(&OldIrql);
493  Index = 0;
494  do {
495  ListHead = &KiTimerTableListHead[Index].Entry;
496  LockQueue = KiAcquireTimerTableLock(Index);
497  NextEntry = ListHead->Flink;
498  while (NextEntry != ListHead) {
499  Timer = CONTAINING_RECORD(NextEntry, KTIMER, TimerListEntry);
500  Address = (PUCHAR)Timer;
501  NextEntry = NextEntry->Flink;
502 
503  //
504  // Check that the timer object is not in the range.
505  //
506 
507  if ((Address > (Start - sizeof(KTIMER))) &&
508  (Address < End)) {
509  KeBugCheckEx(TIMER_OR_DPC_INVALID,
510  0x0,
511  (ULONG_PTR)Address,
512  (ULONG_PTR)Start,
513  (ULONG_PTR)End);
514  }
515 
516  if (Timer->Dpc) {
517 
518  //
519  // Check that the timer DPC object is not in the range.
520  //
521 
522  Address = (PUCHAR)Timer->Dpc;
523  if ((Address > (Start - sizeof(KDPC))) &&
524  (Address < End)) {
525  KeBugCheckEx(TIMER_OR_DPC_INVALID,
526  0x1,
527  (ULONG_PTR)Address,
528  (ULONG_PTR)Start,
529  (ULONG_PTR)End);
530  }
531 
532  //
533  // Check that the timer DPC routine is not in the range.
534  //
535 
536  Address = (PUCHAR)(ULONG_PTR) Timer->Dpc->DeferredRoutine;
537  if (Address >= Start && Address < End) {
538  KeBugCheckEx(TIMER_OR_DPC_INVALID,
539  0x2,
540  (ULONG_PTR)Address,
541  (ULONG_PTR)Start,
542  (ULONG_PTR)End);
543  }
544  }
545  }
546 
547  KiReleaseTimerTableLock(LockQueue);
548  Index += 1;
549  } while(Index < TIMER_TABLE_SIZE);
550 
551  //
552  // Unlock the dispatcher database and lower IRQL to its previous value
553  //
554 
555  KiUnlockDispatcherDatabase(OldIrql);
556 }
