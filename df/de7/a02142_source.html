40 
41 Arguments:
42 
43  TargetDevice - Supplies the device to start balanced read from.
44 
45 Return Value:
46

48  STATUS_INVALID_DEVICE_REQUEST if the volume is not a mirror.
49 
50 --*/
51 
52 {
53  PIRP Irp;
54  KEVENT Event;
55  IO_STATUS_BLOCK Iosb;
56  NTSTATUS Status;
57 
58  KeInitializeEvent( &Event, NotificationEvent, FALSE );
59 
60  Irp = IoBuildDeviceIoControlRequest( FT_BALANCED_READ_MODE,
61  TargetDevice,
62  NULL,
63  0,
64  NULL,
65  0,
66  FALSE,
67  &Event,
68  &Iosb );
69 
70  if ( Irp == NULL ) {
71
