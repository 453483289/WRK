54 /*++
55 
56 Routine Description:
57 
58  This function maps a view in the specified subject process to
59  the section object. The page protection is identical to that
60  of the prototype PTE.
61 
62  This function is a kernel mode interface to allow LPC to map
63  a section given the section pointer to map.
64 
65  This routine assumes all arguments have been probed and captured.
66 
67 Arguments:
68 
69  SectionToMap - Supplies a pointer to the section object.
70 
71  BaseAddress - Supplies a pointer to a variable that will receive
72  the base address of the view. If the initial value
73  of this argument is not null, then the view will
74  be allocated starting at the specified virtual
75  address rounded down to the next 64kb address
76  boundary. If the initial value of this argument is
77  null, then the operating system will determine
78  where to allocate the view using the information
79  specified by the ZeroBits argument value and the
80  section allocation attributes (i.e. based and tiled).
81 
82  SectionOffset - Supplies the offset from the beginning of the section
83  to the view in bytes. This value must be a multiple
84  of 256k.
85 
86  ViewSize - Supplies a pointer to a variable that will receive
87  the actual size in bytes of the view. The initial
88  values of this argument specifies the size of the view
89  in bytes and is rounded up to the next host page size
90  boundary and must be less than or equal to 256k.
91 
92 Return Value:
93 
94  NTSTATUS.
95 
96 Environment:
97 
98  Kernel mode, APC_LEVEL or below.
99 
100 --*/

255  //

262  // This routine releases the PFN lock.
263  //
264

266

268  }
269  }
270  else {
271  UNLOCK_PFN (OldIrql);
272  }
273

275  KeBugCheckEx (MEMORY_MANAGEMENT,
276  0x778,
277  (ULONG_PTR)PointerPte,
278  0,
279  0);
280  }
281 
282  //
283  // Check to see if the TB needs to be flushed. Note that due to natural
284  // TB traffic and the number of system cache views, this is an extremely
285  // rare operation.
286  //
287 
288  if (MiCompareTbFlushTimeStamp ((ULONG)(PointerPte + 1)->u.List.NextEntry, MM_FLUSH_COUNTER_MASK)) {
289  MI_FLUSH_ENTIRE_TB (7);
290  }
291 
292  //
293  // Zero this explicitly now since the number of pages may be only 1.
294  //
295 
296  (PointerPte + 1)->u.List.NextEntry = 0;
297 
298  *CapturedBase = MiGetVirtualAddressMappedByPte (PointerPte);
299 
300  ProtoPte = &Subsection->SubsectionBase[PteOffset];
301 
302  LastProto = &Subsection->SubsectionBase[Subsection->PtesInSubsection];
303 
304  LastPte = PointerPte + NumberOfPages;
305 
306 #if DBG
307 
308  for (PointerPte2 = PointerPte + 2; PointerPte2 < LastPte; PointerPte2 += 1) {
309  ASSERT (PointerPte2->u.Long == 0);
310  }
311 
312 #endif
313 
314  while (PointerPte < LastPte) {
315 
316  if (ProtoPte >= LastProto) {
317 
318  //
319  // Handle extended subsections.
320  //
321 
322  Subsection = Subsection->NextSubsection;
323  ProtoPte = Subsection->SubsectionBase;
324  LastProto = &Subsection->SubsectionBase[
325  Subsection->PtesInSubsection];
326  }
327  PteContents.u.Long = MiProtoAddressForKernelPte (ProtoPte);
328  MI_WRITE_INVALID_PTE (PointerPte, PteContents);
329 
330  ASSERT (((ULONG_PTR)PointerPte & (MM_COLOR_MASK << PTE_SHIFT)) ==
331  (((ULONG_PTR)ProtoPte & (MM_COLOR_MASK << PTE_SHIFT))));
332 
333  PointerPte += 1;
334  ProtoPte += 1;
335  }
336 
337  return STATUS_SUCCESS;
338 }

456  IN PVOID BaseAddress,
457  IN PVOID SectionToUnmap,

461 /*++
462 
463 Routine Description:
464 
465  This function unmaps a view from the system cache.
466 
467  NOTE: When this function is called, no pages may be locked in
468  the cache for the specified view.
469 
470 Arguments:
471 
472  BaseAddress - Supplies the base address of the section in the
473  system cache.
474 
475  SectionToUnmap - Supplies a pointer to the section which the
476  base address maps.
477 
478  AddToFront - Supplies TRUE if the unmapped pages should be
479  added to the front of the standby list (i.e., their
480  value in the cache is low). FALSE otherwise.
481 
482 Return Value:
483 
484  None.
485 
486 Environment:
487 
488  Kernel mode.
489 
490 --*/
491 
492 {
493  LONG i;
494  LONG j;
495  PMMPTE PointerPte;
496  PMMPTE PointerPde;
497  PMMPTE LastPte;
498  PMMPFN Pfn1;
499  PMMPFN Pfn2;
500  PMMPTE FirstPte;
501  PMMPTE ProtoPte;
502  MMPTE PteContents;
503  MMPTE ProtoPteContents;
504  KIRQL OldIrql;
505  PCONTROL_AREA ControlArea;
506  PFILE_OBJECT FilePointer;
507  ULONG WsHeld;
508  PFN_NUMBER PageFrameIndex;
509  PFN_NUMBER PageTableFrameIndex;
510  PMSUBSECTION Subsection;
511  PETHREAD CurrentThread;
512  MMPTE PteArray[X256K / PAGE_SIZE];
513 
514  ASSERT (KeGetCurrentIrql() <= APC_LEVEL);
515 
516  WsHeld = FALSE;
517 
518  CurrentThread = PsGetCurrentThread ();
519 
520  PointerPte = MiGetPteAddress (BaseAddress);
521  LastPte = PointerPte + (X256K / PAGE_SIZE);
522 
523  FirstPte = PointerPte;
524  PageTableFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (MiGetPteAddress (PointerPte));
525  Pfn2 = MI_PFN_ELEMENT (PageTableFrameIndex);
526 
527  //
528  // Get the control area for the segment which is mapped here.
529  //
530 
531  ControlArea = ((PSECTION)SectionToUnmap)->Segment->ControlArea;
532  FilePointer = ControlArea->FilePointer;
533  Subsection = NULL;
534  ProtoPte = NULL;
535  i = 0;
536  j = 0;
537 
538  ASSERT ((ControlArea->u.Flags.Image == 0) &&
539  (ControlArea->u.Flags.PhysicalMemory == 0) &&
540  (ControlArea->u.Flags.GlobalOnlyPerSession == 0));
541 
542 #if defined(_X86PAE_)
543 
544  //
545  // PAE PTEs are written in 2 separate writes so the working set pushlock
546  // must always be held even to examine PTEs in prototype format - because
547  // a trim could be happening in parallel (writing the low half and then
548  // the upper half which would cause otherwise cause us to read the
549  // PTE contents split in the middle).
550  //
551 
552  WsHeld = TRUE;
553  LOCK_SYSTEM_WS (CurrentThread);
554 
555 #endif
556 
557  do {
558 
559  //
560  // The cache is organized in chunks of 256k bytes, clear
561  // the first chunk then check to see if this is the last chunk.
562  //
563  // The page table page is always resident for the system cache.
564  // Check each PTE: it is in one of three states, either valid or
565  // prototype PTE format or zero.
566  //
567 
568  PteContents = *PointerPte;
569  PteArray[i].u.Long = PteContents.u.Long;
570 
571  if (PteContents.u.Hard.Valid == 1) {
572 
573  //
574  // The PTE is valid.
575  //
576 
577  if (!WsHeld) {
578  WsHeld = TRUE;
579  LOCK_SYSTEM_WS (CurrentThread);
580  continue;
581  }
582 
583  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (&PteContents);
584  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
585 
586  //
587  // Remove the working set list entry.
588  //
589 
590  MiTerminateWsle (BaseAddress, &MmSystemCacheWs, Pfn1->u1.WsIndex);
591 
592  //
593  // Decrement the view count for every subsection this view spans.
594  // But make sure it's only done once per subsection in a given view.
595  //
596  // The subsections can only be decremented after all the
597  // PTEs have been cleared and PFN sharecounts decremented so no
598  // prototype PTEs will be valid if it is indeed the final subsection
599  // dereference. This is critical so the dereference segment
600  // thread doesn't free pool containing valid prototype PTEs.
601  //
602 
603  if (FilePointer != NULL) {
604 
605  ASSERT (Pfn1->u3.e1.PrototypePte);
606  ASSERT (Pfn1->OriginalPte.u.Soft.Prototype);
607 
608  ProtoPte = Pfn1->PteAddress;
609 
610  if (Subsection == NULL) {
611  Subsection = (PMSUBSECTION) MiGetSubsectionAddress (&Pfn1->OriginalPte);
612  }
613  }
614 
615  //
616  // The PFN sharecount will be decremented below
617  // when the PFN lock is held.
618  //
619  }
620  else if (PteContents.u.Soft.Prototype == 1) {
621 
622  //
623  // Decrement the view count for every subsection this view
624  // spans. But make sure it's only done once per subsection
625  // in a given view.
626  //
627 
628  if (FilePointer != NULL) {
629  ProtoPte = MiPteToProto (&PteContents);
630  if (Subsection == NULL) {
631 
632  //
633  // The prototype PTE may be paged out - but the contents
634  // must be accessed to reconstruct the subsection pointer.
635  //
636  // Acquire the PFN lock to prevent the prototype PTE
637  // from changing as we will potentially reference through
638  // it to an actual PFN, etc.
639  //
640 
641  PointerPde = MiGetPteAddress (ProtoPte);
642 
643  LOCK_PFN (OldIrql);
644 
645  if (PointerPde->u.Hard.Valid == 0) {
646 
647  if (WsHeld) {
648  MiMakeSystemAddressValidPfnSystemWs (ProtoPte,
649  OldIrql);
650  }
651  else {
652  MiMakeSystemAddressValidPfn (ProtoPte, OldIrql);
653  }
654 
655  //
656  // Ok to march on even though locks were released and
657  // reacquired.
658  //
659  }
660 
661  ProtoPteContents = *ProtoPte;
662 
663  if (ProtoPteContents.u.Hard.Valid == 1) {
664  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (&ProtoPteContents);
665  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
666  ProtoPte = &Pfn1->OriginalPte;
667  }
668  else if ((ProtoPteContents.u.Soft.Transition == 1) &&
669  (ProtoPteContents.u.Soft.Prototype == 0)) {
670  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_TRANSITION_PTE (&ProtoPteContents);
671  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
672  ProtoPte = &Pfn1->OriginalPte;
673  }
674  else {
675  ASSERT (ProtoPteContents.u.Soft.Prototype == 1);
676  }
677 
678  Subsection = (PMSUBSECTION) MiGetSubsectionAddress (ProtoPte);
679  UNLOCK_PFN (OldIrql);
680  }
681  }
682  }
683  else {
684  ASSERT (PteContents.u.Long == 0);
685 
686  //
687  // All the remaining PTEs in the view must also be zero.
688  //
689 
690 #if DBG
691  ASSERT (RtlCompareMemoryUlong (PointerPte, (LastPte - PointerPte) * sizeof (MMPTE), 0) == (LastPte - PointerPte) * sizeof (MMPTE));
692 #endif
693  break;
694  }
695 
696  MI_WRITE_ZERO_PTE (PointerPte);
697 
698  i += 1;
699  PointerPte += 1;
700  BaseAddress = (PVOID)((PCHAR)BaseAddress + PAGE_SIZE);
701 
702  } while (PointerPte < LastPte);
703 
704  if (WsHeld) {
705  UNLOCK_SYSTEM_WS (CurrentThread);
706  }
707 
708  FirstPte->u.List.NextEntry = MM_EMPTY_PTE_LIST;
709 
710  (FirstPte+1)->u.List.NextEntry = (KeReadTbFlushTimeStamp() & MM_FLUSH_COUNTER_MASK);
711 
712  if ((Subsection != NULL) && (Subsection->ControlArea != ControlArea)) {
713  KeBugCheckEx (MEMORY_MANAGEMENT,
714  0x782,
715  (ULONG_PTR) &PteArray,
716  (ULONG_PTR) BaseAddress,
717  (ULONG_PTR) SectionToUnmap);
718  }
719 
720  LOCK_PFN (OldIrql);
721 
722  //
723  // Free this entry to the end of the list.
724  //
725 
726  MmLastFreeSystemCache->u.List.NextEntry = FirstPte - MmSystemCachePteBase;
727  MmLastFreeSystemCache = FirstPte;
728 
729  //
730  // Decrement the share counts for any valid PFNs that were found.
731  //
732 
733  MmFrontOfList = AddToFront;
734 
735  for ( ; j < i; j += 1) {
736 
737  if (PteArray[j].u.Hard.Valid == 0) {
738  continue;
739  }
740 
741  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (&PteArray[j]);
742  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
743 
744  //
745  // Capture the state of the modified bit for this PTE.
746  //
747 
748  MI_CAPTURE_DIRTY_BIT_TO_PFN (&PteArray[j], Pfn1);
749 
750  //
751  // Decrement the share and valid counts of the page table
752  // page which maps this PTE.
753  //
754 
755  MiDecrementShareCountInline (Pfn2, PageTableFrameIndex);
756 
757  //
758  // Decrement the share count for the physical page.
759  //
760 
761 #if DBG
762  if (ControlArea->NumberOfMappedViews == 1) {
763  ASSERT (Pfn1->u2.ShareCount == 1);
764  }
765 #endif
766 
767  MiDecrementShareCountInline (Pfn1, PageFrameIndex);
768  }
769 
770  MmFrontOfList = FALSE;
771 
772  while (Subsection != NULL) {
773 
774  ASSERT (ProtoPte != NULL);
775 
776  ASSERT ((Subsection->NumberOfMappedViews >= 1) ||
777  (Subsection->u.SubsectionFlags.SubsectionStatic == 1));
778 
779  MiRemoveViewsFromSection (Subsection,
780  Subsection->PtesInSubsection);
781 
782  if ((ProtoPte >= Subsection->SubsectionBase) &&
783  (ProtoPte < Subsection->SubsectionBase + Subsection->PtesInSubsection)) {
784 
785  break;
786  }
787 
788  Subsection = (PMSUBSECTION) Subsection->NextSubsection;
789 
790  //
791  // The only way the subsection could be NULL is if we overshoot
792  // the subsection list due to a incorrect ending prototype computation.
793  // This should only occur is someone has corrupted the PTEs.
794  //
