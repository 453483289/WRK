472  }

475  }

482  }
483

493  //
494

496  RtlCopyMemory (Buffer, MemInfo, (ULONG_PTR)String - (ULONG_PTR)MemInfo);

498  status = GetExceptionCode();
499  }
500 
501  ExFreePool (MemInfo);
502

600 /*++
601 
602 Routine Description:
603 
604  This routine logs the PFN numbers of all ActiveAndValid pages.
605 
606 Arguments:
607 
608  None.
609 
610 Return Value:
611 
612  NTSTATUS.
613 
614 Environment:
615 
616  Kernel mode. PASSIVE level. No locks held.
617 
618 --*/
619 
620 {
621  ULONG i;
622  PKTHREAD Thread;
623  PFN_NUMBER StartPage;
624  PFN_NUMBER EndPage;
625  ULONG_PTR MemSnapLocal[(sizeof(MMPFN_MEMSNAP_INFORMATION)/sizeof(ULONG_PTR)) + (MI_MAXIMUM_PFNID_RUN / (8*sizeof(ULONG_PTR))) ];
626  PMMPFN_MEMSNAP_INFORMATION MemSnap;
627  PMMPFN Pfn1;
628  PMMPFN FirstPfn;
629  PMMPFN LastPfn;
630  PMMPFN MaxPfn;
631  PMMPFN InitialPfn;
632  RTL_BITMAP BitMap;
633  PULONG ActualBits;
634 
635  ASSERT (KeGetCurrentIrql() == PASSIVE_LEVEL);
636 
637  ASSERT ((MI_MAXIMUM_PFNID_RUN % (8 * sizeof(ULONG_PTR))) == 0);
638 
639  MemSnap = (PMMPFN_MEMSNAP_INFORMATION) MemSnapLocal;
640 
641  ActualBits = (PULONG)(MemSnap + 1);
642 
643  RtlInitializeBitMap (&BitMap, ActualBits, MI_MAXIMUM_PFNID_RUN);
644 
645  MemSnap->Count = 0;
646  RtlClearAllBits (&BitMap);
647 
648  Thread = KeGetCurrentThread ();
649 
650  KeEnterGuardedRegionThread (Thread);
651 
652  MI_LOCK_DYNAMIC_MEMORY_SHARED ();
653 
654  for (i = 0; i < MmPhysicalMemoryBlock->NumberOfRuns; i += 1) {
655 
656  StartPage = MmPhysicalMemoryBlock->Run[i].BasePage;
657  EndPage = StartPage + MmPhysicalMemoryBlock->Run[i].PageCount;
658  FirstPfn = MI_PFN_ELEMENT (StartPage);
659  LastPfn = MI_PFN_ELEMENT (EndPage);
660 
661  //
662  // Find the first valid PFN and start the run there.
663  //
664 
665  for (Pfn1 = FirstPfn; Pfn1 < LastPfn; Pfn1 += 1) {
666  if (Pfn1->u3.e1.PageLocation == ActiveAndValid) {
667  break;
668  }
669  }
670 
671  if (Pfn1 == LastPfn) {
672 
673  //
674  // No valid PFNs in this block, move on to the next block.
675  //
676 
677  continue;
678  }
679 
680  MaxPfn = LastPfn;
681  InitialPfn = NULL;
682 
683  do {
684  if (Pfn1->u3.e1.PageLocation == ActiveAndValid) {
685  if (InitialPfn == NULL) {
686  MemSnap->InitialPageFrameIndex = MI_PFN_ELEMENT_TO_INDEX (Pfn1);
687  InitialPfn = Pfn1;
688  MaxPfn = InitialPfn + MI_MAXIMUM_PFNID_RUN;
689  }
690  RtlSetBit (&BitMap, (ULONG) (Pfn1 - InitialPfn));
691  }
692 
693  Pfn1 += 1;
694 
695  if ((Pfn1 >= MaxPfn) && (InitialPfn != NULL)) {
696 
697  //
698  // Log the bitmap as we're at then end of it.
699  //
700 
701  ASSERT ((Pfn1 - InitialPfn) == MI_MAXIMUM_PFNID_RUN);
702  MemSnap->Count = MI_MAXIMUM_PFNID_RUN;
703  PerfInfoLogBytes (PERFINFO_LOG_TYPE_MEMORYSNAPLITE,
704  MemSnap,
705  sizeof(MemSnapLocal));
706 
707  InitialPfn = NULL;
708  MaxPfn = LastPfn;
709  RtlClearAllBits (&BitMap);
710  }
711  } while (Pfn1 < LastPfn);
712 
713  //
714  // Dump any straggling bitmap entries now as this range is finished.
715  //
716 
717  if (InitialPfn != NULL) {
718 
719  ASSERT (Pfn1 == LastPfn);
720  ASSERT (Pfn1 < MaxPfn);
721  ASSERT (Pfn1 > InitialPfn);
722 
723  MemSnap->Count = Pfn1 - InitialPfn;
724  PerfInfoLogBytes (PERFINFO_LOG_TYPE_MEMORYSNAPLITE,
725  MemSnap,
726  sizeof(MMPFN_MEMSNAP_INFORMATION) +
727  (ULONG) ((MemSnap->Count + 7) / 8));
728 
729  RtlClearAllBits (&BitMap);
730  }
731  }
732 
733  MI_UNLOCK_DYNAMIC_MEMORY_SHARED ();
734 
735  KeLeaveGuardedRegionThread (Thread);
736 
737  return STATUS_SUCCESS;
738 }

773  KIRQL OldIrql;

775  PMMPFN EndPfn;

782

784  BufferLast = BufferPointer + PFN_ID_BUFFERS;
785  RtlZeroMemory (PfnIdBuffer, sizeof(PfnIdBuffer));
786 
787  Thread = KeGetCurrentThread ();
788 
789  KeEnterGuardedRegionThread (Thread);
790 
791  MI_LOCK_DYNAMIC_MEMORY_SHARED ();
792 
793  //
794  // Walk through the ranges and identify pages until
795  // the buffer is full or we've run out of pages.
796  //
797 
798  for (i = 0; i < MmPhysicalMemoryBlock->NumberOfRuns; i += 1) {
799 
800  PageFrameIndex = MmPhysicalMemoryBlock->Run[i].BasePage;
801  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
802 
803  EndPfn = Pfn1 + MmPhysicalMemoryBlock->Run[i].PageCount;
804 
805  LOCK_PFN (OldIrql);
806 
807  while (Pfn1 < EndPfn) {
808 
809  MiIdentifyPfn (Pfn1, BufferPointer);
810 
811  BufferPointer += 1;
812 
813  if (BufferPointer == BufferLast) {
814 
815  //
816  // Release and reacquire the PFN lock so it's not held so long.
817  //
818 
819  UNLOCK_PFN (OldIrql);
820 
821  //
822  // Log the buffered entries.
823  //
824 
825  BufferPointer = &PfnIdBuffer[0];
826  do {
827 
828  PerfInfoLogBytes (PERFINFO_LOG_TYPE_PAGEINMEMORY,
829  BufferPointer,
830  sizeof(PfnIdBuffer[0]));
831 
832  BufferPointer += 1;
833 
834  } while (BufferPointer < BufferLast);
835 
836  //
837  // Reset the buffer to the beginning and zero it.
838  //
839 
840  BufferPointer = &PfnIdBuffer[0];
841  RtlZeroMemory (PfnIdBuffer, sizeof(PfnIdBuffer));
842 
843  LOCK_PFN (OldIrql);
844  }
845  Pfn1 += 1;
846  }
847 
848  UNLOCK_PFN (OldIrql);
849  }
850 
851  //
852  // Note that releasing this mutex here means the last entry can be
853  // inserted out of order if we are preempted and another thread starts
854  // the same operation (or if we're on an MP machine). The PERF module
855  // must handle this properly as any synchronization provided by this
856  // routine is purely a side effect not deliberate.
857  //
858 
859  MI_UNLOCK_DYNAMIC_MEMORY_SHARED ();
860  KeLeaveGuardedRegionThread (Thread);
861 
862  if (BufferPointer != &PfnIdBuffer[0]) {
863 
864  BufferLast = BufferPointer;
865  BufferPointer = &PfnIdBuffer[0];
866 
867  do {
868 
869  PerfInfoLogBytes (PERFINFO_LOG_TYPE_PAGEINMEMORY,
870  BufferPointer,
871  sizeof(PfnIdBuffer[0]));
872 
873  BufferPointer += 1;
874 
875  } while (BufferPointer < BufferLast);
876  }
877 
878  return STATUS_SUCCESS;
879 }

891

893 
894 Arguments:
895 
896  Pfn1 - Supplies the PFN element of the page frame number being queried.
897

901 
902  None.
903

906  Kernel mode. PFN lock held.
907

915  PVOID VirtualAddress;

955  // So don't bother for now.
956  //

963  }

968  //
969 
970  PfnIdentity->u1.e1.Pinned = 1;
971  MI_INCREMENT_IDENTIFY_COUNTER (10);
972  }

974

976  case ModifiedPageList:
977  case ModifiedNoWritePageList:
978  if (Pfn1->u3.e2.ReferenceCount >= 1) {
979 
980  //
981  // This page is pinned, presumably for an ongoing I/O.
982  //
983

985  MI_INCREMENT_IDENTIFY_COUNTER (11);
986  }
987 
988  if ((Pfn1->u3.e1.PageLocation == ModifiedPageList) &&
989  (MI_IS_PFN_DELETED (Pfn1)) &&
990  (Pfn1->u2.ShareCount == 0)) {
991 
992  //
993  // This page may be a modified write completing in the
994  // context of the modified writer thread. If the
995  // address space was deleted while the I/O was in
996  // progress, the frame will be released now. More
997  // importantly, the frame's containing frame is
998  // meaningless as it may have already been freed
999  // and reused.
1000  //
1001  // We can't tell what this page was being used for
1002  // since its address space is gone, so just call it
1003  // process private for now.
1004  //
1005

1007  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_PROCESSPRIVATE;
1008 
1009  return;
1010  }
1011 
1012  if ((Pfn1->u3.e1.PageLocation == StandbyPageList) &&
1013  (MI_IS_PFN_DELETED (Pfn1)) &&
1014  (Pfn1->u3.e1.ReadInProgress == 1)) {
1015

1017  // Don't bother trying to get fancy and decode any prototype
1018  // PTE encoding here in the PFN or original PTE. This is a
1019  // case where the user deleted the virtual address space
1020  // while one of its threads was in the middle of reading it
1021  // in !
1022  //

1025  PfnIdentity->u1.e1.Pinned = 1;
1026  VirtualAddress = MiGetVirtualAddressMappedByPte (Pfn1->PteAddress);
1027  PfnIdentity->u2.VirtualAddress = PAGE_ALIGN (VirtualAddress);
1028  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_PROCESSPRIVATE;

1030  return;
1031  }
1032 
1033  break;
1034 
1035  case TransitionPage:
1036 
1037  //
1038  // This page is pinned due to a straggling I/O - the virtual
1039  // address has been deleted but an I/O referencing it has not
1040  // completed.
1041  //
1042 
1043  PfnIdentity->u1.e1.Pinned = 1;
1044  MI_INCREMENT_IDENTIFY_COUNTER (11);
1045  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_PROCESSPRIVATE;

1048  default:

1050  DbgPrintEx (DPFLTR_MM_ID, DPFLTR_ERROR_LEVEL,

1055  break;

1057  }

1059  //
1060  // Capture differing information based on the type of page being examined.
1061  //
1062 
1063  //
1064  // General purpose stress shows 40% of the pages are prototypes so
1065  // for speed, check for these first.
1066  //
1067

1069 
1070  MI_INCREMENT_IDENTIFY_COUNTER (12);
1071

1073 
1074  //
1075  // Demand zero or (equivalently) pagefile backed.

1077  // There are some hard problems here preventing more indepth

1080  // 1. The PFN contains a backpointer to the prototype PTE - but
1081  // there is no definitive way to get to the SEGMENT or
1082  // CONTROL_AREA from this.
1083  //
1084  // 2. The prototype PTE pointer itself may be paged out and
1085  // the PFN lock is held right now.
1086  //
1087 
1088  MI_INCREMENT_IDENTIFY_COUNTER (13);
1089 
1090  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_PAGEFILEMAPPED;
1091  return;
1092  }
1093 
1094  MI_INCREMENT_IDENTIFY_COUNTER (14);
1095 
1096  //
1097  // Backed by a mapped file.
1098  //
1099

1102  ASSERT (ControlArea->u.Flags.File == 1);
1103  FilePointer = ControlArea->FilePointer;
1104  ASSERT (FilePointer != NULL);

1109  //
1110

1114  PfnIdentity->u1.e2.Offset = (MiStartingOffset (Subsection, Pfn1->PteAddress) >> MMSECTOR_SHIFT);

1118  //
1119  // The only time we should be here (a valid PFN with no subsection)
1120  // is if we are the segment dereference thread putting pages into

1138  }

1141 
1142  //
1143  // This mapped file has a name.
1144  //
1145 
1146  MI_INCREMENT_IDENTIFY_COUNTER (16);
1147  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_FILE;
1148  return;

1150

1154  //
1155 
1156  MI_INCREMENT_IDENTIFY_COUNTER (17);
1157  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_FILE;
1158  return;
1159  }

1165  // This is initial nonpaged pool.

1187  //
1188 
1189  MI_INCREMENT_IDENTIFY_COUNTER (19);

1278  MI_INCREMENT_IDENTIFY_COUNTER (25);

1299  // Carefully check here as this could be a legitimate frame as well.

1337  return;
1338  }
1339  }

1346  //

1355  // a top level page directory page for a process that has been
1356  // swapped out but is still waiting for the transition/modified
1357  // page table pages to be reclaimed. ie: until that happens, the

1359  // page are pointing at the EPROCESS pool page.
1360  //

1363  (Pfn1->PteAddress <= (PMMPTE) PTE_TOP))
1364  {
1365  PageFrameIndex = Pfn1->u4.PteFrame;
1366  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
1367  i += 1;
1368  }
1369  else {
1370  MI_INCREMENT_IDENTIFY_COUNTER (41);
1371  break;
1372  }
1373  }
1374

1376 
1377  PfnIdentity->u1.e3.PageDirectoryBase = PageFrameIndex;
1378 
1379 #if defined(_X86PAE_)
1380 
1381  //
1382  // PAE is unique because the 3rd level is not defined as only a mini
1383  // 4 entry 3rd level is in use. Check for that explicitly, noting that
1384  // it takes one extra walk to get to the top. Top level PAE pages (the
1385  // ones that contain only the 4 PDPTE pointers) are treated above as
1386  // active pinned pages, not as pagetable pages because each one is shared
1387  // across 127 processes and resides in the system global space.
1388  //
1389 
1390  if (i == _MI_PAGING_LEVELS + 1) {
1391 
1392  //
1393  // Had to walk all the way to the top. Must be a data page.
1394  //
1395 
1396  MI_INCREMENT_IDENTIFY_COUNTER (29);
1397  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_PROCESSPRIVATE;
1398  return;
1399  }
1400 
1401 #else
1402 
1403  if (i == _MI_PAGING_LEVELS) {
1404 
1405  //
1406  // Had to walk all the way to the top. Must be a data page.
1407  //
1408 
1409  MI_INCREMENT_IDENTIFY_COUNTER (29);
1410  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_PROCESSPRIVATE;
1411  return;
1412  }
1413 
1414 #endif
1415 
1416  //
1417  // Must have been a page in the hierarchy (not a data page) as we arrived
1418  // at the top early.
1419  //
1420 
1421  MI_INCREMENT_IDENTIFY_COUNTER (30);
1422  PfnIdentity->u1.e1.UseDescription = MMPFNUSE_PAGETABLE;
1423 
1424  return;

1428 MiSnapCopyMemory (
1429  IN PVOID Destination,
1430  IN PFN_NUMBER PageFrameIndex,
1431  IN SIZE_T NumberOfBytes
1432  )
1433 // nonpaged helper routine since it acquires a spinlock but is used from
1434 // paged code callers.
1435 {
1436  PVOID Va;
1437  KIRQL OldIrql;
1438  PEPROCESS CurrentProcess;
1439  LOGICAL AtDpc;
1440 
1441  CurrentProcess = PsGetCurrentProcess ();
1442 
1443  //
1444  // Determine if we're already at DISPATCH to avoid "raising" IRQL to the
1445  // existing level. In addition to avoiding the overhead of a HAL call on
1446  // some platforms, this also avoids the verifier irql stack trace logging
1447  // which consumes large amounts of kernel stack (ie, can cause overflow).
1448  //
1449 
1450  if (KeGetCurrentIrql () == DISPATCH_LEVEL) {
1451  AtDpc = TRUE;
1452  OldIrql = MM_NOIRQL;
1453  Va = MiMapPageInHyperSpaceAtDpc (CurrentProcess, PageFrameIndex);
1454  }
1455  else {
1456  AtDpc = FALSE;
1457 
1458  //
1459  // Note the call below can also set OldIrql to MM_NOIRQL (when
1460  // superpages are used) so a distinct local (AtDpc) must be used.
1461  //
1462 
1463  Va = MiMapPageInHyperSpace (CurrentProcess, PageFrameIndex, &OldIrql);
1464  }
1465 
1466  RtlCopyMemory (Destination, Va, NumberOfBytes);
1467 
1468  if (AtDpc == TRUE) {
1469  MiUnmapPageInHyperSpaceFromDpc (CurrentProcess, Va);
1470  }
