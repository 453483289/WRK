82 /*++
83 
84 Routine Description:
85 
86  This routine cracks a path. It picks off the first element in the
87  given path name and provides both it and the remaining part. A path
88  is a set of file names separated by backslashes. If a name begins
89  with a backslash, the FirstName is the string immediately following
90  the backslash. Here are some examples:
91 
92  Path FirstName RemainingName
93  ---- --------- -------------
94  empty empty empty
95 
96  \ empty empty
97 
98  A A empty
99 
100  \A A empty
101 
102  A\B\C\D\E A B\C\D\E
103 
104  *A? *A? empty
105 
106 
107  Note that both output strings use the same string buffer memory of the
108  input string, and are not necessarily null terminated.
109 
110  Also, this routine makes no judgement as to the legality of each
111  file name component. This must be done separatly when each file name
112  is extracted.
113 
114 Arguments:
115 
116  Path - The full path name to crack.
117 
118  FirstName - The first name in the path. Don't allocate a buffer for
119  this string.
120 
121  RemainingName - The rest of the path. Don't allocate a buffer for this
122  string.
123 
124 Return Value:
125 
126  None.
127 
128 --*/
129 
130 {
131  ULONG i = 0;
132  ULONG PathLength;
133  ULONG FirstNameStart;
134 
135  PAGED_CODE();
136 
137  //
138  // Make both output strings empty for now
139  //
140 
141  FirstName->Length = 0;
142  FirstName->MaximumLength = 0;
143  FirstName->Buffer = NULL;
144 
145  RemainingName->Length = 0;
146  RemainingName->MaximumLength = 0;
147  RemainingName->Buffer = NULL;
148 
149  PathLength = Path.Length / sizeof(WCHAR);
150 
151  //
152  // Check for an empty input string
153  //
154 
155  if (PathLength == 0) {
156 
157  return;
158  }
159 
160  //
161  // Skip over a starting backslash, and make sure there is more.
162  //
163 
164  if ( Path.Buffer[0] == L'\\' ) {
165 
166  i = 1;
167  }
168 
169  //
170  // Now run down the input string until we hit a backslash or the end
171  // of the string, remembering where we started;
172  //
173 
174  for ( FirstNameStart = i;
175  (i < PathLength) && (Path.Buffer[i] != L'\\');
176  i += 1 ) {
177 
178  NOTHING;
179  }
180 
181  //
182  // At this point all characters up to (but not including) i are
183  // in the first part. So setup the first name
184  //
185 
186  FirstName->Length = (USHORT)((i - FirstNameStart) * sizeof(WCHAR));
187  FirstName->MaximumLength = FirstName->Length;
188  FirstName->Buffer = &Path.Buffer[FirstNameStart];
189 
190  //
191  // Now the remaining part needs a string only if the first part didn't
192  // exhaust the entire input string. We know that if anything is left
193  // that is must start with a backslash. Note that if there is only
194  // a trailing backslash, the length will get correctly set to zero.
195  //
196 
197  if (i < PathLength) {
198 
199  RemainingName->Length = (USHORT)((PathLength - (i + 1)) * sizeof(WCHAR));
200  RemainingName->MaximumLength = RemainingName->Length;
201  RemainingName->Buffer = &Path.Buffer[i + 1];
202  }
203 
204  //
205  // And return to our caller
206  //
207 
208  return;
209 }

278 /*++
279 
280 Routine Description:
281 
282  This routine simple returns whether the two names are exactly equal.
283  If the two names are known to be constant, this routine is much
284  faster than FsRtlIsNameInExpression.
285 
286 Arguments:
287 
288  ConstantNameA - Constant name.
289 
290  ConstantNameB - Constant name.
291 
292  IgnoreCase - TRUE if the Names should be Upcased before comparing.
293 
294  UpcaseTable - If supplied, use this table for case insensitive compares,
295  otherwise, use the default system upcase table.
296 
297 Return Value:
298 
299  BOOLEAN - TRUE if the two names are lexically equal.
300 
301 --*/
302 
303 {
304  ULONG Index;
305  ULONG NameLength;
306  BOOLEAN FreeStrings = FALSE;
307 
308  UNICODE_STRING LocalNameA;
309  UNICODE_STRING LocalNameB;
310 
311  PAGED_CODE();
312 
313  //
314  // If the names aren't even the same size, then return FALSE right away.
315  //
316 
317  if ( ConstantNameA->Length != ConstantNameB->Length ) {
318 
319  return FALSE;
320  }
321 
322  NameLength = ConstantNameA->Length / sizeof(WCHAR);
323 
324  //
325  // If we weren't given an upcase table, we have to upcase the names
326  // ourselves.
327  //
328 
329  if ( IgnoreCase && !ARGUMENT_PRESENT(UpcaseTable) ) {
330 
331  NTSTATUS Status;
332 
333  Status = RtlUpcaseUnicodeString( &LocalNameA, ConstantNameA, TRUE );
334 
335  if ( !NT_SUCCESS(Status) ) {
336 
337  ExRaiseStatus( Status );
338  }
339 
340  Status = RtlUpcaseUnicodeString( &LocalNameB, ConstantNameB, TRUE );
341 
342  if ( !NT_SUCCESS(Status) ) {
343 
344  RtlFreeUnicodeString( &LocalNameA );
345 
346  ExRaiseStatus( Status );
347  }
348 
349  ConstantNameA = &LocalNameA;
350  ConstantNameB = &LocalNameB;
351 
352  IgnoreCase = FALSE;
353  FreeStrings = TRUE;
354  }
355 
356  //
357  // Do either case sensitive or insensitive compare.
358  //
359 
360  if ( !IgnoreCase ) {
361 
362  BOOLEAN BytesEqual;
363 
364  BytesEqual = (BOOLEAN) RtlEqualMemory( ConstantNameA->Buffer,
365  ConstantNameB->Buffer,
366  ConstantNameA->Length );
367 
368  if ( FreeStrings ) {
369 
370  RtlFreeUnicodeString( &LocalNameA );
371  RtlFreeUnicodeString( &LocalNameB );
372  }
373 
374  return BytesEqual;
375 
376  } else {
377 
378  for (Index = 0; Index < NameLength; Index += 1) {
379 
380  if ( UpcaseTable[ConstantNameA->Buffer[Index]] !=
381  UpcaseTable[ConstantNameB->Buffer[Index]] ) {
382 
383  return FALSE;
384  }
385  }
386 
387  return TRUE;
388  }
389 }

464 BOOLEAN
465 FsRtlIsNameInExpressionPrivate (

467  IN PUNICODE_STRING Name,

472 /*++
473 
474 Routine Description:
475 
476  This routine compares a Dbcs name and an expression and tells the caller
477  if the name is in the language defined by the expression. The input name
478  cannot contain wildcards, while the expression may contain wildcards.
479 
480  Expression wild cards are evaluated as shown in the nondeterministic
481  finite automatons below. Note that ~* and ~? are DOS_STAR and DOS_QM.
482 
483 
484  ~* is DOS_STAR, ~? is DOS_QM, and ~. is DOS_DOT
485 
486 
487  S
488  <-----<
489  X | | e Y
490  X * Y == (0)----->-(1)->-----(2)-----(3)
491 
492 
493  S-.
494  <-----<
495  X | | e Y
496  X ~* Y == (0)----->-(1)->-----(2)-----(3)
497 
498 
499 
500  X S S Y
501  X ?? Y == (0)---(1)---(2)---(3)---(4)
502 
503 
504 
505  X . . Y
506  X ~.~. Y == (0)---(1)----(2)------(3)---(4)
507  | |________|
508  | ^ |
509  |_______________|
510  ^EOF or .^
511 
512 
513  X S-. S-. Y
514  X ~?~? Y == (0)---(1)-----(2)-----(3)---(4)
515  | |________|
516  | ^ |
517  |_______________|
518  ^EOF or .^
519 
520 
521 
522  where S is any single character
523 
524  S-. is any single character except the final .
525 
526  e is a null character transition
527 
528  EOF is the end of the name string
529 
530  In words:
531 
532  * matches 0 or more characters.
533 
534  ? matches exactly 1 character.
535 
536  DOS_STAR matches 0 or more characters until encountering and matching
537  the final . in the name.
538 
539  DOS_QM matches any single character, or upon encountering a period or
540  end of name string, advances the expression to the end of the
541  set of contiguous DOS_QMs.
542 
543  DOS_DOT matches either a . or zero characters beyond name string.
544 
545 Arguments:
546 
547  Expression - Supplies the input expression to check against
548  (Caller must already upcase if passing CaseInsensitive TRUE.)
549 
550  Name - Supplies the input name to check for.
551 
552  CaseInsensitive - TRUE if Name should be Upcased before comparing.
553 
554 Return Value:
555 
556  BOOLEAN - TRUE if Name is an element in the set of strings denoted
557  by the input Expression and FALSE otherwise.
558 
559 --*/
560 
561 {
562  USHORT NameOffset;
563  USHORT ExprOffset;
564 
565  ULONG SrcCount;
566  ULONG DestCount;
567  ULONG PreviousDestCount;
568  ULONG MatchesCount;
569 
570  WCHAR NameChar = 0, ExprChar;
571 
572  USHORT LocalBuffer[MATCHES_ARRAY_SIZE * 2];
573 
574  USHORT *AuxBuffer = NULL;
575  USHORT *PreviousMatches;
576  USHORT *CurrentMatches;
577 
578  USHORT MaxState;
579  USHORT CurrentState;
580 
581  BOOLEAN NameFinished = FALSE;
582 
583  //
584  // The idea behind the algorithm is pretty simple. We keep track of
585  // all possible locations in the regular expression that are matching
586  // the name. If when the name has been exhausted one of the locations
587  // in the expression is also just exhausted, the name is in the language
588  // defined by the regular expression.
589  //
590 
591  PAGED_CODE();
592 
593  DebugTrace(+1, Dbg, "FsRtlIsNameInExpression\n", 0);
594  DebugTrace( 0, Dbg, " Expression = %Z\n", Expression );
595  DebugTrace( 0, Dbg, " Name = %Z\n", Name );
596  DebugTrace( 0, Dbg, " CaseInsensitive = %08lx\n", CaseInsensitive );
597 
598  //
599  // If one string is empty return FALSE. If both are empty return TRUE.
600  //
601 
602  if ( (Name->Length == 0) || (Expression->Length == 0) ) {
603 
604  return (BOOLEAN)(!(Name->Length + Expression->Length));
605  }
606 
607  //
608  // Special case by far the most common wild card search of *
609  //
610 
611  if ((Expression->Length == 2) && (Expression->Buffer[0] == L'*')) {
612 
613  return TRUE;
614  }
615 
616  ASSERT( !IgnoreCase || ARGUMENT_PRESENT(UpcaseTable) );
617 
618  //
619  // Also special case expressions of the form *X. With this and the prior
620  // case we have covered virtually all normal queries.
621  //
622 
623  if (Expression->Buffer[0] == L'*') {
624 
625  UNICODE_STRING LocalExpression;
626 
627  LocalExpression = *Expression;
628 
629  LocalExpression.Buffer += 1;
630  LocalExpression.Length -= 2;
631 
632  //
633  // Only special case an expression with a single *
634  //
635 
636  if ( !FsRtlDoesNameContainWildCards( &LocalExpression ) ) {
637 
638  ULONG StartingNameOffset;
639 
640  if (Name->Length < (USHORT)(Expression->Length - sizeof(WCHAR))) {
641 
642  return FALSE;
643  }
644 
645  StartingNameOffset = ( Name->Length -
646  LocalExpression.Length ) / sizeof(WCHAR);
647 
648  //
649  // Do a simple memory compare if case sensitive, otherwise
650  // we have got to check this one character at a time.
651  //
652 
653  if ( !IgnoreCase ) {
654 
655  return (BOOLEAN) RtlEqualMemory( LocalExpression.Buffer,
656  Name->Buffer + StartingNameOffset,
657  LocalExpression.Length );
658 
659  } else {
660 
661  for ( ExprOffset = 0;
662  ExprOffset < (USHORT)(LocalExpression.Length / sizeof(WCHAR));
663  ExprOffset += 1 ) {
664 
665  NameChar = Name->Buffer[StartingNameOffset + ExprOffset];
666  NameChar = UpcaseTable[NameChar];
667 
668  ExprChar = LocalExpression.Buffer[ExprOffset];
669 
670  ASSERT( ExprChar == UpcaseTable[ExprChar] );
671 
672  if ( NameChar != ExprChar ) {
673 
674  return FALSE;
675  }
676  }
677 
678  return TRUE;
679  }
680  }
681  }
682 
683  //
684  // Walk through the name string, picking off characters. We go one
685  // character beyond the end because some wild cards are able to match
686  // zero characters beyond the end of the string.
687  //
688  // With each new name character we determine a new set of states that
689  // match the name so far. We use two arrays that we swap back and forth
690  // for this purpose. One array lists the possible expression states for
691  // all name characters up to but not including the current one, and other
692  // array is used to build up the list of states considering the current
693  // name character as well. The arrays are then switched and the process
694  // repeated.
695  //
696  // There is not a one-to-one correspondence between state number and
697  // offset into the expression. This is evident from the NFAs in the
698  // initial comment to this function. State numbering is not continuous.
699  // This allows a simple conversion between state number and expression
700  // offset. Each character in the expression can represent one or two
701  // states. * and DOS_STAR generate two states: ExprOffset*2 and
702  // ExprOffset*2 + 1. All other expression characters can produce only
703  // a single state. Thus ExprOffset = State/2.
704  //
705  //
706  // Here is a short description of the variables involved:
707  //
708  // NameOffset - The offset of the current name char being processed.
709  //
710  // ExprOffset - The offset of the current expression char being processed.
711  //
712  // SrcCount - Prior match being investigated with current name char
713  //
714  // DestCount - Next location to put a matching assuming current name char
715  //
716  // NameFinished - Allows one more iteration through the Matches array
717  // after the name is exhausted (to come *s for example)
718  //
719  // PreviousDestCount - This is used to prevent entry duplication, see coment
720  //
721  // PreviousMatches - Holds the previous set of matches (the Src array)
722  //
723  // CurrentMatches - Holds the current set of matches (the Dest array)
724  //
725  // AuxBuffer, LocalBuffer - the storage for the Matches arrays
726  //
727 
728  //
729  // Set up the initial variables
730  //
731 
732  PreviousMatches = &LocalBuffer[0];
733  CurrentMatches = &LocalBuffer[MATCHES_ARRAY_SIZE];
734 
735  PreviousMatches[0] = 0;
736  MatchesCount = 1;
737 
738  NameOffset = 0;
739 
740  MaxState = (USHORT)(Expression->Length * 2);
741 
742  while ( !NameFinished ) {
743 
744  if ( NameOffset < Name->Length ) {
745 
746  NameChar = Name->Buffer[NameOffset / sizeof(WCHAR)];
747 
748  NameOffset += sizeof(WCHAR);;
749 
750  } else {
751 
752  NameFinished = TRUE;
753 
754  //
755  // if we have already exhausted the expression, cool. Don't
756  // continue.
757  //
758 
759  if ( PreviousMatches[MatchesCount-1] == MaxState ) {
760 
761  break;
762  }
763  }
764 
765 
766  //
767  // Now, for each of the previous stored expression matches, see what
768  // we can do with this name character.
769  //
770 
771  SrcCount = 0;
772  DestCount = 0;
773  PreviousDestCount = 0;
774 
775  while ( SrcCount < MatchesCount ) {
776 
777  USHORT Length;
778 
779  //
780  // We have to carry on our expression analysis as far as possible
781  // for each character of name, so we loop here until the
782  // expression stops matching. A clue here is that expression
783  // cases that can match zero or more characters end with a
784  // continue, while those that can accept only a single character
785  // end with a break.
786  //
787 
788  ExprOffset = (USHORT)((PreviousMatches[SrcCount++] + 1) / 2);
789 
790 
791  Length = 0;
792 
793  while ( TRUE ) {
794 
795  if ( ExprOffset == Expression->Length ) {
796 
797  break;
798  }
799 
800  //
801  // The first time through the loop we don't want
802  // to increment ExprOffset.
803  //
804 
805  ExprOffset = (USHORT)(ExprOffset + Length);
806  Length = sizeof(WCHAR);
807 
808  CurrentState = (USHORT)(ExprOffset * 2);
809 
810  if ( ExprOffset == Expression->Length ) {
811 
812  CurrentMatches[DestCount++] = MaxState;
813  break;
814  }
815 
816  ExprChar = Expression->Buffer[ExprOffset / sizeof(WCHAR)];
817 
818  ASSERT( !IgnoreCase || !((ExprChar >= L'a') && (ExprChar <= L'z')) );
819 
820  //
821  // Before we get started, we have to check for something
822  // really gross. We may be about to exhaust the local
823  // space for ExpressionMatches[][], so we have to allocate
824  // some pool if this is the case. Yuk!
825  //
826 
827  if ( (DestCount >= MATCHES_ARRAY_SIZE - 2) &&
828  (AuxBuffer == NULL) ) {
829 
830  ULONG ExpressionChars;
831 
832  ExpressionChars = Expression->Length / sizeof(WCHAR);
833 
834  AuxBuffer = FsRtlpAllocatePool( PagedPool,
835  (ExpressionChars+1) *
836  sizeof(USHORT)*2*2 );
837 
838  RtlCopyMemory( AuxBuffer,
839  CurrentMatches,
840  MATCHES_ARRAY_SIZE * sizeof(USHORT) );
841 
842  CurrentMatches = AuxBuffer;
843 
844  RtlCopyMemory( AuxBuffer + (ExpressionChars+1)*2,
845  PreviousMatches,
846  MATCHES_ARRAY_SIZE * sizeof(USHORT) );
847 
848  PreviousMatches = AuxBuffer + (ExpressionChars+1)*2;
849  }
850 
851  //
852  // * matches any character zero or more times.
853  //
854 
855  if (ExprChar == L'*') {
856 
857  CurrentMatches[DestCount++] = CurrentState;
858  CurrentMatches[DestCount++] = CurrentState + 3;
859  continue;
860  }
861 
862  //
863  // DOS_STAR matches any character except . zero or more times.
864  //
865 
866  if (ExprChar == DOS_STAR) {
867 
868  BOOLEAN ICanEatADot = FALSE;
869 
870  //
871  // If we are at a period, determine if we are allowed to
872  // consume it, ie. make sure it is not the last one.
873  //
874 
875  if ( !NameFinished && (NameChar == '.') ) {
876 
877  USHORT Offset;
878 
879  for ( Offset = NameOffset;
880  Offset < Name->Length;
881  Offset = (USHORT)(Offset + Length) ) {
882 
883  if (Name->Buffer[Offset / sizeof(WCHAR)] == L'.') {
884 
885  ICanEatADot = TRUE;
