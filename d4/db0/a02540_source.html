360 /*++
361 
362 Routine Description:
363 
364  This function locates the specified number of unused PTEs
365  within the non paged portion of system space.
366 
367 Arguments:
368 
369  NumberOfPtes - Supplies the number of PTEs to locate.
370 
371  SystemPtePoolType - Supplies the PTE type of the pool to expand, one of
372  SystemPteSpace or NonPagedPoolExpansion.
373 
374 Return Value:
375 
376  Returns the address of the first PTE located.
377  NULL if no system PTEs can be located.
378 
379 Environment:
380 
381  Kernel mode, DISPATCH_LEVEL or below.
382 
383 --*/
384 
385 {
386  ULONG i;
387  PMMPTE PointerPte;
388  ULONG Index;
389  ULONG TimeStamp;
390  PTE_QUEUE_POINTER Value;
391 
392  if (SystemPtePoolType == SystemPteSpace) {
393 
394  if (NumberOfPtes <= MM_PTE_TABLE_LIMIT) {
395  Index = MmSysPteTables [NumberOfPtes];
396  ASSERT (NumberOfPtes <= MmSysPteIndex[Index]);
397 
398  if (ExRemoveHeadNBQueue (MiSystemPteNBHead[Index], (PULONG64)&Value) == TRUE) {
399  InterlockedDecrement ((PLONG)&MmSysPteListBySizeCount[Index]);
400 
401  PointerPte = UnpackPTEPointer (&Value);
402 
403  TimeStamp = UnpackPTETimeStamp (&Value);
404 
405  ASSERT (PointerPte >= MmSystemPtesStart[SystemPtePoolType]);
406  ASSERT (PointerPte <= MmSystemPtesEnd[SystemPtePoolType]);
407 
408 #if DBG
409  PointerPte->u.List.NextEntry = 0xABCDE;
410  NumberOfPtes = MmSysPteIndex[Index];
411  ASSERT (NumberOfPtes != 0);
412  PointerPte += NumberOfPtes;
413 
414  do {
415  PointerPte -= 1;
416  ASSERT (PointerPte->u.Hard.Valid == 0);
417  NumberOfPtes -= 1;
418  } while (NumberOfPtes != 0);
419 #endif
420 
421  i = MmSysPteMinimumFree[Index];
422 
423  if (MmSysPteListBySizeCount[Index] < i) {
424  MiFeedSysPtePool (Index);
425  }
426 
427  //
428  // The last thing is to check whether the TB needs flushing.
429  //
430 
431  if (MiCompareTbFlushTimeStamp (TimeStamp, SYSPTES_FLUSH_COUNTER_MASK)) {
432  MI_FLUSH_ENTIRE_TB (0xC);
433  }
434 
435  if (MmTrackPtes & 0x2) {
436  MiCheckPteReserve (PointerPte, MmSysPteIndex[Index]);
437  }
438 
439  return PointerPte;
440  }
441 
442  //
443  // Fall through and go the long way to satisfy the PTE request.
444  //
445 
446  NumberOfPtes = MmSysPteIndex [Index];
447  }
448  }
449 
450  PointerPte = MiReserveAlignedSystemPtes (NumberOfPtes,
451  SystemPtePoolType,
452  0);
453 
454  if (PointerPte == NULL) {
455  MiSystemPteAllocationFailed += 1;
456  }
457 
458 #if DBG
459  if (PointerPte != NULL) {
460 
461  ASSERT (NumberOfPtes != 0);
462  PointerPte += NumberOfPtes;
463 
464  do {
465  PointerPte -= 1;
466  ASSERT (PointerPte->u.Hard.Valid == 0);
467  NumberOfPtes -= 1;
468  } while (NumberOfPtes != 0);
469  }
470 #endif
471 
472  return PointerPte;
473 }

482 Routine Description:
483 
484  This routine adds PTEs to the nonblocking queue lists.

486 Arguments:

548 /*++
549 
550 Routine Description:
551 
552  This function locates the specified number of unused PTEs to locate
553  within the non paged portion of system space.
554 
555 Arguments:
556 
557  NumberOfPtes - Supplies the number of PTEs to locate.
558 
559  SystemPtePoolType - Supplies the PTE type of the pool to expand, one of
560  SystemPteSpace or NonPagedPoolExpansion.
561 
562  Alignment - Supplies the virtual address alignment for the address
563  the returned PTE maps. For example, if the value is 64K,
564  the returned PTE will map an address on a 64K boundary.
565  An alignment of zero means to align on a page boundary.
566 
567 Return Value:
568 
569  Returns the address of the first PTE located.
570  NULL if no system PTEs can be located.
571 
572 Environment:
573 
574  Kernel mode, DISPATCH_LEVEL or below.
575 
576 --*/
577 
578 {
579  PMMPTE PointerPte;
580  PMMPTE PointerFollowingPte;
581  PMMPTE Previous;
582  ULONG_PTR SizeInSet;
583  KIRQL OldIrql;
584  ULONG MaskSize;
585  ULONG NumberOfRequiredPtes;
586  ULONG OffsetSum;
587  ULONG PtesToObtainAlignment;
588  PMMPTE NextSetPointer;
589  ULONG_PTR LeftInSet;
590  ULONG_PTR PteOffset;
591  PVOID VaFlushList[MM_MAXIMUM_FLUSH_COUNT];
592  PVOID BaseAddress;
593  ULONG TimeStamp;
594  ULONG j;
595 
596  MaskSize = (Alignment - 1) >> (PAGE_SHIFT - PTE_SHIFT);
597 
598  OffsetSum = (Alignment >> (PAGE_SHIFT - PTE_SHIFT));
599 
600 #if defined (_X86_)
601 restart:
602 #endif
603 
604  //
605  // The nonpaged PTE pool uses the invalid PTEs to define the pool
606  // structure. A global pointer points to the first free set
607  // in the list, each free set contains the number free and a pointer
608  // to the next free set. The free sets are kept in an ordered list
609  // such that the pointer to the next free set is always greater
610  // than the address of the current free set.
611  //
612  // As to not limit the size of this pool, two PTEs are used
613  // to define a free region. If the region is a single PTE, the
614  // prototype field within the PTE is set indicating the set
615  // consists of a single PTE.
616  //
617  // The page frame number field is used to define the next set
618  // and the number free. The two flavors are:
619  //
620  // o V
621  // n l
622  // e d
623  // +-----------------------+-+----------+
624  // | next set |0|0 0|
625  // +-----------------------+-+----------+
626  // | number in this set |0|0 0|
627  // +-----------------------+-+----------+
628  //
629  //
630  // +-----------------------+-+----------+
631  // | next set |1|0 0|
632  // +-----------------------+-+----------+
633  // ...
634  //
635 
636  //
637  // Acquire the system space lock to synchronize access.
638  //
639 
640  MiLockSystemSpace (OldIrql);
641 
642  PointerPte = &MmFirstFreeSystemPte[SystemPtePoolType];
643 
644  if (PointerPte->u.List.NextEntry == MM_EMPTY_PTE_LIST) {
645 
646  //
647  // End of list and none found.
648  //
649 
650  MiUnlockSystemSpace (OldIrql);
651 #if defined (_X86_)
652  if ((SystemPtePoolType == SystemPteSpace) &&
653  (MiRecoverExtraPtes (NumberOfPtes) == TRUE)) {
654 
655  goto restart;
656  }
657 #endif
658  MmPteFailures[SystemPtePoolType] += 1;
659  return NULL;
660  }
661 
662  Previous = PointerPte;
663 
664  PointerPte = MmSystemPteBase + PointerPte->u.List.NextEntry;
665 
666  if (Alignment <= PAGE_SIZE) {
667 
668  //
669  // Don't deal with alignment issues.
670  //
671 
672  while (TRUE) {
673 
674  if (PointerPte->u.List.OneEntry) {
675 
676  if (NumberOfPtes == 1) {
677  goto ExactFit;
678  }
679 
680  goto NextEntry;
681  }
682 
683  PointerFollowingPte = PointerPte + 1;
684  SizeInSet = (ULONG_PTR) PointerFollowingPte->u.List.NextEntry;
685 
686  if (NumberOfPtes < SizeInSet) {
687 
688  //
689  // Get the PTEs from this set and reduce the size of the
690  // set. Note that the size of the current set cannot be 1.
691  //
692 
693  if ((SizeInSet - NumberOfPtes) == 1) {
694 
695  //
696  // Collapse to the single PTE format.
697  //
698 
699  PointerPte->u.List.OneEntry = 1;
700  }
701  else {
702 
703  //
704  // Get the required PTEs from the end of the set.
705  //
706 
707  PointerFollowingPte->u.List.NextEntry = SizeInSet - NumberOfPtes;
708  }
709 
710  MmTotalFreeSystemPtes[SystemPtePoolType] -= NumberOfPtes;
711 #if DBG
712  if (MmDebug & MM_DBG_SYS_PTES) {
713  ASSERT (MmTotalFreeSystemPtes[SystemPtePoolType] ==
714  MiCountFreeSystemPtes (SystemPtePoolType));
715  }
716 #endif
717 
718  //
719  // Release the lock and flush the TB.
720  //
721 
722  MiUnlockSystemSpace (OldIrql);
723 
724  PointerPte += (SizeInSet - NumberOfPtes);
725  break;
726  }
727 
728  if (NumberOfPtes == SizeInSet) {
729 
730 ExactFit:
731 
732  //
733  // Satisfy the request with this complete set and change
734  // the list to reflect the fact that this set is gone.
735  //
736 
737  Previous->u.List.NextEntry = PointerPte->u.List.NextEntry;
738 
739  MmTotalFreeSystemPtes[SystemPtePoolType] -= NumberOfPtes;
740 #if DBG
741  if (MmDebug & MM_DBG_SYS_PTES) {
742  ASSERT (MmTotalFreeSystemPtes[SystemPtePoolType] ==
743  MiCountFreeSystemPtes (SystemPtePoolType));
744  }
745 #endif
746 
747  //
748  // Release the lock and flush the TB.
749  //
750 
751  MiUnlockSystemSpace (OldIrql);
752  break;
753  }
754 
755 NextEntry:
756 
757  //
758  // Point to the next set and try again.
759  //
760 
761  if (PointerPte->u.List.NextEntry == MM_EMPTY_PTE_LIST) {
762 
763  //
764  // End of list and none found.
765  //
766 
767  MiUnlockSystemSpace (OldIrql);
768 #if defined (_X86_)
769  if ((SystemPtePoolType == SystemPteSpace) &&
770  (MiRecoverExtraPtes (NumberOfPtes) == TRUE)) {
771 
772  goto restart;
773  }
774 #endif
775  MmPteFailures[SystemPtePoolType] += 1;
776  return NULL;
777  }
778  Previous = PointerPte;
779  PointerPte = MmSystemPteBase + PointerPte->u.List.NextEntry;
780  ASSERT (PointerPte > Previous);
781  }
782  }
783  else {
784 
785  //
786  // Deal with the alignment issues.
787  //
788 
789  while (TRUE) {
790 
791  if (PointerPte->u.List.OneEntry) {
792 
793  //
794  // Initializing PointerFollowingPte is not needed for
795  // correctness, but without it the compiler cannot compile
796  // this code W4 to check for use of uninitialized variables.
797  //
798 
799  PointerFollowingPte = NULL;
800  SizeInSet = 1;
801  }
802  else {
803  PointerFollowingPte = PointerPte + 1;
804  SizeInSet = (ULONG_PTR) PointerFollowingPte->u.List.NextEntry;
805  }
806 
807  PtesToObtainAlignment = (ULONG)
808  (((OffsetSum - ((ULONG_PTR)PointerPte & MaskSize)) & MaskSize) >>
809  PTE_SHIFT);
810 
811  NumberOfRequiredPtes = NumberOfPtes + PtesToObtainAlignment;
812 
813  if (NumberOfRequiredPtes < SizeInSet) {
814 
815  //
816  // Get the PTEs from this set and reduce the size of the
817  // set. Note that the size of the current set cannot be 1.
