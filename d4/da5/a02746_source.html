227  OUT PUNICODE_STRING *RegistryPath

229 /*++
230 
231 Routine Description:
232 
233  This routine is a callback into the driver to retrieve the list of
234  guids or data blocks that the driver wants to register with WMI. This
235  routine may not pend or block. Driver should NOT call
236  ClassWmiCompleteRequest.
237 
238 Arguments:
239 
240  DeviceObject is the device whose data block is being queried
241 
242  *RegFlags returns with a set of flags that describe the guids being
243  registered for this device. If the device wants enable and disable
244  collection callbacks before receiving queries for the registered
245  guids then it should return the WMIREG_FLAG_EXPENSIVE flag. Also the
246  returned flags may specify WMIREG_FLAG_INSTANCE_PDO in which case
247  the instance name is determined from the PDO associated with the
248  device object. Note that the PDO must have an associated devnode. If
249  WMIREG_FLAG_INSTANCE_PDO is not set then Name must return a unique
250  name for the device.
251 
252  InstanceName returns with the instance name for the guids if
253  WMIREG_FLAG_INSTANCE_PDO is not set in the returned *RegFlags. The
254  caller will call ExFreePool with the buffer returned.
255 
256  *RegistryPath returns with the registry path of the driver
257 
258 Return Value:
259 
260  status
261 
262 --*/

264  ANSI_STRING AnsiString;
265  NTSTATUS Status;
266 
267  PAGED_CODE();
268 
269  UNREFERENCED_PARAMETER (DeviceObject);
270  UNREFERENCED_PARAMETER (RegFlags);
271 
272  *RegistryPath = &WmipRegistryPath;
273 
274  RtlInitAnsiString(&AnsiString, "SMBiosData");
275 
276  Status = RtlAnsiStringToUnicodeString(InstanceName, &AnsiString, TRUE);
277 
278  return(Status);

281 NTSTATUS
282 WmipQueryWmiDataBlock(
283  IN PDEVICE_OBJECT DeviceObject,
284  IN PIRP Irp,
285  IN ULONG GuidIndex,
286  IN ULONG InstanceIndex,
287  IN ULONG InstanceCount,
288  IN OUT PULONG InstanceLengthArray,
289  IN ULONG BufferAvail,
290  OUT PUCHAR Buffer
291  )
292 /*++
293 
294 Routine Description:
295 
296  This routine is a callback into the driver to query for the contents of
297  all instances of a data block. When the driver has finished filling the
298  data block it must call IoWMICompleteRequest to complete the irp. The
299  driver can return STATUS_PENDING if the irp cannot be completed
300  immediately.
301 
302 Arguments:
303 
304  DeviceObject is the device whose data block is being queried. In the case
305  of the PnPId guid this is the device object of the device on whose
306  behalf the request is being processed.
307 
308  Irp is the Irp that makes this request
309 
310  GuidIndex is the index into the list of guids provided when the
311  device registered
312 
313  InstanceCount is the number of instances expected to be returned for
314  the data block.
315 
316  InstanceLengthArray is a pointer to an array of ULONG that returns the
317  lengths of each instance of the data block. If this is NULL then
318  there was not enough space in the output buffer to fulfill the request
319  so the irp should be completed with the buffer needed.
320 
321  BufferAvail on entry has the maximum size available to write the data
322  blocks.
323 
324  Buffer on return is filled with the returned data blocks. Note that each
325  instance of the data block must be aligned on a 8 byte boundary.
326 
327 
328 Return Value:
329 
330  status
331 
332 --*/
333 {
334  NTSTATUS status = STATUS_UNSUCCESSFUL;
335  ULONG sizeNeeded = 0, sizeSMBios;
336  PUCHAR BufferPtr;
337  PSMBIOSVERSIONINFO SMBiosVersionInfo;
338  PULONG TableSize = NULL;
339 
340  PAGED_CODE();
341 
342  switch (GuidIndex)
343  {
344  case SmbiosDataGuidIndex:
345  {
346  //
347  // SMBIOS data table query
348 #if defined(_AMD64_) || defined(i386)
349  WmipAssert((InstanceIndex == 0) && (InstanceCount == 1));
350 
351  sizeNeeded = sizeof(SMBIOSVERSIONINFO) + sizeof(ULONG);
352  if (BufferAvail >= sizeNeeded)
353  {
354  sizeSMBios = BufferAvail - sizeNeeded;
355  SMBiosVersionInfo = (PSMBIOSVERSIONINFO)Buffer;
356  TableSize = (PULONG) (Buffer + sizeof(SMBIOSVERSIONINFO));
357  BufferPtr = Buffer + sizeNeeded;
358  }
359  else
360  {
361  sizeSMBios = 0;
362  BufferPtr = NULL;
363  SMBiosVersionInfo = NULL;
364  }
365 
366  status = WmipGetSMBiosTableData(BufferPtr,
367  &sizeSMBios,
368  SMBiosVersionInfo);
369 
370  sizeNeeded += sizeSMBios;
371 
372  if (NT_SUCCESS(status))
373  {
374  *(TableSize) = sizeSMBios;
375  *InstanceLengthArray = sizeNeeded;
376  }
377 #else
378  status = STATUS_WMI_GUID_NOT_FOUND;
379 #endif
380  break;
381  }
382 
383  case PnPIdGuidIndex:
384  {
385  PDEVICE_OBJECT pDO;
386  UNICODE_STRING instancePath;
387  PREGENTRY regEntry;
388  ULONG dataBlockSize, paddedDataBlockSize, padSize;
389  ULONG i;
390 
391  regEntry = WmipFindRegEntryByDevice(DeviceObject, FALSE);
392  if (regEntry != NULL)
393  {
394  pDO = regEntry->PDO;
395  WmipAssert(pDO != NULL);
396 
397  if (pDO != NULL)
398  {
399  status = WmipPDOToDeviceInstanceName(pDO, &instancePath);
400  if (NT_SUCCESS(status))
401  {
402  dataBlockSize = instancePath.Length + sizeof(USHORT);
403  paddedDataBlockSize = (dataBlockSize + 7) & ~7;
404  padSize = paddedDataBlockSize - dataBlockSize;
405  sizeNeeded = paddedDataBlockSize * InstanceCount;
406  if (sizeNeeded <= BufferAvail)
407  {
408  for (i = InstanceIndex;
409  i < (InstanceIndex + InstanceCount);
410  i++)
411  {
412  *InstanceLengthArray++ = dataBlockSize;
413  *((PUSHORT)Buffer) = instancePath.Length;
414  Buffer += sizeof(USHORT);
415  RtlCopyMemory(Buffer,
416  instancePath.Buffer,
417  instancePath.Length);
418  Buffer += instancePath.Length;
419  RtlZeroMemory(Buffer, padSize);
420  Buffer += padSize;
421  }
422  } else {
423  status = STATUS_BUFFER_TOO_SMALL;
424  }
425 
426  RtlFreeUnicodeString(&instancePath);
427 
428  } else {
429  status = STATUS_WMI_GUID_NOT_FOUND;
430  }
431  } else {
432  status = STATUS_UNSUCCESSFUL;
433  }
434 
435  WmipUnreferenceRegEntry(regEntry);
436  } else {
437  WmipAssert(FALSE);
438  status = STATUS_WMI_GUID_NOT_FOUND;
439  }
440 
441  break;

719 NTSTATUS
720 IoWMISystemControl(
721  IN PWMILIB_INFO WmiLibInfo,
722  IN PDEVICE_OBJECT DeviceObject,
723  IN PIRP Irp
724  )
725 /*++
726 
727 Routine Description:
728 
729  Dispatch routine for IRP_MJ_SYSTEM_CONTROL. This routine will process
730  all wmi requests received, forwarding them if they are not for this
731  driver or determining if the guid is valid and if so passing it to
732  the driver specific function for handing out wmi requests.
733 
734 Arguments:
735 
736  WmiLibInfo has the WMI information control block
737 
738  DeviceObject - Supplies a pointer to the device object for this request.
739 
740  Irp - Supplies the Irp making the request.
741 
742 Return Value:
743 
744  status
745 
746 --*/
747 
748 {
749  PIO_STACK_LOCATION irpStack = IoGetCurrentIrpStackLocation(Irp);
750  ULONG bufferSize;
751  PUCHAR buffer;
752  NTSTATUS status;
753  ULONG retSize;
754  UCHAR minorFunction;
755  ULONG guidIndex = 0;
756  ULONG instanceCount = 0;
757  ULONG instanceIndex = 0;
758 
759  PAGED_CODE();
760 
761  //
762  // If the irp is not a WMI irp or it is not targetted at this device
763  // or this device has not registered with WMI then just forward it on.
764  minorFunction = irpStack->MinorFunction;
765  if ((minorFunction > IRP_MN_REGINFO_EX) ||
766  (irpStack->Parameters.WMI.ProviderId != (ULONG_PTR)DeviceObject) ||
767  (((minorFunction != IRP_MN_REGINFO) &&
768  ((minorFunction != IRP_MN_REGINFO_EX))) &&
769  (WmiLibInfo->GuidList == NULL)))
770  {
771  //
772  // IRP is not for us so forward if there is a lower device object
773  if (WmiLibInfo->LowerDeviceObject != NULL)
774  {
775  IoSkipCurrentIrpStackLocation(Irp);
776  return(IoCallDriver(WmiLibInfo->LowerDeviceObject, Irp));
777  } else {
778  status = STATUS_INVALID_DEVICE_REQUEST;
779  Irp->IoStatus.Status = status;
780  IoCompleteRequest(Irp, IO_NO_INCREMENT);
781  return(status);
782  }
783  }
784 
785  buffer = (PUCHAR)irpStack->Parameters.WMI.Buffer;
786  bufferSize = irpStack->Parameters.WMI.BufferSize;
787 
788  if ((minorFunction != IRP_MN_REGINFO) &&
789  (minorFunction != IRP_MN_REGINFO_EX))
790  {
791  //
792  // For all requests other than query registration info we are passed
793  // a guid. Determine if the guid is one that is supported by the
794  // device.
795  if (WmipFindGuid(WmiLibInfo->GuidList,
796  WmiLibInfo->GuidCount,
797  (LPGUID)irpStack->Parameters.WMI.DataPath,
798  &guidIndex,
799  &instanceCount))
800  {
801  status = STATUS_SUCCESS;
802  } else {
803  status = STATUS_WMI_GUID_NOT_FOUND;
804  }
805 
806  if (NT_SUCCESS(status) &&
807  ((minorFunction == IRP_MN_QUERY_SINGLE_INSTANCE) ||
808  (minorFunction == IRP_MN_CHANGE_SINGLE_INSTANCE) ||
809  (minorFunction == IRP_MN_CHANGE_SINGLE_ITEM) ||
810  (minorFunction == IRP_MN_EXECUTE_METHOD)))
811  {
812  instanceIndex = ((PWNODE_SINGLE_INSTANCE)buffer)->InstanceIndex;
813
