299 SepReferenceLogonSession(
300  IN PLUID LogonId,
301  OUT PSEP_LOGON_SESSION_REFERENCES *ReturnSession
302  )
303 
304 /*++
305 
306 Routine Description:
307 
308  This routine increments the reference count of a logon session
309  tracking record.
310 
311 
312 
313 Arguments:
314 
315  LogonId - Pointer to the logon session ID whose logon track is
316  to be incremented.
317 
318  ReturnSession - The found session is returned here on success.
319 
320 Return Value:
321 
322  STATUS_SUCCESS - The reference count was successfully incremented.
323 
324  STATUS_NO_SUCH_LOGON_SESSION - The specified logon session doesn't
325  exist in the reference monitor's database.
326 
327 --*/
328 
329 {
330 
331  ULONG SessionArrayIndex;
332  PSEP_LOGON_SESSION_REFERENCES *Previous, Current;
333  ULONG Refs;
334 
335  PAGED_CODE();
336 
337  SessionArrayIndex = SepLogonSessionIndex( LogonId );
338 
339  Previous = &SepLogonSessions[ SessionArrayIndex ];
340 
341  //
342  // Protect modification of reference monitor database
343  //
344 
345  SepRmAcquireDbWriteLock(SessionArrayIndex);
346 
347 
348  //
349  // Now walk the list for our logon session array hash index.
350  //
351 
352  Current = *Previous;
353 
354  while (Current != NULL) {
355 
356  //
357  // If we found it, increment the reference count and return
358  //
359 
360  if (RtlEqualLuid( LogonId, &Current->LogonId) ) {
361 
362  Refs = InterlockedIncrement (&Current->ReferenceCount);
363 
364  SepRmReleaseDbWriteLock(SessionArrayIndex);
365 
366  *ReturnSession = Current;
367  return STATUS_SUCCESS;
368  }
369 
370  Current = Current->Next;
371  }
372 
373  SepRmReleaseDbWriteLock(SessionArrayIndex);
374 
375  //
376  // Bad news, someone asked us to increment the reference count of
377  // a logon session we didn't know existed. This might be a new
378  // token being created, so return an error status and let the caller
379  // decide if it warrants a bugcheck or not.
380  //
381 
382  return STATUS_NO_SUCH_LOGON_SESSION;
383 }

638  IN PLUID LogonId
639  )
640 
641 /*++
642 
643 Routine Description:
644 
645  This routine decrements the reference count of a logon session
646  tracking record.
647 
648  If the reference count is decremented to zero, then there is no
649  possibility for any more tokens to exist for the logon session.
650  In this case, the LSA is notified that a logon session has
651  terminated.
652 
653 
654 
655 Arguments:
656 
657  LogonId - Pointer to the logon session ID whose logon track is
658  to be decremented.
659 
660 Return Value:
661 
662  None.
663 
664 --*/
665 
666 {
667 
668  ULONG SessionArrayIndex;
669  PSEP_LOGON_SESSION_REFERENCES *Previous, Current;
670  PDEVICE_MAP pDeviceMap = NULL;
671  ULONG Refs;
672 
673 
674  PAGED_CODE();
675 
676  SessionArrayIndex = SepLogonSessionIndex( LogonId );
677 
678  Previous = &SepLogonSessions[ SessionArrayIndex ];
679 
680  //
681  // Protect modification of reference monitor database
682  //
683 
684  SepRmAcquireDbWriteLock(SessionArrayIndex);
685 
686 
687  //
688  // Now walk the list for our logon session array hash index.
689  //
690 
691  Current = *Previous;
692 
693  while (Current != NULL) {
694 
695  //
696  // If we found it, decrement the reference count and return
697  //
698 
699  if (RtlEqualLuid( LogonId, &Current->LogonId) ) {
700  Refs = InterlockedDecrement (&Current->ReferenceCount);
701  if (Refs == 0) {
702 
703  //
704  // Pull it from the list
705  //
706 
707  *Previous = Current->Next;
708 
709  //
710  // No longer need to protect our pointer to this
711  // record.
712  //
713 
714  SepRmReleaseDbWriteLock(SessionArrayIndex);
715 
716  //
717  // If the Device Map exist for this LUID,
718  // dereference the pointer to the Device Map
719  //
720  if (Current->pDeviceMap != NULL) {
721 
722  //
723  // Dereference our reference on the device map
724  // our reference should be the last reference,
725  // thus the system will delete the device map
726  // for the LUID
727  //
728  pDeviceMap = Current->pDeviceMap;
729  Current->pDeviceMap = NULL;
730  }
731 
732 
733  //
734  // Make all the contents of the LUID's device map temporary,
735  // so that the names go away from the Object Manager's
736  // namespace
737  // Remove our reference on the LUID's device map
738  //
739  if (pDeviceMap != NULL) {
740  SepCleanupLUIDDeviceMapDirectory( LogonId );
741  ObfDereferenceDeviceMap( pDeviceMap );
742  }
743 
744  //
745  // Asynchronoously inform file systems that this logon session
746  // is going away, if at least one FS expressed interest in this
747  // logon session.
748  //
749 
750  if (Current->Flags & SEP_TERMINATION_NOTIFY) {
751  SepInformFileSystemsOfDeletedLogon( LogonId );
752  }
753 
754  //
755  // Deallocate the logon session track record.
756  //
757 
758  ExFreePool( (PVOID)Current );
759 
760  //
761  // Inform the LSA about the deletion of this logon session.
762  //
763 
764  SepInformLsaOfDeletedLogon( LogonId );
765 
766 
767 
768  return;
769 
770  }
771 
772  //
773  // reference count was decremented, but not to zero.
774  //
775 
776  SepRmReleaseDbWriteLock(SessionArrayIndex);
777 
778  return;
779  }
780 
781  Previous = &Current->Next;
782  Current = *Previous;
783  }
784 
785  SepRmReleaseDbWriteLock(SessionArrayIndex);
786 
787  //
788  // Bad news, someone asked us to decrement the reference count of
789  // a logon session we didn't know existed.
790  //
791 
792  KeBugCheckEx( DEREF_UNKNOWN_LOGON_SESSION, 0, 0, 0, 0 );
793 
794  return;
795 
796 }
