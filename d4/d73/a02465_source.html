89 /*++
90 
91 Routine Description:
92 
93  This function is called to copy the specified context frame to the
94  specified exception and trap frames for the continue system service.
95 
96 Arguments:
97 
98  ContextRecord - Supplies a pointer to a context record.
99 
100  ExceptionFrame - Supplies a pointer to an exception frame.
101 
102  TrapFrame - Supplies a pointer to a trap frame.
103 
104 Return Value:
105 
106  STATUS_ACCESS_VIOLATION is returned if the context record is not readable
107  from user mode.
108 
109  STATUS_DATATYPE_MISALIGNMENT is returned if the context record is not
110  properly aligned.
111 
112  STATUS_SUCCESS is returned if the context frame is copied successfully
113  to the specified exception and trap frames.
114 
115 --*/
116 
117 {
118 
119  KIRQL OldIrql;
120  NTSTATUS Status;
121 
122  //
123  // Synchronize with other context operations.
124  //
125  // If the current IRQL is less than APC_LEVEL, then raise IRQL to APC level.
126  //
127 
128  OldIrql = KeGetCurrentIrql();
129  if (OldIrql < APC_LEVEL) {
130  KfRaiseIrql(APC_LEVEL);
131  }
132 
133  //
134  // If the previous mode was not kernel mode, then use wrapper function
135  // to copy context to kernel frames. Otherwise, copy context to kernel
136  // frames directly.
137  //
138 
139  Status = STATUS_SUCCESS;
140  if (KeGetPreviousMode() != KernelMode) {
141  try {
142  KiContinuePreviousModeUser(ContextRecord,
143  ExceptionFrame,
144  TrapFrame);
145 
146  } except(EXCEPTION_EXECUTE_HANDLER) {
147  Status = GetExceptionCode();
148  }
149 
150  } else {
151  KeContextToKframes(TrapFrame,
152  ExceptionFrame,
153  ContextRecord,
154  ContextRecord->ContextFlags,
155  KernelMode);
156  }
157 
158  //
159  // If the old IRQL was less than APC level, then lower the IRQL to its
160  // previous value.
161  //
162 
163  if (OldIrql < APC_LEVEL) {
164  KeLowerIrql(OldIrql);
165  }
166 
167  return Status;

181 /*++
182 
183 Routine Description:
184 
185  This function is called to raise an exception. The exception can be
186  raised as a first or second chance exception.
187 
188 Arguments:
189 
190  ExceptionRecord - Supplies a pointer to an exception record.
191 
192  ContextRecord - Supplies a pointer to a context record.
193 
194  ExceptionFrame - Supplies a pointer to an exception frame.
195 
196  TrapFrame - Supplies a pointer to a trap frame.
197 
198  FirstChance - Supplies a boolean value that specifies whether this is
199  the first (TRUE) or second (FALSE) chance for the exception.
200 
201 Return Value:
202 
203  STATUS_ACCESS_VIOLATION is returned if either the exception or the context
204  record is not readable from user mode.
205 
206  STATUS_DATATYPE_MISALIGNMENT is returned if the exception record or the
207  context record are not properly aligned.
208 
209  STATUS_INVALID_PARAMETER is returned if the number of exception parameters
210  is greater than the maximum allowable number of exception parameters.
211 
212  STATUS_SUCCESS is returned if the exception is dispatched and handled.
213 
214 --*/
215 
216 {
217 
218  CONTEXT ContextRecord2;
219  EXCEPTION_RECORD ExceptionRecord2;
220  ULONG Length;
221  ULONG Parameters;
222  KPROCESSOR_MODE PreviousMode;
223 
224  //
225  // Get the previous processor mode and probe the specified exception and
226  // context records if necessary.
227  //
228 
229  PreviousMode = KeGetPreviousMode();
230  if (PreviousMode != KernelMode) {
231  try {
232  ProbeForReadSmallStructure(ContextRecord,
233  sizeof(CONTEXT),
234  sizeof(UCHAR));
235 
236  Parameters = ProbeAndReadUlong(&ExceptionRecord->NumberParameters);
237  if (Parameters > EXCEPTION_MAXIMUM_PARAMETERS) {
238  return STATUS_INVALID_PARAMETER;
239  }
240 
241  //
242  // The exception record structure is defined unlike others with
243  // trailing information as being its maximum size rather than
244  // just a single trailing element.
245  //
246 
247  Length = FIELD_OFFSET(EXCEPTION_RECORD, ExceptionInformation[Parameters]);
248 
249  __assume(Length != 0);
250 
251  ProbeForRead(ExceptionRecord, Length, sizeof(UCHAR));
252 
253  //
254  // Copy the exception and context record to local storage so an
255  // access violation cannot occur during exception dispatching.
256  //
257 
258  RtlCopyMemory(&ContextRecord2, ContextRecord, sizeof(CONTEXT));
259  RtlCopyMemory(&ExceptionRecord2, ExceptionRecord, Length);
260  ContextRecord = &ContextRecord2;
261  ExceptionRecord = &ExceptionRecord2;
262 
263  //
264  // Make sure the number of parameter is correct in the copied
265  // exception record.
266  //
267 
268  ExceptionRecord->NumberParameters = Parameters;
269 
270  } except(EXCEPTION_EXECUTE_HANDLER) {
271  return GetExceptionCode();
272  }
273  }
274 
275  //
276  // Move information from the context record to the exception and trap
277  // frames.
278  //
279 
280  KeContextToKframes(TrapFrame,
281  ExceptionFrame,
282  ContextRecord,
283  ContextRecord->ContextFlags,
284  PreviousMode);
285 
286  //
287  // Make sure the reserved bit is clear in the exception code and
288  // perform exception dispatching.
289  //
290  // N.B. The reserved bit is used to differentiate internally generated
291  // codes from codes generated by application programs.
292  //
293 
294  ExceptionRecord->ExceptionCode &= ~KI_EXCEPTION_INTERNAL;
295  KiDispatchException(ExceptionRecord,
296  ExceptionFrame,
297  TrapFrame,
298  PreviousMode,
299  FirstChance);
300 
301  return STATUS_SUCCESS;
302 }
