39 /*++
40 
41 Routine Description:
42 
43  Open regkey KeyName, read the value specified by ValueName
44  and return the value.
45 
46 Arguments:
47 
48  KeyName - name of key to open
49 
50  ValueName - name of value to read
51 
52  ValueType - type of value to read (REG_DWORD etc.)
53 
54  ValueLength - size in bytes of the value to read
55 
56  ValueBuffer - pointer to returned value
57 
58  LengthRequired - if the passed buffer is not sufficient to hold
59  the value, this param will return the actual size
60  in bytes required.
61 
62 Return Value:
63 
64  NTSTATUS - Standard Nt Result Code
65 
66 Notes:
67 
68 --*/
69 {
70  UNICODE_STRING usKey, usValue;
71  OBJECT_ATTRIBUTES ObjectAttributes = { 0 };
72 
73  //
74  // we will read-in data up to 64 bytes in stack buffer
75  //
76 
77  CHAR KeyInfo[sizeof(KEY_VALUE_PARTIAL_INFORMATION) + 64];
78  PKEY_VALUE_PARTIAL_INFORMATION pKeyInfo;
79  HANDLE hKey = NULL;
80  NTSTATUS Status = STATUS_SUCCESS;
81  NTSTATUS CloseStatus;
82  ULONG ResultLength;
83 
84  PAGED_CODE();
85 
86  RtlInitUnicodeString( &usKey, KeyName );
87 
88  InitializeObjectAttributes(
89  &ObjectAttributes,
90  &usKey,
91  OBJ_CASE_INSENSITIVE | OBJ_KERNEL_HANDLE,
92  NULL,
93  NULL
94  );
95 
96  Status = ZwOpenKey(
97  &hKey,
98  KEY_QUERY_VALUE | OBJ_KERNEL_HANDLE,
99  &ObjectAttributes
100  );
101 
102  if (NT_SUCCESS( Status ))
103  {
104  RtlInitUnicodeString( &usValue, ValueName );
105 
106  Status = ZwQueryValueKey(
107  hKey,
108  &usValue,
109  KeyValuePartialInformation,
110  KeyInfo,
111  sizeof(KeyInfo),
112  &ResultLength
113  );
114 
115  if (NT_SUCCESS( Status ))
116  {
117  pKeyInfo = (PKEY_VALUE_PARTIAL_INFORMATION)KeyInfo;
118 
119  if (( pKeyInfo->Type == ValueType) &&
120  ( pKeyInfo->DataLength == ValueLength ))
121  {
122  switch (ValueType)
123  {
124  case REG_DWORD:
125  *((PULONG)ValueBuffer) = *((PULONG) (pKeyInfo->Data));
126  break;
127 
128  case REG_BINARY:
129  RtlCopyMemory( ValueBuffer, pKeyInfo->Data, ValueLength );
130  break;
131 
132  default:
133  Status = STATUS_INVALID_PARAMETER;
134  break;
135  }
136  }
137  else
138  {
139  Status = STATUS_OBJECT_TYPE_MISMATCH;
140  }
141  }
142 
143  CloseStatus = ZwClose(hKey);
144 
145  ASSERT( NT_SUCCESS( CloseStatus ));
146  }
147 
148  return Status;
149 }
