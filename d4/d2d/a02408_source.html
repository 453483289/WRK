41 /*++
42 
43 Routine Description:
44 
45  This function initializes a kernel APC object. The thread, kernel
46  routine, and optionally a normal routine, processor mode, and normal
47  context parameter are stored in the APC object.
48 
49 Arguments:
50 
51  Apc - Supplies a pointer to a control object of type APC.
52 
53  Thread - Supplies a pointer to a dispatcher object of type thread.
54 
55  Environment - Supplies the environment in which the APC will execute.
56  Valid values for this parameter are: OriginalApcEnvironment,
57  AttachedApcEnvironment, CurrentApcEnvironment, or InsertApcEnvironment
58 
59  KernelRoutine - Supplies a pointer to a function that is to be
60  executed at IRQL APC_LEVEL in kernel mode.
61 
62  RundownRoutine - Supplies an optional pointer to a function that is to be
63  called if the APC is in a thread's APC queue when the thread terminates.
64 
65  NormalRoutine - Supplies an optional pointer to a function that is
66  to be executed at IRQL 0 in the specified processor mode. If this
67  parameter is not specified, then the ProcessorMode and NormalContext
68  parameters are ignored.
69 
70  ApcMode - Supplies the processor mode in which the function specified
71  by the NormalRoutine parameter is to be executed.
72 
73  NormalContext - Supplies a pointer to an arbitrary data structure which is
74  to be passed to the function specified by the NormalRoutine parameter.
75 
76 Return Value:
77 
78  None.
79 
80 --*/
81 
82 {
83 
84  ASSERT(Environment <= InsertApcEnvironment);
85 
86  //
87  // Initialize standard control object header.
88  //
89 
90  Apc->Type = ApcObject;
91  Apc->Size = sizeof(KAPC);
92 
93  //
94  // Initialize the APC environment, thread address, kernel routine address,
95  // rundown routine address, normal routine address, processor mode, and
96  // normal context parameter. If the normal routine address is null, then
97  // the processor mode is defaulted to KernelMode and the APC is a special
98  // APC. Otherwise, the processor mode is taken from the argument list.
99  //
100 
101  if (Environment == CurrentApcEnvironment) {
102  Apc->ApcStateIndex = Thread->ApcStateIndex;
103 
104  } else {
105 
106  ASSERT((Environment <= Thread->ApcStateIndex) || (Environment == InsertApcEnvironment));
107 
108  Apc->ApcStateIndex = (CCHAR)Environment;
109  }
110 
111  Apc->Thread = Thread;
112  Apc->KernelRoutine = KernelRoutine;
113  Apc->RundownRoutine = RundownRoutine;
114  Apc->NormalRoutine = NormalRoutine;
115  if (ARGUMENT_PRESENT(NormalRoutine)) {
116  Apc->ApcMode = ApcMode;
117  Apc->NormalContext = NormalContext;
118 
119  } else {
120  Apc->ApcMode = KernelMode;
121  Apc->NormalContext = NIL;
122  }
123 
124  Apc->Inserted = FALSE;
125  return;
126 }

129 KeFlushQueueApc (
130  __inout PKTHREAD Thread,
131  __in KPROCESSOR_MODE ApcMode
132  )
133 
134 /*++
135 
136 Routine Description:
137 
138  This function flushes the APC queue selected by the specified processor
139  mode for the specified thread. An APC queue is flushed by removing the
140  listhead from the list, scanning the APC entries in the list, setting
141  their inserted variables to FALSE, and then returning the address of the
142  doubly linked list as the function value.
143 
144 Arguments:
145 
146  Thread - Supplies a pointer to a dispatcher object of type thread.
147 
148  ApcMode - Supplies the processor mode of the APC queue that is to
149  be flushed.
150 
151 Return Value:
152 
153  The address of the first entry in the list of APC objects that were flushed
154  from the specified APC queue.
155 
156 --*/
157 
158 {
159 
160  PKAPC Apc;
161  PLIST_ENTRY FirstEntry;
162  KLOCK_QUEUE_HANDLE LockHandle;
163  PLIST_ENTRY NextEntry;
164 
165  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
166 
167  //
168  // If the APC mode is user mode, then acquire the thread APC queue lock
169  // to ensure that no further APCs are queued after a possible setting of
170  // the thread APC queueable state.
171  //
172 
173  if (ApcMode == UserMode) {
174  KeAcquireInStackQueuedSpinLockRaiseToSynch(&Thread->ApcQueueLock, &LockHandle);
175  if (IsListEmpty(&Thread->ApcState.ApcListHead[ApcMode])) {
176  KeReleaseInStackQueuedSpinLock(&LockHandle);
177  return NULL;
178  }
179 
180  } else {
181  if (IsListEmpty(&Thread->ApcState.ApcListHead[ApcMode])) {
182  return NULL;
183 
184  } else {
185  KeAcquireInStackQueuedSpinLockRaiseToSynch(&Thread->ApcQueueLock, &LockHandle);
186  }
187  }
188 
189  //
190  // Get address of first APC in the list and check if the list is
191  // empty or contains entries that should be flushed. If entries
192  // should be flushed, then scan the list of APC objects and set their
193  // inserted state to FALSE.
194  //
195 
196  FirstEntry = Thread->ApcState.ApcListHead[ApcMode].Flink;
197  if (FirstEntry == &Thread->ApcState.ApcListHead[ApcMode]) {
198  FirstEntry = (PLIST_ENTRY)NULL;
199 
200  } else {
201  RemoveEntryList(&Thread->ApcState.ApcListHead[ApcMode]);
202  NextEntry = FirstEntry;
203  do {
204  Apc = CONTAINING_RECORD(NextEntry, KAPC, ApcListEntry);
205  Apc->Inserted = FALSE;
206  NextEntry = NextEntry->Flink;
207  } while (NextEntry != FirstEntry);
208 
209  //
210  // Reinitialize the header so the current thread may safely attach
211  // to another process.
212  //
213 
214  InitializeListHead(&Thread->ApcState.ApcListHead[ApcMode]);
215  }
216 
217  //
218  // Unlock the thread APC queue lock, lower IRQL to its previous value,
219  // and return address of the first entry in list of APC objects that
220  // were flushed.
221  //
222 
223  KeReleaseInStackQueuedSpinLock(&LockHandle);
224  return FirstEntry;
225 }

228 KeInsertQueueApc (

230  __in_opt PVOID SystemArgument1,
231  __in_opt PVOID SystemArgument2,
232  __in KPRIORITY Increment
233  )

235 /*++
236 
237 Routine Description:
238 
239  This function inserts an APC object into the APC queue specifed by the
240  thread and processor mode fields of the APC object. If the APC object
241  is already in an APC queue or APC queuing is disabled, then no operation
242  is performed. Otherwise the APC object is inserted in the specified queue
243  and appropriate scheduling decisions are made.
244 
245 Arguments:
246 
247  Apc - Supplies a pointer to a control object of type APC.
248 
249  SystemArgument1, SystemArgument2 - Supply a set of two arguments that
250  contain untyped data provided by the executive.
251 
252  Increment - Supplies the priority increment that is to be applied if
253  queuing the APC causes a thread wait to be satisfied.
254 
255 Return Value:
256 
257  If the APC object is already in an APC queue or APC queuing is disabled,
258  then a value of FALSE is returned. Otherwise a value of TRUE is returned.
259 
260 --*/
261 
262 {
263 
264  BOOLEAN Inserted;
265  KLOCK_QUEUE_HANDLE LockHandle;
266  PRKTHREAD Thread;
267 
268  ASSERT_APC(Apc);
269  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
270 
271  //
272  // Raise IRQL to SYNCH_LEVEL and acquire the thread APC queue lock.
273  //
274 
275  Thread = Apc->Thread;
276  KeAcquireInStackQueuedSpinLockRaiseToSynch(&Thread->ApcQueueLock, &LockHandle);
277 
278  //
279  // If APC queuing is disabled or the APC is already inserted, then set
280  // inserted to FALSE. Otherwise, set the system parameter values in the
281  // APC object, insert the APC in the thread APC queue, and set inserted to
282  // true.
283  //
284 
285  if ((Thread->ApcQueueable == FALSE) ||
286  (Apc->Inserted == TRUE)) {
287  Inserted = FALSE;
288 
289  } else {
290  Apc->Inserted = TRUE;
291  Apc->SystemArgument1 = SystemArgument1;
292  Apc->SystemArgument2 = SystemArgument2;
293  KiInsertQueueApc(Apc, Increment);
294  Inserted = TRUE;
295  }
296 
297  //
298  // Unlock the thread APC queue lock, exit the scheduler, and return
299  // whether the APC was inserted.
300  //
301 
302  KeReleaseInStackQueuedSpinLockFromDpcLevel(&LockHandle);
303  KiExitDispatcher(LockHandle.OldIrql);
304  return Inserted;
305 }

308 KeRemoveQueueApc (
309  __inout PKAPC Apc
310  )

312 /*++
313 
314 Routine Description:
315 
316  This function removes an APC object from an APC queue. If the APC object
317  is not in an APC queue, then no operation is performed. Otherwise the
318  APC object is removed from its current queue and its inserted state is
319  set FALSE.
320 
321 Arguments:
322 
323  Apc - Supplies a pointer to a control object of type APC.
324 
325 Return Value:
326 
327  If the APC object is not in an APC queue, then a value of FALSE is returned.
328  Otherwise a value of TRUE is returned.
329 
330 --*/
331 
332 {
333 
334  PKAPC_STATE ApcState;
335  BOOLEAN Inserted;
336  KLOCK_QUEUE_HANDLE LockHandle;
337  PRKTHREAD Thread;
338 
339  ASSERT_APC(Apc);
340  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
341 
342  //
343  // Raise IRQL to SYNCH_LEVEL and acquire the thread APC queue lock.
344  //
345 
346  Thread = Apc->Thread;
347  KeAcquireInStackQueuedSpinLockRaiseToSynch(&Thread->ApcQueueLock, &LockHandle);
348 
349  //
350  // If the APC object is in an APC queue, then remove it from the queue
351  // and set its inserted state to FALSE. If the queue becomes empty, set
352  // the APC pending state to FALSE.
353  //
354 
355  Inserted = Apc->Inserted;
356  if (Inserted != FALSE) {
357  Apc->Inserted = FALSE;
358  ApcState = Thread->ApcStatePointer[Apc->ApcStateIndex];
359  KiLockDispatcherDatabaseAtSynchLevel();
360  if (RemoveEntryList(&Apc->ApcListEntry) != FALSE) {
361  if (Apc->ApcMode == KernelMode) {
362  ApcState->KernelApcPending = FALSE;
363 
364  } else {
365  ApcState->UserApcPending = FALSE;
366  }
367  }
368 
369  KiUnlockDispatcherDatabaseFromSynchLevel();
370  }
371 
372  //
373  // Release the thread APC queue lock, lower IRQL to its previous value,
374  // and return whether an APC object was removed from the APC queue.
375  //
376 
377  KeReleaseInStackQueuedSpinLock(&LockHandle);
378  return Inserted;
379 }
