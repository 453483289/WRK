35 /*++
36 
37 Routine Description:
38 
39  This function sends the specified API message over the specified
40  port. It is the caller's responsibility to format the API message
41  prior to calling this function.
42 
43  If the SuspendProcess flag is supplied, then all threads in the
44  calling process are first suspended. Upon receipt of the reply
45  message, the threads are resumed.
46 
47 Arguments:
48 
49  ApiMsg - Supplies the API message to send.
50 
51  SuspendProcess - A flag that if set to true, causes all of the
52  threads in the process to be suspended prior to the call,
53  and resumed upon receipt of a reply.
54 
55 Return Value:
56 
57  NTSTATUS.
58 
59 --*/
60 
61 {
62  NTSTATUS st;
63  PEPROCESS Process;
64 
65  PAGED_CODE();
66 
67  if ( SuspendProcess ) {
68  SuspendProcess = DbgkpSuspendProcess();
69  }
70 
71  ApiMsg->ReturnedStatus = STATUS_PENDING;
72 
73  Process = PsGetCurrentProcess();
74 
75  PS_SET_BITS (&Process->Flags, PS_PROCESS_FLAGS_CREATE_REPORTED);
76 
77  st = DbgkpQueueMessage (Process, PsGetCurrentThread (), ApiMsg, 0, NULL);
78 
79  ZwFlushInstructionCache (NtCurrentProcess (), NULL, 0);
80  if ( SuspendProcess ) {
81  DbgkpResumeProcess();
82  }
83 
84  return st;
85 }

153 DECLSPEC_NOINLINE
154 BOOLEAN
155 DbgkForwardException(
156  IN PEXCEPTION_RECORD ExceptionRecord,
157  IN BOOLEAN DebugException,
158  IN BOOLEAN SecondChance
159  )
160 
161 /*++
162 
163 Routine Description:
164 
165  This function is called forward an exception to the calling process's
166  debug or subsystem exception port.
167 
168 Arguments:
169 
170  ExceptionRecord - Supplies a pointer to an exception record.
171 
172  DebugException - Supplies a boolean variable that specifies whether
173  this exception is to be forwarded to the process's
174  DebugPort(TRUE), or to its ExceptionPort(FALSE).
175 
176 Return Value:
177 
178  TRUE - The process has a DebugPort or an ExceptionPort, and the reply
179  received from the port indicated that the exception was handled.
180 
181  FALSE - The process either does not have a DebugPort or
182  ExceptionPort, or the process has a port, but the reply received
183  from the port indicated that the exception was not handled.
184 
185 --*/
186 
187 {
188  PEPROCESS Process;
189  PVOID Port;
190  DBGKM_APIMSG m;
191  PDBGKM_EXCEPTION args;
192  NTSTATUS st;
193  BOOLEAN LpcPort;
194 
195  PAGED_CODE();
196 
197  args = &m.u.Exception;
198 
199  //
200  // Initialize the debug LPC message with default information.
201  //
202 
203  DBGKM_FORMAT_API_MSG(m,DbgKmExceptionApi,sizeof(*args));
204 
205  //
206  // Get the address of the destination LPC port.
207  //
208 
209  Process = PsGetCurrentProcess();
210  if (DebugException) {
211  if (PsGetCurrentThread()->CrossThreadFlags&PS_CROSS_THREAD_FLAGS_HIDEFROMDBG) {
212  Port = NULL;
213  } else {
214  Port = Process->DebugPort;
215  }
216  LpcPort = FALSE;
217  } else {
218  Port = Process->ExceptionPort;
219  m.h.u2.ZeroInit = LPC_EXCEPTION;
220  LpcPort = TRUE;
221  }
222 
223  //
224  // If the destination LPC port address is NULL, then return FALSE.
225  //
226 
227  if (Port == NULL) {
228  return FALSE;
229  }
230 
231  //
232  // Fill in the remainder of the debug LPC message.
233  //
234 
235  args->ExceptionRecord = *ExceptionRecord;
236  args->FirstChance = !SecondChance;
237 
238  //
239  // Send the debug message to the destination LPC port.
240  //
241 
242  if (LpcPort) {
243  st = DbgkpSendApiMessageLpc(&m,Port,DebugException);
244  } else {
245  st = DbgkpSendApiMessage(&m,DebugException);
246  }
247 
248 
249  //
250  // If the send was not successful, then return a FALSE indicating that
251  // the port did not handle the exception. Otherwise, if the debug port
252  // is specified, then look at the return status in the message.
253  //
254 
255  if (!NT_SUCCESS(st) ||
256  ((DebugException) &&
257  (m.ReturnedStatus == DBG_EXCEPTION_NOT_HANDLED || !NT_SUCCESS(m.ReturnedStatus)))) {
258  return FALSE;
259 
260  } else {
261  return TRUE;
262  }
263 }
