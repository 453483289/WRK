37 /*++
38 
39 Routine Description:
40 
41  This routine moves the selected contents of the specified trap and
42  exception frames into the specified context frame according to the
43  specified context flags.
44 
45 Arguments:
46 
47  TrapFrame - Supplies a pointer to a trap frame from which volatile
48  context should be copied into the context record.
49 
50  ExceptionFrame - Supplies a pointer to an exception frame from which
51  context should be copied into the context record.
52 
53  ContextRecord - Supplies a pointer to the context frame that receives
54  the context copied from the trap and exception frames.
55 
56 Return Value:
57 
58  None.
59 
60 --*/
61 
62 {
63 
64  ULONG ContextFlags;
65  KIRQL OldIrql;
66 
67  //
68  // Raise IRQL to APC_LEVEL to guarantee that a consistent set of context
69  // is transferred from the trap and exception frames.
70  //
71 
72  OldIrql = KeGetCurrentIrql();
73  if (OldIrql < APC_LEVEL) {
74  KfRaiseIrql(APC_LEVEL);
75  }
76 
77  //
78  // Set control information if specified.
79  //
80 
81  ContextFlags = ContextRecord->ContextFlags;
82  if ((ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL) {
83 
84  //
85  // Set registers RIP, CS, RSP, SS, and EFlags.
86  //
87 
88  ContextRecord->Rip = TrapFrame->Rip;
89  ContextRecord->SegCs = TrapFrame->SegCs;
90  ContextRecord->SegSs = TrapFrame->SegSs;
91  ContextRecord->Rsp = TrapFrame->Rsp;
92  ContextRecord->EFlags = TrapFrame->EFlags;
93  }
94 
95  //
96  // Set segment register contents if specified.
97  //
98 
99  if ((ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS) {
100 
101  //
102  // Set segment registers GS, FS, ES, DS.
103  //
104 
105  ContextRecord->SegDs = KGDT64_R3_DATA | RPL_MASK;
106  ContextRecord->SegEs = KGDT64_R3_DATA | RPL_MASK;
107  ContextRecord->SegFs = KGDT64_R3_CMTEB | RPL_MASK;
108  ContextRecord->SegGs = KGDT64_R3_DATA | RPL_MASK;
109  }
110 
111  //
112  // Set integer register contents if specified.
113  //
114 
115  if ((ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER) {
116 
117  //
118  // Set integer registers RAX, RCX, RDX, RSI, RDI, R8, R9, R10, RBX,
119  // RBP, R11, R12, R13, R14, and R15.
120  //
121 
122  ContextRecord->Rax = TrapFrame->Rax;
123  ContextRecord->Rcx = TrapFrame->Rcx;
124  ContextRecord->Rdx = TrapFrame->Rdx;
125  ContextRecord->R8 = TrapFrame->R8;
126  ContextRecord->R9 = TrapFrame->R9;
127  ContextRecord->R10 = TrapFrame->R10;
128  ContextRecord->R11 = TrapFrame->R11;
129  ContextRecord->Rbp = TrapFrame->Rbp;
130 
131  ContextRecord->Rbx = ExceptionFrame->Rbx;
132  ContextRecord->Rdi = ExceptionFrame->Rdi;
133  ContextRecord->Rsi = ExceptionFrame->Rsi;
134  ContextRecord->R12 = ExceptionFrame->R12;
135  ContextRecord->R13 = ExceptionFrame->R13;
136  ContextRecord->R14 = ExceptionFrame->R14;
137  ContextRecord->R15 = ExceptionFrame->R15;
138  }
139 
140  //
141  // Set floating point context if specified.
142  //
143  //
144 
145  if ((ContextFlags & CONTEXT_FLOATING_POINT) == CONTEXT_FLOATING_POINT) {
146 
147  //
148  // If the specified mode is user, then save the legacy floating
149  // point state.
150  //
151 
152  if ((TrapFrame->SegCs & MODE_MASK) == UserMode) {
153  KeSaveLegacyFloatingPointState(&ContextRecord->FltSave);
154  }
155

157  // Set XMM registers Xmm0-Xmm15 and the XMM CSR contents.
158  //
159 
160  ContextRecord->Xmm0 = TrapFrame->Xmm0;
161  ContextRecord->Xmm1 = TrapFrame->Xmm1;
162  ContextRecord->Xmm2 = TrapFrame->Xmm2;
163  ContextRecord->Xmm3 = TrapFrame->Xmm3;
164  ContextRecord->Xmm4 = TrapFrame->Xmm4;
165  ContextRecord->Xmm5 = TrapFrame->Xmm5;
166 
167  ContextRecord->Xmm6 = ExceptionFrame->Xmm6;
168  ContextRecord->Xmm7 = ExceptionFrame->Xmm7;
169  ContextRecord->Xmm8 = ExceptionFrame->Xmm8;
170  ContextRecord->Xmm9 = ExceptionFrame->Xmm9;

172  ContextRecord->Xmm11 = ExceptionFrame->Xmm11;

179  }
180 
181  //
182  //
183  // Set debug register contents if requested.
184  //
185 
186  if ((ContextFlags & CONTEXT_DEBUG_REGISTERS) == CONTEXT_DEBUG_REGISTERS) {
187 
188  //
189  // Set the debug registers DR0, DR1, DR2, DR3, DR6, and DR7.
190  //
191 
192  if ((TrapFrame->Dr7 & DR7_ACTIVE) != 0) {
193  ContextRecord->Dr0 = TrapFrame->Dr0;
194  ContextRecord->Dr1 = TrapFrame->Dr1;
195  ContextRecord->Dr2 = TrapFrame->Dr2;
196  ContextRecord->Dr3 = TrapFrame->Dr3;
197  ContextRecord->Dr6 = TrapFrame->Dr6;
198  ContextRecord->Dr7 = TrapFrame->Dr7;
199  if ((TrapFrame->Dr7 & DR7_LAST_BRANCH) != 0) {
200  ContextRecord->LastBranchToRip = TrapFrame->LastBranchToRip;
201  ContextRecord->LastBranchFromRip = TrapFrame->LastBranchFromRip;
202  ContextRecord->LastExceptionToRip = TrapFrame->LastExceptionToRip;
203  ContextRecord->LastExceptionFromRip = TrapFrame->LastExceptionFromRip;
204 
205  } else {
206  ContextRecord->LastBranchToRip = 0;
207  ContextRecord->LastBranchFromRip = 0;
208  ContextRecord->LastExceptionToRip = 0;
209  ContextRecord->LastExceptionFromRip = 0;
210  }
211 
212  } else {
213  ContextRecord->Dr0 = 0;
214  ContextRecord->Dr1 = 0;
215  ContextRecord->Dr2 = 0;
216  ContextRecord->Dr3 = 0;
217  ContextRecord->Dr6 = 0;
218  ContextRecord->Dr7 = 0;
219  ContextRecord->LastBranchToRip = 0;
220  ContextRecord->LastBranchFromRip = 0;
221  ContextRecord->LastExceptionToRip = 0;
222  ContextRecord->LastExceptionFromRip = 0;
223  }
224  }
225 
226  //
227  // Lower IRQL to its previous value.
228  //
229 
230  if (OldIrql < APC_LEVEL) {
231  KeLowerIrql(OldIrql);
232  }
233 
234  return;
235 }

246 /*++
247 
248 Routine Description:
249 
250  This routine moves the selected contents of the specified context frame
251  into the specified trap and exception frames according to the specified
252  context flags.
253 
254 Arguments:
255 
256  TrapFrame - Supplies a pointer to a trap frame that receives the volatile
257  context from the context record.
258 
259  ExceptionFrame - Supplies a pointer to an exception frame that receives
260  the nonvolatile context from the context record.
261 
262  ContextRecord - Supplies a pointer to a context frame that contains the
263  context that is to be copied into the trap and exception frames.
264 
265  ContextFlags - Supplies the set of flags that specify which parts of the
266  context frame are to be copied into the trap and exception frames.
267 
268  PreviousMode - Supplies the processor mode for which the exception and
269  trap frames are being built.
270 
271 Return Value:
272 
273  If the context operation is a set context and the legacy floating state is
274  switched for the current thread, then the address of the legacy floating
275  save area is returned as the function value. Otherwise, NULL is returned.
276 
277 --*/
278 
279 {
280 
281  PXMM_SAVE_AREA32 XmmSaveArea;
282 
283  //
284  // Set control information if specified.
285  //
286 
287  XmmSaveArea = NULL;
288  if ((ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL) {
289  TrapFrame->EFlags = SANITIZE_EFLAGS(ContextRecord->EFlags, PreviousMode);
290  TrapFrame->Rip = ContextRecord->Rip;
291  TrapFrame->Rsp = ContextRecord->Rsp;
292  }
293 
294  //
295  // The segment registers DS, ES, FS, and GS are never restored from saved
296  // data. However, SS and CS are restored from the trap frame. Make sure
297  // that these segment registers have the proper values.
298  //
299 
300  if (PreviousMode == UserMode) {
301  TrapFrame->SegSs = KGDT64_R3_DATA | RPL_MASK;
302  if (ContextRecord->SegCs != (KGDT64_R3_CODE | RPL_MASK)) {
303  TrapFrame->SegCs = KGDT64_R3_CMCODE | RPL_MASK;
304 
305  } else {
306  TrapFrame->SegCs = KGDT64_R3_CODE | RPL_MASK;
307  }
308 
309  } else {
310  TrapFrame->SegCs = KGDT64_R0_CODE;
311  TrapFrame->SegSs = KGDT64_R0_DATA;
312  }
313 
314  TrapFrame->Rip = SANITIZE_VA(TrapFrame->Rip, TrapFrame->SegCs, PreviousMode);
315 
316  //
317  // Set integer registers contents if specified.
318  //
319 
320  if ((ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER) {
321 
322  //
323  // Set integer registers RAX, RCX, RDX, RSI, RDI, R8, R9, R10, RBX,
324  // RBP, R11, R12, R13, R14, and R15.
325  //
326 
327  TrapFrame->Rax = ContextRecord->Rax;
328  TrapFrame->Rcx = ContextRecord->Rcx;
329  TrapFrame->Rdx = ContextRecord->Rdx;
330  TrapFrame->R8 = ContextRecord->R8;
331  TrapFrame->R9 = ContextRecord->R9;
332  TrapFrame->R10 = ContextRecord->R10;
333  TrapFrame->R11 = ContextRecord->R11;
334  TrapFrame->Rbp = ContextRecord->Rbp;
335 
336  ExceptionFrame->Rbx = ContextRecord->Rbx;
337  ExceptionFrame->Rsi = ContextRecord->Rsi;
338  ExceptionFrame->Rdi = ContextRecord->Rdi;
339  ExceptionFrame->R12 = ContextRecord->R12;
340  ExceptionFrame->R13 = ContextRecord->R13;
341  ExceptionFrame->R14 = ContextRecord->R14;
342  ExceptionFrame->R15 = ContextRecord->R15;
343  }
344 
345  //
346  // Set floating register contents if requested.
347  //
348 
349  if ((ContextFlags & CONTEXT_FLOATING_POINT) == CONTEXT_FLOATING_POINT) {
350 
351  //
352  // Set XMM registers Xmm0-Xmm15 and the XMM CSR contents.
353  //
354 
355  TrapFrame->Xmm0 = ContextRecord->Xmm0;
356  TrapFrame->Xmm1 = ContextRecord->Xmm1;
357  TrapFrame->Xmm2 = ContextRecord->Xmm2;
358  TrapFrame->Xmm3 = ContextRecord->Xmm3;
359  TrapFrame->Xmm4 = ContextRecord->Xmm4;
360  TrapFrame->Xmm5 = ContextRecord->Xmm5;
361

363  ExceptionFrame->Xmm7 = ContextRecord->Xmm7;
364  ExceptionFrame->Xmm8 = ContextRecord->Xmm8;
365  ExceptionFrame->Xmm9 = ContextRecord->Xmm9;
366  ExceptionFrame->Xmm10 = ContextRecord->Xmm10;
367  ExceptionFrame->Xmm11 = ContextRecord->Xmm11;
368  ExceptionFrame->Xmm12 = ContextRecord->Xmm12;
369  ExceptionFrame->Xmm13 = ContextRecord->Xmm13;
370  ExceptionFrame->Xmm14 = ContextRecord->Xmm14;
371  ExceptionFrame->Xmm15 = ContextRecord->Xmm15;
372

375  //
376

378

380  // If the specified mode is user, then set the legacy floating point

383  // Clear all reserved bits in legacy floating state.
384  //
385  // N.B. The legacy floating state is restored if and only if the
386  // request mode is user.
387  //

389  // state so it will get restored if the legacy state is
390  // restored.
391  //
392

394  XmmSaveArea = &ContextRecord->FltSave;
395  ContextRecord->FltSave.MxCsr = ReadMxCsr();

398  }

403  //
404

406 
407  //
408  // Set the debug registers DR0, DR1, DR2, DR3, DR6, and DR7.
409  //
410 
411  TrapFrame->Dr0 = SANITIZE_DRADDR(ContextRecord->Dr0, PreviousMode);
412  TrapFrame->Dr1 = SANITIZE_DRADDR(ContextRecord->Dr1, PreviousMode);
413  TrapFrame->Dr2 = SANITIZE_DRADDR(ContextRecord->Dr2, PreviousMode);
414  TrapFrame->Dr3 = SANITIZE_DRADDR(ContextRecord->Dr3, PreviousMode);
415  TrapFrame->Dr6 = 0;
416  TrapFrame->Dr7 = SANITIZE_DR7(ContextRecord->Dr7, PreviousMode);
417  TrapFrame->LastBranchToRip = ContextRecord->LastBranchToRip;
418  TrapFrame->LastBranchFromRip = ContextRecord->LastBranchFromRip;
419  TrapFrame->LastExceptionToRip = ContextRecord->LastExceptionToRip;
420  TrapFrame->LastExceptionFromRip = ContextRecord->LastExceptionFromRip;
421  if (PreviousMode != KernelMode) {
422  KeGetCurrentThread()->Header.DebugActive =
423  (BOOLEAN)((TrapFrame->Dr7 & DR7_ACTIVE) != 0);
424  }
425  }
426 
427  return XmmSaveArea;
428 }

430 VOID
431 KiDispatchException (
432  IN PEXCEPTION_RECORD ExceptionRecord,
433  IN PKEXCEPTION_FRAME ExceptionFrame,
434  IN PKTRAP_FRAME TrapFrame,
435  IN KPROCESSOR_MODE PreviousMode,
436  IN BOOLEAN FirstChance
437  )
438 
439 /*++
440 
441 Routine Description:
442 
443  This function is called to dispatch an exception to the proper mode and
444  to cause the exception dispatcher to be called. If the previous mode is
445  kernel, then the exception dispatcher is called directly to process the
446  exception. Otherwise the exception record, exception frame, and trap
447  frame contents are copied to the user mode stack. The contents of the
448  exception frame and trap are then modified such that when control is
449  returned, execution will commense in user mode in a routine which will
450  call the exception dispatcher.
451 
452 Arguments:
453 
454  ExceptionRecord - Supplies a pointer to an exception record.
455 
456  ExceptionFrame - Supplies a pointer to an exception frame. For NT386,
457  this should be NULL.
458 
459  TrapFrame - Supplies a pointer to a trap frame.
460 
461  PreviousMode - Supplies the previous processor mode.
462 
463  FirstChance - Supplies a boolean value that specifies whether this is
464  the first (TRUE) or second (FALSE) chance for the exception.
465 
466 Return Value:
467 
468  None.
469 
470 --*/
471 
472 {
473 
474  CONTEXT ContextRecord;
475  BOOLEAN DebugService;
476  EXCEPTION_RECORD ExceptionRecord1;
477  BOOLEAN ExceptionWasForwarded = FALSE;
478  ULONG64 FaultingRsp;
479  PMACHINE_FRAME MachineFrame;
480  ULONG64 UserStack1;
481  ULONG64 UserStack2;
482 
483  //
484  // Move machine state from trap and exception frames to a context frame
485  // and increment the number of exceptions dispatched.
486  //
487 
488  KeGetCurrentPrcb()->KeExceptionDispatchCount += 1;
489  ContextRecord.ContextFlags = CONTEXT_FULL | CONTEXT_DEBUG_REGISTERS | CONTEXT_SEGMENTS;
490  KeContextFromKframes(TrapFrame, ExceptionFrame, &ContextRecord);
491 
492  //
493  // If the exception is a break point, then convert the break point to a
494  // fault.
495  //
496 
497  if (ExceptionRecord->ExceptionCode == STATUS_BREAKPOINT) {
498  ContextRecord.Rip -= 1;
499  }
500 
501  //
502  // If the exception is an internal general protect fault, invalid opcode,
503  // or integer divide by zero, then attempt to resolve the problem without
504  // actually raising an exception.
505  //
506 
507  if (KiPreprocessFault(ExceptionRecord,
508  TrapFrame,
509  &ContextRecord,
510  PreviousMode) != FALSE) {
511 
512  goto Handled1;
513  }
514 
515  //
516  // Select the method of handling the exception based on the previous mode.
517  //
518 
519  if (PreviousMode == KernelMode) {
520 
521  //
522  // Previous mode was kernel.
523  //
524  // If the kernel debugger is active, then give the kernel debugger
525  // the first chance to handle the exception. If the kernel debugger
526  // handles the exception, then continue execution. Otherwise, attempt
527  // to dispatch the exception to a frame based handler. If a frame
528  // based handler handles the exception, then continue execution.
529  //
530  // If a frame based handler does not handle the exception, give the
531  // kernel debugger a second chance, if it's present.
532  //
533  // If the exception is still unhandled call bugcheck.
534  //
535 
536  if (FirstChance != FALSE) {
537  if ((KiDebugRoutine)(TrapFrame,
538  ExceptionFrame,
539  ExceptionRecord,
540  &ContextRecord,
541  PreviousMode,
542  FALSE) != FALSE) {
543 
544  goto Handled1;
545  }
546 
547  //
548  // Kernel debugger didn't handle exception.
549  //
550  // If interrupts are disabled, then bugcheck.
551  //
552 
553  if (RtlDispatchException(ExceptionRecord, &ContextRecord) != FALSE) {
554  goto Handled1;
555  }
556  }
557 
558  //
559  // This is the second chance to handle the exception.
560  //
561 
562  if ((KiDebugRoutine)(TrapFrame,
563  ExceptionFrame,
564  ExceptionRecord,
565  &ContextRecord,
566  PreviousMode,
567  TRUE) != FALSE) {
568 
569  goto Handled1;
570  }
571 
572  KeBugCheckEx(KMODE_EXCEPTION_NOT_HANDLED,
573  ExceptionRecord->ExceptionCode,
574  (ULONG64)ExceptionRecord->ExceptionAddress,
575  ExceptionRecord->ExceptionInformation[0],
576  ExceptionRecord->ExceptionInformation[1]);
577 
578  } else {
579 
580  //
581  // Previous mode was user.
582  //
583  // If this is the first chance and the current process has a debugger
584  // port, then send a message to the debugger port and wait for a reply.
585  // If the debugger handles the exception, then continue execution. Else
586  // transfer the exception information to the user stack, transition to
587  // user mode, and attempt to dispatch the exception to a frame based
588  // handler. If a frame based handler handles the exception, then continue
589  // execution with the continue system service. Else execute the
590  // NtRaiseException system service with FirstChance == FALSE, which
591  // will call this routine a second time to process the exception.
592  //
593  // If this is the second chance and the current process has a debugger
594  // port, then send a message to the debugger port and wait for a reply.
595  // If the debugger handles the exception, then continue execution. Else
596  // if the current process has a subsystem port, then send a message to
597  // the subsystem port and wait for a reply. If the subsystem handles the
598  // exception, then continue execution. Else terminate the process.
599  //
600  // If the current process is a wow64 process, an alignment fault has
601  // occurred, and the AC bit is set in EFLAGS, then clear AC in EFLAGS
602  // and continue execution. Otherwise, attempt to resolve the exception.
603  //
604 
605  if ((PsGetCurrentProcess()->Wow64Process != NULL) &&
606  (ExceptionRecord->ExceptionCode == STATUS_DATATYPE_MISALIGNMENT) &&
607  ((TrapFrame->EFlags & EFLAGS_AC_MASK) != 0)) {
608 
609  TrapFrame->EFlags &= ~EFLAGS_AC_MASK;
610  goto Handled2;
611  }
612 
613  //
614  // If the exception happened while executing 32-bit code, then convert
615  // the exception to a wow64 exception. These codes are translated later
616  // by wow64.
617  //
618 
619  if ((ContextRecord.SegCs & 0xfff8) == KGDT64_R3_CMCODE) {
620 
621  switch (ExceptionRecord->ExceptionCode) {
622  case STATUS_BREAKPOINT:
623  ExceptionRecord->ExceptionCode = STATUS_WX86_BREAKPOINT;
624  break;
625 
626  case STATUS_SINGLE_STEP:
627  ExceptionRecord->ExceptionCode = STATUS_WX86_SINGLE_STEP;
628  break;
629  }
630 
631  //
632  // Clear the upper 32-bits of the stack address and 16-byte
633  // align the stack address.
634  //
635 
636  FaultingRsp = (ContextRecord.Rsp & 0xfffffff0UI64);
637 
638  } else {
639  FaultingRsp = ContextRecord.Rsp;
640  }
641 
642  if (FirstChance == TRUE) {
643 
644  //
645  // This is the first chance to handle the exception.
646  //
647  // If the current processor is not being debugged and user mode
648  // exceptions are not being ignored, or this is a debug service,
649  // then attempt to handle the exception via the kernel debugger.
650  //
651 
652 
653  DebugService = KdIsThisAKdTrap(ExceptionRecord,
654  &ContextRecord,
655  UserMode);
656 
657  if (((PsGetCurrentProcess()->DebugPort == NULL) &&
658  (KdIgnoreUmExceptions == FALSE)) ||
659  (DebugService == TRUE)) {
660 
661  //
662  // Attempt to handle the exception with the kernel debugger.
663  //
664 
665  if ((KiDebugRoutine)(TrapFrame,
666  ExceptionFrame,
667  ExceptionRecord,
668  &ContextRecord,
669  PreviousMode,
670  FALSE) != FALSE) {
671 
672  goto Handled1;
673  }
674  }
675 
676  if ((ExceptionWasForwarded == FALSE) &&
677  (DbgkForwardException(ExceptionRecord, TRUE, FALSE))) {
678 
679  goto Handled2;
680  }
681 
682  //
683  // Clear the trace flag in the trap frame so a spurious trace
684  // trap is guaranteed not to occur in the trampoline code.
685  //
686 
687  TrapFrame->EFlags &= ~EFLAGS_TF_MASK;
688 
689  //
690  // Transfer exception information to the user stack, transition
691  // to user mode, and attempt to dispatch the exception to a frame
692  // based handler.
693  //
694 
695  ExceptionRecord1.ExceptionCode = STATUS_ACCESS_VIOLATION;
696 
697  repeat:
698  try {
699 
700  //
701  // Compute address of aligned machine frame, compute address
702  // of exception record, compute address of context record,
703  // and probe user stack for writeability.
704  //
705 
706  MachineFrame =
707  (PMACHINE_FRAME)((FaultingRsp - sizeof(MACHINE_FRAME)) & ~STACK_ROUND);
708 
709  UserStack1 = (ULONG64)MachineFrame - EXCEPTION_RECORD_LENGTH;
710  UserStack2 = UserStack1 - CONTEXT_LENGTH;
711  ProbeForWriteSmallStructure((PVOID)UserStack2,
712  sizeof(MACHINE_FRAME) + EXCEPTION_RECORD_LENGTH + CONTEXT_LENGTH,
713  STACK_ALIGN);
714 
715  //
716  // Fill in machine frame information.
717  //
718 
719  MachineFrame->Rsp = FaultingRsp;
720  MachineFrame->Rip = ContextRecord.Rip;
721 
722  //
723  // Copy exception record to the user stack.
724  //
725 
726  *(PEXCEPTION_RECORD)UserStack1 = *ExceptionRecord;
727 
728  //
729  // Copy context record to the user stack.
730  //
731 
732  *(PCONTEXT)UserStack2 = ContextRecord;
733 
734  //
735  // Set the address of the new stack pointer in the current
736  // trap frame.
737  //
738 
739  TrapFrame->Rsp = UserStack2;
740 
741  //
742  // Set the user mode 64-bit code selector.
