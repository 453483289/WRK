27 NtLoadDriver (
28  __in PUNICODE_STRING DriverServiceName
29  )
30 
31 /*++
32 
33 Routine Description:
34 
35  This service dynamically loads a device or file system driver into
36  the currently running system. It requires that the caller have the
37  appropriate privilege to execute this service.
38 
39 Arguments:
40 
41  DriverServiceName - Specifies the name of the node in the registry
42  associated with the driver to be loaded.
43 
44 Return Value:
45 
46  The status returned is the final completion status of the load operation.
47 
48 --*/
49 
50 {
51  KPROCESSOR_MODE requestorMode;
52  UNICODE_STRING driverServiceName;
53  PWCHAR nameBuffer = (PWCHAR) NULL;
54  LOAD_PACKET loadPacket;
55  PETHREAD CurrentThread;
56 
57  PAGED_CODE();
58 
59  //
60  // Get the previous mode; i.e., the mode of the caller.
61  //
62 
63  CurrentThread = PsGetCurrentThread ();
64  requestorMode = KeGetPreviousModeByThread(&CurrentThread->Tcb);
65 
66  if (requestorMode != KernelMode) {
67 
68  //
69  // The caller's access mode is not kernel so check to ensure that
70  // the caller has the privilege to load a driver and probe and
71  // capture the name of the driver service entry.
72  //
73 
74  if (!SeSinglePrivilegeCheck( SeLoadDriverPrivilege, requestorMode )) {
75  return STATUS_PRIVILEGE_NOT_HELD;
76  }
77 
78  //
79  // The caller has the appropriate privilege to load and unload
80  // drivers, so capture the driver service name string so that it
81  // can be used to locate the driver from the registry node.
82  //
83 
84  try {
85 
86  ProbeAndReadUnicodeStringEx( &driverServiceName, DriverServiceName );
87 
88  if (!driverServiceName.Length) {
89  return STATUS_INVALID_PARAMETER;
90  }
91 
92  ProbeForRead( driverServiceName.Buffer,
93  driverServiceName.Length,
94  sizeof( WCHAR ) );
95 
96  nameBuffer = ExAllocatePoolWithQuota( PagedPool,
97  driverServiceName.Length );
98 
99  RtlCopyMemory( nameBuffer,
100  driverServiceName.Buffer,
101  driverServiceName.Length );
102 
103  driverServiceName.Buffer = nameBuffer;
104 
105  } except(EXCEPTION_EXECUTE_HANDLER) {
106 
107  //
108  // An exception was incurred while attempting to capture the
109  // input name string or while attempting to allocate the name
110  // string buffer. Simply clean everything up and return an
111  // appropriate error status code.
112  //
113 
114  if (nameBuffer) {
115  ExFreePool( nameBuffer );
116  }
117  return GetExceptionCode();
118  }
119  } else {
120  driverServiceName = *DriverServiceName;
121  }
122 
123  //
124  // Because drivers may wish to create a system thread and execute in
125  // its context, the remainder of this service must be executed in the
126  // context of the primary system process. This is accomplished by
127  // queueing a request to one of the EX worker threads and having it
128  // invoke the I/O system routine to complete this work.
129  //
130  // Fill in a request packet and queue it to the worker thread then, so
131  // that it can actually do the load.
132  //
133 
134  KeInitializeEvent( &loadPacket.Event, NotificationEvent, FALSE );
135  loadPacket.DriverObject = (PDRIVER_OBJECT) NULL;
136  loadPacket.DriverServiceName = &driverServiceName;
137 
138  if (PsGetCurrentProcessByThread(CurrentThread) == PsInitialSystemProcess) {
139 
140  //
141  // If we are already in the system process, just use this thread.
142  //
143 
144  IopLoadUnloadDriver(&loadPacket);
145 
146  } else {
147 
148  ExInitializeWorkItem( &loadPacket.WorkQueueItem,
149  IopLoadUnloadDriver,
150  &loadPacket );
151 
152  ExQueueWorkItem( &loadPacket.WorkQueueItem, DelayedWorkQueue );
153 
154  KeWaitForSingleObject( &loadPacket.Event,
155  UserRequest,
156  KernelMode,
157  FALSE,
158  (PLARGE_INTEGER) NULL );
159 
160  }
161 
162  //
163  // The load operation is now complete. If a name buffer was allocated,
164  // deallocate it now, and return the final status of the load operation.
165  //
166 
167  if (nameBuffer) {
168  ExFreePool( nameBuffer );
169  }
170 
171  return loadPacket.FinalStatus;
172 }

174 NTSTATUS
175 IopCheckUnloadDriver(
176  IN PDRIVER_OBJECT driverObject,
177  OUT PBOOLEAN unloadDriver
178  )
179 {
180  PDEVICE_OBJECT deviceObject;
181  KIRQL irql;
182 
183  //
184  // Check to see whether the driver has already been marked for an unload
185  // operation by anyone in the past.
186  //
187 
188  irql = KeAcquireQueuedSpinLock( LockQueueIoDatabaseLock );
189 
190  if ((driverObject->DeviceObject == NULL &&
191  (driverObject->Flags & DRVO_UNLOAD_INVOKED)) ||
192  (!(driverObject->Flags & DRVO_BASE_FILESYSTEM_DRIVER) && driverObject->DeviceObject &&
193  driverObject->DeviceObject->DeviceObjectExtension->ExtensionFlags
194  & DOE_UNLOAD_PENDING)) {
195 
196  //
197  // The driver has already been marked for unload or is being
198  // unloaded. Simply return a successful completion status since
199  // the driver is on its way out and therefore has been "marked for
200  // unload".
201  //
202 
203  KeReleaseQueuedSpinLock( LockQueueIoDatabaseLock, irql );
204 
205  ObDereferenceObject( driverObject );
206  return STATUS_SUCCESS;
207  }
208 
209  //
210  // The driver exists, and it implements unload, and it has not, so far,
211  // been marked for an unload operation. Simply mark all of the devices
212  // that the driver owns as being marked for unload. While this is going
213  // on, count the references for each of the devices. If all of the
214  // devices have a zero reference count, then tell the driver that it
215  // should unload itself.
216  //
217 
218  deviceObject = driverObject->DeviceObject;
219  *unloadDriver = TRUE;
220 
221  while (deviceObject) {
222  deviceObject->DeviceObjectExtension->ExtensionFlags |= DOE_UNLOAD_PENDING;
223  if (deviceObject->ReferenceCount || deviceObject->AttachedDevice) {
224  *unloadDriver = FALSE;
225  }
226  deviceObject = deviceObject->NextDevice;
227  }
228 
229  //
230  // If this is a base filesystem driver then delay the unload until all its device objects
231  // are deleted.
232  //
233 
234  if (driverObject->Flags & DRVO_BASE_FILESYSTEM_DRIVER && driverObject->DeviceObject) {
235  *unloadDriver = FALSE;
236  }
237 
238  if (*unloadDriver) {
239  driverObject->Flags |= DRVO_UNLOAD_INVOKED;
240  }
241 
242  KeReleaseQueuedSpinLock( LockQueueIoDatabaseLock, irql );
243  return STATUS_UNSUCCESSFUL;
244 }

254 NTSTATUS
255 IopUnloadDriver(
256  IN PUNICODE_STRING DriverServiceName,
257  IN BOOLEAN InvokedByPnpMgr
258  )
259 /*++
260 
261 Routine Description:
262 
263  This service dynamically unloads a device or file system driver from
264  the currently running system. It requires that the caller have the
265  appropriate privilege to execute this service.
266 
267 Arguments:
268 
269  DriverServiceName - Specifies the name of the node in the registry
270  associated with the driver to be unloaded.
271 
272 Return Value:
273 
274  The status returned is the final completion status of the operation.
275 
276 --*/
277 
278 {
279  KPROCESSOR_MODE requestorMode;
280  UNICODE_STRING driverServiceName;
281  PWCHAR nameBuffer = (PWCHAR) NULL;
282  NTSTATUS status;
283  OBJECT_ATTRIBUTES objectAttributes;
284  HANDLE keyHandle;
285  UNICODE_STRING driverName;
286  HANDLE driverHandle;
287  PDRIVER_OBJECT driverObject;
288  BOOLEAN unloadDriver;
289 
290  PAGED_CODE();
291 
292  //
293  // Get the previous mode; i.e., the mode of the caller.
294  //
295 
296  requestorMode = KeGetPreviousMode();
297 
298  if ((requestorMode != KernelMode) && (InvokedByPnpMgr == FALSE)) {
299 
300  //
301  // The caller's access mode is not kernel so check to ensure that
302  // the caller has the privilege to unload a driver and probe and
303  // capture the name of the driver service entry.
304  //
305 
306  if (!SeSinglePrivilegeCheck( SeLoadDriverPrivilege, requestorMode )) {
307  return STATUS_PRIVILEGE_NOT_HELD;
308  }
309 
310  //
311  // The caller has the appropriate privilege to load and unload
312  // drivers, so capture the driver service name string so that it
313  // can be used to locate the driver from the registry node.
314  //
315 
316  try {
317 
318  ProbeAndReadUnicodeStringEx( &driverServiceName, DriverServiceName );
319 
320  if (!driverServiceName.Length) {
321  return STATUS_INVALID_PARAMETER;
322  }
323 
324  ProbeForRead( driverServiceName.Buffer,
325  driverServiceName.Length,
326  sizeof( WCHAR ) );
327 
328  nameBuffer = ExAllocatePoolWithQuota( PagedPool,
329  driverServiceName.Length );
330 
331  RtlCopyMemory( nameBuffer,
332  driverServiceName.Buffer,
333  driverServiceName.Length );
334 
335  driverServiceName.Buffer = nameBuffer;
336 
337  } except(EXCEPTION_EXECUTE_HANDLER) {
338 
339  //
340  // An exception was incurred while attempting to capture the
341  // input name string or while attempting to allocate the name
342  // string buffer. Simply clean everything up and return an
343  // appropriate error status code.
344  //
345 
346  if (nameBuffer) {
347  ExFreePool( nameBuffer );
348  }
349  return GetExceptionCode();
350  }
351 
352  //
353  // Now that the caller's parameters have been captured and everything
354  // appears to have checked out, actually attempt to unload the driver.
355  // This is done with a previous mode of kernel so that drivers will
356  // not fail to unload because the caller didn't happen to have access
357  // to some resource that the driver needs in order to complete its
358  // unload operation.
359  //
360 
361  status = ZwUnloadDriver( &driverServiceName );
362  ExFreePool( nameBuffer );
363  return status;
364  }
365 
366  //
367  // The caller's mode is now kernel mode. Attempt to actually unload the
368  // driver specified by the indicated registry node. Begin by opening
369  // the registry node for this driver.
370  //
371 
372  status = IopOpenRegistryKey( &keyHandle,
373  (HANDLE) NULL,
374  DriverServiceName,
375  KEY_READ,
376  FALSE );
377  if (!NT_SUCCESS( status )) {
378  return status;
379  }
380 
381  //
382  // Get the optional object name for this driver from the value for this
383  // key. If one exists, then its name overrides the default name of the
384  // driver.
385  //
386 
387  status = IopGetDriverNameFromKeyNode( keyHandle,
388  &driverName );
389  ObCloseHandle( keyHandle, KernelMode );
390  if (!NT_SUCCESS( status )) {
391  return status;
392  }
393 
394  //
395  // Now attempt to open the driver object for the specified driver.
396  //
397 
398  InitializeObjectAttributes( &objectAttributes,
399  &driverName,
400  OBJ_CASE_INSENSITIVE|OBJ_KERNEL_HANDLE,
401  (HANDLE) NULL,
402  (PSECURITY_DESCRIPTOR) NULL );
403 
404  status = ObOpenObjectByName( &objectAttributes,
405  IoDriverObjectType,
406  KernelMode,
407  NULL,
408  FILE_READ_DATA,
409  (PVOID) NULL,
410  &driverHandle );
411 
412  //
413  // Perform some common cleanup by getting rid of buffers that have been
414  // allocated up to this point so that error conditions do not have as
415  // much work to do on each exit path.
416  //
417 
418  ExFreePool( driverName.Buffer );
419 
420  //
421  // If the driver object could not be located in the first place, then
422  // return now before attempting to do anything else.
423  //
424 
425  if (!NT_SUCCESS( status )) {
426  return status;
427  }
428 
429  //
430  // The driver object was located, so convert the handle into a pointer
431  // so that the driver object itself can be examined.
432  //
433 
434  status = ObReferenceObjectByHandle( driverHandle,
435  0,
436  IoDriverObjectType,
437  KernelMode,
438  (PVOID *) &driverObject,
439  NULL );
440  ObCloseHandle( driverHandle , KernelMode);
441 
442  if (!NT_SUCCESS( status )) {
443  return status;
444  }
445 
446  //
447  // Check to see whether or not this driver implements unload. Also,
448  // if the driver has no section associated with it, then it was loaded
449  // be the OS loader and therefore cannot be unloaded. If either is true,
450  // return an appropriate error status code.
451  //
452 
453  if (driverObject->DriverUnload == (PDRIVER_UNLOAD) NULL ||
454  !driverObject->DriverSection) {
455  ObDereferenceObject( driverObject );
456  return STATUS_INVALID_DEVICE_REQUEST;
457  }
458 
459  if (!InvokedByPnpMgr && !IopIsLegacyDriver(driverObject)) {
460 
461  ObDereferenceObject( driverObject );
462  return STATUS_INVALID_DEVICE_REQUEST;
463  }
464 
465  //
466  // Check to see whether the driver has already been marked for an unload
467  // operation by anyone in the past.
468  //
469 
470  status = IopCheckUnloadDriver(driverObject,&unloadDriver);
471 
472  if ( NT_SUCCESS(status) ) {
473  return status;
474  }
475 
476  if (unloadDriver) {
477 
478  if (PsGetCurrentProcess() == PsInitialSystemProcess) {
479 
480  //
481  // The current thread is already executing in the context of the
482  // system process, so simply invoke the driver's unload routine.
483  //
484 
485  driverObject->DriverUnload( driverObject );
486 
487  } else {
488 
489  //
490  // The current thread is not executing in the context of the system
491  // process, which is required in order to invoke the driver's unload
492  // routine. Queue a worker item to one of the worker threads to
493  // get into the appropriate process context and then invoke the
494  // routine.
495  //
496 
497  LOAD_PACKET loadPacket;
498 
499  KeInitializeEvent( &loadPacket.Event, NotificationEvent, FALSE );
500  loadPacket.DriverObject = driverObject;
501  ExInitializeWorkItem( &loadPacket.WorkQueueItem,
502  IopLoadUnloadDriver,
503  &loadPacket );
504  ExQueueWorkItem( &loadPacket.WorkQueueItem, DelayedWorkQueue );
505  (VOID) KeWaitForSingleObject( &loadPacket.Event,
506  Executive,
507  KernelMode,
508  FALSE,
509  (PLARGE_INTEGER) NULL );
510  }
511  ObMakeTemporaryObject( driverObject );
512  ObDereferenceObject( driverObject );
513  }
514 
515  //
516  // The driver has either been unloaded, or it has successfully been
517  // marked for an unload operation. Simply dereference the pointer to
518  // the object and return success.
519  //
520 
521  ObDereferenceObject( driverObject );
522  return STATUS_SUCCESS;
523 }
