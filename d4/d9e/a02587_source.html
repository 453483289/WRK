35 /*++
36 
37 Routine Description:
38 
39  This function opens a handle to a process object with the specified
40  desired access.
41 
42  The object is located either by name, or by locating a thread whose
43  Client ID matches the specified Client ID and then opening that thread's
44  process.
45 
46 Arguments:
47 
48  ProcessHandle - Supplies a pointer to a variable that will receive
49  the process object handle.
50 
51  DesiredAccess - Supplies the desired types of access for the process
52  object.
53 
54  ObjectAttributes - Supplies a pointer to an object attributes structure.
55  If the ObjectName field is specified, then ClientId must not be
56  specified.
57 
58  ClientId - Supplies a pointer to a ClientId that if supplied
59  specifies the thread whose process is to be opened. If this
60  argument is specified, then ObjectName field of the ObjectAttributes
61  structure must not be specified.
62 
63 Return Value:
64 
65  NTSTATUS - Status of call
66 
67 --*/

69 {
70 
71  HANDLE Handle;
72  KPROCESSOR_MODE PreviousMode;
73  NTSTATUS Status;
74  PEPROCESS Process;
75  PETHREAD Thread;
76  CLIENT_ID CapturedCid={0};
77  BOOLEAN ObjectNamePresent;
78  BOOLEAN ClientIdPresent;
79  ACCESS_STATE AccessState;
80  AUX_ACCESS_DATA AuxData;
81  ULONG Attributes;
82 
83  PAGED_CODE();
84 
85  //
86  // Make sure that only one of either ClientId or ObjectName is
87  // present.
88  //
89 
90  PreviousMode = KeGetPreviousMode();
91  if (PreviousMode != KernelMode) {
92 
93  //
94  // Since we need to look at the ObjectName field, probe
95  // ObjectAttributes and capture object name present indicator.
96  //
97 
98  try {
99 
100  ProbeForWriteHandle (ProcessHandle);
101 
102  ProbeForReadSmallStructure (ObjectAttributes,
103  sizeof(OBJECT_ATTRIBUTES),
104  sizeof(ULONG));
105  ObjectNamePresent = (BOOLEAN)ARGUMENT_PRESENT (ObjectAttributes->ObjectName);
106  Attributes = ObSanitizeHandleAttributes (ObjectAttributes->Attributes, UserMode);
107 
108  if (ARGUMENT_PRESENT (ClientId)) {
109  ProbeForReadSmallStructure (ClientId, sizeof (CLIENT_ID), sizeof (ULONG));
110  CapturedCid = *ClientId;
111  ClientIdPresent = TRUE;
112  } else {
113  ClientIdPresent = FALSE;
114  }
115  } except (EXCEPTION_EXECUTE_HANDLER) {
116  return GetExceptionCode();
117  }
118  } else {
119  ObjectNamePresent = (BOOLEAN)ARGUMENT_PRESENT (ObjectAttributes->ObjectName);
120  Attributes = ObSanitizeHandleAttributes (ObjectAttributes->Attributes, KernelMode);
121  if (ARGUMENT_PRESENT (ClientId)) {
122  CapturedCid = *ClientId;
123  ClientIdPresent = TRUE;
124  } else {
125  ClientIdPresent = FALSE;
126  }
127  }
128 
129  if (ObjectNamePresent && ClientIdPresent) {
130  return STATUS_INVALID_PARAMETER_MIX;
131  }
132 
133  //
134  // Create an AccessState here, because the caller may have
135  // DebugPrivilege, which requires us to make special adjustments
136  // to his desired access mask. We do this by modifying the
137  // internal fields in the AccessState to achieve the effect
138  // we desire.
139  //
140 
141  Status = SeCreateAccessState(
142  &AccessState,
143  &AuxData,
144  DesiredAccess,
145  &PsProcessType->TypeInfo.GenericMapping
146  );
147 
148  if ( !NT_SUCCESS(Status) ) {
149 
150  return Status;
151  }
152 
153  //
154  // Check here to see if the caller has SeDebugPrivilege. If
155  // he does, we will allow him any access he wants to the process.
156  // We do this by clearing the DesiredAccess in the AccessState
157  // and recording what we want him to have in the PreviouslyGrantedAccess
158  // field.
159  //
160  // Note that this routine performs auditing as appropriate.
161  //
162 
163  if (SeSinglePrivilegeCheck( SeDebugPrivilege, PreviousMode )) {
164 
165  if ( AccessState.RemainingDesiredAccess & MAXIMUM_ALLOWED ) {
166  AccessState.PreviouslyGrantedAccess |= PROCESS_ALL_ACCESS;
167 
168  } else {
169 
170  AccessState.PreviouslyGrantedAccess |= ( AccessState.RemainingDesiredAccess );
171  }
172 
173  AccessState.RemainingDesiredAccess = 0;
174 
175  }
176 
177  if (ObjectNamePresent) {
178 
179  //
180  // Open handle to the process object with the specified desired access,
181  // set process handle value, and return service completion status.
182  //
183 
184  Status = ObOpenObjectByName(
185  ObjectAttributes,
186  PsProcessType,
187  PreviousMode,
188  &AccessState,
189  0,
190  NULL,
191  &Handle
192  );
193 
194  SeDeleteAccessState( &AccessState );
195 
196  if ( NT_SUCCESS(Status) ) {
197  try {
198  *ProcessHandle = Handle;
199  } except (EXCEPTION_EXECUTE_HANDLER) {
200  return GetExceptionCode ();
201  }
202  }
203 
204  return Status;
205  }
206 
207  if ( ClientIdPresent ) {
208 
209  Thread = NULL;
210  if (CapturedCid.UniqueThread) {
211  Status = PsLookupProcessThreadByCid(
212  &CapturedCid,
213  &Process,
214  &Thread
215  );
216 
217  if (!NT_SUCCESS(Status)) {
218  SeDeleteAccessState( &AccessState );
219  return Status;
220  }
221  } else {
222  Status = PsLookupProcessByProcessId(
223  CapturedCid.UniqueProcess,
224  &Process
225  );
226 
227  if ( !NT_SUCCESS(Status) ) {
228  SeDeleteAccessState( &AccessState );
229  return Status;
230  }
231  }
232 
233  //
234  // OpenObjectByAddress
235  //
236 
237  Status = ObOpenObjectByPointer(
238  Process,
239  Attributes,
240  &AccessState,
241  0,
242  PsProcessType,
243  PreviousMode,
244  &Handle
245  );
246 
247  SeDeleteAccessState( &AccessState );
248 
249  if (Thread) {
250  ObDereferenceObject(Thread);
251  }
252 
253  ObDereferenceObject(Process);
254 
255  if (NT_SUCCESS (Status)) {
256 
257  try {
258  *ProcessHandle = Handle;
259  } except (EXCEPTION_EXECUTE_HANDLER) {
260  return GetExceptionCode ();
261  }
262  }
263 
264  return Status;
265 
266  }
267 
268  return STATUS_INVALID_PARAMETER_MIX;
269 }
