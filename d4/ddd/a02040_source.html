118 Routine Description:
119 
120  Allocates a kcb; first try from our own allocator.
121  If it doesn't work (we have maxed out our number of allocs

124

126 
127 
128 Return Value:
129

131 
132 --*/

136  PCM_KEY_CONTROL_BLOCK kcb = NULL;
137  PCM_ALLOC_PAGE AllocPage;
138 
139  PAGED_CODE();
140 
141  if( !CmpAllocInited ) {
142  //
143  // not initialized
144  //
145  goto AllocFromPool;
146  }
147 
148  LOCK_ALLOC_BUCKET();
149 
150 SearchFreeKcb:
151  //
152  // try to find a free one
153  //
154  if( IsListEmpty(&CmpFreeKCBListHead) == FALSE ) {
155  //
156  // found one
157  //
158  kcb = (PCM_KEY_CONTROL_BLOCK)RemoveHeadList(&CmpFreeKCBListHead);
159  kcb = CONTAINING_RECORD(kcb,
160  CM_KEY_CONTROL_BLOCK,
161  FreeListEntry);
162 
163  AllocPage = (PCM_ALLOC_PAGE)KCB_TO_ALLOC_PAGE( kcb );
164 
165  ASSERT( AllocPage->FreeCount != 0 );
166 
167  AllocPage->FreeCount--;
168 
169  //
170  // set when page was allocated
171  //
172  ASSERT( kcb->PrivateAlloc == 1);
173 
174  UNLOCK_ALLOC_BUCKET();
175  return kcb;
176  }
177 
178  //
179  // we need to allocate a new page as we ran out of free kcbs
180  //
181 
182  //
183  // allocate a new page and insert all kcbs in the freelist
184  //
185  AllocPage = (PCM_ALLOC_PAGE)ExAllocatePoolWithTag(PagedPool, PAGE_SIZE, CM_ALLOCATE_TAG|PROTECTED_POOL);
186  if( AllocPage == NULL ) {
187  //
188  // we might be low on pool; maybe small pool chunks will work
189  //
190  UNLOCK_ALLOC_BUCKET();
191  goto AllocFromPool;
192  }
193 
194  //
195  // set up the page
196  //
197  AllocPage->FreeCount = CM_KCBS_PER_PAGE;
198 
199  //
200  // now the dirty job; insert all kcbs inside the page in the free list
201  //
202  for(j=0;j<CM_KCBS_PER_PAGE;j++) {
203  kcb = (PCM_KEY_CONTROL_BLOCK)((PUCHAR)AllocPage + FIELD_OFFSET(CM_ALLOC_PAGE,AllocPage) + j*CM_KCB_ENTRY_SIZE);
204 
205  //
206  // set it here; only once
207  //
208  kcb->PrivateAlloc = 1;
209  kcb->DelayCloseEntry = NULL;
210 
211  InsertTailList(
212  &CmpFreeKCBListHead,
213  &(kcb->FreeListEntry)
214  );
215  }
216 
217  //
218  // this time will find one for sure
219  //
220  goto SearchFreeKcb;
221 
222 AllocFromPool:
223  kcb = ExAllocatePoolWithTag(PagedPool,
224  sizeof(CM_KEY_CONTROL_BLOCK),
225  CM_KCB_TAG | PROTECTED_POOL);
226 
227  if( kcb != NULL ) {
228  //
229  // clear the private alloc flag
230  //
231  kcb->PrivateAlloc = 0;
232  kcb->DelayCloseEntry = NULL;
233  }
234 
235  return kcb;
236 }
