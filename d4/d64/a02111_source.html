59 /*++
60 
61 Routine Description:
62 
63  This function adds the specified atom to the global atom table.
64 
65 Arguments:
66 
67  AtomName - Supplies a pointer to an unicode atom name.
68 
69  Length - Supplies the length of the atom name in bytes.
70 
71  Atom - Supplies an optional pointer to a variable that receives the atom
72  value.
73 
74 Return Value:
75 
76  The service status is returned as the function value.
77 
78 --*/
79 
80 {
81 
82  PVOID AtomTable = (ExGlobalAtomTableCallout)();
83  PWSTR CapturedAtomName;
84  KPROCESSOR_MODE PreviousMode;
85  RTL_ATOM ReturnAtom;
86  ULONG_PTR StackArray[(COPY_STACK_SIZE / sizeof(ULONG_PTR)) + 1];
87  NTSTATUS Status;
88 
89  PAGED_CODE();
90 
91  //
92  // If the atom table is not defined, then return access denied.
93  //
94 
95  if (AtomTable == NULL) {
96  return STATUS_ACCESS_DENIED;
97  }
98 
99  //
100  // If the atom length is too large, then return invalid parameter.
101  //
102 
103  if (Length > COPY_STACK_SIZE) {
104  return STATUS_INVALID_PARAMETER;
105  }
106 
107  //
108  // If the previous mode is user, then probe and copy the arguments as
109  // appropriate.
110  //
111  // N.B. It is known that the length is less than a page from the above
112  // check and, therefore, a small buffer probe can be used.
113  //
114 
115  PreviousMode = KeGetPreviousMode();
116  CapturedAtomName = AtomName;
117  if (PreviousMode != KernelMode) {
118  try {
119  if (ARGUMENT_PRESENT(Atom)) {
120  ProbeForWriteUshort(Atom);
121  }
122 
123  if (ARGUMENT_PRESENT(AtomName)) {
124  ProbeForRead(AtomName, Length, sizeof(WCHAR));
125  CapturedAtomName = (PWSTR)&StackArray[0];
126  RtlCopyMemory(CapturedAtomName, AtomName, Length);
127  CapturedAtomName[Length / sizeof (WCHAR)] = UNICODE_NULL;
128  }
129 
130  } except (EXCEPTION_EXECUTE_HANDLER) {
131  return GetExceptionCode();
132  }
133  }
134 
135  //
136  // Add the specified name to the global atom table.
137  //
138 
139  Status = RtlAddAtomToAtomTable(AtomTable, CapturedAtomName, &ReturnAtom);
140  if ((ARGUMENT_PRESENT(Atom)) && (NT_SUCCESS(Status))) {
141  if (PreviousMode != KernelMode) {
142  try {
143  *Atom = ReturnAtom;
144 
145  } except (EXCEPTION_EXECUTE_HANDLER) {
146  Status = GetExceptionCode();
147  }
148 
149  } else {
150  *Atom = ReturnAtom;
151  }
152  }
153 
154  return Status;
155 }

309 /*++
310 
311 Routine Description:
312 
313  This function queries the specified information for the specified atom.
314 
315 Arguments:
316 
317  Atom - Supplies the atom identification.
318 
319 
320 Return Value:
321 
322 --*/
323 
324 {
325 
326  ULONG AtomFlags;
327  PVOID AtomTable = (ExGlobalAtomTableCallout)();
328  PATOM_BASIC_INFORMATION BasicInfo;
329  ULONG NameLength;
330  KPROCESSOR_MODE PreviousMode;
331  ULONG RequiredLength;
332  NTSTATUS Status;
333  PATOM_TABLE_INFORMATION TableInfo;
334  ULONG UsageCount;
335 
336  PAGED_CODE();
337 
338  //
339  // If the atom table is not defined, then return access denied.
340  //
341 
342  if (AtomTable == NULL) {
343  return STATUS_ACCESS_DENIED;
344  }
345 
346  //
347  // Query atom information.
348  //
349 
350  Status = STATUS_SUCCESS;
351  try {
352 
353  //
354  // Get previous processor mode and probe output argument if necessary.
355  //
356 
357  PreviousMode = KeGetPreviousMode();
358  if (PreviousMode != KernelMode) {
359  ProbeForWrite(AtomInformation, AtomInformationLength, sizeof(ULONG));
360  if (ARGUMENT_PRESENT(ReturnLength)) {
361  ProbeForWriteUlong(ReturnLength);
362  }
363  }
364 
365  RequiredLength = 0;
366  switch (InformationClass) {
367 
368  //
369  // Query basic atom information.
370  //
371 
372  case AtomBasicInformation:
373  RequiredLength = FIELD_OFFSET(ATOM_BASIC_INFORMATION, Name);
374  if (AtomInformationLength < RequiredLength) {
375  return STATUS_INFO_LENGTH_MISMATCH;
376  }
377 
378  BasicInfo = (PATOM_BASIC_INFORMATION)AtomInformation;
379  UsageCount = 0;
380  NameLength = AtomInformationLength - RequiredLength;
381  BasicInfo->Name[ 0 ] = UNICODE_NULL;
382  Status = RtlQueryAtomInAtomTable(AtomTable,
383  Atom,
384  &UsageCount,
385  &AtomFlags,
386  &BasicInfo->Name[0],
387  &NameLength );
388 
389  if (NT_SUCCESS(Status)) {
390  BasicInfo->UsageCount = (USHORT)UsageCount;
391  BasicInfo->Flags = (USHORT)AtomFlags;
392  BasicInfo->NameLength = (USHORT)NameLength;
393  RequiredLength += NameLength + sizeof(WCHAR);
394  }
395 
396  break;
397 
398  //
399  // Query atom table information.
400  //
401 
402  case AtomTableInformation:
403  RequiredLength = FIELD_OFFSET(ATOM_TABLE_INFORMATION, Atoms);
404  if (AtomInformationLength < RequiredLength) {
405  return STATUS_INFO_LENGTH_MISMATCH;
406  }
407 
408  TableInfo = (PATOM_TABLE_INFORMATION)AtomInformation;
409  Status = RtlQueryAtomsInAtomTable(AtomTable,
410  (AtomInformationLength - RequiredLength) / sizeof(RTL_ATOM),
411  &TableInfo->NumberOfAtoms,
412  &TableInfo->Atoms[0]);
413 
414  if (NT_SUCCESS(Status)) {
415  RequiredLength += TableInfo->NumberOfAtoms * sizeof(RTL_ATOM);
416  }
417 
418  break;
419 
420  //
421  // Invalid information class.
422  //
423 
424  default:
425  Status = STATUS_INVALID_INFO_CLASS;
426  break;
427  }
428 
429  if (ARGUMENT_PRESENT(ReturnLength)) {
430  *ReturnLength = RequiredLength;
431  }
432 
433  } except (EXCEPTION_EXECUTE_HANDLER) {
434  Status = GetExceptionCode();
435  }
436 
437  return Status;
438 }
