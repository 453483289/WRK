34 /*++
35 
36 Routine Description:
37 
38  This function accepts The Client ID of a thread, and returns a
39  referenced pointer to the thread, and possibly a referenced pointer
40  to the process.
41 
42 Arguments:
43 
44  Cid - Specifies the Client ID of the thread.
45 
46  Process - If specified, returns a referenced pointer to the process
47  specified in the Cid.
48 
49  Thread - Returns a referenced pointer to the thread specified in the
50  Cid.
51 
52 Return Value:
53 
54  STATUS_SUCCESS - A process and thread were located based on the contents
55  of the Cid.
56 
57  STATUS_INVALID_CID - The specified Cid is invalid.
58 
59 --*/
60 
61 {
62 
63  PHANDLE_TABLE_ENTRY CidEntry;
64  PETHREAD lThread;
65  PETHREAD CurrentThread;
66  PEPROCESS lProcess;
67  NTSTATUS Status;
68 
69  PAGED_CODE();
70 
71 
72  lThread = NULL;
73 
74  CurrentThread = PsGetCurrentThread ();
75  KeEnterCriticalRegionThread (&CurrentThread->Tcb);
76 
77  CidEntry = ExMapHandleToPointer(PspCidTable, Cid->UniqueThread);
78  if (CidEntry != NULL) {
79  lThread = (PETHREAD)CidEntry->Object;
80  if (!ObReferenceObjectSafe (lThread)) {
81  lThread = NULL;
82  }
83  ExUnlockHandleTableEntry(PspCidTable, CidEntry);
84  }
85 
86  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
87 
88  Status = STATUS_INVALID_CID;
89  if (lThread != NULL) {
90  //
91  // This could be a thread or a process. Check its a thread.
92  //
93  if (lThread->Tcb.Header.Type != ThreadObject ||
94  lThread->Cid.UniqueProcess != Cid->UniqueProcess ||
95  lThread->GrantedAccess == 0) {
96  ObDereferenceObject (lThread);
97  } else {
98  *Thread = lThread;
99  if (ARGUMENT_PRESENT (Process)) {
100  lProcess = THREAD_TO_PROCESS (lThread);
101  *Process = lProcess;
102  //
103  // Since the thread holds a reference to the process this reference does not have to
104  // be protected.
105  //
106  ObReferenceObject (lProcess);
107  }
108  Status = STATUS_SUCCESS;
109  }
110 
111  }
112 
113  return Status;
114 }
