133 NTSTATUS
134 LpcpCreatePort (
135  OUT PHANDLE PortHandle,
136  IN POBJECT_ATTRIBUTES ObjectAttributes,
137  IN ULONG MaxConnectionInfoLength,
138  IN ULONG MaxMessageLength,
139  IN ULONG MaxPoolUsage,
140  IN BOOLEAN Waitable
141  )
142 
143 /*++
144 
145 Routine Description:
146 
147  A server process can create a named connection port with the NtCreatePort
148  service.
149 
150  A connection port is created with the name and SECURITY_DESCRIPTOR
151  specified in the ObjectAttributes structure. A handle to the connection
152  port object is returned in the location pointed to by the PortHandle
153  parameter. The returned handle can then be used to listen for connection
154  requests to that port name, using the NtListenPort service.
155 
156  The standard object architecture defined desired access parameter is not
157  necessary since this service can only create a new port, not access an
158  existing port.
159 
160  Connection ports cannot be used to send and receive messages. They are
161  only valid as a parameter to the NtListenPort service.
162 
163 Arguments:
164 
165  PortHandle - A pointer to a variable that will receive the connection port
166  object handle value.
167 
168  ObjectAttributes - A pointer to a structure that specifies the name of the
169  object, an access control list (SECURITY_DESCRIPTOR) to be applied to
170  the object, and a set of object attribute flags.
171 
172  PUNICODE_STRING ObjectName - An optional pointer to a null terminated
173  port name string. The form of the name is
174  [\name...\name]\port_name. If no name is specified then an
175  unconnected communication port is created rather than a connection
176  port. This is useful for sending and receiving messages between
177  threads of a single process.
178 
179  ULONG Attributes - A set of flags that control the port object
180  attributes.
181 
182  None of the standard values are relevant for this call.
183  Connection ports cannot be inherited, are always placed in the
184  system handle table and are exclusive to the creating process.
185  This field must be zero. Future implementations might support
186  specifying the OBJ_PERMANENT attribute.
187 
188  MaxMessageLength - Specifies the maximum length of messages sent or
189  received on communication ports created from this connection
190  port. The value of this parameter cannot exceed
191  MAX_PORTMSG_LENGTH bytes.
192 
193  MaxPoolUsage - Specifies the maximum amount of NonPaged pool used for
194  message storage.
195 
196  Waitable - Specifies if the event used by the port can be use to wait
197  for LPC messages to arrive asynchronously.
198 
199 Return Value:
200 
201  NTSTATUS - An appropriate status value
202 
203 --*/
204 
205 {
206  PLPCP_PORT_OBJECT ConnectionPort;
207  HANDLE Handle;
208  KPROCESSOR_MODE PreviousMode;
209  NTSTATUS Status;
210  PUNICODE_STRING NamePtr;
211  UNICODE_STRING CapturedObjectName;
212 
213  PAGED_CODE();
214 
215  UNREFERENCED_PARAMETER (MaxPoolUsage);
216 
217  //
218  // Get previous processor mode and probe output arguments if necessary.
219  //
220 
221  PreviousMode = KeGetPreviousMode();
222  RtlInitUnicodeString( &CapturedObjectName, NULL );
223 
224  if (PreviousMode != KernelMode) {
225 
226  try {
227 
228  ProbeForWriteHandle( PortHandle );
229 
230  ProbeForReadSmallStructure( ObjectAttributes,
231  sizeof( OBJECT_ATTRIBUTES ),
232  sizeof( ULONG ));
233 
234  NamePtr = ObjectAttributes->ObjectName;
235 
236  if (NamePtr != NULL) {
237 
238  CapturedObjectName = ProbeAndReadStructure( NamePtr,
239  UNICODE_STRING );
240  }
241 
242  } except( EXCEPTION_EXECUTE_HANDLER ) {
243 
244  return( GetExceptionCode() );
245  }
246 
247  } else {
248 
249  if (ObjectAttributes->ObjectName != NULL) {
250 
251  CapturedObjectName = *(ObjectAttributes->ObjectName);
252  }
253  }
254 
255  //
256  // Make the null buffer indicate an unspecified port name
257  //
258 
259  if (CapturedObjectName.Length == 0) {
260 
261  CapturedObjectName.Buffer = NULL;
262  }
263 
264  //
265  // Allocate and initialize a port object. If an object name was
266  // specified, then this is a connection port. Otherwise this is an
267  // unconnected communication port that a process can use to communicate
268  // between threads.
269  //
270 
271  Status = ObCreateObject( PreviousMode,
272  (Waitable ? LpcWaitablePortObjectType
273  : LpcPortObjectType),
274  ObjectAttributes,
275  PreviousMode,
276  NULL,
277  (Waitable ? sizeof( LPCP_PORT_OBJECT )
278  : FIELD_OFFSET( LPCP_PORT_OBJECT, WaitEvent )),
279  0,
280  0,
281  (PVOID *)&ConnectionPort );
282 
283  if (!NT_SUCCESS( Status )) {
284 
285  return( Status );
286  }
287 
288  //
289  // Zero out the connection port object and then initialize its fields
290  //
291 
292  RtlZeroMemory( ConnectionPort, (Waitable ? sizeof( LPCP_PORT_OBJECT )
293  : FIELD_OFFSET( LPCP_PORT_OBJECT, WaitEvent )));
294 
295  ConnectionPort->ConnectionPort = ConnectionPort;
296  ConnectionPort->Creator = PsGetCurrentThread()->Cid;
297 
298  InitializeListHead( &ConnectionPort->LpcReplyChainHead );
299 
300  InitializeListHead( &ConnectionPort->LpcDataInfoChainHead );
301 
302  //
303  // Named ports get a connection message queue.
304  //
305 
306  if (CapturedObjectName.Buffer == NULL) {
307 
308  ConnectionPort->Flags = UNCONNECTED_COMMUNICATION_PORT;
309  ConnectionPort->ConnectedPort = ConnectionPort;
310  ConnectionPort->ServerProcess = NULL;
311 
312  } else {
313 
314  ConnectionPort->Flags = SERVER_CONNECTION_PORT;
315 
316  ObReferenceObject( PsGetCurrentProcess() );
317  ConnectionPort->ServerProcess = PsGetCurrentProcess();
318  }
319 
320  if ( Waitable ) {
321 
322  ConnectionPort->Flags |= PORT_WAITABLE;
323  }
324 
325  //
326  // All ports get a request message queue.
327  //
328 
329  Status = LpcpInitializePortQueue( ConnectionPort );
330 
331  if (!NT_SUCCESS(Status)) {
332 
333  ObDereferenceObject( ConnectionPort );
334 
335  return(Status);
336  }
337 
338  //
339  // For a waitable port, create the KEVENT that will
340  // be used to signal clients
341  //
342 
343  if (ConnectionPort->Flags & PORT_WAITABLE) {
344 
345  KeInitializeEvent( &ConnectionPort->WaitEvent,
346  NotificationEvent,
347  FALSE );
348  }
349 
350  //
351  // Set the maximum message length and connection info length based on the
352  // zone block size less the structs overhead.
353  //
354 
355  ConnectionPort->MaxMessageLength = (USHORT) (LpcpGetMaxMessageLength() -
356  FIELD_OFFSET( LPCP_MESSAGE, Request ));
357 
358  ConnectionPort->MaxConnectionInfoLength = (USHORT) (ConnectionPort->MaxMessageLength -
359  sizeof( PORT_MESSAGE ) -
360  sizeof( LPCP_CONNECTION_MESSAGE ));
361 
362 #if DBG
363  LpcpTrace(( "Created port %ws (%x) - MaxMsgLen == %x MaxConnectInfoLen == %x\n",
364  CapturedObjectName.Buffer == NULL ? L"** UnNamed **" : ObjectAttributes->ObjectName->Buffer,
365  ConnectionPort,
366  ConnectionPort->MaxMessageLength,
367  ConnectionPort->MaxConnectionInfoLength ));
368 #endif
369 
370  //
371  // Sanity check that the max message length being asked for is not
372  // greater than the max message length possible in the system
373  //
374 
375  if (ConnectionPort->MaxMessageLength < MaxMessageLength) {
376 
377 #if DBG
378  LpcpPrint(( "MaxMessageLength granted is %x but requested %x\n",
379  ConnectionPort->MaxMessageLength,
380  MaxMessageLength ));
381 #endif
382 
383  ObDereferenceObject( ConnectionPort );
384 
385  return STATUS_INVALID_PARAMETER_4;
386  }
387 
388  //
389  // Save the MaxMessageLength to the connection port
390  //
391 
392  ConnectionPort->MaxMessageLength = (USHORT) MaxMessageLength;
393 
394  //
395  // Sanity check that the max connection info length being asked for is
396  // not greater than the maximum possible in the system
397  //
398 
399  if (ConnectionPort->MaxConnectionInfoLength < MaxConnectionInfoLength) {
400 
401 #if DBG
402  LpcpPrint(( "MaxConnectionInfoLength granted is %x but requested %x\n",
403  ConnectionPort->MaxConnectionInfoLength,
404  MaxConnectionInfoLength ));
405 #endif
406 
407  ObDereferenceObject( ConnectionPort );
408 
409  return STATUS_INVALID_PARAMETER_3;
410  }
411 
412  //
413  // Save the max connection length to the port
414  //
415 
416  ConnectionPort->MaxConnectionInfoLength = (USHORT)MaxConnectionInfoLength;
417 
418  //
419  // Insert connection port object in specified object table. Set port
420  // handle value if successful. If not successful, then the port will
421  // have been dereferenced, which will cause it to be freed, after our
422  // delete procedure is called. The delete procedure will undo the work
423  // done to initialize the port. Finally, return the system server status.
424  //
425 
426  Status = ObInsertObject( ConnectionPort,
427  NULL,
428  PORT_ALL_ACCESS,
429  0,
430  (PVOID *)NULL,
431  &Handle );
432 
433  if (NT_SUCCESS( Status )) {
434 
435  //
436  // Set the output variable protected against access faults
437  //
438 
439  try {
440 
441  *PortHandle = Handle;
442 
443  } except( EXCEPTION_EXECUTE_HANDLER ) {
444 
445  NtClose( Handle );
446 
447  Status = GetExceptionCode();
448  }
449  }
450 
451  //
452  // And return to our caller
453  //
454 
455  return Status;
456 }
