44 
45 Routine Description:
46 
47  This function initializes a thread object. The priority, affinity,
48  and initial quantum are taken from the parent process object.
49 
50  N.B. This routine is carefully written so that if an access violation
51  occurs while reading the specified context frame, then no kernel
52  data structures will have been modified. It is the responsibility
53  of the caller to handle the exception and provide necessary clean
54  up.
55 
56  N.B. It is assumed that the thread object is zeroed.
57 
58 Arguments:
59 
60  Thread - Supplies a pointer to a dispatcher object of type thread.
61

63  the context frame for the thread is to be constructed.
64

66  called when the thread is first scheduled for execution.
67

69  called after the system has finished initializing the thread. This
70  parameter is specified if the thread is a system thread and will
71  execute totally in kernel mode.
72

74  which will be passed to the StartRoutine as a parameter. This
75  parameter is specified if the thread is a system thread and will
76  execute totally in kernel mode.
77 
78  ContextFrame - Supplies an optional pointer a context frame which contains
79  the initial user mode state of the thread. This parameter is specified
80  if the thread is a user thread and will execute in user mode. If this
81  parameter is not specified, then the Teb parameter is ignored.
82 
83  Teb - Supplies an optional pointer to the user mode thread environment
84  block. This parameter is specified if the thread is a user thread and
85  will execute in user mode. This parameter is ignored if the ContextFrame
86  parameter is not specified.

89

92  None.

383  //

389 
390 #else

393  // Initialize the ideal processor number.
394  //
395  // N.B. It is guaranteed that the process affinity intersects the process
396  // ideal node affinity.
397  //
398  // N.B. The preferred set, however, must be reduced by the process affinity.

450 #endif

454  //

464  //
465

475  //

479  return;

572  __inout PKTHREAD Thread,
573  __in KPROCESSOR_MODE AlertMode
574  )
575 
576 /*++
577 
578 Routine Description:
579 
580  This function attempts to alert a thread and cause its execution to
581  be continued if it is currently in an alertable Wait state. Otherwise
582  it just sets the alerted variable for the specified processor mode.
583 
584 Arguments:
585 
586  Thread - Supplies a pointer to a dispatcher object of type thread.
587 
588  AlertMode - Supplies the processor mode for which the thread is
589  to be alerted.
590 
591 Return Value:
592 
593  The previous state of the alerted variable for the specified processor
594  mode.
595 
596 --*/
597 
598 {
599 
600  BOOLEAN Alerted;
601  KLOCK_QUEUE_HANDLE LockHandle;
602 
603  ASSERT_THREAD(Thread);
604  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
605 
606  //
607  // Raise IRQL to SYNCH_LEVEL, acquire the thread APC queue lock, and lock
608  // the dispatcher database.
609  //
610 
611  KeAcquireInStackQueuedSpinLockRaiseToSynch(&Thread->ApcQueueLock, &LockHandle);
612  KiLockDispatcherDatabaseAtSynchLevel();
613 
614  //
615  // Capture the current state of the alerted variable for the specified
616  // processor mode.
617  //
618 
619  Alerted = Thread->Alerted[AlertMode];
620 
621  //
622  // If the alerted state for the specified processor mode is Not-Alerted,
623  // then attempt to alert the thread.
624  //
625 
626  if (Alerted == FALSE) {
627 
628  //
629  // If the thread is currently in a Wait state, the Wait is alertable,
630  // and the specified processor mode is less than or equal to the Wait
631  // mode, then the thread is unwaited with a status of "alerted".
632  //
633 
634  if ((Thread->State == Waiting) && (Thread->Alertable == TRUE) &&
635  (AlertMode <= Thread->WaitMode)) {
636  KiUnwaitThread(Thread, STATUS_ALERTED, ALERT_INCREMENT);
637 
638  } else {
639  Thread->Alerted[AlertMode] = TRUE;
640  }
641  }
642 
643  //
644  // Unlock the dispatcher database from SYNCH_LEVEL, release the thread
645  // APC queue lock, exit the scheduler, and return the previous alerted
646  // state for the specified mode.
647  //
648 
649  KiUnlockDispatcherDatabaseFromSynchLevel();
650  KeReleaseInStackQueuedSpinLockFromDpcLevel(&LockHandle);
651  KiExitDispatcher(LockHandle.OldIrql);
652  return Alerted;
653 }

656 KeAlertResumeThread (
657  __inout PKTHREAD Thread
658  )
659 
660 /*++
661 
662 Routine Description:
663 
664  This function attempts to alert a thread in kernel mode and cause its
665  execution to be continued if it is currently in an alertable Wait state.
666  In addition, a resume operation is performed on the specified thread.
667 
668 Arguments:
669 
670  Thread - Supplies a pointer to a dispatcher object of type thread.
671 
672 Return Value:
673 
674  The previous suspend count.
675 
676 --*/
677 
678 {
679 
680  KLOCK_QUEUE_HANDLE LockHandle;
681  ULONG OldCount;
682 
683  ASSERT_THREAD(Thread);
684  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
685 
686  //
687  // Raise IRQL to SYNCH_LEVEL, acquire the thread APC queue lock, and lock
688  // the dispatcher database.
689  //
690 
691  KeAcquireInStackQueuedSpinLockRaiseToSynch(&Thread->ApcQueueLock, &LockHandle);
692  KiLockDispatcherDatabaseAtSynchLevel();
693 
694  //
695  // If the kernel mode alerted state is FALSE, then attempt to alert
696  // the thread for kernel mode.
697  //
698 
699  if (Thread->Alerted[KernelMode] == FALSE) {
700 
701  //
702  // If the thread is currently in a Wait state and the Wait is alertable,
703  // then the thread is unwaited with a status of "alerted". Else set the
704  // kernel mode alerted variable.
705  //
706 
707  if ((Thread->State == Waiting) && (Thread->Alertable == TRUE)) {
708  KiUnwaitThread(Thread, STATUS_ALERTED, ALERT_INCREMENT);
709 
710  } else {
711  Thread->Alerted[KernelMode] = TRUE;
712  }
713  }
714 
715  //
716  // Capture the current suspend count.
717  //
718 
719  OldCount = Thread->SuspendCount;
720 
721  //
722  // If the thread is currently suspended, then decrement its suspend count.
723  //
724 
725  if (OldCount != 0) {
726  Thread->SuspendCount -= 1;
727 
728  //
729  // If the resultant suspend count is zero and the freeze count is
730  // zero, then resume the thread by releasing its suspend semaphore.
731  //
732 
733  if ((Thread->SuspendCount == 0) && (Thread->FreezeCount == 0)) {
734  Thread->SuspendSemaphore.Header.SignalState += 1;
735  KiWaitTest(&Thread->SuspendSemaphore, RESUME_INCREMENT);
736  }
737  }
738 
739  //
740  // Unlock the dispatcher database from SYNCH_LEVEL, release the thread
741  // APC queue lock, exit the scheduler, and return the previous suspend
742  // count.
743  //
744 
745  KiUnlockDispatcherDatabaseFromSynchLevel();
746  KeReleaseInStackQueuedSpinLockFromDpcLevel(&LockHandle);
747  KiExitDispatcher(LockHandle.OldIrql);
748  return OldCount;
749 }
