230 NTSTATUS
231 CmpDoFileSetSize(
232  PHHIVE Hive,
233  ULONG FileType,
234  ULONG FileSize,
235  ULONG OldFileSize
236  )
237 /*++
238 
239 Routine Description:
240 
241  This routine sets the size of a file. It must not return until
242  the size is guaranteed.
243 
244  It is environment specific.
245 
246  Must be running in the context of the cmp worker thread.
247 
248 Arguments:
249 
250  Hive - Hive we are doing I/O for
251 
252  FileType - which supporting file to use
253 
254  FileSize - 32 bit value to set the file's size to
255 
256  OldFileSize - old file size, in order to determine if this is a shrink;
257  - ignored if file type is not primary, or hive doesn't use
258  the mapped views technique
259 
260 Return Value:
261 
262  FALSE if failure
263  TRUE if success
264 
265 --*/
266 {
267  PCMHIVE CmHive;
268  HANDLE FileHandle;
269  NTSTATUS Status;
270  FILE_END_OF_FILE_INFORMATION FileInfo;
271  IO_STATUS_BLOCK IoStatus;
272  BOOLEAN oldFlag;
273  LARGE_INTEGER FileOffset; // where the mapping starts
274 
275  ASSERT(FIELD_OFFSET(CMHIVE, Hive) == 0);
276 
277  CmHive = (PCMHIVE)Hive;
278  FileHandle = CmHive->FileHandles[FileType];
279  if (FileHandle == NULL) {
280  return TRUE;
281  }
282 
283  //
284  // disable hard error popups, to avoid self deadlock on bogus devices
285  //
286  oldFlag = IoSetThreadHardErrorMode(FALSE);
287 
288  FileInfo.EndOfFile.HighPart = 0L;
289  if( FileType == HFILE_TYPE_PRIMARY ) {
290  FileInfo.EndOfFile.LowPart = ROUND_UP(FileSize, CM_FILE_GROW_INCREMENT);
291  } else {
292  FileInfo.EndOfFile.LowPart = FileSize;
293  }
294 
295  ASSERT_PASSIVE_LEVEL();
296 
297  Status = ZwSetInformationFile(
298  FileHandle,
299  &IoStatus,
300  (PVOID)&FileInfo,
301  sizeof(FILE_END_OF_FILE_INFORMATION),
302  FileEndOfFileInformation
303  );
304 
305  if (NT_SUCCESS(Status)) {
306  ASSERT(IoStatus.Status == Status);
307  } else {
308 
309  //
310  // set debugging info
311  //
312  CmRegistryIODebug.Action = CmpIoFileSetSize;
313  CmRegistryIODebug.Handle = FileHandle;
314  CmRegistryIODebug.Status = Status;
315 #if DBG
316  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_TRACE_LEVEL,"CmpFileSetSize:\tHandle=%08lx OldLength = %08lx NewLength=%08lx \n",
317  FileHandle, OldFileSize, FileSize);
318 #endif
319  if( (Status == STATUS_DISK_FULL) && ExIsResourceAcquiredExclusiveLite(&CmpRegistryLock) ) {
320  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_TRACE_LEVEL,"Disk is full while attempting to grow file %lx; will flush upon lock release\n",FileHandle);
321  CmpFlushOnLockRelease = TRUE;;
322  }
323  }
324 
325  //
326  // restore hard error popups mode
327  //
328  IoSetThreadHardErrorMode(oldFlag);
329 
330 
331  //
332  // purge
333  //
334  if( HiveWritesThroughCache(Hive,FileType) && (OldFileSize > FileSize)) {
335  //
336  // first we have to unmap any possible mapped views in the last 256K window
337  // to avoid deadlock on CcWaitOnActiveCount inside CcPurgeCacheSection call below
338  //
339  ULONG Offset = FileSize & (~(_256K - 1));
340  //
341  // we are not allowed to shrink in shared mode.
342  //
343  ASSERT_HIVE_WRITER_LOCK_OWNED((PCMHIVE)Hive);
344 
345  while( Offset < OldFileSize ) {
346  CmpUnmapCmViewSurroundingOffset((PCMHIVE)Hive,Offset);
347  Offset += CM_VIEW_SIZE;
348  }
349 
350  //
351  // we need to take extra precaution here and unmap the very last view too
352  //
353 
354  FileOffset.HighPart = 0;
355  FileOffset.LowPart = FileSize;
356  //
357  // This is a shrink; Inform cache manager of the change of the size
358  //
359  CcPurgeCacheSection( ((PCMHIVE)Hive)->FileObject->SectionObjectPointer, (PLARGE_INTEGER)(((ULONG_PTR)(&FileOffset)) + 1),
360  OldFileSize - FileSize, FALSE );
361 
362  //
363  // Flush out this view to clear out the Cc dirty hints
364  //
365  CcFlushCache( ((PCMHIVE)Hive)->FileObject->SectionObjectPointer, (PLARGE_INTEGER)(((ULONG_PTR)(&FileOffset)) + 1),/*we are private writers*/
366  OldFileSize - FileSize,NULL);
367 
368  }
369 
370  return Status;
371 }

410 BOOLEAN
411 CmpFileRead (
412  PHHIVE Hive,
413  ULONG FileType,
414  PULONG FileOffset,
415  PVOID DataBuffer,
416  ULONG DataLength
417  )
418 /*++
419 
420 Routine Description:
421 
422  This routine reads in a buffer from a file.
423 
424  It is environment specific.
425 
426  NOTE: We assume the handle is opened for asynchronous access,
427  and that we, and not the IO system, are keeping the
428  offset pointer.
429 
430  NOTE: Only 32bit offsets are supported, even though the underlying
431  IO system on NT supports 64 bit offsets.
432 
433 Arguments:
434 
435  Hive - Hive we are doing I/O for
436 
437  FileType - which supporting file to use
438 
439  FileOffset - pointer to variable providing 32bit offset on input,
440  and receiving new 32bit offset on output.
441 
442  DataBuffer - pointer to buffer
443 
444  DataLength - length of buffer
445 
446 Return Value:
447 
448  FALSE if failure
449  TRUE if success
450 
451 --*/
452 {
453  NTSTATUS status;
454  LARGE_INTEGER Offset;
455  IO_STATUS_BLOCK IoStatus;
456  PCMHIVE CmHive;
457  HANDLE FileHandle;
458  ULONG LengthToRead;
459  HANDLE eventHandle = NULL;
460  PKEVENT eventObject = NULL;
461 
462  ASSERT(FIELD_OFFSET(CMHIVE, Hive) == 0);
463  CmHive = (PCMHIVE)Hive;
464  FileHandle = CmHive->FileHandles[FileType];
465  if (FileHandle == NULL) {
466  return TRUE;
467  }
468 
469  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"CmpFileRead:\n"));
470  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"\tHandle=%08lx Offset=%08lx ", FileHandle, *FileOffset));
471  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"Buffer=%p Length=%08lx\n", DataBuffer, DataLength));
472 
473  //
474  // Detect attempt to read off end of 2gig file (this should be irrelevant)
475  //
476  if ((0xffffffff - *FileOffset) < DataLength) {
477  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"CmpFileRead: runoff\n"));
478  return FALSE;
479  }
480 
481  status = CmpCreateEvent(
482  SynchronizationEvent,
483  &eventHandle,
484  &eventObject);
485  if (!NT_SUCCESS(status))
486  return FALSE;
487 
488  //
489  // We'd really like to just call the filesystems and have them do
490  // the right thing. But the filesystem will attempt to lock our
491  // entire buffer into memory, and that may fail for large requests.
492  // So we split our reads into 64k chunks and call the filesystem for
493  // each one.
494  //
495  ASSERT_PASSIVE_LEVEL();
496  while (DataLength > 0) {
497 
498  //
499  // Convert ULONG to Large
500  //
501  Offset.LowPart = *FileOffset;
502  Offset.HighPart = 0L;
503 
504  //
505  // trim request down if necessary.
506  //
507  if (DataLength > MAX_FILE_IO) {
508  LengthToRead = MAX_FILE_IO;
509  } else {
510  LengthToRead = DataLength;
511  }
512 
513  status = ZwReadFile(
514  FileHandle,
515  eventHandle,
516  NULL, // apcroutine
517  NULL, // apccontext
518  &IoStatus,
519  DataBuffer,
520  LengthToRead,
521  &Offset,
522  NULL // key
523  );
524 
525  if (STATUS_PENDING == status) {
526  status = KeWaitForSingleObject(eventObject, Executive,
527  KernelMode, FALSE, NULL);
528  ASSERT(STATUS_SUCCESS == status);
529  status = IoStatus.Status;
530  }
531 
532  //
533  // adjust offsets
534  //
535  *FileOffset = Offset.LowPart + LengthToRead;
536  DataLength -= LengthToRead;
537  DataBuffer = (PVOID)((PCHAR)DataBuffer + LengthToRead);
538 
539  if (NT_SUCCESS(status)) {
540  ASSERT(IoStatus.Status == status);
541  if (IoStatus.Information != LengthToRead) {
542  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"CmpFileRead:\n\t"));
543  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"Failure1: status = %08lx ", status));
544  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"IoInformation = %08lx\n", IoStatus.Information));
545  ObDereferenceObject(eventObject);
546  ZwClose(eventHandle);
547  CmRegistryIODebug.Action = CmpIoFileRead;
548  CmRegistryIODebug.Handle = FileHandle;
549 #if defined(_WIN64)
550  CmRegistryIODebug.Status = (ULONG)IoStatus.Information - LengthToRead;
551 #else
552  CmRegistryIODebug.Status = (ULONG)&IoStatus;
553 #endif
554  return FALSE;
555  }
556  } else {
557  //
558  // set debugging info
559  //
560  CmRegistryIODebug.Action = CmpIoFileRead;
561  CmRegistryIODebug.Handle = FileHandle;
562  CmRegistryIODebug.Status = status;
563 #if DBG
564  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_TRACE_LEVEL,"CmpFileRead:\tFailure2: status = %08lx IoStatus = %08lx\n", status, IoStatus.Status);
565 #endif
566 
567  ObDereferenceObject(eventObject);
568  ZwClose(eventHandle);
569  return FALSE;
570  }
571 
572  }
573  ObDereferenceObject(eventObject);
574  ZwClose(eventHandle);
575  return TRUE;
576 }

578 BOOLEAN
579 CmpFileWriteThroughCache(
580  PHHIVE Hive,
581  ULONG FileType,
582  PCMP_OFFSET_ARRAY offsetArray,
583  ULONG offsetArrayCount
584  )
585 /*++
586 
587 Routine Description:
588 
589  This is routine writes dirty ranges of data using Cc mapped views.
590  The benefit is that writes don't go through Cc Lazy Writer, so there
591  is no danger to be throttled or deferred.
592 
593  It also flushes the cache for the written ranges, guaranteeing that
594  the data was commited to the disk upon return.
595 
596 Arguments:
597 
598  Hive - Hive we are doing I/O for
599 
600  FileType - which supporting file to use
601 
602  offsetArray - array of structures where each structure holds a 32bit offset
603  into the Hive file and pointer the a buffer written to that
604  file offset.
605 
606  offsetArrayCount - number of elements in the offsetArray.
607 
608 Return Value:
609 
610  FALSE if failure
611  TRUE if success
612 
613 Note:
614 
615  This routine is intended to deal only with paged bins (i.e. bins crossing the
616  CM_VIEW_SIZE boundary or bins that were added after the last sync)
617 
618 Assumption:
619 
620  We work on the assumption that the data to be written at one iteration never spans
621  over the CM_VIEW_SIZE boundary. HvpFindNextDirtyBlock takes care of that !!!
622 --*/
623 {
624  ULONG i;
625  PVOID DataBuffer;
626  ULONG DataLength;
627  ULONG FileOffset;
628  PCMHIVE CmHive;
629  PVOID Bcb;
630  PVOID FileBuffer;
631  LARGE_INTEGER Offset;
632  IO_STATUS_BLOCK IoStatus;
633 
634  ASSERT_PASSIVE_LEVEL();
635 
636 #if !DBG
637  UNREFERENCED_PARAMETER (FileType);
638 #endif
639 
640  CmHive = (PCMHIVE)CONTAINING_RECORD(Hive, CMHIVE, Hive);
641 
642  ASSERT( ((FileType == HFILE_TYPE_EXTERNAL) && (CmHive->FileObject != NULL)) || HiveWritesThroughCache(Hive,FileType) );
643 
644  Offset.HighPart = 0;
645  //
646  // iterate through the array of data
647  //
648  for(i=0;i<offsetArrayCount;i++) {
649  DataBuffer = offsetArray[i].DataBuffer;
650  DataLength = offsetArray[i].DataLength;
651  FileOffset = offsetArray[i].FileOffset;
652  //
653  // data should never span over CM_VIEW_SIZE boundary
654  //
655  ASSERT( (FileOffset & (~(CM_VIEW_SIZE - 1))) == ((FileOffset + DataLength - 1) & (~(CM_VIEW_SIZE - 1))) );
656 
657  //
658  // unmap any possible mapped view that could overlap with this range ; not needed !!!!
659  //
660 
661  //
662  // map and pin data
663  //
664  Offset.LowPart = FileOffset;
665  try {
666  if( !CcPinRead (CmHive->FileObject,&Offset,DataLength,PIN_WAIT,&Bcb,&FileBuffer) ) {
667  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"CmpFileWriteThroughCache - could not pin read view i= %lu\n",i));
668 #if DBG
669  DbgBreakPoint();
670 #endif //DBG
671  return FALSE;
672  }
673  //
674  // copy data to pinned view; we need to do it inside try except, to protect against devices/volumes
675  // dismounting from under us.
676  //
677  RtlCopyMemory(FileBuffer,DataBuffer, DataLength);
678 
679  } except (EXCEPTION_EXECUTE_HANDLER) {
680  //
681  // in low-memory scenarios, CcPinRead throws a STATUS_INSUFFICIENT_RESOURCES
682  // We want to catch this and treat as a "not enough resources" problem,
683  // rather than letting it to surface the kernel call
684  //
685  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"CmpFileWriteThroughCache : CcPinRead has raised :%08lx\n",GetExceptionCode()));
686  return FALSE;
687  }
688 
689  //
690  // dirty, unpin and flush
691  //
692  CcSetDirtyPinnedData (Bcb,NULL);
693  CcUnpinData( Bcb );
694  CcFlushCache (CmHive->FileObject->SectionObjectPointer,(PLARGE_INTEGER)(((ULONG_PTR)(&Offset)) + 1)/*we are private writers*/,DataLength,&IoStatus);
695  if(!NT_SUCCESS(IoStatus.Status) ) {
696  return FALSE;
697  }
698  }
699 
700  return TRUE;
701 }

703 BOOLEAN
704 CmpFileWrite(
705  PHHIVE Hive,
706  ULONG FileType,
707  PCMP_OFFSET_ARRAY offsetArray,
708  ULONG offsetArrayCount,
709  PULONG FileOffset
710  )
711 /*++
712 
713 Routine Description:
714 
715  This routine writes an array of buffers out to a file.
716 
717  It is environment specific.
718 
719  NOTE: We assume the handle is opened for asynchronous access,
720  and that we, and not the IO system, are keeping the
721  offset pointer.
722 
723  NOTE: Only 32bit offsets are supported, even though the underlying
724  IO system on NT supports 64 bit offsets.
725 
726 Arguments:
727 
728  Hive - Hive we are doing I/O for
729 
730  FileType - which supporting file to use
731 
732  offsetArray - array of structures where each structure holds a 32bit offset
733  into the Hive file and pointer the a buffer written to that
734  file offset.
735 
736  offsetArrayCount - number of elements in the offsetArray.
737 
738  FileOffset - returns the file offset after the last write to the file.
739 
740 Return Value:
741 
742  FALSE if failure
743  TRUE if success
744 
745 --*/
746 {
747  NTSTATUS status;
748  LARGE_INTEGER Offset;
749  PCMHIVE CmHive;
750  HANDLE FileHandle;
751  ULONG LengthToWrite;
752  LONG WaitBufferCount = 0;
753  LONG idx;
754  ULONG arrayCount = 0;
755  PVOID DataBuffer = NULL; // W4 only
756  ULONG DataLength;
757  BOOLEAN ret_val = TRUE;
758  PCM_WRITE_BLOCK WriteBlock = NULL;
759 
760  if (CmpNoWrite) {
761  return TRUE;
762  }
763 
764  ASSERT(FIELD_OFFSET(CMHIVE, Hive) == 0);
765  CmHive = (PCMHIVE)Hive;
766  FileHandle = CmHive->FileHandles[FileType];
767  if (FileHandle == NULL) {
768  return TRUE;
769  }
770 
771  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"CmpFileWrite:\n"));
772  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"\tHandle=%08lx ", FileHandle));
773 
774  //
775  // decide whether we wait for IOs to complete or just issue them and
776  // rely on the CcFlushCache to do the job
777  //
778 
779  // Bring pages being written into memory first to allow disk to write
780  // buffer contiguously.
781  try {
782  for (idx = 0; (ULONG) idx < offsetArrayCount; idx++) {
783  char * start = offsetArray[idx].DataBuffer;
784  char * end = (char *) start + offsetArray[idx].DataLength;
785  while (start < end) {
786  // perftouchbuffer globally declared so that compiler won't try
787  // to remove it and this loop (if its smart enough?).
788  perftouchbuffer += (ULONG) *start;
789  start += PAGE_SIZE;
790  }
791  }
792  } except (EXCEPTION_EXECUTE_HANDLER) {
793  //
794  // we might get STATUS_IN_PAGE_ERROR

950  // some I/Os, but not all of them. This is an error, but we
951  // cannot return from this routine until all the successfully
952  // issued I/Os have completed.
953  //

957  // (log files and hives not using the mapped views technique)
958  //
959  status = KeWaitForMultipleObjects(WaitBufferCount,
960  WriteBlock->EventObjects,
961  WaitAll,
962  Executive,
963  KernelMode,
964  FALSE,
965  NULL,
966  WriteBlock->WaitBlockArray);
967  }

983  }
984

986 }
