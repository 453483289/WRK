38 CmpAddToHiveFileList(
39  PCMHIVE CmHive
40  )
41 /*++
42 
43 Routine Description:
44 
45  Add Hive to list of hives and their files in
46  \registry\machine\system\currentcontrolset\control\hivelist
47 
48 Arguments:
49 
50  HivePath - path to root of hive (e.g. \registry\machine\system)
51 
52  CmHive - pointer to CM_HIVE structure for hive.
53 
54 Return Value:
55 
56  ntstatus
57 
58 --*/
59 {
60 #define NAME_BUFFER_SIZE 512
61  OBJECT_ATTRIBUTES ObjectAttributes;
62  HANDLE KeyHandle;
63  NTSTATUS Status;
64  PUCHAR Buffer;
65  ULONG Length;
66  PWSTR FilePath;
67  WCHAR UnicodeNull=UNICODE_NULL;
68  UNICODE_STRING TempName;
69  UNICODE_STRING HivePath;
70 
71  //
72  // create/open the hive list key
73  //
74  RtlInitUnicodeString(
75  &TempName,
76  HIVE_LIST
77  );
78 
79  InitializeObjectAttributes(
80  &ObjectAttributes,
81  &TempName,
82  OBJ_CASE_INSENSITIVE,
83  (HANDLE)NULL,
84  NULL
85  );
86 
87  Status = ZwCreateKey(
88  &KeyHandle,
89  KEY_READ | KEY_WRITE,
90  &ObjectAttributes,
91  0,
92  NULL,
93  REG_OPTION_VOLATILE,
94  NULL
95  );
96 
97  if (!NT_SUCCESS(Status)) {
98  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"CmpAddToHiveFileList: "));
99  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"Create/Open of Hive list failed status = %08lx\n", Status));
100  return Status;
101  }
102 
103  //
104  // allocate work buffers
105  //
106  Buffer = ExAllocatePool(PagedPool, NAME_BUFFER_SIZE + sizeof(WCHAR));
107  if (Buffer == NULL) {
108  NtClose(KeyHandle);
109  return STATUS_NO_MEMORY;
110  }
111 
112  //
113  // compute name of hive
114  //
115  if (! CmpGetHiveName(CmHive, &HivePath)) {
116  NtClose(KeyHandle);
117  ExFreePool(Buffer);
118  return STATUS_NO_MEMORY;
119  }
120 
121 
122  //
123  // get name of file
124  //
125  if (!(CmHive->Hive.HiveFlags & HIVE_VOLATILE)) {
126  Status = ZwQueryObject(
127  CmHive->FileHandles[HFILE_TYPE_PRIMARY],
128  ObjectNameInformation,
129  (PVOID)Buffer,
130  NAME_BUFFER_SIZE,
131  &Length
132  );
133  Length -= sizeof(UNICODE_STRING);
134  if (!NT_SUCCESS(Status)) {
135  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"CmpAddToHiveFileList: "));
136  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"Query of name2 failed status = %08lx\n", Status));
137  NtClose(KeyHandle);
138  ExFreePool(HivePath.Buffer);
139  ExFreePool(Buffer);
140  return Status;
141  }
142  FilePath = ((POBJECT_NAME_INFORMATION)Buffer)->Name.Buffer;
143  FilePath[Length/sizeof(WCHAR)] = UNICODE_NULL;
144  Length+=sizeof(WCHAR);
145  } else {
146  FilePath = &UnicodeNull;
147  Length = sizeof(UnicodeNull);
148  }
149 
150  //
151  // set entry in list
152  //
153  Status = ZwSetValueKey(
154  KeyHandle,
155  &HivePath,
156  0,
157  REG_SZ,
158  FilePath,
159  Length
160  );
161  if (!NT_SUCCESS(Status)) {
162  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"CmpAddToHiveFileList: "));
163  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BUGCHECK,"Set of entry in Hive list failed status = %08lx\n", Status));
164  }
165 
166  NtClose(KeyHandle);
167  ExFreePool(HivePath.Buffer);
168  ExFreePool(Buffer);
169  return Status;
170 }

232 BOOLEAN
233 CmpGetHiveName(
234  PCMHIVE CmHive,
235  PUNICODE_STRING HiveName
236  )
237 /*++
238 
239 Routine Description:
240 
241  Compute full path to a hive.
242 
243 Arguments:
244 
245  CmHive - pointer to CmHive structure
246 
247  HiveName - supplies pointer to unicode string structure that
248  will be filled in with pointer to name.
249 
250  CALL IS EXPECTED TO FREE BUFFER
251 
252 Return Value:
253 
254  TRUE = it worked, FALSE = it failed (memory)
255 
256 --*/
257 {
258  HCELL_INDEX RootCell;
259  HCELL_INDEX LinkCell;
260  PCM_KEY_NODE LinkKey;
261  PCM_KEY_NODE LinkParent;
262  SIZE_T size;
263  SIZE_T rsize;
264  ULONG KeySize;
265  ULONG ParentSize;
266  PWCHAR p;
267  PCM_KEY_NODE EntryKey;
268 
269  //
270  // First find the link cell.
271  //
272  RootCell = CmHive->Hive.BaseBlock->RootCell;
273  EntryKey = (PCM_KEY_NODE)HvGetCell((PHHIVE)CmHive, RootCell);
274  if( EntryKey == NULL ) {
275  //
276  // we couldn't map a view for the bin containing this cell
277  //
278  return FALSE;
279  }
280  LinkCell = EntryKey->Parent;
281  HvReleaseCell((PHHIVE)CmHive, RootCell);
282 
283  // for master we don't need to count cell usage
284  ASSERT( ((PHHIVE)CmpMasterHive)->ReleaseCellRoutine == NULL );
285  //
286  // Compute the value entry name, which is of the form:
287  // \registry<parent of link node name><link node name>
288  //
289  LinkKey = (PCM_KEY_NODE)HvGetCell((PHHIVE)CmpMasterHive, LinkCell);
290  if( LinkKey == NULL ) {
291  //
292  // we couldn't map a view for the bin containing this cell
293  //
294  return FALSE;
295  }
296  LinkParent = (PCM_KEY_NODE)HvGetCell(
297  (PHHIVE)CmpMasterHive,
298  LinkKey->Parent
299  );
300  if( LinkParent == NULL ) {
301  //
302  // we couldn't map a view for the bin containing this cell
303  //
304  return FALSE;
305  }
306  rsize = wcslen(L"\\REGISTRY\\");
307 
308  KeySize = CmpHKeyNameLen(LinkKey);
309  ParentSize = CmpHKeyNameLen(LinkParent);
310  size = KeySize + ParentSize +
311  (rsize * sizeof(WCHAR)) + sizeof(WCHAR);
312 
313  HiveName->Buffer = ExAllocatePool(PagedPool, size);
314  if (HiveName->Buffer == NULL) {
315  return FALSE;
316  }
317 
318 #pragma prefast(suppress:12005, "no overflow. size is bounded by max key length (512)")
319  HiveName->Length = (USHORT)size;
320  HiveName->MaximumLength = (USHORT)size;
321  p = HiveName->Buffer;
322 
323  RtlCopyMemory(
324  (PVOID)p,
325  (PVOID)L"\\REGISTRY\\",
326  rsize * sizeof(WCHAR)
327  );
328  p += rsize;
329 
330  if (LinkParent->Flags & KEY_COMP_NAME) {
331  CmpCopyCompressedName(p,
332  ParentSize,
333  LinkParent->Name,
334  LinkParent->NameLength);
335  } else {
336  RtlCopyMemory(
337  (PVOID)p,
338  (PVOID)&(LinkParent->Name[0]),
339  ParentSize
340  );
341  }
342 
343  p += ParentSize / sizeof(WCHAR);
344 
345  *p = OBJ_NAME_PATH_SEPARATOR;
346  p++;
347 
348  if (LinkKey->Flags & KEY_COMP_NAME) {
349  CmpCopyCompressedName(p,
350  KeySize,
351  LinkKey->Name,
352  LinkKey->NameLength);
353 
354  } else {
355  RtlCopyMemory(
356  (PVOID)p,
357  (PVOID)&(LinkKey->Name[0]),
358  KeySize
359  );
360  }
361 
362  return TRUE;
363 }
