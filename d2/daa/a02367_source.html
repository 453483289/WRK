33 ;
34 ; BOOLEAN
35 ; KiSwapContext (
36 ; IN PKTHREAD OldThread,
37 ; IN PKTHREAD NewThread
38 ; )
39 ;
40 ; Routine Description:
41 ;
42 ; This function is a small wrapper that marshalls arguments and calls the
43 ; actual swap context routine.
44 ;
45 ; N.B. The old thread lock has been acquired and the dispatcher lock dropped
46 ; before this routine is called.
47 ;
48 ; N.B. The current thread address and the new thread state has been set
49 ; before this routine is called.
50 ;
51 ; Arguments:
52 ;
53 ; OldThread (rcx) - Supplies the address of the old thread.
54 ;
55 ; NewThread (rdx) - Supplies the address of the new thread.
56 ;
57 ; Return Value:
58 ;
59 ; If a kernel APC is pending, then a value of TRUE is returned. Otherwise,
60 ; a value of FALSE is returned.
61 ;
62 ;--
63 
64  NESTED_ENTRY KiSwapContext, _TEXT$00
65 
66  GENERATE_EXCEPTION_FRAME ; generate exception frame
67 
68  mov rbx, gs:[PcCurrentPrcb] ; get current PRCB address
69  mov rdi, rcx ; set old thread address
70  mov rsi, rdx ; set new thread address
71  movzx ecx, byte ptr ThWaitIrql[rdi] ; set APC interrupt bypass disable
72  call SwapContext ; swap context
73 
74  RESTORE_EXCEPTION_STATE ; restore exception state/deallocate
75 
76  ret ; return
77 
78  NESTED_END KiSwapContext, _TEXT$00

84 ;
85 ; This routine is entered as the result of a software interrupt generated
86 ; at DISPATCH_LEVEL. Its function is to process the DPC list, and then

190 ;
191 ; There is a new thread scheduled for execution and the dispatcher lock
192 ; has been acquired. Context switch to the new thread immediately.
193 ;
194 ; N.B. The following routine is entered by falling through from the above
195 ; routine.
196 ;
197 ; N.B. The following routine is carefully written as a nested function that
198 ; appears to have been called directly by the caller of the above
199 ; function which processes the dispatch interrupt.
200 ;
201 ; Arguments:
202 ;
203 ; None.
204 ;
205 
206  NESTED_ENTRY KxDispatchInterrupt, _TEXT$00
207 
208  GENERATE_EXCEPTION_FRAME ; generate exception frame
209 
210  mov rbx, gs:[PcCurrentPrcb] ; get current PRCB address
211  mov rdi, PbCurrentThread[rbx] ; get old thread address
212 
213 ;
214 ; Save current MXCSR and set initial value so swap context will be entered
215 ; with a canonical MXCSR value.
216 ;
217 
218  stmxcsr ExMxCsr[rsp] ; save current MXCSR value
219  ldmxcsr gs:[PcMxCsr] ; set default MXCSR value
220 
221 ;
222 ; Raise IRQL to SYNCH level, set context swap busy for the old thread, and
223 ; acquire the current PRCB lock.
224 ;
225 
226 ifndef NT_UP
227 
228  mov ecx, SYNCH_LEVEL ; set IRQL to SYNCH level
229 
230  SetIrql ;
231 
232  mov byte ptr ThSwapbusy[rdi], 1 ; set context swap busy
233 
234  AcquireSpinLock PbPrcbLock[rbx] ; acquire current PRCB lock
235 
236 endif
237 
238 ;
239 ; Get the next thread address, set the thread state to running, queue the old
240 ; running thread, and swap context to the next thread.
241 ;
242 
243  mov rsi, PbNextThread[rbx] ; get next thread address
244  and qword ptr PbNextThread[rbx], 0 ; clear next thread address
245  mov PbCurrentThread[rbx], rsi ; set current thread address
246  mov byte ptr ThState[rsi], Running ; set new thread state
247  mov byte ptr ThWaitReason[rdi], WrDispatchInt ; set wait reason
248  mov rcx, rdi ; set address of old thread
249  mov rdx, rbx ; set address of current PRCB
250  call KiQueueReadyThread ; queue ready thread for execution
251  mov ecx, APC_LEVEL ; set APC interrupt bypass disable
252  call SwapContext ; call context swap routine
253 
254  ldmxcsr ExMxCsr[rsp] ; restore current MXCSR value
255 
256  RESTORE_EXCEPTION_STATE ; restore exception state/deallocate
257 
258  ret ; return
259 
260  NESTED_END KxDispatchInterrupt, _TEXT$00

262  subttl "Swap Context"
263 ;++
264 ;
265 ; Routine Description:
266 ;
267 ; This routine is called to swap context from one thread to the next. It
268 ; swaps context, flushes the translation buffer, swaps the process address
269 ; space if necessary, and returns to its caller.
270 ;
271 ; N.B. This routine is only called by code within this module and the idle
272 ; thread code and uses special register calling conventions.
273 ;
274 ; Arguments:
275 ;
276 ; al - Supplies a boolean value that determines whether the full legacy
277 ; floating state needs to be saved.
278 ;
279 ; cl - Supplies the APC interrupt bypass disable IRQL value.
280 ;
281 ; rbx - Supplies the address of the current PRCB.
282 ;
283 ; rdi - Supplies the address of previous thread.
284 ;
285 ; rsi - Supplies the address of next thread.
286 ;
287 ; Return value:
288 ;
289 ; al - Supplies the kernel APC pending flag.
290 ;
291 ; rbx - Supplies the address of the current PRCB.
292 ;
293 ; rsi - Supplies the address of current thread.
294 ;
295 ;--
296 
297  NESTED_ENTRY SwapContext, _TEXT$00
298 
299  alloc_stack (KSWITCH_FRAME_LENGTH - (1 * 8)) ; allocate stack frame
300  save_reg rbp, SwRbp ; save nonvolatile register
301 
302  END_PROLOGUE
303 
304  mov SwApcBypass[rsp], cl ; save APC bypass disable
305 
306 ;
307 ; Wait for new thread lock to be dropped.
308 ;
309 ; N.B. It is necessary to wait for the new thread context to be swapped so
310 ; that any context switch away from the new thread on another processor
311 ; is completed before attempting to swap context context to the thread.
312 ;
313 
314 ifndef NT_UP
315 
316 KiSC00: cmp byte ptr ThSwapBusy[rsi], 0 ; check if swap busy for new thread
317 
318  Yield ; yield processor execution
319 
320  jne short KiSC00 ; if ne, context busy for new thread
321 
322 endif
323 
324 ;
325 ; Increment the number of context switches on this processor.
326 ;
327 ; N.B. This increment is done here is force the cache block containing the
328 ; context switch count into the cache as write exclusive. There are
329 ; several other references to this cache block in the following code.
330 ;
331 
332  inc dword ptr PbContextSwitches[rbx] ; processor count
333 
334 ;
335 ; Check for context swap logging.
336 ;
337 
338  mov rax, (PcPerfGlobalGroupMask - PcPrcb)[rbx] ; get global mask address
339  test rax, rax ; test if logging enabled
340  je short KiSC05 ; if e, logging not enabled
341  test dword ptr PERF_CONTEXTSWAP_OFFSET[rax], PERF_CONTEXTSWAP_FLAG ; check flag
342  jz short KiSC05 ; if z, context swap events not enabled
343  mov rcx, rdi ; set address of old thread
344  mov rdx, rsi ; set address of new thread
345  call WmiTraceContextSwap ; call trace routine
346 
347 ;
348 ; If the current thread NPX state is switch, then save the legacy floating
349 ; point state.
350 ;
351 
352 KiSC05: cmp byte ptr ThNpxState[rdi], LEGACY_STATE_SWITCH ; check if switched
353  jne short KiSC10 ; if ne, legacy state not switched
354  mov rbp, ThInitialStack[rdi] ; get previous thread initial stack
355  fxsave [rbp] ; save legacy floating point state
356 
357 ;
358 ; Switch kernel stacks.
359 ;
360 
361 KiSC10: mov ThKernelStack[rdi], rsp ; save old kernel stack pointer
362  mov rsp, ThKernelStack[rsi] ; get new kernel stack pointer
363

365 ; Swap the process address space if the new process is not the same as the
366 ; previous process.
367 ;
368 
369  mov r14, ThApcState + AsProcess[rsi] ; get new process address
370  cmp r14, ThApcState + AsProcess[rdi] ; check if process match
371  je short KiSC20 ; if e, process addresses match

373 ;

375 ;
376 
377 ifndef NT_UP

380  mov rcx, PbSetMember[rbx] ; get processor set member
381  lock xor PrActiveProcessors[rdx], rcx ; clear bit in previous set

383 if DBG

387  int 3 ; debug break - incorrect active mask

389 
390 endif

392 endif

396 ;

399

402 if DBG
403

441 ;
442 ; If the thread requires switching the legacy state, then restore the
443 ; legacy floating state, load the compatilbity mode TEB address, and reload
444 ; the natvie user mode TEB address and segment registers.
445 ;

451 ;
452 ; Set base of compatibility mode TEB.
453 ;
454 ; N.B. The upper 32-bits of the compatibility mode TEB address are always
455 ; zero.
456 ;
457 
458  mov eax, ThTeb[rsi] ; compute compatibility mode TEB address
459  add eax, CmThreadEnvironmentBlockOffset ;
460  mov rcx, (PcGdt - PcPrcb)[rbx] ; get GDT base address
461  mov KgdtBaseLow + KGDT64_R3_CMTEB[rcx], ax ; set CMTEB base address
462  shr eax, 16 ;
463  mov KgdtBaseMiddle + KGDT64_R3_CMTEB[rcx], al ;
464  mov KgdtBaseHigh + KGDT64_R3_CMTEB[rcx], ah ;

466 ;
467 ; If the user segment selectors have been changed, then reload them with
468 ; their cannonical values.
469 ;
470 ; N.B. The following code depends on the values defined in ntamd64.w that
471 ; can be loaded in ds, es, fs, and gs. In particular an "and" operation
472 ; is used for the below comparison.
473 ;
474 
475  mov eax, ds ; compute sum of segment selectors
476  mov ecx, es ;
477  and eax, ecx ;
478  mov ecx, gs ;
479  and eax, ecx ;
480  cmp ax, (KGDT64_R3_DATA or RPL_MASK) ; check if sum matches
481  je short KiSC25 ; if e, sum matches expected value
482  mov ecx, KGDT64_R3_DATA or RPL_MASK ; reload user segment selectors
483  mov ds, ecx ;
484  mov es, ecx ;
485 
486 ;
487 ; N.B. The following reload of the GS selector destroys the system MSR_GS_BASE
488 ; register. Thus this sequence must be done with interrupt off.
489 ;
490 
491  mov eax, (PcSelf - PcPrcb)[rbx] ; get current PCR address
492  mov edx, (PcSelf - PcPrcb + 4)[rbx] ;
493  cli ; disable interrupts
494  mov gs, ecx ; reload GS segment selector
495  mov ecx, MSR_GS_BASE ; get GS base MSR number
496  wrmsr ; write system PCR base address
497  sti ; enable interrupts
498 KiSC25: mov eax, KGDT64_R3_CMTEB or RPL_MASK ; reload FS segment selector
499  mov fs, eax ;
500  mov eax, ThTeb[rsi] ; get low part of user TEB address
501  mov edx, ThTeb + 4[rsi] ; get high part of user TEB address
502  mov (PcTeb - PcPrcb)[rbx], eax ; set user TEB address in PCR
503  mov (PcTeb - PcPrcb + 4)[rbx], edx ;
504  mov ecx, MSR_GS_SWAP ; get GS base swap MSR number
505  wrmsr ; write user TEB base address

522 ; interrupt if APC bypass is disabled.
523 ;

525  cmp byte ptr ThApcState + AsKernelApcPending[rsi], TRUE ; check if APC pending
