182 //
183 // Macros for Upper Casing a Unicode Code Point.
184 //
185 
186 #define LOBYTE(w) ((UCHAR)((w)))
187 #define HIBYTE(w) ((UCHAR)(((USHORT)((w)) >> 8) & 0xFF))
188 #define GET8(w) ((ULONG)(((w) >> 8) & 0xff))
189 #define GETHI4(w) ((ULONG)(((w) >> 4) & 0xf))
190 #define GETLO4(w) ((ULONG)((w) & 0xf))
191 
192 /***************************************************************************\
193 * TRAVERSE844W
194 *
195 * Traverses the 8:4:4 translation table for the given wide character. It
196 * returns the final value of the 8:4:4 table, which is a WORD in length.
197 *
198 * Broken Down Version:
199 * --------------------
200 * Incr = pTable[GET8(wch)];
201 * Incr = pTable[Incr + GETHI4(wch)];
202 * Value = pTable[Incr + GETLO4(wch)];
203 *
204 * DEFINED AS A MACRO.
205 *
206 \***************************************************************************/
207 
208 #define TRAVERSE844W(pTable, wch) \
209  ( (pTable)[(pTable)[(pTable)[GET8((wch))] + GETHI4((wch))] + GETLO4((wch))] )
210 
211 //
212 // NLS_UPCASE - Based on macros in nls.h
213 //
214 // We will have this upcase macro quickly shortcircuit out if the value
215 // is within the normal ANSI range (i.e., < 127). We actually won't bother
216 // with the 5 values above 'z' because they won't happen very often and
217 // coding it this way lets us get out after 1 compare for value less than
218 // 'a' and 2 compares for lowercase a-z.
219 //
220 
221 #define NLS_UPCASE(wch) ( \
222  ((wch) < 'a' ? \
223  (wch) \
224  : \
225  ((wch) <= 'z' ? \
226  (wch) - ('a'-'A') \
227  : \
228  ((WCHAR)((wch) + TRAVERSE844W(Nls844UnicodeUpcaseTable,(wch)))) \
229  ) \
230  ) \
231 )
232 
233 #define NLS_DOWNCASE(wch) ( \
234  ((wch) < 'A' ? \
235  (wch) \
236  : \
237  ((wch) <= 'Z' ? \
238  (wch) + ('a'-'A') \
239  : \
240  ((WCHAR)((wch) + TRAVERSE844W(Nls844UnicodeLowercaseTable,(wch)))) \
241  ) \
242  ) \
243 )
