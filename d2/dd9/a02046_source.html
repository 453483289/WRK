38 /*++
39 
40 Routine Description:
41 
42  This routine interfaces to the NT Object Manager. It is invoked when
43  a Key object (or Key Root object) is closed.
44 
45  It's function is to do cleanup processing by waking up any notifies
46  pending on the handle. This keeps the key object from hanging around
47  forever because a synchronous notify is stuck on it somewhere.
48 
49  All other cleanup, in particular, the freeing of storage, will be
50  done in CmpDeleteKeyObject.
51 
52 Arguments:
53 
54  Process - ignored
55 
56  Object - supplies a pointer to a KeyRoot or Key, thus -> KEY_BODY.
57 
58  GrantedAccess, ProcessHandleCount, SystemHandleCount - ignored
59 
60 Return Value:
61 
62  NONE.
63 
64 --*/

66  PCM_KEY_BODY KeyBody;

68

70

72  UNREFERENCED_PARAMETER (GrantedAccess);
73  UNREFERENCED_PARAMETER (ProcessHandleCount);
74 
75  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_POOL,"CmpCloseKeyObject: Object = %p\n", Object));
76 
77  if( SystemHandleCount > 1 ) {
78  //
79  // There are still has open handles on this key. Do nothing
80  //
81  return;
82  }
83

85 
86  //

100  //

102  if (KeyBody->NotifyBlock != NULL) {
103  //
104  // Post all PostBlocks waiting on the NotifyBlock
105  //
106  NotifyBlock = KeyBody->NotifyBlock;
107  if (IsListEmpty(&(NotifyBlock->PostList)) == FALSE) {
108  LIST_ENTRY DelayedDeref;
109  //
