34 HvFreeHive(
35  PHHIVE Hive

37 /*++
38 
39 Routine Description:
40 
41  Free all of the pieces of a hive.
42 
43 Arguments:
44 
45  Hive - supplies a pointer to hive control structure for hive to free.
46  this structure itself will NOT be freed, but everything it
47  points to will.
48 
49 Return Value:
50 
51  NONE.
52 
53 --*/
54 {
55  PHMAP_DIRECTORY Dir;
56  PHMAP_ENTRY Me;
57  HCELL_INDEX Address;
58  ULONG Type;
59  ULONG Length;
60  PHBIN Bin;
61  ULONG Tables;
62  PFREE_HBIN FreeBin;
63 
64  ASSERT(Hive->Flat == FALSE);
65  ASSERT(Hive->ReadOnly == FALSE);
66  ASSERT(Stable == 0);
67  ASSERT(Volatile == 1);
68 
69  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"HvFreeHive(%ws) :\n", Hive->BaseBlock->FileName));
70  //
71  // Iterate through both types of storage
72  //
73  for (Type = 0; Type <= Volatile; Type++) {
74 
75  Address = HCELL_TYPE_MASK * Type;
76  Length = Hive->Storage[Type].Length + (HCELL_TYPE_MASK * Type);
77 
78  if( Hive->Storage[Type].Map && (Length > (HCELL_TYPE_MASK * Type)) ) {
79 
80  //
81  // Sweep through bin set
82  //
83  do {
84  Me = HvpGetCellMap(Hive, Address);
85  VALIDATE_CELL_MAP(__LINE__,Me,Hive,Address);
86  if (Me->BinAddress & HMAP_DISCARDABLE) {
87  //
88  // hbin is either discarded or discardable, check the tombstone
89  //
90  FreeBin = (PFREE_HBIN)Me->BlockAddress;
91  Address += FreeBin->Size;
92  if (FreeBin->Flags & FREE_HBIN_DISCARDABLE) {
93  CmpFree((PHBIN)HBIN_BASE(Me->BinAddress), FreeBin->Size);
94  } else if(Me->BinAddress & HMAP_INPAGEDPOOL) {
95  //
96  // The bin has been freed, but quota is still charged.
97  // Since the hive is being freed, the quota must be
98  // returned here.
99  //
100  CmpReleaseGlobalQuota(FreeBin->Size);
101  }
102 
103  CmpFree(FreeBin, sizeof(FREE_HBIN));
104  } else {
105  if( Me->BinAddress & HMAP_INPAGEDPOOL ) {
106  ASSERT( Me->BinAddress & HMAP_INPAGEDPOOL );
107 
108  Bin = (PHBIN)HBIN_BASE(Me->BinAddress);
109  Address += HvpGetBinMemAlloc(Hive,Bin,Type);
110 #if DBG
111  if( Type == Stable ) {
112  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"HvFreeHive: BinAddress = 0x%p\t Size = 0x%lx\n", Bin, HvpGetBinMemAlloc(Hive,Bin,Type)));
113  }
114 #endif
115 
116  //
117  // free the actual bin only if it is allocated from paged pool
118  //
119  if(HvpGetBinMemAlloc(Hive,Bin,Type)) {
120  CmpFree(Bin, HvpGetBinMemAlloc(Hive,Bin,Type));
121  }
122  } else {
123  //
124  // bin was mapped into view; advance carefully
125  //
126  Address += HBLOCK_SIZE;
127  }
128 
129  }
130  } while (Address < Length);
131 
132  //
133  // Free map table storage
134  //
135  ASSERT(Hive->Storage[Type].Length != (HCELL_TYPE_MASK * Type));
136  Tables = (((Hive->Storage[Type].Length) / HBLOCK_SIZE)-1) / HTABLE_SLOTS;
137  Dir = Hive->Storage[Type].Map;
138  HvpFreeMap(Hive, Dir, 0, Tables);
139 
140  if (Tables > 0) {
141  CmpFree(Hive->Storage[Type].Map, sizeof(HMAP_DIRECTORY)); // free dir if it exists
142  }
143  }
144  Hive->Storage[Type].Length = 0;
145  }
146 
147  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"\n"));
148  //
149  // Free the base block
150  //
151  (Hive->Free)(Hive->BaseBlock,Hive->BaseBlockAlloc);
152  Hive->BaseBlock = NULL;
153 
154  //
155  // Free the dirty vector
156  //
157  if (Hive->DirtyVector.Buffer != NULL) {
158  CmpFree((PVOID)(Hive->DirtyVector.Buffer), Hive->DirtyAlloc);
159  }
160 
161  HvpFreeHiveFreeDisplay(Hive);
162  LogHiveFree(Hive);
163  return;
164 }

166 VOID
167 HvFreeHivePartial(
168  PHHIVE Hive,
169  HCELL_INDEX Start,
170  HSTORAGE_TYPE Type
171  )
172 /*++
173 
174 Routine Description:
175 
176  Free the memory and associated maps for the end of a hive
177  starting at Start. The baseblock, hive, etc will not be touched.
178 
179 Arguments:
180 
181  Hive - supplies a pointer to hive control structure for hive to
182  partially free.
183 
184  Start - HCELL_INDEX of first bin to free, will free from this
185  bin (inclusive) to the end of the hives stable storage.
186 
187  Type - Type of storage (Stable or Volatile) to be freed.
188 
189 Return Value:
190 
191  NONE.
192 
193 --*/
194 {
195  PHMAP_DIRECTORY Dir;
196  PHMAP_ENTRY Me;
197  HCELL_INDEX Address;
198  ULONG StartTable;
199  ULONG Length;
200  PHBIN Bin;
201  ULONG Tables;
202  ULONG FirstBit;
203  ULONG LastBit;
204  PFREE_HBIN FreeBin;
205 
206  ASSERT(Hive->Flat == FALSE);
207  ASSERT(Hive->ReadOnly == FALSE);
208 
209  Address = Start;
210  Length = Hive->Storage[Type].Length;
211  ASSERT(Address <= Length);
212 
213  if (Address == Length) {
214  return;
215  }
216 
217  //
218  // Sweep through bin set
219  //
220  do {
221  Me = HvpGetCellMap(Hive, Address + (Type*HCELL_TYPE_MASK));
222  VALIDATE_CELL_MAP(__LINE__,Me,Hive,Address + (Type*HCELL_TYPE_MASK));
223  if (Me->BinAddress & HMAP_DISCARDABLE) {
224  FreeBin = (PFREE_HBIN)Me->BlockAddress;
225  if (FreeBin->Flags & FREE_HBIN_DISCARDABLE) {
226  CmpFree((PVOID)HBIN_BASE(Me->BinAddress), FreeBin->Size);
227  } else {
228  //
229  // The bin has been freed, but quota is still charged.
230  // Since the file will now shrink, the quota must be
231  // returned here.
232  //
233  if( Me->BinAddress & HMAP_INPAGEDPOOL) {
234  //
235  // we charge quota only for bins in paged-pool
236  //
237  CmpReleaseGlobalQuota(FreeBin->Size);
238  }
239  }
240  RemoveEntryList(&FreeBin->ListEntry);
241  Address += FreeBin->Size;
242  CmpFree(FreeBin, sizeof(FREE_HBIN));
243 
244  } else {
245  Bin = (PHBIN)HBIN_BASE(Me->BinAddress);
246  Address += HvpGetBinMemAlloc(Hive,Bin,Type);
247 
248  if( Me->BinAddress & HMAP_INPAGEDPOOL && HvpGetBinMemAlloc(Hive,Bin,Type) ) {
249  //
250  // free the bin only if it is allocated from paged pool
251  //
252  CmpFree(Bin, HvpGetBinMemAlloc(Hive,Bin,Type));
253  }
254  }
255  } while (Address < Length);
256 
257  //
258  // Free map table storage
259  //
260  Tables = (((Hive->Storage[Type].Length) / HBLOCK_SIZE) - 1) / HTABLE_SLOTS;
261  Dir = Hive->Storage[Type].Map;
262  if (Start > 0) {
263  StartTable = ((Start-1) / HBLOCK_SIZE) / HTABLE_SLOTS;
264  } else {
265  StartTable = (ULONG)-1;
266  }
267  HvpFreeMap(Hive, Dir, StartTable+1, Tables);
268 
269  //
270  // update hysteresis (eventually queue work item)
271  //
272  if( Type == Stable) {
273  CmpUpdateSystemHiveHysteresis(Hive,(Start&(~HCELL_TYPE_MASK)),Hive->Storage[Type].Length);
274  }
275 
276  Hive->Storage[Type].Length = (Start&(~HCELL_TYPE_MASK));
277 
278  if (Type==Stable) {
279  //
280  // Clear dirty vector for data past Hive->Storage[Stable].Length
281  //
282  FirstBit = Start / HSECTOR_SIZE;
283  LastBit = Hive->DirtyVector.SizeOfBitMap;
284  ASSERT(Hive->DirtyCount == RtlNumberOfSetBits(&Hive->DirtyVector));
285  RtlClearBits(&Hive->DirtyVector, FirstBit, LastBit-FirstBit);
286  Hive->DirtyCount = RtlNumberOfSetBits(&Hive->DirtyVector);
287  }
288 
289  HvpAdjustHiveFreeDisplay(Hive,Hive->Storage[Type].Length,Type);
290 
291  return;
292 }
