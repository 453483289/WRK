28 NtQuerySection(
29  __in HANDLE SectionHandle,
30  __in SECTION_INFORMATION_CLASS SectionInformationClass,
31  __out_bcount(SectionInformationLength) PVOID SectionInformation,
32  __in SIZE_T SectionInformationLength,
33  __out_opt PSIZE_T ReturnLength
34  )
35 
36 /*++
37 
38 Routine Description:
39 
40  This function provides the capability to determine the base address,
41  size, granted access, and allocation of an opened section object.
42 
43 Arguments:
44 
45  SectionHandle - Supplies an open handle to a section object.
46 
47  SectionInformationClass - The section information class about
48  which to retrieve information.
49 
50  SectionInformation - A pointer to a buffer that receives the
51  specified information. The format and content of the
52  buffer depend on the specified section class.
53 
54  SectionInformation Format by Information Class:
55 
56  SectionBasicInformation - Data type is PSECTION_BASIC_INFORMATION.
57 
58  SECTION_BASIC_INFORMATION Structure
59 
60  PVOID BaseAddress - The base virtual address of the
61  section if the section is based.
62 
63  LARGE_INTEGER MaximumSize - The maximum size of the section in
64  bytes.
65 
66  ULONG AllocationAttributes - The allocation attributes flags.
67 
68  AllocationAttributes Flags
69 
70  SEC_BASED - The section is a based section.
71 
72  SEC_FILE - The section is backed by a data file.
73 
74  SEC_RESERVE - All pages of the section were initially
75  set to the reserved state.
76 
77  SEC_COMMIT - All pages of the section were initially
78  to the committed state.
79 
80  SEC_IMAGE - The section was mapped as an executable image file.
81 
82  SECTION_IMAGE_INFORMATION
83 
84  SectionInformationLength - Specifies the length in bytes of the
85  section information buffer.
86 
87  ReturnLength - An optional pointer which, if specified, receives the
88  number of bytes placed in the section information buffer.
89 
90 
91 Return Value:
92 
93  NTSTATUS.
94 
95 --*/
96 
97 {
98  NTSTATUS Status;
99  PSECTION Section;
100  KPROCESSOR_MODE PreviousMode;
101 
102  PAGED_CODE();
103 
104  //
105  // Get previous processor mode and probe output argument if necessary.
106  //
107 
108  PreviousMode = KeGetPreviousMode();
109  if (PreviousMode != KernelMode) {
110 
111  //
112  // Check arguments.
113  //
114 
115  try {
116 
117  ProbeForWrite(SectionInformation,
118  SectionInformationLength,
119  sizeof(ULONG));
120 
121  if (ARGUMENT_PRESENT (ReturnLength)) {
122  ProbeForWriteUlong_ptr (ReturnLength);
123  }
124 
125  } except (EXCEPTION_EXECUTE_HANDLER) {
126 
127  //
128  // If an exception occurs during the probe or capture
129  // of the initial values, then handle the exception and
130  // return the exception code as the status value.
131  //
132 
133  return GetExceptionCode();
134  }
135  }
136 
137  //
138  // Check argument validity.
139  //
140 
141  if ((SectionInformationClass != SectionBasicInformation) &&
142  (SectionInformationClass != SectionImageInformation)) {
143  return STATUS_INVALID_INFO_CLASS;
144  }
145 
146  if (SectionInformationClass == SectionBasicInformation) {
147  if (SectionInformationLength < (ULONG)sizeof(SECTION_BASIC_INFORMATION)) {
148  return STATUS_INFO_LENGTH_MISMATCH;
149  }
150  }
151  else {
152  if (SectionInformationLength < (ULONG)sizeof(SECTION_IMAGE_INFORMATION)) {
153  return STATUS_INFO_LENGTH_MISMATCH;
154  }
155  }
156 
157  //
158  // Reference section object by handle for READ access, get the information
159  // from the section object, dereference the section
160  // object, fill in information structure, optionally return the length of
161  // the information structure, and return service status.
162  //
163 
164  Status = ObReferenceObjectByHandle (SectionHandle,
165  SECTION_QUERY,
166  MmSectionObjectType,
167  PreviousMode,
168  (PVOID *)&Section,
169  NULL);
170 
171  if (NT_SUCCESS(Status)) {
172 
173  try {
174 
175  if (SectionInformationClass == SectionBasicInformation) {
176  ((PSECTION_BASIC_INFORMATION)SectionInformation)->BaseAddress =
177  (PVOID)Section->Address.StartingVpn;
178 
179  ((PSECTION_BASIC_INFORMATION)SectionInformation)->MaximumSize =
180  Section->SizeOfSection;
181 
182  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes =
183  0;
184 
185  if (Section->u.Flags.Image) {
186  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes =
187  SEC_IMAGE;
188  }
189  if (Section->u.Flags.Based) {
190  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes |=
191  SEC_BASED;
192  }
193  if (Section->u.Flags.File) {
194  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes |=
195  SEC_FILE;
196  }
197  if (Section->u.Flags.NoCache) {
198  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes |=
199  SEC_NOCACHE;
200  }
201  if (Section->u.Flags.Reserve) {
202  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes |=
203  SEC_RESERVE;
204  }
205  if (Section->u.Flags.Commit) {
206  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes |=
207  SEC_COMMIT;
208  }
209  if (Section->Segment->ControlArea->u.Flags.GlobalMemory) {
210  ((PSECTION_BASIC_INFORMATION)SectionInformation)->AllocationAttributes |=
211  SEC_GLOBAL;
212  }
213 
214  if (ARGUMENT_PRESENT(ReturnLength)) {
215  *ReturnLength = sizeof(SECTION_BASIC_INFORMATION);
216  }
217  }
218  else {
219 
220  if (Section->u.Flags.Image == 0) {
221  Status = STATUS_SECTION_NOT_IMAGE;
222  }
223  else {
224  *((PSECTION_IMAGE_INFORMATION)SectionInformation) =
225  *Section->Segment->u2.ImageInformation;
226 
227  if (ARGUMENT_PRESENT(ReturnLength)) {
228  *ReturnLength = sizeof(SECTION_IMAGE_INFORMATION);
229  }
230  }
231  }
232 
233  } except (EXCEPTION_EXECUTE_HANDLER) {
234  Status = GetExceptionCode ();
235  }
236 
237  ObDereferenceObject ((PVOID)Section);
238  }
239  return Status;
240 }
