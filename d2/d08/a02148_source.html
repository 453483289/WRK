336 FsRtlGetCompatibilityModeValue (
337  IN PUNICODE_STRING ValueName,
338  IN OUT PULONG Value
339  )
340 
341 /*++
342 
343 Routine Description:
344 
345  Given a unicode value name this routine will go into the registry
346  location for the Chicago compatibility mode information and get the
347  value.
348 
349 Arguments:
350 
351  ValueName - the unicode name for the registry value located in the
352  double space configuration location of the registry.
353  Value - a pointer to the ULONG for the result.
354 
355 Return Value:
356 
357  NTSTATUS
358 
359  If STATUS_SUCCESSFUL is returned, the location *Value will be
360  updated with the DWORD value from the registry. If any failing
361  status is returned, this value is untouched.
362 
363 --*/
364 
365 {
366  HANDLE Handle;
367  NTSTATUS Status;
368  ULONG RequestLength;
369  ULONG ResultLength;
370  UCHAR Buffer[KEY_WORK_AREA];
371  UNICODE_STRING KeyName;
372  OBJECT_ATTRIBUTES ObjectAttributes;
373  PKEY_VALUE_FULL_INFORMATION KeyValueInformation;
374 
375  KeyName.Buffer = COMPATIBILITY_MODE_KEY_NAME;
376  KeyName.Length = sizeof(COMPATIBILITY_MODE_KEY_NAME) - sizeof(WCHAR);
377  KeyName.MaximumLength = sizeof(COMPATIBILITY_MODE_KEY_NAME);
378 
379  InitializeObjectAttributes(&ObjectAttributes,
380  &KeyName,
381  OBJ_CASE_INSENSITIVE,
382  NULL,
383  NULL);
384 
385  Status = ZwOpenKey(&Handle,
386  KEY_READ,
387  &ObjectAttributes);
388 
389  if (!NT_SUCCESS(Status)) {
390 
391  return Status;
392  }
393 
394  RequestLength = KEY_WORK_AREA;
395 
396  KeyValueInformation = (PKEY_VALUE_FULL_INFORMATION)Buffer;
397 
398  while (1) {
399 
400  Status = ZwQueryValueKey(Handle,
401  ValueName,
402  KeyValueFullInformation,
403  KeyValueInformation,
404  RequestLength,
405  &ResultLength);
406 
407  ASSERT( Status != STATUS_BUFFER_OVERFLOW );
408 
409  if (Status == STATUS_BUFFER_OVERFLOW) {
410 
411  //
412  // Try to get a buffer big enough.
413  //
414 
415  if (KeyValueInformation != (PKEY_VALUE_FULL_INFORMATION)Buffer) {
416 
417  ExFreePool(KeyValueInformation);
418  }
419 
420  RequestLength += 256;
421 
422  KeyValueInformation = (PKEY_VALUE_FULL_INFORMATION)
423  ExAllocatePoolWithTag(PagedPool,
424  RequestLength,
425  ' taF');
426 
427  if (!KeyValueInformation) {
428  return STATUS_NO_MEMORY;
429  }
430 
431  } else {
432 
433  break;
434  }
435  }
436 
437  ZwClose(Handle);
438 
439  if (NT_SUCCESS(Status)) {
440 
441  if (KeyValueInformation->DataLength != 0) {
442 
443  PULONG DataPtr;
444 
445  //
446  // Return contents to the caller.
447  //
448 
449  DataPtr = (PULONG)
450  ((PUCHAR)KeyValueInformation + KeyValueInformation->DataOffset);
451  *Value = *DataPtr;
452 
453  } else {
454 
455  //
456  // Treat as if no value was found
457  //
458 
459  Status = STATUS_OBJECT_NAME_NOT_FOUND;
460  }
461  }
462 
463  if (KeyValueInformation != (PKEY_VALUE_FULL_INFORMATION)Buffer) {
464 
465  ExFreePool(KeyValueInformation);
466  }
467 
468  return Status;
469 }
