42 /*++
43 
44 Routine Description:
45 
46  This service opens or creates a file, or opens a device. It is used to
47  establish a file handle to the open device/file that can then be used
48  in subsequent operations to perform I/O operations on. For purposes of
49  readability, files and devices are treated as "files" throughout the
50  majority of this module and the system service portion of the I/O system.
51  The only time a distinction is made is when it is important to determine
52  which is really being accessed. Then a distinction is also made in the
53  comments.
54 
55 Arguments:
56 
57  FileHandle - A pointer to a variable to receive the handle to the open file.
58 
59  DesiredAccess - Supplies the types of access that the caller would like to
60  the file.
61 
62  ObjectAttributes - Supplies the attributes to be used for file object (name,
63  SECURITY_DESCRIPTOR, etc.)
64 
65  IoStatusBlock - Specifies the address of the caller's I/O status block.
66 
67  AllocationSize - Initial size that should be allocated to the file. This
68  parameter only has an affect if the file is created. Further, if
69  not specified, then it is taken to mean zero.
70 
71  FileAttributes - Specifies the attributes that should be set on the file,
72  if it is created.
73 
74  ShareAccess - Supplies the types of share access that the caller would like
75  to the file.
76 
77  CreateDisposition - Supplies the method for handling the create/open.
78 
79  CreateOptions - Caller options for how to perform the create/open.
80 
81  EaBuffer - Optionally specifies a set of EAs to be applied to the file if
82  it is created.
83 
84  EaLength - Supplies the length of the EaBuffer.
85 
86 Return Value:
87 
88  The function value is the final status of the create/open operation.
89 
90 --*/
91 
92 {
93  //
94  // Simply invoke the common I/O file creation routine to do the work.
95  //
96 
97  PAGED_CODE();
98 
99  return IoCreateFile( FileHandle,
100  DesiredAccess,
101  ObjectAttributes,
102  IoStatusBlock,
103  AllocationSize,
104  FileAttributes,
105  ShareAccess,
106  CreateDisposition,
107  CreateOptions,
108  EaBuffer,
109  EaLength,
110  CreateFileTypeNone,
111  (PVOID)NULL,
112  0 );
113 }

133 /*++
134 
135 Routine Description:
136 
137  Creates and opens the server end handle of the first instance of a
138  specific named pipe or another instance of an existing named pipe.
139 
140 Arguments:
141 
142  FileHandle - Supplies a handle to the file on which the service is being
143  performed.
144 
145  DesiredAccess - Supplies the types of access that the caller would like to
146  the file.
147 
148  ObjectAttributes - Supplies the attributes to be used for file object
149  (name, SECURITY_DESCRIPTOR, etc.)
150 
151  IoStatusBlock - Address of the caller's I/O status block.
152 
153  ShareAccess - Supplies the types of share access that the caller would
154  like to the file.
155 
156  CreateDisposition - Supplies the method for handling the create/open.
157 
158  CreateOptions - Caller options for how to perform the create/open.
159 
160  NamedPipeType - Type of named pipe to create (Bitstream or message).
161 
162  ReadMode - Mode in which to read the pipe (Bitstream or message).
163 
164  CompletionMode - Specifies how the operation is to be completed.
165 
166  MaximumInstances - Maximum number of simultaneous instances of the named
167  pipe.
168 
169  InboundQuota - Specifies the pool quota that is reserved for writes to the
170  inbound side of the named pipe.
171 
172  OutboundQuota - Specifies the pool quota that is reserved for writes to
173  the inbound side of the named pipe.
174 
175  DefaultTimeout - Optional pointer to a timeout value that is used if a
176  timeout value is not specified when waiting for an instance of a named
177  pipe.
178 
179 Return Value:
180 
181  The function value is the final status of the create/open operation.
182 
183 --*/
184 
185 {
186  NAMED_PIPE_CREATE_PARAMETERS namedPipeCreateParameters;
187 
188  PAGED_CODE();
189 
190  //
191  // Check whether or not the DefaultTimeout parameter was specified. If
192  // so, then capture it in the named pipe create parameter structure.
193  //
194 
195  if (ARGUMENT_PRESENT( DefaultTimeout )) {
196 
197  //
198  // Indicate that a default timeout period was specified.
199  //
200 
201  namedPipeCreateParameters.TimeoutSpecified = TRUE;
202 
203  //
204  // A default timeout parameter was specified. Check to see whether
205  // the caller's mode is kernel and if not capture the parameter inside
206  // of a try...except clause.
207  //
208 
209  if (KeGetPreviousMode() != KernelMode) {
210  try {
211  ProbeForReadSmallStructure ( DefaultTimeout,
212  sizeof( LARGE_INTEGER ),
213  sizeof( ULONG ) );
214  namedPipeCreateParameters.DefaultTimeout = *DefaultTimeout;
215  } except(EXCEPTION_EXECUTE_HANDLER) {
216 
217  //
218  // Something went awry attempting to access the parameter.
219  // Get the reason for the error and return it as the status
220  // value from this service.
221  //
222 
223  return GetExceptionCode();
224  }
225  } else {
226 
227  //
228  // The caller's mode was kernel so simply store the parameter.
229  //
230 
231  namedPipeCreateParameters.DefaultTimeout = *DefaultTimeout;
232  }
233  } else {
234 
235  //
236  // Indicate that no default timeout period was specified.
237  //
238 
239  namedPipeCreateParameters.TimeoutSpecified = FALSE;
240  }
241 
242  //
243  // Store the remainder of the named pipe-specific parameters in the
244  // structure for use in the call to the common create file routine.
245  //
246 
247  namedPipeCreateParameters.NamedPipeType = NamedPipeType;
248  namedPipeCreateParameters.ReadMode = ReadMode;
249  namedPipeCreateParameters.CompletionMode = CompletionMode;
250  namedPipeCreateParameters.MaximumInstances = MaximumInstances;
251  namedPipeCreateParameters.InboundQuota = InboundQuota;
252  namedPipeCreateParameters.OutboundQuota = OutboundQuota;
253 
254  //
255  // Simply perform the remainder of the service by allowing the common
256  // file creation code to do the work.
257  //
258 
259  return IoCreateFile( FileHandle,
260  DesiredAccess,
261  ObjectAttributes,
262  IoStatusBlock,
263  (PLARGE_INTEGER) NULL,
264  0L,
265  ShareAccess,
266  CreateDisposition,
267  CreateOptions,
268  (PVOID) NULL,
269  0L,
270  CreateFileTypeNamedPipe,
271  &namedPipeCreateParameters,
272  0 );
273 }
