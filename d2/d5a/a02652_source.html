119 
120 Routine Description:
121 
122  This function searches for a matching entry in an inverted function
123  table using the specified control PC value.
124 
125  N.B. It is assumed that appropriate locks are held when this routine
126  is called.
127 
128 Arguments:
129 
130  InvertedTable - Supplies a pointer to an inverted function table.
131 
132  ControlPc - Supplies a PC value to to use in searching the inverted
133  function table.
134 
135  ImageBase - Supplies a pointer to a variable that receives the base
136  address of the corresponding module.
137 
138  SizeOfTable - Supplies a pointer to a variable that receives the size
139  of the function table in bytes.
140 
141 Return Value:
142 
143  If a matching entry is located in the specified function table, then
144  the function table address is returned as the function value. Otherwise,
145  a value of NULL is returned.
146 
147 --*/
148 
149 {
150 
151  PVOID Bound;
152  LONG High;
153  ULONG Index;
154  PINVERTED_FUNCTION_TABLE_ENTRY InvertedEntry;
155  LONG Low;
156  LONG Middle;
157 
158  //
159  // If there are any entries in the specified inverted function table,
160  // then search the table for a matching entry.
161  //
162 
163  if (InvertedTable->CurrentSize != 0) {
164  Low = 0;
165  High = InvertedTable->CurrentSize - 1;
166  while (High >= Low) {
167 
168  //
169  // Compute next probe index and test entry. If the specified
170  // control PC is greater than of equal to the beginning address
171  // and less than the ending address of the inverted function
172  // table entry, then return the address of the function table.
173  // Otherwise, continue the search.
174  //
175 
176  Middle = (Low + High) >> 1;
177  InvertedEntry = &InvertedTable->TableEntry[Middle];
178  Bound = (PVOID)((ULONG_PTR)InvertedEntry->ImageBase + InvertedEntry->SizeOfImage);
179  if (ControlPc < InvertedEntry->ImageBase) {
180  High = Middle - 1;
181 
182  } else if (ControlPc >= Bound) {
183  Low = Middle + 1;
184 
185  } else {
186  *ImageBase = InvertedEntry->ImageBase;
187  *SizeOfTable = InvertedEntry->SizeOfTable;
188  return InvertedEntry->FunctionTable;
189  }
190  }
191  }
192 
193  return NULL;
194 }

434 /*++
435 
436 Routine Description:
437 
438  This function inserts an entry in an inverted function table if there
439  is room in the table. Otherwise, no operation is performed.
440 
441  N.B. It is assumed that appropriate locks are held when this routine
442  is called.
443 
444  N.B. If the inverted function table overflows, then it is treated as
445  a cache. This is unlikely to happen, however.
446 
447 Arguments:
448 
449  InvertedTable - Supplies a pointer to the inverted function table in
450  which the specified entry is to be inserted.
451 
452  ImageBase - Supplies the base address of the containing image.
453 
454  SizeOfImage - Supplies the size of the image.
455 
456 Return Value:
457 
458  None.
459 
460 --*/
461 
462 {
463 
464  ULONG CurrentSize;
465  PRUNTIME_FUNCTION FunctionTable;
466  ULONG Index;
467  ULONG SizeOfTable;
468 
469  //
470  // If the inverted table is not full, then insert the entry in the
471  // specified inverted table.
472  //
473 
474  CurrentSize = InvertedTable->CurrentSize;
475  if (CurrentSize != InvertedTable->MaximumSize) {
476 
477  //
478  // If the inverted table has no entries, then insert the new entry as
479  // the first entry. Otherwise, search the inverted table for the proper
480  // insert position, shuffle the table, and insert the new entry.
481  //
482 
483  Index = 0;
484  if (CurrentSize != 0) {
485  for (Index = 0; Index < CurrentSize; Index += 1) {
486  if (ImageBase < InvertedTable->TableEntry[Index].ImageBase) {
487  break;
488  }
489  }
490 
491  //
492  // If the new entry does not go at the end of the specified table,
493  // then shuffle the table down to make room for the new entry.
494  //
495 
496  if (Index != CurrentSize) {
497  RtlMoveMemory(&InvertedTable->TableEntry[Index + 1],
498  &InvertedTable->TableEntry[Index],
499  (CurrentSize - Index) * sizeof(INVERTED_FUNCTION_TABLE_ENTRY));
500  }
501  }
502 
503  //
504  // Insert the specified entry in the specified inverted function table.
505  //
506 
507  FunctionTable = RtlImageDirectoryEntryToData (ImageBase,
508  TRUE,
509  IMAGE_DIRECTORY_ENTRY_EXCEPTION,
510  &SizeOfTable);
511 
512  InvertedTable->TableEntry[Index].FunctionTable = FunctionTable;
513  InvertedTable->TableEntry[Index].ImageBase = ImageBase;
514  InvertedTable->TableEntry[Index].SizeOfImage = SizeOfImage;
515  InvertedTable->TableEntry[Index].SizeOfTable = SizeOfTable;
516  InvertedTable->CurrentSize += 1;
517 
518  } else {
519  InvertedTable->Overflow = TRUE;
520  }
521 
522  return;
523 }

533 Routine Description:

539

564  // corresponding entry.
565  //
566

572  }

576  // and reduce the size of the table.
577  //
578

583  // remove the specified entry.
584  //
585

587  RtlMoveMemory(&InvertedTable->TableEntry[Index],
588  &InvertedTable->TableEntry[Index + 1],
589  (CurrentSize - Index - 1) * sizeof(INVERTED_FUNCTION_TABLE_ENTRY));
590  }
591 
592  //
593  // Reduce the size of the inverted table.
594  //
595 
596  InvertedTable->CurrentSize -= 1;
597  }
598 
599  return;
