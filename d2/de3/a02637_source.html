84  IN const WCHAR* Buffer,
85  IN ULONG MaximumLength,
86  IN const WCHAR* Format,
87  ...)
88 /*++
89 
90 Routine Description:
91 
92  Scans a source Buffer and places values from that buffer into the parameters
93  as specified by Format.
94 
95 Arguments:
96 
97  Buffer -
98  Contains the source buffer which is to be scanned.
99 
100  MaximumLength -
101  Contains the maximum length in characters for which Buffer is searched.
102  This implies that Buffer need not be UNICODE_NULL terminated.
103 
104  Format -
105  Contains the format string which defines both the acceptable string format
106  contained in Buffer, and the variable parameters which follow.
107 
108 Return Value:
109 
110  Returns the number of parameters filled if the end of the Buffer is reached,
111  else -1 on an error.
112 
113 --*/
114 {
115  va_list ArgList;
116  int FormatItems;
117 
118  va_start(ArgList, Format);
119  for (FormatItems = 0;;) {
120  switch (*Format) {
121  case 0:
122  return (MaximumLength && *Buffer) ? -1 : FormatItems;
123  case '%':
124  Format++;
125  if (*Format != '%') {
126  ULONG Number;
127  int Width;
128  int Long;
129  PVOID Pointer;
130 
131  for (Long = 0, Width = 0;; Format++) {
132  if ((*Format >= '0') && (*Format <= '9')) {
133  Width = Width * 10 + *Format - '0';
134  } else if (*Format == 'l') {
135  Long++;
136  } else if ((*Format == 'X') || (*Format == 'x')) {
137  break;
138  }
139  }
140  Format++;
141  for (Number = 0; Width--; Buffer++, MaximumLength--) {
142  if (!MaximumLength)
143  return -1;
144  Number *= 16;
145  if ((*Buffer >= '0') && (*Buffer <= '9')) {
146  Number += (*Buffer - '0');
147  } else if ((*Buffer >= 'a') && (*Buffer <= 'f')) {
148  Number += (*Buffer - 'a' + 10);
149  } else if ((*Buffer >= 'A') && (*Buffer <= 'F')) {
150  Number += (*Buffer - 'A' + 10);
151  } else {
152  return -1;
153  }
154  }
155  Pointer = va_arg(ArgList, PVOID);
156  if (Long) {
157  *(PULONG)Pointer = Number;
158  } else {
159  *(PUSHORT)Pointer = (USHORT)Number;
160  }
161  FormatItems++;
162  break;
163  }
164  /* no break */
165  default:
166  if (!MaximumLength || (*Buffer != *Format)) {
167  return -1;
168  }
169  Buffer++;
170  MaximumLength--;
171  Format++;
172  break;
173  }
174  }
175 }
