34 
35  This function locks a region of pages within the working set list

37

42

51  pages. The initial value of this argument is

55  perform. One of MAP_PROCESS or MAP_SYSTEM.

306  //

311  // Check to see if the WAS_LOCKED status should be returned.
312  //
313 
314  if ((MapType & MAP_PROCESS) &&

316  WasLocked = TRUE;
317  }
318 
319  if ((MapType & MAP_SYSTEM) &&
320  (MmWsle[WorkingSetIndex].u1.e1.LockedInMemory == 1)) {
321  WasLocked = TRUE;
322  }
323  }
324  }

328  UNLOCK_WS_UNSAFE (Thread, TargetProcess);
329 
330  //
331  // Check to ensure the working set list is still fluid after
332  // the requested number of pages are locked.
333  //
334 
335  if (TargetProcess->Vm.MinimumWorkingSetSize <
336  ((MmWorkingSetList->FirstDynamic + NumberToLock +
337  MM_FLUID_WORKING_SET) - NumberOfAlreadyLocked)) {

340  goto ErrorReturn1;

343  Va = PAGE_ALIGN (CapturedBase);
344 
345  //
346  // Set up an exception handler and touch each page in the specified
347  // range. Mark this thread as the address space mutex owner so it cannot
348  // sneak its stack in as the argument region and trick us into trying to
349  // grow it if the reference faults (as this would cause a deadlock since

354  //

399  //

416  //
417

419 
420  //
421  // Release the working set mutex and fault in the page.
422  //
423 
424  UNLOCK_WS_UNSAFE (Thread, TargetProcess);
425 
426  //
427  // Page in the PDE and make the PTE valid.
428  //
429 
430  try {
431  *(volatile ULONG *)Va;
432  } except (EXCEPTION_EXECUTE_HANDLER) {
433 
434  //
435  // Since all the pages were accessed above with the address
436  // space mutex held and it is still held now, the only way
437  // an exception could occur would be due to a device error,
438  // ie: hardware malfunction, net cable disconnection, CD
439  // being removed, etc.
440  //

442  // is written back to the user now. If this is the very first
443  // page then return a failure status.
444  //
445

447

449  Status = GetExceptionCode ();

454  EndingAddress = (PVOID)((ULONG_PTR)EndingAddress - 1);
455  goto SuccessReturn1;
456  }
457 
458  //

460  //
461

463 
464  //
465  // Ensure the PDE (and any table above it) are still resident.
466  // They could have been trimmed from the working set before the
467  // working set lock was reacquired above.
468  //
469 
470  MiMakePdeExistAndMakeValid (PointerPde,
471  TargetProcess,
472  MM_NOIRQL);
473  }
474 
475  //
476  // The page is now in the working set, lock the page into
477  // the working set.
478  //
479 
480  PointerPte1 = MiGetPteAddress (Va);
481  Pfn1 = MI_PFN_ELEMENT (PointerPte1->u.Hard.PageFrameNumber);
482 
483  Entry = MiLocateWsle (Va, MmWorkingSetList, Pfn1->u1.WsIndex, FALSE);
484 
485  if (Entry >= MmWorkingSetList->FirstDynamic) {
486 
487  SwapEntry = MmWorkingSetList->FirstDynamic;
488 
489  if (Entry != MmWorkingSetList->FirstDynamic) {
490 
491  //
492  // Swap this entry with the one at first dynamic.
493  //
494 
495  MiSwapWslEntries (Entry, SwapEntry, &TargetProcess->Vm, FALSE);
496  }
497 
498  MmWorkingSetList->FirstDynamic += 1;
499  }
500  else {
501  SwapEntry = Entry;
502  }
503 
504  //
505  // Indicate that the page is locked.
506  //
507 
508  if (MapType & MAP_PROCESS) {
509  MmWsle[SwapEntry].u1.e1.LockedInWs = 1;
510  }
511 
512  if (MapType & MAP_SYSTEM) {
513  MmWsle[SwapEntry].u1.e1.LockedInMemory = 1;
514  }
515 
516  //
517  // Increment to the next Va and PTE.
518  //
519 
520  PointerPte += 1;
521  Va = (PVOID)((PCHAR)Va + PAGE_SIZE);
522  }
523 
524  UNLOCK_WS_UNSAFE (Thread, TargetProcess);
525 
526 SuccessReturn1:
527 
528  UNLOCK_ADDRESS_SPACE (TargetProcess);
529 
530  if (Attached == TRUE) {
531  KeUnstackDetachProcess (&ApcState);
532  }
533  ObDereferenceObject (TargetProcess);
534 
535  //
536  // Update return arguments.
537  //
538 
539  //
540  // Establish an exception handler and write the size and base
541  // address.
542  //
543 
544  try {
545  *RegionSize = ((PCHAR)EndingAddress - (PCHAR)PAGE_ALIGN(CapturedBase)) +
546  PAGE_SIZE;
547  *BaseAddress = PAGE_ALIGN(CapturedBase);
548 
549  } except (EXCEPTION_EXECUTE_HANDLER) {
550  return GetExceptionCode();
551  }
552 
553  if (WasLocked) {
554  return STATUS_WAS_LOCKED;
555  }
556 
557  return STATUS_SUCCESS;
558 
559 ErrorReturn:

561 ErrorReturn1:
562  UNLOCK_ADDRESS_SPACE (TargetProcess);
563  if (Attached == TRUE) {
564  KeUnstackDetachProcess (&ApcState);
565  }
566  ObDereferenceObject (TargetProcess);
567  return Status;

570 NTSTATUS
571 NtUnlockVirtualMemory(
572  __in HANDLE ProcessHandle,
573  __inout PVOID *BaseAddress,
574  __inout PSIZE_T RegionSize,
575  __in ULONG MapType
576  )
577 
578 /*++
579 
580 Routine Description:
581 
582  This function unlocks a region of pages within the working set list
583  of a subject process.
584 
585  As a side effect, any pages which are not locked and are in the
586  process's working set are removed from the process's working set.
587  This allows NtUnlockVirtualMemory to remove a range of pages
588  from the working set.
589 
590  The caller of this function must have PROCESS_VM_OPERATION access
591  to the target process.
592 
593  The caller must also have SeLockMemoryPrivilege for MAP_SYSTEM.
594 
595 Arguments:
596 
597  ProcessHandle - Supplies an open handle to a process object.
598 
599  BaseAddress - The base address of the region of pages
600  to be unlocked. This value is rounded down to the
601  next host page address boundary.
602 
603  RegionSize - A pointer to a variable that will receive
604  the actual size in bytes of the unlocked region of
605  pages. The initial value of this argument is
606  rounded up to the next host page size boundary.
607 
608  MapType - A set of flags that describe the type of unlocking to
609  perform. One of MAP_PROCESS or MAP_SYSTEM.
610 
611 Return Value:
612 
613  NTSTATUS.
614 
615 --*/

826  }

828  if (!MiIsAddressValid (Va, TRUE)) {
829 
830  //
831  // This page is not valid, therefore not in working set.
832  //
833 
834  Status = STATUS_NOT_LOCKED;
835  }
836  else {
837 
838  PointerPte = MiGetPteAddress (Va);
839  ASSERT (PointerPte->u.Hard.Valid != 0);
840  Pfn1 = MI_PFN_ELEMENT (PointerPte->u.Hard.PageFrameNumber);
841  Entry = MiLocateWsle (Va, MmWorkingSetList, Pfn1->u1.WsIndex, FALSE);
842  ASSERT (Entry != WSLE_NULL_INDEX);
