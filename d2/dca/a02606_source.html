47 /*++
48 
49 Routine Description:
50 
51  Generates an IPv6 string literal corresponding to the address Addr.
52  The shortened canonical forms are used (RFC 1884 etc).
53  The basic string representation consists of 8 hex numbers
54  separated by colons, with a couple embellishments:
55  - a string of zero numbers (at most one) is replaced
56  with a double-colon.
57  - the last 32 bits are represented in IPv4-style dotted-octet notation
58  if the address is a v4-compatible or ISATAP address.
59 
60  For example,
61  ::
62  ::1
63  ::157.56.138.30
64  ::ffff:156.56.136.75
65  ff01::
66  ff02::2
67  0:1:2:3:4:5:6:7
68 
69 Arguments:
70 
71  S - Receives a pointer to the buffer in which to place the
72  string literal.
73 
74  Addr - Receives the IPv6 address.
75 
76 Return Value:
77 
78  Pointer to the null byte at the end of the string inserted.
79  This can be used by the caller to easily append more information.
80 
81 --*/
82 
83 {
84  int maxFirst, maxLast;
85  int curFirst, curLast;
86  int i;
87  int endHex = 8;
88 
89  // Check for IPv6-compatible, IPv4-mapped, and IPv4-translated
90  // addresses
91  if ((Addr->s6_words[0] == 0) && (Addr->s6_words[1] == 0) &&
92  (Addr->s6_words[2] == 0) && (Addr->s6_words[3] == 0) &&
93  (Addr->s6_words[6] != 0)) {
94  if ((Addr->s6_words[4] == 0) &&
95  ((Addr->s6_words[5] == 0) || (Addr->s6_words[5] == 0xffff)))
96  {
97  // compatible or mapped
98  S += _stprintf(S, _T("::%hs%u.%u.%u.%u"),
99  Addr->s6_words[5] == 0 ? "" : "ffff:",
100  Addr->s6_bytes[12], Addr->s6_bytes[13],
101  Addr->s6_bytes[14], Addr->s6_bytes[15]);
102  return S;
103  }
104  else if ((Addr->s6_words[4] == 0xffff) && (Addr->s6_words[5] == 0)) {
105  // translated
106  S += _stprintf(S, _T("::ffff:0:%u.%u.%u.%u"),
107  Addr->s6_bytes[12], Addr->s6_bytes[13],
108  Addr->s6_bytes[14], Addr->s6_bytes[15]);
109  return S;
110  }
111  }
112 
113 
114  // Find largest contiguous substring of zeroes
115  // A substring is [First, Last), so it's empty if First == Last.
116 
117  maxFirst = maxLast = 0;
118  curFirst = curLast = 0;
119 
120  // ISATAP EUI64 starts with 00005EFE (or 02005EFE)...
121  if (((Addr->s6_words[4] & 0xfffd) == 0) && (Addr->s6_words[5] == 0xfe5e)) {
122  endHex = 6;
123  }
124 
125  for (i = 0; i < endHex; i++) {
126 
127  if (Addr->s6_words[i] == 0) {
128  // Extend current substring
129  curLast = i+1;
130 
131  // Check if current is now largest
132  if (curLast - curFirst > maxLast - maxFirst) {
133 
134  maxFirst = curFirst;
135  maxLast = curLast;
136  }
137  }
138  else {
139  // Start a new substring
140  curFirst = curLast = i+1;
141  }
142  }
143 
144  // Ignore a substring of length 1.
145  if (maxLast - maxFirst <= 1)
146  maxFirst = maxLast = 0;
147 
148  // Write colon-separated words.
149  // A double-colon takes the place of the longest string of zeroes.
150  // All zeroes is just "::".
151 
152  for (i = 0; i < endHex; i++) {
153 
154  // Skip over string of zeroes
155  if ((maxFirst <= i) && (i < maxLast)) {
156 
157  S += _stprintf(S, _T("::"));
158  i = maxLast-1;
159  continue;
160  }
161 
162  // Need colon separator if not at beginning
163  if ((i != 0) && (i != maxLast))
164  S += _stprintf(S, _T(":"));
165 
166  S += _stprintf(S, _T("%x"), RtlUshortByteSwap(Addr->s6_words[i]));
167  }
168 
169  if (endHex < 8) {
170  S += _stprintf(S, _T(":%u.%u.%u.%u"),
171  Addr->s6_bytes[12], Addr->s6_bytes[13],
172  Addr->s6_bytes[14], Addr->s6_bytes[15]);
173  }
174 
175  return S;
176 }

179 RtlIpv6AddressToStringExT(
180  __in const struct in6_addr *Address,
181  __in ULONG ScopeId,

187 /*++
188 
189 Routine Description:
190 
191  This is the extension routine which handles a full address conversion
192  including address, scopeid and port (scopeid and port are optional).
193 
194 Arguments:
195 
196  Address - The address part to be translated.
197 
198  ScopeId - The Scope ID of the address (optional).
199 
200  Port - The port number of the address (optional).
201  Port is in network byte order.
202 
203  AddressString - Pointer to output buffer where we will fill in address string.
204 
205  AddressStringLength - For input, it is the length of the input buffer; for
206  output it is the length we actual returned.
207 Return Value:
208 
209  STATUS_SUCCESS if the operation is successful, error code otherwise.
210 
211 --*/
212 {
213  TCHAR String[INET6_ADDRSTRLEN];
214  LPTSTR S;
215  ULONG Length;
216 
217  if ((Address == NULL) ||
218  (AddressString == NULL) ||
219  (AddressStringLength == NULL)) {
220 
221  return STATUS_INVALID_PARAMETER;
222  }
223  S = String;
224  if (Port) {
225  S += _stprintf(S, _T("["));
226  }
227 
228  //
229  // Now translate this address.
230  //
231  S = RtlIpv6AddressToStringT(Address, S);
232  if (ScopeId != 0) {
233  S += _stprintf(S, _T("%%%u"), ScopeId);
234  }
235  if (Port != 0) {
236  S += _stprintf(S, _T("]:%u"), RtlUshortByteSwap(Port));
237  }
238  Length = (ULONG)(S - String + 1);
239  if (*AddressStringLength < Length) {
240  //
241  // Before return, tell the caller how big
242  // the buffer we need.
243  //
244  *AddressStringLength = Length;
245  return STATUS_INVALID_PARAMETER;
246  }
247  *AddressStringLength = Length;
248  RtlCopyMemory(AddressString, String, Length * sizeof(TCHAR));
249  return STATUS_SUCCESS;
250 
251 }
