98 
99 Routine Description:
100 
101  This routine saves the thread's current non-volatile NPX state,
102  and sets a new initial floating point state for the caller.
103 
104 Arguments:
105 
106  FloatSave - receives the current non-volatile npx state for the thread
107 
108 Return Value:
109 
110 --*/
111 {
112  PKTHREAD Thread;
113  PFX_SAVE_AREA NpxFrame;
114  KIRQL Irql;
115  USHORT ControlWord;
116  ULONG MXCsr;
117  PKPRCB Prcb;
118  PFLOAT_SAVE FloatSave;
119 
120  //
121  // If the system is using floating point emulation, then
122  // return an error
123  //
124 
125  if (!KeI386NpxPresent) {
126  return STATUS_ILLEGAL_FLOAT_CONTEXT;
127  }
128 
129  //
130  // Get the current irql and thread
131  //
132 
133  FloatSave = (PFLOAT_SAVE) PublicFloatSave;
134 
135  Irql = KeGetCurrentIrql();
136  Thread = KeGetCurrentThread();
137 
138  ASSERT (Thread->Header.NpxIrql <= Irql);
139 
140  FloatSave->Flags = 0;
141  FloatSave->Irql = Irql;
142  FloatSave->PreviousNpxIrql = Thread->Header.NpxIrql;
143  FloatSave->Thread = Thread;
144 
145  //
146  // If the irql has changed we need to save the complete floating
147  // state context as the prior level has been interrupted.
148  //
149 
150  if (Thread->Header.NpxIrql != Irql) {
151 
152  //
153  // If this is apc level we don't have anyplace to hold this
154  // context, allocate some heap.
155  //
156 
157  if (Irql == APC_LEVEL) {
158  FloatSave->u.Context = ExAllocatePoolWithTag(NonPagedPool,
159  sizeof (FX_SAVE_AREA) + ALIGN_ADJUST,
160  ' XPN');
161 
162  if (!FloatSave->u.Context) {
163  return STATUS_INSUFFICIENT_RESOURCES;
164  }
165 
166  FloatSave->Flags |= FLOAT_SAVE_FREE_CONTEXT_HEAP;
167 
168  //
169  // ExAllocatePoolWithTag returns an 8 byte aligned pointer.
170  // The FXSAVE instruction requires 16 byte alignment. Adjust
171  // the base address of the save area if needed.
172  //
173 
174  if ((FloatSave->u.ContextAddressAsULONG & ALIGN_ADJUST) != 0) {
175  FloatSave->u.ContextAddressAsULONG += ALIGN_ADJUST;
176  FloatSave->Flags |= FLOAT_SAVE_ALIGN_ADJUSTED;
177  }
178  ASSERT((FloatSave->u.ContextAddressAsULONG & 0xF) == 0);
179 
180  } else {
181 
182  ASSERT (Irql == DISPATCH_LEVEL);
183  FloatSave->u.Context = &KeGetCurrentPrcb()->NpxSaveArea;
184 
185  }
186 
187  FloatSave->Flags |= FLOAT_SAVE_COMPLETE_CONTEXT;
188  }

191  // Stop context switching and allow access to the local fp unit
192  //
193

195  cli
196  mov eax, cr0
197  mov ecx, eax
198  and eax, not (CR0_MP|CR0_EM|CR0_TS)

200  je short sav10

238  sizeof(FX_SAVE_AREA)));
239 
240 
241  //
242  // Save the previous state as required
243  //
244 
245  if (FloatSave->Flags & FLOAT_SAVE_COMPLETE_CONTEXT) {
246 
247  //
248  // Need to save the entire context
249  //
250 
251  if (Thread->NpxState == NPX_STATE_LOADED) {
252  if (KeI386FxsrPresent) {
253  Kix86FxSave((FloatSave->u.Context));
254  } else {
255  Kix86FnSave((FloatSave->u.Context));
256  }
257 
258  FloatSave->u.Context->NpxSavedCpu = 0;
259  FloatSave->u.Context->Cr0NpxState = NpxFrame->Cr0NpxState;
260 
261  } else {
262  RtlCopyMemory (FloatSave->u.Context, NpxFrame, sizeof(FX_SAVE_AREA));
263  FloatSave->u.Context->NpxSavedCpu = 0;
264 
265  }
266 
267  } else {
268 
269  //
270  // Save only the non-volatile state
271  //
272 
273  if (Thread->NpxState == NPX_STATE_LOADED) {
274 
275  _asm {
276  mov eax, FloatSave
277  fnstcw [eax] FLOAT_SAVE.u.Fcw.ControlWord
278  }
279

281  Kix86StMXCsr(&FloatSave->u.Fcw.MXCsr);
282  }
283 
284  } else {
285  //
286  // Save the control word from the npx frame.
287  //
288 
289  if (KeI386FxsrPresent) {
290  FloatSave->u.Fcw.ControlWord = (USHORT) NpxFrame->U.FxArea.ControlWord;
291  FloatSave->u.Fcw.MXCsr = NpxFrame->U.FxArea.MXCsr;
292 
293  } else {
294  FloatSave->u.Fcw.ControlWord = (USHORT) NpxFrame->U.FnArea.ControlWord;
295  }
296  }
297 
298 
299  //
300  // Save Cr0NpxState, but clear CR0_TS as there's not non-volatile
301  // pending fp exceptions
302  //
303 
304  FloatSave->Cr0NpxState = NpxFrame->Cr0NpxState & ~CR0_TS;
305  }
306 
307  //
308  // The previous state is saved. Set an initial default
309  // FP state for the caller
310  //
311 
312  NpxFrame->Cr0NpxState = 0;
313  Thread->NpxState = NPX_STATE_LOADED;
314  Thread->Header.NpxIrql = Irql;
315  ControlWord = 0x27f; // 64bit mode
316  MXCsr = 0x1f80;
317 
318  _asm {
319  fninit
320  fldcw ControlWord
321  }
322 
323  if ((KeI386FxsrPresent) && (KeI386XMMIPresent)) {
324  Kix86LdMXCsr(&MXCsr);
325  }
326 
327  _asm {
328  sti
329  }
330 
331  FloatSave->Flags |= FLOAT_SAVE_VALID;
332  return STATUS_SUCCESS;
333 }

340 /*++
341 
342 Routine Description:
343 
344  This routine restores the thread's current non-volatile NPX state,
345  to the passed in state.
346 
347 Arguments:
348 
349  FloatSave - the non-volatile npx state for the thread to restore
350 
351 Return Value:
352 
353 --*/
354 {
355  PKTHREAD Thread;
356  PFX_SAVE_AREA NpxFrame;
357  ULONG Cr0State;
358  PFLOAT_SAVE FloatSave;
359 
360  ASSERT (KeI386NpxPresent);
361 
362  FloatSave = (PFLOAT_SAVE) PublicFloatSave;
363  Thread = FloatSave->Thread;
364 
365  NpxFrame = (PFX_SAVE_AREA)(((ULONG)(Thread->InitialStack) -
366  sizeof(FX_SAVE_AREA)));
367 
368 
369  //
370  // Verify float save looks like it's from the right context
371  //
372 
373  if ((FloatSave->Flags & (FLOAT_SAVE_VALID | FLOAT_SAVE_RESERVED)) != FLOAT_SAVE_VALID) {
374 
375  //
376  // Invalid floating point save area.
377  //
378 
379  KeBugCheckEx(INVALID_FLOATING_POINT_STATE,
380  0,
381  FloatSave->Flags,
382  0,
383  0);
384  }
385 
386  if (FloatSave->Irql != KeGetCurrentIrql()) {
387 
388  //
389  // Invalid IRQL. IRQL now must be the same as when the
390  // context was saved. (Why? Because we save it in different
391  // places depending on the IRQL at that time).
392  //
393 
394  KeBugCheckEx(INVALID_FLOATING_POINT_STATE,
395  1,
396  FloatSave->Irql,
397  KeGetCurrentIrql(),
398  0);
399  }
400 
401  if (Thread != KeGetCurrentThread()) {
402 
403  //
404  // Invalid Thread. The thread this floating point context
405  // belongs to is not the current thread (or the saved thread
406  // field is trash).
407  //
408 
409  KeBugCheckEx(INVALID_FLOATING_POINT_STATE,
410  2,
411  (ULONG_PTR)Thread,
412  (ULONG_PTR)KeGetCurrentThread(),
413  0);
414  }
415 
416 
417  //
418  // Synchronize with context switches and the npx trap handlers
419  //
420 
421  _asm {
422  cli
423  }
424 
425  //
426  // Restore the required state
427  //
428 
429  if (FloatSave->Flags & FLOAT_SAVE_COMPLETE_CONTEXT) {
430 
431  //
432  // Restore the entire fp state to the threads save area
433  //
434 
435  if (Thread->NpxState == NPX_STATE_LOADED) {
436 
437  //
438  // This state in the fp unit is no longer needed, just disregard it
439  //
440 
441  Thread->NpxState = NPX_STATE_NOT_LOADED;
442  KeGetCurrentPrcb()->NpxThread = NULL;
443  }
444 
445  //
446  // Copy restored state to npx frame
447  //
448 
449  RtlCopyMemory (NpxFrame, FloatSave->u.Context, sizeof(FX_SAVE_AREA));
450 
451  } else {
452 
453  //
454  // Restore the non-volatile state
455  //
456 
457  if (Thread->NpxState == NPX_STATE_LOADED) {
458 
459  //
460  // Init fp state and restore control word
461  //
462 
463  _asm {
464  fninit
465  mov eax, FloatSave
466  fldcw [eax] FLOAT_SAVE.u.Fcw.ControlWord
467  }
468 
469 
470  if ((KeI386FxsrPresent) && (KeI386XMMIPresent)) {
471  Kix86LdMXCsr(&FloatSave->u.Fcw.MXCsr);
472  }
473 
474 
475  } else {
476 
477  //
478  // Fp state not loaded. Restore control word in npx frame
479  //
480 
481  if (KeI386FxsrPresent) {
482  NpxFrame->U.FxArea.ControlWord = FloatSave->u.Fcw.ControlWord;
483  NpxFrame->U.FxArea.StatusWord = 0;
484  NpxFrame->U.FxArea.TagWord = 0;
485  NpxFrame->NpxSavedCpu = 0;
486  NpxFrame->U.FxArea.MXCsr = FloatSave->u.Fcw.MXCsr;
487 
488  } else {
489  NpxFrame->U.FnArea.ControlWord = FloatSave->u.Fcw.ControlWord;
490  NpxFrame->U.FnArea.StatusWord = 0;
491  NpxFrame->U.FnArea.TagWord = 0xffff;
492  }
493 
494  }
495 
496  NpxFrame->Cr0NpxState = FloatSave->Cr0NpxState;
497  }
498 
499  //
500  // Restore NpxIrql and Cr0
501  //
502 
503  Thread->Header.NpxIrql = FloatSave->PreviousNpxIrql;
504  Cr0State = Thread->NpxState | NpxFrame->Cr0NpxState;
505 
506  _asm {
507  mov eax, cr0
508  mov ecx, eax
509  and eax, not (CR0_MP|CR0_EM|CR0_TS)
510  or eax, Cr0State
511  cmp eax, ecx
512  je short res10
513  mov cr0, eax
514 res10:
515  sti
516  }
517 
518  //
519  // Done
520  //
521 
522  if ((FloatSave->Flags & FLOAT_SAVE_FREE_CONTEXT_HEAP) != 0) {
523 
524  //
525  // If FXSAVE area was adjusted for alignment after allocation,
526  // undo that adjustment before freeing.
527  //
528 
529  if ((FloatSave->Flags & FLOAT_SAVE_ALIGN_ADJUSTED) != 0) {
530  FloatSave->u.ContextAddressAsULONG -= ALIGN_ADJUST;
531  }
532  ExFreePool (FloatSave->u.Context);
533  }
534 
535  FloatSave->Flags = 0;
536  return STATUS_SUCCESS;
537 }

575 VOID
576 KiEnableFastSyscallReturn(
577  VOID
578  )
579 
580 /*++
581 
582 Routine Description:
583 
584  The fast syscall/return feature cannot be used until
585  certain processor specific registers have been initialized.
586  This routine is called once the registers are known to
587  have been set on all processors.
588 
589  This routine adjusts the exit path for system calls to
590  use the appropriate sequence for the processor, it does
591  this by adjusting the offset of a branch.
592 
593 Arguments:
594 
595  None.
596 
597 Return Value:
598 
599  None.
600 
601 --*/
602 
603 {
604  //
605  // Adjust the second byte of the two byte branch instruction.
606  // It can never be otherwise, but, make sure we aren't going
607  // to adjust it out of range.
608  //
609 
610  //
611  // The following is a workaround for the fact that in resume
612  // from hibernate the kernel is read only. Basically, we
613  // won't try to do it again, we also don't undo it when
614  // hibernating/suspending.
615  //
616 
617  if ((KiSystemCallExitAdjusted == KiSystemCallExitAdjust) &&
618  KiFastCallCopyDoneOnce) {
619 
620  //
621  // It's already done, don't try to do it again.
622  //
623 
624  return;
625  }
626 
627  if ((KiSystemCallExitAdjust + KiSystemCallExitBranch[1]) < 0x80) {
628 
629  //
630  // It's good, undo any previous adjustment.
631  //
632 
633  KiDisableFastSyscallReturn();
634 
635  //
636  // Adjust the branch.
637  //
638 
639  KiSystemCallExitAdjusted = (UCHAR)KiSystemCallExitAdjust;
640  KiSystemCallExitBranch[1] = (UCHAR) (KiSystemCallExitBranch[1] + KiSystemCallExitAdjusted);
641 
642  KiFastCallCopyDoneOnce = TRUE;
643  }
644 }

785 KiAcquireQueuedSpinLockCheckForFreeze(
786  IN PKSPIN_LOCK_QUEUE QueuedLock,
787  IN PKTRAP_FRAME TrapFrame
788  )
789 
790 /*++
791 
792 Routine Description:
793 
794  This routine is called to acquire a queued spin lock while at high
795  priority. While the lock is not available, a check is made to see
796  if another processor has requested this processor freeze execution.
797 
798  Note: This routine must be called with current IRQL at or above
799  dispatch lever, or interrupts disabled.
800 
801 Arguments:
802 
803  QueuedLock Supplies the address of the queued spinlock.
804  TrapFrame Supplies the address of the trap frame to pass to
805  KiFreezeTargetExecution.
806 
807 Return Value:
808 
809  None.
810 
811 --*/
812 
813 {
814  PKSPIN_LOCK_QUEUE Previous;
815  PKPRCB Prcb;
816  volatile ULONG_PTR * LockPointer;
817 
818  LockPointer = (volatile ULONG_PTR *)&QueuedLock->Lock;
819 
820  Previous = InterlockedExchangePointer(QueuedLock->Lock, QueuedLock);
821 
822  if (Previous == NULL) {
823 
824  //
825  // This processor now owns this lock.
826  //
827 
828  *LockPointer |= LOCK_QUEUE_OWNER;
829 
830  } else {
831 
832  //
833  // Lock is already held, update thew next pointer in the
834  // previous queue entry to point to this new waiter and
835  // wait until the lock is granted.
836  //
837  // The following loop is careful not to write ANYTHING
838  // while waiting unless a freeze execution has been
839  // requested. This includes any stack variables or
840  // return addresses.
841  //
842 
843  *LockPointer |= LOCK_QUEUE_WAIT;
844  Previous->Next = QueuedLock;
845 
846  Prcb = KeGetCurrentPrcb();
847 
848  while (*LockPointer & LOCK_QUEUE_WAIT) {
849  if (Prcb->RequestSummary & IPI_FREEZE) {
850  ULONG OldSummary;
851  ULONG NewSummary;
852  ULONG Summary;
853 
854  OldSummary = Prcb->RequestSummary;
855  NewSummary = OldSummary & ~IPI_FREEZE;
856  Summary = InterlockedCompareExchange((PVOID)&Prcb->RequestSummary,
857  NewSummary,
858  OldSummary);
859 
860  //
861  // If something else edited the RequestSummary, we'll
862  // get it next time around (unless the IPI has been
863  // handled).
864  //
865 
866  if (Summary == OldSummary) {
867 
868  //
869  // IPI_FREEZE cleared in RequestSummary. Now
870  // freeze as requested.
871  //
872 
873  KiFreezeTargetExecution(TrapFrame, NULL);
874  }
875  }

1030  IN PKSPIN_LOCK_QUEUE QueuedLock
1031  )
1032 {
1033  PKSPIN_LOCK_QUEUE Waiter;
1034

1036  // Get the address of the actual lock and strip out the bottom
1037  // two bits which are used for status.
1038  //
1039 
1040  ASSERT((((ULONG_PTR)QueuedLock->Lock) & 3) == LOCK_QUEUE_OWNER);

1042

1044

1046

1048  // Good chance noone is queued on this lock, to be sure
1049  // we need to do an interlocked operation on it.
1050  // Note: This is just an optimization, there is no point
1051  // in doing the interlocked compare exchange if someone
1052  // else has already joined the queue.
1053  //
1054 
1055  Waiter = InterlockedCompareExchangePointer(QueuedLock->Lock,
1056  NULL,
1057  QueuedLock);
1058  }
1059  if (Waiter != QueuedLock) {
1060 
1061  //
1062  // There is another waiter. It is possible for the waiter
1063  // to have only just performed the exchange that put its
1064  // context in the lock and to have not yet updated the
1065  // 'next' pointer in the previous context (which could be
1066  // this context), so we wait for our next pointer to be
1067  // non-null before continuing.
1068  //
1069

1071

1073  KeYieldProcessor();
1074  }

1077  // Pass the lock on to the next in line.
1078  //
1079 
1080  *((PULONG_PTR)&Waiter->Lock) ^= (LOCK_QUEUE_WAIT | LOCK_QUEUE_OWNER);
1081  QueuedLock->Next = NULL;
1082  }
