9 Module Name:
10 
11  cmgquota.c
12 
13 Abstract:
14 
15  The module contains CM routines to support Global Quota
16 
17  Global Quota has little to do with NT's standard per-process/user
18  quota system. Global Quota is waying of controlling the aggregate
19  resource usage of the entire registry. It is used to manage space
20  consumption by objects which user apps create, but which are persistent
21  and therefore cannot be assigned to the quota of a user app.
22 
23  Global Quota prevents the registry from consuming all of paged
24  pool, and indirectly controls how much disk it can consume.
25  Like the release 1 file systems, a single app can fill all the
26  space in the registry, but at least it cannot kill the system.
27 
28  Memory objects used for known short times and protected by
29  serialization, or billable as quota objects, are not counted
30  in the global quota.
31 
32 --*/

297 VOID
298 CmpComputeGlobalQuotaAllowed(
299  VOID
300  )
301 
302 /*++
303 
304 Routine Description:
305 
306  Compute CmpGlobalQuota based on:
307  (a) Size of paged pool
308  (b) Explicit user registry commands to set registry GQ
309 
310 Return Value:
311 
312  NONE.
313 
314 --*/
315 
316 {
317  SIZE_T PagedLimit;
318 
319  PagedLimit = (ULONG)(CM_LIMIT_RATIO(MmSizeOfPagedPoolInBytes));
320 
321  if ((CmRegistrySizeLimitLength != 4) ||
322  (CmRegistrySizeLimitType != REG_DWORD) ||
323  (CmRegistrySizeLimit == 0))
324  {
325  //
326  // If no value at all, or value of wrong type, or set to
327  // zero, use internally computed default
328  //
329  CmpGlobalQuota = (ULONG)(MmSizeOfPagedPoolInBytes / CM_DEFAULT_RATIO);
330 
331  } else if (CmRegistrySizeLimit >= PagedLimit) {
332  //
333  // If more than computed upper bound, use computed upper bound
334  //
335  CmpGlobalQuota = PagedLimit;
336 
337  } else {
338  //
339  // Use the set size
340  //
341  CmpGlobalQuota = CmRegistrySizeLimit;
342 
343  }
344 
345  if (CmpGlobalQuota > CM_WRAP_LIMIT) {
346  CmpGlobalQuota = CM_WRAP_LIMIT;
347  }
348  if (CmpGlobalQuota < CM_MINIMUM_GLOBAL_QUOTA) {
349  CmpGlobalQuota = CM_MINIMUM_GLOBAL_QUOTA;
350  }
351 
352  CmpGlobalQuotaWarning = CM_REGISTRY_WARNING_LEVEL * (CmpGlobalQuota / 100);
353 
354  return;
355 }

529 VOID
530 CmpUpdateSystemHiveHysteresis( PHHIVE Hive,
531  ULONG NewLength,
532  ULONG OldLength
533  )
534 {
535  PCMHIVE CmHive;
536  PWORK_QUEUE_ITEM WorkItem;
537  ULONG CurrentRatio;
538  BOOLEAN DoWorkItem = FALSE;
539 
540  PAGED_CODE();
541 
542  CmHive = (PCMHIVE)CONTAINING_RECORD(Hive,CMHIVE,Hive);
543 
544  if( (!CmpSystemHiveHysteresisCallback) || (CmHive != CmpMachineHiveList[SYSTEM_HIVE_INDEX].CmHive) ) {
545  //
546  // not the system hive, bail out
547  //
548  return;
549  }
550 
551  ASSERT( NewLength != OldLength );
552 
553  //
554  // compute current ratio; account for the header first
555  //
556  CurrentRatio = NewLength + HBLOCK_SIZE;
557  CurrentRatio *= 100;
558  CurrentRatio /= CM_SYSTEM_HIVE_LIMIT_SIZE;
559 
560  if( NewLength > OldLength ) {
561  //
562  // hive is growing
563  //
564  if( (CmpSystemHiveHysteresisHighSeen == FALSE) && (CurrentRatio > CmpSystemHiveHysteresisHigh) ) {
565  //
566  // we reached high; see if low has already been hit and queue work item
567  //
568  CmpSystemHiveHysteresisHighSeen = TRUE;
569  if( TRUE == CmpSystemHiveHysteresisLowSeen ) {
570  //
571  // low to high; queue workitem
572  //
573  CmpSystemHiveHysteresisHitRatio = CurrentRatio;
574  DoWorkItem = TRUE;
575  }
576  }
577  } else {
578  //
579  // hive is shrinking
580  //
581  if( (FALSE == CmpSystemHiveHysteresisLowSeen) && (CurrentRatio < CmpSystemHiveHysteresisLow ) ) {
582  //
583  // we reached low; see if low has been hit and queue work item
584  //
585  CmpSystemHiveHysteresisLowSeen = TRUE;
586  if( TRUE == CmpSystemHiveHysteresisHighSeen ) {
587  //
588  // high to low; queue workitem
589  //
590  CmpSystemHiveHysteresisHitRatio = CurrentRatio;
591  DoWorkItem = TRUE;
592  }
593  }
594  }
595 
596  if( DoWorkItem ) {
597  ASSERT( CmpSystemHiveHysteresisLowSeen && CmpSystemHiveHysteresisHighSeen );
598 
599  WorkItem = ExAllocatePool(NonPagedPool, sizeof(WORK_QUEUE_ITEM));
600  if (WorkItem != NULL) {
601 
602  ExInitializeWorkItem(WorkItem,
603  CmpSystemHiveHysteresisWorker,
604  WorkItem);
605  ExQueueWorkItem(WorkItem, DelayedWorkQueue);
606  }
607  //
608  // reset state so we can fire again later
609  //
610  CmpSystemHiveHysteresisLowSeen = FALSE;
611  CmpSystemHiveHysteresisHighSeen = FALSE;
612  }
613 }

615 ULONG
616 CmRegisterSystemHiveLimitCallback(
617  __in ULONG Low,
618  __in ULONG High,
619  __in PVOID Ref,
620  __in PCM_HYSTERESIS_CALLBACK Callback
621  )
622 /*++
623 
624 Routine Description:
625 
626  This routine registers a hysteresis for the system hive limit ratio.
627  We will call the callback :
628 
629  a. the system hive goes above High from below Low
630  b. the system hive goes below Low from above High
631 
632 Arguments:
633 
634  Low, High - specifies the hysteresis
635 
636  Ref - Context to give back to the callback
637 
638  Callback - callback routine.
639 
640 Return Value:
641 
642  current ratio 0 - 100
643 
644 --*/
645 {
646  ULONG Length;
647 
648  PAGED_CODE();
649 
650  if( CmpMachineHiveList[SYSTEM_HIVE_INDEX].CmHive ) {
651  Length = CmpMachineHiveList[SYSTEM_HIVE_INDEX].CmHive->Hive.BaseBlock->Length + HBLOCK_SIZE;
652 
653  Length *= 100;
654  Length /= CM_SYSTEM_HIVE_LIMIT_SIZE;
655  } else {
656  Length = 0;
657  }
658 
659  //
660  // allow only one call per system uptime.
661  //
662  if( CmpSystemHiveHysteresisCallback == NULL ) {
663  CmpSystemHiveHysteresisLow = Low;
664  CmpSystemHiveHysteresisHigh = High;
665  CmpSystemHiveHysteresisContext = Ref;
666  CmpSystemHiveHysteresisCallback = Callback;
667  //
668  // set state vars
669  //
670  if( Length <= Low ) {
671  CmpSystemHiveHysteresisLowSeen = TRUE;
672  } else {
673  CmpSystemHiveHysteresisLowSeen = FALSE;
674  }
675  if( Length >= High) {
676  CmpSystemHiveHysteresisHighSeen = TRUE;
677  } else {
678  CmpSystemHiveHysteresisHighSeen = FALSE;
679  }
680  }
681  return Length;
682 }
