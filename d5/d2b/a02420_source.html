64 
65 Routine Description:
66 
67  This function conditionally sets the signal state of a gate object and
68  attempts to unwait the first waiter.
69 
70 Arguments:
71 
72  Gate - Supplies a pointer to a dispatcher object of type gate.
73 
74 Return Value:
75 
76  None.
77 
78 --*/
79 
80 {
81 
82  PKTHREAD CurrentThread;
83  PLIST_ENTRY Entry;
84  KIRQL OldIrql;
85  SCHAR Priority;
86  PKQUEUE Queue;
87  PKWAIT_BLOCK WaitBlock;
88  PKTHREAD WaitThread;
89 
90  ASSERT_GATE(Gate);
91 
92  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
93 
94  //
95  // Raise IRQL to SYNCH_LEVEL and acquire the object lock.
96  //
97  // If the object is not already signaled, then attempt to wake a waiter.
98  //
99 
100  CurrentThread = KeGetCurrentThread();
101  do {
102  OldIrql = KeRaiseIrqlToSynchLevel();
103  KiAcquireKobjectLock(Gate);
104  if (Gate->Header.SignalState == 0) {
105 
106  //
107  // If there are any waiters, then remove and ready the first
108  // waiter. Otherwise, set the signal state.
109  //
110 
111  if (IsListEmpty(&Gate->Header.WaitListHead) == FALSE) {
112  Entry = Gate->Header.WaitListHead.Flink;
113  WaitBlock = CONTAINING_RECORD(Entry, KWAIT_BLOCK, WaitListEntry);
114  WaitThread = WaitBlock->Thread;
115 
116  //
117  // Try to acquire the thread lock.
118  //
119  // If the thread lock cannot be acquired, then release the
120  // object lock, lower IRQL to is previous value, and try to
121  // signal the gate again. Otherwise, remove the entry from
122  // the list, set the wait completion status to success, set
123  // the deferred processor number, set the thread state to
124  // deferred ready, release the thread lock, release the
125  // object lock, compute the new thread priority, ready the
126  // thread for execution, and exit the dispatcher.
127  //
128 
129  if (KiTryToAcquireThreadLock(WaitThread)) {
130  RemoveEntryList(Entry);
131  WaitThread->WaitStatus = STATUS_SUCCESS;
132  WaitThread->State = DeferredReady;
133  WaitThread->DeferredProcessor = KeGetCurrentPrcb()->Number;
134  KiReleaseKobjectLock(Gate);
135  KiReleaseThreadLock(WaitThread);
136  Priority = CurrentThread->Priority;
137  if (Priority < LOW_REALTIME_PRIORITY) {
138  Priority = Priority - CurrentThread->PriorityDecrement;
139  if (Priority < CurrentThread->BasePriority) {
140  Priority = CurrentThread->BasePriority;
141  }
142 
143  if (CurrentThread->PriorityDecrement != 0) {
144  CurrentThread->PriorityDecrement = 0;
145  CurrentThread->Quantum = CLOCK_QUANTUM_DECREMENT;
146  }
147  }
148 
149  WaitThread->AdjustIncrement = Priority;
150  WaitThread->AdjustReason = (UCHAR)AdjustBoost;
151 
152  //
153  // If the wait thread is associated with a queue, then
154  // increment the concurrency level.
155  //
156  // N.B. This can be done after dropping all other locks,
157  // but must be done holding the dispatcher lock
158  // since the concurrency count is not accessed with
159  // interlocked operations elsewhere.
160  //
161 
162  if (WaitThread->Queue != NULL) {
163  KiLockDispatcherDatabaseAtSynchLevel();
164  if ((Queue = WaitThread->Queue) != NULL) {
165  Queue->CurrentCount += 1;
166  }
167 
168  KiUnlockDispatcherDatabaseFromSynchLevel();
169  }
170 
171  KiDeferredReadyThread(WaitThread);
172  KiExitDispatcher(OldIrql);
173  return;
174 
175  } else {
176  KiReleaseKobjectLock(Gate);
177  KeLowerIrql(OldIrql);
178  continue;
179  }
180 
181 
182  } else {
183  Gate->Header.SignalState = 1;
184  break;
185  }
186 
187  } else {
188  break;
189  }
190 
191  } while (TRUE);
192 
193  //
194  // Release the object lock and lower IRQL to its previous value.
195  //
196 
197  KiReleaseKobjectLock(Gate);
198  KeLowerIrql(OldIrql);
199  return;
200 }

210 /*++
211 
212 Routine Description:
213 
214  This function waits until the signal state of a gate object is set. If
215  the state of the gate object is signaled when the wait is executed, then
216  no wait will occur.
217 
218 Arguments:
219 
220  Gate - Supplies a pointer to a dispatcher object of type Gate.
221 
222  WaitReason - Supplies the reason for the wait.
223 
224  WaitMode - Supplies the processor mode in which the wait is to occur.
225 
226 Return Value:
227 
228  None.
229 
230 --*/
231 
232 {
233 
234  PKTHREAD CurrentThread;
235  KLOCK_QUEUE_HANDLE LockHandle;
236  PKQUEUE Queue;
237  PKWAIT_BLOCK WaitBlock;
238  NTSTATUS WaitStatus;
239 
240  ASSERT_GATE(Gate);
241 
242  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
243 
244  //
245  // Raise IRQL to SYNCH_LEVEL and acquire the APC queue lock.
246  //
247 
248  CurrentThread = KeGetCurrentThread();
249  do {
250  KeAcquireInStackQueuedSpinLockRaiseToSynch(&CurrentThread->ApcQueueLock,
251  &LockHandle);
252 
253  //
254  // Test to determine if a kernel APC is pending.
255  //
256  // If a kernel APC is pending, the special APC disable count is zero,
257  // and the previous IRQL was less than APC_LEVEL, then a kernel APC
258  // was queued by another processor just after IRQL was raised to
259  // SYNCH_LEVEL, but before the APC queue lock was acquired.
260  //
261  // N.B. This can only happen in a multiprocessor system.
262  //
263 
264  if (CurrentThread->ApcState.KernelApcPending &&
265  (CurrentThread->SpecialApcDisable == 0) &&
266  (LockHandle.OldIrql < APC_LEVEL)) {
267 
268  //
269  // Unlock the APC queue lock and lower IRQL to its previous value.
270  // An APC interrupt will immediately occur which will result in
271  // the delivery of the kernel APC if possible.
272  //
273 
274  KeReleaseInStackQueuedSpinLock(&LockHandle);
275  continue;
276  }
277 
278  //
279  // If the current thread is associated with a queue object, then
280  // acquire the dispatcher lock.
281  //
282 
283  if ((Queue = CurrentThread->Queue) != NULL) {
284  KiLockDispatcherDatabaseAtSynchLevel();
285  }
286 
287  //
288  // Acquire the thread lock and the object lock.
289  //
290  // If the object is already signaled, then clear the signaled state,
291  // release the object lock, release the thread lock, and lower IRQL
292  // to its previous value. Otherwise, set the thread state to gate
293  // wait, set the address of the gate object, insert the thread in the
294  // object wait list, set context swap busy, release the object lock,
295  // release the thread lock, and switch to a new thread.
296  //
297 
298  KiAcquireThreadLock(CurrentThread);
299  KiAcquireKobjectLock(Gate);
300  if (Gate->Header.SignalState != 0) {
301  Gate->Header.SignalState = 0;
302  KiReleaseKobjectLock(Gate);
303  KiReleaseThreadLock(CurrentThread);
304  if (Queue != NULL) {
305  KiUnlockDispatcherDatabaseFromSynchLevel();
306  }
307 
308  KeReleaseInStackQueuedSpinLock(&LockHandle);
309  break;
310 
311  } else {
312  WaitBlock = &CurrentThread->WaitBlock[0];
313  WaitBlock->Object = Gate;
314  WaitBlock->Thread = CurrentThread;
315  CurrentThread->WaitMode = WaitMode;
316  CurrentThread->WaitReason = WaitReason;
317  CurrentThread->WaitIrql = LockHandle.OldIrql;
318  CurrentThread->State = GateWait;
319  CurrentThread->GateObject = Gate;
320  InsertTailList(&Gate->Header.WaitListHead, &WaitBlock->WaitListEntry);
321  KiReleaseKobjectLock(Gate);
322  KiSetContextSwapBusy(CurrentThread);
323  KiReleaseThreadLock(CurrentThread);
324 
325  //
326  // If the current thread is associated with a queue object, then
327  // activate another thread if possible.
328  //
329 
330  if (Queue != NULL) {
331  if ((Queue = CurrentThread->Queue) != NULL) {
332  KiActivateWaiterQueue(Queue);
333  }
334 
335  KiUnlockDispatcherDatabaseFromSynchLevel();
336  }
337 
338  KeReleaseInStackQueuedSpinLockFromDpcLevel(&LockHandle);
339  WaitStatus = (NTSTATUS)KiSwapThread(CurrentThread, KeGetCurrentPrcb());
340  if (WaitStatus == STATUS_SUCCESS) {
341  return;
342  }
343  }
344 
345  } while (TRUE);
346 
347  return;
348 }
