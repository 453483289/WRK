29 KiInitializeUserApc (

38 /*++
39 
40 Routine Description:
41 
42  This function is called to initialize the context for a user mode APC.
43 
44 Arguments:
45 
46  ExceptionFrame - Supplies a pointer to an exception frame.
47 
48  TrapFrame - Supplies a pointer to a trap frame.
49 
50  NormalRoutine - Supplies a pointer to the user mode APC routine.
51 
52  NormalContext - Supplies a pointer to the user context for the APC
53  routine.
54 
55  SystemArgument1 - Supplies the first system supplied value.
56 
57  SystemArgument2 - Supplies the second system supplied value.
58 
59 Return Value:
60 
61  None.
62 
63 --*/
64 
65 {
66 
67  EXCEPTION_RECORD ExceptionRecord;
68  CONTEXT ContextFrame;
69  LONG Length;
70  ULONG UserStack, TopOfStack;
71  PKAPC_RECORD ApcRecord;
72 
73  //
74  // APCs are not defined for V86 mode; however, it is possible a
75  // thread is trying to set it's context to V86 mode - this isn't
76  // going to work, but we don't want to crash the system so we
77  // check for the possibility before hand.
78  //
79 
80  if (TrapFrame->EFlags & EFLAGS_V86_MASK) {
81  return ;
82  }
83 
84  //
85  // Move machine state from trap and exception frames to the context frame.
86  //
87 
88  ContextFrame.ContextFlags = CONTEXT_FULL | CONTEXT_DEBUG_REGISTERS;
89  KeContextFromKframes(TrapFrame, ExceptionFrame, &ContextFrame);
90 
91  //
92  // Transfer the context information to the user stack, initialize the
93  // APC routine parameters, and modify the trap frame so execution will
94  // continue in user mode at the user mode APC dispatch routine.
95  //
96 
97 
98  try {
99  C_ASSERT (CONTEXT_ALIGN >= __alignof(EXCEPTION_REGISTRATION_RECORD));
100  C_ASSERT (CONTEXT_ALIGN >= __alignof(KAPC_RECORD));
101 
102  ASSERT ((TrapFrame->SegCs & MODE_MASK) != KernelMode); // Assert usermode frame
103 
104  //
105  // Compute length of context record and new aligned user stack pointer.
106  // Make sure to include space for a double-word aligned exception registration
107  // record as well. For compatibility the exception registration must follow the
108  // context record.
109  //
110 
111  TopOfStack = (ContextFrame.Esp & ~(__alignof(EXCEPTION_REGISTRATION_RECORD)-1));
112  Length = CONTEXT_ALIGNED_SIZE + sizeof(KAPC_RECORD);
113  UserStack = ((TopOfStack - sizeof (EXCEPTION_REGISTRATION_RECORD)) & ~CONTEXT_ROUND) -
114  Length;
115 
116  //
117  // Probe user stack area for writability and then transfer the context
118  // record to the user stack. Note the minimum alignment is used as
119  // the above code ensures the proper alignment of the addresses.
120  //
121 
122  ProbeForWrite ((PCHAR)UserStack, TopOfStack-UserStack, 1);
123  NT_ASSERT (((ULONG_PTR)(UserStack + sizeof(KAPC_RECORD)) & CONTEXT_ROUND) == 0);
124  RtlCopyMemory ((PULONG)(UserStack + sizeof(KAPC_RECORD)),
125  &ContextFrame, sizeof(CONTEXT));
126 
127  //
128  // Force correct R3 selectors into TrapFrame.
129  //
130 
131  TrapFrame->SegCs = SANITIZE_SEG(KGDT_R3_CODE, UserMode);
132  TrapFrame->HardwareSegSs = SANITIZE_SEG(KGDT_R3_DATA, UserMode);
133  TrapFrame->SegDs = SANITIZE_SEG(KGDT_R3_DATA, UserMode);
134  TrapFrame->SegEs = SANITIZE_SEG(KGDT_R3_DATA, UserMode);
135  TrapFrame->SegFs = SANITIZE_SEG(KGDT_R3_TEB, UserMode);
136  TrapFrame->SegGs = 0;
137  TrapFrame->EFlags = SANITIZE_FLAGS( ContextFrame.EFlags, UserMode );
138 
139  //
140  // If thread is supposed to have IOPL, then force it on in eflags
141  //
142 
143  if (KeGetCurrentThread()->Iopl) {
144  TrapFrame->EFlags |= (EFLAGS_IOPL_MASK & -1); // IOPL = 3
145  }
146 
147  //
148  // Set the address of the user APC routine, the APC parameters, the
149  // new frame pointer, and the new stack pointer in the current trap
150  // frame. Set the continuation address so control will be transferred
151  // to the user APC dispatcher.
152  //
153 
154  TrapFrame->HardwareEsp = UserStack;
155  TrapFrame->Eip = (ULONG)KeUserApcDispatcher;
156  TrapFrame->ErrCode = 0;
157 
158  ApcRecord = (PKAPC_RECORD)UserStack;
159  ApcRecord->NormalRoutine = NormalRoutine;
160  ApcRecord->NormalContext = NormalContext;
161  ApcRecord->SystemArgument1 = SystemArgument1;
162  ApcRecord->SystemArgument2 = SystemArgument2;
163  } except (KiCopyInformation(&ExceptionRecord,
164  (GetExceptionInformation())->ExceptionRecord)) {
165 
166  //
167  // Lower the IRQL to PASSIVE_LEVEL, set the exception address to
168  // the current program address, and raise an exception by calling
169  // the exception dispatcher.
170  //
171  // N.B. The IRQL is lowered to PASSIVE_LEVEL to allow APC interrupts
172  // during the dispatching of the exception. The current thread
173  // will be terminated during the dispatching of the exception,
174  // but lowering of the IRQL is required to enable the debugger
175  // to obtain the context of the current thread.
176  //
177 
178  KeLowerIrql(PASSIVE_LEVEL);
179  ExceptionRecord.ExceptionAddress = (PVOID)(TrapFrame->Eip);
180  KiDispatchException(&ExceptionRecord,
181  ExceptionFrame,
182  TrapFrame,
183  UserMode,
184  TRUE);
185 
186  }
187  return;
188 }
