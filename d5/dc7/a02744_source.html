47 PENTRYHEADER WmipAllocEntry(
48  PCHUNKINFO ChunkInfo
49  )
50 /*++
51 
52 Routine Description:
53 
54  This routine will allocate a single structure within a list of chunks
55  of structures.
56 
57 Arguments:
58 
59  ChunkInfo describes the chunks of structures
60 
61 Return Value:
62 
63  Pointer to structure or NULL if one cannot be allocated. Entry returns
64  with its refcount set to 1
65 
66 --*/
67 {
68  PLIST_ENTRY ChunkList, EntryList, FreeEntryHead;
69  PCHUNKHEADER Chunk;
70  PUCHAR EntryPtr;
71  ULONG EntryCount, ChunkSize;
72  PENTRYHEADER Entry;
73  ULONG i;
74 
75  PAGED_CODE();
76 
77  WmipEnterSMCritSection();
78  ChunkList = ChunkInfo->ChunkHead.Flink;
79 
80  //
81  // Loop over all chunks to see if any chunk has a free entry for us
82  while(ChunkList != &ChunkInfo->ChunkHead)
83  {
84  Chunk = CONTAINING_RECORD(ChunkList, CHUNKHEADER, ChunkList);
85  if (! IsListEmpty(&Chunk->FreeEntryHead))
86  {
87  EntryList = RemoveHeadList(&Chunk->FreeEntryHead);
88  Chunk->EntriesInUse++;
89  WmipLeaveSMCritSection();
90  Entry = (CONTAINING_RECORD(EntryList,
91  ENTRYHEADER,
92  FreeEntryList));
93  WmipAssert(Entry->Flags & FLAG_ENTRY_ON_FREE_LIST);
94  memset(Entry, 0, ChunkInfo->EntrySize);
95  Entry->Chunk = Chunk;
96  Entry->RefCount = 1;
97  Entry->Flags = ChunkInfo->InitialFlags;
98  Entry->Signature = ChunkInfo->Signature;
99 #if DBG
100  InterlockedIncrement(&ChunkInfo->AllocCount);
101 #endif
102  return(Entry);
103  }
104  ChunkList = ChunkList->Flink;
105  }
106  WmipLeaveSMCritSection();
107 
108  //
109  // There are no more free entries in any of the chunks. Allocate a new
110  // chunk if we can
111  ChunkSize = (ChunkInfo->EntrySize * ChunkInfo->EntriesPerChunk) +
112  sizeof(CHUNKHEADER);
113  Chunk = (PCHUNKHEADER)ExAllocatePoolWithTag(PagedPool,
114  ChunkSize,
115  ChunkInfo->Signature);
116  if (Chunk != NULL)
117  {
118  //
119  // Initialize the chunk by building the free list of entries within
120  // it while also initializing each entry.
121  memset(Chunk, 0, ChunkSize);
122 
123  FreeEntryHead = &Chunk->FreeEntryHead;
124  InitializeListHead(FreeEntryHead);
125 
126  EntryPtr = (PUCHAR)Chunk + sizeof(CHUNKHEADER);
127  EntryCount = ChunkInfo->EntriesPerChunk - 1;
128 
129  for (i = 0; i < EntryCount; i++)
130  {
131  Entry = (PENTRYHEADER)EntryPtr;
132  Entry->Chunk = Chunk;
133  Entry->Flags = FLAG_ENTRY_ON_FREE_LIST;
134  InsertHeadList(FreeEntryHead,
135  &((PENTRYHEADER)EntryPtr)->FreeEntryList);
136  EntryPtr = EntryPtr + ChunkInfo->EntrySize;
137  }
138  //
139  // EntryPtr now points to the last entry in the chunk which has not
140  // been placed on the free list. This will be the entry returned
141  // to the caller.
142  Entry = (PENTRYHEADER)EntryPtr;
143  Entry->Chunk = Chunk;
144  Entry->RefCount = 1;
145  Entry->Flags = ChunkInfo->InitialFlags;
146  Entry->Signature = ChunkInfo->Signature;
147 
148  Chunk->EntriesInUse = 1;
149 
150  //
151  // Now place the newly allocated chunk onto the list of chunks
152  WmipEnterSMCritSection();
153  InsertHeadList(&ChunkInfo->ChunkHead, &Chunk->ChunkList);
154  WmipLeaveSMCritSection();
155 
156  } else {
157  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Could not allocate memory for new chunk %x\n",
158  ChunkInfo));
159  Entry = NULL;
160  }
161  return(Entry);
162 }

164 void WmipFreeEntry(
165  PCHUNKINFO ChunkInfo,
166  PENTRYHEADER Entry
167  )
168 /*++
169 
170 Routine Description:
171 
172  This routine will free an entry within a chunk and if the chunk has no
173  more allocated entries then the chunk will be returned to the pool.
174 
175 Arguments:
176 
177  ChunkInfo describes the chunks of structures
178 
179  Entry is the chunk entry to free
180 
181 Return Value:
182 
183 
184 --*/
185 {
186  PCHUNKHEADER Chunk;
187 
188  PAGED_CODE();
189 
190  WmipAssert(Entry != NULL);
191  WmipAssert(! (Entry->Flags & FLAG_ENTRY_ON_FREE_LIST));
192  WmipAssert(Entry->Flags & FLAG_ENTRY_INVALID);
193  WmipAssert(Entry->RefCount == 0);
194  WmipAssert(Entry->Signature == ChunkInfo->Signature);
195 
196  Chunk = Entry->Chunk;
197  WmipAssert(Chunk->EntriesInUse > 0);
198 
199  WmipEnterSMCritSection();
200  if ((--Chunk->EntriesInUse == 0) &&
201  (ChunkInfo->ChunkHead.Blink != &Chunk->ChunkList))
202  {
203  //
204  // We return the chunks memory back to the heap if there are no
205  // more entries within the chunk in use and the chunk was not the
206  // first chunk to be allocated.
207  RemoveEntryList(&Chunk->ChunkList);
208  WmipLeaveSMCritSection();
209  ExFreePoolWithTag(Chunk, ChunkInfo->Signature);
210  } else {
211  //
212  // Otherwise just mark the entry as free and put it back on the
213  // chunks free list.
214 #if DBG
215 // memset(Entry, 0xCCCCCCCC, ChunkInfo->EntrySize);
216 #endif
217  Entry->Flags = FLAG_ENTRY_ON_FREE_LIST;
218  Entry->Signature = 0;
219  InsertTailList(&Chunk->FreeEntryHead, &Entry->FreeEntryList);
220  WmipLeaveSMCritSection();
221  }
222 }

225 ULONG WmipUnreferenceEntry(
226  PCHUNKINFO ChunkInfo,
227  PENTRYHEADER Entry
228  )
229 /*+++
230 
231 Routine Description:
232 
233  This routine will remove a reference count from the entry and if the
234  reference count reaches zero then the entry is removed from its active
235  list and then cleaned up and finally freed.
236 
237 Arguments:
238 
239  ChunkInfo points at structure that describes the entry
240 
241  Entry is the entry to unreference
242 
243 Return Value:
244 
245  New refcount of the entry
246 
247 ---*/
248 {
249  ULONG RefCount;
250 
251  PAGED_CODE();
252 
253  WmipAssert(Entry != NULL);
254  WmipAssert(Entry->RefCount > 0);
255  WmipAssert(Entry->Signature == ChunkInfo->Signature);
256 
257  WmipEnterSMCritSection();
258  InterlockedDecrement((PLONG)&Entry->RefCount);
259  RefCount = Entry->RefCount;
260 
261  if (RefCount == 0)
262  {
263  //
264  // Entry has reached a ref count of 0 so mark it as invalid and remove
265  // it from its active list.
266  Entry->Flags |= FLAG_ENTRY_INVALID;
267 
268  if ((Entry->InUseEntryList.Flink != NULL) &&
269  (Entry->Flags & FLAG_ENTRY_REMOVE_LIST))
270  {
271  RemoveEntryList(&Entry->InUseEntryList);
272  }
273 
274  WmipLeaveSMCritSection();
275 
276  if (ChunkInfo->EntryCleanup != NULL)
277  {
278  //
279  // Call cleanup routine to free anything contained by the entry
280  (*ChunkInfo->EntryCleanup)(ChunkInfo, Entry);
281  }
282 
283  //
284  // Place the entry back on its free list
285  WmipFreeEntry(ChunkInfo, Entry);
286  } else {
287  WmipLeaveSMCritSection();
288  }
289  return(RefCount);
290 }
