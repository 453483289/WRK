81 /*++
82 
83 Routine Description:
84 
85  Perform phase 0 security initialization.
86 
87  This includes:
88 
89  - Initialize LUID allocation
90  - Initialize security global variables
91  - initialize the token object.
92  - Initialize the necessary security components of the boot thread/process
93 
94 
95 Arguments:
96 
97  None.
98 
99 Return Value:
100 
101  TRUE - Initialization was successful.
102 
103  FALSE - Initialization Failed.
104 
105 --*/
106 
107 {
108 
109  PAGED_CODE();
110 
111  //
112  // LUID allocation services are needed by security prior to phase 0
113  // Executive initialization. So, LUID initialization is performed
114  // here
115  //
116 
117  if (ExLuidInitialization() == FALSE) {
118  KdPrint(("Security: Locally Unique ID initialization failed.\n"));
119  return FALSE;
120  }
121 
122  //
123  // Initialize security global variables
124  //
125 
126  if (!SepVariableInitialization()) {
127  KdPrint(("Security: Global variable initialization failed.\n"));
128  return FALSE;
129  }
130 
131  //
132  // Perform Phase 0 Reference Monitor Initialization.
133  //
134 
135  if (!SepRmInitPhase0()) {
136  KdPrint(("Security: Ref Mon state initialization failed.\n"));
137  return FALSE;
138  }
139 
140  //
141  // Initialize the token object type.
142  //
143 
144  if (!SepTokenInitialization()) {
145  KdPrint(("Security: Token object initialization failed.\n"));
146  return FALSE;
147  }
148 
149 // //
150 // // Initialize auditing structures
151 // //
152 //
153 // if (!SepAdtInitializePhase0()) {
154 // KdPrint(("Security: Auditing initialization failed.\n"));
155 // return FALSE;
156 // }
157 //
158  //
159  // Initialize SpinLock and list for the LSA worker thread
160  //
161 
162  //
163  // Initialize the work queue spinlock, list head, and semaphore
164  // for each of the work queues.
165  //
166 
167  if (!SepInitializeWorkList()) {
168  KdPrint(("Security: Unable to initialize work queue\n"));
169  return FALSE;
170  }
171 
172  //
173  // Initialize the security fields of the boot thread.
174  //
175  PsGetCurrentThread()->ImpersonationInfo = NULL;
176  PS_CLEAR_BITS (&PsGetCurrentThread()->CrossThreadFlags, PS_CROSS_THREAD_FLAGS_IMPERSONATING);
177  ObInitializeFastReference (&PsGetCurrentProcess()->Token, NULL);
178 
179  ObInitializeFastReference (&PsGetCurrentProcess()->Token, SeMakeSystemToken());
180 
181  return ( !ExFastRefObjectNull (PsGetCurrentProcess()->Token) );
182 }
