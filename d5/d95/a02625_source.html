122 PVOID
123 RtlpAllocateAtom(
124  IN ULONG NumberOfBytes,
125  IN ULONG Tag
126  )
127 {
128  PRTLP_ATOM_QUOTA AllocatedBlock;
129 
130  NumberOfBytes += sizeof(RTLP_ATOM_QUOTA);
131 
132  AllocatedBlock = ExAllocatePoolWithTag( PagedPool,
133  NumberOfBytes,
134  Tag );
135  if (AllocatedBlock) {
136  AllocatedBlock->QuotaBlock
137  = PsChargeSharedPoolQuota( PsGetCurrentProcess(),
138  NumberOfBytes,
139  0 );
140  if (! AllocatedBlock->QuotaBlock) {
141  ExFreePoolWithTag( AllocatedBlock, Tag );
142  AllocatedBlock = NULL;
143  } else {
144  AllocatedBlock->PagedAmount = NumberOfBytes;
145  AllocatedBlock++;
146  }
147  }
148 
149  return AllocatedBlock;
150 }

168 
169  ExFreePool( AllocatedBlock );
170  return;

313 RtlCreateAtomTable(
314  IN ULONG NumberOfBuckets,
315  OUT PVOID *AtomTableHandle
316  )
317 {
318  NTSTATUS Status;
319  PRTL_ATOM_TABLE p;
320  ULONG Size;
321 
322  RTL_PAGED_CODE();
323  Status = STATUS_SUCCESS;
324  if (*AtomTableHandle == NULL) {
325  if (NumberOfBuckets <= 1) {
326  NumberOfBuckets = RTL_ATOM_TABLE_DEFAULT_NUMBER_OF_BUCKETS;
327  }
328 
329  Size = sizeof( RTL_ATOM_TABLE ) +
330  (sizeof( RTL_ATOM_TABLE_ENTRY ) * (NumberOfBuckets-1));
331 
332  p = (PRTL_ATOM_TABLE)RtlpAllocateAtom( Size, 'TmtA' );
333  if (p == NULL) {
334  Status = STATUS_NO_MEMORY;
335  }
336  else {
337  RtlZeroMemory( p, Size );
338  p->NumberOfBuckets = NumberOfBuckets;
339  if (RtlpInitializeHandleTableForAtomTable( p )) {
340  RtlpInitializeLockAtomTable( p );
341  p->Signature = RTL_ATOM_TABLE_SIGNATURE;
342  *AtomTableHandle = p;
343  }
344  else {
345  Status = STATUS_NO_MEMORY;
346  RtlpFreeAtom( p );
347  }
348  }
349  }
350 
351  return Status;
352 }

355 NTSTATUS
356 RtlDestroyAtomTable(
357  IN PVOID AtomTableHandle
358  )
359 {
360  NTSTATUS Status;
361  PRTL_ATOM_TABLE p = (PRTL_ATOM_TABLE)AtomTableHandle;
362  PRTL_ATOM_TABLE_ENTRY a, aNext, *pa;
363  ULONG i;
364 
365  RTL_PAGED_CODE();
366  Status = STATUS_SUCCESS;
367  if (!RtlpLockAtomTable( p )) {
368  return STATUS_INVALID_PARAMETER;
369  }
370  try {
371  pa = &p->Buckets[ 0 ];
372  for (i=0; i<p->NumberOfBuckets; i++) {
373  aNext = *pa;
374  *pa++ = NULL;
375  while ((a = aNext) != NULL) {
376  aNext = a->HashLink;
377  a->HashLink = NULL;
378  RtlpFreeAtom( a );
379  }
380  }
381  p->Signature = 0;
382  RtlpUnlockAtomTable( p );
383 
384  RtlpDestroyHandleTableForAtomTable( p );
385  RtlpDestroyLockAtomTable( p );
386  RtlZeroMemory( p, sizeof( RTL_ATOM_TABLE ) );
387  RtlpFreeAtom( p );
388  }
389  except (EXCEPTION_EXECUTE_HANDLER) {
390  Status = GetExceptionCode();
391  }
392 
393  return Status;
394 }

438 BOOLEAN
439 RtlpGetIntegerAtom(
440  PWSTR Name,
441  PRTL_ATOM Atom OPTIONAL
442  )
443 {
444  NTSTATUS Status;
445  UNICODE_STRING UnicodeString;
446  PWSTR s;
447  ULONG n;
448  RTL_ATOM Temp;
449 
450  if (((ULONG_PTR)Name & -0x10000) == 0) {
451  Temp = (RTL_ATOM)(USHORT)PtrToUlong(Name);
452  if (Temp >= RTL_ATOM_MAXIMUM_INTEGER_ATOM) {
453  return FALSE;
454  }
455  else {
456  if (Temp == RTL_ATOM_INVALID_ATOM) {
457  Temp = RTL_ATOM_MAXIMUM_INTEGER_ATOM;
458  }
459 
460  if (ARGUMENT_PRESENT( Atom )) {
461  *Atom = Temp;
462  }
463 
464  return TRUE;
465  }
466  }
467  else
468  if (*Name != L'#') {
469  return FALSE;
470  }
471 
472  s = ++Name;
473  while (*s != UNICODE_NULL) {
474  if (*s < L'0' || *s > L'9') {
475  return FALSE;
476  }
477  else {
478  s++;
479  }
480  }
481 
482  n = 0;
483  UnicodeString.Buffer = Name;
484  UnicodeString.Length = (USHORT)((PCHAR)s - (PCHAR)Name);
485  UnicodeString.MaximumLength = UnicodeString.Length;
486  Status = RtlUnicodeStringToInteger( &UnicodeString, 10, &n );
487  if (NT_SUCCESS( Status )) {
488  if (ARGUMENT_PRESENT( Atom )) {
489  if (n == 0 || n > RTL_ATOM_MAXIMUM_INTEGER_ATOM) {
490  *Atom = RTL_ATOM_MAXIMUM_INTEGER_ATOM;
491  }
492  else {
493  *Atom = (RTL_ATOM)n;
494  }
495  }
496 
497  return TRUE;
498  }
499  else {
500  return FALSE;
501  }
502 }

504 PRTL_ATOM_TABLE_ENTRY
505 RtlpHashStringToAtom(
506  IN PRTL_ATOM_TABLE p,
507  IN PWSTR Name,
508  OUT PRTL_ATOM_TABLE_ENTRY **PreviousAtom OPTIONAL,
509  OUT PULONG NameLength
510  )
511 {
512  ULONG Length, Hash;
513  WCHAR c;
514  PWCH s;
515  RTL_ATOM Atom;
516  PRTL_ATOM_TABLE_ENTRY *pa, a;
517 
518  if (((ULONG_PTR)Name & -0x10000) == 0) {
519  Atom = (RTL_ATOM)(USHORT)PtrToUlong(Name);
520  a = NULL;
521  if (Atom >= RTL_ATOM_MAXIMUM_INTEGER_ATOM) {
522  a = RtlpAtomMapAtomToHandleEntry( p,
523  (ULONG)(Atom & (USHORT)~RTL_ATOM_MAXIMUM_INTEGER_ATOM)
524  );
525  }
526 
527  if (ARGUMENT_PRESENT( PreviousAtom )) {
528  *PreviousAtom = NULL;
529  }
530 
531  return a;
532  }
533 
534  s = Name;
535  Hash = 0;
536  while (*s != UNICODE_NULL) {
537  c = *s++;
538  if (c < 'a') {
539  NOTHING;
540 
541  } else if (c > 'z') {
542  c = RtlUpcaseUnicodeChar(c);
543 
544  } else {
545  c -= ('a' - 'A');
546  }
547 
548  Hash = Hash + (c << 1) + (c >> 1) + c;
549  }
550 
551  Length = (ULONG) (s - Name);
552  if (Length > RTL_ATOM_MAXIMUM_NAME_LENGTH) {
553  pa = NULL;
554  a = NULL;
555  }
556  else {
557  pa = &p->Buckets[ Hash % p->NumberOfBuckets ];
558  while (a = *pa) {
559  if (a->NameLength == Length && !_wcsicmp( a->Name, Name )) {
560  break;
561  }
562  else {
563  pa = &a->HashLink;
564  }
565  }
566  }
567 
568  if (ARGUMENT_PRESENT( PreviousAtom )) {
569  *PreviousAtom = pa;
570  }
571 
572  if (a == NULL && ARGUMENT_PRESENT( NameLength )) {
573  *NameLength = Length * sizeof( WCHAR );
574  }
575 
576  return a;
577 }

580 NTSTATUS
581 RtlAddAtomToAtomTable(
582  __in PVOID AtomTableHandle,
583  __in PWSTR AtomName,
584  __inout_opt PRTL_ATOM Atom
585  )
586 {
587  NTSTATUS Status;
588  PRTL_ATOM_TABLE p = (PRTL_ATOM_TABLE)AtomTableHandle;
589  PRTL_ATOM_TABLE_ENTRY a, *pa;
590  ULONG NameLength;
591  RTL_ATOM Temp;
592 
593  RTL_PAGED_CODE();
594  if (!RtlpLockAtomTable( p )) {
595  return STATUS_INVALID_PARAMETER;
596  }
597  try {
598  if (RtlpGetIntegerAtom( AtomName, &Temp )) {
599  if (Temp >= RTL_ATOM_MAXIMUM_INTEGER_ATOM) {
600  Temp = RTL_ATOM_INVALID_ATOM;
601  Status = STATUS_INVALID_PARAMETER;
602  }
603  else {
604  Status = STATUS_SUCCESS;
605  }
606 
607  if (ARGUMENT_PRESENT( Atom )) {
608  *Atom = Temp;
609  }
610  }
611  else
612  if (*AtomName == UNICODE_NULL) {
613  Status = STATUS_OBJECT_NAME_INVALID;
614  }
615  else {
616  a = RtlpHashStringToAtom( p, AtomName, &pa, &NameLength );
617  if (a == NULL) {
618  if (pa != NULL) {
619  Status = STATUS_NO_MEMORY;
620  a = RtlpAllocateAtom( FIELD_OFFSET( RTL_ATOM_TABLE_ENTRY, Name ) +
621  NameLength + sizeof( UNICODE_NULL ),
622  'AmtA'
623  );
624  if (a != NULL) {
625  a->HashLink = NULL;
626  a->ReferenceCount = 1;
627  a->Flags = 0;
628  RtlCopyMemory( a->Name, AtomName, NameLength );
629  a->NameLength = (UCHAR)(NameLength / sizeof( WCHAR ));
630  a->Name[ a->NameLength ] = UNICODE_NULL;
631  if (RtlpCreateHandleForAtom( p, a )) {
632  a->Atom = (RTL_ATOM)a->HandleIndex | RTL_ATOM_MAXIMUM_INTEGER_ATOM;
633  *pa = a;
634  if (ARGUMENT_PRESENT( Atom )) {
635  *Atom = a->Atom;
636  }
637 
638  Status = STATUS_SUCCESS;
639  }
640  else {
641  RtlpFreeAtom( a );
642  }
643  }
644  }
645  else {
646  Status = STATUS_INVALID_PARAMETER;
647  }
648  }
649  else {
650  if (!(a->Flags & RTL_ATOM_PINNED)) {
651  if (a->ReferenceCount == 0xFFFF) {
652  KdPrint(( "RTL: Pinning atom (%x) as reference count about to wrap\n", Atom ));
653  a->Flags |= RTL_ATOM_PINNED;
654  }
655  else {
656  a->ReferenceCount += 1;
657  }
658  }
659 
660  if (ARGUMENT_PRESENT( Atom )) {
661  *Atom = a->Atom;
662  }
663 
664  Status = STATUS_SUCCESS;
665  }
666  }
667  }
668  except (EXCEPTION_EXECUTE_HANDLER) {
669  Status = GetExceptionCode();
670  }
671 
672  RtlpUnlockAtomTable( p );
673 
674  return Status;
675 }

739 NTSTATUS
740 RtlDeleteAtomFromAtomTable(
741  IN PVOID AtomTableHandle,
742  IN RTL_ATOM Atom
743  )
744 {
745  NTSTATUS Status;
746  PRTL_ATOM_TABLE p = (PRTL_ATOM_TABLE)AtomTableHandle;
747  PRTL_ATOM_TABLE_ENTRY a, *pa;
748 
749  RTL_PAGED_CODE();
750  if (!RtlpLockAtomTable( p )) {
751  return STATUS_INVALID_PARAMETER;
752  }
753  try {
754  Status = STATUS_INVALID_HANDLE;
755  if (Atom >= RTL_ATOM_MAXIMUM_INTEGER_ATOM) {
756  a = RtlpAtomMapAtomToHandleEntry( p,
757  (ULONG)(Atom & (USHORT)~RTL_ATOM_MAXIMUM_INTEGER_ATOM)
758  );
759  if (a != NULL && a->Atom == Atom) {
760  Status = STATUS_SUCCESS;
761  if (a->Flags & RTL_ATOM_PINNED) {
762  KdPrint(( "RTL: Ignoring attempt to delete a pinned atom (%x)\n", Atom ));
763  Status = STATUS_WAS_LOCKED; // This is a success status code!
