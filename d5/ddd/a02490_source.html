64  IN PMMPTE PointerPte,
65  IN ULONG_PTR WriteOperation,
66  IN KPROCESSOR_MODE PreviousMode,
67  IN ULONG Protection,
68  IN PVOID TrapInformation,

72 /*++
73 
74 Routine Description:
75 
76 
77 
78 Arguments:
79 
80  PointerPte - Supplies the pointer to the PTE which caused the
81  page fault.
82 
83  WriteOperation - Supplies nonzero if the operation is a write, 0 if
84  the operation is a read.
85 
86  PreviousMode - Supplies the previous mode, one of UserMode or KernelMode.
87 
88  Protection - Supplies the protection mask to check.
89 
90  TrapInformation - Supplies the trap information.
91 
92  CallerHoldsPfnLock - Supplies TRUE if the PFN lock is held, FALSE otherwise.
93 
94 Return Value:
95 
96  Returns TRUE if access to the page is allowed, FALSE otherwise.
97 
98 Environment:
99 
100  Kernel mode, APCs disabled.
101 
102 --*/
103 
104 {
105  MMPTE PteContents;
106  KIRQL OldIrql;
107  PMMPFN Pfn1;
108 
109  //
110  // Check to see if the owner bit allows access to the previous mode.
111  // Access is not allowed if the owner is kernel and the previous
112  // mode is user. Access is also disallowed if the write operation
113  // is true and the write field in the PTE is false.
114  //
115 
116  //
117  // If both an access violation and a guard page violation could
118  // occur for the page, the access violation must be returned.
119  //
120 
121  if (PreviousMode == UserMode) {
122  if (PointerPte > MiHighestUserPte) {
123  return STATUS_ACCESS_VIOLATION;
124  }
125  }
126 
127  PteContents = *PointerPte;
128 
129  if (PteContents.u.Hard.Valid == 1) {
130 
131  //
132  // Valid pages cannot be guard page violations.
133  //
134 
135  if (WriteOperation != 0) {
136 
137  if ((PteContents.u.Hard.Write == 1) ||
138  (PteContents.u.Hard.CopyOnWrite == 1)) {
139 
140  return STATUS_SUCCESS;
141  }
142  return STATUS_ACCESS_VIOLATION;
143  }
144 
145  return STATUS_SUCCESS;
146  }
147 
148  if (WriteOperation != 0) {
149  WriteOperation = 1;
150  }
151 
152  if ((MmReadWrite[Protection] - (CCHAR)WriteOperation) < 10) {
153  return STATUS_ACCESS_VIOLATION;
154  }
155 
156  //
157  // Check for a guard page fault.
158  //
159 
160  if (MI_IS_GUARD (Protection)) {
161 
162  //
163  // If this thread is attached to a different process,
164  // return an access violation rather than a guard
165  // page exception. This prevents problems with unwanted
166  // stack expansion and unexpected guard page behavior
167  // from debuggers.
168  //
169 
170  if (KeIsAttachedProcess ()) {
171  return STATUS_ACCESS_VIOLATION;
172  }
173 
174  //
175  // If this is a user mode SLIST fault then don't remove the guard bit.
176  //
177 
178  if (KeInvalidAccessAllowed (TrapInformation)) {
179  return STATUS_ACCESS_VIOLATION;
180  }
181 
182  //
183  // Check to see if this is a transition PTE. If so, the
184  // PFN database original contents field needs to be updated.
185  //
186 
187  if ((PteContents.u.Soft.Transition == 1) &&
188  (PteContents.u.Soft.Prototype == 0)) {
189 
190  //
191  // Acquire the PFN lock and check to see if the
192  // PTE is still in the transition state. If so,
193  // update the original PTE in the PFN database.
194  //
195 
196  SATISFY_OVERZEALOUS_COMPILER (OldIrql = PASSIVE_LEVEL);
197 
198  if (CallerHoldsPfnLock == FALSE) {
199  LOCK_PFN (OldIrql);
200  }
201 
202  PteContents = *PointerPte;
203  if ((PteContents.u.Soft.Transition == 1) &&
204  (PteContents.u.Soft.Prototype == 0)) {
205 
206  //
207  // Still in transition, update the PFN database.
208  //
209 
210  Pfn1 = MI_PFN_ELEMENT (PteContents.u.Trans.PageFrameNumber);
211 
212  //
213  // Note that forked processes using guard pages only take the
214  // guard page fault when the first thread in either process
215  // access the address. This seems to be the best behavior we
216  // can provide users of this API as we must allow the first
217  // thread to make forward progress and the guard attribute is
218  // stored in the shared fork prototype PTE.
219  //
220 
221  if (PteContents.u.Soft.Protection == MM_NOACCESS) {
222  ASSERT ((Pfn1->u3.e1.PrototypePte == 1) &&
223  (MiLocateCloneAddress (PsGetCurrentProcess (), Pfn1->PteAddress) != NULL));
224  if (CallerHoldsPfnLock == FALSE) {
225  UNLOCK_PFN (OldIrql);
226  }
227  return STATUS_ACCESS_VIOLATION;
228  }
229 
230  ASSERT ((Pfn1->u3.e1.PrototypePte == 0) ||
231  (MiLocateCloneAddress (PsGetCurrentProcess (), Pfn1->PteAddress) != NULL));
232  Pfn1->OriginalPte.u.Soft.Protection =
233  Protection & ~MM_GUARD_PAGE;
234  }
235  if (CallerHoldsPfnLock == FALSE) {
236  UNLOCK_PFN (OldIrql);
237  }
238  }
239 
240  PointerPte->u.Soft.Protection = Protection & ~MM_GUARD_PAGE;
241 
242  return STATUS_GUARD_PAGE_VIOLATION;
243  }
244 
245  return STATUS_SUCCESS;
246 }

256 /*++
257 
258 Routine Description:
259 
260  This routine checks to see if the faulting address is within
261  the stack limits and if so tries to create another guard
262  page on the stack. A stack overflow is returned if the
263  creation of a new guard page fails or if the stack is in
264  the following form:
265 
266  +----------------+ DeallocationBStore
267  | | last page of bstore (always no access)
268  +----------------+
269  | allocated |
270  | |
271  ^ | |
272  | | |
273  BSP | |
274  growth | |
275  | ... |
276  | ... |
277 
278  stack +----------------+ StackBase
279  growth | |
280  | | |
281  v | |
282  | allocated |
283  | |
284  | ... |
285  | |
286  +----------------+
287  | old guard page | <- faulting address is in this page.
288  +----------------+
289  | |
290  +----------------+
291  | | last page of stack (always no access)
292  +----------------+ DeallocationStack
293 
294  In this case, the page before the last page is committed, but
295  not as a guard page and a STACK_OVERFLOW condition is returned.
296 
297 Arguments:
298 
299  FaultingAddress - Supplies the virtual address of the page which
300  was a guard page.
301 
302 Return Value:
303 
304  NTSTATUS.
305 
306 Environment:
307 
308  Kernel mode. No mutexes held.
309 
310 --*/
311

313  PTEB Teb;
314  PPEB Peb;
315  PMMVAD Vad;
316  ULONG GlobalFlag;
317  ULONG_PTR NextPage;
318  SIZE_T RegionSize;
319  SIZE_T AdditionalSize;

322  PVOID MaximumAddress;
323  PVOID *StackLimit;

326  ULONG_PTR PageSize;

328  ULONG ProtectionFlags;
329  ULONG OldProtection;
330  PVOID NextVaToQuery;
331  PKTRAP_FRAME TrapFrame;
332 #if defined(_WIN64)
333  PTEB32 Teb32;
334 
335  Teb32 = NULL;
336 #endif
337 
338  //
339  // Make sure we are not recursing with the address space mutex held.
340  //
341 
342  Thread = PsGetCurrentThread ();
343 
344  if (Thread->AddressSpaceOwner == 1) {
345  ASSERT (KeAreAllApcsDisabled () == TRUE);
346  return STATUS_GUARD_PAGE_VIOLATION;
347  }
348 
349  //
350  // If this thread is attached to a different process,
351  // return an access violation rather than a guard
352  // page exception. This prevents problems with unwanted
353  // stack expansion and unexpected guard page behavior
354  // from debuggers.
355  //
356  // Note we must bail when attached because we reference fields in our
357  // TEB below which have no relationship to the virtual address space
358  // of the process we are attached to. Rather than introduce random
359  // behavior into applications, it's better to consistently return an AV
360  // for this scenario (one thread trying to grow another's stack).
361  //
362 
363  if (KeIsAttachedProcess ()) {
364  return STATUS_GUARD_PAGE_VIOLATION;
365  }
366 
367  TrapFrame = (PKTRAP_FRAME) OpaqueTrapInformation;
368 
369  Teb = Thread->Tcb.Teb;
370 
371  //
372  // Use an exception handler as the TEB is within the user's
373  // address space.
374  //
375 
376  try {
377 
378  StackBase = Teb->NtTib.StackBase;
379  DeallocationStack = Teb->DeallocationStack;
380 
381  RegionSize = (SIZE_T) Teb->GuaranteedStackBytes;
382 
383  } except (EXCEPTION_EXECUTE_HANDLER) {
384 
385  //
386  // An exception has occurred during the referencing of the
387  // TEB or TIB, just return a guard page violation and
388  // don't deal with the stack overflow.
389  //
390 
391  return STATUS_GUARD_PAGE_VIOLATION;
392  }
393 
394  DeallocationStack = PAGE_ALIGN (DeallocationStack);
395 
396  RegionSize = ROUND_TO_PAGES (RegionSize);
397 
398  Process = NULL;
399 
400  PageSize = PAGE_SIZE;
401  ProtectionFlags = PAGE_READWRITE | PAGE_GUARD;
402 
403  StackLimit = &Teb->NtTib.StackLimit;
404 
405  MaximumAddress = DeallocationStack;
406 
407  //
408  // If there is a stack guarantee (RegionSize nonzero), then increase
409  // our guard page size by 1 so that even a subsequent fault that occurs
410  // midway (instead of at the beginning) through the first guard page
411  // will have the extra page to preserve the guarantee.
412  //
413  // At the same time, reduce the MaximumAddress so that the
414  // highest possible guard page can still be used (otherwise our
415  // size increase would falsely disallow this).
416  //
417 
418  if (RegionSize != 0) {
419  RegionSize += PAGE_SIZE;
420  MaximumAddress = (PVOID) ((PCHAR) DeallocationStack - PAGE_SIZE);
421  }
422 
423  if (RegionSize < GUARD_PAGE_SIZE) {
424  RegionSize = GUARD_PAGE_SIZE;
425  }
426 
427  AdditionalSize = RegionSize - GUARD_PAGE_SIZE;
428 
429  NextPage = (ULONG_PTR) PAGE_ALIGN (FaultingAddress) - RegionSize;
430 
431  //
432  // The faulting address must be below the stack base and
433  // above the stack limit.
434  //
435 
436  if ((FaultingAddress >= StackBase) ||
437  (FaultingAddress < DeallocationStack)) {
438 
439  //
440  // Not within the native stack.
441  //
442 
443 #if defined (_WIN64)
444 
445  //
446  // Also check for the 32-bit stack if this is a Wow64 process.
447  //
448 
449  Process = PsGetCurrentProcessByThread (Thread);
450 
451  if (Process->Wow64Process != NULL) {
452 
453  try {
454 
455  Teb32 = WOW64_GET_TEB32 (Teb);
456 
457  if (Teb32 == NULL) {
458  ExRaiseStatus (STATUS_GUARD_PAGE_VIOLATION);
459  }

462  sizeof(TEB32),
463  sizeof(ULONG));
464 
465  StackBase = (PVOID) (ULONG_PTR) Teb32->NtTib.StackBase;
466  DeallocationStack = (PVOID) (ULONG_PTR) Teb32->DeallocationStack;
467  RegionSize = (SIZE_T) Teb32->GuaranteedStackBytes;
468 
469  } except (EXCEPTION_EXECUTE_HANDLER) {
470 
471  //
472  // An exception has occurred during the referencing of the
473  // TEB or TIB, just return a guard page violation and
474  // don't deal with the stack overflow.
475  //
476 
477  return STATUS_GUARD_PAGE_VIOLATION;
478  }
479 
480  DeallocationStack = PAGE_4K_ALIGN (DeallocationStack);
481 
482  if ((FaultingAddress >= StackBase) ||
483  (FaultingAddress < DeallocationStack)) {
484 
485  //
486  // Not within the stack.
487  //
488 
489  return STATUS_GUARD_PAGE_VIOLATION;
490  }
491 
492  PageSize = PAGE_4K;
493  MaximumAddress = DeallocationStack;
494 
495  //
496  // If there is a stack guarantee (RegionSize nonzero), then
497  // increase our guard page size by 1 so that even a subsequent
498  // fault that occurs midway (instead of at the beginning)
499  // through the first guard page will have the extra page
500  // to preserve the guarantee.
501  //
502  // At the same time, reduce the MaximumAddress so that the
503  // highest possible guard page can still be used (otherwise our
504  // size increase would falsely disallow this).
505  //
506 
507  if (RegionSize != 0) {
508  RegionSize += PAGE_4K;
509  MaximumAddress = (PVOID) ((PCHAR) DeallocationStack - PAGE_4K);
510  }
511 
512  RegionSize = ROUND_TO_4K_PAGES (RegionSize);
513 
514  if (RegionSize < PAGE_4K) {
515  RegionSize = PAGE_4K;
516  }
517 
518  StackLimit = (PVOID *)&Teb32->NtTib.StackLimit;
519 
520  AdditionalSize = RegionSize - PAGE_4K;
521 
522  NextPage = (ULONG_PTR) PAGE_4K_ALIGN (FaultingAddress) - RegionSize;
523 
524 #if EMULATE_USERMODE_STACK_4K
525 
526  //
527  // Don't set the guard bit in the native PTE - just set
528  // it in the AltPte.
529  //
530 
531  ProtectionFlags &= ~PAGE_GUARD;
532 #endif
533 
534  goto ExtendTheStack;
535  }
536 
537 #endif
538  //
539  // Not within the stack.
540  //
541 
542  return STATUS_GUARD_PAGE_VIOLATION;
543  }
544 
545 #if defined (_WIN64)
546 ExtendTheStack:
547 #endif
548 
549  //
550  // If the image was marked for no stack extensions,
551  // return stack overflow immediately.
552  //
553 
554  Process = PsGetCurrentProcessByThread (Thread);
555  Peb = Process->Peb;
556 
557  try {
558  GlobalFlag = Peb->NtGlobalFlag;
559  } except (EXCEPTION_EXECUTE_HANDLER) {
560 
561  //
562  // An exception has occurred during the referencing of the
563  // PEB, just return a guard page violation and
564  // don't deal with the stack overflow.
565  //
566 
567  return STATUS_GUARD_PAGE_VIOLATION;
568  }
569 
570  //
571  // NextPage is within the current stack, if there is ample
572  // room for another guard page then attempt to commit it.
573  //
574 
575  if ((NextPage - PageSize) <= (ULONG_PTR) MaximumAddress) {
576 
577  //
578  // There is no more room for expansion, attempt to
579  // commit the page(s) before the last page of the stack.
580  //
581 
582  NextPage = (ULONG_PTR) DeallocationStack + PageSize;
583 
584  status = ZwAllocateVirtualMemory (NtCurrentProcess (),
585  (PVOID *)&NextPage,
586  0,
587  &RegionSize,
588  MEM_COMMIT,
589  PAGE_READWRITE);
590 
591  if (NT_SUCCESS (status)) {
592 
593  try {
594 #if defined(_WIN64)
595  if (Teb32 != NULL) {
596  *(PULONG) StackLimit = (ULONG) NextPage;
597  }
598  else
599 #endif

601 
602  } except (EXCEPTION_EXECUTE_HANDLER) {
603 
604  //
605  // An exception has occurred during the referencing of the
606  // TIB, just return a guard page violation and
607  // don't deal with the stack overflow.
608  //
609 
610  return STATUS_GUARD_PAGE_VIOLATION;
611  }
612  }
613 
614  return STATUS_STACK_OVERFLOW;
615  }
616 
617  if ((GlobalFlag & FLG_DISABLE_STACK_EXTENSION)
618 #if defined(_WIN64)
619  //
620  // Don't do this for emulation processes because it's not worth the
621  // work to clear the guard bits in the native stack, the emulation
622  // stack and the backing store PTEs.
623  //
624  && (Teb32 == NULL)
625 #endif
626  ) {
627 
628  //
629  // Since guard pages may get inserted anywhere in the stack (including
630  // prior to the committed region), make sure the PTE is not committed
631  // before failing to grow it.
632  //
633 
634  LOCK_ADDRESS_SPACE (Process);
635 
636  Vad = MiCheckForConflictingVad (Process,
637  NextPage,
638  NextPage);
639 
640  if (Vad != NULL) {
641 
642  LOCK_WS_SHARED (Thread, Process);
643 
644  if (MiQueryAddressState ((PVOID) NextPage,
645  Vad,
646  Process,
647  &OldProtection,
648  &NextVaToQuery) != MEM_RESERVE) {
649 
650  //
651  // The state is committed (not reserved) so allow the growth.
652  //
653 
654  GlobalFlag = 0;
655  }
656 
657  UNLOCK_WS_SHARED (Thread, Process);
658  }
659 
660  UNLOCK_ADDRESS_SPACE (Process);
661 
662  if (GlobalFlag & FLG_DISABLE_STACK_EXTENSION) {
663 
664  //
665  // We're handing back an overflow so remove any guard bits now
666  // so the app can handle it.
667  //
668 
669  MiClearStackGuardBits (FaultingAddress,
670  StackBase,
671  DeallocationStack);
672 
673  return STATUS_STACK_OVERFLOW;
674  }
675  }
676 
677  try {
678 #if defined(_WIN64)
679 
680  if (Teb32 != NULL) {
681 
682  //
683  // Update the 32-bit stack limit.
684  //
685 
686  *(PULONG) StackLimit = (ULONG) (NextPage + RegionSize);
687  }
688  else
689 #endif
690  *StackLimit = (PVOID)(NextPage + RegionSize);
691 
692  } except (EXCEPTION_EXECUTE_HANDLER) {
693 
694  //
695  // An exception has occurred during the referencing of the
696  // TEB or TIB, just return a guard page violation and
697  // don't deal with the stack overflow.
698  //
699 
700  return STATUS_GUARD_PAGE_VIOLATION;
701  }
702 
703  //
704  // Set the guard page(s). For Wow64 processes the protection
705  // will not contain the PAGE_GUARD bit. This is ok since in these
706  // cases we will set the bit for the top emulated 4K page.
707  //
708 
709  status = ZwAllocateVirtualMemory (NtCurrentProcess (),
710  (PVOID *)&NextPage,
711  0,
712  &RegionSize,
713  MEM_COMMIT,
714  ProtectionFlags);
715 
716  if (NT_SUCCESS (status) || (status == STATUS_ALREADY_COMMITTED)) {
717 
718 #if EMULATE_USERMODE_STACK_4K
719 
720  if (Teb32 != NULL) {
721 
722  LOCK_ADDRESS_SPACE (Process);
723 
724  MiProtectFor4kPage ((PVOID)NextPage,
725  RegionSize,
726  (MM_READWRITE | MM_GUARD_PAGE),
727  ALT_CHANGE,
728  Process);
729 
730  UNLOCK_ADDRESS_SPACE (Process);
731  }
732 
733 #endif
734 
735  //
736  // The guard page is now committed or stack space is
737  // already present, return success.
738  //
739 
740  return STATUS_PAGE_FAULT_GUARD_PAGE;
741  }
742 
743  //
744  // The attempt to commit guard page(s) failed. If there are any
745  // guard pages between the base of the attempt and the faulting
746  // stack address, then strip their guard bits now so the application
747  // can handle the overflow exception without hitting successive
748  // exceptions. Note if an application desires successive exceptions,
749  // it must explicitly set guard page protection on the desired
750  // virtual addresses upon return.
751  //
752 
753  if (AdditionalSize != 0) {
754 
755  NextPage += PageSize;
756 
757  RegionSize -= PageSize;
758 
759  ZwProtectVirtualMemory (NtCurrentProcess (),
760  (PVOID *)&NextPage,
761  &RegionSize,
762  PAGE_READWRITE,
763  &OldProtection);
764  }
765 
766  return STATUS_STACK_OVERFLOW;
767 }
