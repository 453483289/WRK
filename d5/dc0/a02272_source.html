344 //
345 // Hibernation file layout:
346 // Page 0 - PO_MEMORY_IMAGE
347 // Page 1 - Free page array
348 // Page 2 - KPROCESSOR_CONTEXT
349 // Page 3 - first memory_range_array page
350 //
351 // PO_MEMORY_IMAGE:
352 // Header in file which contains some information to identify
353 // the hibernation, as well as a couple of checksums.
354 //
355 // Free page array:
356 // A page full of page numbers which identify 4MBs worth of
357 // system pages that are not in the restoration image. These
358 // pages are used by the loader (to keep itself out of the way)
359 // when restoring the memory image.
360 //
361 // KPROCESSOR_CONTEST
362 // The context of the processor which hibernated the system.
363 // Rest of page is empty.
364 //
365 // memory_range_array
366 // A page which contains an array of memory_range_array elements
367 // where element 0 is a Link entry, and all other elements are
368 // Range entries. The Link entry is used to link to the next
369 // such page, and to supply a count of the number of Range entries
370 // in the current page. The range entries each describe one
371 // physical memory range which needs restoration and its location
372 // in the file.
373 //
374

376  PFN_NUMBER PageNo;
377  PFN_NUMBER StartPage;
378  PFN_NUMBER EndPage;
379  ULONG CheckSum;
380 } PO_MEMORY_RANGE_ARRAY_RANGE;
