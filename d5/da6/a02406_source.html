29 ; )

31 ; Routine Description:
32 ;
33 ; This routine zeros the specified page of memory using normal moves.
34 ;
35 ; Arguments:
36 ;
37 ; PageBase (rcx) - Supplies the address of the pages to zero.
38 ;
39 ; Return Value:
40 ;
41 ; None.
42 ;

44 
45  LEAF_ENTRY KeZeroSinglePage, _TEXT$00
46 
47  xor eax, eax ; clear register
48  mov edx, PAGE_SIZE / 64 ; set number of 64 byte blocks
49 
50  align 16
51 
52 KiZP10: mov [rcx], rax ; zero 64-byte block
53  mov 8[rcx], rax ;
54  mov 16[rcx], rax ;
55  add rcx, 64 ; advance to next block
56  mov (24 - 64)[rcx], rax ;
57  mov (32 - 64)[rcx], rax ;
58  dec edx ; decrement loop count
59  mov (40 - 64)[rcx], rax ;
60  mov (48 - 64)[rcx], rax ;
61  mov (56 - 64)[rcx], rax ;
62  jnz short KiZP10 ; if nz, more bytes to zero
63  ret ; return
64

79 ;
80 ; Arguments:
81 ;
82 ; PageBase (rcx) - Supplies the address of the pages to zero.
83 ;
84 ; NumberOfBytes (rdx) - Supplies the number of bytes to zero. Always a PAGE_SIZE multiple.
85 ;
86 ; Return Value:
87 ;
88 ; None.

92  LEAF_ENTRY KeZeroPages, _TEXT$00
93 
94  xor eax, eax ; clear register
95  shr rdx, 7 ; number of 128 byte chunks (loop count)
96 
97  align 16
98 
99 KiZS10: movnti [rcx], rax ; zero 128-byte block
100  movnti [rcx + 8], rax ;
101  movnti [rcx + 16], rax ;
102  movnti [rcx + 24], rax ;
103  movnti [rcx + 32], rax ;
104  movnti [rcx + 40], rax ;
105  movnti [rcx + 48], rax ;
106  movnti [rcx + 56], rax ;
107  add rcx, 128 ; advance to next block
108  movnti [rcx - 64], rax ;
109  movnti [rcx - 56], rax ;
110  movnti [rcx - 48], rax ;
111  movnti [rcx - 40], rax ;
112  movnti [rcx - 32], rax ;
113  movnti [rcx - 24], rax ;
114  movnti [rcx - 16], rax ;
115  movnti [rcx - 8], rax ;
116  dec rdx ; decrement loop count
117  jnz short KiZS10 ; if nz, more bytes to zero
118  lock or byte ptr [rsp], 0 ; flush data to memory
119  ret ; return

121  LEAF_END KeZeroPages, _TEXT$00

151 ;
152

155
