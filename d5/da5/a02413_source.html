260 /*++
261 
262 Routine Description:
263 
264  This function freezes the execution of the current running processor.
265  If a trapframe is supplied to current state is saved into the prcb
266  for the debugger.
267 
268 Arguments:
269 
270  TrapFrame - Supplies a pointer to the trap frame that describes the
271  trap.
272 
273  ExceptionFrame - Supplies a pointer to the exception frame that

275

277 
278  None.
279 
280 --*/
281 
282 {
283 
284 #if !defined(NT_UP)
285

287  PKPRCB Prcb;
288  BOOLEAN Enable;
289  KCONTINUE_STATUS Status;
290  EXCEPTION_RECORD ExceptionRecord;
291 
292  //
293  // If the freeze lock is not held, then a debugger freeze was missed.
294  //
295 
296  if ((KiFreezeExecutionLock == 0) &&
297  (KiFreezeLockBackup == 0) &&
298  (KeBugCheckActive == FALSE)) {
299 #if DBG
300 
301  DbgBreakPoint();
302 
303 #endif
304 
305  return;
306  }
307 
308  Enable = KeDisableInterrupts();
309  KeRaiseIrql(HIGH_LEVEL, &OldIrql);
310  Prcb = KeGetCurrentPrcb();
311  Prcb->IpiFrozen = TARGET_FROZEN;
312  Prcb->SkipTick = TRUE;
313  if (TrapFrame != NULL) {
314  KiSaveProcessorState(TrapFrame, ExceptionFrame);
315  }
316 
317  //
318  // Sweep the data cache in case this is a system crash and the bugcheck
319  // code is attempting to write a crash dump file.
320  //

322  KeSweepCurrentDcache();
323

325  // Wait for person requesting us to freeze to
326  // clear our frozen flag
327  //
328

331 
332  //
333  // This processor has been made the active processor
334  //

354  // processor then reselect master
355  //
356

360  }
361  }

363  }
364

367  }
368

370  KeFlushCurrentTb();
371  KeSweepCurrentIcache();
372  KeLowerIrql(OldIrql);
373  KeEnableInterrupts(Enable);
374 
375 #else
376 
377  UNREFERENCED_PARAMETER(TrapFrame);
378  UNREFERENCED_PARAMETER(ExceptionFrame);
379 
380 #endif // !define(NT_UP)
