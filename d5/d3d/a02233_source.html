61 //
62 // Structure that defines the header of the Eventlog record. This is the
63 // fixed-sized portion before all the variable-length strings, binary
64 // data and pad bytes.
65 //
66 // TimeGenerated is the time it was generated at the client.
67 // TimeWritten is the time it was put into the log at the server end.
68 //
69 
70 typedef struct _EVENTLOGRECORD {
71  ULONG Length; // Length of full record
72  ULONG Reserved; // Used by the service
73  ULONG RecordNumber; // Absolute record number
74  ULONG TimeGenerated; // Seconds since 1-1-1970
75  ULONG TimeWritten; // Seconds since 1-1-1970
76  ULONG EventID;
77  USHORT EventType;
78  USHORT NumStrings;
79  USHORT EventCategory;
80  USHORT ReservedFlags; // For use with paired events (auditing)
81  ULONG ClosingRecordNumber; // For use with paired events (auditing)
82  ULONG StringOffset; // Offset from beginning of record
83  ULONG UserSidLength;
84  ULONG UserSidOffset;
85  ULONG DataLength;
86  ULONG DataOffset; // Offset from beginning of record
87  //
88  // Then follow:
89  //
90  // WCHAR SourceName[]
91  // WCHAR Computername[]
92  // SID UserSid
93  // WCHAR Strings[]
94  // BYTE Data[]
95  // CHAR Pad[]
96  // ULONG Length;
97  //
98 } EVENTLOGRECORD, *PEVENTLOGRECORD;
