159 /*++
160 
161 Routine Description:
162 
163  This function creates a mutant object, sets its initial count to one
164  (signaled), and opens a handle to the object with the specified desired
165  access.
166 
167 Arguments:
168 
169  MutantHandle - Supplies a pointer to a variable that will receive the
170  mutant object handle.
171 
172  DesiredAccess - Supplies the desired types of access for the mutant
173  object.
174 
175  ObjectAttributes - Supplies a pointer to an object attributes structure.
176 
177  InitialOwner - Supplies a boolean value that determines whether the
178  creator of the object desires immediate ownership of the object.
179 
180 --*/
181 
182 {
183 
184  HANDLE Handle;
185  PVOID Mutant;
186  KPROCESSOR_MODE PreviousMode;
187  NTSTATUS Status;
188 
189  //
190  // Establish an exception handler, probe the output handle address, and
191  // attempt to create a mutant object. If the probe fails, then return the
192  // exception code as the service status. Otherwise return the status value
193  // returned by the object insertion routine.
194  //
195 
196  try {
197 
198  //
199  // Get previous processor mode and probe output handle address if
200  // necessary.
201  //
202 
203  PreviousMode = KeGetPreviousMode();
204  if (PreviousMode != KernelMode) {
205  ProbeForWriteHandle(MutantHandle);
206  }
207 
208  //
209  // Allocate mutant object.
210  //
211 
212  Status = ObCreateObject(PreviousMode,
213  ExMutantObjectType,
214  ObjectAttributes,
215  PreviousMode,
216  NULL,
217  sizeof(KMUTANT),
218  0,
219  0,
220  &Mutant);
221 
222  //
223  // If the mutant object was successfully allocated, then initialize
224  // the mutant object and attempt to insert the mutant object in the
225  // current process' handle table.
226  //
227 
228  if (NT_SUCCESS(Status)) {
229  KeInitializeMutant((PKMUTANT)Mutant, InitialOwner);
230  Status = ObInsertObject(Mutant,
231  NULL,
232  DesiredAccess,
233  0,
234  NULL,
235  &Handle);
236 
237  //
238  // If the mutant object was successfully inserted in the current
239  // process' handle table, then attempt to write the mutant object
240  // handle value. If the write attempt fails, then do not report
241  // an error. When the caller attempts to access the handle value,
242  // an access violation will occur.
243  //
244 
245  if (NT_SUCCESS(Status)) {
246  try {
247  *MutantHandle = Handle;
248 
249  } except(ExSystemExceptionFilter()) {
250  }
251  }
252  }
253 
254  //
255  // If an exception occurs during the probe of the output handle address,
256  // then always handle the exception and return the exception code as the
257  // status value.
258  //
259 
260  } except(ExSystemExceptionFilter()) {
261  return GetExceptionCode();
262  }
263 
264  //
265  // Return service status.
266  //
267 
268  return Status;
269 }

368 NtQueryMutant (
369  __in HANDLE MutantHandle,
370  __in MUTANT_INFORMATION_CLASS MutantInformationClass,
371  __out_bcount(MutantInformationLength) PVOID MutantInformation,
372  __in ULONG MutantInformationLength,
373  __out_opt PULONG ReturnLength
374  )
375 
376 /*++
377 
378 Routine Description:
379 
380  This function queries the state of a mutant object and returns the
381  requested information in the specified record structure.
382 
383 Arguments:
384 
385  MutantHandle - Supplies a handle to a mutant object.
386 
387  MutantInformationClass - Supplies the class of information being
388  requested.
389 
390  MutantInformation - Supplies a pointer to a record that is to receive
391  the requested information.
392 
393  MutantInformationLength - Supplies the length of the record that is
394  to receive the requested information.
395 
396  ReturnLength - Supplies an optional pointer to a variable that will
397  receive the actual length of the information that is returned.
398 
399 --*/
400 
401 {
402 
403  BOOLEAN Abandoned;
404  BOOLEAN OwnedByCaller;
405  LONG Count;
406  PVOID Mutant;
407  KPROCESSOR_MODE PreviousMode;
408  NTSTATUS Status;
409 
410  //
411  // Establish an exception handler, probe the output arguments, reference
412  // the mutant object, and return the specified information. If the probe
413  // fails, then return the exception code as the service status. Otherwise
414  // return the status value returned by the reference object by handle
415  // routine.
416  //
417 
418  try {
419 
420  //
421  // Get previous processor mode and probe output arguments if necessary.
422  //
423 
424  PreviousMode = KeGetPreviousMode();
425  if (PreviousMode != KernelMode) {
426  ProbeForWriteSmallStructure(MutantInformation,
427  sizeof(MUTANT_BASIC_INFORMATION),
428  sizeof(ULONG));
429 
430  if (ARGUMENT_PRESENT(ReturnLength)) {
431  ProbeForWriteUlong(ReturnLength);
432  }
433  }
434 
435  //
436  // Check argument validity.
437  //
438 
439  if (MutantInformationClass != MutantBasicInformation) {
440  return STATUS_INVALID_INFO_CLASS;
441  }
442 
443  if (MutantInformationLength != sizeof(MUTANT_BASIC_INFORMATION)) {
444  return STATUS_INFO_LENGTH_MISMATCH;
445  }
446 
447  //
448  // Reference mutant object by handle.
449  //
450 
451  Status = ObReferenceObjectByHandle(MutantHandle,
452  MUTANT_QUERY_STATE,
453  ExMutantObjectType,
454  PreviousMode,
455  &Mutant,
456  NULL);
457 
458  //
459  // If the reference was successful, then read the current state and
460  // abandoned status of the mutant object, dereference mutant object,
461  // fill in the information structure, and return the length of the
462  // information structure if specified. If the write of the mutant
463  // information or the return length fails, then do not report an error.
464  // When the caller accesses the information structure or length an
465  // access violation will occur.
466  //
467 
468  if (NT_SUCCESS(Status)) {
469  Count = KeReadStateMutant((PKMUTANT)Mutant);
470  Abandoned = ((PKMUTANT)Mutant)->Abandoned;
471  OwnedByCaller = (BOOLEAN)((((PKMUTANT)Mutant)->OwnerThread ==
472  KeGetCurrentThread()));
473 
474  ObDereferenceObject(Mutant);
475  try {
476  ((PMUTANT_BASIC_INFORMATION)MutantInformation)->CurrentCount = Count;
477  ((PMUTANT_BASIC_INFORMATION)MutantInformation)->OwnedByCaller = OwnedByCaller;
478  ((PMUTANT_BASIC_INFORMATION)MutantInformation)->AbandonedState = Abandoned;
479  if (ARGUMENT_PRESENT(ReturnLength)) {
480  *ReturnLength = sizeof(MUTANT_BASIC_INFORMATION);
481  }
482 
483  } except(ExSystemExceptionFilter()) {
484  }
485  }
486 
487  //
488  // If an exception occurs during the probe of the output arguments, then
489  // always handle the exception and return the exception code as the status
490  // value.
491  //
492 
493  } except(ExSystemExceptionFilter()) {
494  return GetExceptionCode();
495  }
496 
497  //
498  // Return service status.
499  //
500 
501  return Status;
502 }

504 NTSTATUS
505 NtReleaseMutant (
506  __in HANDLE MutantHandle,
507  __out_opt PLONG PreviousCount
508  )
509 
510 /*++
511 
512 Routine Description:
513 
514  This function releases a mutant object.
515 
516 Arguments:
517 
518  Mutant - Supplies a handle to a mutant object.
519 
520  PreviousCount - Supplies an optional pointer to a variable that will
521  receive the previous mutant count.
522 
523 --*/
524 
525 {
526 
527  LONG Count;
528  PVOID Mutant;
529  KPROCESSOR_MODE PreviousMode;
530  NTSTATUS Status;
531 
532  //
533  // Establish an exception handler, probe the previous count address if
534  // specified, reference the mutant object, and release the mutant object.
535  // If the probe fails, then return the exception code as the service
536  // status. Otherwise return the status value returned by the reference
537  // object by handle routine.
538  //
539 
540  try {
541 
542  //
543  // Get previous processor mode and probe previous count address
544  // if necessary.
545  //
546 
547  PreviousMode = KeGetPreviousMode();
548  if ((PreviousMode != KernelMode) && (ARGUMENT_PRESENT(PreviousCount))) {
549  ProbeForWriteLong(PreviousCount);
550  }
551 
552  //
553  // Reference mutant object by handle.
554  //
555  // Note that the desired access is specified as zero since only the
556  // owner can release a mutant object.
557  //
558 
559  Status = ObReferenceObjectByHandle(MutantHandle,
560  0,
561  ExMutantObjectType,
562  PreviousMode,
563  &Mutant,
564  NULL);
565 
566  //
567  // If the reference was successful, then release the mutant object. If
568  // an exception occurs because the caller is not the owner of the mutant
569  // object, then dereference mutant object and return the exception code
570  // as the service status. Otherise write the previous count value if
571  // specified. If the write of the previous count fails, then do not
572  // report an error. When the caller attempts to access the previous
573  // count value, an access violation will occur.
574  //
575 
576  if (NT_SUCCESS(Status)) {
577  try {
578  PERFINFO_DECLARE_OBJECT(Mutant);
579  Count = KeReleaseMutant((PKMUTANT)Mutant, MUTANT_INCREMENT, FALSE, FALSE);
580  ObDereferenceObject(Mutant);
581  if (ARGUMENT_PRESENT(PreviousCount)) {
582  try {
583  *PreviousCount = Count;
584 
585  } except(ExSystemExceptionFilter()) {
586  }
587  }
588 
589  //
590  // If an exception occurs because the caller is not the owner of
591  // the mutant object, then always handle the exception, dereference
592  // the mutant object, and return the exception code as the status
593  // value.
594  //
595 
596  } except(ExSystemExceptionFilter()) {
597  ObDereferenceObject(Mutant);
598  return GetExceptionCode();
599  }
600  }
601 
602  //
603  // If an exception occurs during the probe of the previous count, then
604  // always handle the exception and return the exception code as the status
605  // value.
606  //
607 
608  } except(ExSystemExceptionFilter()) {
609  return GetExceptionCode();
610  }
611 
612  //
613  // Return service status.
614  //
615 
616  return Status;
617 }
