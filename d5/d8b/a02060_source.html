91 CmpUnmapCmView(
92  IN PCMHIVE CmHive,
93  IN PCM_VIEW_OF_FILE CmView,
94  IN BOOLEAN MapIsValid,
95  IN BOOLEAN MoveToEnd
96  )
97 /*++
98 
99 Routine Description:
100 
101  Unmaps the view by marking all the bins that maps inside of it as invalid.
102 
103 Arguments:
104 
105  Hive - Hive containing the section
106 
107  CmView - pointer to the view to operate on
108 
109  MapIsValid - Hive's map has been successfully initialized (and not yet freed)
110 
111  MoveToEnd - moves the view to the end of the LRUList after unmapping
112  This is normally TRUE, unless we want to be able to iterate through
113  the entire list and unmap views in the same time
114 
115 
116 Return Value:
117 
118  <none>
119 
120 --*/
121 {
122 
123  ULONG Offset;
124  ULONG_PTR Address;
125  ULONG_PTR AddressEnd;
126  PHMAP_ENTRY Me;
127 
128  CM_PAGED_CODE();
129 
130  ASSERT( (CmView->FileOffset + CmView->Size) != 0 && (CmView->ViewAddress != 0));
131  //
132  // it is forbidden to unmap a view still in use!
133  //
134  ASSERT( CmView->UseCount == 0 );
135 
136  ASSERT_VIEW_LOCK_OWNED(CmHive);
137  //
138  // only if the map is still valid
139  //
140  if( MapIsValid == TRUE ) {
141  Offset = CmView->FileOffset;
142 
143  AddressEnd = Address = (ULONG_PTR)(CmView->ViewAddress);
144  AddressEnd += CmView->Size;
145 
146  if( Offset == 0 ) {
147  //
148  // we are at the beginning, we have to skip the base block
149  //
150  Address += HBLOCK_SIZE;
151  } else {
152  //
153  // we are in the middle of the file. just adjust the offset
154  //
155  Offset -= HBLOCK_SIZE;
156  }
157 
158  while((Address < AddressEnd) && (Offset < CmHive->Hive.Storage[Stable].Length))
159  {
160  Me = HvpGetCellMap(&(CmHive->Hive), Offset);
161  VALIDATE_CELL_MAP(__LINE__,Me,&(CmHive->Hive),Offset);
162 
163  if( Me->BinAddress & HMAP_INPAGEDPOOL ) {
164  //
165  // if bin is mapped in paged pool for some ubiquitous reason,
166  // leave it like that (don't alter it's mapping).
167  //
168  } else {
169  //
170  // Invalidate the bin
171  //
172  Me->BinAddress &= (~HMAP_INVIEW);
173 
174  // we don't need to set it - just for debug purposes
175  ASSERT( (Me->CmView = NULL) == NULL );
176  }
177 
178  Offset += HBLOCK_SIZE;
179  Address += HBLOCK_SIZE;
180  }
181  }
182 
183  //
184  // Invalidate the view
185  //
186 
187  CcUnpinData( CmView->Bcb );
188 
189  CmView->FileOffset = 0;
190  CmView->Size = 0;
191  CmView->ViewAddress = 0;
192  CmView->Bcb = NULL;
193  CmView->UseCount = 0;
194 
195  if( MoveToEnd == TRUE ) {
196  //
197  // remove the view from the LRU list
198  //
199  RemoveEntryList(&(CmView->LRUViewList));
200 
201  //
202  // add it to the end of LRU list
203  //
204  InsertTailList(
205  &(CmHive->LRUViewListHead),
206  &(CmView->LRUViewList)
207  );
208  }
209 }

219 Warning:

223 Routine Description:

293 NTSTATUS
294 CmpMapCmView(
295  IN PCMHIVE CmHive,
296  IN ULONG FileOffset,
297  OUT PCM_VIEW_OF_FILE *CmView,
298  IN BOOLEAN MapInited
299  )
300 /*++
301 
302 Warning:
303 
304  This function should be called with the hivelock held!!!
305 
306 Routine Description:
307 
308  Unmaps the view by marking all the bins that maps inside of it as invalid.
309 
310 Arguments:
311 
312  CmHive - Hive containing the section
313 
314  FileOffset - Offset where to map the view
315 
316  CmView - pointer to the view to operate on
317 
318  MapInited - when TRUE, we can rely on the map info.
319 
320 Return Value:
321 
322  status of the operation
323 
324 --*/
325 {
326 
327  PHMAP_ENTRY Me;
328  NTSTATUS Status = STATUS_SUCCESS;
329  LARGE_INTEGER SectionOffset;
330  ULONG Offset;
331  ULONG_PTR Address;
332  ULONG_PTR AddressEnd;
333  ULONG_PTR BinAddress;
334  PHBIN Bin;
335  LONG PrevMappedBinSize;
336  BOOLEAN FirstTry = TRUE;
337 
338  CM_PAGED_CODE();
339 
340  ASSERT_VIEW_LOCK_OWNED(CmHive);
341 
342  if( CmHive->MappedViews == 0 ){
343  //
344  // we've run out of views; all are pinned
345  //
346  ASSERT( IsListEmpty(&(CmHive->LRUViewListHead)) == TRUE );
347  *CmView = CmpAllocateCmView(CmHive);
348 
349  } else {
350  //
351  // Remove the last view from LRU list (i.e. the LEAST recently used)
352  //
353  *CmView = (PCM_VIEW_OF_FILE)CmHive->LRUViewListHead.Blink;
354  *CmView = CONTAINING_RECORD( *CmView,
355  CM_VIEW_OF_FILE,
356  LRUViewList);
357 
358 
359  if( (*CmView)->ViewAddress != 0 ) {
360  PCM_VIEW_OF_FILE TempCmView = NULL;
361  //
362  // the last view is mapped
363  //
364  if( CmHive->MappedViews < CmMaxViewsPerHive ) {
365  //
366  // we are still allowed to add views
367  //
368  TempCmView = CmpAllocateCmView(CmHive);
369  }
370  if( TempCmView == NULL ) {
371  //
372  // we couldn't allocate a new view, or we need to use an existent one
373  //
374  if( (*CmView)->UseCount != 0 ) {
375  BOOLEAN FoundView = FALSE;
376  //
377  // view is in use; try walking to the top and find an unused view
378  //
379  while( (*CmView)->LRUViewList.Blink != &CmHive->LRUViewListHead ) {
380  *CmView = CONTAINING_RECORD( (*CmView)->LRUViewList.Blink,
381  CM_VIEW_OF_FILE,
382  LRUViewList);
383  if( (*CmView)->UseCount == 0 ) {
384  //
385  // this one is free go ahead and use it !
386  // first unmap, then signal that we found it
387  //
388  if( (*CmView)->ViewAddress != 0 ) {
389  //
390  // unnmap only if mapped
391  //
392  CmpUnmapCmView(CmHive,(*CmView),TRUE,TRUE);
393  }
394  FoundView = TRUE;
395  break;
396 
397  }
398  }
399 
400  if( FoundView == FALSE ) {
401  //
402  // no luck, all views are in use allocate a new one (we are forced to do so)
403  //
404  *CmView = CmpAllocateCmView(CmHive);
405  }
406  } else {
407  //
408  // unmap it!
409  //
410  CmpUnmapCmView(CmHive,(*CmView),TRUE,TRUE);
411  }
412  } else {
413  //
414  // we successfully allocated a new view
415  //
416  (*CmView) = TempCmView;
417  }
418  }
419  }
420 
421  if( (*CmView) == NULL ) {
422  //
423  // we're running low on resources
424  //
425  return STATUS_INSUFFICIENT_RESOURCES;
426  }
427 
428 #if DBG
429  {
430  UNICODE_STRING HiveName;
431  RtlInitUnicodeString(&HiveName, (PCWSTR)CmHive->Hive.BaseBlock->FileName);
432  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"CmpMapCmView for hive (%p) (%.*S),",CmHive,HiveName.Length / sizeof(WCHAR),HiveName.Buffer));
433  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP," FileOfset = %lx ... ",FileOffset));
434  }
435 #endif
436  //
437  // On this call, FileOffset must be a multiple of CM_VIEW_SIZE
438  //
439 
440 
441  //
442  // adjust the file offset to respect the CM_VIEW_SIZE alignment
443  //
444  Offset = ((FileOffset+HBLOCK_SIZE) & ~(CM_VIEW_SIZE - 1) );
445  SectionOffset.LowPart = Offset;
446  SectionOffset.HighPart = 0;
447 
448  (*CmView)->Size = CM_VIEW_SIZE;//(FileOffset + Size) - Offset;
449 
450  if( (Offset + (*CmView)->Size) > (CmHive->Hive.Storage[Stable].Length + HBLOCK_SIZE ) ){
451  (*CmView)->Size = CmHive->Hive.Storage[Stable].Length + HBLOCK_SIZE - Offset;
452  }
453 
454 RetryToMap:
455 
456  try {
457 
458  ASSERT( (*CmView)->Size != 0 );
459  ASSERT( (*CmView)->ViewAddress == NULL );
460  ASSERT( (*CmView)->UseCount == 0 );
461 
462  if (!CcMapData( CmHive->FileObject,
463  (PLARGE_INTEGER)&SectionOffset,
464  (*CmView)->Size,
465  MAP_WAIT | MAP_NO_READ,
466  (PVOID *)(&((*CmView)->Bcb)),
467  (PVOID *)(&((*CmView)->ViewAddress)) )) {
468  Status = STATUS_CANT_WAIT;
469  }
470  } except (EXCEPTION_EXECUTE_HANDLER) {
471  //
472  // in low-memory scenarios, CcMapData throws a STATUS_IN_PAGE_ERROR
473  // this happens when the IO issued to touch the just-mapped data fails (usually with
474  // STATUS_INSUFFICIENT_RESOURCES; We want to catch this and treat as a
475  // "not enough resources" problem, rather than letting it to surface the kernel call
476  //
477  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"CmpMapCmView : CcMapData has raised :%08lx\n",GetExceptionCode()));
478  Status = STATUS_INSUFFICIENT_RESOURCES;
479  }
480 
481  if(!NT_SUCCESS(Status) ){
482  if( FirstTry == TRUE ) {
483  //
484  // unmap all unnecessary views and try again
485  //
486  FirstTry = FALSE;
487  CmpUnmapUnusedViews(CmHive);
488  Status = STATUS_SUCCESS;
489  goto RetryToMap;
490  }
491  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"Fail\n"));
492  ASSERT(FALSE);
493  return Status;
494  }
495 
496  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"Succedeed Address = %p Size = %lx\n",(*CmView)->ViewAddress,(*CmView)->Size));
497 
498  (*CmView)->FileOffset = SectionOffset.LowPart;
499 
500  ASSERT( Offset == (*CmView)->FileOffset);
501 
502  AddressEnd = Address = (ULONG_PTR)((*CmView)->ViewAddress);
503  AddressEnd += (*CmView)->Size;
504 
505  if( Offset == 0 ) {
506  //
507  // we are at the beginning, we have to skip the base block
508  //
509  Address += HBLOCK_SIZE;
510  } else {
511  //
512  // we are in the middle of the file. just adjust the offset
513  //
514  Offset -= HBLOCK_SIZE;
515  }
516 
517  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_BIN_MAP,"CmpMapCmView :: Address = %p AddressEnd = %p ; Size = %lx\n",Address,AddressEnd,(*CmView)->Size));
518 
519  //
520  // here we can optimize not to touch all the bins!!!
521  //
522 
523  //
524  // we don't know yet if the first bin is mapped.
525  //
526  PrevMappedBinSize = 0;
527  BinAddress = Address;
528  while(Address < AddressEnd)
529  {
530  Me = HvpGetCellMap(&(CmHive->Hive), Offset);
531  VALIDATE_CELL_MAP(__LINE__,Me,&(CmHive->Hive),Offset);
532 
533  if( Me->BinAddress & HMAP_INPAGEDPOOL ) {
534  //
535  // if bin is mapped in paged pool for some reason,
536  // leave it like that (don't alter it's mapping).
537  //
538 
539  //
540  // next mapped bin should start updating his bin address
541  //
542  PrevMappedBinSize = 0;
543  } else {
544  //
545  // at this point the bin should be invalid.
546  //
547  ASSERT_BIN_INVALID(Me);
548 
549  Me->BinAddress |= HMAP_INVIEW;
550  Me->CmView = *CmView;
551 
552  //
553  // set the new BinAddress, but take care to preserve the flags
554  //
555  ASSERT( HBIN_FLAGS(Address) == 0 );
556 
557 
558 
559  //
560  // new bins are Always tagged with this flag (we can start updating BinAddress)
561  //
562  if( MapInited && ( Me->BinAddress & HMAP_NEWALLOC ) ) {
563 
564  //
565  // we are at the beginning of a new bin
566  //
567  BinAddress = Address;
568  } else if( (!MapInited) &&(PrevMappedBinSize == 0) ) {
569  //
570  // we cannot rely on the map to carry the bin flags; we have to fault data in
571  //
572  //
573  // Validate the bin
574  //
575  Bin = (PHBIN)Address;
576  PrevMappedBinSize = (LONG)Bin->Size;
577  //
578  // we are at the beginning of a new bin
579  //
580  BinAddress = Address;
581  }
582 
583  //
584  // common sense
585  //
586  ASSERT( (!MapInited) || ((PrevMappedBinSize >=0) && (PrevMappedBinSize%HBLOCK_SIZE == 0)) );
587 
588  Me->BinAddress = ( HBIN_BASE(BinAddress) | HBIN_FLAGS(Me->BinAddress) );
589  if( (Me->BinAddress & HMAP_DISCARDABLE) == 0 ) {
590  //
591  // for discardable bins do not alter this member, as it contains
592  // the address of the free bin
593  //
594  Me->BlockAddress = Address;
595  }
596 
597  if( !MapInited ) {
598  //
599  // compute the remaining size of this bin; next iteration will update BinAddress only if
600  // this variable reaches 0
601  //
602  PrevMappedBinSize -= HBLOCK_SIZE;
603  }
604 
605  ASSERT_BIN_INVIEW(Me);
606  }
607 
608  Offset += HBLOCK_SIZE;
609  Address += HBLOCK_SIZE;
610  }
611 
612  return Status;
613 }

617  IN PCMHIVE CmHive,
618  IN ULONG FileOffset
619  )
620 /*++
621 
622 Routine Description:
623 
624  Parses the mapped view list and if it finds one surrounding this offset, unmaps it.
625 
626 Arguments:
627 
628  CmHive - Hive in question
629 
630  FileOffset - the offset in question
631 
632 Return Value:
633 
634  none
635 
636 Note:
637 
638  Offset is an absolute value,
639 --*/
640 {
641  PCM_VIEW_OF_FILE CmView;
642  USHORT NrViews;
643  BOOLEAN UnMap = FALSE;
644 
645  CM_PAGED_CODE();
646 
647  ASSERT_VIEW_LOCK_OWNED(CmHive);
648  //
649  // Walk through the LRU list and compare view addresses
650  //
651  CmView = (PCM_VIEW_OF_FILE)CmHive->LRUViewListHead.Flink;
652 
653  for(NrViews = CmHive->MappedViews;NrViews;NrViews--) {
654  CmView = CONTAINING_RECORD( CmView,
655  CM_VIEW_OF_FILE,
656  LRUViewList);
657 
658  if( ((CmView->Size + CmView->FileOffset) != 0) && (CmView->ViewAddress != 0) ) {
659  //
660  // view is valid
661  //
662  if( (CmView->FileOffset <= FileOffset) && ((CmView->FileOffset + CmView->Size) > FileOffset) ) {
663  //
664  // the file offset is surrounded by this view
665  //
666  UnMap = TRUE;
667  break;
668  }
669  }
670 
671  CmView = (PCM_VIEW_OF_FILE)CmView->LRUViewList.Flink;
672  }
673 
674  if( UnMap == TRUE ) {
675  // unmap the view anyway (this implies unpinning).
676  ASSERT_VIEW_MAPPED( CmView );
677  ASSERT( CmView->UseCount == 0 );
678  CmpUnmapCmView(CmHive,CmView,TRUE,TRUE);
679  }
680 }
