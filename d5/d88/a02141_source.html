89 /*++
90 
91 Routine Description:
92 
93  This routine does a fast cached read bypassing the usual file system
94  entry routine (i.e., without the Irp). It is used to do a copy read
95  of a cached file object. For a complete description of the arguments
96  see CcCopyRead.
97 
98 Arguments:
99 
100  FileObject - Pointer to the file object being read.
101 
102  FileOffset - Byte offset in file for desired data.
103 
104  Length - Length of desired data in bytes.
105 
106  Wait - FALSE if caller may not block, TRUE otherwise
107 
108  Buffer - Pointer to output buffer to which data should be copied.
109 
110  IoStatus - Pointer to standard I/O status block to receive the status
111  for the transfer.
112 
113 Return Value:
114 
115  FALSE - if Wait was supplied as FALSE and the data was not delivered, or
116  if there is an I/O error.
117 
118  TRUE - if the data is being delivered
119 
120 --*/

123  PFSRTL_COMMON_FCB_HEADER Header;
124  BOOLEAN Status = TRUE;
125  ULONG PageCount = ADDRESS_AND_SIZE_TO_SPAN_PAGES( FileOffset->QuadPart, Length );
126  LARGE_INTEGER BeyondLastByte;
127  PDEVICE_OBJECT targetVdo;
128 
129  PAGED_CODE();
130 
131  UNREFERENCED_PARAMETER (DeviceObject);
132 
133  //
134  // Special case a read of zero length
135  //
136 
137  if (Length != 0) {
138 
139  //
140  // Check for overflow. Returning false here will re-route this request through the
141  // IRP based path, but this isn't performance critical.
142  //
143 
144  if (MAXLONGLONG - FileOffset->QuadPart < (LONGLONG)Length) {
145 
146  IoStatus->Status = STATUS_INVALID_PARAMETER;
147  IoStatus->Information = 0;
148 
149  return FALSE;
150  }
151 
152  BeyondLastByte.QuadPart = FileOffset->QuadPart + (LONGLONG)Length;
153  Header = (PFSRTL_COMMON_FCB_HEADER)FileObject->FsContext;
154 
155  //
156  // Enter the file system
157  //
158 
159  FsRtlEnterFileSystem();
160 
161  //
162  // Increment performance counters and get the resource
163  //
164 
165  if (Wait) {
166 
167  HOT_STATISTIC(CcFastReadWait) += 1;
168 
169  //
170  // Acquired shared on the common fcb header
171  //
172 
173  (VOID)ExAcquireResourceSharedLite( Header->Resource, TRUE );
174 
175  } else {
176 
177  HOT_STATISTIC(CcFastReadNoWait) += 1;
178 
179  //
180  // Acquired shared on the common fcb header, and return if we
181  // don't get it
182  //
183 
184  if (!ExAcquireResourceSharedLite( Header->Resource, FALSE )) {
185 
186  FsRtlExitFileSystem();
187 
188  CcFastReadResourceMiss += 1;
189 
190  return FALSE;
191  }
192  }
193

195  // Now that the File is acquired shared, we can safely test if it
196  // is really cached and if we can do fast i/o and if not, then
197  // release the fcb and return.
198  //
199 
200  if ((FileObject->PrivateCacheMap == NULL) ||
201  (Header->IsFastIoPossible == FastIoIsNotPossible)) {
202 
203  ExReleaseResourceLite( Header->Resource );
204  FsRtlExitFileSystem();
205 
206  HOT_STATISTIC(CcFastReadNotPossible) += 1;
207 
208  return FALSE;
209  }
210 
211  //
212  // Check if fast I/O is questionable and if so then go ask the
213  // file system the answer
214  //
215 
216  if (Header->IsFastIoPossible == FastIoIsQuestionable) {
217 
218  PFAST_IO_DISPATCH FastIoDispatch;
219 
220  ASSERT(!KeIsExecutingDpc());
221 
222  targetVdo = IoGetRelatedDeviceObject( FileObject );
223  FastIoDispatch = targetVdo->DriverObject->FastIoDispatch;
224 
225 
226  //
227  // All file systems that set "Is Questionable" had better support
228  // fast I/O
229  //
230 
231  ASSERT(FastIoDispatch != NULL);
232  ASSERT(FastIoDispatch->FastIoCheckIfPossible != NULL);
233 
234  //
235  // Call the file system to check for fast I/O. If the answer is
236  // anything other than GoForIt then we cannot take the fast I/O
237  // path.
238  //
239 
240  if (!FastIoDispatch->FastIoCheckIfPossible( FileObject,
241  FileOffset,
242  Length,
243  Wait,
244  LockKey,
245  TRUE, // read operation
246  IoStatus,
247  targetVdo )) {
248 
249  //
250  // Fast I/O is not possible so release the Fcb and return.
251  //
252 
253  ExReleaseResourceLite( Header->Resource );
254  FsRtlExitFileSystem();
255 
256  HOT_STATISTIC(CcFastReadNotPossible) += 1;
257 
258  return FALSE;
259  }
260  }
261 
262  //
263  // Check for read past file size.
264  //
265 
266  if ( BeyondLastByte.QuadPart > Header->FileSize.QuadPart ) {
267 
268  if ( FileOffset->QuadPart >= Header->FileSize.QuadPart ) {
269  IoStatus->Status = STATUS_END_OF_FILE;
270  IoStatus->Information = 0;
271 
272  ExReleaseResourceLite( Header->Resource );
273  FsRtlExitFileSystem();
274 
275  return TRUE;
276  }
277 
278  Length = (ULONG)( Header->FileSize.QuadPart - FileOffset->QuadPart );
279  }
280 
281  //
282  // We can do fast i/o so call the cc routine to do the work and then
283  // release the fcb when we've done. If for whatever reason the
284  // copy read fails, then return FALSE to our caller.
285  //
286  // Also mark this as the top level "Irp" so that lower file system
287  // levels will not attempt a pop-up
288  //
289

291 
292  try {
293 
294  if (Wait && ((BeyondLastByte.HighPart | Header->FileSize.HighPart) == 0)) {
295 
296  CcFastCopyRead( FileObject,
297  FileOffset->LowPart,
298  Length,
299  PageCount,
300  Buffer,
301  IoStatus );
302 
303  FileObject->Flags |= FO_FILE_FAST_IO_READ;
304 
305  ASSERT( (IoStatus->Status == STATUS_END_OF_FILE) ||
306  ((FileOffset->LowPart + IoStatus->Information) <= Header->FileSize.LowPart));
307 
308  } else {
309 
310  Status = CcCopyRead( FileObject,
311  FileOffset,
312  Length,
313  Wait,
314  Buffer,
315  IoStatus );
316 
317  FileObject->Flags |= FO_FILE_FAST_IO_READ;
318 
319  ASSERT( !Status || (IoStatus->Status == STATUS_END_OF_FILE) ||
320  ((LONGLONG)(FileOffset->QuadPart + IoStatus->Information) <= Header->FileSize.QuadPart));
321  }
322 
323  if (Status) {
324 
325  FileObject->CurrentByteOffset.QuadPart = FileOffset->QuadPart + IoStatus->Information;
326  }
327 
328  } except( FsRtlIsNtstatusExpected(GetExceptionCode())
329  ? EXCEPTION_EXECUTE_HANDLER
330  : EXCEPTION_CONTINUE_SEARCH ) {
331 
332  Status = FALSE;
333  }
334 
335  PsGetCurrentThread()->TopLevelIrp = 0;
336 
337  ExReleaseResourceLite( Header->Resource );
338  FsRtlExitFileSystem();
339  return Status;
340 
341  } else {
342 
343  //
344  // A zero length transfer was requested.
345  //
346 
347  IoStatus->Status = STATUS_SUCCESS;
348  IoStatus->Information = 0;
349 
350  return TRUE;
351  }
352 }

367 /*++
368 
369 Routine Description:
370 
371  This routine does a fast cached write bypassing the usual file system
372  entry routine (i.e., without the Irp). It is used to do a copy write
373  of a cached file object. For a complete description of the arguments
374  see CcCopyWrite.
375 
376 Arguments:
377 
378  FileObject - Pointer to the file object being write.
379 
380  FileOffset - Byte offset in file for desired data.
381 
382  Length - Length of desired data in bytes.
383 
384  Wait - FALSE if caller may not block, TRUE otherwise
385 
386  Buffer - Pointer to output buffer to which data should be copied.
387 
388  IoStatus - Pointer to standard I/O status block to receive the status
389  for the transfer.
390 
391 Return Value:
392 
393  FALSE - if Wait was supplied as FALSE and the data was not delivered, or
394  if there is an I/O error.
395 
396  TRUE - if the data is being delivered
397 
398 --*/
399 
400 {
401  PFSRTL_COMMON_FCB_HEADER Header;
402  BOOLEAN AcquiredShared = FALSE;
403  BOOLEAN Status = TRUE;
404  BOOLEAN FileSizeChanged = FALSE;
405  BOOLEAN WriteToEndOfFile = (BOOLEAN)((FileOffset->LowPart == FILE_WRITE_TO_END_OF_FILE) &&
406  (FileOffset->HighPart == -1));
407 
408  PAGED_CODE();
409  UNREFERENCED_PARAMETER( DeviceObject );
410 
411  //
412  // Get a real pointer to the common fcb header
413  //
414 
415  Header = (PFSRTL_COMMON_FCB_HEADER)FileObject->FsContext;
416 
417  //
418  // Do we need to verify the volume? If so, we must go to the file
419  // system. Also return FALSE if FileObject is write through, the
420  // File System must do that.
421  //
422 
423  if (CcCanIWrite( FileObject, Length, Wait, FALSE ) &&
424  !FlagOn(FileObject->Flags, FO_WRITE_THROUGH) &&
425  CcCopyWriteWontFlush(FileObject, FileOffset, Length)) {
426 
427  //
428  // Assume our transfer will work
429  //
430 
431  IoStatus->Status = STATUS_SUCCESS;
432  IoStatus->Information = Length;
433 
434  //
435  // Special case the zero byte length
436  //
437 
438  if (Length != 0) {
439 
440  //
441  // Enter the file system
442  //
443 
444  FsRtlEnterFileSystem();
445 
446  //
447  // Split into separate paths for increased performance. First
448  // we have the faster path which only supports Wait == TRUE and
449  // 32 bits. We will make an unsafe test on whether the fast path
450  // is ok, then just return FALSE later if we were wrong. This
451  // should virtually never happen.
452  //
453  // IMPORTANT NOTE: It is very important that any changes made to
454  // this path also be applied to the 64-bit path
455  // which is the else of this test!
456  //
457 
458  if (Wait && (Header->AllocationSize.HighPart == 0)) {
459 
460  ULONG Offset, NewFileSize;
461  ULONG OldFileSize = 0;
462  ULONG OldValidDataLength = 0;
463  LOGICAL Wrapped;
464 
465  //
466  // Make our best guess on whether we need the file exclusive
467  // or shared. Note that we do not check FileOffset->HighPart
468  // until below.
469  //
470 
471  NewFileSize = FileOffset->LowPart + Length;
472 
473  if (WriteToEndOfFile || (NewFileSize > Header->ValidDataLength.LowPart)) {
474 
475  //
476  // Acquired shared on the common fcb header
477  //
478 
479  ExAcquireResourceExclusiveLite( Header->Resource, TRUE );
480 
481  } else {
482 
483  //
484  // Acquired shared on the common fcb header
485  //
486 
487  ExAcquireResourceSharedLite( Header->Resource, TRUE );
488 
489  AcquiredShared = TRUE;
490  }
491 
492  //
493  // We have the fcb shared now check if we can do fast i/o
494  // and if the file space is allocated, and if not then
495  // release the fcb and return.
496  //
497 
498  if (WriteToEndOfFile) {
499 
500  Offset = Header->FileSize.LowPart;
501  NewFileSize = Header->FileSize.LowPart + Length;
502  Wrapped = NewFileSize < Header->FileSize.LowPart;
503 
504  } else {
505 
506  Offset = FileOffset->LowPart;
507  NewFileSize = FileOffset->LowPart + Length;
508  Wrapped = (NewFileSize < FileOffset->LowPart) || (FileOffset->HighPart != 0);
509  }
510 
511  //
512  // Now that the File is acquired shared, we can safely test
513  // if it is really cached and if we can do fast i/o and we
514  // do not have to extend. If not then release the fcb and
515  // return.
516  //
517  // Get out if we have too much to zero. This case is not important
518  // for performance, and a file system supporting sparseness may have
519  // a way to do this more efficiently.
520  //
521 
522  if ((FileObject->PrivateCacheMap == NULL) ||
523  (Header->IsFastIoPossible == FastIoIsNotPossible) ||
524  (NewFileSize > Header->AllocationSize.LowPart) ||
525  (Offset >= (Header->ValidDataLength.LowPart + 0x2000)) ||
526  (Header->AllocationSize.HighPart != 0) || Wrapped) {
527 
528  ExReleaseResourceLite( Header->Resource );
529  FsRtlExitFileSystem();
530 
531  return FALSE;
532  }
533 
534  //
535  // If we will be extending ValidDataLength, we will have to
536  // get the Fcb exclusive, and make sure that FastIo is still
537  // possible. We should only execute this block of code very
538  // rarely, when the unsafe test for ValidDataLength failed
539  // above.
540  //
541 
542  if (AcquiredShared && (NewFileSize > Header->ValidDataLength.LowPart)) {
543 
544  ExReleaseResourceLite( Header->Resource );
545 
546  ExAcquireResourceExclusiveLite( Header->Resource, TRUE );
547 
548  //
549  // If writing to end of file, we must recalculate new size.
550  //
551 
552  if (WriteToEndOfFile) {
553 
554  Offset = Header->FileSize.LowPart;
555  NewFileSize = Header->FileSize.LowPart + Length;
556  Wrapped = NewFileSize < Header->FileSize.LowPart;
557  }
558 
559  if ((FileObject->PrivateCacheMap == NULL) ||
560  (Header->IsFastIoPossible == FastIoIsNotPossible) ||
561  (NewFileSize > Header->AllocationSize.LowPart) ||
562  (Header->AllocationSize.HighPart != 0) || Wrapped) {
563 
564  ExReleaseResourceLite( Header->Resource );
565  FsRtlExitFileSystem();
566 
567  return FALSE;
568  }
569  }
570 
571  //
572  // Check if fast I/O is questionable and if so then go ask
573  // the file system the answer
574  //
575 
576  if (Header->IsFastIoPossible == FastIoIsQuestionable) {
577 
578  PDEVICE_OBJECT targetVdo = IoGetRelatedDeviceObject( FileObject );
579  PFAST_IO_DISPATCH FastIoDispatch = targetVdo->DriverObject->FastIoDispatch;
580  IO_STATUS_BLOCK IoStatus;
581 
582  //
583  // All file system then set "Is Questionable" had better
584  // support fast I/O
585  //
586 
587  ASSERT(FastIoDispatch != NULL);
588  ASSERT(FastIoDispatch->FastIoCheckIfPossible != NULL);
589 
590  //
591  // Call the file system to check for fast I/O. If the
592  // answer is anything other than GoForIt then we cannot
593  // take the fast I/O path.
594  //
595 
596  ASSERT(FILE_WRITE_TO_END_OF_FILE == 0xffffffff);
597 
598  if (!FastIoDispatch->FastIoCheckIfPossible( FileObject,
599  FileOffset->QuadPart != (LONGLONG)-1 ?
600  FileOffset : &Header->FileSize,
601  Length,
602  TRUE,
603  LockKey,
604  FALSE, // write operation
605  &IoStatus,
606  targetVdo )) {
607 
608  //
609  // Fast I/O is not possible so release the Fcb and
610  // return.
611  //
612 
613  ExReleaseResourceLite( Header->Resource );
614  FsRtlExitFileSystem();
615 
616  return FALSE;
617  }
618  }
619 
620  //
621  // Now see if we will change FileSize. We have to do it now
622  // so that our reads are not nooped.
623  //
624 
625  if (NewFileSize > Header->FileSize.LowPart) {
626 
627  FileSizeChanged = TRUE;
628  OldFileSize = Header->FileSize.LowPart;
629  OldValidDataLength = Header->ValidDataLength.LowPart;
630  Header->FileSize.LowPart = NewFileSize;
631  }
632 
633  //
634  // We can do fast i/o so call the cc routine to do the work
635  // and then release the fcb when we've done. If for whatever
636  // reason the copy write fails, then return FALSE to our
637  // caller.
638  //
639  // Also mark this as the top level "Irp" so that lower file
640  // system levels will not attempt a pop-up
641  //
642 
643  PsGetCurrentThread()->TopLevelIrp = FSRTL_FAST_IO_TOP_LEVEL_IRP;
644 
645  try {
646 
647  //
648  // See if we have to do some zeroing
649  //
650 
651  if (Offset > Header->ValidDataLength.LowPart) {
652 
653  LARGE_INTEGER ZeroEnd;
654 
655  ZeroEnd.LowPart = Offset;
656  ZeroEnd.HighPart = 0;
657 
658  CcZeroData( FileObject,
659  &Header->ValidDataLength,
660  &ZeroEnd,
661  TRUE );
662  }
663 
664  CcFastCopyWrite( FileObject,
665  Offset,
666  Length,
667  Buffer );
668 
669  } except( FsRtlIsNtstatusExpected(GetExceptionCode())
670  ? EXCEPTION_EXECUTE_HANDLER
671  : EXCEPTION_CONTINUE_SEARCH ) {
672 
673  Status = FALSE;
674  }
675 
676  PsGetCurrentThread()->TopLevelIrp = 0;
677 
678  //
679  // If we succeeded, see if we have to update FileSize or
680  // ValidDataLength.
681  //
682 
683  if (Status) {
684 
685  //
686  // In the case of ValidDataLength, we really have to
687  // check again since we did not do this when we acquired
688  // the resource exclusive.
689  //
690 
691  if (NewFileSize > Header->ValidDataLength.LowPart) {
692 
693  Header->ValidDataLength.LowPart = NewFileSize;
694  }
695 
696  //
697  // Set this handle as having modified the file
698  //
699 
700  FileObject->Flags |= FO_FILE_MODIFIED;
701 
702  if (FileSizeChanged) {
703 
704  CcGetFileSizePointer(FileObject)->LowPart = NewFileSize;
705 
706  FileObject->Flags |= FO_FILE_SIZE_CHANGED;
707  }
708 
709  //
710  // Also update the file position pointer
711  //
712 
713  FileObject->CurrentByteOffset.LowPart = Offset + Length;
714  FileObject->CurrentByteOffset.HighPart = 0;
715 
716  //
717  // If we did not succeed, then we must restore the original
718  // FileSize while holding the PagingIoResource exclusive if
719  // it exists.
720  //
721 
722  } else if (FileSizeChanged) {
723 
724  if ( Header->PagingIoResource != NULL ) {
725 
726  (VOID)ExAcquireResourceExclusiveLite( Header->PagingIoResource, TRUE );
727  Header->FileSize.LowPart = OldFileSize;
728  Header->ValidDataLength.LowPart = OldValidDataLength;
729  ExReleaseResourceLite( Header->PagingIoResource );
730 
731  } else {
732 
733  Header->FileSize.LowPart = OldFileSize;
734  Header->ValidDataLength.LowPart = OldValidDataLength;
735  }
736  }
737 
738  //
739  // Here is the 64-bit or no-wait path.
740  //
741 
742  } else {
743 
744  LARGE_INTEGER Offset, NewFileSize;
745  LARGE_INTEGER OldFileSize = {0};
746  LARGE_INTEGER OldValidDataLength = {0};
747 
748  ASSERT(!KeIsExecutingDpc());
749 
750  //
751  // Make our best guess on whether we need the file exclusive
752  // or shared.
753  //
754 
755  NewFileSize.QuadPart = FileOffset->QuadPart + (LONGLONG)Length;
756 
757  if (WriteToEndOfFile || (NewFileSize.QuadPart > Header->ValidDataLength.QuadPart)) {
758 
759  //
760  // Acquired shared on the common fcb header, and return
761  // if we don't get it.
762  //
763 
764  if (!ExAcquireResourceExclusiveLite( Header->Resource, Wait )) {
765 
766  FsRtlExitFileSystem();
767 
768  return FALSE;
769  }
770 
771  } else {
772 
773  //
774  // Acquired shared on the common fcb header, and return
775  // if we don't get it.
776  //
777 
778  if (!ExAcquireResourceSharedLite( Header->Resource, Wait )) {
779 
780  FsRtlExitFileSystem();
781 
782  return FALSE;
783  }
784 
785  AcquiredShared = TRUE;
786  }
787 
788 
789  //
790  // We have the fcb shared now check if we can do fast i/o
791  // and if the file space is allocated, and if not then
792  // release the fcb and return.
793  //
794 
795  if (WriteToEndOfFile) {
796 
797  Offset = Header->FileSize;
798  NewFileSize.QuadPart = Header->FileSize.QuadPart + (LONGLONG)Length;
799 
800  } else {
801 
802  Offset = *FileOffset;
803  NewFileSize.QuadPart = FileOffset->QuadPart + (LONGLONG)Length;
804  }
805

807  // Now that the File is acquired shared, we can safely test
808  // if it is really cached and if we can do fast i/o and we
809  // do not have to extend. If not then release the fcb and
810  // return.
