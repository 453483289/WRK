119 CmpDumpOneKeyBody(
120  IN PCM_KEY_CONTROL_BLOCK kcb,
121  IN PCM_KEY_BODY KeyBody,
122  IN PUNICODE_STRING Name,
123  IN PVOID Context
124  )
125 {
126  //
127  // sanity check: this should be a KEY_BODY
128  //
129  ASSERT_KEY_OBJECT(KeyBody);
130 
131  if( !Context ) {
132  //
133  // NtQueryOpenSubKeys : dump it's name and owning process
134  //
135  {
136  PEPROCESS Process;
137  PUCHAR ImageName = NULL;
138 
139 
140  if( NT_SUCCESS(PsLookupProcessByProcessId(KeyBody->ProcessID,&Process))) {
141  ImageName = PsGetProcessImageFileName(Process);
142  } else {
143  Process = NULL;
144  }
145 
146  if( !ImageName ) {
147  ImageName = (PUCHAR)"Unknown";
148  }
149  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"Process %p (PID = %lx ImageFileName = %s) (KCB = %p) :: Key %wZ \n",
150  Process,KeyBody->ProcessID,ImageName,kcb,Name);
151  if( Process ) {
152  ObDereferenceObject (Process);
153  }
154  }
155  } else {
156  //
157  // NtQueryOpenSubKeysEx: build up the return buffer; make sure we only touch it
158  // inside a try except as it's user-mode buffer
159  //
160  PQUERY_OPEN_SUBKEYS_CONTEXT QueryContext = (PQUERY_OPEN_SUBKEYS_CONTEXT)Context;
161  PKEY_OPEN_SUBKEYS_INFORMATION SubkeysInfo = (PKEY_OPEN_SUBKEYS_INFORMATION)(QueryContext->Buffer);
162  ULONG SizeNeeded;
163 
164  //
165  // we need to ignore the one created by us inside NtQueryOpenSubKeysEx
166  //
167  if( QueryContext->KeyBodyToIgnore != KeyBody ) {
168  //
169  // update RequiredSize; we do this regardless if we have room or not in the buffer
170  // reserve for one entry in the array and the unicode name buffer
171  //
172  SizeNeeded = (sizeof(KEY_PID_ARRAY) + (ULONG)(Name->Length));
173  QueryContext->RequiredSize += SizeNeeded;
174 
175  //
176  // if we have encountered an error (overflow, or else) at some previous iteration, no point going on
177  //
178  if( NT_SUCCESS(QueryContext->StatusCode) ) {
179  //
180  // see if we have enough room for current entry.
181  //
182  if( (QueryContext->UsedLength + SizeNeeded) > QueryContext->BufferLength ) {
183  //
184  // buffer not big enough;
185  //
186  QueryContext->StatusCode = STATUS_BUFFER_OVERFLOW;
187  } else {
188  //
189  // we have established we have enough room; create/add a new entry to the key array
190  // and build up unicode name buffer. copy key name to it.
191  // array elements are at the beginning of the user buffer, while name buffers start at
192  // the end and continue backwards, as long as there is enough room.
193  //
194  try {
195  //
196  // protect user mode memory
197  //
198  SubkeysInfo->KeyArray[SubkeysInfo->Count].PID = KeyBody->ProcessID;
199  SubkeysInfo->KeyArray[SubkeysInfo->Count].KeyName.Length = Name->Length;
200  SubkeysInfo->KeyArray[SubkeysInfo->Count].KeyName.MaximumLength = Name->Length;
201  SubkeysInfo->KeyArray[SubkeysInfo->Count].KeyName.Buffer = (PWSTR)((PUCHAR)QueryContext->CurrentNameBuffer - Name->Length);
202  RtlCopyMemory( SubkeysInfo->KeyArray[SubkeysInfo->Count].KeyName.Buffer,
203  Name->Buffer,
204  Name->Length);
205  //
206  // update array count and work vars inside the querycontext
207  //
208  SubkeysInfo->Count++;
209  QueryContext->CurrentNameBuffer = (PUCHAR)QueryContext->CurrentNameBuffer - Name->Length;
210  QueryContext->UsedLength += SizeNeeded;
211  } except (EXCEPTION_EXECUTE_HANDLER) {
212  QueryContext->StatusCode = GetExceptionCode();
213  }
214  }
215  }
216  }

223 CmpDumpKeyBodyList(
224  IN PCM_KEY_CONTROL_BLOCK kcb,
225  IN PULONG Count,
226  IN PVOID Context
227  )
228 {
229 
230  PCM_KEY_BODY KeyBody;
231  PUNICODE_STRING Name;
232  ULONG i;
233  BOOLEAN KeyBodyArrayEmpty = TRUE;
234 
235  ASSERT_KCB_LOCKED_EXCLUSIVE(kcb);
236 
237  if( kcb->RefCount == 0 ) {
238  //
239  // this kcb is in the delay close, ignore it.
240  //
241  ASSERT( kcb->DelayedCloseIndex != CmpDelayedCloseSize );
242  return;
243  }
244 
245  for(i=0;i<CMP_LOCK_FREE_KEY_BODY_ARRAY_SIZE;i++) {
246  if( kcb->KeyBodyArray[i] != NULL ) {
247  KeyBodyArrayEmpty = FALSE;
248  break;
249  }
250  }
251 
252  if( (IsListEmpty(&(kcb->KeyBodyListHead)) == TRUE) && (KeyBodyArrayEmpty == TRUE) ) {
253  //
254  // Nobody has this subkey open, but for sure some subkey must be
255  // open. nicely return.
256  //
257  return;
258  }
259 
260 
261  Name = CmpConstructName(kcb);
262  if( !Name ){
263  // we're low on resources
264  if( Context != NULL ) {
265  ((PQUERY_OPEN_SUBKEYS_CONTEXT)Context)->StatusCode = STATUS_INSUFFICIENT_RESOURCES;
266  }
267  return;
268  }
269 
270  //
271  // now iterate through the list of KEY_BODYs referencing this kcb
272  //
273  KeyBody = (PCM_KEY_BODY)kcb->KeyBodyListHead.Flink;
274  while( KeyBody != (PCM_KEY_BODY)(&(kcb->KeyBodyListHead)) ) {
275  KeyBody = CONTAINING_RECORD(KeyBody,
276  CM_KEY_BODY,
277  KeyBodyList);
278  // dump it.
279  CmpDumpOneKeyBody(kcb,KeyBody,Name,Context);
280 
281  // count it
282  (*Count)++;
283 
284  KeyBody = (PCM_KEY_BODY)KeyBody->KeyBodyList.Flink;
285  }
286 
287  //
288  // now dump the ones that are on the fast array
289  //
290  for(i=0;i<CMP_LOCK_FREE_KEY_BODY_ARRAY_SIZE;i++) {
291  KeyBody = kcb->KeyBodyArray[i];
292  if( (KeyBody != NULL) &&
293  ((ULONG_PTR)KeyBody != CMP_FAST_KEY_BODY_ARRAY_DUMP) &&
294  ((ULONG_PTR)KeyBody != CMP_FAST_KEY_BODY_ARRAY_FLUSH) ) {
295  //
296  // avoid races
297  //
298  if( InterlockedCompareExchangePointer(&(kcb->KeyBodyArray[i]),
299  (PVOID)CMP_FAST_KEY_BODY_ARRAY_DUMP,
300  KeyBody) == KeyBody ) {
301  // dump it.
302  CmpDumpOneKeyBody(kcb,KeyBody,Name,Context);
303 
304  // count it
305  (*Count)++;
306  //
307  // set it back
308  //
309  InterlockedCompareExchangePointer(&(kcb->KeyBodyArray[i]),
310  KeyBody,
311  (PVOID)CMP_FAST_KEY_BODY_ARRAY_DUMP);
312  }
313  }
314  }
315 
316  ExFreePoolWithTag(Name, CM_NAME_TAG | PROTECTED_POOL);

434 VOID CmpCleanUpKCBCacheTable(PCM_KEY_CONTROL_BLOCK KeyControlBlock,
435  BOOLEAN RegLockHeldEx)
436 /*++
437 Routine Description:
438 
439  Kicks out of cache all kcbs with RefCount == 0
440 
441 Arguments:
442 
443  KeyControlBlock - when present, it's already locked EX.
444 
445 Return Value:
446 
447 --*/
448 {
449  ULONG i;
450  PCM_KEY_HASH *Current;
451  PCM_KEY_CONTROL_BLOCK kcb;
452  ULONG ThisKcbHashIndex = CmpHashTableSize;
453  ULONG ParentKcbHashIndex = CmpHashTableSize;
454  BOOLEAN CacheChanged;
455 
456  CM_PAGED_CODE();
457 
458  ASSERT( (KeyControlBlock && CmpIsKCBLockedExclusive(KeyControlBlock) &&
459  ((KeyControlBlock->ParentKcb == NULL) || (CmpIsKCBLockedExclusive(KeyControlBlock->ParentKcb)))
460  )
461  || (CmpTestRegistryLockExclusive() == TRUE) );
462 
463  //
464  // make sure all delayed dereferences are serviced first
465  //
466  CmpRunDownDelayDerefKCBEngine(KeyControlBlock,RegLockHeldEx);
467 
468  if(KeyControlBlock!= NULL) {
469  ThisKcbHashIndex = GET_HASH_INDEX(KeyControlBlock->ConvKey);
470  if( KeyControlBlock->ParentKcb != NULL ) {
471  ParentKcbHashIndex = GET_HASH_INDEX(KeyControlBlock->ParentKcb->ConvKey);
472  }
473  }
474 TryAgain:
475  CacheChanged = FALSE;
476  for (i=0; i<CmpHashTableSize; i++) {
477  if( (KeyControlBlock == NULL) || ((ThisKcbHashIndex != i) && (ParentKcbHashIndex != i)) ) {
478  //
479  // deadlock avoidance
480  //
481  if( CmpKCBLockForceAcquireAllowed(ParentKcbHashIndex,ThisKcbHashIndex,i) ) {
482  CmpLockHashEntryByIndexExclusive(i);
483  } else if( CmpTryToLockHashEntryByIndexExclusive(i) == FALSE ) {
484  //
485  // couldn't get EX access to this entry; we need to skip it
486  //
487  continue;
488  }
489  }
490  Current = &(CmpCacheTable[i].Entry);
491  while (*Current) {
492  kcb = CONTAINING_RECORD(*Current, CM_KEY_CONTROL_BLOCK, KeyHash);
493  if( (kcb->RefCount == 0) &&
494  ((KeyControlBlock == NULL) || (KeyControlBlock->KeyHive == kcb->KeyHive)) ) { // only interested in kcbs below this hive's root
495 
496  //
497  // This kcb is in DelayClose case, remove it.
498  //
499  CmpRemoveFromDelayedClose(kcb);
500  CmpCleanUpKcbCacheWithLock(kcb,RegLockHeldEx);
501  CacheChanged = TRUE;
502  //
503  // The HashTable is changed, start over in this index again.
504  //
505  Current = &(CmpCacheTable[i].Entry);
506  continue;
507  }
508  Current = &(kcb->NextHash);
509  }
510  if( (KeyControlBlock == NULL) || ((ThisKcbHashIndex != i) && (ParentKcbHashIndex != i)) ) {
511  CmpUnlockHashEntryByIndex(i);
512  }
513  if( CacheChanged ) {
514  goto TryAgain;
515  }
516  }

522 ULONG
523 CmpSearchForOpenSubKeys(
524  PCM_KEY_CONTROL_BLOCK KeyControlBlock,
525  SUBKEY_SEARCH_TYPE SearchType,
526  BOOLEAN RegLockHeldEx,
527  PVOID SearchContext
528  )
529 /*++
530 Routine Description:
531 
532  This routine searches the KCB tree for any open handles to keys that
533  are subkeys of the given key.
534 
535  It is used by CmRestoreKey to verify that the tree being restored to
536  has no open handles.
537 
538 Arguments:
539 
540  KeyControlBlock - Supplies the key control block for the key for which
541  open subkeys are to be found.
542 
543  SearchType - the type of the search
544  SearchIfExist - exits at the first open subkey found ==> returns 1 if any subkey is open
545 
546  SearchAndDeref - Forces the keys underneath the Key referenced KeyControlBlock to
547  be marked as not referenced (see the REG_FORCE_RESTORE flag in CmRestoreKey);
548  return value indicates number of keys that could NOT be dereferenced, and thus are
549  still open.
550 
551  SearchAndCount - Counts all open subkeys - returns the number of them
552 
553  SearchAndTagNoDelayClose - Flag the subkey so as not to place it in the delayed closed
554  table.
555 
556  SearchAndRehash (NT_RENAME_KEY) - Rehash subkeys in KCB hash table, if needed.
557 
558 Return Value:
559 
560  Returns the number of open sub keys.
561 
562 --*/
563 {
564  ULONG i;
565  PCM_KEY_HASH Current;
566  PCM_KEY_CONTROL_BLOCK kcb;
567  PCM_KEY_CONTROL_BLOCK Parent;
568  ULONG LevelDiff, l;
569  ULONG Count = 0;
570  ULONG ThisKcbHashIndex = CmpHashTableSize;
571  ULONG ParentKcbHashIndex = CmpHashTableSize;
572 
573  //
574  // Registry lock should be held exclusively, so no need to KCB lock
575  //
576  ASSERT( ( CmpIsKCBLockedExclusive(KeyControlBlock) &&
577  ((KeyControlBlock->ParentKcb == NULL) || CmpIsKCBLockedExclusive(KeyControlBlock->ParentKcb))
578  ) ||
579  (CmpTestRegistryLockExclusive() == TRUE) );
580 
581  //
582  // First, clean up all subkeys in the cache
583  //
584  CmpCleanUpKCBCacheTable(((SearchType == SearchIfExist) || (SearchType == SearchAndCount))?KeyControlBlock:NULL,RegLockHeldEx);
585 
586  if( (KeyControlBlock->RefCount == 1) && (SearchType != SearchAndRehash) ) {
587  //
588  // There is only one open handle, so there must be no open subkeys.
589  //
590  Count = 0;
591  } else {
592  //
593  // Now search for an open subkey handle.
594  //
595  Count = 0;
596 
597  //
598  // dump the root first if we were asked to do so.
599  //
600  if(SearchType == SearchAndCount) {
601  CmpDumpKeyBodyList(KeyControlBlock,&Count,SearchContext);
602  }
603  ThisKcbHashIndex = GET_HASH_INDEX(KeyControlBlock->ConvKey);
604  if( KeyControlBlock->ParentKcb != NULL ) {
605  ParentKcbHashIndex = GET_HASH_INDEX(KeyControlBlock->ParentKcb->ConvKey);
606  }
607 
608  for (i=0; i<CmpHashTableSize; i++) {
609 
610  if( (ThisKcbHashIndex != i) && (ParentKcbHashIndex != i) ) {
611  //
612  // we can wait for the bucket lock in case we are going upwards of what we already own
613  //
614  if( CmpKCBLockForceAcquireAllowed(ParentKcbHashIndex,ThisKcbHashIndex,i) ) {
615  CmpLockHashEntryByIndexExclusive(i);
616  } else {
617  //
618  // deadlock avoidance; bail out with error if we can't get EX access here
619  //
620  if( CmpTryToLockHashEntryByIndexExclusive(i) == FALSE ) {
621  ASSERT(CmpTestRegistryLockExclusive() == FALSE);
622  ASSERT( (SearchType == SearchIfExist) || (SearchType == SearchAndCount) );
623  if( SearchType == SearchIfExist ) {
624  //
625  // assume one already exist
626  //
627  return 1;
628  }
629  //
630  // else skip this entry;
631  //
632  continue;
633  }
634  }
635  }
636 
637 StartDeref:
638 
639  Current = CmpCacheTable[i].Entry;
640  while (Current) {
641  kcb = CONTAINING_RECORD(Current, CM_KEY_CONTROL_BLOCK, KeyHash);
642  if (kcb->TotalLevels > KeyControlBlock->TotalLevels) {
643  LevelDiff = kcb->TotalLevels - KeyControlBlock->TotalLevels;
644 
645  Parent = kcb;
646  for (l=0; l<LevelDiff; l++) {
647  Parent = Parent->ParentKcb;
648  }
649 
650  if (Parent == KeyControlBlock) {
651  //
652  // Found a match;
653  //
654  if( SearchType == SearchIfExist ) {
655  Count = 1;
656  if( (ThisKcbHashIndex != i) && (ParentKcbHashIndex != i) ) {
657  CmpUnlockHashEntryByIndex(i);
658  }
659  return Count;
660  } else if(SearchType == SearchAndTagNoDelayClose) {
661  kcb->ExtFlags |= CM_KCB_NO_DELAY_CLOSE;
662  } else if(SearchType == SearchAndDeref) {
663  //
664  // Mark the key as deleted, remove it from cache, but don't add it
665  // to the Delay Close table (we want the key to be visible only to
666  // the one(s) that have open handles on it.
667  //
668 
669  ASSERT_CM_LOCK_OWNED_EXCLUSIVE();
670 
671  //
672  // don't mess with read only kcbs; this prevents a potential hack when
673  // trying to FORCE_RESTORE over WPA keys.
674  //
675  if( !CmIsKcbReadOnly(kcb) ) {
676  //
677  // flush any pending notifies as the kcb won't be around any longer
678  //
679  CmpFlushNotifiesOnKeyBodyList(kcb,TRUE);
680 
681  CmpCleanUpSubKeyInfo(kcb->ParentKcb);
682  kcb->Delete = TRUE;
683  // Cache the pointer to the next hash block before unlinking the
684  // current one.
685  Current = kcb->NextHash;
686  CmpRemoveKeyControlBlock(kcb);
687  kcb->KeyCell = HCELL_NIL;
688  //
689  // Continue the search.
690  //
691  continue;
692  } else {
693  // We cannot close the key. Record it as still open.
694  ++Count;
695  }
696  } else if(SearchType == SearchAndCount) {
697  //
698  // here do the dumping and count incrementing stuff
699  //
700  CmpDumpKeyBodyList(kcb,&Count,SearchContext);
701 
702  } else if( SearchType == SearchAndRehash ) {
703  //
704  // every kcb which has the one passed as a parameter
705  // as an ancestor needs to be moved to the right location

794 BOOLEAN
795 CmpRehashKcbSubtree(
796  PCM_KEY_CONTROL_BLOCK Start,
797  PCM_KEY_CONTROL_BLOCK End
798  )
799 /*++
800 Routine Description:
801 
802  Walks the path between End and Start and rehashed all kcbs that need
803  rehashing;
804 
805  Assumptions: It is a priori taken that Start is an ancestor of End;
806 
807  Works in two steps:
808  1. walks the path backwards from End to Start, reverting the back-link
809  (we use the ParentKcb member in the kcb structure for that). I.e. we build a
810  forward path from Start to End
811  2.Walks the forward path built at 1, rehashes kcbs whos need rehashing and restores
812  the parent relationship.
813 
814 Arguments:
815 
816  KeyControlBlock - where we start
817 
818  kcb - where we stop
819 
820 Return Value:
821 
822  TRUE if at least one kcb has been rehashed
823 
824 --*/
825 {
826  PCM_KEY_CONTROL_BLOCK Parent;
827  PCM_KEY_CONTROL_BLOCK Current;
828  PCM_KEY_CONTROL_BLOCK TmpKcb;
829  ULONG ConvKey;
830  BOOLEAN Result;
831 
832  CM_PAGED_CODE();
833 
834 #if DBG
835  //
836  // make sure Start is an ancestor of End;
837  //
838  {
839  ULONG LevelDiff = End->TotalLevels - Start->TotalLevels;
840 
841  ASSERT( (LONG)LevelDiff >= 0 );
842 
843  TmpKcb = End;
844  for(;LevelDiff; LevelDiff--) {
845  TmpKcb = TmpKcb->ParentKcb;
846  }
847 
848  ASSERT( TmpKcb == Start );
849  }
850 
851 #endif
852  //
853  // Step 1: walk the path backwards (using the parentkcb link) and
854  // revert it, until we reach Start. It is assumed that Start is an
855  // ancestor of End (the caller must not call this function otherwise !!!)
856  //
857  Current = NULL;
858  Parent = End;
859  while( Current != Start ) {
860  //
861  // revert the link
862  //
863  TmpKcb = Parent->ParentKcb;
864  Parent->ParentKcb = Current;
865  Current = Parent;
866  Parent = TmpKcb;
867 
868  ASSERT( Current->TotalLevels >= Start->TotalLevels );
869  }
870 
871  ASSERT( Current == Start );
872 
873  //
874  // Step 2: Walk the forward path built at 1 and rehash the kcbs that need
875  // caching; At the same time, restore the links (parent relationships)
876  //
877  Result = FALSE;
878  while( Current != NULL ) {
879  //
880  // see if we need to rehash this kcb;
881  //
882  //
883  // restore the parent relationship; need to do this first so
884  // CmpComputeKcbConvKey works OK
885  //
886  TmpKcb = Current->ParentKcb;
887  Current->ParentKcb = Parent;
888 
889  ConvKey = CmpComputeKcbConvKey(Current);
890  if( ConvKey != Current->ConvKey ) {
891  //
892  // rehash the kcb by removing it from hash, and then inserting it
893  // again with th new ConvKey
894  //
895  CmpRemoveKeyHash(&(Current->KeyHash));
896  Current->ConvKey = ConvKey;
897  CmpInsertKeyHash(&(Current->KeyHash),FALSE);
898  Result = TRUE;
899  }
900 
901  //
902  // advance forward
903  //
904  Parent = Current;
905  Current = TmpKcb;
906  }
907 
908  ASSERT( Parent == End );
909 
910  return Result;
911 }

913 BOOLEAN
914 CmpReferenceKeyControlBlock(
915  PCM_KEY_CONTROL_BLOCK KeyControlBlock
916  )
917 {
918  // Note: this is called only under KCB lock
919  LONG RefCount;
920 
921  ASSERT_KCB_LOCKED( KeyControlBlock );
922 
923  if( (KeyControlBlock->RefCount == 0) &&
924  (CmpIsKCBLockedExclusive(KeyControlBlock) == FALSE) &&
925  (CmpTryConvertKCBLockSharedToExclusive(KeyControlBlock) == FALSE) ) {
926  //
927  // need to get the lock exclusive as we are changing the cache table
928  // if we're here, the only way this kcb can go away from under us is if the delay close worker fires
929  // while we are dropping the lock; we set DelayCloseIndex to 1 (different than 0 that the worker uses)
930  // to signal the worker to ignore this kcb; then after we get the lock exclusive, we put it back to 0
931  // (if it's still 1); We're the only ones playing with that value.
932  //
933 
934  KeyControlBlock->DelayedCloseIndex = 1;
935 
936  // add an artificial refcount so it doesn't go away from under us while we drop the lock
937  InterlockedIncrement( (PLONG)&KeyControlBlock->RefCount );
938  CmpUpgradeKCBLockToExclusive(KeyControlBlock);
939 
940  // remove the artificial refcount we added above
941  InterlockedDecrement( (PLONG)&KeyControlBlock->RefCount);
942  if( KeyControlBlock->DelayedCloseIndex == 1 ) {
943  KeyControlBlock->DelayedCloseIndex = 0;
944  } else {
945  // else someone beat us here
946  ASSERT( (KeyControlBlock->DelayedCloseIndex == CmpDelayedCloseSize) ||
947  (KeyControlBlock->DelayedCloseIndex == 0) );
948  }
949  }
950  RefCount = (InterlockedIncrement( (PLONG)&KeyControlBlock->RefCount )) & 0xffff;
951  if (RefCount == 0) {
952  //
953  // We have maxed out the ref count on this key. Probably
954  // some bogus app has opened the same key 64K times without
955  // ever closing it. Just fail the call
956  //
957  InterlockedDecrement( (PLONG)&KeyControlBlock->RefCount);
958  return FALSE;
959  } else if( KeyControlBlock->DelayedCloseIndex == 0 ) {
960  if( (CmpIsKCBLockedExclusive(KeyControlBlock) == FALSE) &&
961  (CmpTryConvertKCBLockSharedToExclusive(KeyControlBlock) == FALSE) ) {
962  //
963  // we need to promote this kcb to EX; safe since we're already added a reference on it.
964  //
965  CmpUpgradeKCBLockToExclusive(KeyControlBlock);
966  }
967  if( KeyControlBlock->DelayedCloseIndex == 0 ) {
968  CmpRemoveFromDelayedClose(KeyControlBlock);
969  }
970  }
971 
972  LogKCBReference(KeyControlBlock,1);
973  return TRUE;

2003 BOOLEAN
2004 CmpSearchKeyControlBlockTree(
2005  PKCB_WORKER_ROUTINE WorkerRoutine,
2006  PVOID Context1,
2007  PVOID Context2
2008  )
2009 /*++
2010 
2011 Routine Description:
2012 
2013  Traverse the kcb tree. We will visit all nodes unless WorkerRoutine
2014  tells us to stop part way through.
2015 
2016  For each node, call WorkerRoutine(..., Context1, Contex2). If it returns
2017  KCB_WORKER_DONE, we are done, simply return. If it returns
2018  KCB_WORKER_CONTINUE, just continue the search. If it returns KCB_WORKER_DELETE,
2019  the specified KCB is marked as deleted.
2020  If it returns KCB_WORKER_ERROR we bail out and signal the error to the caller.
2021 
2022  This routine has the side-effect of removing all delayed-close KCBs.
2023 
2024 Arguments:
2025 
2026  WorkerRoutine - applied to nodes witch Match.
2027 
2028  Context1 - data we pass through
2029 
2030  Context2 - data we pass through
2031 
2032 
2033 Return Value:
2034 
2035  NONE.
2036 
2037 --*/
2038 {
2039  PCM_KEY_CONTROL_BLOCK Current;
2040  PCM_KEY_HASH *Prev;
2041  ULONG WorkerResult;
2042  ULONG i;
2043 
2044  //
2045  // Walk the hash table
2046  //
2047  for (i=0; i<CmpHashTableSize; i++) {
2048  CmpLockHashEntryByIndexExclusive(i);
2049  Prev = &(CmpCacheTable[i].Entry);
2050  while (*Prev) {
2051  Current = CONTAINING_RECORD(*Prev,
2052  CM_KEY_CONTROL_BLOCK,
2053  KeyHash);
2054  ASSERT_KCB(Current);
2055  ASSERT(!Current->Delete);
2056  if (Current->RefCount == 0) {
2057  //
2058  // This kcb is in DelayClose case, remove it.
2059  //
2060  CmpRemoveFromDelayedClose(Current);
2061  CmpCleanUpKcbCacheWithLock(Current,FALSE);
2062 
2063  //
2064  // The HashTable is changed, start over in this index again.
2065  //
2066  Prev = &(CmpCacheTable[i].Entry);
2067  continue;
2068  }
2069 
2070  WorkerResult = (WorkerRoutine)(Current, Context1, Context2);
2071  if (WorkerResult == KCB_WORKER_DONE) {
2072  CmpUnlockHashEntryByIndex(i);
2073  return TRUE;
2074  } else if (WorkerResult == KCB_WORKER_ERROR) {
2075  CmpUnlockHashEntryByIndex(i);
2076  return FALSE;
2077  } else if (WorkerResult == KCB_WORKER_DELETE) {
2078  ASSERT(Current->Delete);
2079  *Prev = Current->NextHash;
2080  continue;
2081  } else {
2082  ASSERT(WorkerResult == KCB_WORKER_CONTINUE);
2083  Prev = &Current->NextHash;
2084  }
2085  }
2086  CmpUnlockHashEntryByIndex(i);
2087  }
