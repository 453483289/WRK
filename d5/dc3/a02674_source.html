128 PRTL_TRACE_DATABASE
129 RtlTraceDatabaseCreate (
130  IN ULONG Buckets,
131  IN SIZE_T MaximumSize OPTIONAL,
132  IN ULONG Flags, // OPTIONAL in User mode
133  IN ULONG Tag, // OPTIONAL in User mode
134  IN RTL_TRACE_HASH_FUNCTION HashFunction OPTIONAL
135  )
136 /*++
137 
138 Routine Description:
139 
140  This routine creates a trace database, that is a hash table
141  to store stack traces.
142 
143 Arguments:
144 
145  Buckets - no of buckets of the hash table with simple chaining.
146 
147  MaximumSize - maximum amount of memory that the database can use.
148  When limit is hit, database add operations will start to fail.
149  If the value is zero then no limit will be imposed.
150 
151  Flags - flags to control if allocation in K mode is done in P or NP
152  pool. The possible bits that can be used right now are:
153  RTL_TRACE_USE_PAGED_POOL
154  RTL_TRACE_USE_NONPAGED_POOL
155 
156  Tag - tag used for K mode allocations.
157 
158  HashFunction - hash function to be used. If null passed a standard hash
159  function will be provided by the module.
160 
161 Return Value:
162 
163  Pointer to an initialized trace database structure.
164 
165 Environment:
166 
167  Any.
168 
169 --*/
170 {
171  PVOID RawArea;
172  SIZE_T RawSize;
173  PRTL_TRACE_DATABASE Database;
174  PRTL_TRACE_SEGMENT Segment;
175  ULONG FirstFlags;
176 
177  //
178  // Prepare trace database flags. The first segment of
179  // the database will be allocated in nonpaged pool
180  // no matter what flags are used because it contains
181  // kernel synchronization objects that need to be in
182  // that pool.
183  //
184 
185  Flags |= RTL_TRACE_IN_KERNEL_MODE;
186  FirstFlags = RTL_TRACE_IN_KERNEL_MODE | RTL_TRACE_USE_NONPAGED_POOL;
187 
188  //
189  // Allocate first segment of trace database that will contain
190  // DATABASE, SEGMENT, buckets of the hash table and later traces.
191  //
192 
193  RawSize = sizeof (RTL_TRACE_DATABASE) +
194  sizeof (RTL_TRACE_SEGMENT) +
195  Buckets * sizeof (PRTL_TRACE_BLOCK);
196 
197  RawSize += RTL_TRACE_SIZE_INCREMENT;
198  RawSize &= ~(RTL_TRACE_SIZE_INCREMENT - 1);
199 
200  RawArea = RtlpTraceDatabaseAllocate (
201  RawSize,
202  FirstFlags,
203  Tag);
204 
205  if (RawArea == NULL) {
206  return NULL;
207  }
208 
209  Database = (PRTL_TRACE_DATABASE)RawArea;
210  Segment = (PRTL_TRACE_SEGMENT)(Database + 1);
211 
212  //
213  // Initialize the database
214  //
215 
216  Database->Magic = RTL_TRACE_DATABASE_MAGIC;
217  Database->Flags = Flags;
218  Database->Tag = Tag;
219  Database->SegmentList = NULL;
220  Database->MaximumSize = MaximumSize;
221  Database->CurrentSize = RTL_TRACE_SIZE_INCREMENT;
222  Database->Owner = NULL;
223 
224  Database->NoOfHits = 0;
225  Database->NoOfTraces = 0;
226  RtlZeroMemory (Database->HashCounter, sizeof (Database->HashCounter));
227 
228  if (! RtlpTraceDatabaseInitializeLock (Database)) {
229  RtlpTraceDatabaseFree (RawArea, Tag);
230  return NULL;
231  }
232 
233  Database->NoOfBuckets = Buckets;
234 
235  if (HashFunction == NULL) {
236  Database->HashFunction = RtlpTraceStandardHashFunction;
237  }
238  else {
239  Database->HashFunction = HashFunction;
240  }
241 
242  //
243  // Initialize first segment of the database
244  //
245 
246  Segment->Magic = RTL_TRACE_SEGMENT_MAGIC;
247  Segment->Database = Database;
248  Segment->NextSegment = NULL;
249  Segment->TotalSize = RTL_TRACE_SIZE_INCREMENT;
250 
251  Database->SegmentList = Segment;
252 
253  //
254  // Initialize the buckets of the database.
255  //
256 
257  Database->Buckets = (PRTL_TRACE_BLOCK *)(Segment + 1);
258  RtlZeroMemory (Database->Buckets, Database->NoOfBuckets * sizeof(PRTL_TRACE_BLOCK));
259 
260  //
261  // Initialize free pointer for segment
262  //
263 
264  Segment->SegmentStart = (PCHAR)RawArea;
265  Segment->SegmentEnd = Segment->SegmentStart + RTL_TRACE_SIZE_INCREMENT;
266  Segment->SegmentFree = (PCHAR)(Segment + 1) + Database->NoOfBuckets * sizeof(PRTL_TRACE_BLOCK);
267 
268  return Database;
269 }

271 BOOLEAN
272 RtlTraceDatabaseDestroy (
273  IN PRTL_TRACE_DATABASE Database
274  )
275 /*++
276 
277 Routine Description:
278 
279  This routine destroys a trace database. It takes care of
280  deallocating everything, uninitializing synchronization
281  objects, etc.
282 
283 Arguments:
284 
285  Database - trace database
286 
287 Return Value:
288 
289  TRUE if destroy operation was successful. FALSE otherwise.
290 
291 Environment:
292 
293  Any.
294 
295 --*/
296 {
297  PRTL_TRACE_SEGMENT Current;
298  BOOLEAN Success;
299  BOOLEAN SomethingFailed = FALSE;
300  PRTL_TRACE_SEGMENT NextSegment;
301 
302  //
303  // Sanity checks.
304  //
305 
306  TRACE_ASSERT (Database && Database->Magic == RTL_TRACE_DATABASE_MAGIC);
307  TRACE_ASSERT (RtlTraceDatabaseValidate (Database));
308 
309  //
310  // Uninitialize the database lock. Even if we fail
311  // we will continue to release memory for all segments.
312  //
313  // N.B. We cannot acquire the lock here for the last time because this
314  // has as a side effect elevating the irql (in K mode) and then the
315  // function will return with raised irql.
316  //
317 
318  Success = RtlpTraceDatabaseUninitializeLock (Database);
319 
320  if (! Success) {
321  SomethingFailed = TRUE;
322  }
323 
324  //
325  // Traverse the list of segments and release memory one by one.
326  // Special attention with the last segment because it contains
327  // the database structure itself and we do not want to shoot.
328  // ourselves in the foot.
329  //
330 
331  for (Current = Database->SegmentList;
332  Current != NULL;
333  Current = NextSegment) {
334 
335  //
336  // We save the next segment before freeing the structure.
337  //
338 
339  NextSegment = Current->NextSegment;
340 
341  //
342  // If this is the last segment we need to offset Current pointer
343  // by the size of the database structure.
344  //
345 
346  if (NextSegment == NULL) {
347 
348  Current = (PRTL_TRACE_SEGMENT) ((PRTL_TRACE_DATABASE)Current - 1);
349  }
350 
351  Success = RtlpTraceDatabaseFree (Current, Database->Tag);
352 
353  if (! Success) {
354 
355  DbgPrint ("Trace database: failed to release segment %p \n", Current);
356  SomethingFailed = TRUE;
357  }
358  }
359 
360  if (SomethingFailed) {
361  return FALSE;
362  }
363  else {
364  return TRUE;
365  }
366 }

369 RtlTraceDatabaseValidate (
370  IN PRTL_TRACE_DATABASE Database
371  )
372 /*++
373 
374 Routine Description:
375 
376  This routine validates the correctness of a trace database.
377  It is intended to be used for testing purposes.
378 
379 Arguments:
380 
381  Database - trace database
382 
383 Return Value:
384 
385  TRUE if the database is ok. For most of the inconsistencies this
386  function will break in the debugger.
387 
388 Environment:
389 
390  Any.
391 
392 --*/
393 {
394  PRTL_TRACE_SEGMENT Segment;
395  PRTL_TRACE_BLOCK Block;
396  ULONG Index;
397 
398  TRACE_ASSERT (Database != NULL);
399  TRACE_ASSERT (Database->Magic == RTL_TRACE_DATABASE_MAGIC);
400 
401  RtlpTraceDatabaseAcquireLock (Database);
402 
403  //
404  // Check all segments.
405  //
406 
407  for (Segment = Database->SegmentList;
408  Segment != NULL;
409  Segment = Segment->NextSegment) {
410 
411  TRACE_ASSERT (Segment->Magic == RTL_TRACE_SEGMENT_MAGIC);
412  }
413 
414  //
415  // Check all blocks.
416  //
417 
418  for (Index = 0; Index < Database->NoOfBuckets; Index++) {
419 
420  for (Block = Database->Buckets[Index];
421  Block != NULL;
422  Block = Block->Next) {
423 
424  TRACE_ASSERT (Block->Magic == RTL_TRACE_BLOCK_MAGIC);
425  }
426  }
427 
428  RtlpTraceDatabaseReleaseLock (Database);
429  return TRUE;
430 }

553 BOOLEAN
554 RtlpTraceDatabaseInternalAdd (
555  IN PRTL_TRACE_DATABASE Database,
556  IN ULONG Count,
557  IN PVOID * Trace,
558  OUT PRTL_TRACE_BLOCK * TraceBlock OPTIONAL
559  )
560 /*++
561 
562 Routine Description:
563 
564  This is the internal routine to add a trace. See RtlTraceDatabaseAdd
565  for more details.
566 
567 Arguments:
568 
569  Database - trace database
570 
571  Count - size of trace (in PVOIDs)
572 
573  Trace - trace
574 
575  TraceBlock - address of block where the trace is stored
576 
577 Return Value:
578 
579  TRUE if trace was added.
580 
581 Environment:
582 
583  Called from RtlTraceDatabaseAdd. Assumes the database lock
584  is held.
585 
586 --*/
587 {
588  PRTL_TRACE_BLOCK Block;
589  PRTL_TRACE_SEGMENT Segment;
590  PRTL_TRACE_SEGMENT TopSegment;
591  ULONG HashValue;
592  SIZE_T RequestSize;
593 
594  //
595  // Check if the block is already in the database (hash table).
596  // If it is increase the number of hits and return.
597  //
598 
599  if (RtlpTraceDatabaseInternalFind (Database, Count, Trace, &Block)) {
600 
601  Block->Count += 1;
602 
603  if (TraceBlock) {
604  *TraceBlock = Block;
605  }
606 
607  Database->NoOfHits += 1;
608  return TRUE;
609  }
610 
611  //
612  // We need to create a new block. First we need to figure out
613  // if the current segment can accommodate the new block.
614  //
615 
616  RequestSize = sizeof(*Block) + Count * sizeof(PVOID);
617 
618  TopSegment = Database->SegmentList;
619  if (RequestSize > (SIZE_T)(TopSegment->SegmentEnd - TopSegment->SegmentFree)) {
620 
621  //
622  // If the database has a maximum size and that limit
623  // has been reached then fail the call.
624  //
625 
626  if (Database->MaximumSize > 0) {
627  if (Database->CurrentSize > Database->MaximumSize) {
628 
629  if (TraceBlock) {
630  *TraceBlock = NULL;
631  }
632 
633  return FALSE;
634  }
635  }
636 
637  //
638  // Allocate a new database segment. Fail call if cannot
639  // allocate.
640  //
641 
642  Segment = RtlpTraceSegmentCreate (RTL_TRACE_SIZE_INCREMENT,
643  Database->Flags,
644  Database->Tag);
645 
646  if (Segment == NULL) {
647 
648  if (TraceBlock) {
649  *TraceBlock = NULL;
650  }
651 
652  return FALSE;
653  }
654 
655  //
656  // Add the new segment to the database.
657  //
658 
659  Segment->Magic = RTL_TRACE_SEGMENT_MAGIC;
660  Segment->Database = Database;
661  Segment->TotalSize = RTL_TRACE_SIZE_INCREMENT;
662  Segment->SegmentStart = (PCHAR)Segment;
663  Segment->SegmentEnd = Segment->SegmentStart + RTL_TRACE_SIZE_INCREMENT;
664  Segment->SegmentFree = (PCHAR)(Segment + 1);
665 
666  Segment->NextSegment = Database->SegmentList;
667  Database->SegmentList = Segment;
668  TopSegment = Database->SegmentList;
669 
670  Database->CurrentSize += RTL_TRACE_SIZE_INCREMENT;
671  }
672 
673  if (RequestSize > (SIZE_T)(TopSegment->SegmentEnd - TopSegment->SegmentFree)) {
674 
675  DbgPrint ("Trace database: failing attempt to save biiiiig trace (size %u) \n",
676  Count);
677 
678  if (TraceBlock) {
679  *TraceBlock = NULL;
680  }
681 
682  return FALSE;
683  }
684 
685  //
686  // Finally we can allocate our block.
687  //
688 
689  Block = (PRTL_TRACE_BLOCK)(TopSegment->SegmentFree);
690  TopSegment->SegmentFree += RequestSize;
691 
692  //
693  // Fill the block with the new trace.
694  //
695 
696  Block->Magic = RTL_TRACE_BLOCK_MAGIC;
697  Block->Size = Count;
698  Block->Count = 1;
699  Block->Trace = (PVOID *)(Block + 1);
700 
701  Block->UserCount = 0;
702  Block->UserSize = 0;
703 
704  //
705  // Copy the trace
706  //
707 
708  RtlMoveMemory (Block->Trace, Trace, Count * sizeof(PVOID));
709 
710  //
711  // Add the block to corresponding bucket.
712  //
713 
714  HashValue = (Database->HashFunction) (Count, Trace);
715  HashValue %= Database->NoOfBuckets;
716  Database->HashCounter[HashValue / (Database->NoOfBuckets / 16)] += 1;
717 
718  Block->Next = Database->Buckets[HashValue];
719  Database->Buckets[HashValue] = Block;
720 
721  //
722  // Loooong function. Finally return success.
723  //
724 
725  if (TraceBlock) {
726  *TraceBlock = Block;
727  }
728 
729  Database->NoOfTraces += 1;
730  return TRUE;
731 }

733 BOOLEAN
734 RtlpTraceDatabaseInternalFind (
735  PRTL_TRACE_DATABASE Database,
736  IN ULONG Count,
737  IN PVOID * Trace,
738  OUT PRTL_TRACE_BLOCK * TraceBlock OPTIONAL
739  )
740 /*++
741 
742 Routine Description:
743 
744  This internal routine searches for a trace in the database.
745 
746 Arguments:
747 
748  Database - trace database
749 
750  Count - size of trace (in PVOIDs)
751 
752  Trace - trace
753 
754  TraceBlock - element where the trace is stored.
755 
756 Return Value:
757 
758  TRUE if the trace was found.
759 
760 Environment:
761 
762  Called from RtlTraceDatabaseFind. Assumes the database lock is held.
763 
764 --*/
765 {
766  ULONG HashValue;
767  PRTL_TRACE_BLOCK Current;
768  ULONG Index;
769  ULONG RequestSize;
770 
771  //
772  // Find the bucket to search into.
773  //
774 
775  HashValue = (Database->HashFunction) (Count, Trace);
776  Database->HashCounter[HashValue % 16] += 1;
777  HashValue %= Database->NoOfBuckets;
778 
779  //
780  // Traverse the list of blocks for the found bucket
781  //
782 
783  for (Current = Database->Buckets[HashValue];
784  Current != NULL;
785  Current = Current->Next) {
786 
787  //
788  // If the size of the trace matches we might have a chance
789  // to find an equal trace.
790  //
791 
792  if (Count == Current->Size) {
793 
794  //
795  // Figure out if the whole trace matches.
796  //
797 
798  for (Index = 0; Index < Count; Index++) {
799  if (Current->Trace[Index] != Trace[Index]) {
800  break;
801  }
802  }
803 
804  //
805  // If the trace matched completely we have found an entry.
806  //
807 
808  if (Index == Count) {
809  if (TraceBlock) {
810  *TraceBlock = Current;
811  }
812 
813  return TRUE;
814  }
815  }
816  }
817 
818  //
819  // If we traversed the whole list for the hashed bucket and did not
820  // find anything we will fail the call.
821  //
822 
823  if (TraceBlock) {
824  *TraceBlock = NULL;
825  }
826 
827  return FALSE;
828 }

830 ULONG
831 RtlpTraceStandardHashFunction (
832  IN ULONG Count,
833  IN PVOID * Trace
834  )
835 /*++
836 
837 Routine Description:
838 
839  This routine is a simple hash function for stack traces in
840  the case the caller of RtlTraceDatabaseCreate does not provide
841  one. The function just xor's together all the pointers in the
842  trace.
843 
844 Arguments:
845 
846  Count - size of trace (in PVOIDs)
847 
848  Trace - trace
849 
850 Return Value:
851 
852  Hash value. This needs to be reduced to the number of buckets
853  in the hash table by a modulo operation (or something similar).
854 
855 Environment:
856 
857  Called internally by RtlpTraceDatabaseInternalAdd/Find.
858 
859 --*/
860 {
861  ULONG_PTR Value = 0;
862  ULONG Index;
863  unsigned short * Key;
864 
865  Key = (unsigned short *)Trace;
866  for (Index = 0; Index < Count * (sizeof (PVOID) / sizeof(*Key)); Index += 2) {
867 
868  Value += Key[Index] ^ Key[Index + 1];
869  }
870 
871  return PtrToUlong ((PVOID)Value);
872 }

1222  RTL_TRACE_ENUMERATE function (zeroed initially) to keep the state of
1223  the enumeration. Since the trace database does not support delete

1243 
1244  User/Kernel mode.

1252  TRACE_ASSERT (Database->Magic == RTL_TRACE_DATABASE_MAGIC);
