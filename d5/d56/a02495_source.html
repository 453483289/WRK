136 /*++
137 
138 Routine Description:
139 
140  This function returns the next physical page number from the largest
141  largest free descriptor. If there are not enough physical pages left
142  to satisfy the request then a bugcheck is executed since the system
143  cannot be initialized.
144 
145 Arguments:
146 
147  PagesNeeded - Supplies the number of pages needed.
148 
149  UseSlush - Supplies TRUE if the slush descriptor can be used (ie, the
150  allocation will never be freed and will always be cachable).
151 
152 Return Value:
153 
154  The base of the range of physically contiguous pages.
155 
156 Environment:
157 
158  Kernel mode, Phase 0 only.
159 
160 --*/
161 
162 {
163  PFN_NUMBER PageFrameIndex;
164 
165  if (UseSlush == TRUE) {
166 
167  if ((MxSlushDescriptor1 != NULL) &&
168  (PagesNeeded <= MxSlushDescriptor1->PageCount)) {
169 
170  PageFrameIndex = MxSlushDescriptor1->BasePage;
171 
172  MxSlushDescriptor1->BasePage += (ULONG) PagesNeeded;
173  MxSlushDescriptor1->PageCount -= (ULONG) PagesNeeded;
174 
175  return PageFrameIndex;
176  }
177 
178  if ((MxSlushDescriptor2 != NULL) &&
179  (PagesNeeded <= MxSlushDescriptor2->PageCount)) {
180 
181  PageFrameIndex = MxSlushDescriptor2->BasePage;
182 
183  MxSlushDescriptor2->BasePage += (ULONG) PagesNeeded;
184  MxSlushDescriptor2->PageCount -= (ULONG) PagesNeeded;
185 
186  return PageFrameIndex;
187  }
188  }
189 
190  //
191  // Examine the free descriptor to see if enough usable memory is available.
192  //
193 
194  if (PagesNeeded > MxFreeDescriptor->PageCount) {
195 
196  KeBugCheckEx (INSTALL_MORE_MEMORY,
197  MmNumberOfPhysicalPages,
198  MxFreeDescriptor->PageCount,
199  MxOldFreeDescriptor.PageCount,
200  PagesNeeded);
201  }
202 
203  PageFrameIndex = MxFreeDescriptor->BasePage;
204 
205  MxFreeDescriptor->BasePage += (ULONG) PagesNeeded;
206  MxFreeDescriptor->PageCount -= (ULONG) PagesNeeded;
207 
208  return PageFrameIndex;
209 }

242 MxConvertToLargePage (
243  IN PVOID VirtualAddress,
244  IN PVOID EndVirtualAddress
245  )
246 
247 /*++
248 
249 Routine Description:
250 
251  This function converts the backing for the supplied virtual address range
252  to a large page mapping.
253 
254 Arguments:
255 
256  VirtualAddress - Supplies the virtual address to convert to a large page.
257 
258  EndVirtualAddress - Supplies the end virtual address to convert to a
259  large page.
260 
261 Return Value:
262 
263  None.
264 
265 Environment:
266 
267  Kernel mode, Phase 1 only.
268 
269 --*/
270 
271 {
272  ULONG i;
273  MMPTE TempPde;
274  PMMPTE PointerPde;
275  PMMPTE LastPde;
276  PMMPTE PointerPte;
277  KIRQL OldIrql;
278  PMMPFN Pfn1;
279  PFN_NUMBER PageFrameIndex;
280  LOGICAL ValidPteFound;
281  PFN_NUMBER LargePageBaseFrame;
282 
283  ASSERT (MxMapLargePages != 0);
284 
285  PointerPde = MiGetPdeAddress (VirtualAddress);
286  LastPde = MiGetPdeAddress (EndVirtualAddress);
287 
288  TempPde = ValidKernelPde;
289  TempPde.u.Hard.LargePage = 1;
290  TempPde.u.Hard.Global = 1;
291 
292  LOCK_PFN (OldIrql);
293 
294  for ( ; PointerPde <= LastPde; PointerPde += 1) {
295 
296  ASSERT (PointerPde->u.Hard.Valid == 1);
297 
298  if (PointerPde->u.Hard.LargePage == 1) {
299  continue;
300  }
301 
302  PointerPte = MiGetVirtualAddressMappedByPte (PointerPde);
303 
304  //
305  // Here's a nasty little hack - the page table page mapping the kernel
306  // and HAL (built by the loader) does not necessarily fill all the
307  // page table entries (ie: any number of leading entries may be zero).
308  //
309  // To deal with this, walk forward until a nonzero entry is found
310  // and re-index the large page based on this.
311  //
312 
313  ValidPteFound = FALSE;
314  LargePageBaseFrame = (ULONG)-1;
315  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
316 
317  ASSERT ((PageFrameIndex & (MM_PFN_MAPPED_BY_PDE - 1)) == 0);
318 
319  for (i = 0; i < PTE_PER_PAGE; i += 1) {
320 
321  ASSERT ((PointerPte->u.Long == 0) ||
322  (ValidPteFound == FALSE) ||
323  (PageFrameIndex == MI_GET_PAGE_FRAME_FROM_PTE (PointerPte)));
324  if (PointerPte->u.Hard.Valid == 1) {
325  if (ValidPteFound == FALSE) {
326  ValidPteFound = TRUE;
327  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
328  LargePageBaseFrame = PageFrameIndex - i;
329  }
330  }
331  PointerPte += 1;
332  PageFrameIndex += 1;
333  }
334 
335  if (ValidPteFound == FALSE) {
336  continue;
337  }
338 
339  TempPde.u.Hard.PageFrameNumber = LargePageBaseFrame;
340 
341  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPde);
342 
343  MI_WRITE_VALID_PTE_NEW_PAGE (PointerPde, TempPde);
344 
345  MI_FLUSH_ENTIRE_TB (0xF);
346 
347  if (((PageFrameIndex >= MxOldSlushDescriptor1.BasePage) &&
348  (PageFrameIndex < MxOldSlushDescriptor1.BasePage + MxOldSlushDescriptor1.PageCount)) ||
349 
350  ((PageFrameIndex >= MxOldSlushDescriptor2.BasePage) &&
351  (PageFrameIndex < MxOldSlushDescriptor2.BasePage + MxOldSlushDescriptor2.PageCount))) {
352 
353  //
354  // Excess slush is given to expansion nonpaged pool here to
355  // ensure that it is always mapped fully cached since the
356  // rest of the large page is inserted as fully cached into
357  // the TB.
358  //
359 
360  MiAddExpansionNonPagedPool (PageFrameIndex, 1, TRUE);
361  }
362  else {
363  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
364  Pfn1->u2.ShareCount = 0;
365  Pfn1->u3.e2.ReferenceCount = 1;
366  Pfn1->u3.e1.PageLocation = StandbyPageList;
367  MI_SET_PFN_DELETED (Pfn1);
368  MiDecrementReferenceCount (Pfn1, PageFrameIndex);
369  }
370  }
371 
372  UNLOCK_PFN (OldIrql);
373 }

375 MiIsRegularMemory (
376  IN PLOADER_PARAMETER_BLOCK LoaderBlock,
377  IN PFN_NUMBER PageFrameIndex
378  )
379 
380 /*++
381 
382 Routine Description:
383 
384  This routine checks whether the argument page frame index represents
385  regular memory in the loader descriptor block. It is only used very
386  early during Phase0 init because the MmPhysicalMemoryBlock is not yet
387  initialized.
388 
389 Arguments:
390 
391  LoaderBlock - Supplies a pointer to the firmware setup loader block.
392 
393  PageFrameIndex - Supplies the page frame index to check.
394 
395 Return Value:
396 
397  TRUE if the frame represents regular memory, FALSE if not.
398 
399 Environment:
400 
401  Kernel mode.
402 
403 --*/
404 
405 {
406  PLIST_ENTRY NextMd;
407  PMEMORY_ALLOCATION_DESCRIPTOR MemoryDescriptor;
408 
409  NextMd = LoaderBlock->MemoryDescriptorListHead.Flink;
410 
411  while (NextMd != &LoaderBlock->MemoryDescriptorListHead) {
412 
413  MemoryDescriptor = CONTAINING_RECORD (NextMd,
414  MEMORY_ALLOCATION_DESCRIPTOR,
415  ListEntry);
416 
417  if (PageFrameIndex >= MemoryDescriptor->BasePage) {
418 
419  if (PageFrameIndex < MemoryDescriptor->BasePage + MemoryDescriptor->PageCount) {
420 
421  if ((MemoryDescriptor->MemoryType == LoaderFirmwarePermanent) ||
422  (MemoryDescriptor->MemoryType == LoaderBBTMemory) ||
423  (MemoryDescriptor->MemoryType == LoaderSpecialMemory)) {
424 
425  //
426  // This page lies in a memory descriptor for which we will
427  // never create PFN entries, hence return FALSE.
428  //
429 
430  break;
431  }
432 
433  return TRUE;
434  }
435  }
436  else {
437 
438  //
439  // Since the loader memory list is sorted in ascending order,
440  // the requested page must not be in the loader list at all.
441  //
442 
443  break;
444  }
445 
446  NextMd = MemoryDescriptor->ListEntry.Flink;
447  }
448 
449  //
450  // The final check before returning FALSE is to ensure that the requested
451  // page wasn't one of the ones we used to normal-map the loader mappings,
452  // etc.
453  //
454 
455  if ((PageFrameIndex >= MxOldFreeDescriptor.BasePage) &&
456  (PageFrameIndex < MxOldFreeDescriptor.BasePage + MxOldFreeDescriptor.PageCount)) {
457 
458  return TRUE;
459  }
460 
461  if ((PageFrameIndex >= MxOldSlushDescriptor1.BasePage) &&
462  (PageFrameIndex < MxOldSlushDescriptor1.BasePage + MxOldSlushDescriptor1.PageCount)) {
463 
464  return TRUE;
465  }
466 
467  if ((PageFrameIndex >= MxOldSlushDescriptor2.BasePage) &&
468  (PageFrameIndex < MxOldSlushDescriptor2.BasePage + MxOldSlushDescriptor2.PageCount)) {
469 
470  return TRUE;
471  }
472 
473  if ((PageFrameIndex >= MiSlushDescriptorBase) &&
474  (PageFrameIndex < MiSlushDescriptorBase + MiSlushDescriptorCount)) {
475 
476  return TRUE;
477  }
478 
479  return FALSE;
480 }

482 VOID
483 MiReportPhysicalMemory (
484  VOID
485  )
486 
487 /*++
488 
489 Routine Description:
490 
491  This routine is called during Phase 0 initialization once the
492  MmPhysicalMemoryBlock has been constructed. It's job is to decide
493  which large page ranges to enable later and also to construct a
494  large page comparison list so any requests which are not fully cached
495  can check this list in order to refuse conflicting requests.
496 
497 Arguments:
498 
499  None.
500 
501 Return Value:
502 
503  None.
504 
505 Environment:
506 
507  Kernel mode. Phase 0 only.
508 
509  This is called before any non-MmCached allocations are made.
510 
511 --*/
512 
513 {
514  ULONG i, j;
515  PMMPTE PointerPte;
516  LOGICAL EntryFound;
517  PFN_NUMBER count;
518  PFN_NUMBER Page;
519  PFN_NUMBER LastPage;
520  PFN_NUMBER PageFrameIndex;
521  PFN_NUMBER LastPageFrameIndex;
522 
523  //
524  // Examine the physical memory block to see whether large pages should
525  // be enabled. The key point is that all the physical pages within a
526  // given large page range must have the same cache attributes (MmCached)
527  // in order to maintain TB coherency. This can be done provided all
528  // the pages within the large page range represent real RAM (as described
529  // by the loader) so that memory management can control it. If any
530  // portion of the large page range is not RAM, it is possible that it
531  // may get used as noncached or writecombined device memory and
532  // therefore large pages cannot be used.
533  //
534 
535  if (MxMapLargePages == 0) {
536  return;
537  }
538 
539  for (i = 0; i < MiLargeVaRangeIndex; i += 1) {
540  PointerPte = MiGetPteAddress (MiLargeVaRanges[i].VirtualAddress);
541  ASSERT (PointerPte->u.Hard.Valid == 1);
542  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
543 
544  PointerPte = MiGetPteAddress (MiLargeVaRanges[i].EndVirtualAddress);
545  ASSERT (PointerPte->u.Hard.Valid == 1);
546  LastPageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
547 
548  //
549  // Round the start down to a page directory boundary and the end to
550  // the last page directory entry before the next boundary.
551  //
552 
553  PageFrameIndex &= ~(MM_PFN_MAPPED_BY_PDE - 1);
554  LastPageFrameIndex |= (MM_PFN_MAPPED_BY_PDE - 1);
555 
556  EntryFound = FALSE;
557 
558  j = 0;
559  do {
560 
561  count = MmPhysicalMemoryBlock->Run[j].PageCount;
562  Page = MmPhysicalMemoryBlock->Run[j].BasePage;
563 
564  LastPage = Page + count;
565 
566  if ((PageFrameIndex >= Page) && (LastPageFrameIndex < LastPage)) {
567  EntryFound = TRUE;
568  break;
569  }
570 
571  j += 1;
572 
573  } while (j != MmPhysicalMemoryBlock->NumberOfRuns);
574 
575  if (EntryFound == FALSE) {
576 
577  //
578  // No entry was found that completely spans this large page range.
579  // Zero it so this range will not be converted into large pages
580  // later.
581  //
582 
583  DbgPrintEx (DPFLTR_MM_ID, DPFLTR_WARNING_LEVEL,
584  "MM: Loader/HAL memory block indicates large pages cannot be used\n");
585 
586  MiLargeVaRanges[i].VirtualAddress = NULL;
587 
588  //
589  // Don't use large pages for anything if any individual range
590  // could not be used. This is because 2 separate ranges may
591  // share a straddling large page. If the first range was unable
592  // to use large pages, but the second one does ... then only part
593  // of the first range will get large pages if we enable large
594  // pages for the second range. This would be very bad as we use
595  // the MI_IS_PHYSICAL macro everywhere and assume the entire
596  // range is in or out, so disable all large pages here instead.
597  //
598 
599  while (i != 0) {
600 
601  i -= 1;
602 
603  if (MiLargeVaRanges[i].VirtualAddress != NULL) {
604 
605  PointerPte = MiGetPteAddress (MiLargeVaRanges[i].VirtualAddress);
606  ASSERT (PointerPte->u.Hard.Valid == 1);
607  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
608 
609  PointerPte = MiGetPteAddress (MiLargeVaRanges[i].EndVirtualAddress);
610  ASSERT (PointerPte->u.Hard.Valid == 1);
611  LastPageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (PointerPte);
612 
613  //
614  // Round the start down to a page directory boundary and
615  // the end to the last page directory entry before the
616  // next boundary.
617  //
618 
619  PageFrameIndex &= ~(MM_PFN_MAPPED_BY_PDE - 1);
620  LastPageFrameIndex |= (MM_PFN_MAPPED_BY_PDE - 1);
621 
622  MiRemoveCachedRange (PageFrameIndex, LastPageFrameIndex);
623  }
624  }
625 
626  MiLargeVaRangeIndex = 0;
627  break;
628  }
629  else {
630  MiAddCachedRange (PageFrameIndex, LastPageFrameIndex);
631  }
632  }
633 }

727 
728 /*++
729 
730 Routine Description:
731 
732  This function computes the number of bytes needed to span the initial
733  nonpaged pool and PFN database plus the color arrays. It rounds this up
734  to a large page boundary and carves the memory from the free descriptor.
735 
736  If the physical memory is too sparse to use large pages for this, then
737  fall back to using small pages.
738 
739 Arguments:
740 
741  None.
742 
743 Return Value:
744 
745  None.
746 
747 Environment:
748 
749  Kernel mode, INIT only.
750 
751 --*/
752 
753 {
754  PFN_NUMBER i;
755  PFN_NUMBER BasePage;
756  PFN_NUMBER LastPage;
757  UCHAR Associativity;
758  SIZE_T NumberOfBytes;
759  SIZE_T PfnAllocation;
760  SIZE_T MaximumNonPagedPoolInBytesLimit;
761 
762  MaximumNonPagedPoolInBytesLimit = 0;
763 
764  //
765  // Non-paged pool comprises 2 chunks. The initial nonpaged pool grows
766  // up and the expansion nonpaged pool expands downward.
767  //
768  // Initial non-paged pool is constructed so virtual addresses
769  // are also physically contiguous.
770  //
771 
772  if ((MmSizeOfNonPagedPoolInBytes >> PAGE_SHIFT) >
773  (7 * (MxTotalFreePages >> 3))) {
774 
775  //
776  // More than 7/8 of memory allocated to nonpagedpool, reset to 0.
777  //
778 
779  MmSizeOfNonPagedPoolInBytes = 0;
780  }
781 
782  if (MmSizeOfNonPagedPoolInBytes < MmMinimumNonPagedPoolSize) {
783 
784  //
785  // Calculate the size of nonpaged pool.
786  // Use the minimum size, then for every MB above 16mb add extra pages.
787  //
788 
789  MmSizeOfNonPagedPoolInBytes = MmMinimumNonPagedPoolSize;
790 
791  MmSizeOfNonPagedPoolInBytes +=
792  ((MxTotalFreePages - _x16mbnp)/_x1mbnp) *
793  MmMinAdditionNonPagedPoolPerMb;
794  }
795 
796  if (MmSizeOfNonPagedPoolInBytes > MM_MAX_INITIAL_NONPAGED_POOL) {
797  MmSizeOfNonPagedPoolInBytes = MM_MAX_INITIAL_NONPAGED_POOL;
798  }
799 
800  //
801  // If the registry specifies a total nonpaged pool percentage cap, enforce
802  // it here.
803  //
804 
805  if (MmMaximumNonPagedPoolPercent != 0) {
806 
807  if (MmMaximumNonPagedPoolPercent < 5) {
808  MmMaximumNonPagedPoolPercent = 5;
809  }
810  else if (MmMaximumNonPagedPoolPercent > 80) {
811  MmMaximumNonPagedPoolPercent = 80;
812  }
813 
814  //
815  // Use the registry-expressed percentage value.
816  //
817 
818  MaximumNonPagedPoolInBytesLimit =
