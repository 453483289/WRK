139 //
140 // This defines the data structure that is used to pass a handle from
141 // um to km. In 32bit code a handle has 32bits and in 64bit code a handle
142 // has 64 bits and both call into the kernel which is 64bits. In order to
143 // ensure that the data structures compile to the same size on 32 and 64
144 // bit systems we define the union with a dummy 64bit value so the field is
145 // forced to be 64 bits in all code. Note that the object manager always
146 // ignores the top 32bits of the handle in order to support 32 bit code
147 // that only maintains 32 bit handles
148 //
149 typedef union
150 {
151  HANDLE Handle;
152  ULONG64 Handle64;
153  ULONG32 Handle32;
154 } HANDLE3264, *PHANDLE3264;
