343 NtQueryDirectoryObject (
344  __in HANDLE DirectoryHandle,
345  __out_bcount_opt(Length) PVOID Buffer,
346  __in ULONG Length,
347  __in BOOLEAN ReturnSingleEntry,
348  __in BOOLEAN RestartScan,
349  __inout PULONG Context,
350  __out_opt PULONG ReturnLength
351  )
352 
353 /*++
354 
355 Routine Description:
356 
357  This function returns information regarding a specified object
358  directory.
359 
360 Arguments:
361 
362  DirectoryHandle - Supplies a handle to the directory being queried
363 
364  Buffer - Supplies the output buffer to receive the directory
365  information. On return this contains one or more OBJECT DIRECTORY
366  INFORMATION structures, the last one being null. And then this is
367  followed by the string names for the directory entries.
368 
369  Length - Supplies the length, in bytes, of the user supplied output
370  buffer
371 
372  ReturnSingleEntry - Indicates if this routine should just return
373  one entry in the directory
374 
375  RestartScan - Indicates if we are to restart the scan or continue
376  relative to the enumeration context passed in as the next
377  parameter
378 
379  Context - Supplies an enumeration context that must be resupplied
380  to this routine on subsequent calls to keep the enumeration
381  in sync
382 
383  ReturnLength - Optionally receives the length, in bytes, that this
384  routine has stuffed into the output buffer
385 
386 Return Value:
387 
388  An appropriate status value.
389 
390 --*/
391 
392 {
393  POBJECT_DIRECTORY Directory;
394  POBJECT_DIRECTORY_ENTRY DirectoryEntry;
395  POBJECT_HEADER ObjectHeader;
396  POBJECT_HEADER_NAME_INFO NameInfo;
397  UNICODE_STRING ObjectName;
398  POBJECT_DIRECTORY_INFORMATION DirInfo;
399  PWCH NameBuffer;
400  KPROCESSOR_MODE PreviousMode;
401  NTSTATUS Status;
402  ULONG Bucket, EntryNumber, CapturedContext;
403  ULONG TotalLengthNeeded, LengthNeeded, EntriesFound;
404  PCHAR TempBuffer;
405  OBP_LOOKUP_CONTEXT LookupContext;
406 
407  PAGED_CODE();
408 
409  ObpValidateIrql( "NtQueryDirectoryObject" );
410 
411  ObpInitializeLookupContext( &LookupContext );
412 
413  //
414  // Get previous processor mode and probe output arguments if necessary.
415  //
416 
417  PreviousMode = KeGetPreviousMode();
418 
419  if (PreviousMode != KernelMode) {
420 
421  try {
422 
423  ProbeForWrite( Buffer, Length, sizeof( WCHAR ) );
424  ProbeForWriteUlong( Context );
425 
426  if (ARGUMENT_PRESENT( ReturnLength )) {
427 
428  ProbeForWriteUlong( ReturnLength );
429  }
430 
431  if (RestartScan) {
432 
433  CapturedContext = 0;
434 
435  } else {
436 
437  CapturedContext = *Context;
438  }
439 
440  } except( EXCEPTION_EXECUTE_HANDLER ) {
441 
442  return( GetExceptionCode() );
443  }
444 
445  } else {
446 
447  if (RestartScan) {
448 
449  CapturedContext = 0;
450 
451  } else {
452 
453  CapturedContext = *Context;
454  }
455  }
456 
457  //
458  // Test for 64 bit if Length + sizeof( OBJECT_DIRECTORY_INFORMATION ) is less than Length
459  // Return STATUS_INVALID_PARAMETER if there is an overflow
460  //
461 
462  if (ObpIsOverflow( Length, sizeof( OBJECT_DIRECTORY_INFORMATION ))) {
463 
464  return( STATUS_INVALID_PARAMETER );
465  }
466 
467  //
468  // Allocate space for a temporary work buffer, make sure we got it,
469  // and then zero it out. Make sure the buffer is large enough to
470  // hold at least one dir info record. This will make the logic work
471  // better when the a bad length is passed in.
472  //
473 
474  TempBuffer = ExAllocatePoolWithQuotaTag( PagedPool | POOL_QUOTA_FAIL_INSTEAD_OF_RAISE,
475  Length + sizeof( OBJECT_DIRECTORY_INFORMATION ),
476  'mNbO' );
477 
478  if (TempBuffer == NULL) {
479 
480  return( STATUS_INSUFFICIENT_RESOURCES );
481  }
482 
483  RtlZeroMemory( TempBuffer, Length );
484 
485  //
486  // Reference the directory object
487  //
488 
489  Status = ObReferenceObjectByHandle( DirectoryHandle,
490  DIRECTORY_QUERY,
491  ObpDirectoryObjectType,
492  PreviousMode,
493  (PVOID *)&Directory,
494  NULL );
495 
496  if (!NT_SUCCESS( Status )) {
497 
498  ExFreePool( TempBuffer );
499 
500  return( Status );
501  }
502 
503  //
504  // Lock down the directory structures for the life of this
505  // procedure
506  //
507 
508  ObpLockDirectoryShared(Directory, &LookupContext);
509 
510  //
511  // DirInfo is used to march through the output buffer filling
512  // in directory information. We'll start off by making sure
513  // there is room for a NULL entry at end.
514  //
515 
516  DirInfo = (POBJECT_DIRECTORY_INFORMATION)TempBuffer;
517 
518  TotalLengthNeeded = sizeof( *DirInfo );
519 
520  //
521  // Keep track of the number of entries found and actual
522  // entry that we are processing
523  //
524 
525  EntryNumber = 0;
526  EntriesFound = 0;
527 
528  //
529  // By default we'll say there are no more entries until the
530  // following loop put in some data
531  //
532 
533  Status = STATUS_NO_MORE_ENTRIES;
534 
535  //
536  // Our outer loop processes each hash bucket in the directory object
537  //
538 
539  for (Bucket=0; Bucket<NUMBER_HASH_BUCKETS; Bucket++) {
540 
541  DirectoryEntry = Directory->HashBuckets[ Bucket ];
542 
543  //
544  // For this hash bucket we'll zip through its list of entries.
545  // This is a singly linked list so when the next pointer is null
546  // (i.e., false) we at the end of the hash list
547  //
548 
549  while (DirectoryEntry) {
550 
551  //
552  // The captured context is simply the entry count unless the
553  // user specified otherwise we start at zero, which means
554  // the first entry is always returned in the enumeration.
555  // If we have an match based on the entry index then we
556  // process this entry. We bump the captured context further
557  // done in the code.
558  //
559 
560  if (CapturedContext == EntryNumber++) {
561 
562  //
563  // For this directory entry we'll get a pointer to the
564  // object body and see if it has an object name. If it
565  // doesn't have a name then we'll give it an empty name.
566  //
567 
568  ObjectHeader = OBJECT_TO_OBJECT_HEADER( DirectoryEntry->Object );
569  NameInfo = OBJECT_HEADER_TO_NAME_INFO( ObjectHeader );
570 
571  if (NameInfo != NULL) {
572 
573  ObjectName = NameInfo->Name;
574 
575  } else {
576 
577  RtlInitUnicodeString( &ObjectName, NULL );
578  }
579 
580  //
581  // Now compute the length needed for this entry. This would
582  // be the size of the object directory information record,
583  // plus the size of the object name and object type name both
584  // null terminated.
585  //
586 
587  LengthNeeded = sizeof( *DirInfo ) +
588  ObjectName.Length + sizeof( UNICODE_NULL ) +
589  ObjectHeader->Type->Name.Length + sizeof( UNICODE_NULL );
590 
591  //
592  // If there isn't enough room then take the following error
593  // path. If the user wanted a single entry then tell the
594  // caller what length is really needed and say the buffer was
595  // too small. Otherwise the user wanted multiple entries,
596  // so we'll just say there are more entries in the directory.
597  // In both cases we drop down the entry number because we
598  // weren't able to fit it in on this call
599  //
600 
601  if ((TotalLengthNeeded + LengthNeeded) > Length) {
602 
603  if (ReturnSingleEntry) {
604 
605  TotalLengthNeeded += LengthNeeded;
606 
607  Status = STATUS_BUFFER_TOO_SMALL;
608 
609  } else {
610 
611  Status = STATUS_MORE_ENTRIES;
612  }
613 
614  EntryNumber -= 1;
615  goto querydone;
616  }
617 
618  //
619  // The information will fit in the buffer. So now fill
620  // in the output buffer. We temporarily put in pointers
621  // to the name buffer as stored in the object and object
622  // type. We copy the data buffer to the user buffer
623  // right before we return to the caller
624  //
625 
626  try {
627 
628  DirInfo->Name.Length = ObjectName.Length;
629  DirInfo->Name.MaximumLength = (USHORT)(ObjectName.Length+sizeof( UNICODE_NULL ));
630  DirInfo->Name.Buffer = ObjectName.Buffer;
631 
632  DirInfo->TypeName.Length = ObjectHeader->Type->Name.Length;
633  DirInfo->TypeName.MaximumLength = (USHORT)(ObjectHeader->Type->Name.Length+sizeof( UNICODE_NULL ));
634  DirInfo->TypeName.Buffer = ObjectHeader->Type->Name.Buffer;
635 
636  Status = STATUS_SUCCESS;
637 
638  } except( EXCEPTION_EXECUTE_HANDLER ) {
639 
640  Status = GetExceptionCode();
641  }
642 
643  if (!NT_SUCCESS( Status )) {
644 
645  goto querydone;
646  }
647 
648  //
649  // Update the total number of bytes needed in this query.
650  // Push the dir info pointer to the next output location,
651  // and indicate how many entries we've processed
652  //
653  //
654 
655  TotalLengthNeeded += LengthNeeded;
656 
657  DirInfo++;
658  EntriesFound++;
659 
660  //
661  // If we are to return only one entry then move on to the
662  // post processing phase, otherwise indicate that we're
663  // processing the next entry and go back to the top of
664  // the inner loop
665  //
666 
667  if (ReturnSingleEntry) {
668 
669  goto querydone;
670 
671  } else {
672 
673  //
674  // Bump the captured context by one entry.
675  //
676 
677  CapturedContext++;
678  }
679  }
680 
681  //
682  // Get the next directory entry from the singly linked hash
683  // bucket chain
684  //
685 
686  DirectoryEntry = DirectoryEntry->ChainLink;
687  }
688  }
689 
690  //
691  // At this point we've processed the directory entries and the first
692  // part of the output buffer now contains a bunch of object directory
693  // information records, but the pointers in them refer to the wrong
694  // copies. So now we have some fixup to do.
695  //
696 
697 querydone:
698 
699  try {
700 
701  //
702  // We'll only do this post processing if we've been successful
703  // so far. Note that this means we could be returning in the
704  // user's output buffer system address that are meaningless, but
705  // then getting back an error status should tell the caller to
706  // forget about everything in the output buffer. Given back
707  // a system address also isn't harmful because there is nothing
708  // that the user can really do with it.
709  //
710 
711  if (NT_SUCCESS( Status )) {
712 
713  //
714  // Null terminate the string of object directory information
715  // records and point to where the actual names will go
716  //
717 
718  RtlZeroMemory( DirInfo, sizeof( *DirInfo ));
719 
720  DirInfo++;
721 
722  NameBuffer = (PWCH)DirInfo;
723 
724  //
725  // Now for every entry that we've put in the output buffer
726  // DirInfo will point to the entry and EntriesFound kept the
727  // count. Note that we are guaranteed space because of
728  // the math we did earlier in computing TotalLengthNeeded.
729  //
730 
731  DirInfo = (POBJECT_DIRECTORY_INFORMATION)TempBuffer;
732 
733  while (EntriesFound--) {
734 
735  //
736  // Copy over the object name, set the dir info pointer into
737  // the user's buffer, then null terminate the string. Note
738  // that we are really copying the data into our temp buffer
739  // but the pointer fix up is for the user's buffer which
740  // we'll copy into right after this loop.
741  //
742 
743  RtlCopyMemory( NameBuffer,
744  DirInfo->Name.Buffer,
745  DirInfo->Name.Length );
746 
747  DirInfo->Name.Buffer = (PVOID)((ULONG_PTR)Buffer + ((ULONG_PTR)NameBuffer - (ULONG_PTR)TempBuffer));
748  NameBuffer = (PWCH)((ULONG_PTR)NameBuffer + DirInfo->Name.Length);
749  *NameBuffer++ = UNICODE_NULL;
750 
751  //
752  // Do the same copy with the object type name
753  //
754 
755  RtlCopyMemory( NameBuffer,
756  DirInfo->TypeName.Buffer,
757  DirInfo->TypeName.Length );
758 
759  DirInfo->TypeName.Buffer = (PVOID)((ULONG_PTR)Buffer + ((ULONG_PTR)NameBuffer - (ULONG_PTR)TempBuffer));
760  NameBuffer = (PWCH)((ULONG_PTR)NameBuffer + DirInfo->TypeName.Length);
761  *NameBuffer++ = UNICODE_NULL;
762 
763  //
764  // Move on to the next dir info record
765  //
766 
767  DirInfo++;
768  }
769 
770  //
771  // Set the enumeration context to the entry number of the next
772  // entry to return.
773  //
774 
775  *Context = EntryNumber;
776  }
777 
778  //
779  // Copy over the results from our temp buffer to the users buffer.
780  // But adjust the amount copied just in case the total length needed
781  // exceeds the length we allocated.
782  //
783 
784  RtlCopyMemory( Buffer,
785  TempBuffer,
786  (TotalLengthNeeded <= Length ? TotalLengthNeeded : Length) );
787 
788  //
789  // In all cases we'll tell the caller how much space if really needed
790  // provided the user asked for this information
791  //
792 
793  if (ARGUMENT_PRESENT( ReturnLength )) {
794 
795  *ReturnLength = TotalLengthNeeded;
796  }
797 
798  } except( EXCEPTION_EXECUTE_HANDLER ) {
799 
800  //
801  // Fall through, since we do not want to undo what we have done.
802  //
803  }
804 
805  //
806  // Unlock the directory structures, dereference the directory object,
807  // free up our temp buffer, and return to our caller
808  //
809 
810  ObpUnlockDirectory( Directory, &LookupContext);
811 
812  ObDereferenceObject( Directory );
813 
814  ExFreePool( TempBuffer );
815 
816  return( Status );
817 }

830 /*++
831 
832 Routine Description:
833 
834  This routine will lookup a single directory entry in a given directory.
835  If it founds an object into that directory with the give name,
836  that object will be referenced and the name will be referenced too, in order
837  to prevent them going away when the directory is unlocked.
838  The referenced object is saved in the LookupContext, and the references will be released
839  at the next lookup, or when ObpReleaseLookupContext is called.
840 
841 Arguments:
842 
843  Directory - Supplies the directory being searched
844 
845  Name - Supplies the name of entry we're looking for
846 
847  Attributes - Indicates if the lookup should be case insensitive
848  or not
849 
850  SearchShadow - If TRUE, and the object name is not found in the current directory,
851  it will search the object into the shadow directory.
852 
853  LookupContext - The lookup context for this call. This structure must be initialized
854  before calling first time ObpLookupDirectoryEntry.
855 
856 Return Value:
857 
858  Returns a pointer to the corresponding object body if found and NULL
859  otherwise.
860 --*/
861 
862 {
863  POBJECT_DIRECTORY_ENTRY *HeadDirectoryEntry;
864  POBJECT_DIRECTORY_ENTRY DirectoryEntry;
865  POBJECT_HEADER ObjectHeader;
866  POBJECT_HEADER_NAME_INFO NameInfo;
867  PWCH Buffer;
868  ULONG Wchar;
869  ULONG HashIndex;
870  ULONG HashValue;
871  ULONG WcharLength;
872  BOOLEAN CaseInSensitive;
873  POBJECT_DIRECTORY_ENTRY *LookupBucket;
874  PVOID Object = NULL;
875 
876  PAGED_CODE();
877 
878  if (ObpLUIDDeviceMapsEnabled == 0) {
879 
880  SearchShadow = FALSE; // Disable global devmap search
881  }
882 
883  //
884  // The caller needs to specify both a directory and a name otherwise
885  // we can't process the request
886  //
887
