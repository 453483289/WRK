342 /*++
343 
344 Routine Description:
345 
346  This function swaps the address space to another process by flushing the
347  the translation buffer and establishings a new directory table base. It
348  also swaps the I/O permission map to the new process.
349 
350  N.B. There is code similar to this code in swap context.
351 
352  N.B. This code is executed at DPC level.
353 
354 Arguments:
355 
356  NewProcess - Supplies a pointer to the new process object.
357 
358  Oldprocess - Supplies a pointer to the old process object.
359 
360 Return Value:
361 
362  None.
363 
364 --*/
365 
366 {
367 
368  //
369  // Clear the processor bit in the old process.
370  //
371 
372 #if !defined(NT_UP)
373 
374  PKPRCB Prcb;
375  KAFFINITY SetMember;
376 
377  Prcb = KeGetCurrentPrcb();
378  SetMember = Prcb->SetMember;
379  InterlockedXor64((LONG64 volatile *)&OldProcess->ActiveProcessors, SetMember);
380 
381  ASSERT((OldProcess->ActiveProcessors & SetMember) == 0);
382 
383  //
384  // Set the processor bit in the new process.
385  //
386 
387  InterlockedXor64((LONG64 volatile *)&NewProcess->ActiveProcessors, SetMember);
388 
389  ASSERT((NewProcess->ActiveProcessors & SetMember) != 0);
390 
391 #endif
392 
393  //
394  // Load the new directory table base.
395  //
396 
397  WriteCR3(NewProcess->DirectoryTableBase[0]);
398 
399 #if defined(NT_UP)
400 
401  UNREFERENCED_PARAMETER(OldProcess);
402 
403 #endif // !defined(NT_UP)
404 
405  return TRUE;
406 }

414  IN PKEXCEPTION_FRAME ExceptionFrame

416 
417 /*++
418 
419 Routine Description:
420 
421  This function checks for a pending freeze execution request from
422  another processor. If such a request is pending then execution is
423  frozen.
424 
425 Arguments:
426 
427  TrapFrame - Supplies a pointer to a trap frame.
428 
429  ExceptionFrame - Supplies a pointer to an exception frame.
430 
431 Return Value:
432 
433  None.
434 
435 --*/
436 
437 {
438 
439  PKPRCB Prcb;
440 
441  Prcb = KeGetCurrentPrcb();
442  if (Prcb->IpiFrozen == TARGET_FREEZE) {
443  KiFreezeTargetExecution(TrapFrame, ExceptionFrame);
444  if (KiResumeForReboot != FALSE) {
445 
446  //
447  // If previous mode was user, then set the code and stack
448  // selectors and the stack pointer to valid kernel values.
449  //
450 
451  if (TrapFrame->SegCs != KGDT64_R0_CODE) {
452  TrapFrame->SegCs = KGDT64_R0_CODE;
453  TrapFrame->SegSs = KGDT64_R0_DATA;
454  TrapFrame->Rsp = Prcb->RspBase;
455  }
456 
457  TrapFrame->Rip = (ULONG64)KiWaitForReboot;
458 
459  }
460  return TRUE;
461 
462  } else {
463  return FALSE;
464  }
465 }
