64 /*++
65 
66 Routine Description:
67 
68  This function creates a symbolic link object, sets it initial value to
69  value specified in the LinkTarget parameter, and opens a handle to the
70  object with the specified desired access.
71 
72 Arguments:
73 
74  LinkHandle - Supplies a pointer to a variable that will receive the
75  symbolic link object handle.
76 
77  DesiredAccess - Supplies the desired types of access for the symbolic link
78  object.
79 
80  ObjectAttributes - Supplies a pointer to an object attributes structure.
81 
82  LinkTarget - Supplies the target name for the symbolic link object.
83 
84 Return Value:
85 
86  An appropriate status value
87 
88 --*/
89 
90 {
91  KPROCESSOR_MODE PreviousMode;
92  NTSTATUS Status;
93  POBJECT_SYMBOLIC_LINK SymbolicLink;
94  PVOID Object;
95  HANDLE Handle;
96  UNICODE_STRING CapturedLinkTarget;
97 
98  PAGED_CODE();
99 
100  //
101  // Get previous processor mode and probe output arguments if necessary.
102  //
103 
104  PreviousMode = KeGetPreviousMode();
105 
106  if (PreviousMode != KernelMode) {
107 
108  try {
109 
110  ProbeForReadSmallStructure( ObjectAttributes,
111  sizeof( OBJECT_ATTRIBUTES ),
112  sizeof( ULONG ));
113 
114  ProbeForReadSmallStructure( LinkTarget,
115  sizeof( *LinkTarget ),
116  sizeof( UCHAR ));
117 
118  CapturedLinkTarget = *LinkTarget;
119 
120  ProbeForRead( CapturedLinkTarget.Buffer,
121  CapturedLinkTarget.MaximumLength,
122  sizeof( UCHAR ));
123 
124  ProbeForWriteHandle( LinkHandle );
125 
126  } except( EXCEPTION_EXECUTE_HANDLER ) {
127 
128  return( GetExceptionCode() );
129  }
130 
131  } else {
132 
133  CapturedLinkTarget = *LinkTarget;
134  }
135 
136  //
137  // Check if there is an odd MaximumLength
138  //
139 
140  if (CapturedLinkTarget.MaximumLength % sizeof( WCHAR )) {
141 
142  //
143  // Round down the MaximumLength to a valid even size
144  //
145 
146  CapturedLinkTarget.MaximumLength = (CapturedLinkTarget.MaximumLength / sizeof( WCHAR )) * sizeof( WCHAR );
147  }
148 
149  //
150  // Error if link target name length is odd, the length is greater than
151  // the maximum length, or zero and creating.
152  //
153 
154  if ((CapturedLinkTarget.MaximumLength == 0) ||
155  (CapturedLinkTarget.Length > CapturedLinkTarget.MaximumLength) ||
156  (CapturedLinkTarget.Length % sizeof( WCHAR ))) {
157 
158  KdPrint(( "OB: Invalid symbolic link target - %wZ\n", &CapturedLinkTarget ));
159 
160  return( STATUS_INVALID_PARAMETER );
161  }
162 
163  //
164  // Create the symbolic link object
165  //
166 
167  Status = ObCreateObject( PreviousMode,
168  ObpSymbolicLinkObjectType,
169  ObjectAttributes,
170  PreviousMode,
171  NULL,
172  sizeof( *SymbolicLink ),
173  0,
174  0,
175  (PVOID *)&SymbolicLink );
176 
177  if (!NT_SUCCESS( Status )) {
178 
179  return( Status );
180  }
181 
182  //
183  // Fill in symbolic link object with link target name string
184  //
185 
186  KeQuerySystemTime( &SymbolicLink->CreationTime );
187 
188  SymbolicLink->DosDeviceDriveIndex = 0;
189  SymbolicLink->LinkTargetObject = NULL;
190 
191  RtlInitUnicodeString( &SymbolicLink->LinkTargetRemaining, NULL );
192 
193  SymbolicLink->LinkTarget.MaximumLength = CapturedLinkTarget.MaximumLength;
194  SymbolicLink->LinkTarget.Length = CapturedLinkTarget.Length;
195  SymbolicLink->LinkTarget.Buffer = (PWCH)ExAllocatePoolWithTag( PagedPool,
196  CapturedLinkTarget.MaximumLength,
197  'tmyS' );
198 
199  if (SymbolicLink->LinkTarget.Buffer == NULL) {
200 
201  ObDereferenceObject( SymbolicLink );
202 
203  return STATUS_NO_MEMORY;
204  }
205 
206  try {
207 
208  RtlCopyMemory( SymbolicLink->LinkTarget.Buffer,
209  CapturedLinkTarget.Buffer,
210  CapturedLinkTarget.MaximumLength );
211 
212  } except( EXCEPTION_EXECUTE_HANDLER ) {
213 
214  ObDereferenceObject( SymbolicLink );
215 
216  return( GetExceptionCode() );
217  }
218 
219  //
220  // Insert symbolic link object in the current processes object table,
221  // set symbolic link handle value and return status.
222  //
223 
224  Status = ObInsertObject( SymbolicLink,
225  NULL,
226  DesiredAccess,
227  0,
228  (PVOID *)&Object,
229  &Handle );
230 
231  try {
232 
233  *LinkHandle = Handle;
234 
235  } except( EXCEPTION_EXECUTE_HANDLER ) {
236 
237  //
238  // Fall through, since we do not want to undo what we have done.
239  //
240  }
241 
242  return( Status );
243 }

330 NTSTATUS
331 NtQuerySymbolicLinkObject (
332  __in HANDLE LinkHandle,
333  __inout PUNICODE_STRING LinkTarget,
334  __out_opt PULONG ReturnedLength
335  )
336 
337 /*++
338 
339 Routine Description:
340 
341  This function queries the state of a symbolic link object and returns the
342  requested information in the specified record structure.
343 
344 Arguments:
345 
346  LinkHandle - Supplies a handle to a symbolic link object. This handle
347  must have SYMBOLIC_LINK_QUERY access granted.
348 
349  LinkTarget - Supplies a pointer to a record that is to receive the
350  target name of the symbolic link object.
351 
352  ReturnedLength - Optionally receives the maximum length, in bytes, of
353  the link target on return
354 
355 Return Value:
356 
357  An appropriate status value
358 
359 --*/
360 
361 {
362  KPROCESSOR_MODE PreviousMode;
363  NTSTATUS Status;
364  POBJECT_SYMBOLIC_LINK SymbolicLink;
365  UNICODE_STRING CapturedLinkTarget;
366 
367  //
368  // Get previous processor mode and probe output arguments if necessary.
369  //
370 
371  PAGED_CODE();
372 
373  PreviousMode = KeGetPreviousMode();
374 
375  if (PreviousMode != KernelMode) {
376 
377  try {
378 
379  ProbeForReadSmallStructure( LinkTarget,
380  sizeof( *LinkTarget ),
381  sizeof( WCHAR ) );
382 
383  ProbeForWriteUshort( &LinkTarget->Length );
384 
385  ProbeForWriteUshort( &LinkTarget->MaximumLength );
386 
387  CapturedLinkTarget = *LinkTarget;
388 
389  ProbeForWrite( CapturedLinkTarget.Buffer,
390  CapturedLinkTarget.MaximumLength,
391  sizeof( UCHAR ) );
392 
393  if (ARGUMENT_PRESENT( ReturnedLength )) {
394 
395  ProbeForWriteUlong( ReturnedLength );
396  }
397 
398  } except( EXCEPTION_EXECUTE_HANDLER ) {
399 
400  return( GetExceptionCode() );
401  }
402 
403  } else {
404 
405  CapturedLinkTarget = *LinkTarget;
406  }
407 
408  //
409  // Reference symbolic link object by handle, read current state, deference
410  // symbolic link object, fill in target name structure and return service
411  // status.
412  //
413 
414  Status = ObReferenceObjectByHandle( LinkHandle,
415  SYMBOLIC_LINK_QUERY,
416  ObpSymbolicLinkObjectType,
417  PreviousMode,
418  (PVOID *)&SymbolicLink,
419  NULL );
420 
421  if (NT_SUCCESS( Status )) {
422 
423  POBJECT_HEADER ObjectHeader;
424 
425  ObjectHeader = OBJECT_TO_OBJECT_HEADER( SymbolicLink );
426 
427  ObpLockObject( ObjectHeader );
428 
429  //
430  // If the caller wants a return length and what we found can easily
431  // fit in the output buffer then we copy into the output buffer all
432  // the bytes from the link.
433  //
434  // If the caller did not want a return length and we found can still
435  // easily fit in the output buffer then copy over the bytes that just
436  // make up the string and nothing extra
437  //
438 
439  if ((ARGUMENT_PRESENT( ReturnedLength ) &&
440  (SymbolicLink->LinkTarget.MaximumLength <= CapturedLinkTarget.MaximumLength))
441 
442  ||
443 
444  (!ARGUMENT_PRESENT( ReturnedLength ) &&
445  (SymbolicLink->LinkTarget.Length <= CapturedLinkTarget.MaximumLength)) ) {
446 
447  try {
448 
449  RtlCopyMemory( CapturedLinkTarget.Buffer,
450  SymbolicLink->LinkTarget.Buffer,
451  ARGUMENT_PRESENT( ReturnedLength ) ? SymbolicLink->LinkTarget.MaximumLength
452  : SymbolicLink->LinkTarget.Length );
453 
454  LinkTarget->Length = SymbolicLink->LinkTarget.Length;
455 
456  if (ARGUMENT_PRESENT( ReturnedLength )) {
457 
458  *ReturnedLength = SymbolicLink->LinkTarget.MaximumLength;
459  }
460 
461  } except( EXCEPTION_EXECUTE_HANDLER ) {
462 
463  //
464  // Fall through, since we do cannot undo what we have done.
465  //
466  }
467 
468  } else {
469 
470  //
471  // The output buffer is just too small for the link target, but
472  // we'll tell the user how much is needed if they asked for that
473  // return value
474  //
475 
476  if (ARGUMENT_PRESENT( ReturnedLength )) {
477 
478  try {
479 
480  *ReturnedLength = SymbolicLink->LinkTarget.MaximumLength;
481 
482  } except( EXCEPTION_EXECUTE_HANDLER ) {
483 
484  //
485  // Fall through, since we do cannot undo what we have done.
486  //
487  }
488  }
489 
490  Status = STATUS_BUFFER_TOO_SMALL;
491  }
492 
493  ObpUnlockObject( ObjectHeader );
494 
495  ObDereferenceObject( SymbolicLink );
496  }
497 
498  return( Status );
499 }

502 NTSTATUS
503 ObpParseSymbolicLink (
504  IN PVOID ParseObject,
505  IN PVOID ObjectType,
506  IN PACCESS_STATE AccessState,
507  IN KPROCESSOR_MODE AccessMode,
508  IN ULONG Attributes,
509  IN OUT PUNICODE_STRING CompleteName,
510  IN OUT PUNICODE_STRING RemainingName,
511  IN OUT PVOID Context OPTIONAL,
512  IN PSECURITY_QUALITY_OF_SERVICE SecurityQos OPTIONAL,
513  OUT PVOID *Object
514  )
515 
516 /*++
517 
518 Routine Description:
519 
520  This is the call back routine for parsing symbolic link objects. It is invoked
521  as part of ObpLookupObjectName
522 
523 Arguments:
524 
525  ParseObject - This will actually be a symbolic link object
526 
527  ObjectType - Specifies the type of the object to lookup
528 
529  AccessState - Current access state, describing already granted access
530  types, the privileges used to get them, and any access types yet to
531  be granted. The access masks may not contain any generic access
532  types.
533 
534  AccessMode - Specifies the callers processor mode
535 
536  Attributes - Specifies the attributes for the lookup (e.g., case
537  insensitive)
538 
539  CompleteName - Supplies a pointer to the complete name that we are trying
540  to open. On return this could be modified to fit the new reparse
541  buffer
542 
543  RemainingName - Supplies a pointer to the remaining name that we are
544  trying to open. On return this will point to what remains after
545  we processed the symbolic link.
546 
547  Context - Unused
548 
549  SecurityQos - Unused
550 
551  Object - Receives a pointer to the symbolic link object that we
552  resolve to
553 
554 Return Value:
555 
556  STATUS_REPARSE_OBJECT if the parse object is already a snapped
557  symbolic link meaning we've modified the remaining name and
558  and have returned the target object of the symbolic link
559 
560 
561  STATUS_REPARSE if the parse object has not been snapped. In this
562  case the Complete name has been modified with the link target
563  name added in front of the remaining name. The parameters
564  remaining name and object must now be ignored by the caller
565 
566  An appropriate error value
567 
568 --*/
569 
570 {
571  ULONG NewLength;
572  USHORT Length;
573  USHORT MaximumLength;
574  PWCHAR NewName, NewRemainingName;
575  ULONG InsertAmount;
576  NTSTATUS Status;
577  POBJECT_SYMBOLIC_LINK SymbolicLink;
578  PUNICODE_STRING LinkTargetName;
579 
580  UNREFERENCED_PARAMETER (Context);
581  UNREFERENCED_PARAMETER (AccessState);
582  UNREFERENCED_PARAMETER (SecurityQos);
583  UNREFERENCED_PARAMETER (Attributes);
584 
585  PAGED_CODE();
586 
587  //
588  // This routine needs to be synchonized with the delete symbolic link
589  // operation. Which uses the root directory mutex.
590  //
591 
592  *Object = NULL;
593 
594  //
595  // If there isn't any name remaining and the caller gave us
596  // an object type then we'll reference the parse object. If
597  // this is successful then that's the object we return. Otherwise
598  // if the status is anything but a type mismatch then we'll
599  // return that error status
600  //
601 
602  if (RemainingName->Length == 0) {
603 
604  if ( ObjectType ) {
605 
606  Status = ObReferenceObjectByPointer( ParseObject,
607  0,
608  ObjectType,
609  AccessMode );
610 
611  if (NT_SUCCESS( Status )) {
612 
613  *Object = ParseObject;
614 
615  return Status;
616 
617  } else if (Status != STATUS_OBJECT_TYPE_MISMATCH) {
618 
619  return Status;
620  }
621  }
622 
623  //
624  // If the remaining name does not start with a "\" then
625  // its is malformed and we'll call it a type mismatch
626  //
627 
628  } else if (*(RemainingName->Buffer) != OBJ_NAME_PATH_SEPARATOR) {
629 
630  return STATUS_OBJECT_TYPE_MISMATCH;
631  }
632 
633  //
634  // A symbolic link has been encountered. See if this link has been snapped
635  // to a particular object.
636  //
637 
638  SymbolicLink = (POBJECT_SYMBOLIC_LINK)ParseObject;
639 
640  if (SymbolicLink->LinkTargetObject != NULL) {
641 
642  //
643  // This is a snapped link. Get the remaining portion of the
644  // symbolic link target, if any.
645  //
646 
647  LinkTargetName = &SymbolicLink->LinkTargetRemaining;
648 
649  if (LinkTargetName->Length == 0) {
650 
651  //
652  // Remaining link target string is zero, so return to caller
653  // quickly with snapped object pointer and remaining object name
654  // which we haven't touched yet.
655  //
656 
657  *Object = SymbolicLink->LinkTargetObject;
658 
659  return STATUS_REPARSE_OBJECT;
660  }
661 
662  //
663  // We have a snapped symbolic link that has additional text.
664  // Insert that in front of the current remaining name, preserving
665  // and text between CompleteName and RemainingName
666  //
667 
668  InsertAmount = LinkTargetName->Length;
669 
670  if ((LinkTargetName->Buffer[ (InsertAmount / sizeof( WCHAR )) - 1 ] == OBJ_NAME_PATH_SEPARATOR)
671 
672  &&
673 
674  (*(RemainingName->Buffer) == OBJ_NAME_PATH_SEPARATOR)) {
675 
676  //
677  // Both the link target name ends in a "\" and the remaining
678  // starts with a "\" but we only need one when we're done
679  //
680 
681  InsertAmount -= sizeof( WCHAR );
682  }
683 
684  //
685  // We need to bias the difference between two
686  // pointers with * sizeof(wchar) because the difference is in wchar
687  // and we need the length in bytes
688  //
689 
690  NewLength = (ULONG)(((RemainingName->Buffer - CompleteName->Buffer) * sizeof( WCHAR )) +
691  InsertAmount +
692  RemainingName->Length);
693 
694  if (NewLength > 0xFFF0) {
695 
696  return STATUS_NAME_TOO_LONG;
697  }
698 
699  Length = (USHORT)NewLength;
700 
701  //
702  // Now check if the new computed length is too big for the input
703  // buffer containing the complete name
704  //
705 
706  if (CompleteName->MaximumLength <= Length) {
707 
708  //
709  // The new concatenated name is larger than the buffer supplied for
710  // the complete name. Allocate space for this new string
711  //
712 
713  MaximumLength = Length + sizeof( UNICODE_NULL );
714  NewName = ExAllocatePoolWithTag( OB_NAMESPACE_POOL_TYPE, MaximumLength, 'mNbO' );
715 
716  if (NewName == NULL) {
717 
718  return STATUS_INSUFFICIENT_RESOURCES;
719  }
720 
721  //
722  // Calculate the pointer within this buffer for the remaining
723  // name. This value has not been biased by the new link
724  // target name
725  //
726 
727  NewRemainingName = NewName + (RemainingName->Buffer - CompleteName->Buffer);
728 
729  //
730  // Copy over all the names that we've processed so far
731  //
732 
733  RtlCopyMemory( NewName,
734  CompleteName->Buffer,
735  ((RemainingName->Buffer - CompleteName->Buffer) * sizeof( WCHAR )));
736 
737  //
738  // If we have some remaining names then those over at the
739  // the location offset to hold the link target name
740  //
741 
742  if (RemainingName->Length != 0) {
743 
744  RtlCopyMemory( (PVOID)((PUCHAR)NewRemainingName + InsertAmount),
745  RemainingName->Buffer,
746  RemainingName->Length );
747  }
748 
749  //
750  // Now insert the link target name
751  //
752 
753  RtlCopyMemory( NewRemainingName, LinkTargetName->Buffer, InsertAmount );
754 
755  //
756  // Free the old complete name buffer and reset the input
757  // strings to use the new buffer
758  //
759 
760  ExFreePool( CompleteName->Buffer );
761 
762  CompleteName->Buffer = NewName;
763  CompleteName->Length = Length;
764  CompleteName->MaximumLength = MaximumLength;
765 
766  RemainingName->Buffer = NewRemainingName;
767  RemainingName->Length = Length - (USHORT)((PCHAR)NewRemainingName - (PCHAR)NewName);
768  RemainingName->MaximumLength = RemainingName->Length + sizeof( UNICODE_NULL );
769 
770  } else {
771 
772  //
773  // Insert extra text associated with this symbolic link name before
774  // existing remaining name, if any.
775  //
776  // First shove over the remaining name to make a hole for the
777  // link target name
778  //
779 
780  if (RemainingName->Length != 0) {
781 
782  RtlMoveMemory( (PVOID)((PUCHAR)RemainingName->Buffer + InsertAmount),
783  RemainingName->Buffer,
784  RemainingName->Length );
785  }
786 
787  //
788  // Now insert the link target name
789  //
790 
791  RtlCopyMemory( RemainingName->Buffer, LinkTargetName->Buffer, InsertAmount );
792 
793  //
794  // Adjust input strings to account for this inserted text
795  //
796 
797  CompleteName->Length = (USHORT)(CompleteName->Length + LinkTargetName->Length);
798 
799  RemainingName->Length = (USHORT)(RemainingName->Length + LinkTargetName->Length);
800  RemainingName->MaximumLength += RemainingName->Length + sizeof( UNICODE_NULL );
801 
802  CompleteName->Buffer[ CompleteName->Length / sizeof( WCHAR ) ] = UNICODE_NULL;
803  }
804 
805  //
806  // Return the object address associated with snapped symbolic link
807  // and the reparse object status code.
808  //
809 
810  *Object = SymbolicLink->LinkTargetObject;
811 
812  return STATUS_REPARSE_OBJECT;
813  }
814 
815  //
816  // The symbolic has not yet been snapped
817  //
818  // Compute the size of the new name and check if the name will
819  // fit in the existing complete name buffer.
820  //
821 
822  LinkTargetName = &SymbolicLink->LinkTarget;
823 
824  InsertAmount = LinkTargetName->Length;
825 
826  if ((InsertAmount != 0)
827  &&
828  (LinkTargetName->Buffer[ (InsertAmount / sizeof( WCHAR )) - 1 ] == OBJ_NAME_PATH_SEPARATOR)
829  &&
830  (RemainingName->Length != 0)
831  &&
832  (*(RemainingName->Buffer) == OBJ_NAME_PATH_SEPARATOR)) {
833 
834  //
835  // Both the link target name ends in a "\" and the remaining
836  // starts with a "\" but we only need one when we're done
837  //
838 
839  InsertAmount -= sizeof( WCHAR );
840  }
841 
842  NewLength = InsertAmount + RemainingName->Length;
843 
844  if (NewLength > 0xFFF0) {
845 
846  return STATUS_NAME_TOO_LONG;
847  }
848 
849  Length = (USHORT)NewLength;
850 
851  if (CompleteName->MaximumLength <= Length) {
852 
853  //
854  // The new concatenated name is larger than the buffer supplied for
855  // the complete name.
856  //
857 
858  MaximumLength = Length + sizeof( UNICODE_NULL );
859  NewName = ExAllocatePoolWithTag( OB_NAMESPACE_POOL_TYPE, MaximumLength, 'mNbO' );
860 
861  if (NewName == NULL) {
862 
863  return STATUS_INSUFFICIENT_RESOURCES;
864  }
865 
866  } else {
867 
868  MaximumLength = CompleteName->MaximumLength;
869  NewName = CompleteName->Buffer;
870  }
871 
872  //
873  // Concatenate the symbolic link name with the remaining name,
874  // if any. What this does is overwrite the front of the complete
875  // name up to the remaining name with the links target name
876  //
877 
878  if (RemainingName->Length != 0) {
879 
880  RtlMoveMemory( (PVOID)((PUCHAR)NewName + InsertAmount),
