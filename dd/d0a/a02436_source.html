81 /*++
82 
83 Routine Description:
84 
85  The specified i/o access map will be set to match the
86  definition specified by IoAccessMap (i.e. enable/disable
87  those ports) before the call returns. The change will take
88  effect on all processors.
89 
90  Ke386SetIoAccessMap does not give any process enhanced I/O
91  access, it merely defines a particular access map.
92 
93 Arguments:
94 
95  MapNumber - Number of access map to set. Map 0 is fixed.
96 
97  IoAccessMap - Pointer to bitvector (64K bits, 8K bytes) which
98  defines the specified access map. Must be in
99  non-paged pool.
100 
101 Return Value:
102 
103  TRUE if successful. FALSE if failure (attempt to set a map
104  which does not exist, attempt to set map 0)
105 
106 --*/
107 
108 {
109  MAPINFO MapInfo;
110 
111  //
112  // Reject illegal requests
113  //
114 
115  if ((MapNumber > IOPM_COUNT) || (MapNumber == IO_ACCESS_MAP_NONE)) {
116  return FALSE;
117  }
118 
119  MapInfo.MapSource = IoAccessMap;
120  MapInfo.MapNumber = MapNumber;
121  MapInfo.Process = KeGetCurrentThread()->ApcState.Process;
122 
123  KeGenericCallDpc (KiSetIoMap,
124  &MapInfo);
125 
126  return TRUE;
127 }

136 /*++
137 
138 Routine Description:
139 
140  copy the specified map into this processor's TSS.
141  This procedure runs at IPI level.
142 
143 Arguments:
144 
145  Dpc - DPC used to initiate this call
146  DeferredContext - Context
147  SystemArgument1 - System context, Used to signal completion of this call
148  SystemArgument2 - System context
149 
150 Return Value:
151 
152  none
153 
154 --*/
155 
156 {
157 
158  PKPROCESS CurrentProcess;
159  PKPCR Pcr;
160  PKPRCB Prcb;
161  PVOID pt;
162  PMAPINFO MapInfo;
163 
164  UNREFERENCED_PARAMETER (Dpc);
165  UNREFERENCED_PARAMETER (SystemArgument2);
166 
167  MapInfo = DeferredContext;
168 
169  //
170  // Copy the IOPM map and load the map for the current process.
171  // We only do this if the current process is running on this processor.
172  //
173 
174  Pcr = KiPcr ();
175  Prcb = Pcr->Prcb;
176  CurrentProcess = Prcb->CurrentThread->ApcState.Process;
177 
178  pt = &(Pcr->TSS->IoMaps[MapInfo->MapNumber-1].IoMap);
179  RtlCopyMemory (pt, MapInfo->MapSource, IOPM_SIZE);
180  Pcr->TSS->IoMapBase = CurrentProcess->IopmOffset;
181 
182  //
183  // Signal that all processing has been done
184  //
185 
186  KeSignalCallDpcDone (SystemArgument1);
187 
188  return;
189 }

197 /*++
198 
199 Routine Description:
200 
201  The specified i/o access map will be dumped into the buffer.
202  map 0 is a constant, but will be dumped anyway.
203 
204 Arguments:
205 
206  MapNumber - Number of access map to set. map 0 is fixed.
207 
208  IoAccessMap - Pointer to buffer (64K bits, 8K bytes) which
209  is to receive the definition of the access map.
210  Must be in non-paged pool.
211 
212 Return Value:
213 
214  TRUE if successful. FALSE if failure (attempt to query a map
215  which does not exist)
216 
217 --*/
218 
219 {
220 
221  ULONG i;
222  PVOID Map;
223  KIRQL OldIrql;
224  PUCHAR p;
225 
226  //
227  // Reject illegal requests
228  //
229 
230  if (MapNumber > IOPM_COUNT) {
231  return FALSE;
232  }
233 
234 
235  //
236  // Copy out the map
237  //
238 
239  if (MapNumber == IO_ACCESS_MAP_NONE) {
240 
241  //
242  // no access case, simply return a map of all 1s
243  //
244 
245  p = (PUCHAR)IoAccessMap;
246  for (i = 0; i < IOPM_SIZE; i++) {
247  p[i] = (UCHAR)-1;
248  }
249 
250  } else {
251 
252  //
253  // Raise to DISPATCH_LEVEL to obtain read access to the structure
254  //
255 
256  KeRaiseIrql (DISPATCH_LEVEL, &OldIrql);
257 
258  //
259  // normal case, just copy the bits
260  //
261 
262  Map = (PVOID)&(KiPcr ()->TSS->IoMaps[MapNumber-1].IoMap);
263  RtlCopyMemory ((PVOID)IoAccessMap, Map, IOPM_SIZE);
264 
265  //
266  // Restore IRQL.
267  //
268 
269  KeLowerIrql (OldIrql);
270  }
271 
272  return TRUE;
273 }

280 /*++
281 
282 Routine Description:
283 
284  Set the i/o access map which controls user mode i/o access
285  for a particular process.
286 
287 Arguments:
288 
289  Process - Pointer to kernel process object describing the
290  process which for which a map is to be set.
291 
292  MapNumber - Number of the map to set. Value of map is
293  defined by Ke386IoSetAccessProcess. Setting MapNumber
294  to IO_ACCESS_MAP_NONE will disallow any user mode i/o
295  access from the process.
296 
297 Return Value:
298 
299  TRUE if success, FALSE if failure (illegal MapNumber)
300 
301 --*/
302 
303 {
304  MAPINFO MapInfo;
305  USHORT MapOffset;
306 
307  //
308  // Reject illegal requests
309  //
310 
311  if (MapNumber > IOPM_COUNT) {
312  return FALSE;
313  }
314 
315  MapOffset = KiComputeIopmOffset (MapNumber);
316 
317  //
318  // Do the update on all processors at DISPATCH_LEVEL
319  //
320 
321  MapInfo.Process = Process;
322  MapInfo.MapOffset = MapOffset;
323 
324  KeGenericCallDpc (KiLoadIopmOffset,
325  &MapInfo);
326 
327  return TRUE;
328 }

398 /*++
399 
400 Routine Description:
401 
402  Gives IOPL to the specified process.
403 
404  All threads created from this point on will get IOPL. The current
405  process will get IOPL. Must be called from context of thread and
406  process that are to have IOPL.
407 
408  Iopl (to be made a boolean) in KPROCESS says all
409  new threads to get IOPL.
410 
411  Iopl (to be made a boolean) in KTHREAD says given
412  thread to get IOPL.
413 
414  N.B. If a kernel mode only thread calls this procedure, the
415  result is (a) pointless and (b) will break the system.
416 
417 Arguments:
418 
419  None.
420 
421 Return Value:
422 
423  None.
424 
425 --*/
426 
427 {
428 
429  PKTHREAD Thread;
430  PKPROCESS Process2;
431  PKTRAP_FRAME TrapFrame;
432  CONTEXT Context;
433 
434  //
435  // get current thread and Process2, set flag for IOPL in both of them
436  //
437 
438  Thread = KeGetCurrentThread();
439  Process2 = Thread->ApcState.Process;
440 
441  Process2->Iopl = 1;
442  Thread->Iopl = 1;
443 
444  //
445  // Force IOPL to be on for current thread
446  //
447 
448  TrapFrame = (PKTRAP_FRAME)((PUCHAR)Thread->InitialStack -
449  ALIGN_UP(sizeof(KTRAP_FRAME),KTRAP_FRAME_ALIGN) -
450  sizeof(FX_SAVE_AREA));
451 
452  Context.ContextFlags = CONTEXT_CONTROL;
453  KeContextFromKframes(TrapFrame,
454  NULL,
455  &Context);
456 
457  Context.EFlags |= (EFLAGS_IOPL_MASK & -1); // IOPL == 3
458 
459  KeContextToKframes(TrapFrame,
460  NULL,
461  &Context,
462  CONTEXT_CONTROL,
463  UserMode);
464 
465  return;
466 }
