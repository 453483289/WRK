35 VOID
36 CcMdlRead (
37  __in PFILE_OBJECT FileObject,
38  __in PLARGE_INTEGER FileOffset,
39  __in ULONG Length,
40  __out PMDL *MdlChain,
41  __out PIO_STATUS_BLOCK IoStatus

44 /*++
45 
46 Routine Description:
47 
48  This routine attempts to lock the specified file data in the cache
49  and return a description of it in an Mdl along with the correct
50  I/O status. It is *not* safe to call this routine from Dpc level.
51 
52  This routine is synchronous, and raises on errors.
53 
54  As each call returns, the pages described by the Mdl are
55  locked in memory, but not mapped in system space. If the caller
56  needs the pages mapped in system space, then it must map them.
57 
58  Note that each call is a "single shot" which should be followed by
59  a call to CcMdlReadComplete. To resume an Mdl-based transfer, the
60  caller must form one or more subsequent calls to CcMdlRead with
61  appropriately adjusted parameters.
62 
63 Arguments:
64 
65  FileObject - Pointer to the file object for a file which was
66  opened with NO_INTERMEDIATE_BUFFERING clear, i.e., for
67  which CcInitializeCacheMap was called by the file system.
68 
69  FileOffset - Byte offset in file for desired data.
70 
71  Length - Length of desired data in bytes.
72 
73  MdlChain - On output it returns a pointer to an Mdl chain describing
74  the desired data. Note that even if FALSE is returned,
75  one or more Mdls may have been allocated, as may be ascertained
76  by the IoStatus.Information field (see below).
77 
78  IoStatus - Pointer to standard I/O status block to receive the status
79  for the transfer. (STATUS_SUCCESS guaranteed for cache
80  hits, otherwise the actual I/O status is returned.) The
81  I/O Information Field indicates how many bytes have been
82  successfully locked down in the Mdl Chain.
83 
84 Return Value:
85 
86  None
87 
88 Raises:
89 
90  STATUS_INSUFFICIENT_RESOURCES - If a pool allocation failure occurs.
91 
92 --*/

139  // so we don't send Greg the uninitialized data...

143

253  // Unmap the data now, now that the pages are locked down.
254  //

260  // Now link the Mdl into the caller's chain
261  //

269  }
270  MdlTemp->Next = Mdl;
271  }
272  Mdl = NULL;
273 
274  //
275  // Assume we did not get all the data we wanted, and set FOffset
276  // to the end of the returned data.
277  //
278 
279  FOffset = BeyondLastByte;
280 
281  //
282  // Update number of bytes transferred.
283  //
284 
285  Information += ReceivedLength;
286 
287  //
288  // Calculate length left to transfer.
289  //
290 
291  Length -= ReceivedLength;
292  }
293  }
294  finally {
295 
296  CcMissCounter = &CcThrowAway;
297 
298  //
299  // Restore the readahead hints.
300  //
301 
302  MmResetPageFaultReadAhead( Thread, SavedState );
303 
304  if (AbnormalTermination()) {
305 
306  //
307  // We may have failed to allocate an Mdl while still having
308  // data mapped.
309  //
310 
311  if (Vacb != NULL) {
312  CcFreeVirtualAddress( Vacb );
313  }
314 
315  if (Mdl != NULL) {
316  IoFreeMdl( Mdl );
317  }
318 
319  //
320  // Otherwise loop to deallocate the Mdls
321  //
322

324  MdlTemp = (*MdlChain)->Next;
325 
326  DebugTrace( 0, mm, "MmUnlockPages/IoFreeMdl:\n", 0 );
327  DebugTrace( 0, mm, " Mdl = %08lx\n", *MdlChain );
328 
329  MmUnlockPages( *MdlChain );
330  IoFreeMdl( *MdlChain );
331 
332  *MdlChain = MdlTemp;
333  }
334 
335  DebugTrace(-1, me, "CcMdlRead -> Unwinding\n", 0 );
336 
337  }
338  else {
339 
340  //
341  // Now enable read ahead if it looks like we got any misses, and do
342  // the first one.
343  //
344 
345  if (!FlagOn( FileObject->Flags, FO_RANDOM_ACCESS ) &&
346  !PrivateCacheMap->Flags.ReadAheadEnabled &&
347  (SavedMissCounter != 0)) {
348 
349  CC_SET_PRIVATE_CACHE_MAP (PrivateCacheMap, PRIVATE_CACHE_MAP_READ_AHEAD_ENABLED);
350  CcScheduleReadAhead( FileObject, FileOffset, OriginalLength );
351  }
352 
353  //
354  // Now that we have described our desired read ahead, let's
355  // shift the read history down.
356  //
357 
358  PrivateCacheMap->FileOffset1 = PrivateCacheMap->FileOffset2;
359  PrivateCacheMap->BeyondLastByte1 = PrivateCacheMap->BeyondLastByte2;
360  PrivateCacheMap->FileOffset2 = *FileOffset;
361  PrivateCacheMap->BeyondLastByte2.QuadPart =
362  FileOffset->QuadPart + (LONGLONG)OriginalLength;
363 
364  IoStatus->Status = STATUS_SUCCESS;
365  IoStatus->Information = Information;
366  }
367  }
368 
369 
370  DebugTrace( 0, me, " <MdlChain = %08lx\n", *MdlChain );
371  DebugTrace2(0, me, " <IoStatus = %08lx, %08lx\n", IoStatus->Status,
372  IoStatus->Information );
373  DebugTrace(-1, me, "CcMdlRead -> VOID\n", 0 );
374 
375  return;

472 VOID
473 CcPrepareMdlWrite (
474  __in PFILE_OBJECT FileObject,
475  __in PLARGE_INTEGER FileOffset,
476  __in ULONG Length,
477  __out PMDL *MdlChain,
478  __out PIO_STATUS_BLOCK IoStatus
479  )
480 
481 /*++
482 
483 Routine Description:
484 
485  This routine attempts to lock the specified file data in the cache
486  and return a description of it in an Mdl along with the correct
487  I/O status. Pages to be completely overwritten may be satisfied
488  with emtpy pages. It is *not* safe to call this routine from Dpc level.
489 
490  This call is synchronous and raises on error.
491 
492  When this call returns, the caller may immediately begin
493  to transfer data into the buffers via the Mdl.
494 
495  When the call returns with TRUE, the pages described by the Mdl are
496  locked in memory, but not mapped in system space. If the caller
497  needs the pages mapped in system space, then it must map them.
498  On the subsequent call to CcMdlWriteComplete the pages will be
499  unmapped if they were mapped, and in any case unlocked and the Mdl
500  deallocated.
501 
502 Arguments:
503 
504  FileObject - Pointer to the file object for a file which was
505  opened with NO_INTERMEDIATE_BUFFERING clear, i.e., for
506  which CcInitializeCacheMap was called by the file system.
507 
508  FileOffset - Byte offset in file for desired data.
509 
510  Length - Length of desired data in bytes.
511 
512  MdlChain - On output it returns a pointer to an Mdl chain describing
513  the desired data. Note that even if FALSE is returned,
514  one or more Mdls may have been allocated, as may be ascertained
515  by the IoStatus.Information field (see below).
516 
517  IoStatus - Pointer to standard I/O status block to receive the status
518  for the in-transfer of the data. (STATUS_SUCCESS guaranteed
519  for cache hits, otherwise the actual I/O status is returned.)
520  The I/O Information Field indicates how many bytes have been
521  successfully locked down in the Mdl Chain.
522 
523 Return Value:
524 
525  None
526 
527 --*/
528 
529 {
530  PSHARED_CACHE_MAP SharedCacheMap;
531  PVOID CacheBuffer;
532  LARGE_INTEGER FOffset;
533  PMDL Mdl = NULL;
534  PMDL MdlTemp;
535  LARGE_INTEGER Temp;
536  ULONG SavedState = 0;
537  ULONG ZeroFlags = 0;
538  ULONG Information = 0;
539 
540  KLOCK_QUEUE_HANDLE LockHandle;
541  ULONG ActivePage;
542  ULONG PageIsDirty;
543  PVACB Vacb = NULL;
544 
545  DebugTrace(+1, me, "CcPrepareMdlWrite\n", 0 );
546  DebugTrace( 0, me, " FileObject = %08lx\n", FileObject );
547  DebugTrace2(0, me, " FileOffset = %08lx, %08lx\n", FileOffset->LowPart,
548  FileOffset->HighPart );
549  DebugTrace( 0, me, " Length = %08lx\n", Length );
550 
551  //
552  // Get pointer to SharedCacheMap.
553  //
554 
555  SharedCacheMap = FileObject->SectionObjectPointer->SharedCacheMap;
556 
557  if (!FlagOn(FileObject->Flags, FO_WRITE_THROUGH) &&
558  CcForceWriteThrough( FileObject, Length, SharedCacheMap, FALSE )) {
559 
560  ExRaiseStatus( STATUS_INSUFFICIENT_RESOURCES );
561  }
562 
563  //
564  // See if we have an active Vacb, that we need to free.
565  //
566 
567  GetActiveVacb( SharedCacheMap, LockHandle.OldIrql, Vacb, ActivePage, PageIsDirty );
568 
569  //
570  // If there is an end of a page to be zeroed, then free that page now,
571  // so it does not cause our data to get zeroed. If there is an active
572  // page, free it so we have the correct ValidDataGoal.
573  //
574 
575  if ((Vacb != NULL) || (SharedCacheMap->NeedToZero != NULL)) {
576 
577  CcFreeActiveVacb( SharedCacheMap, Vacb, ActivePage, PageIsDirty );
578  Vacb = NULL;
579  }
580 
581  FOffset = *FileOffset;
582 
583  //
584  // Put try-finally around the loop to deal with exceptions
585  //
586 
587  try {
588 
589  //
590  // Not all of the transfer will come back at once, so we have to loop
591  // until the entire transfer is complete.
592  //
593 
594  while (Length != 0) {
595 
596  ULONG ReceivedLength;
597  LARGE_INTEGER BeyondLastByte;
598 
599  //
600  // Map and see how much we could potentially access at this
601  // FileOffset, then cut it down if it is more than we need.
602  //
603 
604  CacheBuffer = CcGetVirtualAddress( SharedCacheMap,
605  FOffset,
606  &Vacb,
607  &ReceivedLength );
608 
609  if (ReceivedLength > Length) {
610  ReceivedLength = Length;
611  }
612 
613  BeyondLastByte.QuadPart = FOffset.QuadPart + (LONGLONG)ReceivedLength;
614 
615  //
616  // At this point we can calculate the ZeroFlags.
617  //
618 
619  //
620  // We can always zero middle pages, if any.
621  //
622 
623  ZeroFlags = ZERO_MIDDLE_PAGES;
624 
625  //
626  // See if we are completely overwriting the first or last page.
627  //
628 
629  if (((FOffset.LowPart & (PAGE_SIZE - 1)) == 0) &&
630  (ReceivedLength >= PAGE_SIZE)) {
631  ZeroFlags |= ZERO_FIRST_PAGE;
632  }
633 
634  if ((BeyondLastByte.LowPart & (PAGE_SIZE - 1)) == 0) {
635  ZeroFlags |= ZERO_LAST_PAGE;
636  }
637 
638  //
639  // See if the entire transfer is beyond valid data length,
640  // or at least starting from the second page.
641  //
642 
643  Temp = FOffset;
644  Temp.LowPart &= ~(PAGE_SIZE -1);
645  KeAcquireInStackQueuedSpinLock( &SharedCacheMap->BcbSpinLock, &LockHandle );
646  Temp.QuadPart = SharedCacheMap->ValidDataGoal.QuadPart - Temp.QuadPart;
647  KeReleaseInStackQueuedSpinLock( &LockHandle );
648 
649  if (Temp.QuadPart <= 0) {
650  ZeroFlags |= ZERO_FIRST_PAGE | ZERO_MIDDLE_PAGES | ZERO_LAST_PAGE;
651  } else if ((Temp.HighPart == 0) && (Temp.LowPart <= PAGE_SIZE)) {
652  ZeroFlags |= ZERO_MIDDLE_PAGES | ZERO_LAST_PAGE;
653  }
654 
655  (VOID)CcMapAndRead( SharedCacheMap,
656  &FOffset,
657  ReceivedLength,
658  ZeroFlags,
659  TRUE,
660  CacheBuffer );
661 
662  //
663  // Now attempt to allocate an Mdl to describe the mapped data.
664  //
665 
666  DebugTrace( 0, mm, "IoAllocateMdl:\n", 0 );
667  DebugTrace( 0, mm, " BaseAddress = %08lx\n", CacheBuffer );
668  DebugTrace( 0, mm, " Length = %08lx\n", ReceivedLength );
669 
670  Mdl = IoAllocateMdl( CacheBuffer,
671  ReceivedLength,
672  FALSE,
673  FALSE,
674  NULL );
675 
676  DebugTrace( 0, mm, " <Mdl = %08lx\n", Mdl );
677 
678  if (Mdl == NULL) {
679  DebugTrace( 0, 0, "Failed to allocate Mdl\n", 0 );
680 
681  ExRaiseStatus( STATUS_INSUFFICIENT_RESOURCES );
682  }
683 
684  DebugTrace( 0, mm, "MmProbeAndLockPages:\n", 0 );
685  DebugTrace( 0, mm, " Mdl = %08lx\n", Mdl );
686 
687  MmDisablePageFaultClustering(&SavedState);
688  MmProbeAndLockPages( Mdl, KernelMode, IoWriteAccess );
689  MmEnablePageFaultClustering(SavedState);
690  SavedState = 0;
691 
692  //
693  // Now that some data (maybe zeros) is locked in memory and
694  // set dirty, it is safe, and necessary for us to advance
695  // valid data goal, so that we will not subsequently ask
696  // for a zero page. Note if we are extending valid data,
697  // our caller has the file exclusive.
698  //
699 
700  KeAcquireInStackQueuedSpinLock( &SharedCacheMap->BcbSpinLock, &LockHandle );
701  if (BeyondLastByte.QuadPart > SharedCacheMap->ValidDataGoal.QuadPart) {
702  SharedCacheMap->ValidDataGoal = BeyondLastByte;
703  }
704  KeReleaseInStackQueuedSpinLock( &LockHandle );
705 
706  //
707  // Unmap the data now, now that the pages are locked down.
708  //
709 
710  CcFreeVirtualAddress( Vacb );
711  Vacb = NULL;
712 
713  //
714  // Now link the Mdl into the caller's chain
715  //
716 
717  if ( *MdlChain == NULL ) {
718  *MdlChain = Mdl;
719  } else {
720  MdlTemp = CONTAINING_RECORD( *MdlChain, MDL, Next );
721  while (MdlTemp->Next != NULL) {
722  MdlTemp = MdlTemp->Next;
723  }
724  MdlTemp->Next = Mdl;
725  }
726  Mdl = NULL;
727 
728  //
729  // Assume we did not get all the data we wanted, and set FOffset
730  // to the end of the returned data.
731  //
732 
733  FOffset = BeyondLastByte;
734 
735  //
736  // Update number of bytes transferred.
737  //
738 
739  Information += ReceivedLength;
740 
741  //
742  // Calculate length left to transfer.
743  //
744 
745  Length -= ReceivedLength;
746  }
747  }
748  finally {
749 
750  if (AbnormalTermination()) {
751 
752  if (SavedState != 0) {
753  MmEnablePageFaultClustering(SavedState);
754  }
755 
756  if (Vacb != NULL) {
757  CcFreeVirtualAddress( Vacb );
758  }
759 
760  if (Mdl != NULL) {
761  IoFreeMdl( Mdl );
762  }
763 
764  //
765  // Otherwise loop to deallocate the Mdls
766  //
767 
768  FOffset = *FileOffset;
769  while (*MdlChain != NULL) {
770  MdlTemp = (*MdlChain)->Next;
771 
772  DebugTrace( 0, mm, "MmUnlockPages/IoFreeMdl:\n", 0 );
773  DebugTrace( 0, mm, " Mdl = %08lx\n", *MdlChain );
774 
775  MmUnlockPages( *MdlChain );
776 
777  //
778  // Extract the File Offset for this part of the transfer, and

780  // marked them dirty. Ignore the only exception (allocation

850 VOID
851 CcMdlWriteComplete2 (
852  IN PFILE_OBJECT FileObject,
853  IN PLARGE_INTEGER FileOffset,
854  IN PMDL MdlChain
855  )
856 
857 /*++
858 
859 Routine Description:
860 
861  This routine must be called at IPL0 after a call to CcPrepareMdlWrite.
862  The caller supplies the ActualLength of data that it actually wrote
863  into the buffer, which may be less than or equal to the Length specified
864  in CcPrepareMdlWrite.
865 
866  This call does the following:
867 
868  Makes sure the data up to ActualLength eventually gets written.
869  If WriteThrough is FALSE, the data will not be written immediately.
870  If WriteThrough is TRUE, then the data is written synchronously.
871 
872  Unmaps the pages (if mapped), unlocks them and deletes the MdlChain
873 
874 Arguments:
875 
876  FileObject - Pointer to the file object for a file which was
877  opened with NO_INTERMEDIATE_BUFFERING clear, i.e., for
878  which CcInitializeCacheMap was called by the file system.
879 
880  FileOffset - Original file offset read above.
881 
882  MdlChain - same as returned from corresponding call to CcPrepareMdlWrite.
883 
884 Return Value:
885 
886  None
887 
888 --*/
889 
890 {
891  PMDL MdlNext;
892  PMDL Mdl;
893  PSHARED_CACHE_MAP SharedCacheMap;
894  LARGE_INTEGER FOffset;
895  IO_STATUS_BLOCK IoStatus;
896  KIRQL OldIrql;
897  NTSTATUS StatusToRaise = STATUS_SUCCESS;
898  BOOLEAN First = FALSE;
899 
900  DebugTrace(+1, me, "CcMdlWriteComplete\n", 0 );
901  DebugTrace( 0, me, " FileObject = %08lx\n", FileObject );
902  DebugTrace( 0, me, " MdlChain = %08lx\n", MdlChain );
903 
904  SharedCacheMap = FileObject->SectionObjectPointer->SharedCacheMap;
905 
906  //
907  // Deallocate the Mdls
908  //
909 
910  FOffset.QuadPart = *(LONGLONG UNALIGNED *)FileOffset;
911  Mdl = MdlChain;
912 
913  //
914  // If the MDL is unlocked, this is a retry.
915  //
916 
917  if (FlagOn( MdlChain->MdlFlags, MDL_PAGES_LOCKED )) {
918  First = TRUE;
919  }
920 
921  while (Mdl != NULL) {
922 
923  MdlNext = Mdl->Next;
924 
925  DebugTrace( 0, mm, "MmUnlockPages/IoFreeMdl:\n", 0 );
926  DebugTrace( 0, mm, " Mdl = %08lx\n", Mdl );
927 
928  //
929  // Now clear the dirty bits in the Pte and set them in the
930  // Pfn. The Mdls will not be locked on repeated completion
931  // attempts.
932  //
933 
934  if (First) {
935  MmUnlockPages( Mdl );
936  }
937 
938  //
939  // Extract the File Offset for this part of the transfer.
940  //
941 
942  if (FlagOn(FileObject->Flags, FO_WRITE_THROUGH)) {
943 
944  MmFlushSection ( FileObject->SectionObjectPointer,
945  &FOffset,
946  Mdl->ByteCount,
947  &IoStatus,
948  MM_FLUSH_ACQUIRE_FILE );
949 
950  ASSERT( IoStatus.Status != STATUS_ENCOUNTERED_WRITE_IN_PROGRESS );
951 
952  //
953  // If we got an I/O error, remember it.
954  //
955 
956  if (!NT_SUCCESS(IoStatus.Status)) {
957  StatusToRaise = IoStatus.Status;
958  }
959 
960  } else {
961 
962  //
963  // Ignore the only exception (allocation error), and console
964  // ourselves for having tried.
965  //
966 
967  CcSetDirtyInMask( SharedCacheMap, &FOffset, Mdl->ByteCount );
968  }
969 
970  FOffset.QuadPart = FOffset.QuadPart + (LONGLONG)(Mdl->ByteCount);
971 
972  Mdl = MdlNext;
973  }
974 
975  //
976  // Remove our open count and check to see if this makes the shared cache
977  // map eligible for lazy close.
978  //
979  // We do this now so, on failure, old filesystems which did not expect
980  // writethrough to raise continue to work. They will be within exception
981  // handling with the Mdl still in the IRP.
982  //
983  // Note that non-writethrough is the only one that needs the cache map,
984  // and it'll always work. Removing the open count for writethrough
985  // could be a minor win.
986  //
987 
988  if (First) {
989 
990  CcAcquireMasterLock( &OldIrql );
991 
992  CcDecrementOpenCount( SharedCacheMap, 'ldmC' );
993 
994  if ((SharedCacheMap->OpenCount == 0) &&
995  !FlagOn(SharedCacheMap->Flags, WRITE_QUEUED) &&
996  (SharedCacheMap->DirtyPages == 0)) {
997 
998  //
999  // Move to the dirty list.
1000  //
1001 
1002  RemoveEntryList( &SharedCacheMap->SharedCacheMapLinks );
1003  InsertTailList( &CcDirtySharedCacheMapList.SharedCacheMapLinks,
1004  &SharedCacheMap->SharedCacheMapLinks );
1005 
1006  //
1007  // Make sure the Lazy Writer will wake up, because we
1008  // want him to delete this SharedCacheMap.
1009  //
1010 
1011  LazyWriter.OtherWork = TRUE;
1012  if (!LazyWriter.ScanActive) {
1013  CcScheduleLazyWriteScan( FALSE );
1014  }
1015  }
1016 
1017  CcReleaseMasterLock( OldIrql );
1018  }
1019 
1020  //
1021  // If we got an I/O error, raise it now. Note that we have not free'd the Mdl
1022  // yet so the owning filesystem can retry the completion.
1023  //
1024 
1025  if (!NT_SUCCESS(StatusToRaise)) {
1026  ExRaiseStatus( FsRtlNormalizeNtstatus( StatusToRaise,
1027  STATUS_UNEXPECTED_IO_ERROR ));
1028  }
1029 
1030  //
1031  // Otherwise, free the Mdl chain and clean everything up.
1032  //
1033 
1034  Mdl = MdlChain;
1035  while (Mdl != NULL) {
1036 
1037  MdlNext = Mdl->Next;
1038  IoFreeMdl( Mdl );
1039  Mdl = MdlNext;
1040  }
1041 
1042  DebugTrace(-1, me, "CcMdlWriteComplete -> TRUE\n", 0 );
1043 
1044  return;
1045 }
