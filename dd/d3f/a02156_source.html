9 Module Name:
10 
11  Tunnel.c
12 
13 Abstract:
14 
15  The tunnel package provides a set of routines that allow compatibility
16  with applications that rely on filesystems being able to "hold onto"
17  file meta-info for a short period of time after deletion/renaming and
18  reinstantiating a new directory entry with that meta-info if a
19  create/rename occurs to cause a file of that name to appear again in a
20  short period of time.
21 
22  Note that this violates POSIX rules. This package should not be used
23  on POSIX fileobjects, i.e. fileobjects that have case-sensitive names.
24 
25  Entries are keyed by directory and one of the short/long names. An opaque
26  rock of information is also associated (create time, last write time, etc.).
27  This is expected to vary on a per-filesystem basis.
28 
29  A TUNNEL variable should be initialized for every volume in the system
30  at mount time. Thereafter, each delete/rename-out should add to the tunnel
31  and each create/rename-in should read from the tunnel. Each directory
32  deletion should also notify the package so that all associated entries can
33  be flushed. The package is responsible for cleaning out aged entries.
34 
35  Tunneled information is in the paged pool.
36 
37  Concurrent access to the TUNNEL variable is controlled by this package.
38  Callers are responsible for synchronizing access to the FsRtlDeleteTunnelCache
39  call.
40 
41  The functions provided in this package are as follows:
42 
43  o FsRtlInitializeTunnel - Initializes the TUNNEL package (called once per boot)
44 
45  o FsRtlInitializeTunnelCache - Initializes a TUNNEL structure (called once on mount)
46 
47  o FsRtlAddToTunnelCache - Adds a new key/value pair to the tunnel
48 
49  o FsRtlFindInTunnelCache - Finds and returns a key/value from the tunnel
50 
51  o FsRtlDeleteKeyFromTunnelCache - Deletes all entries with a given
52  directory key from the tunnel
53 
54  o FsRtlDeleteTunnelCache - Deletes a TUNNEL structure
55 
56 --*/

363 /*++
364 
365 Routine Description:
366 
367  Initializes the global part of the tunneling package.
368 
369 Arguments:
370 
371  None
372 
373 Return Value:
374 
375  None
376 
377 --*/
378 {
379  UNICODE_STRING ValueName;
380  USHORT LookasideDepth;
381 
382  PAGED_CODE();
383 
384  if (MmIsThisAnNtAsSystem()) {
385 
386  TunnelMaxEntries = 1024;
387 
388  }
389 
390  //
391  // Query our configurable parameters
392  //
393  // Don't worry about failure in retrieving from the registry. We've gotten
394  // this far so fall back on defaults even if there was a problem with resources.
395  //
396 
397  ValueName.Buffer = TUNNEL_SIZE_VALUE_NAME;
398  ValueName.Length = sizeof(TUNNEL_SIZE_VALUE_NAME) - sizeof(WCHAR);
399  ValueName.MaximumLength = sizeof(TUNNEL_SIZE_VALUE_NAME);
400  (VOID) FsRtlGetTunnelParameterValue(&ValueName, &TunnelMaxEntries);
401 
402  ValueName.Buffer = TUNNEL_AGE_VALUE_NAME;
403  ValueName.Length = sizeof(TUNNEL_AGE_VALUE_NAME) - sizeof(WCHAR);
404  ValueName.MaximumLength = sizeof(TUNNEL_AGE_VALUE_NAME);
405  (VOID) FsRtlGetTunnelParameterValue(&ValueName, &TunnelMaxAge);
406 
407  if (TunnelMaxAge == 0) {
408 
409  //
410  // If the registry has been set so the timeout is zero, we should force
411  // the number of entries to zero also. This preserves expectations and lets
412  // us key off of max entries alone in performing the hard disabling of the
413  // caching code.
414  //
415 
416  TunnelMaxEntries = 0;
417  }
418 
419  //
420  // Convert from seconds to 10ths of msecs, the internal resolution
421  //
422 
423  TunnelMaxAge *= 10000000;
424 
425  //
426  // Build the lookaside list for common node allocation
427  //
428 
429  if (TunnelMaxEntries > MAXUSHORT) {
430 
431  //
432  // User is hinting a big need to us
433  //
434 
435  LookasideDepth = MAX_LOOKASIDE_DEPTH;
436 
437  } else {
438 
439  LookasideDepth = ((USHORT)TunnelMaxEntries)/16;
440  }
441 
442  if (LookasideDepth == 0 && TunnelMaxEntries) {
443 
444  //
445  // Miniscule number of entries allowed. Lookaside 'em all.
446  //
447 
448  LookasideDepth = (USHORT)TunnelMaxEntries + 1;
449  }
450 
451  if (LookasideDepth > MAX_LOOKASIDE_DEPTH) {
452 
453  //
454  // Finally, restrict the depth to something reasonable.
455  //
456 
457  LookasideDepth = MAX_LOOKASIDE_DEPTH;
458  }
459 
460  ExInitializePagedLookasideList( &TunnelLookasideList,
461  NULL,
462  NULL,
463  0,
464  LOOKASIDE_NODE_SIZE,
465  'LnuT',
466  LookasideDepth );
467 
468  return;
469 }

521 //
522 // *** SPEC
523 //
524 // FsRtlAddToTunnelCache - add information to a tunnel cache
525 //
526 // FsRtlAddToTunnelCache is called by file systems when a name disappears from a
527 // directory. This typically occurs in both the delete and the rename paths. When
528 // a name is deleted, all information needed to be cached is extracted from the file
529 // and passed in a single buffer. This information is stored keyed by the directory key
530 // (a ULONG that is unique to the directory) and the short-name of the file.
531 //
532 // The caller is required to synchronize this call against FsRtlDeleteTunnelCache.
533 //
534 // Arguments:
535 // Cache pointer to cache initialized by FsRtlInitializeTunnelCache
536 // DirectoryKey ULONG unique ID of the directory containing the deleted file
537 // ShortName UNICODE_STRING* short (8.3) name of the file
538 // LongName UNICODE_STRING* full name of the file
539 // DataLength ULONG length of data to be cached with these names
540 // Data VOID* data that will be cached.
541 //
542 // It is acceptable for the Cache to ignore this request based upon memory constraints.
543 //
544 // Change: W95 maintains 10 items in the tunnel cache. Since we are a potential server
545 // this should be much higher. The max count would be initialized from the registry with
546 // a proposed default of 1024. Adds which run into the limit would cause least recently
547 // inserted recycling (i.e., off of the top of the timer queue).
548 //
549 // Change: Key should be by the name removed, not necessarily the short name. If a long name
550 // is removed, it would be incorrect to miss the tunnel. Use KeyByShortName boolean to specify
551 // which.
552 //
553 // Change: Specify that Data, ShortName, and LongName are copied for storage.
554 //
555 
556 VOID
557 FsRtlAddToTunnelCache (
558  __in TUNNEL *Cache,
559  __in ULONGLONG DirKey,
560  __in UNICODE_STRING *ShortName,
561  __in UNICODE_STRING *LongName,
562  __in BOOLEAN KeyByShortName,
563  __in ULONG DataLength,
564  __in_bcount(DataLength) VOID *Data
565  )
566 /*++
567 
568 Routine Description:
569 
570  Adds an entry to the tunnel cache keyed by
571 
572  DirectoryKey ## (KeyByShortName ? ShortName : LongName)
573 
574  ShortName, LongName, and Data are copied and stored in the tunnel. As a side
575  effect, if there are too many entries in the tunnel cache, this routine will
576  initiate expiration in the tunnel cache.
577 
578 Arguments:
579 
580  Cache - a tunnel cache initialized by FsRtlInitializeTunnelCache()
581 
582  DirKey - the key value of the directory the name appeared in
583 
584  ShortName - (optional if !KeyByShortName) the 8.3 name of the file
585 
586  LongName - (optional if KeyByShortName) the long name of the file
587 
588  KeyByShortName - specifies which name is keyed in the tunnel cache
589 
590  DataLength - specifies the length of the opaque data segment (file
591  system specific) which contains the tunneling information for this
592  file
593 
594  Data - pointer to the opaque tunneling data segment
595 
596 Return Value:
597 
598  None
599 
600 --*/
601 {
602  LONG Compare;
603  ULONG NodeSize;
604  PUNICODE_STRING NameKey;
605  PRTL_SPLAY_LINKS *Links;
606  LIST_ENTRY FreePoolList;
607 
608  PTUNNEL_NODE Node = NULL;
609  PTUNNEL_NODE NewNode = NULL;
610  BOOLEAN AllocatedFromPool = FALSE;
611 
612  PAGED_CODE();
613 
614  //
615  // If MaxEntries is 0 then tunneling is disabled.
616  //
617 
618  if (TunnelMaxEntries == 0) return;
619 
620  InitializeListHead(&FreePoolList);
621 
622  //
623  // Grab a new node for this data
624  //
625 
626  NodeSize = sizeof(TUNNEL_NODE) + ShortName->Length + LongName->Length + DataLength;
627 
628  if (LOOKASIDE_NODE_SIZE >= NodeSize) {
629 
630  NewNode = ExAllocateFromPagedLookasideList(&TunnelLookasideList);
631  }
632 
633  if (NewNode == NULL) {
634 
635  //
636  // Data doesn't fit in lookaside nodes
637  //
638 
639  NewNode = ExAllocatePoolWithTag(PagedPool|POOL_COLD_ALLOCATION, NodeSize, 'PnuT');
640 
641  if (NewNode == NULL) {
642 
643  //
644  // Give up tunneling this entry
645  //
646 
647  return;
648  }
649 
650  AllocatedFromPool = TRUE;
651  }
652 
653  //
654  // Traverse the cache to find our insertion point
655  //
656 
657  NameKey = (KeyByShortName ? ShortName : LongName);
658 
659  ExAcquireFastMutex(&Cache->Mutex);
660 
661  Links = &Cache->Cache;
662 
663  while (*Links) {
664 
665  Node = CONTAINING_RECORD(*Links, TUNNEL_NODE, CacheLinks);
666 
667  Compare = FsRtlCompareNodeAndKey(Node, DirKey, NameKey);
668 
669  if (Compare > 0) {
670 
671  Links = &RtlLeftChild(&Node->CacheLinks);
672 
673  } else {
674 
675  if (Compare < 0) {
676 
677  Links = &RtlRightChild(&Node->CacheLinks);
678 
679  } else {
680 
681  break;
682  }
683  }
684  }
685 
686  //
687  // Thread new data into the splay tree
688  //
689 
690  RtlInitializeSplayLinks(&NewNode->CacheLinks);
691 
692  if (Node) {
693 
694  //
695  // Not inserting first node in tree
696  //
697 
698  if (*Links) {
699 
700  //
701  // Entry exists in the cache, so replace by swapping all splay links
702  //
703 
704  RtlRightChild(&NewNode->CacheLinks) = RtlRightChild(*Links);
705  RtlLeftChild(&NewNode->CacheLinks) = RtlLeftChild(*Links);
706 
707  if (RtlRightChild(*Links)) RtlParent(RtlRightChild(*Links)) = &NewNode->CacheLinks;
708  if (RtlLeftChild(*Links)) RtlParent(RtlLeftChild(*Links)) = &NewNode->CacheLinks;
709 
710  if (!RtlIsRoot(*Links)) {
711 
712  //
713  // Change over the parent links. Note that we've messed with *Links now
714  // since it is pointing at the parent member.
715  //
716 
717  RtlParent(&NewNode->CacheLinks) = RtlParent(*Links);
718 
719  if (RtlIsLeftChild(*Links)) {
720 
721  RtlLeftChild(RtlParent(*Links)) = &NewNode->CacheLinks;
722 
723  } else {
724 
725  RtlRightChild(RtlParent(*Links)) = &NewNode->CacheLinks;
726  }
727 
728  } else {
729 
730  //
731  // Set root of the cache
732  //
733 
734  Cache->Cache = &NewNode->CacheLinks;
735  }
736 
737  //
738  // Free old node
739  //
740 
741  RemoveEntryList(&Node->ListLinks);
742 
743  FsRtlFreeTunnelNode(Node, &FreePoolList);
744 
745  Cache->NumEntries--;
746 
747  } else {
748 
749  //
750  // Simple insertion as a leaf
751  //
752 
753  NewNode->CacheLinks.Parent = &Node->CacheLinks;
754  *Links = &NewNode->CacheLinks;
755  }
756 
757  } else {
758 
759  Cache->Cache = &NewNode->CacheLinks;
760  }
761 
762  //
763  // Thread onto the timer list
764  //
765 
766  KeQuerySystemTime(&NewNode->CreateTime);
767  InsertTailList(&Cache->TimerQueue, &NewNode->ListLinks);
768 
769  Cache->NumEntries++;
770 
771  //
772  // Stash tunneling information
773  //
774 
775  NewNode->DirKey = DirKey;
776 
777  if (KeyByShortName) {
778 
779  NewNode->Flags = TUNNEL_FLAG_KEY_SHORT;
780 
781  } else {
782 
783  NewNode->Flags = 0;
784  }
785 
786  //
787  // Initialize the internal UNICODE_STRINGS to point at the buffer segments. For various
788  // reasons (UNICODE APIs are incomplete, we're avoiding calling any allocate routine more
789  // than once, UNICODE strings are not guaranteed to be null terminated) we have to do a lot
790  // of this by hand.
791  //
792  // The data is layed out like this in the allocated block:
793  //
794  // -----------------------------------------------------------------------------------
795  // | TUNNEL_NODE | Node->ShortName.Buffer | Node->LongName.Buffer | Node->TunnelData |
796  // -----------------------------------------------------------------------------------
797  //
798 
799  NewNode->ShortName.Buffer = (PWCHAR)((PCHAR)NewNode + sizeof(TUNNEL_NODE));
800  NewNode->LongName.Buffer = (PWCHAR)((PCHAR)NewNode + sizeof(TUNNEL_NODE) + ShortName->Length);
801 
802  NewNode->ShortName.Length = NewNode->ShortName.MaximumLength = ShortName->Length;
803  NewNode->LongName.Length = NewNode->LongName.MaximumLength = LongName->Length;
804 
805  if (ShortName->Length) {
806 
807  RtlCopyMemory(NewNode->ShortName.Buffer, ShortName->Buffer, ShortName->Length);
808  }
809 
810  if (LongName->Length) {
811 
812  RtlCopyMemory(NewNode->LongName.Buffer, LongName->Buffer, LongName->Length);
813  }
814 
815  NewNode->TunnelData = (PVOID)((PCHAR)NewNode + sizeof(TUNNEL_NODE) + ShortName->Length + LongName->Length);
816 
817  NewNode->TunnelDataLength = DataLength;
818 
819  RtlCopyMemory(NewNode->TunnelData, Data, DataLength);
820 
821  if (AllocatedFromPool) {
822 
823  SetFlag(NewNode->Flags, TUNNEL_FLAG_NON_LOOKASIDE);
824  }
825 
826  //
827  // Clean out the cache, release, and then drop any pool memory we need to
828  //
829 
830  FsRtlPruneTunnelCache(Cache, &FreePoolList);
831 
832  ExReleaseFastMutex(&Cache->Mutex);
833 
834  FsRtlEmptyFreePoolList(&FreePoolList);
835 
836  return;
837 }
