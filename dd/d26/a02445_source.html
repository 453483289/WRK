275 VOID
276 KiInitializeMTRR (
277  IN BOOLEAN LastProcessor
278  )
279 /*++
280 
281 Routine Description:
282 
283  Called to incrementally initialize the physical range
284  database feature. First processor's MTRR set is read into the
285  physical range database.
286 
287 Arguments:
288 
289  LastProcessor - If set this is the last processor to execute this routine
290  such that when this processor finishes, the initialization is complete.
291 
292 Return Value:
293 
294  None - if there was a problem the function
295  KeSetPhysicalCacheTypeRange type is disabled.
296 
297 --*/
298 {
299  BOOLEAN Status;
300  ULONG Index;
301  MTRR_DEFAULT Default;
302  MTRR_CAPABILITIES Capabilities;
303  NEW_RANGE NewRange;
304  MTRR_VARIABLE_BASE MtrrBase;
305  MTRR_VARIABLE_MASK MtrrMask;
306  ULONGLONG Base, Mask, Length;
307  PKPRCB Prcb;
308 
309  Status = TRUE;
310  RtlZeroMemory (&NewRange, sizeof (NewRange));
311  NewRange.Status = STATUS_UNSUCCESSFUL;
312 
313  //
314  // If this is the first processor, initialize some fields
315  //
316 
317  if (KeGetPcr()->Number == 0) {
318  KeInitializeSpinLock (&KiRangeLock);
319 
320  KiRangeInfo.Capabilities.u.QuadPart = RDMSR(MTRR_MSR_CAPABILITIES);
321  KiRangeInfo.Default.u.QuadPart = RDMSR(MTRR_MSR_DEFAULT);
322  KiRangeInfo.DefaultCachedType = MTRR_TYPE_MAX;
323 
324  //
325  // If h/w mtrr support is not enabled, disable OS support
326  //
327 
328  if (!KiRangeInfo.Default.u.hw.MtrrEnabled ||
329  KiRangeInfo.Capabilities.u.hw.VarCnt == 0 ||
330  KiRangeInfo.Default.u.hw.Type != MTRR_TYPE_UC) {
331 
332  DBGMSG("MTRR feature disabled.\n");
333  Status = FALSE;
334 
335  } else {
336 
337  //
338  // If USWC type is supported by hardware, but the MTRR
339  // feature is not set in KeFeatureBits, it is because
340  // the HAL indicated USWC should not be used on this
341  // machine. (Possibly due to shared memory clusters).
342  //
343 
344  if (KiRangeInfo.Capabilities.u.hw.UswcSupported &&
345  ((KeFeatureBits & KF_MTRR) == 0)) {
346 
347  DBGMSG("KiInitializeMTRR: MTRR use globally disabled on this machine.\n");
348  KiRangeInfo.Capabilities.u.hw.UswcSupported = 0;
349  }
350 
351  //
352  // Allocate initial range type database
353  //
354 
355  KiRangeInfo.NoRange = 0;
356  KiRangeInfo.MaxRange = (UCHAR) KiRangeInfo.Capabilities.u.hw.VarCnt + GROW_RANGE_TABLE;
357 
358  //
359  // Don't allocate a new range on reinitialization from
360  // hibernate.
361  //
362 
363  if (KiRangeInfo.Ranges == NULL) {
364  KiRangeInfo.Ranges = ExAllocatePoolWithTag (NonPagedPool,
365  sizeof(ONE_RANGE) * KiRangeInfo.MaxRange,
366  ' eK');
367  }
368  if (KiRangeInfo.Ranges != NULL) {
369  RtlZeroMemory (KiRangeInfo.Ranges,
370  sizeof(ONE_RANGE) * KiRangeInfo.MaxRange);
371  }
372  }
373  }
374 
375  //
376  // Workaround for cpu signatures 611, 612, 616 and 617
377  // - if the request for setting a variable MTRR specifies
378  // an address which is not 4M aligned or length is not
379  // a multiple of 4M then possible problem for INVLPG inst.
380  // Detect if workaround is required
381  //
382 
383  Prcb = KeGetCurrentPrcb();
384  if (Prcb->CpuType == 6 &&
385  (Prcb->CpuStep == 0x0101 || Prcb->CpuStep == 0x0102 ||
386  Prcb->CpuStep == 0x0106 || Prcb->CpuStep == 0x0107 )) {
387 
388  if (strcmp((PCHAR)Prcb->VendorString, "GenuineIntel") == 0) {
389 
390  //
391  // Only do this if it's an Intel part, other
392  // manufacturers may have the same stepping
393  // numbers but no bug.
394  //
395 
396  KiRangeInfo.MtrrWorkaround = TRUE;
397  }
398  }
399 
400  //
401  // If MTRR support disabled on first processor or if
402  // buffer not allocated then fall through
403  //
404 
405  if (!KiRangeInfo.Ranges){
406  Status = FALSE;
407  Capabilities.u.QuadPart = 0; // satisfy no_opt compilation
408  } else {
409 
410  //
411  // Verify MTRR support is symmetric
412  //
413 
414  Capabilities.u.QuadPart = RDMSR(MTRR_MSR_CAPABILITIES);
415 
416  if ((Capabilities.u.hw.UswcSupported) &&
417  ((KeFeatureBits & KF_MTRR) == 0)) {
418  DBGMSG ("KiInitializeMTRR: setting UswcSupported FALSE\n");
419  Capabilities.u.hw.UswcSupported = 0;
420  }
421 
422  Default.u.QuadPart = RDMSR(MTRR_MSR_DEFAULT);
423 
424  if (Default.u.QuadPart != KiRangeInfo.Default.u.QuadPart ||
425  Capabilities.u.QuadPart != KiRangeInfo.Capabilities.u.QuadPart) {
426  DBGMSG ("KiInitializeMTRR: asymmetric mtrr support\n");
427  Status = FALSE;
428  }
429  }
430 
431  NewRange.Status = STATUS_SUCCESS;
432 
433  //
434  // MTRR registers should be identically set on each processor.
435  // Ranges should be added to the range database only for one
436  // processor.
437  //
438 
439  if (Status && (KeGetPcr()->Number == 0)) {
440 #if IDBG
441  KiDumpMTRR ("Processor MTRR:", NULL);
442 #endif
443 
444  //
445  // Read current MTRR settings for various cached range types
446  // and add them to the range database
447  //
448 
449  for (Index=0; Index < Capabilities.u.hw.VarCnt; Index++) {
450 
451  MtrrBase.u.QuadPart = RDMSR(MTRR_MSR_VARIABLE_BASE+Index*2);
452  MtrrMask.u.QuadPart = RDMSR(MTRR_MSR_VARIABLE_MASK+Index*2);
453 
454  Mask = MtrrMask.u.QuadPart & KiMtrrMaskMask;
455  Base = MtrrBase.u.QuadPart & KiMtrrMaskBase;
456 
457  //
458  // Note - the variable MTRR Mask does NOT contain the length
459  // spanned by the variable MTRR. Thus just checking the Valid
460  // Bit should be sufficient for identifying a valid MTRR.
461  //
462 
463  if (MtrrMask.u.hw.Valid) {
464 
465  Length = KiMaskToLength(Mask);
466 
467  //
468  // Check for non-contiguous MTRR mask.
469  //
470 
471  if ((Mask + Length) & KiMtrrOverflowMask) {
472  DBGMSG ("KiInitializeMTRR: Found non-contiguous MTRR mask!\n");
473  Status = FALSE;
474  }
475 
476  //
477  // Add this MTRR to the range database
478  //
479 
480  Base &= Mask;
481  KiAddRange (
482  &NewRange,
483  Base,
484  Base + Length - 1,
485  (UCHAR) MtrrBase.u.hw.Type
486  );
487 
488  //
489  // Check for default cache type
490  //
491 
492  if (MtrrBase.u.hw.Type == MTRR_TYPE_WB) {
493  KiRangeInfo.DefaultCachedType = MTRR_TYPE_WB;
494  }
495 
496  if (KiRangeInfo.DefaultCachedType == MTRR_TYPE_MAX &&
497  MtrrBase.u.hw.Type == MTRR_TYPE_WT) {
498  KiRangeInfo.DefaultCachedType = MTRR_TYPE_WT;
499  }
500  }
501  }
502 
503  //
504  // If a default type for "cached" was not found, assume write-back
505  //
506 
507  if (KiRangeInfo.DefaultCachedType == MTRR_TYPE_MAX) {
508  DBGMSG ("KiInitializeMTRR: assume write-back\n");
509  KiRangeInfo.DefaultCachedType = MTRR_TYPE_WB;
510  }
511  }
512 
513  //
514  // Done
515  //
516 
517  if (!NT_SUCCESS(NewRange.Status)) {
518  Status = FALSE;
519  }
520 
521  if (!Status) {
522  DBGMSG ("KiInitializeMTRR: OS support for MTRRs disabled\n");
523  if (KiRangeInfo.Ranges != NULL) {
524  ExFreePool (KiRangeInfo.Ranges);
525  KiRangeInfo.Ranges = NULL;
526  }
527  } else {
528 
529  // if last processor indicate initialization complete
530  if (LastProcessor) {
531  KiRangeInfo.RangesValid = TRUE;
532  }
533  }
534 }

541 
542 Routine Description:
543 
544  This function reloads the MTRR registers to be the current
545  known values. This is used on a system wakeup to ensure the

547 
548  N.B. The caller must have the PAGELK code locked

552  none

586 NTSTATUS
587 KeSetPhysicalCacheTypeRange (
588  IN PHYSICAL_ADDRESS PhysicalAddress,
589  IN ULONG NumberOfBytes,
590  IN MEMORY_CACHING_TYPE CacheType
591  )
592 /*++
593 
594 Routine Description:
595 
596  This function sets a physical range to a particular cache type.
597  If the system does not support setting cache policies based on
598  physical ranges, no action is taken.
599 
600 Arguments:
601 
602  PhysicalAddress - The starting address of the range being set
603 
604  NumberOfBytes - The length, in bytes, of the range being set
605 
606  CacheType - The caching type for which the physical range is
607  to be set to.
608 
609  NonCached:
610  Setting ranges to be NonCached is done for
611  book keeping reasons. A return of SUCCESS when
612  setting a range NonCached does not mean it has
613  been physically set to as NonCached. The caller
614  must use a cache-disabled virtual pointer for
615  any NonCached range.
616 
617  Cached:
618  A successful return indicates that the physical
619  range has been set to cached. This mode requires
620  the caller to be at irql < dispatch_level.
621 
622  FrameBuffer:
623  A successful return indicates that the physical
624  range has been set to be framebuffer cached.
625  This mode requires the caller to be at irql <
626  dispatch_level.
627 
628  USWCCached:
629  This type is to be satisfied only via PAT and
630  fails for the MTRR interface.
631 
632 Return Value:
633 
634  STATUS_SUCCESS - if success, the cache attributes of the physical range
635  have been set.
636 
637  STATUS_NOT_SUPPORTED - either feature not supported or not yet initialized,
638  or MmWriteCombined type not supported and is
639  requested, or input range does not match restrictions
640  imposed by workarounds for current processor stepping
641  or is below 1M (in the fixed MTRR range), or not yet
642  initialized.
643 
644  STATUS_UNSUCCESSFUL - Unable to satisfy request due to
645  - Unable to map software image into limited # of
646  hardware MTRRs.
647  - irql was not < DISPATCH_LEVEL.
648  - Failure due to other internal error (out of memory).
649 
650  STATUS_INVALID_PARAMETER - Incorrect input memory type.
651 
652 --*/
653 {
654  KIRQL OldIrql;
655  NEW_RANGE NewRange;
656  BOOLEAN RemoveThisType[MTRR_TYPE_MAX];
657  BOOLEAN EffectRangeChange, AddToRangeDatabase;
658 
659  //
660  // If caller has requested the MmUSWCCached memory type then fail
661  // - MmUSWCCached is supported via PAT and not otherwise
662  //
663 
664  if (CacheType == MmUSWCCached) {
665  return STATUS_NOT_SUPPORTED;
666  }
667 
668  //
669  // Addresses above 4GB, below 1MB or not page aligned and
670  // page length are not supported.
671  //
672 
673  if ((PhysicalAddress.HighPart != 0) ||
674  (PhysicalAddress.LowPart < (1 * 1024 * 1024)) ||
675  (PhysicalAddress.LowPart & 0xfff) ||
676  (NumberOfBytes & 0xfff) ) {
677  return STATUS_NOT_SUPPORTED;
678  }
679 
680  ASSERT (NumberOfBytes != 0);
681 
682  //
683  // If the processor is a AMD K6 with MTRR support then perform
684  // processor specific implementation.
685  //
686 
687  if (KeFeatureBits & KF_AMDK6MTRR) {
688 
689  if ((CacheType != MmWriteCombined) && (CacheType != MmNonCached)) {
690  return STATUS_NOT_SUPPORTED;
691  }
692 
693  return KiAmdK6MtrrSetMemoryType(PhysicalAddress.LowPart,
694  NumberOfBytes,
695  CacheType);
696  }
697 
698  //
699  // If processor doesn't have the memory type range feature
700  // return not supported.
701  //
702 
703  if (!KiRangeInfo.RangesValid) {
704  return STATUS_NOT_SUPPORTED;
705  }
706 
707  //
708  // Workaround for cpu signatures 611, 612, 616 and 617
709  // - if the request for setting a variable MTRR specifies
710  // an address which is not 4M aligned or length is not
711  // a multiple of 4M then return status not supported
712  //
713 
714  if ((KiRangeInfo.MtrrWorkaround) &&
715  ((PhysicalAddress.LowPart & 0x3fffff) ||
716  (NumberOfBytes & 0x3fffff))) {
717 
718  return STATUS_NOT_SUPPORTED;
719  }
720 
721  RtlZeroMemory (&NewRange, sizeof (NewRange));
722  NewRange.Base = PhysicalAddress.QuadPart;
723  NewRange.Limit = NewRange.Base + NumberOfBytes - 1;
724 
725  //
726  // Determine what the new mtrr range type is. If setting NonCached then
727  // the database need not be updated to reflect the virtual change. This
728  // is because non-cached virtual pointers are mapped as cache disabled.
729  //
730 
731  EffectRangeChange = TRUE;
732  AddToRangeDatabase = TRUE;
733  switch (CacheType) {
734  case MmNonCached:
735  NewRange.Type = MTRR_TYPE_UC;
736 
737  //
738  // NonCached ranges do not need to be reflected into the h/w state
739  // as all non-cached ranges are mapped with cache-disabled pointers.
740  // This also means that cache-disabled ranges do not need to
741  // be put into mtrrs, or held in the range, regardless of the default
742  // range type.
743  //
744 
745  EffectRangeChange = FALSE;
746  AddToRangeDatabase = FALSE;
747  break;
748 
749  case MmCached:
750  NewRange.Type = KiRangeInfo.DefaultCachedType;
751  break;
752 
753  case MmWriteCombined:
754  NewRange.Type = MTRR_TYPE_USWC;
755 
756  //
757  // If USWC type isn't supported, then request can not be honored
758  //
759 
760  if (!KiRangeInfo.Capabilities.u.hw.UswcSupported) {
761  DBGMSG ("KeSetPhysicalCacheTypeRange: USWC not supported\n");
762  return STATUS_NOT_SUPPORTED;
763  }
764  break;
765 
766  default:
767  DBGMSG ("KeSetPhysicalCacheTypeRange: no such cache type\n");
768  return STATUS_INVALID_PARAMETER;
769  break;
770  }
771 
772  NewRange.Status = STATUS_SUCCESS;
773 
774  //
775  // The default type is UC thus the range is still mapped using
776  // a Cache Disabled VirtualPointer and hence it need not be added.
777  //
778 
779  //
780  // If h/w needs updated, lock down the code required to effect the change
781  //
782 
783  if (EffectRangeChange) {
784  if (KeGetCurrentIrql() >= DISPATCH_LEVEL) {
785 
786  //
787  // Code can not be locked down. Supplying a new range type requires
788  // that the caller calls at irql < dispatch_level.
789  //
790 
791  DBGMSG ("KeSetPhysicalCacheTypeRange failed due to calling IRQL == DISPATCH_LEVEL\n");
792  return STATUS_UNSUCCESSFUL;
793  }
794 
795  MmLockPageableSectionByHandle(ExPageLockHandle);
796  }
797 
798  //
799  // Serialize the range type database
800  //
801 
802  KeAcquireSpinLock (&KiRangeLock, &OldIrql);
803 
804  //
805  // If h/w is going to need updated, then start an effective range change
806  //
807 
808  if (EffectRangeChange) {
809  KiStartEffectiveRangeChange (&NewRange);
810  }
811 
812  if (NT_SUCCESS (NewRange.Status)) {
813 
814  //
815  // If the new range is NonCached, then don't remove standard memory
816  // caching types
817  //
818 
819  memset (RemoveThisType, TRUE, MTRR_TYPE_MAX);
820  if (NewRange.Type != MTRR_TYPE_UC) {
821  //

1198  //

1216  }

1229  // the same type, combine them.
1230  //

1321  //

1376  //
