45 /*++
46 
47 Routine Description:
48 
49  This routine returns a valid DOS path for the given device object.
50  This caller of this routine must call ExFreePool on DosName->Buffer
51  when it is no longer needed.
52 
53 Arguments:
54

56 
57  DosName - Returns the DOS name for the volume

65 {
66  PDEVICE_OBJECT volumeDeviceObject = VolumeDeviceObject;
67  PMOUNTDEV_NAME name;

69  KEVENT event;
70  PIRP irp;
71  IO_STATUS_BLOCK ioStatus;

79  //

86  KeInitializeEvent(&event, NotificationEvent, FALSE);

92  }
93

95  if (status == STATUS_PENDING) {
96  KeWaitForSingleObject(&event, Executive, KernelMode, FALSE, NULL);

98  }
99 
100  if (!NT_SUCCESS(status)) {

102  }
103

105  status = IoGetDeviceObjectPointer(&mountmgrName, FILE_READ_ATTRIBUTES,

114  deviceObject, name, 512,

123  if (status == STATUS_PENDING) {
124  KeWaitForSingleObject(&event, Executive, KernelMode, FALSE, NULL);
125  status = ioStatus.Status;
126  }
127

145  if (!paths) {

149 
150  KeInitializeEvent(&event, NotificationEvent, FALSE);
151  irp = IoBuildDeviceIoControlRequest(IOCTL_MOUNTMGR_QUERY_DOS_VOLUME_PATH,
152  deviceObject, name, 512,
153  paths, len, FALSE, &event, &ioStatus);
154  if (!irp) {
155  ExFreePool(paths);
156  ObDereferenceObject(fileObject);
157  return STATUS_INSUFFICIENT_RESOURCES;
158  }
159 
160  status = IoCallDriver(deviceObject, irp);
161  if (status == STATUS_PENDING) {
162  KeWaitForSingleObject(&event, Executive, KernelMode, FALSE, NULL);
163  status = ioStatus.Status;
164  }
165 
166  if (!NT_SUCCESS(status)) {
167  ExFreePool(paths);
168  ObDereferenceObject(fileObject);
169  return status;
170  }
171 
172  DosName->Length = (USHORT) paths->MultiSzLength - 2*sizeof(WCHAR);
173  DosName->MaximumLength = DosName->Length + sizeof(WCHAR);
174  DosName->Buffer = (PWCHAR) paths;
175 
176  RtlCopyMemory(paths, paths->MultiSz, DosName->Length);
177  DosName->Buffer[DosName->Length/sizeof(WCHAR)] = 0;
178 
179  ObDereferenceObject(fileObject);
180 
181  return STATUS_SUCCESS;
182 }
