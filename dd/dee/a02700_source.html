184 //
185 // BOOLEAN
186 // SepBadImpersonationLevel(
187 // IN SECURITY_IMPERSONATION_LEVEL ImpersonationLevel,
188 // IN BOOLEAN ServerIsRemote
189 // )
190 //
191 // Routine Description:
192 //
193 // Determine whether a client is trying to impersonate inappropriately
194 // This routine should only be called if a thread requesting impersonation
195 // is itself already impersonating a client of its own. This routine
196 // indicates whether the client is attempting to violate the level of
197 // impersonation granted to it by its client.
198 //
199 // Arguments:
200 //
201 // ImpersonationLevel - Is the impersonation level of the client's
202 // effective token.
203 //
204 // ServerIsRemote - Is a boolean flag indicating whether the client
205 // is requesting impersonation services to a remote system. TRUE
206 // indicates the session is a remote session, FALSE indicates the
207 // session is a local session. Delegation level is necessary to
208 // achieve a remote session.
209 //
210 // Return Value:
211 //
212 // TRUE - Indicates that the impersonation level of the client's client
213 // token is inappropriate for the attempted impersonation.
214 // An error (STATUS_BAD_IMPERSONATION_LEVEL) should be generated.
215 //
216 // FALSE - Indicates the impersonation attempt is not bad, and should
217 // be allowed.
218 //
219 //
220 
221 #define SepBadImpersonationLevel(IL,SIR) (( \
222  ((IL) == SecurityAnonymous) || ((IL) == SecurityIdentification) || \
223  ( (SIR) && ((IL) != SecurityDelegation) ) \
224  ) ? TRUE : FALSE )
