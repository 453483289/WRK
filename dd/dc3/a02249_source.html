235 //
236 // Define DPC importance.
237 //
238 // LowImportance - Queue DPC at end of target DPC queue.
239 // MediumImportance - Queue DPC at end of target DPC queue.
240 // HighImportance - Queue DPC at front of target DPC DPC queue.
241 //
242 // If there is currently a DPC active on the target processor, or a DPC
243 // interrupt has already been requested on the target processor when a
244 // DPC is queued, then no further action is necessary. The DPC will be
245 // executed on the target processor when its queue entry is processed.
246 //
247 // If there is not a DPC active on the target processor and a DPC interrupt
248 // has not been requested on the target processor, then the exact treatment
249 // of the DPC is dependent on whether the host system is a UP system or an
250 // MP system.
251 //
252 // UP system.
253 //
254 // If the DPC is of medium or high importance, the current DPC queue depth
255 // is greater than the maximum target depth, or current DPC request rate is
256 // less the minimum target rate, then a DPC interrupt is requested on the
257 // host processor and the DPC will be processed when the interrupt occurs.
258 // Otherwise, no DPC interupt is requested and the DPC execution will be
259 // delayed until the DPC queue depth is greater that the target depth or the
260 // minimum DPC rate is less than the target rate.
261 //
262 // MP system.
263 //
264 // If the DPC is being queued to another processor and the depth of the DPC
265 // queue on the target processor is greater than the maximum target depth or
266 // the DPC is of high importance, then a DPC interrupt is requested on the
267 // target processor and the DPC will be processed when the interrupt occurs.
268 // Otherwise, the DPC execution will be delayed on the target processor until
269 // the DPC queue depth on the target processor is greater that the maximum
270 // target depth or the minimum DPC rate on the target processor is less than
271 // the target minimum rate.
272 //
273 // If the DPC is being queued to the current processor and the DPC is not of
274 // low importance, the current DPC queue depth is greater than the maximum
275 // target depth, or the minimum DPC rate is less than the minimum target rate,
276 // then a DPC interrupt is request on the current processor and the DPV will
277 // be processed whne the interrupt occurs. Otherwise, no DPC interupt is
278 // requested and the DPC execution will be delayed until the DPC queue depth
279 // is greater that the target depth or the minimum DPC rate is less than the
280 // target rate.
281 //
282 
283 typedef enum _KDPC_IMPORTANCE {
284  LowImportance,
285  MediumImportance,
286  HighImportance
287 } KDPC_IMPORTANCE;

445 // Data structure used to represent client security context for a thread.
446 // This data structure is used to support impersonation.
447 //
448 // THE FIELDS OF THIS DATA STRUCTURE SHOULD BE CONSIDERED OPAQUE
449 // BY ALL EXCEPT THE SECURITY ROUTINES.
450 //
451 
452 typedef struct _SECURITY_CLIENT_CONTEXT {
453  SECURITY_QUALITY_OF_SERVICE SecurityQos;
454  PACCESS_TOKEN ClientToken;
455  BOOLEAN DirectlyAccessClientToken;
456  BOOLEAN DirectAccessEffectiveOnly;
457  BOOLEAN ServerIsRemote;
458  TOKEN_CONTROL ClientTokenControl;
459  } SECURITY_CLIENT_CONTEXT, *PSECURITY_CLIENT_CONTEXT;
460 
461 //
462 // where
463 //
464 // SecurityQos - is the security quality of service information in effect
465 // for this client. This information is used when directly accessing
466 // the client's token. In this case, the information here over-rides
467 // the information in the client's token. If a copy of the client's
468 // token is requested, it must be generated using this information,
469 // not the information in the client's token. In all cases, this
470 // information may not provide greater access than the information
471 // in the client's token. In particular, if the client's token is
472 // an impersonation token with an impersonation level of
473 // "SecurityDelegation", but the information in this field indicates
474 // an impersonation level of "SecurityIdentification", then
475 // the server may only get a copy of the token with an Identification
476 // level of impersonation.
477 //
478 // ClientToken - If the DirectlyAccessClientToken field is FALSE,
479 // then this field contains a pointer to a duplicate of the
480 // client's token. Otherwise, this field points directly to
481 // the client's token.
482 //
483 // DirectlyAccessClientToken - This boolean flag indicates whether the
484 // token pointed to by ClientToken is a copy of the client's token
485 // or is a direct reference to the client's token. A value of TRUE
486 // indicates the client's token is directly accessed, FALSE indicates
487 // a copy has been made.
488 //
489 // DirectAccessEffectiveOnly - This boolean flag indicates whether the
490 // the disabled portions of the token that is currently directly
491 // referenced may be enabled. This field is only valid if the
492 // DirectlyAccessClientToken field is TRUE. In that case, this
493 // value supersedes the EffectiveOnly value in the SecurityQos
494 // FOR THE CURRENT TOKEN ONLY! If the client changes to impersonate
495 // another client, this value may change. This value is always
496 // minimized by the EffectiveOnly flag in the SecurityQos field.
497 //
498 // ServerIsRemote - If TRUE indicates that the server of the client's
499 // request is remote. This is used for determining the legitimacy
500 // of certain levels of impersonation and to determine how to
501 // track context.
502 //
503 // ClientTokenControl - If the ServerIsRemote flag is TRUE, and the
504 // tracking mode is DYNAMIC, then this field contains a copy of
505 // the TOKEN_SOURCE from the client's token to assist in deciding
506 // whether the information at the remote server needs to be
507 // updated to match the current state of the client's security
508 // context.
509 //
510 //
511 // NOTE: At some point, we may find it worthwhile to keep an array of
512 // elements in this data structure, where each element of the
513 // array contains {ClientToken, ClientTokenControl} fields.
514 // This would allow efficient handling of the case where a client
515 // thread was constantly switching between a couple different
516 // contexts - presumably impersonating client's of its own.
517 //
518 // end_ntifs end_ntosp
