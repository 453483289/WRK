396 ;*** StorState - emulate FRSTOR [address]
397 ;
398 ; ARGUMENTS
399 ; (esi) = where to get the environment
400 ; (ebx) = PcTeb
401 ;
402 ;
403 ; DESCRIPTION
404 ; This routine emulates an 80387 FRSTOR (restore state)
405 
406 StorState:
407 ;First we set up the status word so that [CURstk] is initialized.
408 ;The floating-point registers are stored in logical ST(0) - ST(7) order,
409 ;not physical register order. We don't do a full load of the environment
410 ;because we're not ready to use the tag word yet.
411 
412  mov ax, [esi].E32_StatusWord
413  call SetEmStatusWord ;Initialize [CURstk]
414  add esi,size Env80x87_32 ;Skip over environment
415 
416 ;Load of temp real has one difference from real math chip: it is an invalid
417 ;operation to load an unsupported format. By ensuring the exception is
418 ;masked, we will convert unsupported format to Indefinite. Note that the
419 ;mask and [CURerr] will be completely restored by the FLDENV at the end.
420 
421  mov [ebx].CWmask,3FH ;Mask off invalid operation exception
422  mov edi,[ebx].CURstk
423  mov ebp,NumLev
424 FrstorLoadLoop:
425  push esi
426  call LoadTempReal ;In emload.asm
427  pop esi
428  add esi,10 ;Point to next temp real
429 ;;; NextStackElem edi,Frstor
430  cmp edi,INITstk
431  jae short fr_wrap
432  add edi,Reg87Len
433 fr_continue:
434  dec ebp
435  jnz short FrstorLoadLoop
436  sub esi,NumLev*10+size Env80x87_32 ;Point to start of env.
437 ;
438 ; Stor Environment
439 ; (esi) = where to get environment
440 ; (ebx) = PcTeb
441 ;
442 
443  mov ax, [esi].E32_StatusWord
444  call SetEmStatusWord ; set up status word
445  mov ax, [esi].E32_ControlWord
446  call SetControlWord
447  mov ax, [esi].E32_TagWord
448  call UseTagWord
449 
450  mov eax, [esi].E32_CodeOff
451  mov [ebx].PrevCodeOff, eax
452  mov eax, [esi].E32_DataOff
453  mov [ebx].PrevDataOff, eax
454  ret

458  jmp short fr_continue

462 ;
463 ; Given user status word in ax, set into emulator.
464 ; Destroys ebx only.

467 SetEmStatusWord:
468  and ax,7F7FH
469  mov cx,ax
470  and cx,3FH ; set up CURerr in case user
471  mov [ebx].CURerr,cl ; wants to force an exception
472  mov ecx, eax
473  and ecx, not (7 shl 11) ; remove stack field.
474  mov [ebx].StatusWord, cx
475 
476  sub ah, 8 ; adjust for emulator's stack layout
477  and ah, 7 shl 3
478  mov al, ah
479  shr ah, 1
480  add al, ah ; stack field * 3 * 4
481 .erre Reg87Len eq 12
482  and eax, 255 ; eax is now 12*stack number
483  add eax, BEGstk
484  mov [ebx].CURstk, eax
485  ret
486 
487 SetControlWord:
488  and ax,0F3FH ; Limit to valid values
489  mov [ebx].ControlWord, ax ; Store in the emulated control word
490  not al ;Flip mask bits for fast compare
491  and al,3FH ;Limit to valid mask bits
492  mov [ebx].ErrMask,al
493  and eax,(RoundControl + PrecisionControl) shl 8
494 .erre RoundControl eq 1100B
495 .erre PrecisionControl eq 0011B
496  shr eax,6 ;Put PC and RC in bits 2-5
497  mov ecx,_Ki387RoundModeTable

508 ;*** UseTagWord - Set up tags using tag word from environment
509 ;
510 ; ARGUMENTS
511 ; ax - should contain the tag word
512 ;
513 ; Destroys ax,bx,cx,dx,di
514 
515 UseTagWord:
516  ror ax, 2 ; mov Tag(0) into top bits of ax
517  mov edi,INITstk
518  mov ecx, NumLev
519 UseTagLoop:
520  mov dl,bTAG_EMPTY
521  cmp ah, 0c0h ;Is register to be tagged Empty?
522  jae short SetTag ;Yes, go mark it
523  mov dl,[ebx+edi].bTag ;Get current tag
524  cmp dl,bTAG_EMPTY ;Is register currently Empty?
525  je short SetTagNotEmpty ;If so, go figure out tag for it
526 SetTag:
527  mov [ebx+edi].bTag,dl
528 UseTagLoopCheck:
529  sub edi, Reg87Len
530  shl eax, 2
531  loop UseTagLoop
532  ret
533 
534 SetTagEmpty:
535  mov [ebx+edi].bTag, bTAG_EMPTY
536  jmp short UseTagLoopCheck
537 
538 SetTagNotEmpty:
539 ;Register is currently tagged empty, but new tag word says it is not empty.
540 ;Figure out a new tag for it. The rules are:
541 ;
542 ;1. Everything is either normalized or zero--unnormalized formats cannot
543 ;get in. So if the high half mantissa is zero, the number is zero.
544 ;
545 ;2. Although the exponent bias is different, NANs and Infinities are in
546 ;standard IEEE format - exponent is TexpMax, mantissa indicates NAN vs.
547 ;infinity (mantissa for infinity is 800..000H).
548 ;
549 ;3. Denormals have an exponent less than TexpMin.
550 ;
551 ;4. If the low half of the mantissa is zero, it is tagged bTAG_SNGL
552 ;
553 ;5. Everything else is bTAG_VALID
554 
555  cmp [ebx+edi].lManHi, 0
556  mov dl,bTAG_ZERO ;Try zero first
557  jz short SetTag ;Is mantissa zero?
558  mov edx,[ebx+edi].ExpSgn
559  mov dl,bTAG_DEN
560  cmp edx,TexpMin shl 16 ;Is it denormal?
561  jl short SetTag
562  cmp [ebx+edi].lManLo,0 ;Is low half zero?
563 .erre bTAG_VALID eq 1
564 .erre bTAG_SNGL eq 0
565  setnz dl ;if low half==0 then dl=0 else dl=1
566  cmp edx,TexpMax shl 16 ;Is it NAN or Infinity?
567  jl short SetTag ;If not, it's valid
568 .erre (bTAG_VALID - bTAG_SNGL) shl TAG_SHIFT eq (bTAG_NAN - bTAG_INF)
569  shl dl,TAG_SHIFT
570  add dl,bTAG_INF - bTAG_SNGL
571 ;If the low bits were zero we have just changed bTAG_SNGL to bTAG_INF
572 ;If the low bits weren't zero, we changed bTAG_VALID to bTAG_NAN
573 ;See if infinity is really possible: is high half 80..00H?
574  cmp [ebx+edi].lManHi,1 shl 31 ;Is it infinity?
575  jz short SetTag ;Store tag for infinity or NAN
576  mov dl,bTAG_NAN
577  jmp short SetTag

585 LoadTempReal:
586  mov ebx,[esi+4] ;Get high half of mantissa
587  mov cx,[esi+8] ;Get exponent and sign
588  mov esi,[esi] ;Get low half of mantissa
589  mov eax,ecx
590  and ch,7FH ;Mask off sign bit
591  shl ecx,16 ;Move exponent to high end
592  mov ch,ah ;Restore sign
593  jz short ZeroOrDenorm80
594 ;Check for unsupported format: unnormals (MSB not set)
595  or ebx,ebx
596  jns short Unsupported
597  sub ecx,(IexpBias-TexpBias) shl 16 ;Correct the bias
598  cmp ecx,TexpMax shl 16
599  jge short NANorInf80
600 SetupTag:
601  or esi,esi ;Any bits in low half?
602 .erre bTAG_VALID eq 1
603 .erre bTAG_SNGL eq 0
604  setnz cl ;if low half==0 then cl=0 else cl=1
605  jmp short SaveStack
606 
607 NANorInf80:
608  mov cl,bTAG_NAN
609  cmp ebx,1 shl 31 ;Only 1 bit set means infinity
610  jnz short SaveStack
611  or esi,esi
612  jnz short SaveStack
613  mov cl,bTAG_INF
614  jmp short SaveStack
615 
616 ZeroOrDenorm80:
617 ;Exponent is zero. Number is either zero or denormalized
618  or ebx,ebx
619  jnz short ShortNorm80 ;Are top 32 bits zero?
620  or esi,esi ;Are low 32 bits zero too?
621  jnz LongNorm80
622  mov cl,bTAG_ZERO
623  jmp short SaveStack
624 
625 ;This code accepts and works correctly with pseudo-denormals (MSB already set)
626 LongNorm80:
627  xchg ebx,esi ;Shift up 32 bits
628  sub ecx,32 shl 16 ;Correct exponent
629 ShortNorm80:
630  add ecx,(TexpBias-IexpBias+1-31) shl 16 ;Fix up bias
631  bsr edx,ebx ;Scan for MSB
632 ;Bit number in edx ranges from 0 to 31
633  mov cl,dl
634  not cl ;Convert bit number to shift count
635  shld ebx,esi,cl
636  shl esi,cl
637  shl edx,16 ;Move exp. adjustment to high end
638  add ecx,edx ;Adjust exponent
639  jmp short SetUpTag
640 
641 SaveStack:
642  mov eax, PCR[PcTeb]
643  mov [eax].CURstk,edi
644  mov [eax+edi].lManLo,esi
645  mov [eax+edi].lManHi,ebx
646  mov [eax+edi].ExpSgn,ecx
647  mov ebx, eax ; (ebx) = PcTeb
648  ret
