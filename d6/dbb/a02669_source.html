53 /*++
54 
55 Routine Description:
56 
57  Parses the string S as an IPv6 address. See RFC 1884.
58  The basic string representation consists of 8 hex numbers
59  separated by colons, with a couple embellishments:
60  - a string of zero numbers (at most one) may be replaced
61  with a double-colon. Double-colons are allowed at beginning/end
62  of the string.
63  - the last 32 bits may be represented in IPv4-style dotted-octet notation.
64 
65  For example,
66  ::
67  ::1
68  ::157.56.138.30
69  ::ffff:156.56.136.75
70  ff01::
71  ff02::2
72  0:1:2:3:4:5:6:7
73 
74 Arguments:
75 
76  S - RFC 1884 string representation of an IPv6 address.
77 
78  Terminator - Receives a pointer to the character that terminated
79  the conversion.
80 
81  Addr - Receives the IPv6 address.
82 
83 Return Value:
84 
85  TRUE if parsing was successful. FALSE otherwise.
86 
87 --*/

89 {
90  enum { Start, InNumber, AfterDoubleColon } state = Start;
91  const TCHAR *number = NULL;
92  BOOLEAN sawHex;
93  ULONG numColons = 0, numDots = 0, numDigits = 0;
94  ULONG sawDoubleColon = 0;
95  ULONG i = 0;
96  TCHAR c;
97 
98  // There are a several difficulties here. For one, we don't know
99  // when we see a double-colon how many zeroes it represents.
100  // So we just remember where we saw it and insert the zeroes
101  // at the end. For another, when we see the first digits
102  // of a number we don't know if it is hex or decimal. So we
103  // remember a pointer to the first character of the number
104  // and convert it after we see the following character.
105 
106  while (c = *S) {
107 
108  switch (state) {
109  case Start:
110  if (c == _T(':')) {
111

113

115  goto Finish;
116  if (numColons > 0)
117  goto Finish;
118  if (S[1] != _T(':'))
119  goto Finish;
120 
121  sawDoubleColon = 1;
122  numColons = 2;
123  Addr->s6_words[i++] = 0; // pretend it was 0::
124  S++;
125  state = AfterDoubleColon;
126 
127  } else
128  case AfterDoubleColon:
129  if (ISDIGIT(c)) {
130 
131  sawHex = FALSE;
132  number = S;
133  state = InNumber;
134  numDigits = 1;
135 
136  } else if (ISXDIGIT(c)) {
137 
138  if (numDots > 0)
139  goto Finish;
140 
141  sawHex = TRUE;
142  number = S;
143  state = InNumber;
144  numDigits = 1;
145 
146  } else
147  goto Finish;
148  break;
149 
150  case InNumber:
151  if (ISDIGIT(c)) {
152 
153  numDigits++;
154 
155  // remain in InNumber state
156 
157  } else if (ISXDIGIT(c)) {
158 
159  numDigits++;
160 
161  if (numDots > 0)
162  goto Finish;
163 
164  sawHex = TRUE;
165  // remain in InNumber state;
166 
167  } else if (c == _T(':')) {
168 
169  if (numDots > 0)
170  goto Finish;
171  if (numColons > 6)
172  goto Finish;
173 
174  if (S[1] == _T(':')) {

177  goto Finish;
178  if (numColons > 5)
179  goto Finish;
180

182  numColons += 2;
183  S++;
184  state = AfterDoubleColon;
185 
186  } else {
187  numColons++;
188  state = Start;
189  }
190 
191  } else if (c == _T('.')) {
192 
193  if (sawHex)
194  goto Finish;
195  if (numDots > 2)
196  goto Finish;
197  if (numColons > 6)
198  goto Finish;
199  numDots++;
200  state = Start;
201 
202  } else
203  goto Finish;
204  break;
205  }
206 
207  // If we finished a number, parse it.
208 
209  if ((state != InNumber) && (number != NULL)) {
210 
211  // Note either numDots > 0 or numColons > 0,
212  // because something terminated the number.
213 
214  if (numDots == 0) {
215  if (numDigits > 4)
216  return STATUS_INVALID_PARAMETER;
217  Addr->s6_words[i++] =
218  RtlUshortByteSwap((USHORT) _tcstol(number, NULL, 16));
219  } else {
220  ULONG Temp;
221  if (numDigits > 3)
222  return STATUS_INVALID_PARAMETER;
223  Temp = _tcstol(number, NULL, 10);
224  if (Temp > 255)
225  return STATUS_INVALID_PARAMETER;
226  Addr->s6_bytes[2*i + numDots-1] = (UCHAR) Temp;
227  }
228  }
229 
230  S++;
231  }
232 
233 Finish:
234  *Terminator = S;
235 
236  // Check that we have a complete address.
237 
238  if (numDots == 0)
239  ;
240  else if (numDots == 3)
241  numColons++;
242  else
243  return STATUS_INVALID_PARAMETER;
244 
245  if (sawDoubleColon)
246  ;
247  else if (numColons == 7)
248  ;
249  else
250  return STATUS_INVALID_PARAMETER;
251 
252  // Parse the last number, if necessary.
253 
254  if (state == InNumber) {
255 
256  if (numDots == 0) {
257  if (numDigits > 4)
258  return STATUS_INVALID_PARAMETER;
259  Addr->s6_words[i] =
260  RtlUshortByteSwap((USHORT) _tcstol(number, NULL, 16));
261  } else {

264  return STATUS_INVALID_PARAMETER;
265  Temp = _tcstol(number, NULL, 10);
266  if (Temp > 255)
267  return STATUS_INVALID_PARAMETER;
268  Addr->s6_bytes[2*i + numDots] = (UCHAR) Temp;
269  }
270 
271  } else if (state == AfterDoubleColon) {
272 
273  Addr->s6_words[i] = 0; // pretend it was ::0
274 
275  } else
276  return STATUS_INVALID_PARAMETER;
277

279 
280  if (sawDoubleColon) {
281 
282  RtlMoveMemory(&Addr->s6_words[sawDoubleColon + 8 - numColons],
283  &Addr->s6_words[sawDoubleColon],
284  (numColons - sawDoubleColon) * sizeof(USHORT));
285  RtlZeroMemory(&Addr->s6_words[sawDoubleColon],
286  (8 - numColons) * sizeof(USHORT));
287  }
288 
289  return STATUS_SUCCESS;
290 }

301 /*++
302 
303 Routine Description:
304 
305  Parsing a human-readable string to Address, port number and scope id.
306 
307  The syntax is address%scope-id or [address%scope-id]:port, where
308  the scope-id and port are optional.
309  Note that since the IPv6 address format uses a varying number
310  of ':' characters, the IPv4 convention of address:port cannot
311  be supported without the braces.
312 
313 Arguments:
314 
315  AddressString - Points to the zero-terminated human-readable string.
316 
317  Address - Receive address part (in6_addr) of this address.
318 
319  ScopeId - Receive scopeid of this address. If there is no scope id in
320  the address string, 0 is returned.
321 
322  Port - Receive port number of this address. If there is no port number
323  in the string, 0 is returned. Port is returned in network byte order.
324 
325 Return Value:
326 
327  NT_STATUS - STATUS_SUCCESS if successful, NT error code if not.
328 
329 --*/
330 
331 {
332  LPTSTR Terminator;
333  ULONG TempScopeId;
334  USHORT TempPort;
335  TCHAR Ch;
336  BOOLEAN ExpectBrace;
337 
338  //
339  // Quick sanity checks.
340  //
341  if ((AddressString == NULL) ||
342  (Address == NULL) ||
343  (ScopeId == NULL) ||
344  (Port == NULL)) {
345  return STATUS_INVALID_PARAMETER;
346  }
347 
348  TempPort = 0;
349  TempScopeId = 0;
350  ExpectBrace = FALSE;
351  if (*AddressString == _T('[')) {
352  ExpectBrace = TRUE;
353  AddressString++;
354  }
355 
356  if (!NT_SUCCESS(RtlIpv6StringToAddressT(AddressString,
357  &Terminator,
358  Address))) {
359  return STATUS_INVALID_PARAMETER;
360  }
361 
362  //
363  // We have parsed the address, check for a scope-id.
364  //
365  if (*Terminator == _T('%')) {
366  Terminator++;
367  Ch = *Terminator;
368  if (!ISDIGIT(Ch)) {
369  return STATUS_INVALID_PARAMETER;
370  }
371  while ((Ch != 0) && (Ch != _T(']'))) {
372  if (!ISDIGIT(Ch)) {
373  return STATUS_INVALID_PARAMETER;
374  }
375  //
376  // first check the possibility of overflow
377  //
378  if (((ULONGLONG)TempScopeId * 10 + Ch - _T('0')) >
379  0xFFFFFFFF) {
380  return STATUS_INVALID_PARAMETER;
381  }
382  TempScopeId = 10 * TempScopeId + (Ch - _T('0'));
383  Terminator++;
384  Ch = *Terminator;
385  }
386 
387  }
388 
389  //
390  // When we come here, the current char should either be the
391  // end of the string or ']' if expectbrace is true.
392  //
393  if (*Terminator == _T(']')) {
394  if (!ExpectBrace) {
395  return STATUS_INVALID_PARAMETER;
396  }
397  ExpectBrace = FALSE;
398  Terminator++;
399  //
400  // See if we have a port to parse.
401  //
402  if (*Terminator == _T(':')) {
403  USHORT Base;
404  Terminator++;
405  Base = 10;
406  if (*Terminator == _T('0')) {
407  Base = 8;
408  Terminator++;
409  if ((*Terminator == _T('x')) ||
410  (*Terminator == _T('X'))) {
411  Base = 16;
412  Terminator++;
413  }
414  }
415  Ch = *Terminator;
416  while (Ch != 0) {
417  if (ISDIGIT(Ch) && (Ch - _T('0')) < Base) {
418  //
419  // check the possibility for overflow first
420  //
421  if (((ULONG)TempPort * Base + Ch - _T('0')) >
422  0xFFFF) {
423  return STATUS_INVALID_PARAMETER;
424  }
425  TempPort = (TempPort * Base) + (Ch - _T('0'));
426  } else if (Base == 16 && ISXDIGIT(Ch)) {
427  //
428  // check the possibility for overflow
429  //
430  if ((((ULONG)TempPort << 4) + Ch + 10 -
431  (ISLOWER(Ch)? _T('a') : _T('A'))) > 0xFFFF) {
432  return STATUS_INVALID_PARAMETER;
433  }
434  TempPort = (TempPort << 4);
435  TempPort += Ch + 10 - (ISLOWER(Ch)? _T('a') : _T('A'));
436  } else {
437  return STATUS_INVALID_PARAMETER;
438  }
439  Terminator++;
440  Ch = *Terminator;
441  }
442  }
443  }
444 
445  //
446  // We finished parsing address, scope id and port number. We are expecting the
447  // end of the string.
448  //
449  if ((*Terminator != 0) || ExpectBrace) {
450  return STATUS_INVALID_PARAMETER;
451  }
452 
453  //
454  // Now construct the address.
455  //
456  *Port = RtlUshortByteSwap(TempPort);
457  *ScopeId = TempScopeId;
458  return STATUS_SUCCESS;
459 }

462 RtlIpv4StringToAddressT(
463  IN LPCTSTR String,
464  IN BOOLEAN Strict,
465  OUT LPCTSTR *Terminator,
466  OUT struct in_addr *Addr

468 
469 /*++
470 
471 Routine Description:
472 
473  This function interprets the character string specified by the cp
474  parameter. This string represents a numeric Internet address
475  expressed in the Internet standard ".'' notation. The value
476  returned is a number suitable for use as an Internet address. All
477  Internet addresses are returned in network order (bytes ordered from
478  left to right).
479 
480  Internet Addresses
481 
482  Values specified using the "." notation take one of the following
483  forms:
484 
485  a.b.c.d a.b.c a.b a
486 
487  When four parts are specified, each is interpreted as a byte of data
488  and assigned, from left to right, to the four bytes of an Internet
489  address. Note that when an Internet address is viewed as a 32-bit
490  integer quantity on the Intel architecture, the bytes referred to
491  above appear as "d.c.b.a''. That is, the bytes on an Intel
492  processor are ordered from right to left.
493 
494  Note: The following notations are only used by Berkeley, and nowhere
495  else on the Internet. In the interests of compatibility with their
496  software, they are supported as specified.
497 
498  When a three part address is specified, the last part is interpreted
499  as a 16-bit quantity and placed in the right most two bytes of the
500  network address. This makes the three part address format
501  convenient for specifying Class B network addresses as
502  "128.net.host''.
503 
504  When a two part address is specified, the last part is interpreted
505  as a 24-bit quantity and placed in the right most three bytes of the
506  network address. This makes the two part address format convenient
507  for specifying Class A network addresses as "net.host''.
508 
509  When only one part is given, the value is stored directly in the
510  network address without any byte rearrangement.
511 
512 Arguments:
513 
514  String - A character string representing a number expressed in the
515  Internet standard "." notation.
516 
517  Strict - If TRUE, the string must be dotted-decimal with 4 parts.
518  Otherwise, any of the four forms are allowed, with decimal,
519  octal, or hex.
520 
521  Terminator - Receives a pointer to the character that terminated
522  the conversion.
523 
524  Addr - Receives a pointer to the structure to fill in with
525  a suitable binary representation of the Internet address given.
526 
527 Return Value:
528 
529  TRUE if parsing was successful. FALSE otherwise.
530 
531 --*/
532 
533 {
534  ULONG val, n;
535  LONG base;
536  TCHAR c;
537  ULONG parts[4], *pp = parts;
538  BOOLEAN sawDigit;
539 
540 again:
541  //
542  // We must see at least one digit for address to be valid.
543  //
544  sawDigit=FALSE;
545 
546  //
547  // Collect number up to ``.''.
548  // Values are specified as for C:
549  // 0x=hex, 0=octal, other=decimal.
550  //
551  val = 0;
552  base = 10;
553  if (*String == _T('0')) {
554  String++;
555  if (ISDIGIT(*String)) {
556  base = 8;
557  } else if (*String == _T('x') || *String == _T('X')) {
558  base = 16;
559  String++;
560  } else {
561  //
562  // It is still decimal but we saw the digit
563  // and it was 0.
564  //
565  sawDigit = TRUE;
566  }
567  }
568  if (Strict && (base != 10)) {
569  *Terminator = String;
570  return STATUS_INVALID_PARAMETER;
571  }
572 
573  while ((c = *String)!= 0) {
574  ULONG newVal;
575  if (ISDIGIT(c) && ((c - _T('0')) < base)) {
576  newVal = (val * base) + (c - _T('0'));
577  } else if ((base == 16) && ISXDIGIT(c)) {
578  newVal = (val << 4) + (c + 10 - (ISLOWER(c) ? _T('a') : _T('A')));
579  } else {
580  break;
581  }
582 
583  //
584  // Protect from overflow
585  //
586  if (newVal < val) {
587  *Terminator = String;
588  return STATUS_INVALID_PARAMETER;
589  }
590  String++;
591  sawDigit = TRUE;
592  val = newVal;
593  }
594  if (*String == _T('.')) {
595  //
596  // Internet format:
597  // a.b.c.d
598  // a.b.c (with c treated as 16-bits)
599  // a.b (with b treated as 24 bits)
600  //
601  if (pp >= parts + 3) {
602  *Terminator = String;
603  return STATUS_INVALID_PARAMETER;
604  }
605  *pp++ = val, String++;
606 
607  //
608  // Check if we saw at least one digit.
609  //
610  if (!sawDigit) {
611  *Terminator = String;
612  return STATUS_INVALID_PARAMETER;
613  }
614 
615  goto again;
616  }
617 
618  //
619  // Check if we saw at least one digit.
620  //
621  if (!sawDigit) {
622  *Terminator = String;
623  return STATUS_INVALID_PARAMETER;
624  }
625  *pp++ = val;
626 
627  //
628  // Concoct the address according to
629  // the number of parts specified.
630  //
631  n = (ULONG)(pp - parts);
632  if (Strict && (n != 4)) {
633  *Terminator = String;
634  return STATUS_INVALID_PARAMETER;
635  }
636  switch ((int) n) {
637 
638  case 1: /* a -- 32 bits */
639  val = parts[0];
640  break;
641 
642  case 2: /* a.b -- 8.24 bits */
643  if ((parts[0] > 0xff) || (parts[1] > 0xffffff)) {
644  *Terminator = String;
645  return STATUS_INVALID_PARAMETER;
646  }
647  val = (parts[0] << 24) | (parts[1] & 0xffffff);
648  break;
649 
650  case 3: /* a.b.c -- 8.8.16 bits */
651  if ((parts[0] > 0xff) || (parts[1] > 0xff) ||
652  (parts[2] > 0xffff)) {
653  *Terminator = String;
654  return STATUS_INVALID_PARAMETER;
655  }
656  val = (parts[0] << 24) | ((parts[1] & 0xff) << 16) |
657  (parts[2] & 0xffff);
658  break;
659 
660  case 4: /* a.b.c.d -- 8.8.8.8 bits */
661  if ((parts[0] > 0xff) || (parts[1] > 0xff) ||
662  (parts[2] > 0xff) || (parts[3] > 0xff)) {
663  *Terminator = String;
664  return STATUS_INVALID_PARAMETER;
665  }
666  val = (parts[0] << 24) | ((parts[1] & 0xff) << 16) |
667  ((parts[2] & 0xff) << 8) | (parts[3] & 0xff);
668  break;
669 
670  default:
671  *Terminator = String;
672  return STATUS_INVALID_PARAMETER;
673  }
674 
675  val = RtlUlongByteSwap(val);
676  *Terminator = String;
677  Addr->s_addr = val;
678 
679  return STATUS_SUCCESS;
680 }
