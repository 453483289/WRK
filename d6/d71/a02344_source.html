39

41  FileHandle parameter. The information returned in the buffer is defined
42  by the FsInformationClass parameter. The legal values for this parameter
43  are as follows:

46 
47  o FileFsSizeInformation
48 
49  o FileFsDeviceInformation

57

59 
60  FsInformation - Supplies a buffer to receive the requested information
61  returned about the volume.
62 
63  Length - Supplies the length, in bytes, of the FsInformation buffer.
64 
65  FsInformationClass - Specifies the type of information which should be
66  returned about the volume.

203  // Make a special check here to determine whether this is a synchronous

278  }

286  //
287  // An error occurred attempting to write into one of the caller's
288  // buffers. Simply indicate that the error occurred, and fall
289  // through.
290  //
291

296  // If this operation was performed as synchronous I/O, then release

301  IopReleaseFileObjectLock( fileObject );
302  }
303 
304  //
305  // Now simply cleanup and return the final status of the operation.
306  //
307 
308  ObDereferenceObject( fileObject );

334

339 
340  //
341  // An exception was incurred while allocating the intermediary
342  // system buffer or while copying the caller's data into the
343  // buffer. Cleanup and return an appropriate error status code.
344  //
345 
346  status = GetExceptionCode();
347

415  ExFreePool( event );

417

419

423  irp->Tail.Overlay.Thread = CurrentThread;

428  //

433  } else {

493  // IRP.
494  //

502  //
503

505  irp,

521 
522  status = IopSynchronousApiServiceTail( status,
523  event,
524  irp,
525  requestorMode,
526  &localIoStatus,
527  IoStatusBlock );
528  }

530  return status;

542 /*++
543 
544 Routine Description:
545 
546  This service changes information about the volume "mounted" on the device
547  specified by the FileHandle parameter. The information to be changed is
548  in the FsInformation buffer. Its contents are defined by the FsInformation-
549  Class parameter, whose values may be as follows:
550 
551  o FileFsLabelInformation

564 
565  FsInformationClass - Specifies the type of information which should be
566  changed about the volume.

794  irp->RequestorMode = requestorMode;

801  irp->UserEvent = (PKEVENT) NULL;
802  irp->UserIosb = IoStatusBlock;
803  } else {
804  irp->UserEvent = event;
