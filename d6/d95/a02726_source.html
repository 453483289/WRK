26 //
27 // VFM_ flags control how a verifier message is handled.
28 //
29 // VFM_FLAG_INITIALIZED - Set when the error template has been updated with
30 // error-class information
31 //
32 // VFM_FLAG_BEEP - Set if error should beep in debugger
33 //
34 // VFM_FLAG_ZAPPED - Set if error was "zapped" (ie prints instead of
35 // stops) via debugger
36 //
37 // VFM_FLAG_CLEARED - Set if error was cleared (disabled) in debugger
38 //
39 // VFM_DEPLOYMENT_FAILURE - Set if the error is severe enough to warrant
40 // removal of the driver from a production system
41 //
42 // VFM_LOGO_FAILURE - Set if the error should disallow certification
43 // for the hardware or the driver
44 //
45 // VFM_IGNORE_DRIVER_LIST - Set if error should fire regardless of whether
46 // the offending driver is being verified or not.
47 //
48 
49 #define VFM_FLAG_INITIALIZED 0x00000001
50 #define VFM_FLAG_BEEP 0x00000002
51 #define VFM_FLAG_ZAPPED 0x00000004
52 #define VFM_FLAG_CLEARED 0x00000008
53 #define VFM_DEPLOYMENT_FAILURE 0x00000010
54 #define VFM_LOGO_FAILURE 0x00000020
55 #define VFM_IGNORE_DRIVER_LIST 0x00000040
