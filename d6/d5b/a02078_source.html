83  )
84 /*++
85 
86 Routine Description:
87 
88  Find a child object in an object list. Child List must be sorted
89  based on the name. (for new hives format)
90 
91 Arguments:
92 
93  Hive - pointer to hive control structure for hive of interest
94 
95  List - pointer to mapped in list structure
96 
97  Count - number of elements in list structure
98 
99  Name - name of child object to find
100 
101  ChildIndex - pointer to variable to receive index for child;
102 
103  CellIndex - pointer to receive the index of the child.
104  On return, this is:
105  HCELL_INDEX for the found cell
106  HCELL_NIL if not found
107 
108 
109 Return Value:
110 
111  TRUE - success
112  FALSE - error, insufficient resources
113 
114 Notes:
115 
116  ChildIndex is always filled with the position where Name should be in the list.
117  The difference whether Name is in the list or not is made upon CellIndex
118  - CellIndex == HCELL_NIL ==> Name not found in the list
119  - CellIndex <> HCELL_NIL ==> Name already exists in the list
120 
121 --*/
122 {
123  PCM_KEY_VALUE pchild;
124  UNICODE_STRING Candidate;
125  LONG Result;
126  PCELL_DATA List = NULL;
127  ULONG Current;
128  HCELL_INDEX CellToRelease = HCELL_NIL;
129  BOOLEAN ReturnValue = FALSE;
130 
131  CM_PAGED_CODE();
132 
133  if (ChildList->Count != 0) {
134  List = (PCELL_DATA)HvGetCell(Hive,ChildList->List);
135  if( List == NULL ) {
136  //
137  // we could not map the view containing the cell
138  //
139  *CellIndex = HCELL_NIL;
140  return FALSE;
141  }
142 
143  //
144  // old plain hive; simulate a for
145  //
146  Current = 0;
147 
148  while( TRUE ) {
149 
150  if( CellToRelease != HCELL_NIL ) {
151  HvReleaseCell(Hive,CellToRelease);
152  CellToRelease = HCELL_NIL;
153  }
154  pchild = (PCM_KEY_VALUE)HvGetCell(Hive, List->u.KeyList[Current]);
155  if( pchild == NULL ) {
156  //
157  // we could not map the view containing the cell
158  //
159  *CellIndex = HCELL_NIL;
160  ReturnValue = FALSE;
161  goto JustReturn;
162  }
163  CellToRelease = List->u.KeyList[Current];
164 
165  if (pchild->Flags & VALUE_COMP_NAME) {
166  Result = CmpCompareCompressedName(Name,
167  pchild->Name,
168  pchild->NameLength,
169  0);
170  } else {
171  Candidate.Length = pchild->NameLength;
172  Candidate.MaximumLength = Candidate.Length;
173  Candidate.Buffer = pchild->Name;
174  Result = RtlCompareUnicodeString(Name,
175  &Candidate,
176  TRUE);
177  }
178 
179  if (Result == 0) {
180  //
181  // Success, return data to caller and exit
182  //
183 
184  if (ARGUMENT_PRESENT(ChildIndex)) {
185  *ChildIndex = Current;
186  }
187  *CellIndex = List->u.KeyList[Current];
188  ReturnValue = TRUE;
189  goto JustReturn;
190  }
191  //
192  // compute the next index to try: old'n plain hive; go on
193  //
194  Current++;
195  if( Current == ChildList->Count ) {
196  //
197  // we've reached the end of the list
198  //
199  if (ARGUMENT_PRESENT(ChildIndex)) {
200  *ChildIndex = Current;
201  }
202  //
203  // nicely return
204  //
205  *CellIndex = HCELL_NIL;
206  ReturnValue = TRUE;
207  goto JustReturn;
208  }
209  }
210  }
211  //
212  // in the new design we shouldn't get here; we should exit the while loop with return
213  //
214  ASSERT( ChildList->Count == 0 );
215 
216  // add it first; as it's the only one
217  if (ARGUMENT_PRESENT(ChildIndex)) {
218  *ChildIndex = 0;
219  }
220  *CellIndex = HCELL_NIL;
221  return TRUE;
222 
223 JustReturn:
224  if( List != NULL ) {
225  HvReleaseCell(Hive,ChildList->List);
226  }
227  if( CellToRelease != HCELL_NIL ) {
228  HvReleaseCell(Hive,CellToRelease);
229  }
230  return ReturnValue;
231

234 BOOLEAN
235 CmpGetValueData(IN PHHIVE Hive,
236  IN PCM_KEY_VALUE Value,
237  OUT PULONG realsize,
238  IN OUT PVOID *Buffer,
239  OUT PBOOLEAN Allocated,
240  OUT PHCELL_INDEX CellToRelease
241  )
242 /*++
243 
244 Routine Description:
245 
246  Retrieves the real valueData, given the key value.
247 
248 
249 Arguments:
250 
251  Hive - pointer to hive control structure for hive of interest
252 
253  Value - CM_KEY_VALUE to retrieve the data for.
254 
255  realsize - the actual size of the data (in bytes)
256 
257  Buffer - pointer to the data; if the cell is a BIG_CELL
258  we should allocate a buffer
259 
260  Allocated - here we signal the caller that he has to free the
261  buffer on return;
262  TRUE - a new buffer was allocated to gather together the BIG_CELL data
263  FALSE - Buffer points directly in the hive, the caller shouldn't free it
264 
265  CellToRelease - Cell to release after finishing work with Buffer
266 
267 Return Value:
268 
269  TRUE - success
270 
271  FALSE - not enough resources available; (to map a cell or to allocate the buffer)
272 
273 Notes:
274 
275  The caller is responsible to remove the buffer, when Allocated is set on TRUE on return;
276 
277 --*/
278 {
279 
280  CM_PAGED_CODE();
281 
282  ASSERT_KEY_VALUE(Value);
283  //
284  // normally we don't allocate buffer
285  //
286  *Allocated = FALSE;
287  *Buffer = NULL;
288  *CellToRelease = HCELL_NIL;
289 
290  //
291  // check for small values
292  //
293  if( CmpIsHKeyValueSmall(*realsize, Value->DataLength) == TRUE ) {
294  //
295  // data is stored inside the cell
296  //
297  *Buffer = &Value->Data;
298  return TRUE;
299  }
300 
301  //
302  // check for big values
303  //
304  if( CmpIsHKeyValueBig(Hive,*realsize) == TRUE ) {
305  //
306  //
307  //
308  PCM_BIG_DATA BigData = NULL;
309  PUCHAR WorkBuffer = NULL;
310  ULONG Length;
311  USHORT i;
312  PUCHAR PartialData;
313  PHCELL_INDEX Plist = NULL;
314  BOOLEAN bRet = TRUE;
315 
316  try {
317  BigData = (PCM_BIG_DATA)HvGetCell(Hive,Value->Data);
318  if( BigData == NULL ) {
319  //
320  // cannot map view containing the cell; bail out
321  //
322  bRet = FALSE;
323  leave;
324  }
325 
326  ASSERT_BIG_DATA(BigData);
327 
328  Plist = (PHCELL_INDEX)HvGetCell(Hive,BigData->List);
329  if( Plist == NULL ) {
330  //
331  // cannot map view containing the cell; bail out
332  //
333  bRet = FALSE;
334  leave;
335  }
336 
337  Length = Value->DataLength;
338  //
339  // sanity check
340  //
341  ASSERT( Length <= (ULONG)(BigData->Count * CM_KEY_VALUE_BIG) );
342 
343  //
344  // allocate a buffer to merge bring all the pieces together
345  //
346  WorkBuffer = (PUCHAR)ExAllocatePoolWithTag(PagedPool, Length, CM_POOL_TAG);
347  if( WorkBuffer == NULL ){
348  bRet = FALSE;
349  leave;
350  }
351 
352  for(i=0;i<BigData->Count;i++) {
353  //
354  // sanity check
355  //
356  ASSERT( Length > 0 );
357 
358  PartialData = (PUCHAR)HvGetCell(Hive,Plist[i]);
359  if( PartialData == NULL ){
360  //
361  // cannot map view containing the cell; bail out
362  //
363  ExFreePool(WorkBuffer);
364  bRet = FALSE;
365  leave;
366  }
367 
368  //
369  // copy this piece of data to the work buffer
370  //
371  RtlCopyMemory(WorkBuffer + CM_KEY_VALUE_BIG*i,PartialData,(Length>CM_KEY_VALUE_BIG)?CM_KEY_VALUE_BIG:Length);
372  HvReleaseCell(Hive,Plist[i]);
373 
374  //
375  // adjust the data still to copy.
376  // All cells in Plist should be of size CM_KEY_VALUE_BIG, except the last one, which is the remaining
377  //
378  Length -= CM_KEY_VALUE_BIG;
379  }
380  } finally {
381  if( BigData != NULL ) {
382  HvReleaseCell(Hive,Value->Data);
383  if( Plist != NULL ) {
384  HvReleaseCell(Hive,BigData->List);
385  }
386  }
387  }
388  if( !bRet ) {
389  return FALSE;
390  }
391  //
392  // if we are here; we successfuly have copied all data into WorkBuffer.
393  // update the return buffer and return; Caller is responsible to free the return buffer
394  // We signal the caller by setting Allocated on TRUE
395  //
396  *Buffer = WorkBuffer;
397  *Allocated = TRUE;
398  return TRUE;
399  }
400 
401  //
402  // normal, old plain case
403  //
404  *Buffer = HvGetCell(Hive,Value->Data);
405  if( *Buffer == NULL ) {
406  //
407  // insufficient resources to map the view containing this cell
408  //
409  return FALSE;
410  }
411  //
412  // signal to the caller to release this cell after finishing with buffer
413  //
414  *CellToRelease = Value->Data;
415 
416  return TRUE;
417 }

486 NTSTATUS
487 CmpAddValueToList(
488  IN PHHIVE Hive,
489  IN HCELL_INDEX ValueCell,
490  IN ULONG Index,
491  IN ULONG Type,
492  IN OUT PCHILD_LIST ChildList
493  )
494 /*++
495 
496 Routine Description:
497 
498  Adds a value to the value list, keeping the list sorted
499  (for new hives format)
500 
501 Arguments:
502 
503  Hive - pointer to hive control structure for hive of interest
504 
505  ValueCell - value index
506 
507  Index - index at which to add the value
508 
509  ChildList - pointer to the list of values
510 
511 
512 Return Value:
513 
514  STATUS_SUCCESS - success
515 
516  STATUS_INSUFFICIENT_RESOURCES - an error occured
517 
518 --*/
519 {
520  HCELL_INDEX NewCell;
521  ULONG count;
522  ULONG AllocateSize;
523  ULONG i;
524  PCELL_DATA pdata;
525 
526  CM_PAGED_CODE();
527 
528  //
529  // we have the lock exclusive or nobody is operating inside this hive
530  //
531  ASSERT_HIVE_FLUSHER_LOCKED((PCMHIVE)Hive);
532 
533  //
534  // sanity check for index range
535  //
536  ASSERT( (((LONG)Index) >= 0) && (Index <= ChildList->Count) );
537 
538  count = ChildList->Count;
539  count++;
540  if (count > 1) {
541 
542  ASSERT_CELL_DIRTY(Hive,ChildList->List);
543 
544  if (count < CM_MAX_REASONABLE_VALUES) {
545 
546  //
547  // A reasonable number of values, allocate just enough
548  // space.
549  //
550 
551  AllocateSize = count * sizeof(HCELL_INDEX);
552  } else {
553 
554  //
555  // An excessive number of values, pad the allocation out
556  // to avoid fragmentation. (if there's this many values,
557  // there'll probably be more pretty soon)
558  //
559  AllocateSize = ROUND_UP(count, CM_MAX_REASONABLE_VALUES) * sizeof(HCELL_INDEX);
560  if (AllocateSize > HBLOCK_SIZE) {
561  AllocateSize = ROUND_UP(AllocateSize, HBLOCK_SIZE);
562  }
563  }
564 
565  NewCell = HvReallocateCell(
566  Hive,
567  ChildList->List,
568  AllocateSize
569  );
570  } else {
571  NewCell = HvAllocateCell(Hive, sizeof(HCELL_INDEX), Type,ValueCell);
572  }
573 
574  //
575  // put ourselves on the list
576  //
577  if (NewCell != HCELL_NIL) {
578  // sanity
579  ChildList->List = NewCell;
580 
581  pdata = HvGetCell(Hive, NewCell);
582  if( pdata == NULL ) {
583  //
584  // we couldn't map a view for the bin containing this cell
585  //
586 
587  //
588  // normally this shouldn't happen as we just allocated ValueCell
589  // i.e. the bin containing NewCell should be mapped in memory at this point.
590  //
591  ASSERT( FALSE );
592  return STATUS_INSUFFICIENT_RESOURCES;
593  }
594 
595  //
596  // make room for the new cell; move values in the reverse order !
597  // adding at the end makes this a nop
598  //
599  for( i = count - 1; i > Index; i-- ) {
600  pdata->u.KeyList[i] = pdata->u.KeyList[i-1];
601  }
602  pdata->u.KeyList[Index] = ValueCell;
603  ChildList->Count = count;
604 
605  HvReleaseCell(Hive,NewCell);
606  // sanity
607  ASSERT_CELL_DIRTY(Hive,ValueCell);
608 
609  } else {
610  return STATUS_INSUFFICIENT_RESOURCES;
611  }
612 
613  return STATUS_SUCCESS;
614 }

709 /*++
710 
711 Routine Description:
712 
713  Marks the cell(s) storing the value data as dirty;
714  Knows how to deal with bigcells
715 
716 Arguments:
717 
718  Hive - pointer to hive control structure for hive of interest
719 
720  Value - CM_KEY_VALUE to retrieve the data for.
721 
722 Return Value:
723 
724  TRUE - success
725  FALSE - failure to mark all the cells involved;
726 
727 --*/
728 {
729  ULONG realsize;
730 
731  CM_PAGED_CODE();
732 
733  //
734  // we have the lock exclusive or nobody is operating inside this hive
735  //
736  ASSERT_HIVE_FLUSHER_LOCKED((PCMHIVE)Hive);
737 
738  ASSERT_KEY_VALUE(Value);
739 
740  if( Value->Data != HCELL_NIL ) {
741  //
742  // Could be that value was just partially initialized (CmpSetValueKeyNew case)
743  //
744  //
745  // check for small values
746  //
747  if( CmpIsHKeyValueSmall(realsize, Value->DataLength) == TRUE ) {
748  //
749  // data is stored inside the cell
750  //
751  return TRUE;
752  }
753 
754  //
755  // check for big values
756  //
757  if( CmpIsHKeyValueBig(Hive,realsize) == TRUE ) {
758  //
759  //
760  //
761  PCM_BIG_DATA BigData;
762  PHCELL_INDEX Plist;
763  USHORT i;
764 
765  BigData = (PCM_BIG_DATA)HvGetCell(Hive,Value->Data);
766  if( BigData == NULL ) {
767  //
768  // cannot map view containing the cell; bail out
769  //
770  return FALSE;
771  }
772 
773  ASSERT_BIG_DATA(BigData);
774 
775  if( BigData->List != HCELL_NIL ) {
776  Plist = (PHCELL_INDEX)HvGetCell(Hive,BigData->List);
777  if( Plist == NULL ) {
778  //
779  // cannot map view containing the cell; bail out
780  //
781  HvReleaseCell(Hive,Value->Data);
782  return FALSE;
783  }
784 
785 
786  for(i=0;i<BigData->Count;i++) {
787  //
788  // mark this chunk dirty
789  //
790  if( Plist[i] != HCELL_NIL ) {
791  if (! HvMarkCellDirty(Hive, Plist[i],FALSE)) {
792  HvReleaseCell(Hive,Value->Data);
793  HvReleaseCell(Hive,BigData->List);
794  return FALSE;
795  }
796  }
797  }
798  //
799  // mark the list as dirty
800  //
801  if (! HvMarkCellDirty(Hive, BigData->List,FALSE)) {
802  HvReleaseCell(Hive,Value->Data);
803  HvReleaseCell(Hive,BigData->List);
804  return FALSE;
805  }
806  //
807  // we can safely remove it here as it is now dirty/pinned
808  //
809  HvReleaseCell(Hive,BigData->List);
810  }
811  //
812  // we don't need this cell anymore
813  //
814  HvReleaseCell(Hive,Value->Data);
815  //
816  // fall through to mark the cell itself as dirty
817  //
818  }
819 
820  //
821  // Data is a HCELL_INDEX; mark it dirty
822  //
823  if (! HvMarkCellDirty(Hive, Value->Data,FALSE)) {
824  return FALSE;
825  }
826  }
827 
828  return TRUE;
829 }
