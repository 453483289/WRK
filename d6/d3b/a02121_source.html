159 ExpComputeLookasideDepth (
160  IN PGENERAL_LOOKASIDE Lookaside,
161  IN ULONG Misses,
162  IN ULONG ScanPeriod

165 /*++
166 
167 Routine Description:
168 
169  This function computes the target depth of a lookaside list given the
170  total allocations and misses during the last scan period and the current
171  depth.
172 
173 Arguments:
174 
175  Lookaside - Supplies a pointer to a lookaside list descriptor.
176 
177  Misses - Supplies the total number of allocate misses.
178 
179  ScanPeriod - Supplies the scan period in seconds.
180 
181 Return Value:
182 
183  None.
184 
185 --*/
186 
187 {
188 
189  ULONG Allocates;
190  ULONG Delta;
191  USHORT MaximumDepth;
192  ULONG Ratio;
193  LONG Target;
194 
195  //
196  // Compute the total number of allocations and misses per second for
197  // this scan period.
198  //
199 
200  Allocates = Lookaside->TotalAllocates - Lookaside->LastTotalAllocates;
201  Lookaside->LastTotalAllocates = Lookaside->TotalAllocates;
202 
203  //
204  // If the verifier is enabled, disable lookasides so driver problems can
205  // be isolated. Otherwise, compute the target lookaside list depth.
206  //
207 
208  if (ExMinimumLookasideDepth == 0) {
209  Target = 0;
210 
211  } else {
212 
213  //
214  // If the allocate rate is less than the minimum threshold, then lower
215  // the maximum depth of the lookaside list. Otherwise, if the miss rate
216  // is less than .5%, then lower the maximum depth. Otherwise, raise the
217  // maximum depth based on the miss rate.
218  //
219 
220  MaximumDepth = Lookaside->MaximumDepth;
221  Target = Lookaside->Depth;
222  if (Allocates < (ScanPeriod * MINIMUM_ALLOCATION_THRESHOLD)) {
223  if ((Target -= 10) < MINIMUM_LOOKASIDE_DEPTH) {
224  Target = MINIMUM_LOOKASIDE_DEPTH;
225  }
226 
227  } else {
228 
229  //
230  // N.B. The number of allocates is guaranteed to be greater than
231  // zero because of the above test.
232  //
233  // N.B. It is possible that the number of misses are greater than the
234  // number of allocates, but this won't cause the an incorrect
235  // computation of the depth adjustment.
236  //
237 
238  Ratio = (Misses * 1000) / Allocates;
239  if (Ratio < 5) {
240  if ((Target -= 1) < MINIMUM_LOOKASIDE_DEPTH) {
241  Target = MINIMUM_LOOKASIDE_DEPTH;
242  }
243 
244  } else {
245  if ((Delta = ((Ratio * (MaximumDepth - Target)) / (1000 * 2)) + 5) > 30) {
246  Delta = 30;
247  }
248 
249  if ((Target += Delta) > MaximumDepth) {
250  Target = MaximumDepth;
251  }
252  }
253  }
254  }
255 
256  Lookaside->Depth = (USHORT)Target;
257  return;
258 }

260 VOID
261 ExpScanGeneralLookasideList (
262  IN PLIST_ENTRY ListHead,
263  IN PKSPIN_LOCK SpinLock
264  )
265 
266 /*++
267 
268 Routine Description:
269 
270  This function scans the specified list of general lookaside descriptors
271  and adjusts the maximum depth as necessary.
272 
273 Arguments:
274 
275  ListHead - Supplies the address of the listhead for a list of lookaside
276  descriptors.
277 
278  SpinLock - Supplies the address of the spinlock to be used to synchronize
279  access to the list of lookaside descriptors.
280 
281 Return Value:
282 
283  None.
284 
285 --*/
286 
287 {
288 
289  PLIST_ENTRY Entry;
290  PPAGED_LOOKASIDE_LIST Lookaside;
291  ULONG Misses;
292  KIRQL OldIrql;
293 
294 #ifdef NT_UP
295 
296  UNREFERENCED_PARAMETER (SpinLock);
297 
298 #endif
299 
300  //
301  // Raise IRQL and acquire the specified spinlock.
302  //
303 
304  ExAcquireSpinLock(SpinLock, &OldIrql);
305 
306  //
307  // Scan the specified list of lookaside descriptors and adjust the
308  // maximum depth as necessary.
309  //
310  // N.B. All lookaside list descriptors are treated as if they were
311  // paged descriptors even though they may be nonpaged descriptors.
312  // This is possible since both structures are identical except
313  // for the locking fields which are the last structure fields.
314  //
315 
316  Entry = ListHead->Flink;
317  while (Entry != ListHead) {
318  Lookaside = CONTAINING_RECORD(Entry,
319  PAGED_LOOKASIDE_LIST,
320  L.ListEntry);
321 
322  //
323  // Compute target depth of lookaside list.
324  //
325 
326  Misses = Lookaside->L.AllocateMisses - Lookaside->L.LastAllocateMisses;
327  Lookaside->L.LastAllocateMisses = Lookaside->L.AllocateMisses;
328  ExpComputeLookasideDepth(&Lookaside->L, Misses, 3);
329  Entry = Entry->Flink;
330  }
331 
332  //
333  // Release spinlock, lower IRQL, and return function value.
334  //
335 
336  ExReleaseSpinLock(SpinLock, OldIrql);
337  return;
338 }

347 Routine Description:
348 
349  This function scans the current set of paged and nonpaged pool lookaside

372  //
373  // Scan the current set of lookaside descriptors and adjust the maximum
374  // depth as necessary. Either a set of per processor small pool lookaside
375  // lists or the global small pool lookaside lists are scanned during a
376  // scan period.
377  //
378  // N.B. All lookaside list descriptors are treated as if they were
379  // paged descriptors even though they may be nonpaged descriptors.
380  // This is possible since both structures are identical except
381  // for the locking fields which are the last structure fields.
382  //
383 
384  ScanPeriod = (1 + 1 + 1) * KeNumberProcessors;
385  if (ExpPoolScanCount == (ULONG)KeNumberProcessors) {
386 
387  //
388  // Adjust the maximum depth for the global set of system lookaside
389  // descriptors.
390  //
391 
392  Prcb = KeGetCurrentPrcb();
393  for (Index = 0; Index < LookasideMaximumList; Index += 1) {
394  Lookaside = Prcb->PPLookasideList[Index].L;
395  if (Lookaside != NULL) {
396  Misses = Lookaside->AllocateMisses - Lookaside->LastAllocateMisses;
397  Lookaside->LastAllocateMisses = Lookaside->AllocateMisses;
398  ExpComputeLookasideDepth(Lookaside, Misses, ScanPeriod);
399  }
400  }
401 
402  //
403  // Adjust the maximum depth for the global set of small pool lookaside
404  // descriptors.
405  //
406 
407  for (Index = 0; Index < POOL_SMALL_LISTS; Index += 1) {
408 
409  //
410  // Compute target depth of nonpaged lookaside list.
411  //
412 
413  Lookaside = &ExpSmallNPagedPoolLookasideLists[Index];
414  Hits = Lookaside->AllocateHits - Lookaside->LastAllocateHits;
415  Lookaside->LastAllocateHits = Lookaside->AllocateHits;
416  Misses =
417  Lookaside->TotalAllocates - Lookaside->LastTotalAllocates - Hits;
418 
419  ExpComputeLookasideDepth(Lookaside, Misses, ScanPeriod);
420 
421  //
422  // Compute target depth of paged lookaside list.
423  //
424 
425  Lookaside = &ExpSmallPagedPoolLookasideLists[Index];
426  Hits = Lookaside->AllocateHits - Lookaside->LastAllocateHits;
427  Lookaside->LastAllocateHits = Lookaside->AllocateHits;
428  Misses =
429  Lookaside->TotalAllocates - Lookaside->LastTotalAllocates - Hits;
430 
431  ExpComputeLookasideDepth(Lookaside, Misses, ScanPeriod);
432  }
433 
434  } else {
435 
436  //
437  // Adjust the maximum depth for the global set of per processor
438  // system lookaside descriptors.
439  //
440 
441  Prcb = KiProcessorBlock[ExpPoolScanCount];
442  for (Index = 0; Index < LookasideMaximumList; Index += 1) {
443  Lookaside = Prcb->PPLookasideList[Index].P;
444  if (Lookaside != NULL) {
445  Misses = Lookaside->AllocateMisses - Lookaside->LastAllocateMisses;
446  Lookaside->LastAllocateMisses = Lookaside->AllocateMisses;
447  ExpComputeLookasideDepth(Lookaside, Misses, ScanPeriod);
448  }
449  }
450 
451  //

453  // lookaside descriptors.
454  //
455 
456  for (Index = 0; Index < POOL_SMALL_LISTS; Index += 1) {
457 
458  //
459  // Compute target depth of nonpaged lookaside list.
460  //
461 
462  Lookaside = Prcb->PPNPagedLookasideList[Index].P;
463  Hits = Lookaside->AllocateHits - Lookaside->LastAllocateHits;
464  Lookaside->LastAllocateHits = Lookaside->AllocateHits;
465  Misses =
466  Lookaside->TotalAllocates - Lookaside->LastTotalAllocates - Hits;
467 
468  ExpComputeLookasideDepth(Lookaside, Misses, ScanPeriod);
469 
470  //
471  // Compute target depth of paged lookaside list.
472  //
473 
474  Lookaside = Prcb->PPPagedLookasideList[Index].P;
475  Hits = Lookaside->AllocateHits - Lookaside->LastAllocateHits;
476  Lookaside->LastAllocateHits = Lookaside->AllocateHits;
477  Misses =
478  Lookaside->TotalAllocates - Lookaside->LastTotalAllocates - Hits;
479 
480  ExpComputeLookasideDepth(Lookaside, Misses, ScanPeriod);
481  }
482  }
483 
484  ExpPoolScanCount += 1;
485  if (ExpPoolScanCount > (ULONG)KeNumberProcessors) {
486  ExpPoolScanCount = 0;
487  }
488 
489  return;
