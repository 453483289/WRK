44 /*++
45 
46 Routine Description:
47 
48  This function returns the address of the NT Header.
49 
50  This function is a bit complicated.
51  It is this way because RtlImageNtHeader that it replaces was hard to understand,
52  and this function retains compatibility with RtlImageNtHeader.
53 
54  RtlImageNtHeader was #ifed such as to act different in each of the three
55  boot loader, kernel, usermode flavors.
56 
57  boot loader -- no exception handling
58  usermode -- limit msdos header to 256meg, catch any exception accessing the msdos-header
59  or the pe header
60  kernel -- don't cross user/kernel boundary, don't catch the exceptions,
61  no 256meg limit
62 
63 Arguments:
64 
65  Flags - RTL_IMAGE_NT_HEADER_EX_FLAG_NO_RANGE_CHECK -- don't be so picky
66  about the image, for compatibility with RtlImageNtHeader
67  Base - Supplies the base of the image.
68  Size - The size of the view, usually larger than the size of the file on disk.
69  This is available from NtMapViewOfSection but not from MapViewOfFile.
70  OutHeaders -
71 
72 Return Value:
73 
74  STATUS_SUCCESS -- everything ok
75  STATUS_INVALID_IMAGE_FORMAT -- bad filesize or signature value
76  STATUS_INVALID_PARAMETER -- bad parameters
77 
78 --*/
79 
80 {
81  PIMAGE_NT_HEADERS NtHeaders = 0;
82  ULONG e_lfanew = 0;
83  BOOLEAN RangeCheck = 0;
84  NTSTATUS Status = 0;
85  const ULONG ValidFlags =
86  RTL_IMAGE_NT_HEADER_EX_FLAG_NO_RANGE_CHECK;
87 
88  if (OutHeaders != NULL) {
89  *OutHeaders = NULL;
90  }
91  if (OutHeaders == NULL) {
92  Status = STATUS_INVALID_PARAMETER;
93  goto Exit;
94  }
95  if ((Flags & ~ValidFlags) != 0) {
96  Status = STATUS_INVALID_PARAMETER;
97  goto Exit;
98  }
99  if (Base == NULL || Base == (PVOID)(LONG_PTR)-1) {
100  Status = STATUS_INVALID_PARAMETER;
101  goto Exit;
102  }
103 
104  RangeCheck = ((Flags & RTL_IMAGE_NT_HEADER_EX_FLAG_NO_RANGE_CHECK) == 0);
105  if (RangeCheck) {
106  if (Size < sizeof(IMAGE_DOS_HEADER)) {
107  Status = STATUS_INVALID_IMAGE_FORMAT;
108  goto Exit;
109  }
110  }
111 
112  //
113  // Exception handling is not available in the boot loader, and exceptions
114  // were not historically caught here in kernel mode. Drivers are considered
115  // trusted, so we can't get an exception here due to a bad file, but we
116  // could take an inpage error.
117  //
118 #define EXIT goto Exit
119  if (((PIMAGE_DOS_HEADER)Base)->e_magic != IMAGE_DOS_SIGNATURE) {
120  Status = STATUS_INVALID_IMAGE_FORMAT;
121  EXIT;
122  }
123  e_lfanew = ((PIMAGE_DOS_HEADER)Base)->e_lfanew;
124  if (RangeCheck) {
125  if (e_lfanew >= Size
126 #define SIZEOF_PE_SIGNATURE 4
127  || e_lfanew >= (MAXULONG - SIZEOF_PE_SIGNATURE - sizeof(IMAGE_FILE_HEADER))
128  || (e_lfanew + SIZEOF_PE_SIGNATURE + sizeof(IMAGE_FILE_HEADER)) >= Size
129  ) {
130  Status = STATUS_INVALID_IMAGE_FORMAT;
131  EXIT;
132  }
133  }
134 
135  NtHeaders = (PIMAGE_NT_HEADERS)((PCHAR)Base + e_lfanew);
136 
137  //
138  // In kernelmode, do not cross from usermode address to kernelmode address.
139  //
140  if (Base < MM_HIGHEST_USER_ADDRESS) {
141  if ((PVOID)NtHeaders >= MM_HIGHEST_USER_ADDRESS) {
142  Status = STATUS_INVALID_IMAGE_FORMAT;
143  EXIT;
144  }
145  //
146  // Note that this check is slightly overeager since IMAGE_NT_HEADERS has
147  // a builtin array of data_directories that may be larger than the image
148  // actually has. A better check would be to add FileHeader.SizeOfOptionalHeader,
149  // after ensuring that the FileHeader does not cross the u/k boundary.
150  //
151  if ((PVOID)((PCHAR)NtHeaders + sizeof (IMAGE_NT_HEADERS)) >= MM_HIGHEST_USER_ADDRESS) {
152  Status = STATUS_INVALID_IMAGE_FORMAT;
153  EXIT;
154  }
155  }
156 
157  if (NtHeaders->Signature != IMAGE_NT_SIGNATURE) {
158  Status = STATUS_INVALID_IMAGE_FORMAT;
159  EXIT;
160  }
161  Status = STATUS_SUCCESS;
162 
163 Exit:
164  if (NT_SUCCESS(Status)) {
165  *OutHeaders = NtHeaders;
166  }
167  return Status;
168 }
169 #undef EXIT
