37 NTSTATUS
38 NtSetInformationToken (
39  __in HANDLE TokenHandle,
40  __in TOKEN_INFORMATION_CLASS TokenInformationClass,
41  __in_bcount(TokenInformationLength) PVOID TokenInformation,
42  __in ULONG TokenInformationLength
43  )
44 
45 /*++
46 
47 
48 Routine Description:
49 
50  Modify information in a specified token.
51 
52 Arguments:
53 
54  TokenHandle - Provides a handle to the token to operate on.
55 
56  TokenInformationClass - The token information class being set.
57 
58  TokenInformation - The buffer containing the new values for the
59  specified class of information. The buffer must be aligned
60  on at least a longword boundary. The actual structures
61  provided are dependent upon the information class specified,
62  as defined in the TokenInformationClass parameter
63  description.
64 
65  TokenInformation Format By Information Class:
66 
67  TokenUser => This value is not a valid value for this API.
68  The User ID may not be replaced.
69 
70  TokenGroups => This value is not a valid value for this
71  API. The Group IDs may not be replaced. However, groups
72  may be enabled and disabled using NtAdjustGroupsToken().
73 
74  TokenPrivileges => This value is not a valid value for
75  this API. Privilege information may not be replaced.
76  However, privileges may be explicitly enabled and disabled
77  using the NtAdjustPrivilegesToken API.
78 
79  TokenOwner => TOKEN_OWNER data structure.
80  TOKEN_ADJUST_DEFAULT access is needed to replace this
81  information in a token. The owner values that may be
82  specified are restricted to the user and group IDs with an
83  attribute indicating they may be assigned as the owner of
84  objects.
85 
86  TokenPrimaryGroup => TOKEN_PRIMARY_GROUP data structure.
87  TOKEN_ADJUST_DEFAULT access is needed to replace this
88  information in a token. The primary group values that may
89  be specified are restricted to be one of the group IDs
90  already in the token.
91 
92  TokenDefaultDacl => TOKEN_DEFAULT_DACL data structure.
93  TOKEN_ADJUST_DEFAULT access is needed to replace this
94  information in a token. The ACL provided as a new default
95  discretionary ACL is not validated for structural
96  correctness or consistency.
97 
98  TokenSource => This value is not a valid value for this
99  API. The source name and context handle may not be
100  replaced.
101 
102  TokenStatistics => This value is not a valid value for this
103  API. The statistics of a token are read-only.
104 
105  TokenSessionId => ULONG to set the token session. Must have
106  TOKEN_ADJUST_SESSIONID and TCB privilege.
107 
108  TokenSessionReference => ULONG. Must be zero. Must have
109  TCB privilege to dereference the logon session. This info class
110  will remove a reference for the logon session, and mark the token
111  as not referencing the session.
112 
113  TokenAuditPolicy => TOKEN_AUDIT_POLICY structure. This sets the per
114  user policy for the token, and all tokens derived from it. Requires
115  TCB privilege.
116 
117  TokenParent => TOKEN_PARENT structure. The parent id can be set
118  by a caller with TCB privilege. The token id cannot.
119 
120  TokenInformationLength - Indicates the length, in bytes, of the
121  TokenInformation buffer. This is only the length of the primary
122  buffer. All extensions of the primary buffer are self describing.
123 
124 Return Value:
125 
126  STATUS_SUCCESS - The operation was successful.
127 
128  STATUS_INVALID_OWNER - The ID specified to be an owner (or
129  default owner) is not one the caller may assign as the owner
130  of an object.
131 
132  STATUS_INVALID_INFO_CLASS - The specified information class is
133  not one that may be specified in this API.
134 
135  STATUS_ALLOTTED_SPACE_EXCEEDED - The space allotted for storage
136  of the default discretionary access control and the primary
137  group ID is not large enough to accept the new value of one
138  of these fields.
139 
140 --*/

188  //

190  //

192  (TokenInformationClass != TokenPrimaryGroup) &&
193  (TokenInformationClass != TokenSessionId) &&
194  (TokenInformationClass != TokenDefaultDacl) &&

196  (TokenInformationClass != TokenAuditPolicy) &&

210  DesiredAccess |= TOKEN_ADJUST_SESSIONID;

213  Status = ObReferenceObjectByHandle(
214  TokenHandle, // Handle
215  DesiredAccess, // DesiredAccess
216  SeTokenObjectType, // ObjectType
217  PreviousMode, // AccessMode
218  (PVOID *)&Token, // Object
219  NULL // GrantedAccess
220  );
221 
222  if ( !NT_SUCCESS(Status) ) {
223  return Status;
224  }

227  //
228  // Case on information class.
229  //
230 
231  switch ( TokenInformationClass ) {
232 
233  case TokenOwner:
234 
235  //
236  // Make sure the buffer is large enough to hold the
237  // necessary information class data structure.
238  //
239 
240  if (TokenInformationLength < (ULONG)sizeof(TOKEN_OWNER)) {
241 
242  ObDereferenceObject( Token );
243  return STATUS_INFO_LENGTH_MISMATCH;
244  }
245 
246  //
247  // Capture and copy
248 
249  try {
250 
251  //
252  // Capture Owner SID
253  //
254 
255  CapturedOwner = ((PTOKEN_OWNER)TokenInformation)->Owner;
256  Status = SeCaptureSid(
257  CapturedOwner,
258  PreviousMode,
259  NULL, 0,
260  PagedPool,
261  TRUE,
262  &CapturedOwner
263  );
264 
265  } except(EXCEPTION_EXECUTE_HANDLER) {
266 
267  ObDereferenceObject( Token );
268  return GetExceptionCode();
269  }
270 
271  if (!NT_SUCCESS(Status)) {
272  ObDereferenceObject( Token );
273  return Status;
274  }
275 
276  Index = 0;
277 
278  //
279  // Gain write access to the token.
280  //
281 
282  SepAcquireTokenWriteLock( Token );
283 
284  //
285  // Walk through the list of user and group IDs looking
286  // for a match to the specified SID. If one is found,
287  // make sure it may be assigned as an owner. If it can,
288  // then set the index in the token's OwnerIndex field.
289  // Otherwise, return invalid owner error.
290  //
291 
292  while (Index < Token->UserAndGroupCount) {
293 
294  try {
295 
296  Found = RtlEqualSid(
297  CapturedOwner,
298  Token->UserAndGroups[Index].Sid
299  );
300 
301  if ( Found ) {
302

304 
305  Token->DefaultOwnerIndex = Index;
306  TokenModified = TRUE;
307  Status = STATUS_SUCCESS;
308

311  Status = STATUS_INVALID_OWNER;
312 
313  } //endif assignable
314 
315  SepReleaseTokenWriteLock( Token, TokenModified );
316  ObDereferenceObject( Token );

318  return Status;
319 
320  } //endif Found
321 
322  } except(EXCEPTION_EXECUTE_HANDLER) {
323 
324  SepReleaseTokenWriteLock( Token, TokenModified );
325  ObDereferenceObject( Token );
326  SeReleaseSid( CapturedOwner, PreviousMode, TRUE);
327  return GetExceptionCode();
328 
329  } //endtry
330 
331  Index += 1;
332 
333  } //endwhile
334 
335  SepReleaseTokenWriteLock( Token, TokenModified );
336  ObDereferenceObject( Token );
337  SeReleaseSid( CapturedOwner, PreviousMode, TRUE);
338  return STATUS_INVALID_OWNER;
339 
340  case TokenPrimaryGroup:
341 
342  //
343  // Assuming everything works out, the strategy is to move everything
344  // in the Dynamic part of the token (exept the primary group) to
345  // the beginning of the dynamic part, freeing up the entire end of
346  // the dynamic part for the new primary group.
347  //
348 
349  //
350  // Make sure the buffer is large enough to hold the
351  // necessary information class data structure.
352  //
353 
354  if (TokenInformationLength < (ULONG)sizeof(TOKEN_PRIMARY_GROUP)) {
355 
356  ObDereferenceObject( Token );
357  return STATUS_INFO_LENGTH_MISMATCH;
358  }
359 
360  //
361  // Capture And Validate TOKEN_PRIMARY_GROUP and corresponding SID.
362  //
363 
364  try {
365 
366  CapturedPrimaryGroup =
367  ((PTOKEN_PRIMARY_GROUP)TokenInformation)->PrimaryGroup;
368 
369  Status = SeCaptureSid(
370  CapturedPrimaryGroup,
371  PreviousMode,
372  NULL, 0,
373  PagedPool,
374  TRUE,
375  &CapturedPrimaryGroup
376  );
377 
378  } except(EXCEPTION_EXECUTE_HANDLER) {
379 
380  ObDereferenceObject( Token );
381  return GetExceptionCode();
382  }
383 
384  if (!NT_SUCCESS(Status)) {
385  ObDereferenceObject( Token );
386  return Status;
387  }
388 
389  if (!SepIdAssignableAsGroup( Token, CapturedPrimaryGroup )) {
390  ObDereferenceObject( Token );
391  SeReleaseSid( CapturedPrimaryGroup, PreviousMode, TRUE);
392  return STATUS_INVALID_PRIMARY_GROUP;
393  }
394

396 
397  //
398  // Gain write access to the token.
399  //
400 
401  SepAcquireTokenWriteLock( Token );
402

404  // See if there is enough room in the dynamic part of the token
405  // to replace the current Primary Group with the one specified.
406  //
407 
408  if (Token->DefaultDacl) {
409  NewLength += Token->DefaultDacl->AclSize;
410  }
411 
412  if (NewLength > Token->DynamicCharged) {
413 
414  SepReleaseTokenWriteLock( Token, TokenModified );
415  ObDereferenceObject( Token );
416  SeReleaseSid( CapturedPrimaryGroup, PreviousMode, TRUE);
417  return STATUS_ALLOTTED_SPACE_EXCEEDED;
418  }
419

422  //
423 
424  Status = SepExpandDynamic( Token, NewLength );
425 
426  if (!NT_SUCCESS (Status)) {
427  SepReleaseTokenWriteLock( Token, TokenModified );
428  ObDereferenceObject( Token );
429  SeReleaseSid( CapturedPrimaryGroup, PreviousMode, TRUE);
430  return Status;
431  }
432 
433 
434  //
435  // Free up the existing primary group
436  //
437 
438  SepFreePrimaryGroup( Token );
439 
440  //

442  //

447 
448  //
449  // All done.
450  //
451 
452  SepReleaseTokenWriteLock( Token, TokenModified );
453  ObDereferenceObject( Token );
454  SeReleaseSid( CapturedPrimaryGroup, PreviousMode, TRUE);
455  return STATUS_SUCCESS;

464  // the dynamic part for the new default Dacl.

466

468  // Make sure the buffer is large enough to hold the

527

529  ObDereferenceObject( Token );

531  SeReleaseAcl( CapturedDefaultDacl, PreviousMode, TRUE);
532  }

537  // Expand the tokens dynamic buffer if we have to
538  //
539  Status = SepExpandDynamic( Token, NewLength );
540 
541  if (!NT_SUCCESS (Status)) {
542  SepReleaseTokenWriteLock( Token, TokenModified );
543  ObDereferenceObject( Token );
544  if (ARGUMENT_PRESENT(CapturedDefaultDacl)) {
545  SeReleaseAcl( CapturedDefaultDacl, PreviousMode, TRUE);
546  }
547  return Status;
548  }

551  //

554 
555  //

557  //
558

570  ObDereferenceObject( Token );
571  if (ARGUMENT_PRESENT(CapturedDefaultDacl)) {
572  SeReleaseAcl( CapturedDefaultDacl, PreviousMode, TRUE);
573  }

751  }

787
