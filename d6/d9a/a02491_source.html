172 // holding the fewest possible nodes that can possibly be contained in a
173 // balanced binary tree with the given number of levels. After the first
174 // two levels, each level n is obviously occupied by a root node, plus
175 // one subtree the size of level n-1, and another subtree which is the
176 // size of n-2, i.e.:
177 //
178 // MiWorstCaseFill[n] = 1 + MiWorstCaseFill[n-1] + MiWorstCaseFill[n-2]

180 // The efficiency of a typical balanced binary tree will normally fall
181 // between the two extremes, typically closer to the best case. Note

212 TABLE_SEARCH_RESULT
213 MiFindNodeOrParent (
214  IN PMM_AVL_TABLE Table,
215  IN ULONG_PTR StartingVpn,
216  OUT PMMADDRESS_NODE *NodeOrParent
217  )
218 
219 /*++
220 
221 Routine Description:
222 
223  This routine is used by all of the routines of the generic
224  table package to locate the a node in the tree. It will
225  find and return (via the NodeOrParent parameter) the node
226  with the given key, or if that node is not in the tree it
227  will return (via the NodeOrParent parameter) a pointer to
228  the parent.
229 
230 Arguments:
231 
232  Table - The generic table to search for the key.
233 
234  StartingVpn - The starting virtual page number.
235 
236  NodeOrParent - Will be set to point to the node containing the
237  the key or what should be the parent of the node
238  if it were in the tree. Note that this will *NOT*
239  be set if the search result is TableEmptyTree.
240 
241 Return Value:
242 
243  TABLE_SEARCH_RESULT - TableEmptyTree: The tree was empty. NodeOrParent
244  is *not* altered.
245 
246  TableFoundNode: A node with the key is in the tree.
247  NodeOrParent points to that node.
248 
249  TableInsertAsLeft: Node with key was not found.
250  NodeOrParent points to what would
251  be parent. The node would be the
252  left child.
253 
254  TableInsertAsRight: Node with key was not found.
255  NodeOrParent points to what would
256  be parent. The node would be
257  the right child.
258 
259 Environment:
260 
261  Kernel mode. The PFN lock is held for some of the tables.
262 
263 --*/
264 
265 {
266 #if DBG
267  ULONG NumberCompares = 0;
268 #endif
269  PMMADDRESS_NODE Child;
270  PMMADDRESS_NODE NodeToExamine;
271 
272  if (Table->NumberGenericTableElements == 0) {
273  return TableEmptyTree;
274  }
275 
276  NodeToExamine = (PMMADDRESS_NODE) Table->BalancedRoot.RightChild;
277 
278  do {
279 
280  //
281  // Make sure the depth of tree is correct.
282  //
283 
284  ASSERT(++NumberCompares <= Table->DepthOfTree);
285 
286  //
287  // Compare the buffer with the key in the tree element.
288  //
289 
290  if (StartingVpn < NodeToExamine->StartingVpn) {
291 
292  Child = NodeToExamine->LeftChild;
293 
294  if (Child != NULL) {
295  NodeToExamine = Child;
296  }
297  else {
298 
299  //
300  // Node is not in the tree. Set the output
301  // parameter to point to what would be its
302  // parent and return which child it would be.
303  //
304 
305  *NodeOrParent = NodeToExamine;
306  return TableInsertAsLeft;
307  }
308  }
309  else if (StartingVpn <= NodeToExamine->EndingVpn) {
310 
311  //
312  // This is the node.
313  //
314 
315  *NodeOrParent = NodeToExamine;
316  return TableFoundNode;
317  }
318  else {
319 
320  Child = NodeToExamine->RightChild;
321 
322  if (Child != NULL) {
323  NodeToExamine = Child;
324  }
325  else {
326 
327  //
328  // Node is not in the tree. Set the output
329  // parameter to point to what would be its
330  // parent and return which child it would be.
331  //
332 
333  *NodeOrParent = NodeToExamine;
334  return TableInsertAsRight;
335  }
336  }
337 
338  } while (TRUE);
339 }

349 /*++
350 
351 Routine Description:
352 
353  The function determines if any addresses between a given starting and
354  ending address is contained within a virtual address descriptor.
355 
356 Arguments:
357 
358  StartVpn - Supplies the virtual address to locate a containing
359  descriptor.
360 
361  EndVpn - Supplies the virtual address to locate a containing
362  descriptor.
363 
364 Return Value:
365 
366  Returns a pointer to the first conflicting virtual address descriptor
367  if one is found, otherwise a NULL value is returned.
368 
369 --*/
370 
371 {
372  PMMADDRESS_NODE Node;
373 
374  if (Table->NumberGenericTableElements == 0) {
375  return NULL;
376  }
377 
378  Node = (PMMADDRESS_NODE) Table->BalancedRoot.RightChild;
379  ASSERT (Node != NULL);
380 
381  do {
382 
383  if (Node == NULL) {
384  return NULL;
385  }
386 
387  if (StartVpn > Node->EndingVpn) {
388  Node = Node->RightChild;
389  }
390  else if (EndVpn < Node->StartingVpn) {
391  Node = Node->LeftChild;
392  }
393  else {
394 
395  //
396  // The starting address is less than or equal to the end VA
397  // and the ending address is greater than or equal to the
398  // start va. Return this node.
399  //
400 
401  return Node;
402  }
403 
404  } while (TRUE);
405 }

451 VOID
452 MiPromoteNode (
453  IN PMMADDRESS_NODE C
454  )
455 
456 /*++
457 
458 Routine Description:
459 
460  This routine performs the fundamental adjustment required for balancing
461  the binary tree during insert and delete operations. Simply put, the
462  designated node is promoted in such a way that it rises one level in
463  the tree and its parent drops one level in the tree, becoming now the
464  child of the designated node. Generally the path length to the subtree
465  "opposite" the original parent. Balancing occurs as the caller chooses
466  which nodes to promote according to the balanced tree algorithms from
467  Knuth.
468 
469  This is not the same as a splay operation, typically a splay "promotes"
470  a designated node twice.
471 
472  Note that the pointer to the root node of the tree is assumed to be
473  contained in a MMADDRESS_NODE structure itself, to allow the
474  algorithms below to change the root of the tree without checking
475  for special cases. Note also that this is an internal routine,
476  and the caller guarantees that it never requests to promote the
477  root itself.
478 
479  This routine only updates the tree links; the caller must update
480  the balance factors as appropriate.
481 
482 Arguments:
483 
484  C - pointer to the child node to be promoted in the tree.
485 
486 Return Value:
487 
488  None.
489 
490 --*/
491 
492 {
493  PMMADDRESS_NODE P;
494  PMMADDRESS_NODE G;
495 
496  //
497  // Capture the current parent and grandparent (may be the root).
498  //
499 
500  P = SANITIZE_PARENT_NODE (C->u1.Parent);
501  G = SANITIZE_PARENT_NODE (P->u1.Parent);
502 
503  //
504  // Break down the promotion into two cases based upon whether C
505  // is a left or right child.
506  //
507 
508  if (P->LeftChild == C) {
509 
510  //
511  // This promotion looks like this:
512  //
513  // G G
514  // | |
515  // P C
516  // / \ => / \
517  // C z x P
518  // / \ / \
519  // x y y z
520  //
521 
522  P->LeftChild = C->RightChild;
523 
524  if (P->LeftChild != NULL) {
525 
526  P->LeftChild->u1.Parent = MI_MAKE_PARENT (P, P->LeftChild->u1.Balance);
527  }
528 
529  C->RightChild = P;
530 
531  //
532  // Fall through to update parent and G <-> C relationship in
533  // common code.
534  //
535 
536  }
537  else {
538 
539  ASSERT(P->RightChild == C);
540 
541  //
542  // This promotion looks like this:
543  //
544  // G G
545  // | |
546  // P C
547  // / \ => / \
548  // x C P z
549  // / \ / \
550  // y z x y
551  //
552 
553  P->RightChild = C->LeftChild;
554 
555  if (P->RightChild != NULL) {
556  P->RightChild->u1.Parent = MI_MAKE_PARENT (P, P->RightChild->u1.Balance);
557  }
558 
559  C->LeftChild = P;
560  }
561 
562  //
563  // Update parent of P, for either case above.
564  //
565 
566  P->u1.Parent = MI_MAKE_PARENT (C, P->u1.Balance);
567 
568  //
569  // Finally update G <-> C links for either case above.
570  //
571 
572  if (G->LeftChild == P) {
573  G->LeftChild = C;
574  }
575  else {
576  ASSERT(G->RightChild == P);
577  G->RightChild = C;
578  }
579  C->u1.Parent = MI_MAKE_PARENT (G, C->u1.Balance);
580 }

583 ULONG
584 MiRebalanceNode (
585  IN PMMADDRESS_NODE S
586  )
587 
588 /*++
589 
590 Routine Description:
591 
592  This routine performs a rebalance around the input node S, for which the
593  Balance factor has just effectively become +2 or -2. When called, the
594  Balance factor still has a value of +1 or -1, but the respective longer
595  side has just become one longer as the result of an insert or delete
596  operation.
597 
598  This routine effectively implements steps A7.iii (test for Case 1 or
599  Case 2) and steps A8 and A9 of Knuth's balanced insertion algorithm,
600  plus it handles Case 3 identified in the delete section, which can
601  only happen on deletes.
602 
603  The trick is, to convince yourself that while traveling from the
604  insertion point at the bottom of the tree up, that there are only
605  these two cases, and that when traveling up from the deletion point,
606  that there are just these three cases. Knuth says it is obvious!
607 
608 Arguments:
609 
610  S - pointer to the node which has just become unbalanced.
611 
612 Return Value:
613 
614  TRUE if Case 3 was detected (causes delete algorithm to terminate).
615 
616 Environment:
617 
618  Kernel mode. The PFN lock is held for some of the tables.
619 
620 --*/
621 
622 {
623  PMMADDRESS_NODE R, P;
624  SCHAR a;
625 
626  PRINT("rebalancing node %p bal=%x start=%x end=%x\n",
627  S,
628  S->u1.Balance,
629  S->StartingVpn,
630  S->EndingVpn);
631 
632  //
633  // The parent node is never the argument node.
634  //
635 
636  ASSERT (SANITIZE_PARENT_NODE(S->u1.Parent) != S);
637 
638  //
639  // Capture which side is unbalanced.
640  //
641 
642  a = (SCHAR) S->u1.Balance;
643 
644  if (a == +1) {
645  R = S->RightChild;
646  }
647  else {
648  R = S->LeftChild;
649  }
650 
651  //
652  // If the balance of R and S are the same (Case 1 in Knuth) then a single
653  // promotion of R will do the single rotation. (Step A8, A10)
654  //
655  // Here is a diagram of the Case 1 transformation, for a == +1 (a mirror
656  // image transformation occurs when a == -1), and where the subtree
657  // heights are h and h+1 as shown (++ indicates the node out of balance):
658  //
659  // | |
660  // S++ R
661  // / \ / \
662  // (h) R+ ==> S (h+1)
663  // / \ / \
664  // (h) (h+1) (h) (h)
665  //
666  // Note that on an insert we can hit this case by inserting an item in the
667  // right subtree of R. The original height of the subtree before the insert
668  // was h+2, and it is still h+2 after the rebalance, so insert rebalancing
669  // may terminate.
670  //
671  // On a delete we can hit this case by deleting a node from the left subtree
672  // of S. The height of the subtree before the delete was h+3, and after the
673  // rebalance it is h+2, so rebalancing must continue up the tree.
674  //
675 
676  if ((SCHAR) R->u1.Balance == a) {
677 
678  MiPromoteNode (R);
679  R->u1.Balance = 0;
680  S->u1.Balance = 0;
681 
682  return FALSE;
683  }
684 
685  //
686  // Otherwise, we have to promote the appropriate child of R twice (Case 2
687  // in Knuth). (Step A9, A10)
688  //
689  // Here is a diagram of the Case 2 transformation, for a == +1 (a mirror
690  // image transformation occurs when a == -1), and where the subtree
691  // heights are h and h-1 as shown. There are actually two minor subcases,
692  // differing only in the original balance of P (++ indicates the node out
693  // of balance).
694  //
695  // | |
696  // S++ P
697  // / \ / \
698  // / \ / \
699  // / \ / \
700  // (h) R- ==> S- R
701  // / \ / \ / \
702  // P+ (h) (h)(h-1)(h) (h)
703  // / \
704  // (h-1) (h)
705  //
706  //
707  // | |
708  // S++ P
709  // / \ / \
710  // / \ / \
711  // / \ / \
712  // (h) R- ==> S R+
713  // / \ / \ / \
714  // P- (h) (h) (h)(h-1)(h)
715  // / \
716  // (h) (h-1)
717  //
718  // Note that on an insert we can hit this case by inserting an item in the
719  // left subtree of R. The original height of the subtree before the insert
720  // was h+2, and it is still h+2 after the rebalance, so insert rebalancing
721  // may terminate.
722  //
723  // On a delete we can hit this case by deleting a node from the left subtree
724  // of S. The height of the subtree before the delete was h+3, and after the
725  // rebalance it is h+2, so rebalancing must continue up the tree.
726  //
727 
728  if ((SCHAR) R->u1.Balance == -a) {
729 
730  //
731  // Pick up the appropriate child P for the double rotation (Link(-a,R)).
732  //
733 
734  if (a == 1) {
735  P = R->LeftChild;
736  }
737  else {
738  P = R->RightChild;
739  }
740 
741  //
742  // Promote him twice to implement the double rotation.
743  //
744 
745  MiPromoteNode (P);
746  MiPromoteNode (P);
747 
748  //
749  // Now adjust the balance factors.
750  //
751 
752  S->u1.Balance = 0;
753  R->u1.Balance = 0;
754  if ((SCHAR) P->u1.Balance == a) {
755  PRINT("REBADJ A: Node %p, Bal %x -> %x\n", S, S->u1.Balance, -a);
756  COUNT_BALANCE_MAX ((SCHAR)-a);
757  S->u1.Balance = (ULONG_PTR) -a;
758  }
759  else if ((SCHAR) P->u1.Balance == -a) {
760  PRINT("REBADJ B: Node %p, Bal %x -> %x\n", R, R->u1.Balance, a);
761  COUNT_BALANCE_MAX ((SCHAR)a);
762  R->u1.Balance = (ULONG_PTR) a;
763  }
764 
765  P->u1.Balance = 0;
766  return FALSE;
767  }
768 
769  //
770  // Otherwise this is Case 3 which can only happen on Delete (identical
771  // to Case 1 except R->u1.Balance == 0). We do a single rotation, adjust
772  // the balance factors appropriately, and return TRUE. Note that the
773  // balance of S stays the same.
774  //
775  // Here is a diagram of the Case 3 transformation, for a == +1 (a mirror
776  // image transformation occurs when a == -1), and where the subtree
777  // heights are h and h+1 as shown (++ indicates the node out of balance):
778  //
779  // | |
780  // S++ R-
781  // / \ / \
782  // (h) R ==> S+ (h+1)
783  // / \ / \
784  // (h+1)(h+1) (h) (h+1)
785  //
786  // This case can not occur on an insert, because it is impossible for
787  // a single insert to balance R, yet somehow grow the right subtree of
788  // S at the same time. As we move up the tree adjusting balance factors
789  // after an insert, we terminate the algorithm if a node becomes balanced,
790  // because that means the subtree length did not change!
791  //
792  // On a delete we can hit this case by deleting a node from the left
793  // subtree of S. The height of the subtree before the delete was h+3,
794  // and after the rebalance it is still h+3, so rebalancing may terminate
795  // in the delete path.
796  //
797 
798  MiPromoteNode (R);
799  PRINT("REBADJ C: Node %p, Bal %x -> %x\n", R, R->u1.Balance, -a);
800  COUNT_BALANCE_MAX ((SCHAR)-a);
801  R->u1.Balance = -a;
802  return TRUE;

814 
815 Routine Description:
816 
817  This routine deletes the specified node from the balanced tree, rebalancing
818  as necessary. If the NodeToDelete has at least one NULL child pointers,
819  then it is chosen as the EasyDelete, otherwise a subtree predecessor or
820  successor is found as the EasyDelete. In either case the EasyDelete is
821  deleted and the tree is rebalanced. Finally if the NodeToDelete was
822  different than the EasyDelete, then the EasyDelete is linked back into the
823  tree in place of the NodeToDelete.
824 
825 Arguments:
826 
827  NodeToDelete - Pointer to the node which the caller wishes to delete.
828 
829  Table - The generic table in which the delete is to occur.
830 
831 Return Value:
832 
833  None.
834 
835 Environment:
836 
837  Kernel mode. The PFN lock is held for some of the tables.
838 
839 --*/
840 
841 {
842  PMMADDRESS_NODE Parent;
843  PMMADDRESS_NODE EasyDelete;
844  PMMADDRESS_NODE P;
845  SCHAR a;
846 
847  //
848  // If the NodeToDelete has at least one NULL child pointer, then we can
849  // delete it directly.
850  //
851 
852  if ((NodeToDelete->LeftChild == NULL) ||
853  (NodeToDelete->RightChild == NULL)) {
854 
855  EasyDelete = NodeToDelete;
856  }
857 
858  //
859  // Otherwise, we may as well pick the longest side to delete from (if one is
860  // is longer), as that reduces the probability that we will have to
861  // rebalance.
862  //
863 
864  else if ((SCHAR) NodeToDelete->u1.Balance >= 0) {
865 
866  //
867  // Pick up the subtree successor.
868  //
869 
870  EasyDelete = NodeToDelete->RightChild;
871  while (EasyDelete->LeftChild != NULL) {
872  EasyDelete = EasyDelete->LeftChild;
873  }
874  }
875  else {
876 
877  //
878  // Pick up the subtree predecessor.
879  //
880 
881  EasyDelete = NodeToDelete->LeftChild;
882  while (EasyDelete->RightChild != NULL) {
883  EasyDelete = EasyDelete->RightChild;
884  }
885  }
886

888  // Rebalancing must know which side of the first parent the delete occurred
889  // on. Assume it is the left side and otherwise correct below.
890  //
891 
892  a = -1;
893 
894  //
895  // Now we can do the simple deletion for the no left child case.
896  //
897 
898  if (EasyDelete->LeftChild == NULL) {
899 
900  Parent = SANITIZE_PARENT_NODE (EasyDelete->u1.Parent);
901 
902  if (MiIsLeftChild(EasyDelete)) {

907  a = 1;
908  }

914  //
915  // Now we can do the simple deletion for the no right child case,
916  // plus we know there is a left child.
917  //

924  if (MiIsLeftChild(EasyDelete)) {
925  Parent->LeftChild = EasyDelete->LeftChild;
926  }

933  EasyDelete->LeftChild->u1.Balance);

937  // For delete rebalancing, set the balance at the root to 0 to properly

939  // if the depth of the tree actually decreased.
940  //

946  // Loop until the tree is balanced.
947  //

954  // the parent.
955  //

963  // and terminate the scan since the subtree length has not changed.
964  // (This may be the root, since we set Balance to 0 above!)
965  //
966 
967  }

976  // the tree really has one less level.
977  //
978 
979  if (Table->BalancedRoot.u1.Balance != 0) {
980  Table->DepthOfTree -= 1;
981  }
982

984 
985  //
986  // Otherwise we made the short side 2 levels less than the long side,
987  // and rebalancing is required. On return, some node has been promoted
988  // to above node P. If Case 3 from Knuth was not encountered, then we
989  // want to effectively resume rebalancing from P's original parent which
990  // is effectively its grandparent now.
991  //
992 
993  }
994  else {

996  //
997  // We are done if Case 3 was hit, i.e., the depth of this subtree is

999  //

1001  if (MiRebalanceNode(P)) {
1002  break;
1003  }
1004 
1005  P = SANITIZE_PARENT_NODE (P->u1.Parent);
1006  }
1007

1009  if (MiIsRightChild(P)) {
1010  a = 1;
1011  }
1012  P = SANITIZE_PARENT_NODE (P->u1.Parent);
1013  }
1014 
1015  //
1016  // Finally, if we actually deleted a predecessor/successor of the
1017  // NodeToDelete, we will link him back into the tree to replace
1018  // NodeToDelete before returning. Note that NodeToDelete did have
1019  // both child links filled in, but that may no longer be the case

1021  //
1022

1024 
1025  //

1027  // to just overlay the EasyDelete node with the NodeToDelete like the
1028  // rtl avl code does.
1029  //
1030  // Copy just the links, preserving the rest of the original EasyDelete
1031  // VAD.
1032  //
1033 
1034  EasyDelete->u1.Parent = NodeToDelete->u1.Parent;
1035  EasyDelete->LeftChild = NodeToDelete->LeftChild;
1036  EasyDelete->RightChild = NodeToDelete->RightChild;
1037 
1038  if (MiIsLeftChild(NodeToDelete)) {
1039  Parent = SANITIZE_PARENT_NODE (EasyDelete->u1.Parent);
1040  Parent->LeftChild = EasyDelete;
1041  }
1042  else {
1043  ASSERT(MiIsRightChild(NodeToDelete));
1044  Parent = SANITIZE_PARENT_NODE (EasyDelete->u1.Parent);
1045  Parent->RightChild = EasyDelete;
1046  }
1047  if (EasyDelete->LeftChild != NULL) {
1048  EasyDelete->LeftChild->u1.Parent = MI_MAKE_PARENT (EasyDelete,
1049  EasyDelete->LeftChild->u1.Balance);
1050  }
1051  if (EasyDelete->RightChild != NULL) {
1052  EasyDelete->RightChild->u1.Parent = MI_MAKE_PARENT (EasyDelete,
1053  EasyDelete->RightChild->u1.Balance);

1061  //

1064  (Table->NumberGenericTableElements <= MiBestCaseFill[Table->DepthOfTree]));
