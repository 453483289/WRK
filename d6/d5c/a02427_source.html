165  ; Get the family and stepping (cpuid fn=1). If processor family
166  ; is less than 0xf, the format returned is as below:
167  ; 3 2 1
168  ; 10987654321098765432109876543210
169  ; --------------------------------
170  ; ppffffmmmmssss
171  ; where
172  ; pp = Processor Type
173  ; ffff = Family
174  ; mmmm = Model
175  ; ssss = Stepping
176  ;
177  ; This is transformed and saved in the PRCB as
178  ;
179  ; PRCB->CpuStep = 0000mmmm0000ssss v v
180  ; ->CpuID = 00000001 | | v
181  ; ->CpuType = 0000ffff | | | v
182  ; | | | |
183  ; ie the dword that contains all this looks like 0M0S010F
184  ;
185 
186 
187  ; If the processor family is 0xf or greater, the format returned is:
188  ; 3 2 1
189  ; 10987654321098765432109876543210
190  ; --------------------------------
191  ; RRRRFFFFFFFFMMMMRRppffffmmmmssss
192  ; where
193  ; pp = Processor Type
194  ; ffff = Family
195  ; mmmm = Model
196  ; ssss = Stepping
197  ; MMMM = Extended Model
198  ; FFFFFFFF = Extended Family
199  ; RRRR, RR = Reserved
200 
201  ; This is transformed and saved in the PRCB as
202  ;
203  ; PRCB->CpuStep = EEEEEEEE0000ssss v v
204  ; ->CpuID = 00000001 | | v
205  ; ->CpuType = XXXXXXXX | | | v
206  ; | | | |
207  ; ie the dword that contains all this looks like EE0S01XX
208  ;
209  ; where
210  ; EEEEEEEE = ((MMMM) << 4)8 bits + (mmmm)zero extended 8 bits
211  ; XXXXXXXX = (FFFFFFFF) + (ffff)zero extended 8 bits
212 
213  ; The value for Extended Family cannot go beyond F0H inorder to support
214  ; a maximum value of FFH for the final Family value(XXXXXXXX).
215 
216  ; The maximum value of Extended Model is FH and the maximum value for
217  ; the final Model value(EEEEEEEE) is FFH
218  ;
219  ;
220  ; If the processor vendor is GenuineIntel and Family=6,then Extended Model
221  ; field is valid and Extended Family is reserved.
222  ;

447 ; BOOLEAN
448 ; Check486AStepping (
449 ; VOID
450 ; )
451 ;
452 ; Routine Description:
453 ;
454 ; This routine checks for 486 A Stepping.
455 ;
456 ; It takes advantage of the fact that on the A-step of the i486
457 ; processor, the ET bit in CR0 could be set or cleared by software,
458 ; but was not used by the hardware. On B or C -step, ET bit in CR0
459 ; is now hardwired to a "1" to force usage of the 386 math coprocessor
460 ; protocol.
461 ;
462 ; Arguments:
463 ;
464 ; None.
465 ;
466 ; Return Value:
467 ;
468 ; Carry Flag clear if B or later stepping.
469 ; Carry Flag set if A or earlier stepping.
470 ;
471 ;--
472  public Check486AStepping
473 Check486AStepping proc near
474  mov eax, cr0 ; reset ET bit in cr0
475  and eax, NOT CR0_ET
476  mov cr0, eax
477 
478  mov eax, cr0 ; get cr0 back
479  test eax, CR0_ET ; if ET bit still set?
480  jnz short cas10 ; if nz, yes, still set, it's NOT A step
481  stc
482  ret
483 
484 cas10: clc
485  ret
486 Check486AStepping endp

583 ; Routine Description:
584 ;
585 ; This routine checks for 486 C Stepping.
586 ;
587 ; This routine takes advantage of the fact that FSCALE produces
588 ; wrong result with Denormal or Pseudo-denormal operand on 486
589 ; C and earlier steps.
590 ;
591 ; If the value contained in ST(1), second location in the floating
592 ; point stack, is between 1 and 11, and the value in ST, top of the
593 ; floating point stack, is either a pseudo-denormal number or a
594 ; denormal number with the underflow exception unmasked, the FSCALE
595 ; instruction produces an incorrect result.
596 ;
597 ; Arguments:
598 ;
599 ; None.
600 ;
601 ; Return Value:
602 ;
603 ; Carry Flag clear if D or later stepping.
604 ; Carry Flag set if C stepping.
605 ;
606 ;--

608 FpControl equ [ebp - 2]
609 RealLongSt1 equ [ebp - 10]
610 PseudoDenormal equ [ebp - 20]
611 FscaleResult equ [ebp - 30]

614 Check486CStepping proc
615 
616  push ebp
617  mov ebp, esp
618  sub esp, 30 ; Allocate space for temp real variables
619 
620  mov eax, cr0 ; Don't trap while doing math
621  and eax, NOT (CR0_ET+CR0_MP+CR0_TS+CR0_EM)
622  mov cr0, eax
623 
624 ;
625 ; Initialize the local FP variables to predefined values.
626 ; RealLongSt1 = 1.0 * (2 ** -1) = 0.5 in normalized double precision FP form
627 ; PseudoDenormal = a unsupported format by IEEE.
628 ; Sign bit = 0
629 ; Exponent = 000000000000000B
630 ; Significand = 100000...0B
631 ; FscaleResult = The result of FSCALE instruction. Depending on 486 step,
632 ; the value will be different:
633 ; Under C and earlier steps, 486 returns the original value
634 ; in ST as the result. The correct returned value should be
635 ; original significand and an exponent of 0...01.
636 ;
637 
638  mov dword ptr RealLongSt1, REALLONG_LOW
639  mov dword ptr RealLongSt1 + 4, REALLONG_HIGH
640  mov dword ptr PseudoDenormal, PSEUDO_DENORMAL_LOW
641  mov dword ptr PseudoDenormal + 4, PSEUDO_DENORMAL_MID
642  mov word ptr PseudoDenormal + 8, PSEUDO_DENORMAL_HIGH
643 
644 .387
645  fnstcw FpControl ; Get FP control word
646  fwait
647  or word ptr FpControl, 0FFh ; Mask all the FP exceptions
648  fldcw FpControl ; Set FP control
649 
650  fld qword ptr RealLongSt1 ; 0 < ST(1) = RealLongSt1 < 1
651  fld tbyte ptr PseudoDenormal; Denormalized operand. Note, i486
652  ; won't report denormal exception
653  ; on 'FLD' instruction.
654  ; ST(0) = Extended Denormalized operand
655  fscale ; try to trigger 486Cx errata
656  fstp tbyte ptr FscaleResult ; Store ST(0) in FscaleResult
657  cmp word ptr FscaleResult + 8, PSEUDO_DENORMAL_HIGH
658  ; Is Exponent changed?
659  jz short c4ds00 ; if z, no, it is C step
660  clc
661  jmp short c4ds10
662 c4ds00: stc
663 c4ds10: mov esp, ebp
664  pop ebp
665  ret

671 ; BOOLEAN
672 ; Check386B0 (
673 ; VOID
674 ; )
675 ;
676 ; Routine Description:
677 ;
678 ; This routine checks for 386 B0 or earlier stepping.
679 ;
680 ; It takes advantage of the fact that the bit INSERT and
681 ; EXTRACT instructions that existed in B0 and earlier versions of the
682 ; 386 were removed in the B1 stepping. When executed on the B1, INSERT
683 ; and EXTRACT cause an int 6 (invalid opcode) exception. This routine
684 ; can therefore discriminate between B1/later 386s and B0/earlier 386s.
685 ; It is intended to be used in sequence with other checks to determine
686 ; processor stepping by exercising specific bugs found in specific
687 ; steppings of the 386.
688 ;
689 ; Arguments:
690 ;
691 ; None.
692 ;
693 ; Return Value:
694 ;
695 ; Carry Flag clear if B1 or later stepping
696 ; Carry Flag set if B0 or prior
697 ;
698 ;--
699 
700 Check386B0 proc
701 
702  push ebx
703 
704  mov ebx, PCR[PcIdt] ; Address of IDT
705  push dword ptr [ebx+30h]
706  push dword ptr [ebx+34h] ; Save Trap06 handler
707 
708  mov eax, offset TemporaryInt6
709  mov word ptr [ebx+30h], ax ; Set LowWord
710  shr eax, 16
711  mov word ptr [ebx+36h], ax ; Set HighWord
712 
713 
714 ;
715 ; Attempt execution of Extract Bit String instruction. Execution on
716 ; B0 or earlier with length (CL) = 0 will return 0 into the destination
717 ; (CX in this case). Execution on B1 or later will fail either due to
718 ; taking the invalid opcode trap, or if the opcode is valid, we don't
719 ; expect CX will be zeroed by any new instruction supported by newer
720 ; steppings. The dummy int 6 handler will clears the Carry Flag and
721 ; returns execution to the appropriate label. If the instruction
722 ; actually executes, CX will *probably* remain unchanged in any new
723 ; stepping that uses the opcode for something else. The nops are meant
724 ; to handle newer steppings with an unknown instruction length.
725 ;
726 
727  xor eax,eax
728  mov edx,eax
729  mov ecx,0ff00h ; Extract length (CL) == 0, (CX) != 0
730 
731 b1c50: db 0fh, 0a6h, 0cah ; xbts cx,dx,ax,cl
732  nop
733  nop
734  nop
735  nop
736  nop
737  stc ; assume B0
738  jecxz short b1c70 ; jmp if B0
739 b1c60: clc
740 b1c70: pop dword ptr [ebx+34h] ; restore old int 6 vector
741  pop dword ptr [ebx+30h]
742

744  ret

785 ;
786 ; This routine checks for 386 D1 Stepping.
787 ;
788 ; It takes advantage of the fact that on pre-D1 386, if a REPeated
789 ; MOVS instruction is executed when single-stepping is enabled,
790 ; a single step trap is taken every TWO moves steps, but should
791 ; occur each move step.
792 ;
793 ; NOTE: This routine cannot distinguish between a D0 stepping and a D1
794 ; stepping. If a need arises to make this distinction, this routine
795 ; will need modification. D0 steppings will be recognized as D1.
796 ;
797 ; Arguments:
798 ;
799 ; None.

801 ; Return Value:

804 ; Carry Flag set if B1 or prior
805 ;

822 ; Execution on B1 or earlier with length (CX) > 1 will trace over two
823 ; iterations before accepting the trace trap. Execution on D1 or later
824 ; will accept the trace trap after a single iteration. The dummy int 1

826 ; instruction. Examination of (CX) will reveal the stepping.
