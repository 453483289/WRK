167 /*++
168 
169 Routine Description:
170 
171  This function flushes multiple entries from the translation buffer
172  on all processors that are currently running threads which are
173  children of the current process or flushes a multiple entries from
174  the translation buffer on all processors in the host configuration.
175 
176 Arguments:
177 
178  Number - Supplies the number of TB entries to flush.
179 
180  Virtual - Supplies a pointer to an array of virtual addresses that
181  are within the pages whose translation buffer entries are to be
182  flushed.
183 
184  AllProcessors - Supplies a boolean value that determines which
185  translation buffers are to be flushed.
186 
187 Return Value:
188 
189  The previous contents of the specified page table entry is returned
190  as the function value.
191 
192 --*/
193 
194 {
195 
196  PKAFFINITY Barrier;
197  PVOID *End;
198  KIRQL OldIrql;
199  PKPRCB Prcb;
200  PKPROCESS Process;
201  KAFFINITY TargetProcessors;
202 
203  ASSERT((Number != 0) && (Number <= FLUSH_MULTIPLE_MAXIMUM));
204 
205  //
206  // Compute target set of processors.
207  //
208 
209  OldIrql = KeRaiseIrqlToSynchLevel();
210  Prcb = KeGetCurrentPrcb();
211  if (AllProcessors != FALSE) {
212  TargetProcessors = KeActiveProcessors;
213 
214  } else {
215  Process = Prcb->CurrentThread->ApcState.Process;
216  TargetProcessors = Process->ActiveProcessors;
217  }
218 
219  //
220  // Send request to target processors, if any, flush multiple entries in
221  // current TB, and wait for the IPI request barrier.
222  //
223 
224  End = Virtual + Number;
225  TargetProcessors &= ~Prcb->SetMember;
226  if (TargetProcessors != 0) {
227  Barrier = KiIpiSendRequest(TargetProcessors,
228  (LONG64)Virtual,
229  Number,
230  IPI_FLUSH_MULTIPLE);
231 
232  do {
233  KiFlushSingleTb(*Virtual);
234  Virtual += 1;
235  } while (Virtual < End);
236 
237  KiIpiWaitForRequestBarrier(Barrier);
238 
239  } else {
240  do {
241  KiFlushSingleTb(*Virtual);
242  Virtual += 1;
243  } while (Virtual < End);
244  }
245 
246  //
247  // Lower IRQL to its previous value.
248  //
249 
250  KeLowerIrql(OldIrql);
251  return;
252 }
