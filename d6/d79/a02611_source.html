126 /*++
127 
128 Routine Description:
129 
130  This function calls a procedure in another thread/process, using
131  NtGetContext and NtSetContext. Parameters are passed to the
132  target procedure via its stack.
133 
134 Arguments:
135 
136  Process - Handle of the target process
137 
138  Thread - Handle of the target thread within that process
139 
140  CallSite - Address of the procedure to call in the target process.
141 
142  ArgumentCount - Number of 32 bit parameters to pass to the target
143  procedure.
144 
145  Arguments - Pointer to the array of 32 bit parameters to pass.
146 
147  PassContext - TRUE if an additional parameter is to be passed that
148  points to a context record.
149 
150  AlreadySuspended - TRUE if the target thread is already in a suspended
151  or waiting state.
152 
153 Return Value:
154 
155  Status - Status value
156 
157 --*/
158 
159 {
160  NTSTATUS Status;
161  CONTEXT Context;
162  ULONG NewSp;
163  ULONG ArgumentsCopy[5];
164 
165  RTL_PAGED_CODE();
166 
167  if (ArgumentCount > 4)
168  return STATUS_INVALID_PARAMETER;
169 
170  //
171  // If necessary, suspend the guy before with we mess with his stack.
172  //
173  if (!AlreadySuspended) {
174  Status = NtSuspendThread( Thread, NULL );
175  if (!NT_SUCCESS( Status )) {
176  return( Status );
177  }
178  }
179 
180 
181  //
182  // Get the context record for the target thread.
183  //
184 
185  Context.ContextFlags = CONTEXT_FULL;
186  Status = NtGetContextThread( Thread, &Context );
187  if (!NT_SUCCESS( Status )) {
188  if (!AlreadySuspended) {
189  NtResumeThread( Thread, NULL );
190  }
191  return( Status );
192  }
193 
194 
195  //
196  // Pass all parameters on the stack, regardless of whether a
197  // a context record is passed.
198  //
199 
200  //
201  // Put Context Record on stack first, so it is above other args.
202  //
203  NewSp = Context.Esp;
204  if (PassContext) {
205  NewSp -= sizeof( CONTEXT );
206  Status = NtWriteVirtualMemory( Process,
207  (PVOID)NewSp,
208  &Context,
209  sizeof( CONTEXT ),
210  NULL
211  );
212  if (!NT_SUCCESS( Status )) {
213  if (!AlreadySuspended) {
214  NtResumeThread( Thread, NULL );
215  }
216  return( Status );
217  }
218  ArgumentsCopy[0] = NewSp; // pass pointer to context
219  RtlCopyMemory(&(ArgumentsCopy[1]),Arguments,ArgumentCount*sizeof( ULONG ));
220  ArgumentCount++;
221  }
222  else {
223  RtlCopyMemory(ArgumentsCopy,Arguments,ArgumentCount*sizeof( ULONG ));
224  }
225 
226  //
227  // Copy the arguments onto the target stack
228  //
229  if (ArgumentCount) {
230  NewSp -= ArgumentCount * sizeof( ULONG );
231  Status = NtWriteVirtualMemory( Process,
232  (PVOID)NewSp,
233  ArgumentsCopy,
234  ArgumentCount * sizeof( ULONG ),
235  NULL
236  );
237  if (!NT_SUCCESS( Status )) {
238  if (!AlreadySuspended) {
239  NtResumeThread( Thread, NULL );
240  }
241  return( Status );
242  }
243  }
244 
245  //
246  // Set the address of the target code into Eip, the new target stack
247  // into Esp, and reload context to make it happen.
248  //
249  Context.Esp = NewSp;
250  Context.Eip = (ULONG)CallSite;
251  Status = NtSetContextThread( Thread, &Context );
252  if (!AlreadySuspended) {
253  NtResumeThread( Thread, NULL );
254  }
255 
256  return( Status );
257 }
