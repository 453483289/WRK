37 ; This function compares two unaligned blocks of memory and returns the

39 ;
40 ; Arguments:
41 ;
42 ; Source1 (rcx) - Supplies a pointer to the first block of memory to

139 ; IN PVOID Source,

144 ; Routine Description:
145 ;
146 ; This function compares a block of dword aligned memory with a specified
147 ; pattern 32-bits at a time.
148 ;
149 ; N.B. The low two bits of the length are assumed to be zero and are
150 ; ignored.
151 ;
152 ; Arguments:
153 ;
154 ; Source (rcx) - Supplies a pointer to the block of memory to compare.
155 ;
156 ; Length (rdx) - Supplies the length, in bytes, of the memory to compare. compare.
157 ;
158 ; Pattern (r8d) - Supplies the pattern to be compared against.
159 ;
160 ; Return Value:
161 ;
162 ; The number of bytes that compared equal is returned as the function
163 ; value. If all bytes compared equal, then the length of the original
164 ; block of memory is returned.
165 ;
166 ;--

169

171  save_reg rdi, 0 ; save nonvolatile register
172 
173  END_PROLOGUE
174 
175  mov rdi, rcx ; set destination address
176  shr rdx, 2 ; compute number of dwords
177  jz short RlCU10 ; if z, no dwords to compare
178  mov rcx, rdx ; set length of compare in dwords
