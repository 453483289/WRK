86  IN PVOID Context

89 /*++
90 
91 Routine Description:
92 
93  This routine implements the thread reaper. The reaper is responsible
94  for processing terminated threads. This includes:
95 
96  - deallocating their kernel stacks
97 
98  - releasing their process' CreateDelete lock
99 
100  - dereferencing their process
101 
102  - dereferencing themselves
103 
104 Arguments:
105 
106  Context - NOT USED
107 
108 Return Value:
109 
110  None.
111 
112 --*/
113 
114 {
115  PSINGLE_LIST_ENTRY NextEntry;
116  PETHREAD Thread;
117 
118  UNREFERENCED_PARAMETER (Context);
119 
120  //
121  // Remove the current list of threads from the reaper list and place a
122  // marker in the reaper list head. This marker will prevent another
123  // worker thread from being queued until all threads in the reaper list
124  // have been processed.
125  //
126 
127  do {
128 
129  NextEntry = InterlockedExchangePointer (&PsReaperListHead.Next,
130  (PVOID)1);
131 
132  ASSERT ((NextEntry != NULL) && (NextEntry != (PVOID)1));
133 
134  //
135  // Delete the respective kernel stack and dereference each thread
136  // in the reaper list.
137  //
138 
139  do {
140 
141  //
142  // Wait until context is swapped for this thread.
143  //
144 
145  Thread = CONTAINING_RECORD (NextEntry, ETHREAD, ReaperLink);
146  KeWaitForContextSwap (&Thread->Tcb);
147  MmDeleteKernelStack (Thread->Tcb.StackBase,
148  (BOOLEAN)Thread->Tcb.LargeStack);
149 
150  Thread->Tcb.InitialStack = NULL;
151  NextEntry = NextEntry->Next;
152  ObDereferenceObject (Thread);
153 
154  } while ((NextEntry != NULL) && (NextEntry != (PVOID)1));
155 
156  } while (InterlockedCompareExchangePointer (&PsReaperListHead.Next,
157  NULL,
158  (PVOID)1) != (PVOID)1);
159 
160  return;
161 }

164 PspTerminateThreadByPointer(
165  IN PETHREAD Thread,
166  IN NTSTATUS ExitStatus,
167  IN BOOLEAN DirectTerminate
168  )
169 
170 /*++
171 
172 Routine Description:
173 
174  This function causes the specified thread to terminate.
175 
176 Arguments:
177 
178  ThreadHandle - Supplies a referenced pointer to the thread to terminate.
179 
180  ExitStatus - Supplies the exit status associated with the thread.
181 
182  DirectTerminate - TRUE is its ok to exit without queing an APC, FALSE otherwise
183 
184 --*/
185 
186 {
187  NTSTATUS Status;
188  PKAPC ExitApc=NULL;
189  ULONG OldMask;
190 
191  PAGED_CODE();
192 
193  if (Thread->CrossThreadFlags
194  & PS_CROSS_THREAD_FLAGS_BREAK_ON_TERMINATION) {
195  PspCatchCriticalBreak("Terminating critical thread 0x%p (in %s)\n",
196  Thread,
197  THREAD_TO_PROCESS(Thread)->ImageFileName);
198  }
199 
200  if (DirectTerminate && Thread == PsGetCurrentThread()) {
201 
202  ASSERT (KeGetCurrentIrql() < APC_LEVEL);
203 
204  PS_SET_BITS (&Thread->CrossThreadFlags, PS_CROSS_THREAD_FLAGS_TERMINATED);
205 
206  PspExitThread (ExitStatus);
207 
208  //
209  // Never Returns
210  //
211 
212  } else {
213  //
214  // Cross thread deletion of system threads won't work.
215  //
216  if (IS_SYSTEM_THREAD (Thread)) {
217  return STATUS_ACCESS_DENIED;
218  }
219 
220  Status = STATUS_SUCCESS;
221 
222  while (1) {
223  ExitApc = (PKAPC) ExAllocatePoolWithTag (NonPagedPool,
224  sizeof(KAPC),
225  'xEsP');
226  if (ExitApc != NULL) {
227  break;
228  }
229  KeDelayExecutionThread(KernelMode, FALSE, &ShortTime);
230  }
231 
232  //
233  // Mark the thread as terminating and call the exit function.
234  //
235  OldMask = PS_TEST_SET_BITS (&Thread->CrossThreadFlags, PS_CROSS_THREAD_FLAGS_TERMINATED);
236 
237  //
238  // If we are the first to set the terminating flag then queue the APC
239  //
240 
241  if ((OldMask & PS_CROSS_THREAD_FLAGS_TERMINATED) == 0) {
242 
243  KeInitializeApc (ExitApc,
244  PsGetKernelThread (Thread),
245  OriginalApcEnvironment,
246  PsExitSpecialApc,
247  PspExitApcRundown,
248  PspExitNormalApc,
249  KernelMode,
250  ULongToPtr (ExitStatus));
251 
252  if (!KeInsertQueueApc (ExitApc, ExitApc, NULL, 2)) {
253  //
254  // If APC queuing is disabled then the thread is exiting anyway
255  //
256  ExFreePool (ExitApc);
257  Status = STATUS_UNSUCCESSFUL;
258  } else {
259  //
260  // We queued the APC to the thread. Wake up the thread if it was suspended.
261  //
262  KeForceResumeThread (&Thread->Tcb);
263 
264  }
265  } else {
266  ExFreePool (ExitApc);
267  }
268  }
269 
270  return Status;
271 }

279 /*++
280 
281 Routine Description:
282 
283  This function causes the specified process and all of
284  its threads to terminate.
285 
286 Arguments:
287 
288  ProcessHandle - Supplies a handle to the process to terminate.
289 
290  ExitStatus - Supplies the exit status associated with the process.
291 
292 Return Value:
293 
294  NTSTATUS - Status of operation
295 
296 --*/
297 
298 {
299 
300  PETHREAD Thread, Self;
301  PEPROCESS Process;
302  PEPROCESS CurrentProcess;
303  NTSTATUS st;
304  BOOLEAN ProcessHandleSpecified;
305  PAGED_CODE();
306 
307  Self = PsGetCurrentThread();
308  CurrentProcess = PsGetCurrentProcessByThread (Self);
309 
310  if (ARGUMENT_PRESENT (ProcessHandle)) {
311  ProcessHandleSpecified = TRUE;
312  } else {
313  ProcessHandleSpecified = FALSE;
314  ProcessHandle = NtCurrentProcess();
315  }
316 
317  st = ObReferenceObjectByHandle (ProcessHandle,
318  PROCESS_TERMINATE,
319  PsProcessType,
320  KeGetPreviousModeByThread(&Self->Tcb),
321  &Process,
322  NULL);
323 
324  if (!NT_SUCCESS (st)) {
325  return(st);
326  }
327 
328  if (Process->Flags & PS_PROCESS_FLAGS_BREAK_ON_TERMINATION) {
329  PspCatchCriticalBreak ("Terminating critical process 0x%p (%s)\n",
330  Process,
331  Process->ImageFileName);
332  }
333 
334  //
335  // Acquire rundown protection just so we can give the right errors
336  //
337 
338  if (!ExAcquireRundownProtection (&Process->RundownProtect)) {
339  ObDereferenceObject (Process);
340  return STATUS_PROCESS_IS_TERMINATING;
341  }
342 
343  //
344  // Mark process as deleting except for the obscure delete self case.
345  //
346  if (ProcessHandleSpecified) {
347  PS_SET_BITS (&Process->Flags, PS_PROCESS_FLAGS_PROCESS_DELETE);
348  }
349 
350  st = STATUS_NOTHING_TO_TERMINATE;
351 
352  for (Thread = PsGetNextProcessThread (Process, NULL);
353  Thread != NULL;
354  Thread = PsGetNextProcessThread (Process, Thread)) {
355 
356  st = STATUS_SUCCESS;
357  if (Thread != Self) {
358  PspTerminateThreadByPointer (Thread, ExitStatus, FALSE);
359  }
360  }
361 
362  ExReleaseRundownProtection (&Process->RundownProtect);
363 
364 
365  if (Process == CurrentProcess) {
366  if (ProcessHandleSpecified) {
367 
368  ObDereferenceObject (Process);
369 
370  //
371  // Never Returns
372  //
373 
374  PspTerminateThreadByPointer (Self, ExitStatus, TRUE);
375  }
376  } else if (ExitStatus == DBG_TERMINATE_PROCESS) {
377  DbgkClearProcessDebugObject (Process, NULL);
378  }
379 
380  //
381  // If there are no threads in this process then clear out its handle table.
382  // Do the same for processes being debugged. This is so a process can never lock itself into the system
383  // by debugging itself or have a handle open to itself.
384  //
385  if (st == STATUS_NOTHING_TO_TERMINATE || (Process->DebugPort != NULL && ProcessHandleSpecified)) {
386  ObClearProcessHandleTable (Process);
387  st = STATUS_SUCCESS;
388  }
389 
390  ObDereferenceObject(Process);
391 
392  return st;
393 }
