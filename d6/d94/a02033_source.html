962  //
963

966  //
967  // Here's hoping that it is cheaper to call Mm to see if
968  // the page is valid. If not let Mm know how many pages
969  // we are after before doing the move.
970  //

976

991

993  CcFreeVirtualAddress( Vacb );
994

997  // away. Otherwise we must have gotten an I/O error trying
998  // to bring the data in.
999  //
1000 
1001  if (Status == STATUS_ACCESS_VIOLATION) {
1002  ExRaiseStatus( STATUS_INVALID_USER_BUFFER );
1003  }
1004  else {
1005  ExRaiseStatus( FsRtlNormalizeNtstatus( Status,
1006  STATUS_UNEXPECTED_IO_ERROR ));
1007  }
1008  }
1009

1011  // Update number of bytes transferred.
1012  //
1013 
1014  Length -= ReceivedLength;
1015 
1016  //
1017  // Unmap the data now, and calculate length left to transfer.
1018  //
1019 
1020  if (Length != 0) {
1021 
1022  //
1023  // If there is more to go, just free this vacb.
1024  //
1025 
1026  CcFreeVirtualAddress( Vacb );
1027 
1028  } else {

1032  //

1382

1384  Buffer,
1385  &FOffset,
1386  Length,
1387  OptimizeFlags,
1388  FileObject,
1389  &ValidDataLength,
1390  Wait );
1391 
1392  DebugTrace(-1, me, "CcCopyWrite -> %02lx\n", Result );
