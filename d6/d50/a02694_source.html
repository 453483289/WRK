260 SepProbeAndCaptureString_U (
261  IN PUNICODE_STRING SourceString,
262  OUT PUNICODE_STRING *DestString
263  )
264 /*++
265 
266 Routine Description:
267 
268  Helper routine to probe and capture a Unicode string argument.
269 
270  This routine may fail due to lack of memory, in which case,
271  it will return a NULL pointer in the output parameter.
272 
273 Arguments:
274 
275  SourceString - Pointer to a Unicode string to be captured.
276 
277  DestString - Returns a pointer to a captured Unicode string. This
278  will be one contiguous structure, and thus may be freed by
279  a single call to ExFreePool().
280 
281 Return Value:
282 
283  None.
284 
285 --*/
286 {
287 
288  UNICODE_STRING InputString;
289  ULONG Length;
290  NTSTATUS Status;
291 
292  PAGED_CODE();
293 
294  //
295  // Initialize the object name descriptor and capture the specified name
296  // string.
297  //
298 
299  *DestString = NULL;
300 
301  Status = STATUS_SUCCESS;
302  try {
303 
304  //
305  // Probe and capture the name string descriptor and probe the
306  // name string, if necessary.
307  //
308 
309  ProbeAndReadUnicodeStringEx(&InputString, SourceString);
310  ProbeForRead(InputString.Buffer,
311  InputString.Length,
312  sizeof(WCHAR));
313 
314 
315 
316  //
317  // If the length of the string is not an even multiple of the
318  // size of a UNICODE character or cannot be zero terminated,
319  // then return an error.
320  //
321 
322  Length = InputString.Length;
323  if (((Length & (sizeof(WCHAR) - 1)) != 0) ||
324  (Length == (MAXUSHORT - sizeof(WCHAR) + 1))) {
325  Status = STATUS_INVALID_PARAMETER;
326 
327  } else {
328 
329  //
330  // Allocate a buffer for the specified name string.
331  //
332 
333  *DestString = ExAllocatePoolWithTag(
334  PagedPool,
335  InputString.Length + sizeof(UNICODE_STRING),
336  'sUeS');
337 
338  if (*DestString == NULL) {
339  Status = STATUS_INSUFFICIENT_RESOURCES;
340 
341  } else {
342  (*DestString)->Length = InputString.Length;
343  (*DestString)->MaximumLength = InputString.Length;
344  (*DestString)->Buffer = (PWSTR) ((*DestString) + 1);
345 
346  if (InputString.Length != 0) {
347 
348  RtlCopyMemory(
349  (*DestString)->Buffer,
350  InputString.Buffer,
351  InputString.Length);
352  }
353 
354  }
355  }
356 
357  } except(ExSystemExceptionFilter()) {
358  Status = GetExceptionCode();
359  if (*DestString != NULL) {
360  ExFreePool(*DestString);
361  *DestString = NULL;
362  }
363  }
364 
365  return;
366 
367 }

406 NTSTATUS
407 NtPrivilegeObjectAuditAlarm (
408  __in PUNICODE_STRING SubsystemName,
409  __in_opt PVOID HandleId,
410  __in HANDLE ClientToken,
411  __in ACCESS_MASK DesiredAccess,
412  __in PPRIVILEGE_SET Privileges,
413  __in BOOLEAN AccessGranted
414  )
415 /*++
416 
417 Routine Description:
418 
419  This routine is used to generate audit and alarm messages when an
420  attempt is made to perform privileged operations on a protected
421  subsystem object after the object is already opened. This routine may
422  result in several messages being generated and sent to Port objects.
423  This may result in a significant latency before returning. Design of
424  routines that must call this routine must take this potential latency
425  into account. This may have an impact on the approach taken for data
426  structure mutex locking, for example.
427 
428  This API requires the caller have SeAuditPrivilege privilege. The test
429  for this privilege is always against the primary token of the calling
430  process, allowing the caller to be impersonating a client during the
431  call with no ill effects.
432 
433 Arguments:
434 
435  SubsystemName - Supplies a name string identifying the subsystem
436  calling the routine.
437 
438  HandleId - A unique value representing the client's handle to the
439  object.
440 
441  ClientToken - A handle to a token object representing the client that
442  requested the operation. This handle must be obtained from a
443  communication session layer, such as from an LPC Port or Local
444  Named Pipe, to prevent possible security policy violations.
445 
446  DesiredAccess - The desired access mask. This mask must have been
447  previously mapped to contain no generic accesses.
448 
449  Privileges - The set of privileges required for the requested
450  operation. Those privileges that were held by the subject are
451  marked using the UsedForAccess flag of the attributes
452  associated with each privilege.
453 
454  AccessGranted - Indicates whether the requested access was granted or
455  not. A value of TRUE indicates the access was granted. A value of
456  FALSE indicates the access was not granted.
457 
458 Return value:
459 
460 --*/
461 {
462 
463  KPROCESSOR_MODE PreviousMode;
464  PUNICODE_STRING CapturedSubsystemName = NULL;
465  PPRIVILEGE_SET CapturedPrivileges = NULL;
466  ULONG PrivilegeParameterLength;
467  ULONG PrivilegeCount;
468  SECURITY_SUBJECT_CONTEXT SubjectSecurityContext;
469  BOOLEAN Result;
470  PTOKEN Token;
471  NTSTATUS Status;
472  BOOLEAN AuditPerformed;
473 
474  PAGED_CODE();
475 
476  PreviousMode = KeGetPreviousMode();
477 
478  ASSERT(PreviousMode != KernelMode);
479 
480  Status = ObReferenceObjectByHandle(
481  ClientToken, // Handle
482  TOKEN_QUERY, // DesiredAccess
483  SeTokenObjectType, // ObjectType
484  PreviousMode, // AccessMode
485  (PVOID *)&Token, // Object
486  NULL // GrantedAccess
487  );
488 
489  if (!NT_SUCCESS( Status )) {
490  return( Status );
491  }
492 
493  //
494  // If the passed token is an impersonation token, make sure
495  // it is at SecurityIdentification or above.
496  //
497 
498  if (Token->TokenType == TokenImpersonation) {
499 
500  if (Token->ImpersonationLevel < SecurityIdentification) {
501 
502  ObDereferenceObject( (PVOID)Token );
503 
504  return( STATUS_BAD_IMPERSONATION_LEVEL );
505 
506  }
507  }
508 
509  //
510  // Check for SeAuditPrivilege
511  //
512 
513  SeCaptureSubjectContext ( &SubjectSecurityContext );
514 
515  Result = SeCheckAuditPrivilege (
516  &SubjectSecurityContext,
517  PreviousMode
518  );
519 
520  if (!Result) {
521 
522  ObDereferenceObject( (PVOID)Token );
523  SeReleaseSubjectContext ( &SubjectSecurityContext );
524  return(STATUS_PRIVILEGE_NOT_HELD);
525 
526  }
527 
528  try {
529 
530  SepProbeAndCaptureString_U ( SubsystemName,
531  &CapturedSubsystemName );
532 
533  ProbeForReadSmallStructure(
534  Privileges,
535  sizeof(PRIVILEGE_SET),
536  sizeof(ULONG)
537  );
538  PrivilegeCount = Privileges->PrivilegeCount;
539 
540  if (!IsValidPrivilegeCount(PrivilegeCount)) {
541  Status= STATUS_INVALID_PARAMETER;
542  leave ;
543  }
544  PrivilegeParameterLength = (ULONG)sizeof(PRIVILEGE_SET) +
545  ((PrivilegeCount - ANYSIZE_ARRAY) *
546  (ULONG)sizeof(LUID_AND_ATTRIBUTES) );
547 
548  ProbeForRead(
549  Privileges,
550  PrivilegeParameterLength,
551  sizeof(ULONG)
552  );
553 
554  CapturedPrivileges = ExAllocatePoolWithTag( PagedPool,
555  PrivilegeParameterLength,
556  'rPeS'
557  );
558 
559  if (CapturedPrivileges != NULL) {
560 
561  RtlCopyMemory ( CapturedPrivileges,
562  Privileges,
563  PrivilegeParameterLength );
564  CapturedPrivileges->PrivilegeCount = PrivilegeCount;
565  }
566 
567  } except (EXCEPTION_EXECUTE_HANDLER) {
568 
569  Status = GetExceptionCode();
570  }
571  if (!NT_SUCCESS(Status)) {
572 
573  if (CapturedPrivileges != NULL) {
574  ExFreePool( CapturedPrivileges );
575  }
576 
577  if (CapturedSubsystemName != NULL) {
578  SepFreeCapturedString ( CapturedSubsystemName );
579  }
580 
581  SeReleaseSubjectContext ( &SubjectSecurityContext );
582 
583  ObDereferenceObject( (PVOID)Token );
584 
585  return Status;
586 
587  }
588 
589  //
590  // No need to lock the token, because the only thing we're going
591  // to reference in it is the User's Sid, which cannot be changed.
592  //
593 
594  //
595  // SepPrivilegeObjectAuditAlarm will check the global flags
596  // to determine if we're supposed to be auditing here.
597  //
598 
599  AuditPerformed = SepAdtPrivilegeObjectAuditAlarm (
600  CapturedSubsystemName,
601  HandleId,
602  Token, // ClientToken
603  SubjectSecurityContext.PrimaryToken, // PrimaryToken
604  SubjectSecurityContext.ProcessAuditId,
605  DesiredAccess,
606  CapturedPrivileges,
607  AccessGranted
608  );
609 
610  if (CapturedPrivileges != NULL) {
611  ExFreePool( CapturedPrivileges );
612  }
613 
614  if (CapturedSubsystemName != NULL) {
615  SepFreeCapturedString ( CapturedSubsystemName );
616  }
617 
618  SeReleaseSubjectContext ( &SubjectSecurityContext );
619 
620  ObDereferenceObject( (PVOID)Token );
621 
622  return(STATUS_SUCCESS);
623 }
