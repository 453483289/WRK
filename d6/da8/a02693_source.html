63 /*++
64 Routine Description:
65 
66  This routine initializes an ACCESS_STATE structure. This consists
67  of:
68 
69  - zeroing the entire structure
70 
71  - mapping generic access types in the passed DesiredAccess
72  and putting it into the structure
73 
74  - "capturing" the Subject Context, which must be held for the
75  duration of the access attempt (at least until auditing is performed).
76 
77  - Allocating an Operation ID, which is an LUID that will be used
78  to associate different parts of the access attempt in the audit
79  log.
80 
81 Arguments:
82 
83  AccessState - a pointer to the structure to be initialized.
84 
85  AuxData - Supplies a buffer big enough for an AuxData structure
86  so we don't have to allocate one.
87 
88  DesiredAccess - Access mask containing the desired access
89 
90  GenericMapping - Optionally supplies a pointer to a generic mapping
91  that may be used to map any generic access requests that may
92  have been passed in the DesiredAccess parameter.
93 
94  Note that if this parameter is not supplied, it must be filled
95  in at some later point. The IO system does this in IopParseDevice.
96 
97 Return Value:
98 
99  Error if the attempt to allocate an LUID fails.
100 
101  Note that this error may be safely ignored if it is known that all
102  security checks will be performed with PreviousMode == KernelMode.
103  Know what you're doing if you choose to ignore this.
104 
105 --*/
106 
107 {
108  return SeCreateAccessStateEx (PsGetCurrentThread (),
109  PsGetCurrentProcess (),
110  AccessState,
111  AuxData,
112  DesiredAccess,
113  GenericMapping);
114 }

117 SeCreateAccessStateEx(
118  __in_opt PETHREAD Thread,
119  __in PEPROCESS Process,

126 /*++
127 Routine Description:
128 
129  This routine initializes an ACCESS_STATE structure. This consists
130  of:
131 
132  - zeroing the entire structure
133 
134  - mapping generic access types in the passed DesiredAccess
135  and putting it into the structure
136 
137  - "capturing" the Subject Context, which must be held for the
138  duration of the access attempt (at least until auditing is performed).
139 
140  - Allocating an Operation ID, which is an LUID that will be used
141  to associate different parts of the access attempt in the audit
142  log.
143 
144 Arguments:
145 
146  Thread - Optional thread to capture impersonation token from. If
147  NULL no impersonation token is captured.
148 
149  Process - Process to capture primary token from.
150 
151  AccessState - a pointer to the structure to be initialized.
152 
153  AuxData - Supplies a buffer big enough for an AuxData structure
154  so we don't have to allocate one.
155 
156  DesiredAccess - Access mask containing the desired access
157 
158  GenericMapping - Optionally supplies a pointer to a generic mapping
159  that may be used to map any generic access requests that may
160  have been passed in the DesiredAccess parameter.
161 
162  Note that if this parameter is not supplied, it must be filled
163  in at some later point. The IO system does this in IopParseDevice.
164 
165 Return Value:
166 
167  Error if the attempt to allocate an LUID fails.
168 
169  Note that this error may be safely ignored if it is known that all
170  security checks will be performed with PreviousMode == KernelMode.
171  Know what you're doing if you choose to ignore this.
172 
173 --*/
174 
175 {
176 
177  ACCESS_MASK MappedAccessMask;
178  PSECURITY_DESCRIPTOR InputSecurityDescriptor = NULL;
179 
180  PAGED_CODE();
181 
182  //
183  // Don't modify what he passed in
184  //
185 
186  MappedAccessMask = DesiredAccess;
187 
188  //
189  // Map generic access to object specific access iff generic access types
190  // are specified and a generic access mapping table is provided.
191  //
192 
193  if ( ((DesiredAccess & GENERIC_ACCESS) != 0) &&
194  ARGUMENT_PRESENT(GenericMapping) ) {
195 
196  RtlMapGenericMask(
197  &MappedAccessMask,
198  GenericMapping
199  );
200  }
201 
202  RtlZeroMemory(AccessState, sizeof(ACCESS_STATE));
203  RtlZeroMemory(AuxData, sizeof(AUX_ACCESS_DATA));
204 
205  //
206  // Assume RtlZeroMemory has initialized these fields properly
207  //
208 
209  ASSERT( AccessState->SecurityDescriptor == NULL );
210  ASSERT( AccessState->PrivilegesAllocated == FALSE );
211 
212  AccessState->AuxData = AuxData;
213 
214  SeCaptureSubjectContextEx(Thread, Process, &AccessState->SubjectSecurityContext);
215 
216  if (((PTOKEN)EffectiveToken( &AccessState->SubjectSecurityContext ))->TokenFlags & TOKEN_HAS_TRAVERSE_PRIVILEGE ) {
217  AccessState->Flags = TOKEN_HAS_TRAVERSE_PRIVILEGE;
218  }
219 
220  AccessState->RemainingDesiredAccess = MappedAccessMask;
221  AccessState->OriginalDesiredAccess = MappedAccessMask;
222  AuxData->PrivilegesUsed = (PPRIVILEGE_SET)((ULONG_PTR)AccessState +
223  (FIELD_OFFSET(ACCESS_STATE, Privileges)));
224 
225  ExAllocateLocallyUniqueId(&AccessState->OperationID);
226 
227  if (ARGUMENT_PRESENT(GenericMapping)) {
228  AuxData->GenericMapping = *GenericMapping;
229  }
230 
231  return( STATUS_SUCCESS );
232 
233 }

322 NTSTATUS
323 SeAppendPrivileges(
324  __inout PACCESS_STATE AccessState,
325  __in PPRIVILEGE_SET Privileges
326  )
327 /*++
328 
329 Routine Description:
330 
331  This routine takes a privilege set and adds it to the privilege set
332  embedded in an ACCESS_STATE structure.
333 
334  An AccessState may contain up to three embedded privileges. To
335  add more, this routine will allocate a block of memory, copy
336  the current privileges into it, and append the new privilege
337  to that block. A bit is set in the AccessState indicating that
338  the pointer to the privilege set in the structure points to pool
339  memory and must be deallocated.
340 
341 Arguments:
342 
343  AccessState - The AccessState structure representing the current
344  access attempt.
345 
346  Privileges - A pointer to a privilege set to be added.
347 
348 Return Value:
349 
350  STATUS_INSUFFICIENT_RESOURCES - an attempt to allocate pool memory
351  failed.
352 
353 --*/
354 
355 {
356  ULONG NewPrivilegeSetSize;
357  PPRIVILEGE_SET NewPrivilegeSet;
358  PAUX_ACCESS_DATA AuxData;
359 
360  PAGED_CODE();
361 
362  AuxData = (PAUX_ACCESS_DATA)AccessState->AuxData;
363 
364  if (Privileges->PrivilegeCount + AuxData->PrivilegesUsed->PrivilegeCount >
365  INITIAL_PRIVILEGE_COUNT) {
366 
367  //
368  // Compute the total size of the two privilege sets
369  //
370 
371  NewPrivilegeSetSize = SepPrivilegeSetSize( Privileges ) +
372  SepPrivilegeSetSize( AuxData->PrivilegesUsed );
373 
374  NewPrivilegeSet = ExAllocatePoolWithTag( PagedPool, NewPrivilegeSetSize, 'rPeS' );
375 
376  if (NewPrivilegeSet == NULL) {
377  return( STATUS_INSUFFICIENT_RESOURCES );
378  }
379 
380 
381  RtlCopyMemory(
382  NewPrivilegeSet,
383  AuxData->PrivilegesUsed,
384  SepPrivilegeSetSize( AuxData->PrivilegesUsed )
385  );
386 
387  //
388  // Note that this will adjust the privilege count in the
389  // structure for us.
390  //
391 
392  SepConcatenatePrivileges(
393  NewPrivilegeSet,
394  NewPrivilegeSetSize,
395  Privileges
396  );
397 
398  if (AccessState->PrivilegesAllocated) {
399  ExFreePool( AuxData->PrivilegesUsed );
400  }
401 
402  AuxData->PrivilegesUsed = NewPrivilegeSet;
403 
404  //
405  // Mark that we've allocated memory for the privilege set,
406  // so we know to free it when we're cleaning up.
407  //
408 
409  AccessState->PrivilegesAllocated = TRUE;
410 
411  } else {
412 
413  //
414  // Note that this will adjust the privilege count in the
415  // structure for us.
416  //
417 
418  SepConcatenatePrivileges(
419  AuxData->PrivilegesUsed,
420  sizeof(INITIAL_PRIVILEGE_SET),
421  Privileges
422  );
423 
424  }
425 
426  return( STATUS_SUCCESS );
427 
428 }

432 SepConcatenatePrivileges(
433  IN PPRIVILEGE_SET TargetPrivilegeSet,
434  IN ULONG TargetBufferSize,
435  IN PPRIVILEGE_SET SourcePrivilegeSet
436  )

438 /*++
439 
440 Routine Description:
441 
442  Takes two privilege sets and appends the second to the end of the
443  first.
444 
445  There must be enough space left at the end of the first privilege
446  set to contain the second.
447 
448 Arguments:
449 
450  TargetPrivilegeSet - Supplies a buffer containing a privilege set.
451  The buffer must be large enough to contain the second privilege
452  set.
453 
454  TargetBufferSize - Supplies the size of the target buffer.
455 
456  SourcePrivilegeSet - Supplies the privilege set to be copied
457  into the target buffer.
458 
459 Return Value:
460 
461  None
462 
463 --*/
464 
465 {
466  PVOID Base;
467  PVOID Source;
468  ULONG Length;
469 
470  PAGED_CODE();
471 
472  ASSERT( ((ULONG)SepPrivilegeSetSize( TargetPrivilegeSet ) +
473  (ULONG)SepPrivilegeSetSize( SourcePrivilegeSet ) -
474  SEP_PRIVILEGE_SET_HEADER_SIZE ) <=
475  TargetBufferSize
476  );
477 
478  Base = (PVOID)((ULONG_PTR)TargetPrivilegeSet + SepPrivilegeSetSize( TargetPrivilegeSet ));
479 
480  Source = (PVOID) ((ULONG_PTR)SourcePrivilegeSet + SEP_PRIVILEGE_SET_HEADER_SIZE);
481 
482  Length = SourcePrivilegeSet->PrivilegeCount * sizeof(LUID_AND_ATTRIBUTES);
483 
484  RtlMoveMemory(
485  Base,
486  Source,
487  Length
488  );
489 
490  TargetPrivilegeSet->PrivilegeCount += SourcePrivilegeSet->PrivilegeCount;
491 }
