49 NTSTATUS
50 WmipConstructMCAErrorEvent(
51  IN PMCA_EXCEPTION McaException,
52  IN ULONG ErrorLogSize,
53  IN OUT PWNODE_SINGLE_INSTANCE Wnode,
54  IN OUT PMSMCAEvent_Header Header,
55  IN OUT PUCHAR *RawPtr,
56  IN OUT BOOLEAN *IsFatal
57  )
58 /*++
59 
60 Routine Description:
61 
62  This routine attempts to build an MCA error log event from the information
63  in the supplied MCA_EXCEPTION.
64 
65 Arguments:
66 
67  McaException - Pointer to a MCA_EXCEPTION that encapsulates the machine
68  check event.
69 
70  ErrorLogSize - The total size in bytes of the raw error information.
71 
72  Wnode - Pointer to the error event WNODE.
73 
74  Header - Pointer to the common header for all MCA events.
75 
76  RawPtr - Address into which a pointer to the raw MCA data is to
77  be copied.
78 
79 Return Value:
80 
81  STATUS_SUCCESS - If an event is successfully constructed.
82 
83  STATUS_NOT_SUPPORTED - If the routine does not create an event.
84 
85 --*/
86 {
87  PMSMCAEvent_MemoryHierarchyError memoryEvent;
88  PMSMCAEvent_InvalidError invalidEvent;
89  PMSMCAEvent_TLBError tlbEvent;
90  PMSMCAEvent_BusError busEvent;
91  NTSTATUS status;
92  ULONG mcaCode;
93 
94  PAGED_CODE();
95 
96  if (*IsFatal) {
97  if (McaException->ExceptionType == HAL_MCA_RECORD) {
98  if (!McaException->u.Mca.Status.MciStats.UnCorrected) {
99  *IsFatal = FALSE;
100  }
101  }
102  }
103 
104  //
105  // Sanity checks on the size of the event structures. None should be
106  // larger than the size of a memory error event.
107  //
108  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
109  sizeof(MSMCAEvent_InvalidError));
110  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
111  sizeof(MSMCAEvent_MemoryHierarchyError));
112  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
113  sizeof(MSMCAEvent_TLBError));
114  WmipAssert(sizeof(MSMCAEvent_MemoryError) >=
115  sizeof(MSMCAEvent_BusError));
116 
117  //
118  // This routine is only to be called whe the MCA status bits are valid.
119  //
120  WmipAssert(McaException->u.Mca.Status.MciStats.Valid);
121  mcaCode = McaException->u.Mca.Status.MciStats.McaCod;
122  status = STATUS_SUCCESS;
123 
124  if (mcaCode == MCA_CODE_MICROCODE_ROM_PARITY_ERROR ||
125  mcaCode == MCA_CODE_EXTERNAL_ERROR ||
126  mcaCode == MCA_CODE_FRC_ERROR ||
127  mcaCode == MCA_CODE_INTERNAL_TIMER_ERROR) {
128 
129  //
130  // For Microcode ROM parity errors, external errors, FRC errors, and
131  // internal timer errors, we will use the InvalidError event since it
132  // is useful for generating error events with no parameters.
133  //
134  invalidEvent = (PMSMCAEvent_InvalidError)Header;
135 
136  //
137  // Indicate the error type.
138  //
139  if (mcaCode == MCA_CODE_MICROCODE_ROM_PARITY_ERROR) {
140  invalidEvent->Type = (ULONG)MCA_MICROCODE_ROM_PARITY_ERROR;
141  } else if (mcaCode == MCA_CODE_EXTERNAL_ERROR) {
142  invalidEvent->Type = (ULONG)MCA_EXTERNAL_ERROR;
143  } else if (mcaCode == MCA_CODE_FRC_ERROR) {
144  invalidEvent->Type = (ULONG)MCA_FRC_ERROR;
145  } else {
146  invalidEvent->Type = (ULONG)MCA_INTERNALTIMER_ERROR;
147  }
148 
149  //
150  // Fill in the GUID and the size of the data block.
151  //
152  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_InvalidErrGuid;
153  Wnode->SizeDataBlock =
154  FIELD_OFFSET(MSMCAEvent_InvalidError, RawRecord) + ErrorLogSize;
155 
156  //
157  // Copy the address of the raw record into the supplied address.
158  //
159  *RawPtr = invalidEvent->RawRecord;
160 
161  //
162  // Indicate the size of the entire record.
163  //
164  invalidEvent->Size = ErrorLogSize;
165 
166  } else if (mcaCode & MCA_CODE_BUS_ERR_MASK) {
167 
168  //
169  // Bus/Interconnect error. Extract the details of the error information
170  // from the preserved MCI_STATS and save it in the bus error event.
171  //
172 
173  busEvent = (PMSMCAEvent_BusError)Header;
174 
175  if (mcaCode & 0x0100) {
176  busEvent->Type = (ULONG)MCA_BUS_TIMEOUT_ERROR;
177  } else {
178  busEvent->Type = (ULONG)MCA_BUS_ERROR;
179  }
180 
181  busEvent->Participation = ((mcaCode & 0x00000600) >> 9);
182  busEvent->MemoryHierarchyLevel = (mcaCode & 0x00000003);
183  busEvent->RequestType = ((mcaCode & 0x000000F0) >> 4);
184  busEvent->MemOrIo = ((mcaCode & 0x0000000C) >> 2);
185 
186  if (McaException->u.Mca.Status.MciStats.AddressValid) {
187  busEvent->Address = McaException->u.Mca.Address.QuadPart;
188  } else {
189  busEvent->Address = (ULONG64)0;
190  }
191 
192  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_BusErrorGuid;
193  Wnode->SizeDataBlock = FIELD_OFFSET(MSMCAEvent_BusError,
194  RawRecord) + ErrorLogSize;
195 
196  busEvent->Size = ErrorLogSize;
197  *RawPtr = busEvent->RawRecord;
198 
199  } else if (mcaCode & MCA_CODE_MEM_HIERARCHY_ERR_MASK) {
200 
201  //
202  // Memory hierarchy error.
203  //
204 
205  memoryEvent = (PMSMCAEvent_MemoryHierarchyError)Header;
206 
207  memoryEvent->Type = (ULONG)MCA_MEMORYHIERARCHY_ERROR;
208  memoryEvent->TransactionType = ((mcaCode & 0x0000000C) >> 2);
209  memoryEvent->MemoryHierarchyLevel = (mcaCode & 0x00000003);
210  memoryEvent->RequestType = ((mcaCode & 0x000000F0) >> 4);
211 
212  if (McaException->u.Mca.Status.MciStats.AddressValid) {
213  memoryEvent->Address = McaException->u.Mca.Address.QuadPart;
214  } else {
215  memoryEvent->Address = (ULONG64)0;
216  }
217 
218  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_MemoryHierarchyErrorGuid;
219  Wnode->SizeDataBlock = FIELD_OFFSET(MSMCAEvent_MemoryHierarchyError,
220  RawRecord) + ErrorLogSize;
221 
222  memoryEvent->Size = ErrorLogSize;
223  *RawPtr = memoryEvent->RawRecord;
224 
225  } else if (mcaCode & MCA_CODE_TLB_ERR_MASK) {
226 
227  //
228  // TLB error.
229  //
230 
231  tlbEvent = (PMSMCAEvent_TLBError)Header;
232 
233  tlbEvent->Type = (ULONG)MCA_TLB_ERROR;
234  tlbEvent->TransactionType = ((mcaCode & 0x0000000C) >> 2);
235  tlbEvent->MemoryHierarchyLevel = (mcaCode & 0x00000003);
236 
237  if (McaException->u.Mca.Status.MciStats.AddressValid) {
238  tlbEvent->Address = McaException->u.Mca.Address.QuadPart;
239  } else {
240  tlbEvent->Address = (ULONG64)0;
241  }
242 
243  Wnode->WnodeHeader.Guid = WmipMSMCAEvent_TLBErrorGuid;
244  Wnode->SizeDataBlock = FIELD_OFFSET(MSMCAEvent_TLBError,
245  RawRecord) + ErrorLogSize;
246 
247  tlbEvent->Size = ErrorLogSize;
248  *RawPtr = tlbEvent->RawRecord;
249 
250  } else {
251 
252  //
253  // No error, unclassified errors, and internal unclassified errors are
254  // treated the same way... we ignore them.
255  //
256  status = STATUS_NOT_SUPPORTED;
257  }
258 
259  return status;
260 }
