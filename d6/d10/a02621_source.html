47 /*++
48 
49 Routine Description:
50 
51  This function searches the currently active function tables for an entry
52  that corresponds to the specified control PC. This function does not
53  consider the runtime function table.
54 
55 Arguments:
56 
57  ControlPc - Supplies the address of an instruction within the specified
58  function.
59 
60  ImageBase - Supplies the address of a variable that receives the image base
61  if a function table entry contains the specified control PC.
62 
63 Return Value:
64 
65  If there is no entry in the function table for the specified PC, then
66  NULL is returned. Otherwise, the address of the function table entry
67  that corresponds to the specified PC is returned.
68 
69 --*/
70 
71 {
72 
73  ULONG64 BaseAddress;
74  ULONG64 BeginAddress;
75  ULONG64 EndAddress;
76  PRUNTIME_FUNCTION FunctionEntry;
77  PRUNTIME_FUNCTION FunctionTable;
78  LONG High;
79  ULONG Index;
80  LONG Low;
81  LONG Middle;
82  ULONG RelativePc;
83  ULONG SizeOfTable;
84 
85  //
86  // Attempt to find an image that contains the specified control PC. If
87  // an image is found, then search its function table for a function table
88  // entry that contains the specified control PC.
89  // If an image is not found then do not search the dynamic function table.
90  // The dynamic function table has the same lock as the loader lock,
91  // and searching that table could deadlock a caller of RtlpWalkFrameChain
92  //
93 
94  //
95  // attempt to find a matching entry in the loaded module list.
96  //
97 
98  FunctionTable = RtlLookupFunctionTable((PVOID)ControlPc,
99  (PVOID *)ImageBase,
100  &SizeOfTable);
101 
102  //
103  // If a function table is located, then search for a function table
104  // entry that contains the specified control PC.
105  //
106 
107  if (FunctionTable != NULL) {
108  Low = 0;
109  High = (SizeOfTable / sizeof(RUNTIME_FUNCTION)) - 1;
110  RelativePc = (ULONG)(ControlPc - *ImageBase);
111  while (High >= Low) {
112 
113  //
114  // Compute next probe index and test entry. If the specified
115  // control PC is greater than of equal to the beginning address
116  // and less than the ending address of the function table entry,
117  // then return the address of the function table entry. Otherwise,
118  // continue the search.
119  //
120 
121  Middle = (Low + High) >> 1;
122  FunctionEntry = &FunctionTable[Middle];
123 
124  if (RelativePc < FunctionEntry->BeginAddress) {
125  High = Middle - 1;
126 
127  } else if (RelativePc >= FunctionEntry->EndAddress) {
128  Low = Middle + 1;
129 
130  } else {
131  break;
132  }
133  }
134 
135  if (High < Low) {
136  FunctionEntry = NULL;
137  }
138 
139  } else {
140 
141  FunctionEntry = NULL;
142 
143  }
144 
145  return RtlpConvertFunctionEntry(FunctionEntry, *ImageBase);
146 }

158 /*++
159 
160 Routine Description:
161 
162  This routine captures a stack back trace by walking up the stack and
163  recording the information for each frame.
164 
165  N.B. This is an exported function that MUST probe the ability to take
166  page faults.
167 
168 Arguments:
169 
170  FramesToSkip - Supplies the number of frames to skip over at the start
171  of the back trace.
172 
173  FramesToCapture - Supplies the number of frames to be captured.
174 
175  BackTrace - Supplies a pointer to the back trace buffer.
176 
177  BackTraceHash - Optionally supples a pointer to the computed hash value.
178 
179 Return Value:
180 
181  The number of captured frames is returned as the function value.
182 
183 --*/
184 
185 {
186 
187  ULONG FramesFound;
188  ULONG HashValue;
189  ULONG Index;
190  PVOID Trace[2 * MAX_STACK_DEPTH];
191 
192  //
193  // In kernel mode avoid running at IRQL levels where page faults cannot
194  // be taken. The walking code will access various sections from driver
195  // and system images and this will cause page faults. Also the walking
196  // code needs to bail out if the current thread is processing a page
197  // fault since collided faults may occur.
198  //
199 
200  if (MmCanThreadFault() == FALSE) {
201  return 0;
202  }
203 
204  if (PsGetCurrentThread ()->ActiveFaultCount != 0) {
205  return 0;
206  }
207 
208  //
209  // If the number of frames to capture plus the number of frames to skip
210  // (one additional frame is skipped for the call to walk the chain), then
211  // return zero.
212  //
213 
214  FramesToSkip += 1;
215  if ((FramesToCapture + FramesToSkip) >= (2 * MAX_STACK_DEPTH)) {
216  return 0;
217  }
218 
219  //
220  // Capture the stack back trace.
221  //
222 
223  FramesFound = RtlWalkFrameChain(&Trace[0],
224  FramesToCapture + FramesToSkip,
225  0);
226 
227  //
228  // If the number of frames found is less than the number of frames to
229  // skip, then return zero.
230  //
231 
232  if (FramesFound <= FramesToSkip) {
233  return 0;
234  }
235 
236  //
237  // Compute the hash value and transfer the captured trace to the back
238  // trace buffer.
239  //
240 
241  HashValue = 0;
242  for (Index = 0; Index < FramesToCapture; Index += 1) {
243  if (FramesToSkip + Index >= FramesFound) {
244  break;
245  }
246 
247  BackTrace[Index] = Trace[FramesToSkip + Index];
248  HashValue += PtrToUlong(BackTrace[Index]);
249  }
250 
251  if (ARGUMENT_PRESENT(BackTraceHash)) {
252  *BackTraceHash = HashValue;
253  }
254 
255  return (USHORT)Index;
256 }

267 /*++
268 
269 Routine Description:
270 
271  This routine returns the address of the call to the routine that called
272  this routine, and the address of the call to the routine that called
273  the routine that called this routine. For example, if A called B called
274  C which called this routine, the return addresses in B and A would be
275  returned.
276 
277  N.B. This is an exported function that MUST probe the ability to take
278  page faults.
279 
280 Arguments:
281 
282  CallersPc - Supplies a pointer to a variable that receives the address
283  of the caller of the caller of this routine (B).
284 
285  CallersCallersPc - Supplies a pointer to a variable that receives the
286  address of the caller of the caller of the caller of this routine
287  (A).
288 
289 Return Value:
290 
291  None.
292 
293 Note:
294 
295  If either of the calling stack frames exceeds the limits of the stack,
296  they are set to NULL.
297 
298 --*/

300 {
301 
302  CONTEXT ContextRecord;
303  ULONG64 EstablisherFrame;
304  PRUNTIME_FUNCTION FunctionEntry;
305  PVOID HandlerData;
306  ULONG64 HighLimit;
307  ULONG64 ImageBase;
308  ULONG64 LowLimit;
309 
310  //
311  // Assume the function table entries for the various routines cannot be
312  // found or there are not three procedure activation records on the stack.
313  //
314 
315  *CallersPc = NULL;
316  *CallersCallersPc = NULL;
317 
318  //
319  // In kernel mode avoid running at IRQL levels where page faults cannot
320  // be taken. The walking code will access various sections from driver
321  // and system images and this will cause page faults. Also the walking
322  // code needs to bail out if the current thread is processing a page
323  // fault since collided faults may occur.
324  //
325 
326  if (MmCanThreadFault() == FALSE) {
327  return;
328  }
329 
330  if (PsGetCurrentThread ()->ActiveFaultCount != 0) {
331  return;
332  }
333

336  // unwind to the caller of this routine, and lookup function table entry.

342  &ImageBase);
343

345  // Attempt to unwind to the caller of this routine (C).
346  //

375  NULL);

382  //
383 
384  FunctionEntry = RtlpLookupFunctionEntryForStackWalks(ContextRecord.Rip,
385  &ImageBase);
386 
387  if ((FunctionEntry != NULL) &&
388  ((RtlpIsFrameInBounds(&LowLimit, ContextRecord.Rsp, &HighLimit) == TRUE))) {
389 
390  RtlVirtualUnwind(UNW_FLAG_NHANDLER,
391  ImageBase,
392  ContextRecord.Rip,
393  FunctionEntry,
394  &ContextRecord,
395  &HandlerData,
396  &EstablisherFrame,
397  NULL);

409 RtlpWalkFrameChain (
410  OUT PVOID *Callers,
411  IN ULONG Count,
412  IN ULONG Flags,
413  IN ULONG FramesToSkip
414  )
415 
416 /*++
417 
418 Routine Description:
419 
420  This function attempts to walk the call chain and capture a vector with
421  a specified number of return addresses. It is possible that the function
422  cannot capture the requested number of callers, in which case, the number
423  of captured return addresses will be returned.
424 
425  N.B. The ability to take page faults is checked in the wrapper function.
426 
427 Arguments:
428 
429  Callers - Supplies a pointer to an array that is to received the return
430  address values.
431 
432  Count - Supplies the number of frames to be walked.
433 
434  Flags - Supplies the flags value (unused).
435 
436  FramesToSkip - Supplies the number of frames to skip.
437 
438 Return value:
439 
440  The number of captured return addresses.
441 
442 --*/
443 
444 {
445 
446  CONTEXT ContextRecord;
447  ULONG64 EstablisherFrame;
448  PRUNTIME_FUNCTION FunctionEntry;
449  PVOID HandlerData;
450  ULONG64 HighLimit;
451  ULONG64 ImageBase;
452  ULONG Index;
453  ULONG64 LowLimit;
454 
455  //
456  // No flag values are currently supported on amd64 platforms.
457  //
458 
459  if (Flags != 0) {
460  return 0;
461  }
462 
463  //
464  // Get current stack limits and capture the current context.
465  //
466 
467  RtlpGetStackLimits(&LowLimit, &HighLimit);
468  RtlCaptureContext(&ContextRecord);
469 
470  //
471  // Capture the requested number of return addresses if possible.
472  //
473 
474  InterlockedIncrement (&RtlpStackWalksInProgress);
475 
476  Index = 0;
477  try {
478  while ((Index < Count) && (ContextRecord.Rip != 0)) {
479 
480  //
481  // Check the next PC value to make sure it is valid in the
482  // current process.
483  //
484 
485  if ((MmIsSessionAddress((PVOID)ContextRecord.Rip) == TRUE &&
486  MmGetSessionId(PsGetCurrentProcess()) == 0) ||
487  (MmIsAddressValid((PVOID)ContextRecord.Rip) == FALSE)) {
488 
489  break;
490  }
491 
492  //
493  // Lookup the function table entry using the point at which control
494  // left the function.
495  //
496 
497  FunctionEntry = RtlpLookupFunctionEntryForStackWalks(ContextRecord.Rip,
498  &ImageBase);
499 
500  //
501  // If there is a function table entry for the routine and the stack is
502  // within limits, then virtually unwind to the caller of the routine
503  // to obtain the return address. Otherwise, discontinue the stack walk.
504  //
505 
506  if ((FunctionEntry != NULL) &&
507  ((RtlpIsFrameInBounds(&LowLimit, ContextRecord.Rsp, &HighLimit) == TRUE))) {
508 
509  RtlVirtualUnwind(UNW_FLAG_NHANDLER,
510  ImageBase,
511  ContextRecord.Rip,
512  FunctionEntry,
513  &ContextRecord,
514  &HandlerData,
515  &EstablisherFrame,
516  NULL);
517 
518  if (FramesToSkip != 0){
519  FramesToSkip -= 1;
520 
521  } else {
522  Callers[Index] = (PVOID)ContextRecord.Rip;
523  Index += 1;
524  }
525 
526  } else {
527  break;
528  }
529  }
530 
531  } except (RtlpWalkFrameChainExceptionFilter(GetExceptionCode(),
532  GetExceptionInformation())) {
533 
534  Index = 0;
535  }
536 
537  InterlockedDecrement (&RtlpStackWalksInProgress);
538 
539  return Index;
540 }
