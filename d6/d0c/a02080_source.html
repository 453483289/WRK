143 CmpCmdHiveClose(
144  PCMHIVE CmHive
145  )
146 /*++
147 
148 Routine Description:
149 
150  Closes all the file handles for the specified hive
151 Arguments:
152 
153  CmHive - the hive to close
154 
155 Return Value:
156  none
157 --*/
158 {
159  ULONG i;
160  IO_STATUS_BLOCK IoStatusBlock;
161  FILE_BASIC_INFORMATION BasicInfo;
162  LARGE_INTEGER systemtime;
163  BOOLEAN oldFlag;
164 
165  CM_PAGED_CODE();
166 
167  //
168  // disable hard error popups, to workaround ObAttachProcessStack
169  //
170  oldFlag = IoSetThreadHardErrorMode(FALSE);
171 
172  //
173  // Close the files associated with this hive.
174  //
175  ASSERT_CM_LOCK_OWNED_EXCLUSIVE_OR_HIVE_LOADING(CmHive);
176 
177  for (i=0; i<HFILE_TYPE_MAX; i++) {
178  if (CmHive->FileHandles[i] != NULL) {
179  //
180  // attempt to set change the last write time (profile guys are relying on it!)
181  //
182  if( i == HFILE_TYPE_PRIMARY ) {
183  if( NT_SUCCESS(ZwQueryInformationFile(
184  CmHive->FileHandles[i],
185  &IoStatusBlock,
186  &BasicInfo,
187  sizeof(BasicInfo),
188  FileBasicInformation) ) ) {
189 
190  KeQuerySystemTime(&systemtime);
191 
192  if( CmHive->Hive.DirtyFlag ) {
193  BasicInfo.LastWriteTime = systemtime;
194  }
195  BasicInfo.LastAccessTime = systemtime;
196 
197  ZwSetInformationFile(
198  CmHive->FileHandles[i],
199  &IoStatusBlock,
200  &BasicInfo,
201  sizeof(BasicInfo),
202  FileBasicInformation
203  );
204  }
205 
206  CmpTrackHiveClose = TRUE;
207  CmCloseHandle(CmHive->FileHandles[i]);
208  CmpTrackHiveClose = FALSE;
209 
210  } else {
211  CmCloseHandle(CmHive->FileHandles[i]);
212  }
213 
214  CmHive->FileHandles[i] = NULL;
215  }
216  }
217  //
218  // restore hard error popups mode
219  //
220  IoSetThreadHardErrorMode(oldFlag);
221 }

556 VOID
557 CmpLazyFlushWorker(
558  IN PVOID Parameter
559  )
560 
561 /*++
562 
563 Routine Description:
564 
565  Worker routine called to do a lazy flush. Called by an executive worker
566  thread in the system process.
567 
568 Arguments:
569 
570  Parameter - not used.
571 
572 Return Value:
573 
574  None.
575 
576 --*/
577 
578 {
579  BOOLEAN Result = TRUE;
580  BOOLEAN ForceFlush;
581  BOOLEAN PostNewWorker = FALSE;
582  ULONG DirtyCount = 0;
583 
584  CM_PAGED_CODE();
585 
586  UNREFERENCED_PARAMETER (Parameter);
587 
588  if( CmpHoldLazyFlush ) {
589  //
590  // lazy flush mode is disabled
591  //
592  return;
593  }
594 
595  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"CmpLazyFlushWorker: flushing hives\n"));
596 
597  ForceFlush = CmpForceForceFlush;
598  if(ForceFlush == TRUE) {
599  //
600  // something bad happened and we may need to fix hive's use count
601  //
602  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_TRACE_LEVEL,"CmpLazyFlushWorker: Force Flush - getting the reglock exclusive\n"));
603  CmpLockRegistryExclusive();
604  } else {
605  CmpLockRegistry();
606  ENTER_FLUSH_MODE();
607  }
608  if (!HvShutdownComplete) {
609  //
610  // no loads/unloads while flush in progress
611  //
612  LOCK_HIVE_LOAD();
613  PostNewWorker = CmpDoFlushNextHive(ForceFlush,&Result,&DirtyCount);
614  UNLOCK_HIVE_LOAD();
615 
616  if( !PostNewWorker ) {
617  //
618  // we have completed a sweep through the entire list of hives
619  //
620  InterlockedIncrement( (PLONG)&CmpLazyFlushCount );
621  }
622  } else {
623  CmpForceForceFlush = FALSE;
624  }
625 
626  if( ForceFlush == FALSE ) {
627  EXIT_FLUSH_MODE();
628  }
629 
630  CmpLazyFlushPending = FALSE;
631  CmpUnlockRegistry();
632 
633  if( CmpCannotWriteConfiguration ) {
634  //
635  // Disk full; system hive has not been saved at initialization
636  //
637  if(!Result) {
638  //
639  // All hives were saved; No need for disk full warning anymore
640  //
641  CmpCannotWriteConfiguration = FALSE;
642  } else {
643  //
644  // Issue another hard error (if not already displayed) and postpone a lazy flush operation
645  //
646  CmpDiskFullWarning();
647  CmpLazyFlush();
648  }
649  }
650  //
651  // if we have not yet flushed the whole list; or there are still hives dirty from the curently ended iteration.
652  //
653  if( PostNewWorker || (DirtyCount != 0) ) {
654  //
655  // post a new worker to flush the next hive
656  //
657  CmpLazyFlush();
658  }
659 
660 }
