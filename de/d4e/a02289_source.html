17 // -------------------------------------------------------------------------------
18 // Introduction
19 //
20 // SpecStrings.h provides a set of annotations to describe how a function uses its
21 // parameters - the assumptions it makes about them, and the guarantees it makes
22 // upon finishing.
23 //
24 // Annotations may be placed before either a function parameter's type or its return
25 // type, and describe the function's behavior regarding the parameter or return value.
26 // There are two classes of annotations: buffer annotations and advanced annotations.
27 // Buffer annotations describe how functions use their pointer parameters, and
28 // advanced annotations either describe complex/unusual buffer behavior, or provide
29 // additional information about a parameter that is not otherwise expressible.
30 //
31 // -------------------------------------------------------------------------------
32 // Buffer Annotations
33 //
34 // The most important annotations in SpecStrings.h provide a consistent way to annotate
35 // buffer parameters or return values for a function. Each of these annotations describes
36 // a single buffer (which could be a string, a fixed-length or variable-length array,
37 // or just a pointer) that the function interacts with: where it is, how large it is,
38 // how much is initialized, and what the function does with it.
39 //
40 // The appropriate macro for a given buffer can be constructed using the table below.
41 // Just pick the appropriate values from each category, and combine them together
42 // with a leading underscore. Some combinations of values do not make sense as buffer
43 // annotations. Only meaningful annotations can be added to your code; for a list of
44 // these, see the buffer annotation definitions section.
45 //
46 // Only a single buffer annotation should be used for each parameter.
47 //
48 // |------------|------------|---------|--------|----------|---------------|
49 // | Level | Usage | Size | Output | Optional | Parameters |
50 // |------------|------------|---------|--------|----------|---------------|
51 // | <> | <> | <> | <> | <> | <> |
52 // | _deref | _in | _ecount | _full | _opt | (size) |
53 // | _deref_opt | _out | _bcount | _part | | (size,length) |
54 // | | _inout | | | | |
55 // | | | | | | |
56 // |------------|------------|---------|--------|----------|---------------|
57 //
58 // Level: Describes the buffer pointer's level of indirection from the parameter or
59 // return value 'p'.
60 //
61 // <> : p is the buffer pointer.
62 // _deref : *p is the buffer pointer. p must not be NULL.
63 // _deref_opt : *p may be the buffer pointer. p may be NULL, in which case the rest of
64 // the annotation is ignored.
65 //
66 // Usage: Describes how the function uses the buffer.
67 //
68 // <> : The buffer is not accessed. If used on the return value or with _deref, the
69 // function will provide the buffer, and it will be uninitialized at exit.
70 // Otherwise, the caller must provide the buffer. This should only be used
71 // for alloc and free functions.
72 // _in : The function will only read from the buffer. The caller must provide the
73 // buffer and initialize it.
74 // _out : The function will only write to the buffer. If used on the return value or
75 // with _deref, the function will provide the buffer and initialize it.
76 // Otherwise, the caller must provide the buffer, and the function will
77 // initialize it.
78 // _inout : The function may freely read from and write to the buffer. The caller must
79 // provide the buffer and initialize it. If used with _deref, the buffer may
80 // be reallocated by the function.
81 //
82 // Size: Describes the total size of the buffer. This may be less than the space actually
83 // allocated for the buffer, in which case it describes the accessible amount.
84 //
85 // <> : No buffer size is given. If the type specifies the buffer size (such as
86 // with LPSTR and LPWSTR), that amount is used. Otherwise, the buffer is one
87 // element long. Must be used with _in, _out, or _inout.
88 // _ecount : The buffer size is an explicit element count.
89 // _bcount : The buffer size is an explicit byte count.
90 //
91 // Output: Describes how much of the buffer will be initialized by the function. For
92 // _inout buffers, this also describes how much is initialized at entry. Omit this
93 // category for _in buffers; they must be fully initialized by the caller.
94 //
95 // <> : The type specifies how much is initialized. For instance, a function initializing
96 // an LPWSTR must NULL-terminate the string.
97 // _full : The function initializes the entire buffer.
98 // _part : The function initializes part of the buffer, and explicitly indicates how much.
99 //
100 // Optional: Describes if the buffer itself is optional.
101 //
102 // <> : The pointer to the buffer must not be NULL.
103 // _opt : The pointer to the buffer might be NULL. It will be checked before being dereferenced.
104 //
105 // Parameters: Gives explicit counts for the size and length of the buffer.
106 //
107 // <> : There is no explicit count. Use when neither _ecount nor _bcount is used.
108 // (size) : Only the buffer's total size is given. Use with _ecount or _bcount but not _part.
109 // (size,length) : The buffer's total size and initialized length are given. Use with _ecount_part
110 // and _bcount_part.
111 //
112 // -------------------------------------------------------------------------------
113 // Buffer Annotation Examples
114 //
115 // LWSTDAPI_(BOOL) StrToIntExA(
116 // LPCSTR pszString, // No annotation required, const implies __in.
117 // DWORD dwFlags,
118 // __out int *piRet // A pointer whose dereference will be filled in.
119 // );
120 //
121 // void MyPaintingFunction(
122 // __in HWND hwndControl, // An initialized read-only parameter.
123 // __in_opt HDC hdcOptional, // An initialized read-only parameter that might be NULL.
124 // __inout IPropertyStore *ppsStore // An initialized parameter that may be freely used
125 // // and modified.
126 // );
127 //
128 // LWSTDAPI_(BOOL) PathCompactPathExA(
129 // __out_ecount(cchMax) LPSTR pszOut, // A string buffer with cch elements that will
130 // // be NULL terminated on exit.
131 // LPCSTR pszSrc, // No annotation required, const implies __in.
132 // UINT cchMax,
133 // DWORD dwFlags
134 // );
135 //
136 // HRESULT SHLocalAllocBytes(
137 // size_t cb,
138 // __deref_bcount(cb) T **ppv // A pointer whose dereference will be set to an
139 // // uninitialized buffer with cb bytes.
140 // );
141 //
142 // __inout_bcount_full(cb) : A buffer with cb elements that is fully initialized at
143 // entry and exit, and may be written to by this function.
144 //
145 // __out_ecount_part(count, *countOut) : A buffer with count elements that will be
146 // partially initialized by this function. The function indicates how much it
147 // initialized by setting *countOut.
148 //
149 // -------------------------------------------------------------------------------
150 // Advanced Annotations
151 //
152 // Advanced annotations describe behavior that is not expressible with the regular
153 // buffer macros. These may be used either to annotate buffer parameters that involve
154 // complex or conditional behavior, or to enrich existing annotations with additional
155 // information.
156 //
157 // __success(expr) f :
158 // <expr> indicates whether function f succeeded or not. If <expr> is true at exit,
159 // all the function's guarantees (as given by other annotations) must hold. If <expr>
160 // is false at exit, the caller should not expect any of the function's guarantees
161 // to hold. If not used, the function must always satisfy its guarantees. Added
162 // automatically to functions that indicate success in standard ways, such as by
163 // returning an HRESULT.
164 //
165 // __out_awcount(expr, size) p :
166 // Pointer p is a buffer whose size may be given in either bytes or elements. If
167 // <expr> is true, this acts like __out_bcount. If <expr> is false, this acts
168 // like __out_ecount. This should only be used to annotate old APIs.
169 //
170 // __in_awcount(expr, size) p :
171 // Pointer p is a buffer whose size may be given in either bytes or elements. If
172 // <expr> is true, this acts like __in_bcount. If <expr> is false, this acts
173 // like __in_ecount. This should only be used to annotate old APIs.
174 //
175 // __nullterminated p :
176 // Pointer p is a buffer that may be read or written up to and including the first
177 // NULL character or pointer. May be used on typedefs, which marks valid (properly
178 // initialized) instances of that type as being NULL-terminated.
179 //
180 // __nullnullterminated p :
181 // Pointer p is a buffer that may be read or written up to and including the first
182 // sequence of two NULL characters or pointers. May be used on typedefs, which marks
183 // valid instances of that type as being double-NULL terminated.
184 //
185 // __reserved v :
186 // Value v must be 0/NULL, reserved for future use.
187 //
188 // __checkReturn v :
189 // Return value v must not be ignored by callers of this function.
190 //
191 // __typefix(ctype) v :
192 // Value v should be treated as an instance of ctype, rather than its declared type.
193 //
194 // __override f :
195 // Specify C#-style 'override' behavior for overriding virtual methods.
196 //
197 // __callback f :
198 // Function f can be used as a function pointer.
199 //
200 // __format_string p :
201 // Pointer p is a string that contains % markers in the style of printf.
202 //
203 // __blocksOn(resource) f :
204 // Function f blocks on the resource 'resource'.
205 //
206 // __fallthrough :
207 // Annotates switch statement labels where fall-through is desired, to distinguish
208 // from forgotten break statements.
209 //
210 // -------------------------------------------------------------------------------
211 // Advanced Annotation Examples
212 //
213 // __success(return == TRUE) LWSTDAPI_(BOOL)
214 // PathCanonicalizeA(__out_ecount(MAX_PATH) LPSTR pszBuf, LPCSTR pszPath) :
215 // pszBuf is only guaranteed to be NULL-terminated when TRUE is returned.
216 //
217 // typedef __nullterminated WCHAR* LPWSTR : Initialized LPWSTRs are NULL-terminated strings.
218 //
219 // __out_ecount(cch) __typefix(LPWSTR) void *psz : psz is a buffer parameter which will be
220 // a NULL-terminated WCHAR string at exit, and which initially contains cch WCHARs.
221 //
222 // -------------------------------------------------------------------------------

257  // In the primitive __declspec("SAL_*") annotations "SAL" stands for Standard
258  // Annotation Language. These __declspec("SAL_*") annotations are the
259  // primitives the compiler understands and all high-level SpecString MACROs

266  // __notnull p

268  //
269  // Annotates a pointer p. States that pointer p is null. Commonly used
270  // in the negated form __notnull or the possibly null form __maybenull.

278  // __notreadonly l
279  // __mabyereadonly l
280  //
281  // Annotates a location l. States that location l is not modified after
282  // this point. If the annotation is placed on the precondition state of
283  // a function, the restriction only applies until the postcondition state
284  // of the function. __maybereadonly states that the annotated location
285  // may be modified, whereas __notreadonly states that a location must be
286  // modified.
287  //

394  // __exceptthat
395  //
396  // Given a set of annotations Q containing __exceptthat maybeP, the effect of
397  // the except clause is to erase any P or notP annotations (explicit or
398  // implied) within Q at the same level of dereferencing that the except
399  // clause appears, and to replace it with maybeP.
400  //
401  // Example 1: __valid __exceptthat __maybenull on a pointer p means that the
402  // pointer may be null, and is otherwise valid, thus overriding
403  // the implicit notnull annotation implied by __valid on
404  // pointers.
405  //
406  // Example 2: __valid __deref __exceptthat __maybenull on an int **p means
407  // that p is not null (implied by valid), but the elements
408  // pointed to by p could be null, and are otherwise valid.
409  //
