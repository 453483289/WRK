28 NTSTATUS
29 NtImpersonateClientOfPort (
30  __in HANDLE PortHandle,
31  __in PPORT_MESSAGE Message
32  )
33 
34 /*++
35 
36 Routine Description:
37 
38  This procedure is used by the server thread to temporarily acquire the
39  identifier set of a client thread.
40 
41  This service establishes an impersonation token for the calling thread.
42  The impersonation token corresponds to the context provided by the port
43  client. The client must currently be waiting for a reply to the
44  specified message.
45 
46  This service returns an error status code if the client thread is not
47  waiting for a reply to the message. The security quality of service
48  parameters specified by the client upon connection dictate what use the
49  server will have of the client's security context.
50 
51  For complicated or extended impersonation needs, the server may open a
52  copy of the client's token (using NtOpenThreadToken()). This must be
53  done while impersonating the client.
54 
55 Arguments:
56 
57  PortHandle - Specifies the handle of the communication port that the
58  message was received from.
59 
60  Message - Specifies an address of a message that was received from the
61  client that is to be impersonated. The ClientId field of the message
62  identifies the client thread that is to be impersonated. The client
63  thread must be waiting for a reply to the message in order to
64  impersonate the client.
65 
66 Return Value:
67 
68  NTSTATUS - Status code that indicates whether or not the operation was
69  successful.
70 
71 --*/
72 
73 {
74  PLPCP_PORT_OBJECT PortObject;
75  PLPCP_PORT_OBJECT ConnectedPort;
76  KPROCESSOR_MODE PreviousMode;
77  NTSTATUS Status;
78  PETHREAD ClientThread;
79  CLIENT_ID CapturedClientId;
80  ULONG CapturedMessageId;
81  SECURITY_CLIENT_CONTEXT DynamicSecurity;
82 
83  PAGED_CODE();
84 
85  //
86  // Get previous processor mode and probe output arguments if necessary.
87  //
88 
89  PreviousMode = KeGetPreviousMode();
90 
91  if (PreviousMode != KernelMode) {
92 
93  try {
94 
95  ProbeForReadSmallStructure( Message, sizeof( PORT_MESSAGE ), sizeof( ULONG ));
96 
97  CapturedClientId = Message->ClientId;
98  CapturedMessageId = Message->MessageId;
99 
100  } except( EXCEPTION_EXECUTE_HANDLER ) {
101 
102  return( GetExceptionCode() );
103  }
104 
105  } else {
106 
107  CapturedClientId = Message->ClientId;
108  CapturedMessageId = Message->MessageId;
109  }
110 
111  //
112  // Reference the communication port object by handle. Return status if
113  // unsuccessful.
114  //
115 
116  Status = LpcpReferencePortObject( PortHandle, 0,
117  PreviousMode, &PortObject );
118  if (!NT_SUCCESS( Status )) {
119 
120  return( Status );
121  }
122 
123  //
124  // It is an error to try this on any port other than a server
125  // communication port
126  //
127 
128  if ((PortObject->Flags & PORT_TYPE) != SERVER_COMMUNICATION_PORT) {
129 
130  ObDereferenceObject( PortObject );
131 
132  return( STATUS_INVALID_PORT_HANDLE );
133  }
134 
135  //
136  // Translate the ClientId from the connection request into a
137  // thread pointer. This is a referenced pointer to keep the thread
138  // from evaporating out from under us.
139  //
140 
141  Status = PsLookupProcessThreadByCid( &CapturedClientId,
142  NULL,
143  &ClientThread );
144 
145  if (!NT_SUCCESS( Status )) {
146 
147  ObDereferenceObject( PortObject );
148 
149  return( Status );
150  }
151 
152  //
153  // Acquire the mutex that guards the LpcReplyMessage field of
154  // the thread and get the pointer to the message that the thread
155  // is waiting for a reply to.
156  //
157 
158  LpcpAcquireLpcpLock();
159 
160  //
161  // The connected port can be in a state with 0 references in a deletion process.
162  // We need to test this case while referencing it.
163  //
164 
165  ConnectedPort = PortObject->ConnectedPort;
166 
167  if ( ( ConnectedPort == NULL ) ||
168  ( !ObReferenceObjectSafe( ConnectedPort ) ) ) {
169 
170  //
171  // The port is being deleted. Quit this function with
172  // appropriate return status.
173  // We don't need to dereference the connected port because
174  // it is anyway about to be deleted
175  //
176 
177  LpcpReleaseLpcpLock();
178 
179  ObDereferenceObject( PortObject );
180  ObDereferenceObject( ClientThread );
181 
182  return( STATUS_PORT_DISCONNECTED );
183  }
184 
185  //
186  // See if the thread is waiting for a reply to the message
187  // specified on this call, if the user gave us a bad
188  // message id. If not then a bogus message
189  // has been specified, so return failure.
190  //
191 
192  //
193  // The W2k fix searched the client thread in the rundown queue, to make sure
194  // we are not impersonating a port from a different connection. Ones we added the port
195  // to the thread structure to fix other security issues for reply os accessing data
196  // we can use that easy test for the impersonation too, w/o searching the rundown queue
197  //
198 
199  if ((ClientThread->LpcReplyMessageId != CapturedMessageId)
200  ||
201  (CapturedMessageId == 0)
202  ||
203  (!LpcpValidateClientPort( ClientThread,
204  PortObject,
205  LPCP_VALIDATE_REASON_IMPERSONATION)) ) {
206 
207  LpcpReleaseLpcpLock();
208 
209  ObDereferenceObject( PortObject );
210  ObDereferenceObject( ClientThread );
211  ObDereferenceObject( ConnectedPort );
212 
213  return (STATUS_REPLY_MESSAGE_MISMATCH);
214  }
215 
216  //
217  // Test whether the client allows impersonation for this message or not.
218  //
219 
220  if (LpcpGetThreadAttributes(ClientThread) & LPCP_NO_IMPERSONATION) {
221 
222  LpcpReleaseLpcpLock();
223 
224  ObDereferenceObject( PortObject );
225  ObDereferenceObject( ClientThread );
226  ObDereferenceObject( ConnectedPort );
227 
228  return (STATUS_ACCESS_DENIED);
229  }
230 
231  LpcpReleaseLpcpLock();
232 
233  //
234  // If the client requested dynamic security tracking, then the client
235  // security needs to be referenced. Otherwise, (static case)
236  // it is already in the client's port.
237  //
238 
239  if (ConnectedPort->Flags & PORT_DYNAMIC_SECURITY) {
240 
241  //
242  // Impersonate the client with information from the queued message
243  //
244 
245  Status = LpcpGetDynamicClientSecurity( ClientThread,
246  ConnectedPort,
247  &DynamicSecurity );
248 
249  if (!NT_SUCCESS( Status )) {
250 
251  ObDereferenceObject( PortObject );
252  ObDereferenceObject( ClientThread );
253  ObDereferenceObject( ConnectedPort );
254 
255  return( Status );
256  }
257 
258  Status = SeImpersonateClientEx( &DynamicSecurity, NULL );
259 
260  LpcpFreeDynamicClientSecurity( &DynamicSecurity );
261 
262  } else {
263 
264  //
265  // Impersonate the client with information from the client's port
266  //
267 
268  Status = SeImpersonateClientEx( &ConnectedPort->StaticSecurity, NULL );
269 
270  }
271 
272  ObDereferenceObject( PortObject );
273  ObDereferenceObject( ClientThread );
274  ObDereferenceObject( ConnectedPort );
275 
276  //
277  // And return to our caller
278  //
279 
280  return Status;
281 }

285 LpcpFreePortClientSecurity (
286  IN PLPCP_PORT_OBJECT Port
287  )
288 
289 /*++
290 
291 Routine Description:
292 
293  This routine cleans up the captured security context for a client port.
294  The cleanup is typically done when we are deleting a port
295 
296 Arguments:
297 
298  Port - Supplies the client port being deleted
299 
300 Return Value:
301 
302  None.
303 
304 --*/
305 
306 {
307  //
308  // We only do this action if supplied with a client communication port
309  //
310 
311  if ((Port->Flags & PORT_TYPE) == CLIENT_COMMUNICATION_PORT) {
312 
313  //
314  // We only do this action if the port has static security tracking,
315  // and we have a captured client token. The action is to simply
316  // delete the client token.
317  //
318 
319  if (!(Port->Flags & PORT_DYNAMIC_SECURITY)) {
320 
321  if ( Port->StaticSecurity.ClientToken ) {
322 
323  SeDeleteClientSecurity( &(Port)->StaticSecurity );
324  }
325  }
326  }
327 
328  //
329  // And return to our caller
330  //
331 
332  return;
333 }
