35 /*++
36 
37 Routine Description:
38 
39  This function initializes a context record so that it can be used in a
40  subsequent call to create thread.
41 
42 Arguments:
43 
44  Process - Supplies a handle to the process in which a thread is being
45  created.
46 
47  Context - Supplies a pointer to a context record.
48 
49  InitialPc - Supplies an initial program counter value.
50 
51  InitialSp - Supplies an initial stack pointer value.
52 
53 Return Value:
54 
55  STATUS_BAD_INITIAL_STACK is raised if initial stack pointer value is not
56  properly aligned.
57 
58 --*/
59 
60 {
61 
62  RTL_PAGED_CODE();
63 
64  //
65  // Check stack alignment.
66  //
67 
68  if (((ULONG64)InitialSp & 0xf) != 0) {
69  RtlRaiseStatus(STATUS_BAD_INITIAL_STACK);
70  }
71 
72  //
73  // Initialize the EFflags field.
74  //
75 
76  Context->EFlags = EFLAGS_IF_MASK;
77 
78  //
79  // Initialize the integer registers.
80  //
81 
82  Context->Rax = 0L;
83  Context->Rcx = 2L;
84  Context->Rbx = 1L;
85  Context->Rsp = (ULONG64)InitialSp;
86  Context->Rbp = 0L;
87  Context->Rsi = 4L;
88  Context->Rdi = 5L;
89  Context->R8 = 8;
90  Context->R9 = 9;
91  Context->R10 = 10;
92  Context->R11 = 11;
93  Context->R12 = 12;
94  Context->R13 = 13;
95  Context->R14 = 14;
96  Context->R15 = 15;
97 
98  //
99  // Initialize the floating point state.
100  //
101  // N.B. This includes both the legacy and extended floating state.
102  //
103 
104  RtlZeroMemory(&Context->FltSave, sizeof(XMM_SAVE_AREA32));
105  Context->MxCsr = INITIAL_MXCSR;
106  Context->FltSave.ControlWord = INITIAL_FPCSR;
107  Context->FltSave.MxCsr = INITIAL_MXCSR;
108 
109  //

111  //
112

114 
115  //
116  // Set context record flags.
117  //

121  //

123  //
124

130  //
131 
132  Context->R9 = 0xf0e0d0c0a0908070UI64;

137 RtlRemoteCall (

147 /*++
148 
149 Routine Description:
150 
151  This function calls a procedure in another thread/process using the
152  system functions get and set context. Parameters are passed to the
153  target procedure via the nonvolatile registers.
154 
155 Arguments:
156 
157  Process - Supplies an open handle to the target process.
158 
159  Thread - Supplies an open handle to the target thread within the target
160  process.
161 
162  CallSite - Supplies the address of the procedure to call in the target
163  process.
164 
165  ArgumentCount - Supplies the number of parameters to pass to the target
166  procedure.
167 
168  Arguments - Supplies a pointer to the array of parameters to pass.
169 
170  PassContext - Supplies a boolean value that determines whether a parameter
171  is to be passed that points to a context record.
172 
173  AlreadySuspended - Supplies a boolean value that determines whether the
174  target thread is already in a suspended or waiting state.
175 
176 Return Value:
177 
178  Status - Status value
179 
180 --*/
181 
182 {
183 
184  CONTEXT Context;
185  ULONG Index;
186  ULONG64 NewSp;
187  NTSTATUS Status;
188 
189  RTL_PAGED_CODE();
190 
191  //
192  // Check if too many arguments are specified.
193  //
194 
195  if (ArgumentCount > 4) {
196  return STATUS_INVALID_PARAMETER;
197  }
198 
199  //
200  // If necessary, suspend the target thread before getting the thread's
201  // current state.
202  //
203 
204  if (AlreadySuspended == FALSE) {
205  Status = NtSuspendThread(Thread, NULL);
206  if (!NT_SUCCESS(Status)) {
207  return Status;
208  }
209  }
210 
211  //
212  // Get the current context of the target thread.
213  //
214 
215  Context.ContextFlags = CONTEXT_FULL;
216  Status = NtGetContextThread(Thread, &Context);
217  if (!NT_SUCCESS(Status)) {
218  if (AlreadySuspended == FALSE) {
219  NtResumeThread(Thread, NULL);
220  }
221 
222  return Status;
223  }
224 
225  if (AlreadySuspended != FALSE) {
226  Context.Rax = STATUS_ALERTED;
227  }
228 
229  //
230  // Write previous thread context into the stack of the target thread.
231  //
232 
233  NewSp = Context.Rsp - sizeof(CONTEXT);
234  Status = NtWriteVirtualMemory(Process,
235  (PVOID)NewSp,
236  &Context,
237  sizeof(CONTEXT),
238  NULL);
239 
240  if (!NT_SUCCESS(Status)) {
241  if (AlreadySuspended == FALSE) {
242  NtResumeThread(Thread, NULL);
243  }
244 
245  return Status;
246  }
247 
248  //
249  // Pass the parameters to the target thread via the nonvolatile registers
250  // R11-R15.
251  //
252 
253  Context.Rsp = NewSp;
254  if (PassContext != FALSE) {
255  Context.R11 = NewSp;
256  for (Index = 0; Index < ArgumentCount; Index += 1) {
257  (&Context.R12)[Index] = Arguments[Index];
258  }
259 
260  } else {
261  for (Index = 0; Index < ArgumentCount; Index += 1) {
262  (&Context.R11)[Index] = Arguments[Index];
263  }
264  }
265 
266  //
267  // Set the address of the target code into RIP and set the thread context
268  // to cause the target procedure to be executed.
269  //
270 
271  Context.Rip = (ULONG64)CallSite;
272  Status = NtSetContextThread(Thread, &Context);
273  if (AlreadySuspended == FALSE) {
274  NtResumeThread(Thread, NULL);
275  }
276 
277  return Status;
278 }
