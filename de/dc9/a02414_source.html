254 ULONG
255 vDbgPrintExWithPrefixInternal (
256  __in PCH Prefix,
257  __in ULONG ComponentId,
258  __in ULONG Level,
259  __in PCHAR Format,
260  __in va_list arglist,
261  __in BOOLEAN ControlC
262  )
263 
264 /*++
265 
266 Routine Description:
267 
268  This routine provides a "printf" style capability for the kernel
269  debugger.
270 
271 Arguments:
272 
273  Prefix - Supplies a pointer to text that is to prefix the formatted
274  output.
275 
276  ComponentId - Supplies the Id of the calling component.
277 
278  Level - Supplies the output filter level or mask.
279 
280  Format - Supplies a pointer to a "printf" style format string.
281 
282  arglist - Supplies a pointer to a variable argument list.
283 
284  ControlC - Supplies a boolean that determines whether control C is
285  consumed or not.
286 
287 Return Value:
288 
289  Defined as returning a ULONG, actually returns status.
290 
291 --*/
292 
293 {
294 
295  UCHAR Buffer[512];
296  int cb;
297  int Length;
298  STRING Output;
299  NTSTATUS Status;
300 
301  //
302  // If the debug output will be suppressed, then return success
303  // immediately.
304  //
305 
306  if ((ComponentId != -1) &&
307  (NtQueryDebugFilterState(ComponentId, Level) != TRUE)) {
308 
309  return STATUS_SUCCESS;
310  }
311 
312  //
313  // Format the output into a buffer and then print it.
314  //
315 
316  try {
317  cb = strlen(Prefix);
318  if (cb > sizeof(Buffer)) {
319  cb = sizeof(Buffer);
320  }
321 
322  strncpy(Buffer, Prefix, cb);
323  Length = _vsnprintf(Buffer + cb , sizeof(Buffer) - cb, Format, arglist);
324 
325  } except (EXCEPTION_EXECUTE_HANDLER) {
326 
327  return GetExceptionCode();
328  }
329 
330  //
331  // Check if buffer overflow occurred during formatting. If buffer overflow
332  // occurred, then terminate the buffer with an end of line.
333  //
334 
335  if (Length == -1) {
336  Buffer[sizeof(Buffer) - 1] = '\n';
337  Length = sizeof(Buffer);
338 
339  } else {
340  Length += cb;
341  }
342 
343  Output.Buffer = Buffer;
344  Output.Length = (USHORT)Length;
345 
346  //
347  // If APP is being debugged, raise an exception and the debugger
348  // will catch and handle this. Otherwise, kernel debugger service
349  // is called.
350  //
351 
352  Status = DebugPrint(&Output, ComponentId, Level);
353  if ((ControlC == TRUE) &&
354  (Status == STATUS_BREAKPOINT)) {
355 
356  DbgBreakPointWithStatus(DBG_STATUS_CONTROL_C);
357  Status = STATUS_SUCCESS;
358  }
359 
360  return Status;
361 }

370 /*++
371 
372 Routine Description:
373 
374  This function displays the prompt string on the debug console and
375  then reads a line of text from the debug console. The line read is
376  returned in the memory pointed to by the second parameter. The third
377  parameter specifies the maximum number of characters that can be
378  stored in the response area.
379 
380 Arguments:

382  Prompt - Supplies a pointer to the text to display as the prompt.
383 
384  Response - Supplies a pointer to a buffer that receives the response
385  read from the debug console.
386 
387  Length - Supplies the maximum number of characters that can be stored in
388  the response buffer.
389 
390 Return Value:
391 
392  Number of characters stored in the Response buffer including the
393  terminating newline character, but not the ending null character.
394 
395 --*/
396 
397 {
398 
399  STRING Input;
400  STRING Output;
401 
402  //
403  // Output the prompt string and read input.
404  //
405 
406  Input.MaximumLength = (USHORT)Length;
407  Input.Buffer = Response;
408  Output.Length = (USHORT)strlen(Prompt);
409  Output.Buffer = Prompt;
