203 KeInsertQueueDpc (
204  __inout PRKDPC Dpc,
205  __in_opt PVOID SystemArgument1,
206  __in_opt PVOID SystemArgument2

208 
209 /*++
210 
211 Routine Description:
212 
213  This function inserts a DPC object into the DPC queue. If the DPC object
214  is already in the DPC queue, then no operation is performed. Otherwise,
215  the DPC object is inserted in the DPC queue and a dispatch interrupt is
216  requested.
217 
218 Arguments:
219 
220  Dpc - Supplies a pointer to a control object of type DPC.
221 
222  SystemArgument1, SystemArgument2 - Supply a set of two arguments that
223  contain untyped data provided by the executive.
224 
225 Return Value:
226 
227  If the DPC object is already in a DPC queue, then a value of FALSE is
228  returned. Otherwise a value of TRUE is returned.
229 
230 --*/
231 
232 {
233 
234  PKPRCB CurrentPrcb;
235  PKDPC_DATA DpcData;
236  BOOLEAN Inserted;
237 
238 #if !defined(NT_UP)
239 
240  ULONG_PTR Number;
241 
242 #endif
243 
244  KIRQL OldIrql;
245  BOOLEAN RequestInterrupt;
246  PKPRCB TargetPrcb;
247 
248  ASSERT_DPC(Dpc);
249 
250  //
251  // Disable interrupts and acquire the DPC queue lock for the specified
252  // target processor.
253  //
254  // N.B. Disable interrupt cannot be used here since it causes the
255  // software interrupt request code to get confused on some
256  // platforms.
257  //
258 
259  RequestInterrupt = FALSE;
260  KeRaiseIrql(HIGH_LEVEL, &OldIrql);
261  CurrentPrcb = KeGetCurrentPrcb();
262 
263 #if !defined(NT_UP)
264 
265  if (Dpc->Number >= MAXIMUM_PROCESSORS) {
266  Number = Dpc->Number - MAXIMUM_PROCESSORS;
267  TargetPrcb = KiProcessorBlock[Number];
268 
269  } else {
270  Number = CurrentPrcb->Number;
271  TargetPrcb = CurrentPrcb;
272  }
273 
274  DpcData = KiSelectDpcData(TargetPrcb, Dpc);
275  KiAcquireSpinLock(&DpcData->DpcLock);
276 
277 #else
278 
279  TargetPrcb = CurrentPrcb;
280  DpcData = KiSelectDpcData(TargetPrcb, Dpc);
281 
282 #endif
283 
284  //
285  // If the DPC object is not in a DPC queue, then store the system
286  // arguments, insert the DPC object in the DPC queue, increment the
287  // number of DPCs queued to the target processor, increment the DPC
288  // queue depth, set the address of the DPC target DPC spinlock, and
289  // request a dispatch interrupt if appropriate.
290  //
291 
292  Inserted = FALSE;
293  if (InterlockedCompareExchangePointer(&Dpc->DpcData,
294  DpcData,
295  NULL) == NULL) {
296 
297  DpcData->DpcQueueDepth += 1;
298  DpcData->DpcCount += 1;
299  Dpc->SystemArgument1 = SystemArgument1;
300  Dpc->SystemArgument2 = SystemArgument2;
301 
302  //
303  // If the DPC is of high importance, then insert the DPC at the
304  // head of the DPC queue. Otherwise, insert the DPC at the end
305  // of the DPC queue.
306  //
307 
308  Inserted = TRUE;
309  if (Dpc->Importance == HighImportance) {
310  InsertHeadList(&DpcData->DpcListHead, &Dpc->DpcListEntry);
311 
312  } else {
313  InsertTailList(&DpcData->DpcListHead, &Dpc->DpcListEntry);
314  }
315 
316  KeMemoryBarrier();
317 
318  //
319  // If the DPC is a normal DPC, then determine if a DPC interrupt
320  // should be request. Otherwise, check if the DPC thread should
321  // be activated.
322  //
323 
324  if (DpcData == &TargetPrcb->DpcData[DPC_THREADED]) {
325 
326  //
327  // If the DPC thread is not active on the target processor and
328  // a thread activation has not been requested, then request a
329  // dispatch interrupt on the target processor.
330  //
331 
332  if ((TargetPrcb->DpcThreadActive == FALSE) &&
333  (TargetPrcb->DpcThreadRequested == FALSE)) {
334 
335  InterlockedExchange(&TargetPrcb->DpcSetEventRequest, TRUE);
336  TargetPrcb->DpcThreadRequested = TRUE;
337  TargetPrcb->QuantumEnd = TRUE;
338 
339 #if defined(NT_UP)
340 
341  RequestInterrupt = TRUE;
342 
343 #else
344 
345  if (CurrentPrcb != TargetPrcb) {
346  if (((KiIdleSummary & AFFINITY_MASK(Number)) == 0) ||
347  (KeIsIdleHaltSet(TargetPrcb, Number) != FALSE)) {
348 
349  RequestInterrupt = TRUE;
350  }
351 
352  } else {
353  RequestInterrupt = TRUE;
354  }
355 
356 #endif
357 
358  }
359 
360  } else {
361 
362  //
363  // If a DPC routine is not active on the target processor and
364  // an interrupt has not been requested, then request a dispatch
365  // interrupt on the target processor if appropriate.
366  //
367 
368  if ((TargetPrcb->DpcRoutineActive == FALSE) &&
369  (TargetPrcb->DpcInterruptRequested == FALSE)) {
370 
371  //
372  // Request a dispatch interrupt on the current processor if
373  // the DPC is not of low importance, the length of the DPC
374  // queue has exceeded the maximum threshold, or if the DPC
375  // request rate is below the minimum threshold.
376  //
377 
378 #if defined(NT_UP)
379 
380  if ((Dpc->Importance != LowImportance) ||
381  (DpcData->DpcQueueDepth >= TargetPrcb->MaximumDpcQueueDepth) ||
382  (TargetPrcb->DpcRequestRate < TargetPrcb->MinimumDpcRate)) {
383 
384  TargetPrcb->DpcInterruptRequested = TRUE;
385  RequestInterrupt = TRUE;
386  }
387 
388 #endif
389 
390  //
391  // If the DPC is being queued to another processor and the
392  // DPC is of high importance, or the length of the other
393  // processor's DPC queue has exceeded the maximum threshold,
394  // then request a dispatch interrupt. Otherwise, request a
395  // dispatch interrupt on the current processor if the DPC is
396  // not of low importance, the length of the DPC queue has
397  // exceeded the maximum threshold, or if the DPC request rate
398  // is below the minimum threshold.
399  //
400 
401 #if !defined(NT_UP)
402 
403  if (CurrentPrcb != TargetPrcb) {
404  if (((Dpc->Importance == HighImportance) ||
405  (DpcData->DpcQueueDepth >= TargetPrcb->MaximumDpcQueueDepth))) {
406 
407  if (((KiIdleSummary & AFFINITY_MASK(Number)) == 0) ||
408  (KeIsIdleHaltSet(TargetPrcb, Number) != FALSE)) {
409 
410  TargetPrcb->DpcInterruptRequested = TRUE;
411  RequestInterrupt = TRUE;
412  }
413  }
414 
415  } else {
416  if ((Dpc->Importance != LowImportance) ||
417  (DpcData->DpcQueueDepth >= TargetPrcb->MaximumDpcQueueDepth) ||
418  (TargetPrcb->DpcRequestRate < TargetPrcb->MinimumDpcRate)) {
419 
420  TargetPrcb->DpcInterruptRequested = TRUE;
421  RequestInterrupt = TRUE;
422  }
423  }
424 
425 #endif
426 
427  }
428  }
429  }
430 
431  //
432  // Release the DPC lock, request a DPC interrupt if required, enable
433  // interrupts, and return whether the DPC was queued or not.
434  //
435 
436 #if !defined(NT_UP)
437 
438  KiReleaseSpinLock(&DpcData->DpcLock);
439 
440 #endif
441 
442  if (RequestInterrupt == TRUE) {
443 
444 #if defined(NT_UP)
445 
446  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
447 
448 #else
449 
450  if (TargetPrcb != CurrentPrcb) {
451  KiSendSoftwareInterrupt(AFFINITY_MASK(Number), DISPATCH_LEVEL);
452 
453  } else {
454  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
455  }
456 
457 #endif
458 
459  }
460 
461  KeLowerIrql(OldIrql);
462  return Inserted;
463 }

465 BOOLEAN
466 KeRemoveQueueDpc (
467  __inout PRKDPC Dpc
468  )
469 
470 /*++
471 
472 Routine Description:
473 
474  This function removes a DPC object from the DPC queue. If the DPC object
475  is not in the DPC queue, then no operation is performed. Otherwise, the
476  DPC object is removed from the DPC queue and its inserted state is set
477  FALSE.
478 
479 Arguments:
480 
481  Dpc - Supplies a pointer to a control object of type DPC.
482 
483 Return Value:
484 
485  If the DPC object is not in the DPC queue, then a value of FALSE is
486  returned. Otherwise a value of TRUE is returned.
487 
488 --*/
489 
490 {
491 
492  PKDPC_DATA DpcData;
493  BOOLEAN Enable;
494 
495  ASSERT_DPC(Dpc);
496 
497  //
498  // If the DPC object is in the DPC queue, then remove it from the queue
499  // and set its inserted state to FALSE.
500  //
501 
502  Enable = KeDisableInterrupts();
503  DpcData = Dpc->DpcData;
504  if (DpcData != NULL) {
505 
506  //
507  // Acquire the DPC lock of the target processor.
508  //
509 
510 #if !defined(NT_UP)
511 
512  KiAcquireSpinLock(&DpcData->DpcLock);
513 
514 #endif
515 
516  //
517  // If the specified DPC is still in the DPC queue, then remove
518  // it.
519  //
520  // N.B. It is possible for specified DPC to be removed from the
521  // specified DPC queue before the DPC lock is obtained.
522  //
523  //
524 
525  if (DpcData == Dpc->DpcData) {
526  DpcData->DpcQueueDepth -= 1;
527  RemoveEntryList(&Dpc->DpcListEntry);
528  Dpc->DpcData = NULL;
529  }
530 
531  //
532  // Release the DPC lock of the target processor.
533  //
534 
535 #if !defined(NT_UP)
536 
537  KiReleaseSpinLock(&DpcData->DpcLock);
538 
539 #endif
540 
541  }
542 
543  //
544  // Enable interrupts and return whether the DPC was removed from a DPC
545  // queue.
546  //
547 
548  KeEnableInterrupts(Enable);
549  return (DpcData != NULL ? TRUE : FALSE);
550 }

557 /*++
558 
559 Routine Description:
560 
561  This function causes all current DPCs on all processors to execute to
562  completion. This function is used at driver unload to make sure all
563  driver DPC processing has exited the driver image before the code and
564  data is deleted.
565 
566 Arguments:
567 
568  None.
569 
570 Return Value:
571 
572  None.
573 
574 --*/
575 
576 {
577 
578 #if !defined(NT_UP)
579 
580  PKTHREAD CurrentThread;
581  KPRIORITY OldPriority;
582  KAFFINITY ProcessorMask;
583  KAFFINITY SentDpcMask;
584  KAFFINITY CurrentProcessorMask;
585  BOOLEAN SetAffinity;
586  ULONG CurrentProcessor;
587  KIRQL OldIrql;
588 
589 #endif
590 
591  PKPRCB CurrentPrcb;
592 
593  PAGED_CODE ();
594 
595 #if defined(NT_UP)
596 
597  CurrentPrcb = KeGetCurrentPrcb();
598  if ((CurrentPrcb->DpcData[DPC_NORMAL].DpcQueueDepth > 0) ||
599  (CurrentPrcb->DpcData[DPC_THREADED].DpcQueueDepth > 0)) {
600  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
601  }
602 
603 #else
604 
605 #if DBG
606 
607  if (KeActiveProcessors == (KAFFINITY)1) {
608  CurrentPrcb = KeGetCurrentPrcb();
609  if ((CurrentPrcb->DpcData[DPC_NORMAL].DpcQueueDepth > 0) ||
610  (CurrentPrcb->DpcData[DPC_THREADED].DpcQueueDepth > 0)) {
611  KiRequestSoftwareInterrupt(DISPATCH_LEVEL);
612  }
613  return;
614  }
615 
616 #endif
617 
618  //
619  // Set the priority of this thread high so it will run quickly on the
620  // target processor.
621  //
622 
623  CurrentThread = KeGetCurrentThread();
624  OldPriority = KeSetPriorityThread(CurrentThread, HIGH_PRIORITY);
625  ProcessorMask = KeActiveProcessors;
626  SetAffinity = FALSE;
627  SentDpcMask = 0;
628 
629  //
630  // Clear the current processor from the affinity set and switch to the
631  // remaining processors in the affinity set so it can be guaranteed that
632  // respective DPCs have been completed processed.
633  //
634 
635  while (1) {
636  CurrentPrcb = KeGetCurrentPrcb();
637  CurrentProcessor = CurrentPrcb->Number;
638  CurrentProcessorMask = AFFINITY_MASK(CurrentProcessor);
639 
640  //
641  // Check to determine if there are DPCs that haven't been delivered yet.
642  // Low importance DPCs do not run right away and need to be forced to
643  // run now. This only needs to do this once per processor.
644  //
645 
646  if ((SentDpcMask & CurrentProcessorMask) == 0 &&
647  (CurrentPrcb->DpcData[DPC_NORMAL].DpcQueueDepth > 0) || (CurrentPrcb->DpcData[DPC_THREADED].DpcQueueDepth > 0)) {
648 
649  SentDpcMask |= CurrentProcessorMask;
650  KeRaiseIrql(DISPATCH_LEVEL, &OldIrql);
651  KiSendSoftwareInterrupt(CurrentProcessorMask, DISPATCH_LEVEL);
652  KeLowerIrql(OldIrql);
653 
654  //
655  // If processors have not been swapped, then the DPCs must have run
656  // to completion. If processors have been swapped, then just repeat
657  // for the current processor.
658  //
659 
660  if (KeGetCurrentPrcb() != CurrentPrcb) {
661  continue;
662  }
663  }
664 
665  ProcessorMask &= ~CurrentProcessorMask;
666  if (ProcessorMask == 0) {
667  break;
668  }
669 
670  KeSetSystemAffinityThread(ProcessorMask);
671  SetAffinity = TRUE;
672  }
673 
674  //
675  // Restore the affinity of the current thread if it was changed and
676  // restore the original thread priority.
677  //
678 
679  if (SetAffinity) {
680  KeRevertToUserAffinityThread ();
681  }
682 
683  KeSetPriorityThread(CurrentThread, OldPriority);
684 
685 #endif
686 
687  return;
688 }

690 VOID
691 KeGenericCallDpc (
692  __in PKDEFERRED_ROUTINE Routine,
693  __in_opt PVOID Context
694  )
695 
696 /*++
697 
698 Routine Description:
699 
700  This function acquires the DPC call lock, initializes a processor specific
701  DPC for each process with the specified deferred routine and context, and
702  queues the DPC for execution. When all DPCs routines have executed, the
703  DPC call lock is released.
704 
705 Arguments:
706 
707  Routine - Supplies the address of the deferred routine to be called.
708 
709  Context - Supplies the context that is passed to the deferred routine.
710 
711 Return Value:
712 
713  None.
714 
715 --*/
716 
717 {
718 
719  ULONG Barrier;
720 
721 #if !defined(NT_UP)
722 
723  PKDPC Dpc;
724  ULONG Index;
725  ULONG Limit;
726  ULONG Number;
727 
728 #endif
729 
730  KIRQL OldIrql;
731  DEFERRED_REVERSE_BARRIER ReverseBarrier;
732 
733  ASSERT(KeGetCurrentIrql () < DISPATCH_LEVEL);
734 
735  Barrier = KeNumberProcessors;
736  ReverseBarrier.Barrier = Barrier;
737  ReverseBarrier.TotalProcessors = Barrier;
738 
739 #if !defined(NT_UP)
740 
741  Index = 0;
742  Limit = Barrier;
743 
744 #endif
745 
746  //
747  // Switch to processor one to synchronize with other DPCs.
748  //
749 
750 #if !defined(NT_UP)
751 
752  KeSetSystemAffinityThread(1);
753 
754  //
755  // Acquire generic call DPC mutex.
756  //
757 
758  ExAcquireFastMutex(&KiGenericCallDpcMutex);
759 
760 #endif
761 
762  KeRaiseIrql(DISPATCH_LEVEL, &OldIrql);
763 
764  //
765  // Loop through all the target processors, initialize the deferred
766  // routine address, context parameter, barrier parameter, and queue
767  // the DPC to the target processor.
768  //
769 
770 #if !defined(NT_UP)
771 
772  Number = KeGetCurrentProcessorNumber();
773  do {
774 
775  //
776  // If the target processor is not the current processor, then
777  // initialize and queue the generic call DPC.
778  //
779 
780  if (Number != Index) {
781  Dpc = &KiProcessorBlock[Index]->CallDpc;
782  Dpc->DeferredRoutine = Routine;
783  Dpc->DeferredContext = Context;
784  KeInsertQueueDpc(Dpc, &Barrier, &ReverseBarrier);
785  }
786 
787  Index += 1;
788  } while (Index < Limit);
789 
790 #endif
791 
792  //
793  // Call deferred routine on current processor.
794  //
795 
796  (Routine)(&KeGetCurrentPrcb()->CallDpc, Context, &Barrier, &ReverseBarrier);
797 
798  //
799  // Wait for all target DPC routines to finish execution.
800  //
801 
802 #if !defined(NT_UP)
803 
804  while (*((ULONG volatile *)&Barrier) != 0) {
805  KeYieldProcessor();
806  }
807 
808 #endif
809 
810  //
811  // Release generic all DPC mutex and lower IRQL to previous level.
812  //
813 
814  KeLowerIrql(OldIrql);
815 
816 #if !defined(NT_UP)
817 
818  ExReleaseFastMutex(&KiGenericCallDpcMutex);
819  KeRevertToUserAffinityThread();
820 
821 #endif
822  return;
823 }

861 /*++

863 Routine Description:

884
