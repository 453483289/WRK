55 /*++
56 
57 Routine Description:
58 
59  This routine probes and captures a copy of the security descriptor based
60  upon the following tests.
61 
62  if the requestor mode is not kernel mode then
63 
64  probe and capture the input descriptor
65  (the captured descriptor is self-relative)
66 
67  if the requestor mode is kernel mode then
68 
69  if force capture is true then
70 
71  do not probe the input descriptor, but do capture it.
72  (the captured descriptor is self-relative)
73 
74  else
75 
76  do nothing
77  (the input descriptor is expected to be self-relative)
78 
79 Arguments:
80 
81  InputSecurityDescriptor - Supplies the security descriptor to capture.
82  This parameter is assumed to have been provided by the mode specified
83  in RequestorMode.
84 
85  RequestorMode - Specifies the caller's access mode.
86 
87  PoolType - Specifies which pool type to allocate the captured
88  descriptor from
89 
90  ForceCapture - Specifies whether the input descriptor should always be
91  captured
92 
93  OutputSecurityDescriptor - Supplies the address of a pointer to the
94  output security descriptor. The captured descriptor will be
95  self-relative format.
96 
97 Return Value:
98 
99  STATUS_SUCCESS if the operation is successful.
100 
101  STATUS_INVALID_SID - An SID within the security descriptor is not
102  a valid SID.
103 
104  STATUS_INVALID_ACL - An ACL within the security descriptor is not
105  a valid ACL.
106 
107  STATUS_UNKNOWN_REVISION - The revision level of the security descriptor
108  is not one known to this revision of the capture routine.
109 --*/

171  // first.

178  //

219  //
220  // Capture the SecurityDescriptor main record.
221  // It doesn't need probing because requestor mode is kernel.
222  //
223

225  InputSecurityDescriptor,
226  sizeof(SECURITY_DESCRIPTOR_RELATIVE) );

246  //

251  (SECURITY_DESCRIPTOR *)InputSecurityDescriptor
252  );

254  (SECURITY_DESCRIPTOR *)InputSecurityDescriptor

388 
389  } else {

393 
394  }
395

407 
408  try {
409  GroupSubAuthorityCount =
410  ProbeAndReadUchar( &(((SID *)(Captured.Group))->SubAuthorityCount) );

423 
424  }
425

475 
476 
477  try {
