120 /*++
121 
122 Routine Description:
123 
124  This function examines the working set list for the specified
125  working set and locates an entry to contain a new page.
126 
127  If the memory is not tight, the new page is added without removing a page.
128 
129  If memory is tight (or this working set is at its limit), a page
130  is removed from the working set and the new page added in its place.
131 
132 Arguments:
133 
134  WsInfo - Supplies the working set list.
135 
136  PointerPte - Supplies the PTE of the virtual address to insert.
137 
138  Pfn1 - Supplies the PFN entry of the virtual address being inserted. If
139  this pointer has the low bit set, no trimming can be done at this
140  time (because it is a WSLE hash table page insertion). Strip the
141  low bit for these cases.
142 
143  WsleMask - Supplies the mask to logical OR into the new working set entry.
144 
145 Return Value:
146 
147  Returns the working set index which was used to insert the specified entry,
148  or 0 if no index was available.
149 
150 Environment:
151 
152  Kernel mode, APCs disabled, working set lock. PFN lock NOT held.
153 
154 --*/
155 
156 {
157  PVOID VirtualAddress;
158  PMMWSLE Wsle;
159  PMMWSL WorkingSetList;
160  WSLE_NUMBER WorkingSetIndex;
161 
162  WorkingSetList = WsInfo->VmWorkingSetList;
163  Wsle = WorkingSetList->Wsle;
164 
165  //
166  // Update page fault counts.
167  //
168 
169  WsInfo->PageFaultCount += 1;
170  InterlockedIncrement (&KeGetCurrentPrcb ()->MmPageFaultCount);
171 
172  //
173  // Determine if a page should be removed from the working set to make
174  // room for the new page. If so, remove it.
175  //
176 
177  if ((ULONG_PTR)Pfn1 & 0x1) {
178  MiDoReplacement (WsInfo, WsleAllocationDontTrim);
179  if (WorkingSetList->FirstFree == WSLE_NULL_INDEX) {
180  return 0;
181  }
182  Pfn1 = (PMMPFN)((ULONG_PTR)Pfn1 & ~0x1);
183  }
184  else {
185  MiDoReplacement (WsInfo, WsleAllocationAny);
186 
187  if (WorkingSetList->FirstFree == WSLE_NULL_INDEX) {
188 
189  //
190  // Add more pages to the working set list structure.
191  //
192 
193  if (MiAddWorkingSetPage (WsInfo) == FALSE) {
194 
195  //
196  // No page was added to the working set list structure.
197  // We must replace a page within this working set.
198  //
199 
200  MiDoReplacement (WsInfo, WsleAllocationReplace);
201 
202  if (WorkingSetList->FirstFree == WSLE_NULL_INDEX) {
203  MiWsleFailures += 1;
204  return 0;
205  }
206  }
207  }
208  }
209 
210  //
211  // Get the working set entry from the free list.
212  //
213 
214  ASSERT (WorkingSetList->FirstFree <= WorkingSetList->LastInitializedWsle);
215 
216  ASSERT (WorkingSetList->FirstFree >= WorkingSetList->FirstDynamic);
217 
218  WorkingSetIndex = WorkingSetList->FirstFree;
219  WorkingSetList->FirstFree = (WSLE_NUMBER)(Wsle[WorkingSetIndex].u1.Long >> MM_FREE_WSLE_SHIFT);
220 
221  ASSERT ((WorkingSetList->FirstFree <= WorkingSetList->LastInitializedWsle) ||
222  (WorkingSetList->FirstFree == WSLE_NULL_INDEX));
223 
224  ASSERT (WsInfo->WorkingSetSize <= (WorkingSetList->LastInitializedWsle + 1));
225  WsInfo->WorkingSetSize += 1;
226 
227  if (WsInfo->WorkingSetSize > WsInfo->MinimumWorkingSetSize) {
228  InterlockedIncrementSizeT (&MmPagesAboveWsMinimum);
229  }
230 
231  if (WsInfo->WorkingSetSize > WsInfo->PeakWorkingSetSize) {
232  WsInfo->PeakWorkingSetSize = WsInfo->WorkingSetSize;
233  }
234 
235  if (WsInfo == &MmSystemCacheWs) {
236  if (WsInfo->WorkingSetSize + MmTransitionSharedPages > MmTransitionSharedPagesPeak) {
237  MmTransitionSharedPagesPeak = WsInfo->WorkingSetSize + MmTransitionSharedPages;
238  }
239  }
240 
241  if (WorkingSetIndex > WorkingSetList->LastEntry) {
242  WorkingSetList->LastEntry = WorkingSetIndex;
243  }
244 
245  //
246  // The returned entry is guaranteed to be available at this point.
247  //
248 
249  ASSERT (Wsle[WorkingSetIndex].u1.e1.Valid == 0);
250 
251  VirtualAddress = MiGetVirtualAddressMappedByPte (PointerPte);
252 
253  MiUpdateWsle (&WorkingSetIndex, VirtualAddress, WsInfo, Pfn1, WsleMask);
254 
255  if (WsleMask != 0) {
256  Wsle[WorkingSetIndex].u1.Long |= WsleMask;
257  }
258 
259 #if DBG
260  if (MI_IS_SYSTEM_CACHE_ADDRESS (VirtualAddress)) {
261  ASSERT (MmSystemCacheWsle[WorkingSetIndex].u1.e1.Protection == MM_ZERO_ACCESS);
262  }
263 #endif
264 
265  MI_SET_PTE_IN_WORKING_SET (PointerPte, WorkingSetIndex);
266 
267  return WorkingSetIndex;
268 }

276 /*++
277 
278 Routine Description:
279 
280  This function determines whether the working set should be
281  grown or if a page should be replaced. Replacement is
282  done here if deemed necessary.
283 
284 Arguments:
285 
286  WsInfo - Supplies the working set information structure to replace within.
287 
288  Flags - Supplies 0 if replacement is not required.
289  Supplies 1 if replacement is desired.
290  Supplies 2 if working set trimming must not be done here - this
291  is only used for inserting new WSLE hash table pages as a trim
292  would not know how to interpret them.
293 
294 Return Value:
295 
296  None.
297 
298 Environment:
299 
300  Kernel mode, APCs disabled, working set lock. PFN lock NOT held.
301 
302 --*/
303 
304 {
305  KIRQL OldIrql;
306  WSLE_NUMBER PagesTrimmed;
307  ULONG MemoryMaker;
308  PMMWSL WorkingSetList;
309  WSLE_NUMBER CurrentSize;
310  LARGE_INTEGER CurrentTime;
311  PFN_NUMBER Dummy1;
312  PFN_NUMBER Dummy2;
313  WSLE_NUMBER Trim;
314  ULONG TrimAge;
315  ULONG GrowthSinceLastEstimate;
316  WSLE_ALLOCATION_TYPE TrimFlags;
317 
318  TrimFlags = Flags;
319  Flags &= ~WsleAllocationDontTrim;
320 
321  WorkingSetList = WsInfo->VmWorkingSetList;
322  GrowthSinceLastEstimate = 1;
323 
324  PagesTrimmed = 0;
325 
326  //
327  // Determine whether the working set can be grown or whether a
328  // page needs to be replaced.
329  //
330 
331 recheck:
332 
333  if (WsInfo->WorkingSetSize >= WsInfo->MinimumWorkingSetSize) {
334 
335  if ((WsInfo->Flags.ForceTrim == 1) && (TrimFlags != WsleAllocationDontTrim)) {
336 
337  //
338  // The working set manager cannot attach to this process
339  // to trim it. Force a trim now and update the working
340  // set manager's fields properly to indicate a trim occurred.
341  //
342 
343  Trim = WsInfo->Claim >>
344  ((WsInfo->Flags.MemoryPriority == MEMORY_PRIORITY_FOREGROUND)
345  ? MI_FOREGROUND_CLAIM_AVAILABLE_SHIFT
346  : MI_BACKGROUND_CLAIM_AVAILABLE_SHIFT);
347 
348  if (MmAvailablePages < MM_HIGH_LIMIT + 64) {
349  if (WsInfo->WorkingSetSize > WsInfo->MinimumWorkingSetSize) {
350  Trim = (WsInfo->WorkingSetSize - WsInfo->MinimumWorkingSetSize) >> 2;
351  }
352  TrimAge = MI_PASS4_TRIM_AGE;
353  }
354  else {
355  TrimAge = MI_PASS0_TRIM_AGE;
356  }
357 
358  PagesTrimmed += MiTrimWorkingSet (Trim, WsInfo, TrimAge);
359 
360  MiAgeWorkingSet (WsInfo,
361  TRUE,
362  NULL,
363  &Dummy1,
364  &Dummy2);
365 
366  KeQuerySystemTime (&CurrentTime);
367 
368  LOCK_EXPANSION (OldIrql);
369  WsInfo->LastTrimTime = CurrentTime;
370  WsInfo->Flags.ForceTrim = 0;
371  UNLOCK_EXPANSION (OldIrql);
372 
373  goto recheck;
374  }
375 
376  CurrentSize = WsInfo->WorkingSetSize;
377  ASSERT (CurrentSize <= (WorkingSetList->LastInitializedWsle + 1));
378 
379  if ((WsInfo->Flags.MaximumWorkingSetHard) &&
380  (CurrentSize >= WsInfo->MaximumWorkingSetSize)) {
381 
382  //
383  // This is an enforced working set maximum triggering a replace.
384  //
385 
386  MiReplaceWorkingSetEntry (WsInfo, Flags);
387 
388  return;
389  }
390 
391  //
392  // Don't grow if :
393  // - there aren't any pages to take
394  // - or if we are growing too much in this time interval
395  // and there isn't much memory available
396  //
397 
398  MemoryMaker = PsGetCurrentThread()->MemoryMaker;
399 
400  if ((Flags == WsleAllocationReplace) ||
401  ((MmAvailablePages < MM_VERY_HIGH_LIMIT) &&
402  ((MmAvailablePages == 0) ||
403  (MI_WS_GROWING_TOO_FAST(WsInfo)) && (MemoryMaker == 0)))) {
404 
405  //
406  // Can't grow this one.
407  //
408 
409  MiReplacing = TRUE;
410 
411  if ((Flags == WsleAllocationReplace) || (MemoryMaker == 0)) {
412 
413  MiReplaceWorkingSetEntry (WsInfo, Flags);
414 
415  //
416  // Set the must trim flag because this could be a realtime
417  // thread where the fault straddles a page boundary. If
418  // it's realtime, the balance set manager will never get to
419  // run and the thread will endlessly replace one WSL entry
420  // with the other half of the straddler. Setting this flag
421  // guarantees the next fault will guarantee a forced trim
422  // and allow a reasonable available page threshold trim
423  // calculation since GrowthSinceLastEstimate will be
424  // cleared.
425  //
426 
427  LOCK_EXPANSION (OldIrql);
428  WsInfo->Flags.ForceTrim = 1;
429  UNLOCK_EXPANSION (OldIrql);
430 
431  GrowthSinceLastEstimate = 0;
432  }
433  else {
434 
435  //
436  // If we've only trimmed a single page, then don't force
437  // replacement on the next fault. This prevents a single
438  // instruction causing alternating faults on the referenced
439  // code & data in a (realtime) thread from looping endlessly.
440  //
441 
442  if (PagesTrimmed > 1) {
443 
444  LOCK_EXPANSION (OldIrql);
445  WsInfo->Flags.ForceTrim = 1;
446  UNLOCK_EXPANSION (OldIrql);
447  }
448  }
449  }
450  }
451 
452  //
453  // If there isn't enough memory to allow growth, find a good page
454  // to remove and remove it.
455  //
456 
457  WsInfo->GrowthSinceLastEstimate += GrowthSinceLastEstimate;
458 
459  return;
460 }

469 /*++
470 
471 Routine Description:
472 
473  This function enables/disables hard enforcement of the working set minimums
474  and maximums for the specified WsInfo.
475 
476 Arguments:
477 
478  Process - Supplies the target process.
479 
480  Flags - Supplies new flags (MM_WORKING_SET_MAX_HARD_ENABLE, etc).
481 
482 Return Value:
483 
484  NTSTATUS.
485 
486 Environment:
487 
488  Kernel mode, APCs disabled. The working set mutex must NOT be held.
489  The caller guarantees that the target WsInfo cannot go away.
490 
491 --*/
492 
493 {
494  KIRQL OldIrql;
495  PMMSUPPORT WsInfo;
496  PETHREAD Thread;
497  MMSUPPORT_FLAGS PreviousBits;
498  MMSUPPORT_FLAGS TempBits = {0};
499  LOGICAL Attached;
500  NTSTATUS Status;
501  KAPC_STATE ApcState;
502 
503  WsInfo = &Process->Vm;
504  Status = STATUS_SUCCESS;
505 
506  if (Flags & MM_WORKING_SET_MIN_HARD_ENABLE) {
507  Flags &= ~MM_WORKING_SET_MIN_HARD_DISABLE;
508  TempBits.MinimumWorkingSetHard = 1;
509  }
510 
511  if (Flags & MM_WORKING_SET_MAX_HARD_ENABLE) {
512  Flags &= ~MM_WORKING_SET_MAX_HARD_DISABLE;
513  TempBits.MaximumWorkingSetHard = 1;
514  }
515 
516  Attached = FALSE;
517 
518  if (PsGetCurrentProcess() != Process) {
519  KeStackAttachProcess (&Process->Pcb, &ApcState);
520  Attached = TRUE;
521  }
522 
523  Thread = PsGetCurrentThread ();
524 
525  LOCK_WS (Thread, Process);
526 
527  LOCK_EXPANSION (OldIrql);
528 
529  if (Flags & MM_WORKING_SET_MIN_HARD_DISABLE) {
530  WsInfo->Flags.MinimumWorkingSetHard = 0;
531  }
532 
533  if (Flags & MM_WORKING_SET_MAX_HARD_DISABLE) {
534  WsInfo->Flags.MaximumWorkingSetHard = 0;
535  }
536 
537  PreviousBits = WsInfo->Flags;
538 
539  //
540  // If the caller's request will result in hard enforcement of both limits
541  // is enabled, then check whether the current minimum and maximum working
542  // set values will guarantee forward progress even in pathological
543  // scenarios.
544  //
545 
546  if (PreviousBits.MinimumWorkingSetHard == 1) {
547  TempBits.MinimumWorkingSetHard = 1;
548  }
549 
550  if (PreviousBits.MaximumWorkingSetHard == 1) {
551  TempBits.MaximumWorkingSetHard = 1;
552  }
553 
554  if ((TempBits.MinimumWorkingSetHard == 1) &&
555  (TempBits.MaximumWorkingSetHard == 1)) {
556 
557  //
558  // Net result is hard enforcement on both limits so check that the
559  // two limits cannot result in lack of forward progress.
560  //
561 
562  if (WsInfo->MinimumWorkingSetSize + MM_FLUID_WORKING_SET >= WsInfo->MaximumWorkingSetSize) {
563  Status = STATUS_BAD_WORKING_SET_LIMIT;
564  goto bail;
565  }
566  }
567 
568  if (Flags & MM_WORKING_SET_MIN_HARD_ENABLE) {
569  WsInfo->Flags.MinimumWorkingSetHard = 1;
570  }
571 
572  if (Flags & MM_WORKING_SET_MAX_HARD_ENABLE) {
573  WsInfo->Flags.MaximumWorkingSetHard = 1;
574  }
575 
576 bail:
577 
578  UNLOCK_EXPANSION (OldIrql);
579 
580  UNLOCK_WS (Thread, Process);
581 
582  if (Attached) {
583  KeUnstackDetachProcess (&ApcState);
584  }
585 
586  return Status;
587 }

596 /*++
597 
598 Routine Description:
599 
600  This function tries to find a good working set entry to replace.
601 
602 Arguments:
603 
604  WsInfo - Supplies the working set info pointer.
605 
606  Flags - Supplies 0 if replacement is not required.
607 
608  Supplies 1 if replacement is desired. Note replacement cannot
609  be guaranteed (the entire existing working set may
610  be locked down) - if no entry can be released the caller
611  can detect this because MMWSL->FirstFree will not contain
612  any free entries - and so the caller should release the
613  working set mutex and retry the operation.
614 
615 Return Value:
616 
617  None.
618 
619 Environment:
620 
621  Kernel mode, APCs disabled, working set lock. PFN lock NOT held.
622 
623 --*/
624 
625 {
626  WSLE_NUMBER WorkingSetIndex;
627  WSLE_NUMBER FirstDynamic;
628  WSLE_NUMBER LastEntry;
629  PMMWSL WorkingSetList;
630  PMMWSLE Wsle;
631  ULONG NumberOfCandidates;
632  PMMPTE PointerPte;
633  WSLE_NUMBER TheNextSlot;
634  WSLE_NUMBER OldestWorkingSetIndex;
635  LONG OldestAge;
636 
637  WorkingSetList = WsInfo->VmWorkingSetList;
638  Wsle = WorkingSetList->Wsle;
639 
640  //
641  // Toss a page out of the working set.
642  //
643 
644  LastEntry = WorkingSetList->LastEntry;
645  FirstDynamic = WorkingSetList->FirstDynamic;
646  WorkingSetIndex = WorkingSetList->NextSlot;
647  if (WorkingSetIndex > LastEntry || WorkingSetIndex < FirstDynamic) {
648  WorkingSetIndex = FirstDynamic;
649  }
650  TheNextSlot = WorkingSetIndex;
651  NumberOfCandidates = 0;
652 
653  OldestWorkingSetIndex = WSLE_NULL_INDEX;
654  OldestAge = -1;
655 
656  while (TRUE) {
657 
658  //
659  // Keep track of the oldest page along the way in case we
660  // don't find one that's >= MI_IMMEDIATE_REPLACEMENT_AGE
661  // before we've looked at MM_WORKING_SET_LIST_SEARCH
662  // entries.
663  //
664 
665  while (Wsle[WorkingSetIndex].u1.e1.Valid == 0) {
666  WorkingSetIndex += 1;
667  if (WorkingSetIndex > LastEntry) {
668  WorkingSetIndex = FirstDynamic;
669  }
670 
671  if (WorkingSetIndex == TheNextSlot) {
672 
673  if (Flags == WsleAllocationAny) {
674 
675  //
676  // Entire working set list has been searched, increase
677  // the working set size. Note this may result in exceeding
678  // a hard maximum limit on the system cache (only)
679  // in the case where all the other entries (paged pool)
680  // have been locked (so there isn't one to replace).
681  //
682 
683  WsInfo->GrowthSinceLastEstimate += 1;
684  }
685  return;
686  }
687  }
688 
689  if (OldestWorkingSetIndex == WSLE_NULL_INDEX) {
690 
691  //
692  // First time through, so initialize the OldestWorkingSetIndex
693  // to the first valid WSLE. As we go along, this will be re-pointed
694  // at the oldest candidate we come across.
695  //
696 
697  OldestWorkingSetIndex = WorkingSetIndex;
698  OldestAge = -1;
699  }
700 
701  PointerPte = MiGetPteAddress(Wsle[WorkingSetIndex].u1.VirtualAddress);
702 
703  if ((Flags == WsleAllocationReplace) ||
704  ((MI_GET_ACCESSED_IN_PTE(PointerPte) == 0) &&
705  (OldestAge < (LONG) MI_GET_WSLE_AGE(PointerPte, &Wsle[WorkingSetIndex])))) {
706 
707  //
708  // This one is not used and it's older.
709  //
710 
711  OldestAge = MI_GET_WSLE_AGE(PointerPte, &Wsle[WorkingSetIndex]);
712  OldestWorkingSetIndex = WorkingSetIndex;
713  }
714 
715  //
716  // If it's old enough or we've searched too much then use this entry.
717  //
718 
719  if ((Flags == WsleAllocationReplace) ||
720  OldestAge >= MI_IMMEDIATE_REPLACEMENT_AGE ||
721  NumberOfCandidates > MM_WORKING_SET_LIST_SEARCH) {
722 
723  if (OldestWorkingSetIndex != WorkingSetIndex) {
724  WorkingSetIndex = OldestWorkingSetIndex;
725  PointerPte = MiGetPteAddress(Wsle[WorkingSetIndex].u1.VirtualAddress);
726  }
727 
728  if (MiFreeWsle (WorkingSetIndex, WsInfo, PointerPte)) {
729 
730  //
731  // This entry was removed.
732  //
733 
734  WorkingSetList->NextSlot = WorkingSetIndex + 1;
735  break;
736  }
737 
738  //
739  // We failed to remove a page, try the next one.
740  //
741  // Clear the OldestWorkingSetIndex so that
742  // it gets set to the next valid entry above like the
743  // first time around.
744  //
745 
746  WorkingSetIndex = OldestWorkingSetIndex + 1;
747 
748  OldestWorkingSetIndex = WSLE_NULL_INDEX;
749  }
750  else {
751  WorkingSetIndex += 1;
752  }
753 
754  if (WorkingSetIndex > LastEntry) {
755  WorkingSetIndex = FirstDynamic;
756  }
757 
758  NumberOfCandidates += 1;
759 
760 
761  if (WorkingSetIndex == TheNextSlot) {
762 
763  if (Flags == WsleAllocationAny) {
764 
765  //
766  // Entire working set list has been searched, increase
767  // the working set size. Note this may result in exceeding
768  // a hard maximum limit on the system cache (only)
769  // in the case where all the other entries (paged pool)
770  // have been locked (so there isn't one to replace).
771 
772  WsInfo->GrowthSinceLastEstimate += 1;
773  }
774  break;
775  }
776  }
777  return;
778 }

1069  // PFN as 0 and therefore want to set the direct bit.
1070  //

1113  0);

1166  //

1170

1172

1174  ASSERT (WorkingSetIndex >= WorkingSetList->FirstDynamic);

1178  Temp = Wsle[WorkingSetIndex];
1179  MI_LOG_WSLE_CHANGE (WorkingSetList, WorkingSetIndex, Wsle[Index]);
1180  Wsle[WorkingSetIndex] = Wsle[Index];

1183  WorkingSetIndex = Index;
1184  ASSERT (((Wsle[WorkingSetList->FirstFree].u1.Long >> MM_FREE_WSLE_SHIFT)
1185  <= WorkingSetList->LastInitializedWsle) ||
1186  ((Wsle[WorkingSetList->FirstFree].u1.Long >> MM_FREE_WSLE_SHIFT)
1187  == WSLE_NULL_INDEX));
1188  }
1189  else if (Wsle[Index - 1].u1.e1.Valid == 0) {
1190  if ((Wsle[Index - 1].u1.Long >> MM_FREE_WSLE_SHIFT) == Index) {
1191  FreeIndex = Index - 1;
1192  }
1193  }
1194  else if (Wsle[Index + 1].u1.e1.Valid == 0) {

1197  }
1198  }

1202  // Link the WSLE into the free list.
1203  //

1218  }
