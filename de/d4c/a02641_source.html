80 RtlCaptureStackBackTrace(
81  IN ULONG FramesToSkip,
82  IN ULONG FramesToCapture,
83  OUT PVOID *BackTrace,
84  OUT PULONG BackTraceHash
85  )
86 /*++
87 
88 Routine Description:
89 
90  This routine walks up the stack frames, capturing the return address from
91  each frame requested.
92 
93 Arguments:
94 
95  FramesToSkip - frames detected but not included in the stack trace
96 
97  FramesToCapture - frames to be captured in the stack trace buffer.
98  One of the frames will be for RtlCaptureStackBackTrace.
99 
100  BackTrace - stack trace buffer
101 
102  BackTraceHash - very simple hash value that can be used to organize
103  hash tables. It is just an arithmetic sum of the pointers in the
104  stack trace buffer. If NULL then no hash value is computed.
105 
106 Return Value:
107 
108  Number of return addresses returned in the stack trace buffer.
109 
110 --*/
111 {
112  PVOID Trace [2 * MAX_STACK_DEPTH];
113  ULONG FramesFound;
114  ULONG HashValue;
115  ULONG Index;
116 
117  //
118  // One more frame to skip for the "capture" function (RtlWalkFrameChain).
119  //
120 
121  FramesToSkip += 1;
122 
123  //
124  // Sanity checks.
125  //
126 
127  if (FramesToCapture + FramesToSkip >= 2 * MAX_STACK_DEPTH) {
128  return 0;
129  }
130 
131  FramesFound = RtlWalkFrameChain (Trace,
132  FramesToCapture + FramesToSkip,
133  0);
134 
135  if (FramesFound <= FramesToSkip) {
136  return 0;
137  }
138 
139  Index = 0;
140 
141  for (HashValue = 0; Index < FramesToCapture; Index++) {
142 
143  if (FramesToSkip + Index >= FramesFound) {
144  break;
145  }
146 
147  BackTrace[Index] = Trace[FramesToSkip + Index];
148  HashValue += PtrToUlong(BackTrace[Index]);
149  }
150 
151  if (BackTraceHash != NULL) {
152 
153  *BackTraceHash = HashValue;
154  }
155 
156  //
157  // Zero the temporary buffer used to get the stack trace
158  // so that we do not leave garbage on the stack. This will
159  // improve debugging when we need to look manually at a stack.
160  //
161  // N.B. We cannot use here a simple call to RtlZeroMemory because the
162  // compiler will optimize away that call since it is performed
163  // for a buffer that gets out of scope.
164  //
165 
166  {
167  volatile PVOID * Pointer = (volatile PVOID *)Trace;
168  SIZE_T Count = sizeof(Trace) / sizeof (PVOID);
169 
170  while (Count > 0) {
171 
172  *Pointer = NULL;
173 
174  Pointer += 1;
175  Count -= 1;
176  }
177  }
178 
179  return (USHORT)Index;
180 }

198 RtlWalkFrameChain (
199  OUT PVOID *Callers,
200  IN ULONG Count,
201  IN ULONG Flags
202  )
203 
204 /*++
205 
206 Routine Description:
207 
208  This function tries to walk the EBP chain and fill out a vector of
209  return addresses. It is possible that the function cannot fill the
210  requested number of callers. In this case the function will just return
211  with a smaller stack trace. In kernel mode the function should not take
212  any exceptions (page faults) because it can be called at all sorts of
213  irql levels.
214 
215  The `Flags' parameter is used for future extensions. A zero value will be
216  compatible with new stack walking algorithms.
217 
218  A value of 1 for `Flags' means we are running in K-mode and we want to get
219  the user mode stack trace.
220 
221 Return value:
222 
223  The number of identified return addresses on the stack. This can be less
224  then the Count requested.
225 
226 --*/
227 
228 {
229 
230  ULONG_PTR Fp, NewFp, ReturnAddress;
231  ULONG Index;
232  ULONG_PTR StackEnd, StackStart;
233  BOOLEAN Result;
234  BOOLEAN InvalidFpValue;
235 
236  //
237  // Get the current EBP pointer which is supposed to
238  // be the start of the EBP chain.
239  //
240 
241  _asm mov Fp, EBP;
242 
243  StackStart = Fp;
244  InvalidFpValue = FALSE;
245 
246  if (Flags == 0) {
247  if (! RtlpCaptureStackLimits (Fp, &StackStart, &StackEnd)) {
248  return 0;
249  }
250  }
251 
252 
253  try {
254 
255  //
256  // If we need to get the user mode stack trace from kernel mode
257  // figure out the proper limits.
258  //
259 
260  if (Flags == 1) {
261 
262  PKTHREAD Thread = KeGetCurrentThread ();
263  PTEB Teb;
264  PKTRAP_FRAME TrapFrame;
265  ULONG_PTR Esp;
266 
267  TrapFrame = Thread->TrapFrame;
268  Teb = Thread->Teb;
269 
270  //
271  // If this is a system thread, it has no Teb and no kernel mode
272  // stack, so check for it so we don't dereference NULL.
273  //
274  // If there is no trap frame (probably an APC), or it's attached,
275  // or the irql is greater than dispatch, this code can't log a
276  // stack.
277  //
278 
279  if (Teb == NULL ||
280  IS_SYSTEM_ADDRESS((PVOID)TrapFrame) == FALSE ||
281  (PVOID)TrapFrame <= Thread->StackLimit ||
282  (PVOID)TrapFrame >= Thread->StackBase ||
283  KeIsAttachedProcess() ||
284  (KeGetCurrentIrql() >= DISPATCH_LEVEL)) {
285 
286  return 0;
287  }
288 
289  StackStart = (ULONG_PTR)(Teb->NtTib.StackLimit);
290  StackEnd = (ULONG_PTR)(Teb->NtTib.StackBase);
291  Fp = (ULONG_PTR)(TrapFrame->Ebp);
292 
293  if (StackEnd <= StackStart) {
294  return 0;
295  }
296 
297  ProbeForRead (StackStart, StackEnd - StackStart, sizeof (UCHAR));
298  }
299 
300  for (Index = 0; Index < Count; Index += 1) {
301 
302  if (Fp >= StackEnd ||
303  ( (Index == 0)?
304  (Fp < StackStart):
305  (Fp <= StackStart) ) ||
306  StackEnd - Fp < sizeof(ULONG_PTR) * 2) {
307  break;
308  }
309 
310  NewFp = *((PULONG_PTR)(Fp + 0));
311  ReturnAddress = *((PULONG_PTR)(Fp + sizeof(ULONG_PTR)));
312 
313  //
314  // Figure out if the new frame pointer is ok. This validation
315  // should avoid all exceptions in kernel mode because we always
316  // read within the current thread's stack and the stack is
317  // guaranteed to be in memory (no page faults). It is also guaranteed
318  // that we do not take random exceptions in user mode because we always
319  // keep the frame pointer within stack limits.
320  //
321 
322  if (! (Fp < NewFp && NewFp < StackEnd)) {
323 
324  InvalidFpValue = TRUE;
325  }
326 
327  //
328  // Figure out if the return address is ok. If return address
329  // is a stack address or <64k then something is wrong. There is
330  // no reason to return garbage to the caller therefore we stop.
331  //
332 
333  if (StackStart < ReturnAddress && ReturnAddress < StackEnd) {
334  break;
335  }
336 
337  if (Flags == 0 && IS_SYSTEM_ADDRESS((PVOID)ReturnAddress) == FALSE) {
338  break;
339  }
340 
341  //
342  // Store new fp and return address and move on.
343  // If the new FP value is bogus but the return address
344  // looks ok then we still save the address.
345  //
346 
347  Callers[Index] = (PVOID)ReturnAddress;
348 
349  if (InvalidFpValue) {
350 
351  Index += 1;
352  break;
353  }
354  else {
355 
356  Fp = NewFp;
357  }
358  }
359  }
360  except (RtlpWalkFrameChainExceptionFilter (_exception_code(), _exception_info())) {
361 
362  Index = 0;
363  }
364 
365  //
366  // Return the number of return addresses identified on the stack.
367  //
368 
369  return Index;
370 
371 }

387 ULONG
388 RtlCaptureStackContext (
389  OUT PULONG_PTR Callers,
390  OUT PRTL_STACK_CONTEXT Context,
391  IN ULONG Limit
392  )
393 /*++
394 
395 Routine Description:
396 
397  This routine will detect up to `Limit' potential callers from the stack.
398 
399  A potential caller is a pointer (PVOID) that points into one of the
400  regions occupied by modules loaded into the process space (user mode -- dlls)
401  or kernel space (kernel mode -- drivers).
402 
403  Note. Based on experiments you need to save at least 64 pointers to be sure you
404  get a complete stack.
405 
406 Arguments:
407 
408  Callers - vector to be filled with potential return addresses. Its size is
409  expected to be `Limit'. If it is not null then Context should be null.
410 
411  Context - if not null the caller wants the stack context to be saved here
412  as opposed to the Callers parameter.
413 
414  Limit - # of pointers that can be written into Callers and Offsets.
415 
416 Return value:
417 
418  The number of potential callers detected and written into the
419  `Callers' buffer.
420 
421 --*/
422 {
423  ULONG_PTR Current;
424  ULONG_PTR Value;
425  ULONG Index;
426  ULONG_PTR Offset;
427  ULONG_PTR StackStart;
428  ULONG_PTR StackEnd;
429  ULONG_PTR Hint;
430  ULONG_PTR Caller;
431  ULONG_PTR ContextSize;
432 
433  //
434  // Avoid weird conditions. Doing this in an ISR is never a good idea.
435  //
436 
437  if (KeGetCurrentIrql() > DISPATCH_LEVEL) {
438  return 0;
439  }
440 
441  if (Limit == 0) {
442  return 0;
443  }
444 
445  Caller = (ULONG_PTR)_ReturnAddress();
446 
447  if (Context) {
448  Context->Entry[0].Data = Caller;
449  ContextSize = sizeof(RTL_STACK_CONTEXT) + (Limit - 1) * sizeof (RTL_STACK_CONTEXT_ENTRY);
450  }
451  else {
452  Callers[0] = Caller;
453  }
454 
455  //
456  // Get stack limits
457  //
458 
459  _asm mov Hint, EBP;
460 
461 
462  if (! RtlpCaptureStackLimits (Hint, &StackStart, &StackEnd)) {
463  return 0;
464  }
465 
466  //
467  // Synchronize stack traverse pointer to the next word after the first
468  // return address.
469  //
470 
471  for (Current = StackStart; Current < StackEnd; Current += sizeof(ULONG_PTR)) {
472 
473  if (*((PULONG_PTR)Current) == Caller) {
474  break;
475  }
476  }
477 
478  if (Context) {
479  Context->Entry[0].Address = Current;
480  }
481 
482  //
483  // Iterate the stack and pickup potential callers on the way.
484  //
485 
486  Current += sizeof(ULONG_PTR);
487 
488  Index = 1;
489 
490  for ( ; Current < StackEnd; Current += sizeof(ULONG_PTR)) {
491 
492  //
493  // If potential callers buffer is full then wrap this up.
494  //
495 
496  if (Index == Limit) {
497  break;
498  }
499 
500  //
501  // Skip `Callers' buffer because it will give false positives.
502  // It is very likely for this to happen because most probably the buffer
503  // is allocated somewhere upper in the call chain.
504  //
505 
506  if (Context) {
507 
508  if (Current >= (ULONG_PTR)Context && Current < (ULONG_PTR)Context + ContextSize ) {
509  continue;
510  }
511 
512  }
513  else {
514 
515  if ((PULONG_PTR)Current >= Callers && (PULONG_PTR)Current < Callers + Limit ) {
516  continue;
517  }
518  }
519 
520  Value = *((PULONG_PTR)Current);
521 
522  //
523  // Skip small numbers.
524  //
525 
526  if (Value <= 0x10000) {
527  continue;
528  }
529 
530  //
531  // Skip stack pointers.
532  //
533 
534  if (Value >= StackStart && Value <= StackEnd) {
535  continue;
536  }
537 
538  //
539  // Check if `Value' points inside one of the loaded modules.
540  //
541 
542  if (RtlpStkIsPointerInDllRange (Value)) {
543 
544  if (Context) {
545 
546  Context->Entry[Index].Address = Current;
547  Context->Entry[Index].Data = Value;
548  }
549  else {
550 
551  Callers[Index] = Value;
552  }
553 
554  Index += 1;
555  }
556 
557  }
558 
559  if (Context) {
560  Context->NumberOfEntries = Index;
561  }
562 
563  return Index;
564 }

571 //
572 // DLL ranges bitmap
573 //
574 // This range scheme is needed in order to capture stack contexts on x86
575 // machines fast. On IA64 there are totally different algorithms for getting
576 // stack traces.
577 //
578 // Every bit represents 1Mb of virtual space. Since we use the code either
579 // in user mode or kernel mode the first bit of a pointer is not interesting.
580 // Therefore we have to represent 2Gb / 1Mb regions. This totals 256 bytes.
581 //
582 // The bits are set only in loader code paths when a DLL (or driver) gets loaded.
583 // The writing is protected by the loader lock. The bits are read in stack
584 // capturing function.The reading does not require lock protection.
585 //
586 
587 UCHAR RtlpStkDllRanges [2048 / 8];
588 
589 BOOLEAN
590 RtlpStkIsPointerInDllRange (
591  ULONG_PTR Value
592  )
593 {
594  ULONG Index;
595 
596  Value &= ~0x80000000;
597  Index = (ULONG)(Value >> 20);
598 
599  if (RtlpStkDllRanges[Index >> 3] & (UCHAR)(1 << (Index & 7))) {
600 
601  return TRUE;
602  }
603  else {
604 
605  return FALSE;
606  }
607 }

617 #endif

620 #define ROUND_UP(a,b) (((ULONG_PTR)(a) + ((ULONG_PTR)(b) - 1)) & ~((ULONG_PTR)(b) - 1))

626 /*++
627 
628 Routine description:
629 
630  This routine marks the corresponding bits for the loaded dll in the
631  RtlpStkDllRanges variable. This global is used within RtlpDetectDllReferences
632  to save a stack context.
633 
634 Arguments:
635 
636  Loader structure for a loaded dll.
637 
638 Return value:
639 
640  None.
641 
642 Environment:
643 
644  In user mode this function is called from loader code paths. The Peb->LoaderLock
645  is always held while executing this function.
646 
647 --*/
648 {
649  PVOID Base;
650  ULONG Size;
651  ULONG_PTR Current;
652  ULONG_PTR Start;
653  ULONG_PTR End;
654  ULONG Index;
655  ULONG_PTR Value;
656 
657  Base = DllEntry->DllBase;
658  Size = DllEntry->SizeOfImage;
659 
660  //
661  // Find out where is ntdll loaded if we do not know yet.
662  //
663 
664  Start = ROUND_DOWN(Base, SIZE_1_MB);
665  End = ROUND_UP((ULONG_PTR)Base + Size, SIZE_1_MB);
666 
667  for (Current = Start; Current < End; Current += SIZE_1_MB) {
668 
669  Value = Current & ~0x80000000;
670 
671  Index = (ULONG)(Value >> 20);
672 
673  RtlpStkDllRanges[Index >> 3] |= (UCHAR)(1 << (Index & 7));
674  }
675 }

680 RtlpCaptureStackLimits (
681  ULONG_PTR HintAddress,
682  PULONG_PTR StartStack,
683  PULONG_PTR EndStack
684  )
685 /*++
686 
687 Routine Description:
688 
689  This routine figures out what are the stack limits for the current thread.
690  This is used in other routines that need to grovel the stack for various
691  information (e.g. potential return addresses).
692 
693  The function is especially tricky in K-mode where the information kept in
694  the thread structure about stack limits is not always valid because the
695  thread might execute a DPC routine and in this case we use a different stack
696  with different limits.
697 
698 Arguments:
699 
700  HintAddress - Address of a local variable or parameter of the caller of the
701  function that should be the start of the stack.
702 
703  StartStack - start address of the stack (lower value).
704 
705  EndStack - end address of the stack (upper value).
706 
707 Return value:
708 
709  False if some weird condition is discovered, like an End lower than a Start.
710 
711 --*/
712 {
713  //
714  // Avoid weird conditions. Doing this in an ISR is never a good idea.
715  //
716 
717  if (KeGetCurrentIrql() > DISPATCH_LEVEL) {
718  return FALSE;
719  }
720 
721  *StartStack = (ULONG_PTR)(KeGetCurrentThread()->StackLimit);
722  *EndStack = (ULONG_PTR)(KeGetCurrentThread()->StackBase);
723 
724  if (*StartStack <= HintAddress && HintAddress <= *EndStack) {
725 
726  *StartStack = HintAddress;
727  }
728  else {
729 
730  *EndStack = (ULONG_PTR)(KeGetPcr()->Prcb->DpcStack);
731  *StartStack = *EndStack - KERNEL_STACK_SIZE;
732 
733  //
734  // Check if this is within the DPC stack for the current
735  // processor.
736  //
737 
738  if (*EndStack && *StartStack <= HintAddress && HintAddress <= *EndStack) {
739 
740  *StartStack = HintAddress;
741  }
742  else {
743 
744  //
745  // This is not current thread's stack and is not the DPC stack
746  // of the current processor. We will return just the rest of the
747  // stack page containing HintAddress as a valid stack range.
748  //
749 
750  *StartStack = HintAddress;
751 
752  *EndStack = (*StartStack + PAGE_SIZE) & ~((ULONG_PTR)PAGE_SIZE - 1);
753  }
754  }
755 
756  return TRUE;
757 }
