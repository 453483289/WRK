191 RtlpQueryRegistryDirect(
192  IN ULONG ValueType,
193  IN PVOID ValueData,
194  IN ULONG ValueLength,
195  IN OUT PVOID Destination
196  )
197 {
198 
199  if (ValueType == REG_SZ ||
200  ValueType == REG_EXPAND_SZ ||
201  ValueType == REG_MULTI_SZ
202  ) {
203  PUNICODE_STRING DestinationString;
204  USHORT TruncValueLength;
205 
206  //
207  // Truncate ValueLength to be represented in a UNICODE_STRING
208  //
209  if ( ValueLength <= MAX_USTRING ) {
210  TruncValueLength = (USHORT)ValueLength;
211  } else {
212  TruncValueLength = MAX_USTRING;
213 
214 #if DBG
215  DbgPrint("RtlpQueryRegistryDirect: truncating SZ Value length: %x -> %x\n",
216  ValueLength, TruncValueLength);
217 #endif //DBG
218  }
219 
220  DestinationString = (PUNICODE_STRING)Destination;
221  if (DestinationString->Buffer == NULL) {
222 
223  DestinationString->Buffer = RtlAllocateStringRoutine( TruncValueLength );
224  if (!DestinationString->Buffer) {
225  return STATUS_NO_MEMORY;
226  }
227  DestinationString->MaximumLength = TruncValueLength;
228  } else if (TruncValueLength > DestinationString->MaximumLength) {
229  return STATUS_BUFFER_TOO_SMALL;
230  }
231 
232  RtlCopyMemory( DestinationString->Buffer, ValueData, TruncValueLength );
233  DestinationString->Length = (TruncValueLength - sizeof(UNICODE_NULL));
234 
235  } else if (ValueLength <= sizeof( ULONG )) {
236  if (Destination != ValueData && ValueLength)
237  RtlCopyMemory( Destination, ValueData, ValueLength );
238 
239  } else {
240  PULONG DestinationLength;
241 
242  DestinationLength = (PULONG)Destination;
243  if ((LONG)*DestinationLength < 0) {
244  ULONG n = -(LONG)*DestinationLength;
245 
246  if (n < ValueLength) {
247  return STATUS_BUFFER_TOO_SMALL;
248  }
249  RtlCopyMemory( DestinationLength, ValueData, ValueLength );
250 
251  } else {
252  if (*DestinationLength < (2 * sizeof(*DestinationLength) + ValueLength)) {
253  return STATUS_BUFFER_TOO_SMALL;
254  }
255 
256  *DestinationLength++ = ValueLength;
257  *DestinationLength++ = ValueType;
258  RtlCopyMemory( DestinationLength, ValueData, ValueLength );
259  }
260  }
261 
262  return STATUS_SUCCESS;
263 }

269 NTSTATUS
270 RtlpCallQueryRegistryRoutine(
271  IN PRTL_QUERY_REGISTRY_TABLE QueryTable,
272  IN PKEY_VALUE_FULL_INFORMATION KeyValueInformation,
273  IN OUT PULONG PKeyValueInfoLength,
274  IN PVOID Context,
275  IN PVOID Environment OPTIONAL
276  )
277 
278 /*++
279 
280 Routine Description:
281 
282  This function implements the caller out the a caller specified
283  routine. It is reponsible for capturing the arguments for the
284  routine and then calling it. If not specifically disabled, this
285  routine will converted REG_EXPAND_SZ Registry values to REG_SZ by
286  calling RtlExpandEnvironmentStrings_U prior to calling the routine.
287  It will also converted REG_MULTI_SZ registry values into multiple
288  REG_SZ calls to the specified routine.
289 
290  N.B. UNICODE_STRINGs cannot handle strings exceeding MAX_USTRING bytes. This creates
291  issues both for expansion and for returning queries. Whenever this limitation
292  is a encountered, we punt as best we can -- often returning an unexpanded, or perhaps
293  truncated stream -- since this seems to create fewer problems for our callers than
294  if we unexpectedly fail.
295 
296 Arguments:
297 
298  QueryTable - specifies the current query table entry.
299 
300  KeyValueInformation - points to a buffer that contains the information
301  about the current registry value.
302 
303  PKeyValueInfoLength - pointer to the maximum length of the KeyValueInformation
304  buffer. This function will use the
305  unused portion at the end of this buffer for storing null terminated
306  value name strings and the expanded version of REG_EXPAND_SZ values.
307  PKeyValueInfoLength returns an estimate of the space required if
308  STATUS_BUFFER_TOO_SMALL is returned. This estimate can be used to retry
309  with a larger buffer. Two retries may be required if REG_EXPAND_SZ is specified.
310 
311  Context - specifies a 32-bit quantity that is passed uninterpreted to
312  each QueryRoutine called.
313 
314  Environment - optional parameter, that if specified is the environment
315  used when expanding variable values in REG_EXPAND_SZ registry
316  values.
317 
318 Return Value:
319 
320  Status of the operation.
321 
322 --*/
323 {
324  NTSTATUS Status;
325  ULONG ValueType;
326  PWSTR ValueName;
327  PVOID ValueData;
328  ULONG ValueLength;
329  PWSTR s;
330  PCHAR FreeMem;
331  PCHAR EndFreeMem;
332  LONG FreeMemSize;
333  ULONG KeyValueInfoLength;
334  int retries;
335 
336 
337  //
338  // Return 0 length unless we return STATUS_BUFFER_TOO_SMALL.
339  //
340  KeyValueInfoLength = *PKeyValueInfoLength;
341  *PKeyValueInfoLength = 0;
342 
343  //
344  // the registry has signaled no data for this value
345  //
346  if( KeyValueInformation->DataOffset == (ULONG)-1 ) {
347  //
348  // Return success unless this is a required value.
349  //
350  if ( QueryTable->Flags & RTL_QUERY_REGISTRY_REQUIRED ) {
351  return STATUS_OBJECT_NAME_NOT_FOUND;
352  } else {
353  return STATUS_SUCCESS;
354  }
355  }
356 
357  //
358  // Initially assume the entire KeyValueInformation buffer is unused.
359  //
360 
361  FreeMem = (PCHAR)KeyValueInformation;
362  FreeMemSize = KeyValueInfoLength;
363  EndFreeMem = FreeMem + FreeMemSize;
364 
365  if (KeyValueInformation->Type == REG_NONE ||
366  (KeyValueInformation->DataLength == 0 &&
367  KeyValueInformation->Type == QueryTable->DefaultType)
368  ) {
369 
370  //
371  // If there is no registry value then see if they want to default
372  // this value.
373  //
374  if (QueryTable->DefaultType == REG_NONE) {
375  //
376  // No default value specified. Return success unless this is
377  // a required value.
378  //
379  if ( QueryTable->Flags & RTL_QUERY_REGISTRY_REQUIRED ) {
380  return STATUS_OBJECT_NAME_NOT_FOUND;
381  } else {
382  return STATUS_SUCCESS;
383  }
384  }
385 
386  //
387  // Default requested. Setup the value data pointers from the
388  // information in the table entry.
389  //
390 
391  ValueName = QueryTable->Name,
392  ValueType = QueryTable->DefaultType;
393  ValueData = QueryTable->DefaultData;
394  ValueLength = QueryTable->DefaultLength;
395  if (ValueLength == 0) {
396  //
397  // If the length of the value is zero, then calculate the
398  // actual length for REG_SZ, REG_EXPAND_SZ and REG_MULTI_SZ
399  // value types.
400  //
401 
402  s = (PWSTR)ValueData;
403  if (ValueType == REG_SZ || ValueType == REG_EXPAND_SZ) {
404  while (*s++ != UNICODE_NULL) {
405  }
406  ValueLength = (ULONG)((PCHAR)s - (PCHAR)ValueData);
407 
408  } else if (ValueType == REG_MULTI_SZ) {
409  while (*s != UNICODE_NULL) {
410  while (*s++ != UNICODE_NULL) {
411  }
412  }
413  ValueLength = (ULONG)((PCHAR)s - (PCHAR)ValueData) + sizeof( UNICODE_NULL );
414  }
415  }
416 
417  } else {
418  if (!(QueryTable->Flags & RTL_QUERY_REGISTRY_DIRECT)) {
419  LONG ValueSpaceNeeded;
420 
421  //
422  // There is a registry value. Calculate a pointer to the
423  // free memory at the end of the value information buffer,
424  // and its size.
425  //
426  if (KeyValueInformation->DataLength) {
427  FreeMem += KeyValueInformation->DataOffset +
428  KeyValueInformation->DataLength;
429  } else {
430  FreeMem += FIELD_OFFSET(KEY_VALUE_FULL_INFORMATION, Name) +
431  KeyValueInformation->NameLength;
432  }
433  FreeMem = (PCHAR)QuadAlignPtr(FreeMem);
434  FreeMemSize = (ULONG) (EndFreeMem - FreeMem);
435 
436  //
437  // See if there is room in the free memory area for a null
438  // terminated copy of the value name string. If not return
439  // the length we require (so far) and an error.
440  //
441  ValueSpaceNeeded = KeyValueInformation->NameLength + sizeof(UNICODE_NULL);
442  if ( FreeMemSize < ValueSpaceNeeded ) {
443 
444  *PKeyValueInfoLength = (ULONG)(((PCHAR)FreeMem - (PCHAR)KeyValueInformation) + ValueSpaceNeeded);
445  return STATUS_BUFFER_TOO_SMALL;
446  }
447 
448  //
449  // There is room, so copy the string, and null terminate it.
450  //
451 
452  ValueName = (PWSTR)FreeMem;
453  RtlCopyMemory( ValueName,
454  KeyValueInformation->Name,
455  KeyValueInformation->NameLength
456  );
457  *(PWSTR)((PCHAR)ValueName + KeyValueInformation->NameLength) = UNICODE_NULL;
458 
459  //
460  // Update the free memory pointer and size to reflect the space we
461  // just used for the null terminated value name.
462  //
463  FreeMem += ValueSpaceNeeded;
464  FreeMem = (PCHAR)QuadAlignPtr(FreeMem);
465  FreeMemSize = (LONG) (EndFreeMem - FreeMem);
466 
467  } else {
468  ValueName = QueryTable->Name;
469  }
470 
471  //
472  // Get the remaining data for the registry value.
473  //
474 
475  ValueType = KeyValueInformation->Type;
476  ValueData = (PCHAR)KeyValueInformation + KeyValueInformation->DataOffset;
477  ValueLength = KeyValueInformation->DataLength;
478  }
479 
480  //
481  // Unless specifically disabled for this table entry, preprocess
482  // registry values of type REG_EXPAND_SZ and REG_MULTI_SZ
483  //
484 
485  if (!(QueryTable->Flags & RTL_QUERY_REGISTRY_NOEXPAND)) {
486  if (ValueType == REG_MULTI_SZ) {
487  PWSTR ValueEnd;
488 
489  //
490  // For REG_MULTI_SZ value type, call the query routine once
491  // for each null terminated string in the registry value. Fake
492  // like this is multiple REG_SZ values with the same value name.
493  //
494 
495  Status = STATUS_SUCCESS;
496  ValueEnd = (PWSTR)((PCHAR)ValueData + ValueLength) - sizeof(UNICODE_NULL);
497  s = (PWSTR)ValueData;
498  while (s < ValueEnd) {
499  while (*s++ != UNICODE_NULL) {
500  }
501 
502  ValueLength = (ULONG)((PCHAR)s - (PCHAR)ValueData);
503  if (QueryTable->Flags & RTL_QUERY_REGISTRY_DIRECT) {
504  Status = RtlpQueryRegistryDirect( REG_SZ,
505  ValueData,
506  ValueLength,
507  QueryTable->EntryContext
508  );
509  (PUNICODE_STRING)(QueryTable->EntryContext) += 1;
510 
511  } else {
512  Status = (QueryTable->QueryRoutine)( ValueName,
513  REG_SZ,
514  ValueData,
515  ValueLength,
516  Context,
517  QueryTable->EntryContext
518  );
519  }
520 
521  //
522  // We ignore failures where the buffer is too small.
523  //
524  if (Status == STATUS_BUFFER_TOO_SMALL) {
525  Status = STATUS_SUCCESS;
526  }
527 
528  if (!NT_SUCCESS( Status )) {
529  break;
530  }
531 
532  ValueData = (PVOID)s;
533  }
534 
535  return Status;
536  }
537 
538  //
539  // If requested, expand the Value -- but only if the unexpanded value
540  // can be represented with a UNICODE_STRING.
541  //
542  if ((ValueType == REG_EXPAND_SZ) &&
543  (ValueLength >= sizeof(WCHAR)) &&
544  (ValueLength <= MAX_NONNULL_USTRING)) {
545  //
546  // For REG_EXPAND_SZ value type, expand any environment variable
547  // references in the registry value string using the Rtl function.
548  //
549 
550  UNICODE_STRING Source;
551  UNICODE_STRING Destination;
552  PWCHAR Src;
553  ULONG SrcLength;
554  ULONG RequiredLength;
555  BOOLEAN PercentFound;
556 
557  //
558  // Don't expand unless we have to since expansion doubles buffer usage.
559  //
560 
561  PercentFound = FALSE;
562  SrcLength = ValueLength - sizeof(WCHAR);
563  Src = (PWSTR)ValueData;
564  while (SrcLength) {
565  if (*Src == L'%') {
566  PercentFound = TRUE;
567  break;
568  }
569  Src++;
570  SrcLength -= sizeof(WCHAR);
571  }
572 
573  if ( PercentFound ) {
574  Source.Buffer = (PWSTR)ValueData;
575  Source.MaximumLength = (USHORT)ValueLength;
576  Source.Length = (USHORT)(Source.MaximumLength - sizeof(UNICODE_NULL));
577  Destination.Buffer = (PWSTR)FreeMem;
578  Destination.Length = 0;
579 
580  if (FreeMemSize <= 0) {
581  Destination.MaximumLength = 0;
582  } else if (FreeMemSize <= MAX_USTRING) {
583  Destination.MaximumLength = (USHORT)FreeMemSize;
584  Destination.Buffer[FreeMemSize/sizeof(WCHAR) - 1] = UNICODE_NULL;
585  } else {
586  Destination.MaximumLength = MAX_USTRING;
587  Destination.Buffer[MAX_USTRING/sizeof(WCHAR) - 1] = UNICODE_NULL;
588  }
589 
590  Status = RtlExpandEnvironmentStrings_U( Environment,
591  &Source,
592  &Destination,
593  &RequiredLength
594  );
595  ValueType = REG_SZ;
596 
597  if ( NT_SUCCESS(Status) ) {
598  ValueData = Destination.Buffer;
599  ValueLength = Destination.Length + sizeof( UNICODE_NULL );
600  } else {
601  if (Status == STATUS_BUFFER_TOO_SMALL) {
602  *PKeyValueInfoLength = (ULONG)((PCHAR)FreeMem - (PCHAR)KeyValueInformation) + RequiredLength;
603  }
604 //#if DBG
605  if (Status == STATUS_BUFFER_TOO_SMALL) {
606  DbgPrint( "RTL: Expand variables for %wZ failed - Status == %lx Size %x > %x <%x>\n",
607  &Source, Status, *PKeyValueInfoLength, KeyValueInfoLength,
608  Destination.MaximumLength );
609  } else {
610  DbgPrint( "RTL: Expand variables for %wZ failed - Status == %lx\n", &Source, Status );
611  }
612 //#endif // DBG
613  if ( Status == STATUS_BUFFER_OVERFLOW ||
614  Status == STATUS_BUFFER_TOO_SMALL &&
615  ( Destination.MaximumLength == MAX_USTRING
616  || RequiredLength > MAX_NONNULL_USTRING ) ) {
617 
618  // We can't do variable expansion because the required buffer can't be described
619  // by a UNICODE_STRING, so we silently ignore expansion.
620  } else {
621  return Status;
622  }
623  }
624  }
625  }
626  }
627 
628  //
629  // No special process of the registry value required so just call
630  // the query routine.
631  //
632  if (QueryTable->Flags & RTL_QUERY_REGISTRY_DIRECT) {
633  Status = RtlpQueryRegistryDirect( ValueType,
634  ValueData,
635  ValueLength,
636  QueryTable->EntryContext
637  );
638  } else {
639  Status = (QueryTable->QueryRoutine)( ValueName,
640  ValueType,
641  ValueData,
642  ValueLength,
643  Context,
644  QueryTable->EntryContext
645  );
646 
647  }
648 
649  //
650  // At this point we fail silently if the buffer is too small.
651  //
652  if (Status == STATUS_BUFFER_TOO_SMALL) {
653  Status = STATUS_SUCCESS;
654  }
655  return Status;
656 }

706 RtlQueryRegistryValues(
707  IN ULONG RelativeTo,
708  IN PCWSTR Path,
709  IN PRTL_QUERY_REGISTRY_TABLE QueryTable,
710  IN PVOID Context,
711  IN PVOID Environment OPTIONAL
712  )
713 
714 /*++
715 
716 Routine Description:
717 
718  This function allows the caller to query multiple values from the registry
719  sub-tree with a single call. The caller specifies an initial key path,
720  and a table. The table contains one or more entries that describe the
721  key values and subkey names the caller is interested in. This function
722  starts at the initial key and enumerates the entries in the table. For
723  each entry that specifies a value name or subkey name that exists in
724  the registry, this function calls the caller's query routine associated
725  with each table entry. The caller's query routine is passed the value
726  name, type, data and data length, to do with what they wish.
727 
728 Arguments:
729 
730  RelativeTo - specifies that the Path parameter is either an absolute
731  registry path, or a path relative to a predefined key path. The
732  following values are defined:
733 
734  RTL_REGISTRY_ABSOLUTE - Path is an absolute registry path
735  RTL_REGISTRY_SERVICES - Path is relative to \Registry\Machine\System\CurrentControlSet\Services
736  RTL_REGISTRY_CONTROL - Path is relative to \Registry\Machine\System\CurrentControlSet\Control
737  RTL_REGISTRY_WINDOWS_NT - Path is relative to \Registry\Machine\Software\Microsoft\Windows NT\CurrentVersion
738  RTL_REGISTRY_DEVICEMAP - Path is relative to \Registry\Machine\Hardware\DeviceMap
739  RTL_REGISTRY_USER - Path is relative to \Registry\User\CurrentUser
740 
741  RTL_REGISTRY_OPTIONAL - Bit that specifies the key referenced by
742  this parameter and the Path parameter is
743  optional.
744
