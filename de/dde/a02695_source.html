13 Abstract:
14 
15  This module implements routines providing client impersonation to
16  communication session layers (such as LPC Ports).
17 
18  WARNING: The following notes apply to the use of these services:
19 
20  (1) No synchronization of operations to a security context block are
21  performed by these services. The caller of these services must
22  ensure that use of an individual security context block is
23  serialized to prevent simultaneous, incompatible updates.
24 
25  (2) Any or all of these services may create, open, or operate on a
26  token object. This may result in a mutex being acquired at
27  MUTEXT_LEVEL_SE_TOKEN level. The caller must ensure that no
28  mutexes are held at levels that conflict with this action.

65 SepCreateClientSecurity(
66  IN PACCESS_TOKEN Token,
67  IN PSECURITY_QUALITY_OF_SERVICE ClientSecurityQos,
68  IN BOOLEAN ServerIsRemote,

72  OUT PSECURITY_CLIENT_CONTEXT ClientContext

75 /*++
76 
77 Routine Description
78 
79  This routine creates a context block to represent the passed token. The token
80  is expected to be properly referenced when passed to the function. If the call
81  is unsuccessful, the caller is responsible to dereference the token. Also, if the
82  call succeeds but the caller requested SECURITY_STATIC_TRACKING, then we have duplicated
83  the token and the caller is again responsible to dereference the passed token.
84 
85 Arguments
86 
87  Token - The effective token for which the context is constructed.
88 
89  ClientSecurityQos - Points to the security quality of service
90  parameters specified by the client for this communication
91  session.
92 
93  ServerIsRemote - Provides an indication as to whether the session
94  this context block is being used for is an inter-system
95  session or intra-system session. This is reconciled with the
96  impersonation level of the client thread's token (in case the
97  client has a client of his own that didn't specify delegation).
98 
99  TokenType - specifies the type of the passed token.
100 
101  ThreadEffectiveOnly - if the token is an impersonation token, then this
102  is the thread's ImpersonationInfo->EffectiveOnly field value.
103 
104  ImpersonationLevel - the impersonation level of the token.
105 
106  ClientContext - Points to the client security context block to be
107  initialized.
108 
109 Return Value
110 
111 --*/
112 
113 {
114  NTSTATUS Status = STATUS_SUCCESS;
115  PACCESS_TOKEN DuplicateToken;
116 
117  PAGED_CODE();
118 
119  if ( !VALID_IMPERSONATION_LEVEL(ClientSecurityQos->ImpersonationLevel) ) {
120 
121  return STATUS_INVALID_PARAMETER;
122  }
123 
124  //
125  // Make sure the client is not trying to abuse use of a
126  // client of its own by attempting an invalid impersonation.
127  // Also set the ClientContext->DirectAccessEffectiveOnly flag
128  // appropriately if the impersonation is legitimate. The
129  // DirectAccessEffectiveOnly flag value will end up being ignored
130  // if STATIC mode is requested, but this is the most efficient
131  // place to calculate it, and we are optimizing for DYNAMIC mode.
132  //
133 
134  if (TokenType == TokenImpersonation) {
135 
136  if ( ClientSecurityQos->ImpersonationLevel > ImpersonationLevel) {
137 
138  return STATUS_BAD_IMPERSONATION_LEVEL;
139  }
140 
141  if ( SepBadImpersonationLevel(ImpersonationLevel,ServerIsRemote)) {
142 
143  return STATUS_BAD_IMPERSONATION_LEVEL;
144 
145  } else {
146 
147  //
148  // TokenType is TokenImpersonation and the impersonation is legit.
149  // Set the DirectAccessEffectiveOnly flag to be the minimum of
150  // the current thread value and the caller specified value.
151  //
152 
153  ClientContext->DirectAccessEffectiveOnly =
154  ( (ThreadEffectiveOnly || (ClientSecurityQos->EffectiveOnly)) ?
155  TRUE : FALSE );
156  }
157 
158  } else {
159 
160  //
161  // TokenType is TokenPrimary. In this case, the client specified
162  // EffectiveOnly value is always used.
163  //
164 
165  ClientContext->DirectAccessEffectiveOnly =
166  ClientSecurityQos->EffectiveOnly;
167  }
168 
169  //
170  // Copy the token if necessary (i.e., static tracking requested)
171  //
172 
173  if (ClientSecurityQos->ContextTrackingMode == SECURITY_STATIC_TRACKING) {
174 
175  ClientContext->DirectlyAccessClientToken = FALSE;
176 
177  Status = SeCopyClientToken(
178  Token,
179  ClientSecurityQos->ImpersonationLevel,
180  KernelMode,
181  &DuplicateToken
182  );
183 
184  Token = DuplicateToken;
185 
186  //
187  // If there was an error, we're done.
188  //
189  if (!NT_SUCCESS(Status)) {
190  return Status;
191  }
192 
193  } else {
194 
195  ClientContext->DirectlyAccessClientToken = TRUE;
196 
197  if (ServerIsRemote) {
198  //
199  // Get a copy of the client token's control information
200  // so that we can tell if it changes in the future.
201  //
202 
203  SeGetTokenControlInformation( Token,
204  &ClientContext->ClientTokenControl
205  );
206 
207  }
208 
209  }
210 
211  ClientContext->SecurityQos.Length =
212  (ULONG)sizeof(SECURITY_QUALITY_OF_SERVICE);
213 
214  ClientContext->SecurityQos.ImpersonationLevel =
215  ClientSecurityQos->ImpersonationLevel;
216 
217  ClientContext->SecurityQos.ContextTrackingMode =
218  ClientSecurityQos->ContextTrackingMode;
219 
220  ClientContext->SecurityQos.EffectiveOnly =
221  ClientSecurityQos->EffectiveOnly;
222 
223  ClientContext->ServerIsRemote = ServerIsRemote;
224 
225  ClientContext->ClientToken = Token;
226 
227  return STATUS_SUCCESS;
228 }

230 NTSTATUS
231 SeCreateClientSecurity (
232  __in PETHREAD ClientThread,
233  __in PSECURITY_QUALITY_OF_SERVICE ClientSecurityQos,
234  __in BOOLEAN ServerIsRemote,
235  __out PSECURITY_CLIENT_CONTEXT ClientContext
236  )
237 
238 /*++
239 
240 Routine Description:
241 
242  This service initializes a context block to represent a client's
243  security context. This may simply result in a reference to the
244  client's token, or may cause the client's token to be duplicated,
245  depending upon the security quality of service information specified.
246 
247  NOTE
248 
249  The code in this routine is optimized for DYNAMIC context
250  tracking. This is only mode in which direct access to a
251  caller's token is allowed, and the mode expected to be used
252  most often. STATIC context tracking always requires the
253  caller's token to be copied.
254 
255 
256 Arguments:
257 
258  ClientThread - Points to the client's thread. This is used to
259  locate the client's security context (token).
260 
261  ClientSecurityQos - Points to the security quality of service
262  parameters specified by the client for this communication
263  session.
264 
265  ServerIsRemote - Provides an indication as to whether the session
266  this context block is being used for is an inter-system
267  session or intra-system session. This is reconciled with the
268  impersonation level of the client thread's token (in case the
269  client has a client of his own that didn't specify delegation).
270 
271  ClientContext - Points to the client security context block to be
272  initialized.
273 
274 
275 Return Value:
276 
277  STATUS_SUCCESS - The service completed successfully.
278 
279  STATUS_BAD_IMPERSONATION_LEVEL - The client is currently
280  impersonating either an Anonymous or Identification level
281  token, which can not be passed on for use by another server.
282  This status may also be returned if the security context
283  block is for an inter-system communication session and the
284  client thread is impersonating a client of its own using
285  other than delegation impersonation level.
286 
287 
288 --*/
289 
290 {
291  NTSTATUS Status = STATUS_SUCCESS;
292  PACCESS_TOKEN Token;
293  TOKEN_TYPE TokenType;
294  BOOLEAN ThreadEffectiveOnly;
295  SECURITY_IMPERSONATION_LEVEL ImpersonationLevel;
296  PACCESS_TOKEN DuplicateToken;
297 
298  PAGED_CODE();
299 
300  //
301  // Gain access to the client thread's effective token
302  //
303 
304  Token = PsReferenceEffectiveToken(
305  ClientThread,
306  &TokenType,
307  &ThreadEffectiveOnly,
308  &ImpersonationLevel
309  );
310 
311 
312  Status = SepCreateClientSecurity(
313  Token,
314  ClientSecurityQos,
315  ServerIsRemote,
316  TokenType,
317  ThreadEffectiveOnly,
318  ImpersonationLevel,
319  ClientContext );
320 
321  //
322  // If failed, or if a token was copied internally, then deref our token.
323  //
324 
325  if ((!NT_SUCCESS( Status )) || (ClientSecurityQos->ContextTrackingMode == SECURITY_STATIC_TRACKING)) {
326 
327  ObDereferenceObject( Token );
328  }
329 
330  return Status;
331 }
