544 // begin_ntifs
545 //
546 // Common ReadAhead call for Copy Read and Mdl Read.
547 //
548 // ReadAhead should always be invoked by calling the CcReadAhead macro,
549 // which tests first to see if the read is large enough to warrant read
550 // ahead. Measurements have shown that, calling the read ahead routine
551 // actually decreases performance for small reads, such as issued by
552 // many compilers and linkers. Compilers simply want all of the include
553 // files to stay in memory after being read the first time.
554 //
555 
556 #define CcReadAhead(FO,FOFF,LEN) { \
557  if ((LEN) >= 256) { \
558  CcScheduleReadAhead((FO),(FOFF),(LEN)); \
559  } \
560 }

602 // Flags for pinning

604 // Note: The flags for pinning and the flags for mapping cannot overlap unless
605 // the flag has the same meaning.
606 //
607 
608 //
609 // Synchronous Wait - normally specified. This pattern may be specified as TRUE.
610 //
611 
612 #define PIN_WAIT (1)
613 
614 //
615 // Acquire metadata Bcb exclusive (default is shared, Lazy Writer uses exclusive).
616 //
617 // Must be set with PIN_WAIT.
618 //
619 
620 #define PIN_EXCLUSIVE (2)
621 
622 //
623 // Acquire metadata Bcb but do not fault data in. Default is to fault the data in.
624 // This unusual flag is only used by Ntfs for cache coherency synchronization between
625 // compressed and uncompressed streams for the same compressed file.
626 //
627 // Must be set with PIN_WAIT.
628 //
629 
630 #define PIN_NO_READ (4)
631 
632 //
633 // This option may be used to pin data only if the Bcb already exists. If the Bcb
634 // does not already exist - the pin is unsuccessful and no Bcb is returned. This routine
635 // provides a way to see if data is already pinned (and possibly dirty) in the cache,
636 // without forcing a fault if the data is not there.
637 //
638 
639 #define PIN_IF_BCB (8)
640 
641 //
642 // If this option is specified, the caller is responsible for tracking the
643 // dirty ranges and calling MmSetAddressRangeModified on these ranges before
644 // they are flushed. Ranges should only be pinned via this manner if the
645 // entire range will be written or purged (one or the other must occur).
646 //
647 
648 #define PIN_CALLER_TRACKS_DIRTY_DATA (32)
