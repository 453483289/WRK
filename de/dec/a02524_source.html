58 /*++
59 
60 Routine Description:
61 
62  This function takes two user supplied page protections and checks
63  to see if the new protection is compatible with the old protection.
64 
65  protection compatible protections
66 
67  NoAccess NoAccess
68  ReadOnly NoAccess, ReadOnly, ReadWriteCopy
69  ReadWriteCopy NoAccess, ReadOnly, ReadWriteCopy
70  ReadWrite NoAccess, ReadOnly, ReadWriteCopy, ReadWrite
71  Execute NoAccess, Execute
72  ExecuteRead NoAccess, ReadOnly, ReadWriteCopy, Execute, ExecuteRead,
73  ExecuteWriteCopy
74  ExecuteWrite NoAccess, ReadOnly, ReadWriteCopy, Execute, ExecuteRead,
75  ExecuteWriteCopy, ReadWrite, ExecuteWrite
76  ExecuteWriteCopy NoAccess, ReadOnly, ReadWriteCopy, Execute, ExecuteRead,
77  ExecuteWriteCopy
78 
79 Arguments:
80 
81  OldWin32Protect - Supplies the protection to be compatible with.
82 
83  NewWin32Protect - Supplies the protection to check out.
84 
85 Return Value:
86 
87  Returns TRUE if the protection is compatible, FALSE if not.
88 
89 Environment:
90 
91  Kernel Mode.
92 
93 --*/
94 
95 {
96  MM_PROTECTION_MASK Mask;
97  MM_PROTECTION_MASK PteProtection;
98  WIN32_PROTECTION_MASK Win32ProtectMask;
99 
100  PteProtection = MiMakeProtectionMask (OldWin32Protect);
101 
102  if (PteProtection == MM_INVALID_PROTECTION) {
103  return FALSE;
104  }
105 
106  Mask = PteProtection & 0x7;
107 
108  Win32ProtectMask = MmCompatibleProtectionMask[Mask] | PAGE_GUARD | PAGE_NOCACHE | PAGE_WRITECOMBINE;
109 
110  if ((Win32ProtectMask | NewWin32Protect) != Win32ProtectMask) {
111  return FALSE;
112  }
113 
114  return TRUE;
115 }

181 FASTCALL
182 MiMakeProtectionMask (
183  IN WIN32_PROTECTION_MASK Win32Protect
184  )
185 
186 /*++
187 
188 Routine Description:
189 
190  This function takes a user supplied protection and converts it
191  into a 5-bit protection code for the PTE.
192 
193 Arguments:
194 
195  Win32Protect - Supplies the protection.
196 
197 Return Value:
198 
199  Returns the protection code for use in the PTE. Note that
200  MM_INVALID_PROTECTION (-1) is returned for an invalid protection
201  request. Since valid PTE protections fit in 5 bits and are
202  zero-extended, it's easy for callers to distinguish this.
203 
204 Environment:
205 
206  Kernel Mode.
207 
208 --*/
209 
210 {
211  ULONG Field1;
212  ULONG Field2;
213  MM_PROTECTION_MASK ProtectCode;
214 
215  if (Win32Protect >= (PAGE_WRITECOMBINE * 2)) {
216  return MM_INVALID_PROTECTION;
217  }
218 
219  Field1 = Win32Protect & 0xF;
220  Field2 = (Win32Protect >> 4) & 0xF;
221 
222  //
223  // Make sure at least one field is set.
224  //
225 
226  if (Field1 == 0) {
227  if (Field2 == 0) {
228 
229  //
230  // Both fields are zero, return failure.
231  //
232 
233  return MM_INVALID_PROTECTION;
234  }
235  ProtectCode = MmUserProtectionToMask2[Field2];
236  }
237  else {
238  if (Field2 != 0) {
239  //
240  // Both fields are non-zero, return failure.
241  //
242 
243  return MM_INVALID_PROTECTION;
244  }
245  ProtectCode = MmUserProtectionToMask1[Field1];
246  }
247 
248  if (ProtectCode == -1) {
249  return MM_INVALID_PROTECTION;
250  }
251 
252  if (Win32Protect & PAGE_GUARD) {
253 
254  if ((ProtectCode == MM_NOACCESS) ||
255  (Win32Protect & (PAGE_NOCACHE | PAGE_WRITECOMBINE))) {
256 
257  //
258  // Invalid protection -
259  // guard and either no access, no cache or write combine.
260  //
261 
262  return MM_INVALID_PROTECTION;
263  }
264 
265  MI_ADD_GUARD (ProtectCode);
266  }
267 
268  if (Win32Protect & PAGE_NOCACHE) {
269 
270  ASSERT ((Win32Protect & PAGE_GUARD) == 0); // Already checked above
271 
272  if ((ProtectCode == MM_NOACCESS) ||
273  (Win32Protect & PAGE_WRITECOMBINE)) {
274 
275  //
276  // Invalid protection -
277  // nocache and either no access or write combine.
278  //
279 
280  return MM_INVALID_PROTECTION;
281  }
282 
283  MI_ADD_NOCACHE (ProtectCode);
284  }
285 
286  if (Win32Protect & PAGE_WRITECOMBINE) {
287 
288  ASSERT ((Win32Protect & (PAGE_GUARD|PAGE_NOACCESS)) == 0); // Already checked above
289 
290  if (ProtectCode == MM_NOACCESS) {
291 
292  //
293  // Invalid protection, no access and write combine.
294  //
295 
296  return MM_INVALID_PROTECTION;
297  }
298 
299  MI_ADD_WRITECOMBINE (ProtectCode);
300  }
301 
302  return ProtectCode;
303 }

307 MiMakePdeExistAndMakeValid (
308  IN PMMPTE PointerPde,
309  IN PEPROCESS TargetProcess,
310  IN KIRQL OldIrql
311  )

313 /*++
314 
315 Routine Description:
316 
317  This routine examines the specified Page Directory Parent Entry to
318  determine if the page directory page mapped by the PPE exists. If it does,
319  then it examines the specified Page Directory Entry to determine if
320  the page table page mapped by the PDE exists.
321 
322  If the page table page exists and is not currently in memory, the
323  working set pushlock and, if held, the PFN lock are released and the
324  page table page is faulted into the working set. The pushlock and PFN
325  lock are reacquired.
326 
327  If the PDE does not exist, a zero filled PTE is created and it
328  too is brought into the working set.
329 
330 Arguments:
331 
332  PointerPde - Supplies a pointer to the PDE to examine and bring
333  into the working set.
334 
335  TargetProcess - Supplies a pointer to the current process.
336 
337  OldIrql - Supplies the IRQL the caller acquired the PFN lock at or MM_NOIRQL
338  if the caller does not hold the PFN lock.
339 
340 Return Value:
341 
342  None.
343 
344 Environment:
345 
346  Kernel mode, APCs disabled, working set pushlock held.
347 
348 --*/
349 
350 {
351  PMMPTE PointerPte;
352  PMMPTE PointerPpe;
353  PMMPTE PointerPxe;
354 
355  ASSERT (KeAreAllApcsDisabled () == TRUE);
356 
357  PointerPpe = MiGetPteAddress (PointerPde);
358  PointerPxe = MiGetPdeAddress (PointerPde);
359 
360  if ((PointerPxe->u.Hard.Valid == 1) &&
361  (PointerPpe->u.Hard.Valid == 1) &&
362  (PointerPde->u.Hard.Valid == 1)) {
363 
364  //
365  // Already valid.
366  //
367 
368  return;
369  }
370 
371  if (OldIrql != MM_NOIRQL) {
372  UNLOCK_PFN (OldIrql);
373  }
374 
375  //
376  // Page directory parent (or extended parent) entry not valid,
377  // make it valid.
378  //
379 
380  PointerPte = MiGetVirtualAddressMappedByPte (PointerPde);
381 
382  do {
383 
384  ASSERT (KeAreAllApcsDisabled () == TRUE);
385 
386  //
387  // Fault it in, this must be done one level at a time because the
388  // fault handler checks for the preceding level already being valid
389  // for system space addresses.
390  //
391 
392  if (PointerPxe->u.Hard.Valid == 0) {
393  MiMakeSystemAddressValid (PointerPpe, TargetProcess);
394  ASSERT (PointerPxe->u.Hard.Valid == 1);
395  }
396 
397  if (PointerPpe->u.Hard.Valid == 0) {
398  MiMakeSystemAddressValid (PointerPde, TargetProcess);
399  ASSERT (PointerPpe->u.Hard.Valid == 1);
400  }
401 
402  MiMakeSystemAddressValid (PointerPte, TargetProcess);
403 
404  ASSERT (PointerPxe->u.Hard.Valid == 1);
405  ASSERT (PointerPpe->u.Hard.Valid == 1);
406  ASSERT (PointerPde->u.Hard.Valid == 1);
407 
408  } while ((PointerPxe->u.Hard.Valid == 0) ||
409  (PointerPpe->u.Hard.Valid == 0) ||
410  (PointerPde->u.Hard.Valid == 0));
411 
412  if (OldIrql != MM_NOIRQL) {
413  LOCK_PFN (OldIrql);
414  }
415 
416  return;
417 }

421 MiMakeSystemAddressValid (
422  IN PVOID PageTableVirtualAddress,
423  IN PEPROCESS CurrentProcess

425 
426 /*++
427 
428 Routine Description:
429 
430  This routine checks to see if the virtual address is valid, and if
431  not makes it valid.
432 
433 Arguments:
434 
435  PageTableVirtualAddress - Supplies the virtual address to make valid - this
436  MUST be a page table hierarchy address because
437  the process working set pushlock is used to
438  synchronize the validity check below.
439 
440  CurrentProcess - Supplies a pointer to the current process.
441 
442 Return Value:
443 
444  Returns TRUE if the working set pushlock was released and wait performed,
445  FALSE otherwise.
446 
447 Environment:
448 
449  Kernel mode, APCs disabled, working set pushlock held.
450 
451 --*/
452 
453 {
454  NTSTATUS status;
455  LOGICAL WsHeldSafe;
456  LOGICAL WsHeldShared;
457  ULONG Waited;
458  PETHREAD Thread;
459 
460  Waited = FALSE;
461  Thread = NULL;
462 
463  ASSERT (PageTableVirtualAddress > MM_HIGHEST_USER_ADDRESS);
464 
465  ASSERT ((PageTableVirtualAddress < MM_PAGED_POOL_START) ||
466  (PageTableVirtualAddress > MmPagedPoolEnd));
467 
468  ASSERT (KeAreAllApcsDisabled () == TRUE);
469 
470  while (!MiIsAddressValid (PageTableVirtualAddress, TRUE)) {
471 
472  //
473  // The virtual address is not present. Release
474  // the working set pushlock and fault it in.
475  //
476  // The working set pushlock may have been acquired safely or unsafely
477  // by our caller. Handle both cases here and below.
478  //
479 
480  if (Thread == NULL) {
481  Thread = PsGetCurrentThread ();
482  }
483 
484  UNLOCK_WS_REGARDLESS (Thread, CurrentProcess, WsHeldSafe, WsHeldShared);
485 
486  status = MmAccessFault (FALSE, PageTableVirtualAddress, KernelMode, NULL);
487 
488  if (!NT_SUCCESS (status)) {
489  KeBugCheckEx (KERNEL_DATA_INPAGE_ERROR,
490  1,
491  (ULONG)status,
492  (ULONG_PTR)CurrentProcess,
493  (ULONG_PTR)PageTableVirtualAddress);
494  }
495 
496  LOCK_WS_REGARDLESS (Thread, CurrentProcess, WsHeldSafe, WsHeldShared);
497 
498  Waited = TRUE;
499  }
500 
501  return Waited;
502 }
