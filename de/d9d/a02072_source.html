54 /*++
55 
56 Routine Description:
57 
58  Do the actual copy of data for a key into caller's buffer.
59 
60  If KeyInformation is not long enough to hold all requested data,
61  STATUS_BUFFER_OVERFLOW will be returned, and ResultLength will be
62  set to the number of bytes actually required.
63 
64 Arguments:
65 
66  Hive - supplies a pointer to the hive control structure for the hive
67 
68  Node - Supplies pointer to node whose subkeys are to be found
69 
70  KeyInformationClass - Specifies the type of information returned in
71  Buffer. One of the following types:
72 
73  KeyBasicInformation - return last write time, title index, and name.
74  (see KEY_BASIC_INFORMATION structure)
75 
76  KeyNodeInformation - return last write time, title index, name, class.
77  (see KEY_NODE_INFORMATION structure)
78 
79  KeyInformation -Supplies pointer to buffer to receive the data.
80 
81  Length - Length of KeyInformation in bytes.
82 
83  ResultLength - Number of bytes actually written into KeyInformation.
84 
85 Return Value:
86 
87  NTSTATUS
88 
89 --*/
90 {
91  NTSTATUS status;
92  PCELL_DATA pclass;
93  ULONG requiredlength;
94  LONG leftlength;
95  ULONG offset;
96  ULONG minimumlength;
97  PKEY_INFORMATION pbuffer;
98  USHORT NameLength;
99 
100  pbuffer = (PKEY_INFORMATION)KeyInformation;
101  NameLength = CmpHKeyNameLen(Node);
102 
103  switch (KeyInformationClass) {
104 
105  case KeyBasicInformation:
106 
107  //
108  // LastWriteTime, TitleIndex, NameLength, Name
109  //
110 
111  requiredlength = FIELD_OFFSET(KEY_BASIC_INFORMATION, Name) +
112  NameLength;
113 
114  minimumlength = FIELD_OFFSET(KEY_BASIC_INFORMATION, Name);
115 
116  *ResultLength = requiredlength;
117 
118  status = STATUS_SUCCESS;
119 
120  if (Length < minimumlength) {
121 
122  status = STATUS_BUFFER_TOO_SMALL;
123 
124  } else {
125 
126  pbuffer->KeyBasicInformation.LastWriteTime =
127  Node->LastWriteTime;
128 
129  pbuffer->KeyBasicInformation.TitleIndex = 0;
130 
131  pbuffer->KeyBasicInformation.NameLength =
132  NameLength;
133 
134  leftlength = Length - minimumlength;
135 
136  requiredlength = NameLength;
137 
138  if (leftlength < (LONG)requiredlength) {
139  requiredlength = leftlength;
140  status = STATUS_BUFFER_OVERFLOW;
141  }
142 
143  if (Node->Flags & KEY_COMP_NAME) {
144  CmpCopyCompressedName(pbuffer->KeyBasicInformation.Name,
145  leftlength,
146  Node->Name,
147  Node->NameLength);
148  } else {
149  RtlCopyMemory(
150  &(pbuffer->KeyBasicInformation.Name[0]),
151  &(Node->Name[0]),
152  requiredlength
153  );
154  }
155  }
156 
157  break;
158 
159 
160  case KeyNodeInformation:
161  //
162  // LastWriteTime, TitleIndex, ClassOffset, ClassLength
163  // NameLength, Name, Class
164  //
165  requiredlength = FIELD_OFFSET(KEY_NODE_INFORMATION, Name) +
166  NameLength +
167  Node->ClassLength;
168 
169  minimumlength = FIELD_OFFSET(KEY_NODE_INFORMATION, Name);
170 
171  *ResultLength = requiredlength;
172 
173  status = STATUS_SUCCESS;
174 
175  if (Length < minimumlength) {
176 
177  status = STATUS_BUFFER_TOO_SMALL;
178 
179  } else {
180 
181  pbuffer->KeyNodeInformation.LastWriteTime =
182  Node->LastWriteTime;
183 
184  pbuffer->KeyNodeInformation.TitleIndex = 0;
185 
186  pbuffer->KeyNodeInformation.ClassLength =
187  Node->ClassLength;
188 
189  pbuffer->KeyNodeInformation.NameLength =
190  NameLength;
191 
192  leftlength = Length - minimumlength;
193  requiredlength = NameLength;
194 
195  if (leftlength < (LONG)requiredlength) {
196  requiredlength = leftlength;
197  status = STATUS_BUFFER_OVERFLOW;
198  }
199 
200  if (Node->Flags & KEY_COMP_NAME) {
201  CmpCopyCompressedName(pbuffer->KeyNodeInformation.Name,
202  leftlength,
203  Node->Name,
204  Node->NameLength);
205  } else {
206  RtlCopyMemory(
207  &(pbuffer->KeyNodeInformation.Name[0]),
208  &(Node->Name[0]),
209  requiredlength
210  );
211  }
212 
213  if (Node->ClassLength > 0) {
214 
215  offset = FIELD_OFFSET(KEY_NODE_INFORMATION, Name) +
216  NameLength;
217  offset = ALIGN_OFFSET(offset);
218 
219  pbuffer->KeyNodeInformation.ClassOffset = offset;
220 
221  pclass = HvGetCell(Hive, Node->Class);
222  if( pclass == NULL ) {
223  //
224  // we couldn't map this cell
225  //
226  status = STATUS_INSUFFICIENT_RESOURCES;
227  break;
228  }
229 
230  pbuffer = (PKEY_INFORMATION)((PUCHAR)pbuffer + offset);
231 
232  leftlength = (((LONG)Length - (LONG)offset) < 0) ?
233  0 :
234  Length - offset;
235 
236  requiredlength = Node->ClassLength;
237 
238  if (leftlength < (LONG)requiredlength) {
239  requiredlength = leftlength;
240  status = STATUS_BUFFER_OVERFLOW;
241  }
242 
243  RtlCopyMemory(
244  pbuffer,
245  pclass,
246  requiredlength
247  );
248 
249  HvReleaseCell(Hive,Node->Class);
250 
251  } else {
252  pbuffer->KeyNodeInformation.ClassOffset = (ULONG)-1;
253  }
254  }
255 
256  break;
257 
258 
259  case KeyFullInformation:
260  //
261  // LastWriteTime, TitleIndex, ClassOffset, ClassLength,
262  // SubKeys, MaxNameLen, MaxClassLen, Values, MaxValueNameLen,
263  // MaxValueDataLen, Class
264  //
265  requiredlength = FIELD_OFFSET(KEY_FULL_INFORMATION, Class) +
266  Node->ClassLength;
267 
268  minimumlength = FIELD_OFFSET(KEY_FULL_INFORMATION, Class);
269 
270  *ResultLength = requiredlength;
271 
272  status = STATUS_SUCCESS;
273 
274  if (Length < minimumlength) {
275 
276  status = STATUS_BUFFER_TOO_SMALL;
277 
278  } else {
279 
280  pbuffer->KeyFullInformation.LastWriteTime =
281  Node->LastWriteTime;
282 
283  pbuffer->KeyFullInformation.TitleIndex = 0;
284 
285  pbuffer->KeyFullInformation.ClassLength =
286  Node->ClassLength;
287 
288  if (Node->ClassLength > 0) {
289 
290  pbuffer->KeyFullInformation.ClassOffset =
291  FIELD_OFFSET(KEY_FULL_INFORMATION, Class);
292 
293  pclass = HvGetCell(Hive, Node->Class);
294  if( pclass == NULL ) {
295  //
296  // we couldn't map this cell
297  //
298  status = STATUS_INSUFFICIENT_RESOURCES;
299  break;
300  }
301 
302  leftlength = Length - minimumlength;
303  requiredlength = Node->ClassLength;
304 
305  if (leftlength < (LONG)requiredlength) {
306  requiredlength = leftlength;
307  status = STATUS_BUFFER_OVERFLOW;
308  }
309 
310  RtlCopyMemory(
311  &(pbuffer->KeyFullInformation.Class[0]),
312  pclass,
313  requiredlength
314  );
315 
316  HvReleaseCell(Hive,Node->Class);
317 
318  } else {
319  pbuffer->KeyFullInformation.ClassOffset = (ULONG)-1;
320  }
321 
322  pbuffer->KeyFullInformation.SubKeys =
323  Node->SubKeyCounts[Stable] +
324  Node->SubKeyCounts[Volatile];
325 
326  pbuffer->KeyFullInformation.Values =
327  Node->ValueList.Count;
328 
329  pbuffer->KeyFullInformation.MaxNameLen =
330  Node->MaxNameLen;
331 
332  pbuffer->KeyFullInformation.MaxClassLen =
333  Node->MaxClassLen;
334 
335  pbuffer->KeyFullInformation.MaxValueNameLen =
336  Node->MaxValueNameLen;
337 
338  pbuffer->KeyFullInformation.MaxValueDataLen =
339  Node->MaxValueDataLen;
340 
341  }
342 
343  break;
344 
345 
346  default:
347  status = STATUS_INVALID_PARAMETER;
348  break;
349  }
350 
351  return status;
352 }

354 NTSTATUS
355 CmpQueryKeyDataFromCache(
356  PCM_KEY_CONTROL_BLOCK Kcb,
357  KEY_INFORMATION_CLASS KeyInformationClass,
358  PVOID KeyInformation,
359  ULONG Length,
360  PULONG ResultLength
361  )
362 /*++
363 
364 Routine Description:
365 
366  Do the actual copy of data for a key into caller's buffer.
367 
368  If KeyInformation is not long enough to hold all requested data,
369  STATUS_BUFFER_OVERFLOW will be returned, and ResultLength will be
370  set to the number of bytes actually required.
371 
372  Works only for the information cached into kcb. I.e. KeyBasicInformation
373  and KeyCachedInfo
374 
375 
376 Arguments:
377 
378  Kcb - Supplies pointer to the kcb to be queried
379 
380  KeyInformationClass - Specifies the type of information returned in
381  Buffer. One of the following types:
382 
383  KeyBasicInformation - return last write time, title index, and name.
384  (see KEY_BASIC_INFORMATION structure)
385 
386  KeyCachedInformation - return last write time, title index, name ....
387  (see KEY_CACHED_INFORMATION structure)
388 
389  KeyInformation -Supplies pointer to buffer to receive the data.
390 
391  Length - Length of KeyInformation in bytes.
392 
393  ResultLength - Number of bytes actually written into KeyInformation.
394 
395 Return Value:
396 
397  NTSTATUS
398 
399 --*/
400 {
401  NTSTATUS status;
402  PKEY_INFORMATION pbuffer;
403  ULONG requiredlength;
404  USHORT NameLength;
405  PCM_KEY_NODE Node; // this is to be used only in case of cache incoherency
406 
407  CM_PAGED_CODE();
408 
409  //
410  // we cannot afford to return the kcb NameBlock as the key name
411  // for KeyBasicInformation as there are lots of callers expecting
412  // the name to be case-sensitive; KeyCachedInformation is new
413  // and used only by the Win32 layer, which is not case sensitive
414  // Note: future clients of KeyCachedInformation must be made aware
415  // that name is NOT case-sensitive
416  //
417  ASSERT( KeyInformationClass == KeyCachedInformation );
418 
419  //
420  // we are going to need the nameblock; if it is NULL, bail out
421  //
422  if( Kcb->NameBlock == NULL ) {
423  return STATUS_INSUFFICIENT_RESOURCES;
424  }
425 
426  pbuffer = (PKEY_INFORMATION)KeyInformation;
427 
428  if (Kcb->NameBlock->Compressed) {
429  NameLength = CmpCompressedNameSize(Kcb->NameBlock->Name,Kcb->NameBlock->NameLength);
430  } else {
431  NameLength = Kcb->NameBlock->NameLength;
432  }
433 
434  // Assume success
435  status = STATUS_SUCCESS;
436 
437  switch (KeyInformationClass) {
438 
439  case KeyCachedInformation:
440 
441  //
442  // LastWriteTime, TitleIndex,
443  // SubKeys, MaxNameLen, Values, MaxValueNameLen,
444  // MaxValueDataLen, Name
445  //
446  requiredlength = sizeof(KEY_CACHED_INFORMATION);
447 
448  *ResultLength = requiredlength;
449 
450  if (Length < requiredlength) {
451 
452  status = STATUS_BUFFER_TOO_SMALL;
453 
454  } else {
455 
456  pbuffer->KeyCachedInformation.LastWriteTime = Kcb->KcbLastWriteTime;
457 
458  pbuffer->KeyCachedInformation.TitleIndex = 0;
459 
460  pbuffer->KeyCachedInformation.NameLength = NameLength;
461 
462  pbuffer->KeyCachedInformation.Values = Kcb->ValueCache.Count;
463 
464  pbuffer->KeyCachedInformation.MaxNameLen = Kcb->KcbMaxNameLen;
465 
466  pbuffer->KeyCachedInformation.MaxValueNameLen = Kcb->KcbMaxValueNameLen;
467 
468  pbuffer->KeyCachedInformation.MaxValueDataLen = Kcb->KcbMaxValueDataLen;
469 
470  if( !(Kcb->ExtFlags & CM_KCB_INVALID_CACHED_INFO) ) {
471  // there is some cached info
472  if( Kcb->ExtFlags & CM_KCB_NO_SUBKEY ) {
473  pbuffer->KeyCachedInformation.SubKeys = 0;
474  } else if( Kcb->ExtFlags & CM_KCB_SUBKEY_ONE ) {
475  pbuffer->KeyCachedInformation.SubKeys = 1;
476  } else if( Kcb->ExtFlags & CM_KCB_SUBKEY_HINT ) {
477  pbuffer->KeyCachedInformation.SubKeys = Kcb->IndexHint->Count;
478  } else {
479  pbuffer->KeyCachedInformation.SubKeys = Kcb->SubKeyCount;
480  }
481  } else {
482  //
483  // kcb cache is not coherent; get the info from knode
484  //
485  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_TRACE_LEVEL,"Kcb cache incoherency detected, kcb = %p\n",Kcb));
486 
487  Node = (PCM_KEY_NODE)HvGetCell(Kcb->KeyHive,Kcb->KeyCell);
488  if( Node == NULL ) {
489  //
490  // couldn't map view for this cell
491  //
492  status = STATUS_INSUFFICIENT_RESOURCES;
493  break;
494  }
495 
496  pbuffer->KeyCachedInformation.SubKeys = Node->SubKeyCounts[Stable] + Node->SubKeyCounts[Volatile];
497  HvReleaseCell(Kcb->KeyHive,Kcb->KeyCell);
498 
499  }
500 
501  }
502 
503  break;
504 
505  default:
506  status = STATUS_INVALID_PARAMETER;
507  break;
508  }
509 
510  return status;
511 }

538

540 
541  KeyValueInformationClass - Specifies the type of information returned in

571 
572  Hive = KeyControlBlock->KeyHive;

574

576  NameLength = CmpValueNameLen(&pcell->u.KeyValue);

581

593

633  case KeyValueFullInformationAlign64:

644

648  base = requiredlength - realsize;

654 #else

663 #endif

667  }

692  }
693 
694 #endif
695 
696  }

699 
700  *status = STATUS_SUCCESS;
701 
702  if (Length < minimumlength) {
703 
704  *status = STATUS_BUFFER_TOO_SMALL;
705 
706  } else {
707

709 
710  pbuffer->KeyValueFullInformation.Type =
711  pcell->u.KeyValue.Type;
712

714  realsize;

729  requiredlength,
730  pcell->u.KeyValue.Name,
731  pcell->u.KeyValue.NameLength);
732  } else {
733  RtlCopyMemory(
734  &(pbuffer->KeyValueFullInformation.Name[0]),
735  &(pcell->u.KeyValue.Name[0]),
736  requiredlength
737  );
738  }
739 
740  if (realsize > 0) {
741 
742  if (small == TRUE) {

747  ASSERT( datapointer == NULL );
748  ASSERT( BufferAllocated == FALSE );

750  }
751  }
752 
753  pbuffer->KeyValueFullInformation.DataOffset = offset;
754 
755  leftlength = (((LONG)Length - (LONG)offset) < 0) ?
756  0 :
757  Length - offset;
758 
759  requiredlength = realsize;
760 
761  if (leftlength < (LONG)requiredlength) {
762  requiredlength = leftlength;
763  *status = STATUS_BUFFER_OVERFLOW;
764  }
765 
766  ASSERT((small ? (requiredlength <= CM_KEY_VALUE_SMALL) : TRUE));
767

769  try {
770  RtlCopyMemory(
771  ((PUCHAR)pbuffer + offset),
772  datapointer,
773  requiredlength
774  );
775  } finally {

778  }
779  if( CellToRelease != HCELL_NIL ) {

781  }
782  }
783  }
784 
785  } else {
786  pbuffer->KeyValueFullInformation.DataOffset = (ULONG)-1;
787  }
788  }
789 
790  break;
791 
792 
793  case KeyValuePartialInformation:
794 
795  //
796  // TitleIndex, Type, DataLength, Data
797  //
798  small = CmpIsHKeyValueSmall(realsize, pcell->u.KeyValue.DataLength);

800  realsize;
801 
802  minimumlength = FIELD_OFFSET(KEY_VALUE_PARTIAL_INFORMATION, Data);
803 
804  *ResultLength = requiredlength;
805 
806  *status = STATUS_SUCCESS;
807 
808  if (Length < minimumlength) {
809

811 
812  } else {
813 
814  pbuffer->KeyValuePartialInformation.TitleIndex = 0;
815 
816  pbuffer->KeyValuePartialInformation.Type =
817  pcell->u.KeyValue.Type;
818 
819  pbuffer->KeyValuePartialInformation.DataLength =
820  realsize;
821 
822  leftlength = Length - minimumlength;
823  requiredlength = realsize;
824 
825  if (leftlength < (LONG)requiredlength) {
826  requiredlength = leftlength;
827  *status = STATUS_BUFFER_OVERFLOW;
828  }
829 
830  if (realsize > 0) {
831
