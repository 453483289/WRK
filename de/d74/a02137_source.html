440 /*++

442 Routine Description:
443 
444  This function creates the Win32 object type descriptors at system
445  initialization and stores the address of the object type descriptor
446  in local static storage.
447 
448 Arguments:
449 
450  None.
451 
452 Return Value:
453 
454  A value of TRUE is returned if the Win32 object type descriptors are
455  successfully created. Otherwise a value of FALSE is returned.
456

458 
459 {
460 
461  OBJECT_TYPE_INITIALIZER ObjectTypeInitializer;
462  NTSTATUS Status;
463  UNICODE_STRING TypeName;
464 
465  //
466  // Initialize string descriptor.
467  //
468 
469  RtlInitUnicodeString(&TypeName, L"WindowStation");
470 
471  //
472  // Create windowstation object type descriptor.
473  //
474 
475  RtlZeroMemory(&ObjectTypeInitializer, sizeof(ObjectTypeInitializer));
476  ObjectTypeInitializer.Length = sizeof(ObjectTypeInitializer);
477  ObjectTypeInitializer.GenericMapping = ExpWindowStationMapping;
478  ObjectTypeInitializer.SecurityRequired = TRUE;
479  ObjectTypeInitializer.PoolType = NonPagedPool;
480 
481  ObjectTypeInitializer.CloseProcedure = ExpWin32CloseProcedure;
482  ObjectTypeInitializer.DeleteProcedure = ExpWin32DeleteProcedure;
483  ObjectTypeInitializer.OkayToCloseProcedure = ExpWin32OkayToCloseProcedure;
484 
485  ObjectTypeInitializer.ParseProcedure = ExpWin32ParseProcedure;
486  ObjectTypeInitializer.OpenProcedure = ExpWin32OpenProcedure;
487 
488  ObjectTypeInitializer.InvalidAttributes = OBJ_OPENLINK |
489  OBJ_PERMANENT |
490  OBJ_EXCLUSIVE;
491  ObjectTypeInitializer.ValidAccessMask = STANDARD_RIGHTS_REQUIRED;
492  Status = ObCreateObjectType(&TypeName,
493  &ObjectTypeInitializer,
494  (PSECURITY_DESCRIPTOR)NULL,
495  &ExWindowStationObjectType);
496 
497  //
498  // If the windowstation object type descriptor was not successfully
499  // created, then return a value of FALSE.
500  //
501 
502  if (!NT_SUCCESS(Status))

504 
505 
506 
507  //
508  // Initialize string descriptor.
509  //

511  RtlInitUnicodeString(&TypeName, L"Desktop");
512 
513  ObjectTypeInitializer.ParseProcedure = NULL; //Desktop has no Parse Procedure
514 
515  //
516  // Create windowstation object type descriptor.
517  //
518 
519  ObjectTypeInitializer.GenericMapping = ExpDesktopMapping;
520  Status = ObCreateObjectType(&TypeName,
521  &ObjectTypeInitializer,
522  (PSECURITY_DESCRIPTOR)NULL,
523  &ExDesktopObjectType);
524 
525 
526  //
527  // If the desktop object type descriptor was successfully created, then
528  // return a value of TRUE. Otherwise return a value of FALSE.
529  //
530 
531  return (BOOLEAN)(NT_SUCCESS(Status));

536 NTSTATUS
537 ExpWin32SessionCallout(
538  IN PKWIN32_CALLOUT CalloutRoutine,
539  IN PKWIN32_CALLOUT_PARAMETERS Parameters,
540  IN ULONG SessionId,
541  OUT PNTSTATUS CalloutStatus OPTIONAL
542  )
543 /*++
544 
545 Routine Description:
546 
547  This routine calls the specified callout routine in session space, for the
548  specified session.
549 
550 Parameters:
551 
552  CalloutRoutine - Callout routine in session space.
553 
554  Parameters - Parameters to pass the callout routine.
555 
556  SessionId - Specifies the ID of the session in which the specified
557  callout routine is to be called.
558 
559  CalloutStatus - Optionally, supplies the address of a variable to receive
560  the NTSTATUS code returned by the callout routine.
561 
562 Return Value:
563 
564  Status code that indicates whether or not the function was successful.
565 
566 Notes:
567 
568  Returns STATUS_NOT_FOUND if the specified session was not found.
569 
570 --*/
571 {
572  NTSTATUS Status, CallStatus;
573  PVOID OpaqueSession;
574  KAPC_STATE ApcState;
575 
576  PAGED_CODE();
577 
578  //
579  // Make sure we have all the information we need to deliver notification.
580  //
581  if (CalloutRoutine == NULL) {
582  return STATUS_INVALID_PARAMETER;
583  }
584 
585  //
586  // Make sure the callout routine in session space.
587  //
588  ASSERT(MmIsSessionAddress((PVOID)CalloutRoutine));
589 
590  if ((PsGetCurrentProcess()->Flags & PS_PROCESS_FLAGS_IN_SESSION) &&
591  (SessionId == PsGetCurrentProcessSessionId())) {
592  //
593  // If the call is from a user mode process, and we are asked to call the
594  // current session, call directly.
595  //
596  CallStatus = (CalloutRoutine)(Parameters);
597 
598  //
599  // Return the callout status.
600  //
601  if (ARGUMENT_PRESENT(CalloutStatus)) {
602  *CalloutStatus = CallStatus;
603  }
604 
605  Status = STATUS_SUCCESS;
606 
607  } else {
608  //
609  // Reference the session object for the specified session.
610  //
611  OpaqueSession = MmGetSessionById(SessionId);
612  if (OpaqueSession == NULL) {
613  return STATUS_NOT_FOUND;
614  }
615 
616  //
617  // Attach to the specified session.
618  //
619  Status = MmAttachSession(OpaqueSession, &ApcState);
620  if (!NT_SUCCESS(Status)) {
621  KdPrintEx((DPFLTR_SYSTEM_ID, DPFLTR_WARNING_LEVEL,
622  "ExpWin32SessionCallout: "
623  "could not attach to 0x%p, session %d for registered notification callout @ 0x%p\n",
624  OpaqueSession,
625  SessionId,
626  CalloutRoutine));
627  MmQuitNextSession(OpaqueSession);
628  return Status;
629  }
630 
631  //
632  // Dispatch notification to the callout routine.
633  //
634  CallStatus = (CalloutRoutine)(Parameters);
635 
636  //
637  // Return the callout status.
638  //
639  if (ARGUMENT_PRESENT(CalloutStatus)) {
640  *CalloutStatus = CallStatus;
641  }
642 
643  //
644  // Detach from the session.
645  //
646  Status = MmDetachSession(OpaqueSession, &ApcState);
647  ASSERT(NT_SUCCESS(Status));
648 
649  //
650  // Dereference the session object.
651  //
652  Status = MmQuitNextSession(OpaqueSession);
653  ASSERT(NT_SUCCESS(Status));
654  }
655 
656  return Status;
657 }
