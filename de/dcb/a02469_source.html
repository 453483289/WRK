693 /*++
694 
695 Routine Description:
696 
697  This function searches the dispatcher ready queues in an attempt to find
698  a thread that can execute on the specified processor.
699 
700  N.B. This routine is called with the sources PRCB locked and the specified
701  PRCB lock held and returns with both locks held.
702 
703  N.B. This routine is only called when it is known that the ready summary
704  for the specified processor is nonzero.
705 
706 Arguments:
707 
708  Number - Supplies the number of the processor to find a thread for.
709 
710  Prcb - Supplies a pointer to the processor control block whose ready
711  queues are to be examined.
712 
713 Return Value:
714 
715  If a thread is located that can execute on the specified processor, then
716  the address of the thread object is returned. Otherwise a null pointer is
717  returned.
718 
719 --*/
720 
721 {
722 
723  ULONG HighPriority;
724  PRLIST_ENTRY ListHead;
725  PRLIST_ENTRY NextEntry;
726  ULONG PrioritySet;
727  PKTHREAD Thread;
728 
729  //
730  // Initialize the set of priority levels that should be scanned in an
731  // attempt to find a thread that can run on the specified processor.
732  //
733 
734  PrioritySet = Prcb->ReadySummary;
735 
736  ASSERT(PrioritySet != 0);
737 
738  KeFindFirstSetLeftMember(PrioritySet, &HighPriority);
739  do {
740 
741  ASSERT((PrioritySet & PRIORITY_MASK(HighPriority)) != 0);
742  ASSERT(IsListEmpty(&Prcb->DispatcherReadyListHead[HighPriority]) == FALSE);
743 
744  ListHead = &Prcb->DispatcherReadyListHead[HighPriority];
745  NextEntry = ListHead->Flink;
746 
747  ASSERT(NextEntry != ListHead);
748 
749  //
750  // Scan the specified dispatcher ready queue for a suitable
751  // thread to execute.
752  //
753  // N.B. It is not necessary to attempt to find a better candidate
754  // on either multinode or non-multinode systems. For multinode
755  // systems, this routine is called sequentially specifying each
756  // processor on the current node before attempting to schedule
757  // from other processors. For non-multinode systems all threads
758  // run on a single node and there is no node distinction. In
759  // both cases threads are inserted in per-processor ready queues
760  // according to their ideal processor.
761  //
762 
763  do {
764  Thread = CONTAINING_RECORD(NextEntry, KTHREAD, WaitListEntry);
765  if ((Thread->Affinity & AFFINITY_MASK(Number)) != 0) {
766 
767  ASSERT((Prcb->ReadySummary & PRIORITY_MASK(HighPriority)) != 0);
768  ASSERT((KPRIORITY)HighPriority == Thread->Priority);
769  ASSERT(Thread->NextProcessor == Prcb->Number);
770 
771  if (RemoveEntryList(&Thread->WaitListEntry) != FALSE) {
772  Prcb->ReadySummary ^= PRIORITY_MASK(HighPriority);
773  }
774 
775  Thread->NextProcessor = (UCHAR)Number;
776  return Thread;
777  }
778 
779  NextEntry = NextEntry->Flink;
780  } while (NextEntry != ListHead);
781 
782  PrioritySet ^= PRIORITY_MASK(HighPriority);
783  KeFindFirstSetLeftMember(PrioritySet, &HighPriority);
784  } while (PrioritySet != 0);
785 
786  //
787  // No thread could be found, return a null pointer.
788  //
789 
790  return NULL;
791 }
