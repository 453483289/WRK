34 /*++
35 
36 Routine Description:
37 
38  Dispatch the request to the appropriate function. It is the worker
39  function's responsibility to appropriately complete the IRP.
40 
41 Arguments:
42 
43  VolumeDeviceObject - Supplies the volume device object to use.
44 
45  Irp - Supplies the Irp being processed
46 
47 Return Value:
48 
49  NTSTATUS - The status for the IRP
50 
51 --*/
52 
53 {
54  NTSTATUS Status;
55  PIO_STACK_LOCATION IrpSp;
56  PVCB Vcb;
57 
58  PAGED_CODE();
59 
60  //
61  // Get a pointer to the current stack location. This location contains
62  // the function codes and parameters for this particular request.
63  //
64 
65  IrpSp = IoGetCurrentIrpStackLocation( Irp );
66 
67  //
68  // Check for operations associated with our FileSystemDeviceObjects
69  // as opposed to our VolumeDeviceObjects. Only mount is allowed to
70  // continue through the normal dispatch in this case.
71  //
72 
73  if ((((PDEVICE_OBJECT)VolumeDeviceObject)->Size == sizeof(DEVICE_OBJECT)) &&
74  !((IrpSp->MajorFunction == IRP_MJ_FILE_SYSTEM_CONTROL) &&
75  (IrpSp->MinorFunction == IRP_MN_MOUNT_VOLUME))) {
76 
77  if ((IrpSp->MajorFunction == IRP_MJ_CREATE) ||
78  (IrpSp->MajorFunction == IRP_MJ_CLEANUP) ||
79  (IrpSp->MajorFunction == IRP_MJ_CLOSE)) {
80 
81  Status = STATUS_SUCCESS;
82 
83  } else {
84 
85  Status = STATUS_INVALID_DEVICE_REQUEST;
86  }
87 
88  RawCompleteRequest( Irp, Status );
89 
90  return Status;
91  }
92 
93  FsRtlEnterFileSystem();
94 
95  //
96  // Get a pointer to the Vcb. Note that is we are mount a volume this
97  // pointer will not have meaning, but that is OK since we will not
98  // use it in that case.
99  //
100 
101  Vcb = &VolumeDeviceObject->Vcb;
102 
103  //
104  // Case on the function that is being performed by the requestor. We
105  // should only see expected requests since we filled the dispatch table
106  // by hand.
107  //
108 
109  try {
110 
111  switch ( IrpSp->MajorFunction ) {
112 
113  case IRP_MJ_CLEANUP:
114 
115  Status = RawCleanup( Vcb, Irp, IrpSp );
116  break;
117 
118  case IRP_MJ_CLOSE:
119 
120  Status = RawClose( Vcb, Irp, IrpSp );
121  break;
122 
123  case IRP_MJ_CREATE:
124 
125  Status = RawCreate( Vcb, Irp, IrpSp );
126  break;
127 
128  case IRP_MJ_FILE_SYSTEM_CONTROL:
129 
130  Status = RawFileSystemControl( Vcb, Irp, IrpSp );
131  break;
132 
133  case IRP_MJ_PNP:
134 
135  if(IrpSp->MinorFunction == IRP_MN_QUERY_REMOVE_DEVICE) {
136  Status = STATUS_DEVICE_BUSY;
137  RawCompleteRequest(Irp, Status);
138  break;
139  }
140 
141  case IRP_MJ_READ:
142  case IRP_MJ_WRITE:
143  case IRP_MJ_DEVICE_CONTROL:
144 
145  Status = RawReadWriteDeviceControl( Vcb, Irp, IrpSp );
146  break;
147 
148  case IRP_MJ_QUERY_INFORMATION:
149 
150  Status = RawQueryInformation( Vcb, Irp, IrpSp );
151  break;
152 
153  case IRP_MJ_SET_INFORMATION:
154 
155  Status = RawSetInformation( Vcb, Irp, IrpSp );
156  break;
157 
158  case IRP_MJ_QUERY_VOLUME_INFORMATION:
159 
160  Status = RawQueryVolumeInformation( Vcb, Irp, IrpSp );
161  break;
162 
163  default:
164 
165  //
166  // We should never get a request we don't expect.
167  //
168 
169  KdPrint(("Raw: Illegal Irp major function code 0x%x.\n", IrpSp->MajorFunction));
170  KeBugCheckEx( FILE_SYSTEM, 0, 0, 0, 0 );
171  }
172 
173  } except( FsRtlIsNtstatusExpected(GetExceptionCode()) ?
174  EXCEPTION_EXECUTE_HANDLER : EXCEPTION_CONTINUE_SEARCH ) {
175 
176  //
177  // No routine we call should ever generate an exception
178  //
179 
180  Status = GetExceptionCode();
181 
182  KdPrint(("Raw: Unexpected exception %X.\n", Status));
183  }
184 
185  //
186  // And return to our caller
187  //
188 
189  FsRtlExitFileSystem();
190 
191  return Status;
192 }

194 //
195 // Completion routine for read, write, and device control to deal with
196 // verify issues. Implemented in RawDisp.c
197 //
198 
199 NTSTATUS
200 RawCompletionRoutine(
201  IN PDEVICE_OBJECT DeviceObject,
202  IN PIRP Irp,
203  IN PVOID Context
204  )
205 
206 {
207  PIO_STACK_LOCATION IrpSp = IoGetCurrentIrpStackLocation( Irp );
208 
209  //
210  // Simply update the file pointer context in the file object if we
211  // were successful and this was a synrchonous read or write.
212  //
213 
214  if (((IrpSp->MajorFunction == IRP_MJ_READ) ||
215  (IrpSp->MajorFunction == IRP_MJ_WRITE)) &&
216  (IrpSp->FileObject != NULL) &&
217  FlagOn(IrpSp->FileObject->Flags, FO_SYNCHRONOUS_IO) &&
218  NT_SUCCESS(Irp->IoStatus.Status)) {
219 
220  IrpSp->FileObject->CurrentByteOffset.QuadPart =
221  IrpSp->FileObject->CurrentByteOffset.QuadPart +
222  Irp->IoStatus.Information;
223  }
224 
225  //
226  // If IoCallDriver returned PENDING, mark our stack location
227  // with pending.
228  //
229 
230  if ( Irp->PendingReturned ) {
231 
232  IoMarkIrpPending( Irp );
233  }
234 
235  UNREFERENCED_PARAMETER( DeviceObject );
236  UNREFERENCED_PARAMETER( Context );
237 
238  return STATUS_SUCCESS;
239 }
