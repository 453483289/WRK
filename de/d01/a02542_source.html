42 MiInsertVadCharges (
43  IN PMMVAD Vad,
44  IN PEPROCESS CurrentProcess

47 /*++
48 
49 Routine Description:
50 
51  This function charges various quotas for the specified virtual address
52  descriptor.
53 
54 Arguments:
55 
56  Vad - Supplies a pointer to a virtual address descriptor.
57 
58  Process - Supplies a pointer to the current process.
59 
60 Return Value:
61 
62  NTSTATUS.
63 
64 Environment:
65 
66  Kernel mode, address creation mutex held, but not the working set pushlock.
67 
68 --*/
69 
70 {
71  ULONG StartBit;
72  ULONG EndBit;
73  SIZE_T RealCharge;
74  SIZE_T PageCharge;
75  SIZE_T PagesReallyCharged;
76  ULONG FirstPage;
77  ULONG LastPage;
78  SIZE_T PagedPoolCharge;
79  LOGICAL ChargedJobCommit;
80  NTSTATUS Status;
81  RTL_BITMAP VadBitMap;
82 #if (_MI_PAGING_LEVELS >= 3)
83  ULONG FirstPdPage;
84  ULONG LastPdPage;
85 #endif
86 #if (_MI_PAGING_LEVELS >= 4)
87  ULONG FirstPpPage;
88  ULONG LastPpPage;
89 #endif
90 
91  ASSERT (Vad->EndingVpn >= Vad->StartingVpn);
92 
93  ASSERT (CurrentProcess == PsGetCurrentProcess ());
94 
95  //
96  // Commit charge of MAX_COMMIT means don't charge quota.
97  //
98 
99  if (Vad->u.VadFlags.CommitCharge != MM_MAX_COMMIT) {
100 
101  PageCharge = 0;
102  PagedPoolCharge = 0;
103  ChargedJobCommit = FALSE;
104 
105  //
106  // Charge quota for the nonpaged pool for the VAD. This is
107  // done here rather than by using ExAllocatePoolWithQuota
108  // so the process object is not referenced by the quota charge.
109  //
110 
111  Status = PsChargeProcessNonPagedPoolQuota (CurrentProcess, sizeof(MMVAD));
112  if (!NT_SUCCESS(Status)) {
113  return STATUS_COMMITMENT_LIMIT;
114  }
115 
116  //
117  // Charge quota for the prototype PTEs if this is a mapped view.
118  //
119 
120  if ((Vad->u.VadFlags.PrivateMemory == 0) &&
121  (Vad->ControlArea != NULL)) {
122 
123  PagedPoolCharge =
124  (Vad->EndingVpn - Vad->StartingVpn + 1) << PTE_SHIFT;
125 
126  Status = PsChargeProcessPagedPoolQuota (CurrentProcess,
127  PagedPoolCharge);
128 
129  if (!NT_SUCCESS(Status)) {
130  PagedPoolCharge = 0;
131  RealCharge = 0;
132  goto Failed;
133  }
134  }
135 
136  //
137  // Add in the charge for page table pages.
138  //
139 
140  FirstPage = MiGetPdeIndex (MI_VPN_TO_VA (Vad->StartingVpn));
141  LastPage = MiGetPdeIndex (MI_VPN_TO_VA (Vad->EndingVpn));
142 
143  while (FirstPage <= LastPage) {
144 
145 #if (_MI_PAGING_LEVELS >= 3)
146  ASSERT (FirstPage < MmWorkingSetList->MaximumUserPageTablePages);
147 #endif
148  if (!MI_CHECK_BIT (MmWorkingSetList->CommittedPageTables,
149  FirstPage)) {
150  PageCharge += 1;
151  }
152  FirstPage += 1;
153  }
154 
155 #if (_MI_PAGING_LEVELS >= 4)
156 
157  //
158  // Add in the charge for page directory parent pages.
159  //
160 
161  FirstPpPage = MiGetPxeIndex (MI_VPN_TO_VA (Vad->StartingVpn));
162  LastPpPage = MiGetPxeIndex (MI_VPN_TO_VA (Vad->EndingVpn));
163 
164  while (FirstPpPage <= LastPpPage) {
165 
166  if (!MI_CHECK_BIT (MmWorkingSetList->CommittedPageDirectoryParents,
167  FirstPpPage)) {
168  PageCharge += 1;
169  }
170  FirstPpPage += 1;
171  }
172 #endif
173 
174 #if (_MI_PAGING_LEVELS >= 3)
175 
176  //
177  // Add in the charge for page directory pages.
178  //
179 
180  FirstPdPage = MiGetPpeIndex (MI_VPN_TO_VA (Vad->StartingVpn));
181  LastPdPage = MiGetPpeIndex (MI_VPN_TO_VA (Vad->EndingVpn));
182 
183  while (FirstPdPage <= LastPdPage) {
184 
185 #if (_MI_PAGING_LEVELS >= 4)
186  ASSERT (FirstPdPage < MmWorkingSetList->MaximumUserPageDirectoryPages);
187 #endif
188  if (!MI_CHECK_BIT (MmWorkingSetList->CommittedPageDirectories,
189  FirstPdPage)) {
190  PageCharge += 1;
191  }
192  FirstPdPage += 1;
193  }
194 #endif
195 
196  RealCharge = Vad->u.VadFlags.CommitCharge + PageCharge;
197 
198  if (RealCharge != 0) {
199 
200  Status = PsChargeProcessPageFileQuota (CurrentProcess, RealCharge);
201  if (!NT_SUCCESS (Status)) {
202  RealCharge = 0;
203  goto Failed;
204  }
205 
206  if (CurrentProcess->CommitChargeLimit) {
207  if (CurrentProcess->CommitCharge + RealCharge > CurrentProcess->CommitChargeLimit) {
208  if (CurrentProcess->Job) {
209  PsReportProcessMemoryLimitViolation ();
210  }
211  goto Failed;
212  }
213  }
214  if (CurrentProcess->JobStatus & PS_JOB_STATUS_REPORT_COMMIT_CHANGES) {
215  if (PsChangeJobMemoryUsage (PS_JOB_STATUS_REPORT_COMMIT_CHANGES, RealCharge) == FALSE) {
216  goto Failed;
217  }
218  ChargedJobCommit = TRUE;
219  }
220 
221  if (MiChargeCommitment (RealCharge, NULL) == FALSE) {
222  goto Failed;
223  }
224 
225  CurrentProcess->CommitCharge += RealCharge;
226  if (CurrentProcess->CommitCharge > CurrentProcess->CommitChargePeak) {
227  CurrentProcess->CommitChargePeak = CurrentProcess->CommitCharge;
228  }
229 
230  MI_INCREMENT_TOTAL_PROCESS_COMMIT (RealCharge);
231 
232  ASSERT (RealCharge == Vad->u.VadFlags.CommitCharge + PageCharge);
233  MM_TRACK_COMMIT (MM_DBG_COMMIT_INSERT_VAD, Vad->u.VadFlags.CommitCharge);
234  MM_TRACK_COMMIT (MM_DBG_COMMIT_INSERT_VAD_PT, PageCharge);
235  }
236 
237  if (PageCharge != 0) {
238 
239  //
240  // Since the commitment was successful, charge the page
241  // table pages.
242  //
243 
244  PagesReallyCharged = 0;
245 
246  FirstPage = MiGetPdeIndex (MI_VPN_TO_VA (Vad->StartingVpn));
247 
248  while (FirstPage <= LastPage) {
249 
250 #if (_MI_PAGING_LEVELS >= 3)
251  ASSERT (FirstPage < MmWorkingSetList->MaximumUserPageTablePages);
252 #endif
253  if (!MI_CHECK_BIT (MmWorkingSetList->CommittedPageTables,
254  FirstPage)) {
255  MI_SET_BIT (MmWorkingSetList->CommittedPageTables,
256  FirstPage);
257  MmWorkingSetList->NumberOfCommittedPageTables += 1;
258 
259  ASSERT32 (MmWorkingSetList->NumberOfCommittedPageTables <
260  PD_PER_SYSTEM * PDE_PER_PAGE);
261  PagesReallyCharged += 1;
262  }
263  FirstPage += 1;
264  }
265 
266 #if (_MI_PAGING_LEVELS >= 3)
267 
268  //
269  // Charge the page directory pages.
270  //
271 
272  FirstPdPage = MiGetPpeIndex (MI_VPN_TO_VA (Vad->StartingVpn));
273 
274  while (FirstPdPage <= LastPdPage) {
275 
276 #if (_MI_PAGING_LEVELS >= 4)
277  ASSERT (FirstPdPage < MmWorkingSetList->MaximumUserPageDirectoryPages);
278 #endif
279  if (!MI_CHECK_BIT (MmWorkingSetList->CommittedPageDirectories,
280  FirstPdPage)) {
281 
282  MI_SET_BIT (MmWorkingSetList->CommittedPageDirectories,
283  FirstPdPage);
284  MmWorkingSetList->NumberOfCommittedPageDirectories += 1;
285 #if (_MI_PAGING_LEVELS == 3)
286  ASSERT (MmWorkingSetList->NumberOfCommittedPageDirectories <
287  PDE_PER_PAGE);
288 #endif
289  PagesReallyCharged += 1;
290  }
291  FirstPdPage += 1;
292  }
293 #endif
294 
295 #if (_MI_PAGING_LEVELS >= 4)
296 
297  //
298  // Charge the page directory parent pages.
299  //
300 
301  FirstPpPage = MiGetPxeIndex (MI_VPN_TO_VA (Vad->StartingVpn));
302 
303  while (FirstPpPage <= LastPpPage) {
304 
305  if (!MI_CHECK_BIT (MmWorkingSetList->CommittedPageDirectoryParents,
306  FirstPpPage)) {
307 
308  MI_SET_BIT (MmWorkingSetList->CommittedPageDirectoryParents,
309  FirstPpPage);
310  MmWorkingSetList->NumberOfCommittedPageDirectoryParents += 1;
311  ASSERT (MmWorkingSetList->NumberOfCommittedPageDirectoryParents <
312  PDE_PER_PAGE);
313  PagesReallyCharged += 1;
314  }
315  FirstPpPage += 1;
316  }
317 #endif
318 
319  ASSERT (PageCharge == PagesReallyCharged);
320  }
321  }
322 
323  //
324  // Set the relevant fields in the Vad bitmap.
325  //
326 
327  StartBit = (ULONG)(((ULONG_PTR) MI_64K_ALIGN (MI_VPN_TO_VA (Vad->StartingVpn))) / X64K);
328  EndBit = (ULONG) (((ULONG_PTR) MI_64K_ALIGN (MI_VPN_TO_VA (Vad->EndingVpn))) / X64K);
329 
330  //
331  // Initialize the bitmap inline for speed.
332  //
333 
334  VadBitMap.SizeOfBitMap = MiLastVadBit + 1;
335  VadBitMap.Buffer = VAD_BITMAP_SPACE;
336 
337  //
338  // Note VADs like the PEB & TEB start on page (not 64K) boundaries so
339  // for these, the relevant bits may already be set.
340  //
341 
342 #if defined (_WIN64) || defined (_X86PAE_)
343  if (EndBit > MiLastVadBit) {
344  EndBit = MiLastVadBit;
345  }
346 
347  //
348  // Only the first (PAGE_SIZE*8*64K) of VA space on NT64 is bitmapped.
349  //
350 
351  if (StartBit <= MiLastVadBit) {
352  RtlSetBits (&VadBitMap, StartBit, EndBit - StartBit + 1);
353  }
354 #else
355  RtlSetBits (&VadBitMap, StartBit, EndBit - StartBit + 1);
356 #endif
357 
358  if (MmWorkingSetList->VadBitMapHint == StartBit) {
359  MmWorkingSetList->VadBitMapHint = EndBit + 1;
360  }
361 
362  if ((CurrentProcess->VadFreeHint != NULL) &&
363  (((ULONG_PTR)((PMMVAD)CurrentProcess->VadFreeHint)->EndingVpn + MI_VA_TO_VPN (X64K)) >= Vad->StartingVpn)) {
364 
365  CurrentProcess->VadFreeHint = Vad;
366  }
367 
368  return STATUS_SUCCESS;
369 
370 Failed:
371 
372  //
373  // Return any quotas charged thus far.
374  //
375 
376  PsReturnProcessNonPagedPoolQuota (CurrentProcess, sizeof(MMVAD));
377 
378  if (PagedPoolCharge != 0) {
379  PsReturnProcessPagedPoolQuota (CurrentProcess, PagedPoolCharge);
380  }
381 
382  if (RealCharge != 0) {
383  PsReturnProcessPageFileQuota (CurrentProcess, RealCharge);
384  }
385 
386  if (ChargedJobCommit == TRUE) {
387  PsChangeJobMemoryUsage (PS_JOB_STATUS_REPORT_COMMIT_CHANGES, -(SSIZE_T)RealCharge);
388  }
389 
390  return STATUS_COMMITMENT_LIMIT;
391 }

394 VOID
395 MiRemoveVadCharges (
396  IN PMMVAD Vad,
397  IN PEPROCESS CurrentProcess
398  )
399 
400 /*++
401 
402 Routine Description:
403 
404  This function removes a virtual address descriptor's commit charges.
405  The VAD remains in the tree.
406 
407 Arguments:
408 
409  Vad - Supplies a pointer to a virtual address descriptor.
410 
411  Process - Supplies a pointer to the current process.
412 
413 Return Value:
414 
415  None.
416 
417 Environment:
418 
419  Kernel mode, address creation mutex held, but not the working set pushlock.
420 
421 --*/
422 
423 {
424  SIZE_T RealCharge;
425  PLIST_ENTRY Next;
426  PMMSECURE_ENTRY Entry;
427 
428  ASSERT (!MM_ANY_WS_LOCK_HELD (PsGetCurrentThread()));
429 
430  ASSERT (CurrentProcess == PsGetCurrentProcess ());
431 
432  //
433  // Commit charge of MAX_COMMIT means quota was not charged.
434  //
435 
436  if (Vad->u.VadFlags.CommitCharge != MM_MAX_COMMIT) {
437 
438  //
439  // Return the quota charge to the process.
440  //
441 
442  PsReturnProcessNonPagedPoolQuota (CurrentProcess, sizeof(MMVAD));
443 
444  if ((Vad->u.VadFlags.PrivateMemory == 0) &&
445  (Vad->ControlArea != NULL)) {
446  PsReturnProcessPagedPoolQuota (CurrentProcess,
447  (Vad->EndingVpn - Vad->StartingVpn + 1) << PTE_SHIFT);
448  }
449 
450  RealCharge = Vad->u.VadFlags.CommitCharge;
451 
452  if (RealCharge != 0) {
453 
454  PsReturnProcessPageFileQuota (CurrentProcess, RealCharge);
455 
456  MiReturnCommitment (RealCharge);
457  MM_TRACK_COMMIT (MM_DBG_COMMIT_RETURN_VAD, RealCharge);
458  if (CurrentProcess->JobStatus & PS_JOB_STATUS_REPORT_COMMIT_CHANGES) {
459  PsChangeJobMemoryUsage (PS_JOB_STATUS_REPORT_COMMIT_CHANGES, -(SSIZE_T)RealCharge);
460  }
461  CurrentProcess->CommitCharge -= RealCharge;
462 
463  MI_INCREMENT_TOTAL_PROCESS_COMMIT (0 - RealCharge);
464  }
465  }
466 
467  if (Vad->u.VadFlags.NoChange) {
468  if (Vad->u2.VadFlags2.MultipleSecured) {
469 
470  //
471  // Free the outstanding pool allocations.
472  //
473 
474  Next = ((PMMVAD_LONG) Vad)->u3.List.Flink;
475  do {
476  Entry = CONTAINING_RECORD( Next,
477  MMSECURE_ENTRY,
478  List);
479 
480  Next = Entry->List.Flink;
481  ExFreePool (Entry);
482  } while (Next != &((PMMVAD_LONG)Vad)->u3.List);
483  }
484  }
485 
486  if (Vad == CurrentProcess->VadFreeHint) {
487  CurrentProcess->VadFreeHint = MiGetPreviousVad (Vad);
488  }
489 
490  return;
491 }

494 NTSTATUS
495 MiFindEmptyAddressRange (
496  IN SIZE_T SizeOfRange,
497  IN ULONG_PTR Alignment,
498  IN ULONG QuickCheck,
499  IN PVOID *Base
500  )
501 
502 /*++
503 
504 Routine Description:
505 
506  The function examines the virtual address descriptors to locate
507  an unused range of the specified size and returns the starting
508  address of the range.
509 
510 Arguments:
511 
512  SizeOfRange - Supplies the size in bytes of the range to locate.
513 
514  Alignment - Supplies the alignment for the address. Must be
515  a power of 2 and greater than the page_size.
516 
517  QuickCheck - Supplies a zero if a quick check for free memory
518  after the VadFreeHint exists, non-zero if checking
519  should start at the lowest address.
520 
521  Base - Receives the starting address of a suitable range on success.
522 
523 Return Value:
524 
525  NTSTATUS.
526 
527 --*/
528 
529 {
530  ULONG FirstBitValue;
531  ULONG StartPosition;
532  ULONG BitsNeeded;
533  PMMVAD NextVad;
534  PMMVAD FreeHint;
535  PEPROCESS CurrentProcess;
536  PVOID StartingVa;
537  PVOID EndingVa;
538  NTSTATUS Status;
539  RTL_BITMAP VadBitMap;
540 
541  CurrentProcess = PsGetCurrentProcess ();
542 
543  if ((QuickCheck == 0) && (Alignment == X64K)) {
544 
545  //
546  // Initialize the bitmap inline for speed.
547  //
548 
549  VadBitMap.SizeOfBitMap = MiLastVadBit + 1;
550  VadBitMap.Buffer = VAD_BITMAP_SPACE;
551 
552  //
553  // Skip the first bit here as we don't generally recommend
554  // that applications map virtual address zero.
555  //
556 
557  FirstBitValue = *((PULONG)VAD_BITMAP_SPACE);
558 
559  *((PULONG)VAD_BITMAP_SPACE) = (FirstBitValue | 0x1);
560 
561  BitsNeeded = (ULONG) ((MI_ROUND_TO_64K (SizeOfRange)) / X64K);
562 
563  StartPosition = RtlFindClearBits (&VadBitMap,
564  BitsNeeded,
565  MmWorkingSetList->VadBitMapHint);
566 
567  if (FirstBitValue & 0x1) {
568  FirstBitValue = (ULONG)-1;
569  }
570  else {
571  FirstBitValue = (ULONG)~0x1;
572  }
573 
574  *((PULONG)VAD_BITMAP_SPACE) &= FirstBitValue;
575 
576  if (StartPosition != NO_BITS_FOUND) {
577  *Base = (PVOID) (((ULONG_PTR)StartPosition) * X64K);
578 #if DBG
579  if (MiCheckForConflictingVad (CurrentProcess, *Base, (ULONG_PTR)*Base + SizeOfRange - 1) != NULL) {
580  DbgPrintEx (DPFLTR_MM_ID, DPFLTR_ERROR_LEVEL,
581  "MiFindEmptyAddressRange: overlapping VAD %p %p\n", *Base, SizeOfRange);
582  DbgBreakPoint ();
583  }
584 #endif
585  return STATUS_SUCCESS;
586  }
587 
588  FreeHint = CurrentProcess->VadFreeHint;
589 
590  if (FreeHint != NULL) {
591 
592  EndingVa = MI_VPN_TO_VA_ENDING (FreeHint->EndingVpn);
593  NextVad = MiGetNextVad (FreeHint);
594 
595  if (NextVad == NULL) {
596 
597  if (SizeOfRange <
598  (((ULONG_PTR)MM_HIGHEST_USER_ADDRESS + 1) -
599  MI_ROUND_TO_SIZE((ULONG_PTR)EndingVa, Alignment))) {
600  *Base = (PVOID) MI_ROUND_TO_SIZE((ULONG_PTR)EndingVa,
601  Alignment);
602  return STATUS_SUCCESS;
603  }
604  }
605  else {
606  StartingVa = MI_VPN_TO_VA (NextVad->StartingVpn);
607 
608  if (SizeOfRange <
609  ((ULONG_PTR)StartingVa -
610  MI_ROUND_TO_SIZE((ULONG_PTR)EndingVa, Alignment))) {
611 
612  //
613  // Check to ensure that the ending address aligned upwards
614  // is not greater than the starting address.
615  //
616 
617  if ((ULONG_PTR)StartingVa >
618  MI_ROUND_TO_SIZE((ULONG_PTR)EndingVa,Alignment)) {
619 
620  *Base = (PVOID)MI_ROUND_TO_SIZE((ULONG_PTR)EndingVa,
621  Alignment);
622  return STATUS_SUCCESS;
623  }
624  }
625  }
626  }
627  }
628 
629  Status = MiFindEmptyAddressRangeInTree (
630  SizeOfRange,
631  Alignment,
632  &CurrentProcess->VadRoot,
633  (PMMADDRESS_NODE *)&CurrentProcess->VadFreeHint,
634  Base);
635 
636  return Status;
637 }

698 PFILE_OBJECT *
699 MmPerfVadTreeWalk (
700  IN PEPROCESS TargetProcess
701  )
702 
703 /*++
704 
705 Routine Description:
706 
707  This routine walks through the VAD tree to find all files mapped
708  into the specified process. It returns a pointer to a pool allocation
709  containing the referenced file object pointers.
710 
711 Arguments:
712 
713  TargetProcess - Supplies the process to walk. Note this is usually NOT
714  the same as the current process.
715 
716 Return Value:
717 
718  Returns a pointer to a NULL terminated pool allocation containing
719  the file object pointers which have been referenced in the process,
720  NULL if the memory could not be allocated.
721 
722  It is also the responsibility of the caller to dereference each
723  file object in the list and then free the returned pool.
724 
725 Environment:
726 
727  PASSIVE_LEVEL, arbitrary thread context.
728 
729 --*/
730 
731 {
732  PMMVAD Vad;
733  ULONG VadCount;
734  PFILE_OBJECT *File;
735  PFILE_OBJECT *FileObjects;
736  PMM_AVL_TABLE Table;
737  PVOID RestartKey;
738  PMMADDRESS_NODE NewNode;
739 
740  ASSERT (KeGetCurrentIrql () == PASSIVE_LEVEL);
741 
742  Table = &TargetProcess->VadRoot;
743  RestartKey = NULL;
744 
745  LOCK_ADDRESS_SPACE (TargetProcess);
746 
747  if (Table->NumberGenericTableElements == 0) {
748  UNLOCK_ADDRESS_SPACE (TargetProcess);
749  return NULL;
750  }
751 
752  //
753  // Allocate one additional entry for the NULL terminator.
754  //
755 
756  VadCount = (ULONG)(Table->NumberGenericTableElements + 1);
757 
758  FileObjects = (PFILE_OBJECT *) ExAllocatePoolWithTag (
759  PagedPool,
760  VadCount * sizeof(PFILE_OBJECT),
761  '01pM');
762 
763  if (FileObjects == NULL) {
764  UNLOCK_ADDRESS_SPACE (TargetProcess);
765  return NULL;
766  }
767 
768  File = FileObjects;
769 
770  do {
771 
772  NewNode = MiEnumerateGenericTableWithoutSplayingAvl (Table,
773  &RestartKey);
774 
775  if (NewNode == NULL) {
776  break;
777  }
778 
779  Vad = (PMMVAD) NewNode;
780 
781  if ((!Vad->u.VadFlags.PrivateMemory) &&
782  (Vad->ControlArea != NULL) &&
783  (Vad->ControlArea->FilePointer != NULL)) {
784 
785  *File = Vad->ControlArea->FilePointer;
786  ObReferenceObject (*File);
787  File += 1;
788  }
789 
790  } while (TRUE);
791 
792  ASSERT (File < FileObjects + VadCount);
793 
794  UNLOCK_ADDRESS_SPACE (TargetProcess);
795 
796  *File = NULL;
797 
798  return FileObjects;
799 }
