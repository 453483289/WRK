46 /*++
47 
48 Routine Description:
49 
50  This function manages the logging of Audit Records. It provides the
51  single interface to the Audit Logging component from the Audit/Alarm
52  generation routines. The function constructs an Audit Record in
53  self-relative format from the information provided and appends it to
54  the Audit Record Queue, a doubly-linked list of Audit Records awaiting
55  output to the Audit Log. A dedicated thread reads this queue, writing
56  Audit Records to the Audit Log and removing them from the Audit Queue.
57 
58 Arguments:
59 
60  AuditEventType - Specifies the type of the Audit Event described by
61  the audit information provided.
62 
63  AuditInformation - Pointer to buffer containing captured auditing
64  information related to an Audit Event of type AuditEventType.
65 
66 Return Value:
67 
68  STATUS_SUCCESS
69  STATUS_UNSUCCESSFUL - Audit record was not queued
70  STATUS_INSUFFICIENT_RESOURCES - unable to allocate heap
71 
72 --*/
73 
74 {
75  NTSTATUS Status;
76  BOOLEAN ForceQueue;
77  PSEP_LSA_WORK_ITEM AuditWorkItem;
78 
79  PAGED_CODE();
80 
81  AuditWorkItem = ExAllocatePoolWithTag( PagedPool, sizeof( SEP_LSA_WORK_ITEM ), 'iAeS' );
82 
83  if ( AuditWorkItem == NULL ) {
84 
85  SepAuditFailed( STATUS_INSUFFICIENT_RESOURCES );
86  return;
87  }
88 
89  AuditWorkItem->Tag = SepAuditRecord;
90  AuditWorkItem->CommandNumber = LsapWriteAuditMessageCommand;
91  AuditWorkItem->ReplyBuffer = NULL;
92  AuditWorkItem->ReplyBufferLength = 0;
93  AuditWorkItem->CleanupFunction = NULL;
94 
95  //
96  // Build an Audit record in self-relative format from the supplied
97  // Audit Information.
98  //
99 
100  Status = SepAdtMarshallAuditRecord(
101  AuditParameters,
102  (PSE_ADT_PARAMETER_ARRAY *) &AuditWorkItem->CommandParams.BaseAddress,
103  &AuditWorkItem->CommandParamsMemoryType
104  );
105 
106  if (NT_SUCCESS(Status)) {
107 
108  //
109  // Extract the length of the Audit Record. Store it as the length
110  // of the Command Parameters buffer.
111  //
112 
113  AuditWorkItem->CommandParamsLength =
114  ((PSE_ADT_PARAMETER_ARRAY) AuditWorkItem->CommandParams.BaseAddress)->Length;
115 
116  //
117  // If we're going to crash on a discarded audit, ignore the queue bounds
118  // check and force the item onto the queue.
119  //
120 
121  if (SepCrashOnAuditFail || AuditParameters->AuditId == SE_AUDITID_AUDITS_DISCARDED) {
122  ForceQueue = TRUE;
123  } else {
124  ForceQueue = FALSE;
125  }
126 
127  if (!SepQueueWorkItem( AuditWorkItem, ForceQueue )) {
128 
129  ExFreePool( AuditWorkItem->CommandParams.BaseAddress );
130  ExFreePool( AuditWorkItem );
131 
132  //
133  // We failed to put the record on the queue. Take whatever action is
134  // appropriate.
135  //
136 
137  SepAuditFailed( STATUS_UNSUCCESSFUL );
138  }
139 
140  } else {
141 
142  ExFreePool( AuditWorkItem );
143  SepAuditFailed( Status );
144  }
145 }

260 NTSTATUS
261 SepAdtMarshallAuditRecord(
262  IN PSE_ADT_PARAMETER_ARRAY AuditParameters,
263  OUT PSE_ADT_PARAMETER_ARRAY *MarshalledAuditParameters,
264  OUT PSEP_RM_LSA_MEMORY_TYPE RecordMemoryType
265  )
266 
267 /*++
268 
269 Routine Description:
270 
271  This routine will take an AuditParameters structure and create
272  a new AuditParameters structure that is suitable for sending
273  to LSA. It will be in self-relative form and allocated as
274  a single chunk of memory.
275 
276 Arguments:
277 
278 
279  AuditParameters - A filled in set of AuditParameters to be marshalled.
280 
281  MarshalledAuditParameters - Returns a pointer to a block of heap memory
282  containing the passed AuditParameters in self-relative form suitable
283  for passing to LSA.
284 
285  RecordMemoryType -- type of memory returned. currently always uses
286  paged pool (returns SepRmPagedPoolMemory)
287 
288 Return Value:
289 
290  NTSTATUS code
291 
292 --*/
293 
294 {
295  ULONG i;
296  ULONG TotalSize = sizeof( SE_ADT_PARAMETER_ARRAY );
297  PUNICODE_STRING TargetString;
298  PCHAR Base;
299  ULONG BaseIncr;
300  ULONG Size;
301  PSE_ADT_PARAMETER_ARRAY_ENTRY pInParam, pOutParam;
302 
303  PAGED_CODE();
304 
305  ASSERT( AuditParameters );
306 
307  ASSERT(IsValidParameterCount(AuditParameters->ParameterCount));
308 
309  //
310  // Calculate the total size required for the passed AuditParameters
311  // block. This calculation will probably be an overestimate of the
312  // amount of space needed, because data smaller that 2 dwords will
313  // be stored directly in the parameters structure, but their length
314  // will be counted here anyway. The overestimate can't be more than
315  // 24 dwords, and will never even approach that amount, so it isn't
316  // worth the time it would take to avoid it.
317  //
318 
319  for (i=0; i<AuditParameters->ParameterCount; i++) {
320  Size = AuditParameters->Parameters[i].Length;
321  TotalSize += PtrAlignSize( Size );
322  }
323 
324  //
325  // Allocate a big enough block of memory to hold everything.
326  // If it fails, quietly abort, since there isn't much else we
327  // can do.
328  //
329 
330  *MarshalledAuditParameters = ExAllocatePoolWithTag( PagedPool, TotalSize, 'pAeS' );
331 
332  if (*MarshalledAuditParameters == NULL) {
333 
334  *RecordMemoryType = SepRmNoMemory;
335  return(STATUS_INSUFFICIENT_RESOURCES);
336  }
337 
338  *RecordMemoryType = SepRmPagedPoolMemory;
339 
340  RtlCopyMemory (
341  *MarshalledAuditParameters,
342  AuditParameters,
343  sizeof( SE_ADT_PARAMETER_ARRAY )
344  );
345 
346  (*MarshalledAuditParameters)->Length = TotalSize;
347  (*MarshalledAuditParameters)->Flags = SE_ADT_PARAMETERS_SELF_RELATIVE;
348 
349  pInParam = &AuditParameters->Parameters[0];
350  pOutParam = &((*MarshalledAuditParameters)->Parameters[0]);
351 
352  //
353  // Start walking down the list of parameters and marshall them
354  // into the target buffer.
355  //
356 
357  Base = (PCHAR) ((PCHAR)(*MarshalledAuditParameters) + sizeof( SE_ADT_PARAMETER_ARRAY ));
358 
359  for (i=0; i<AuditParameters->ParameterCount; i++, pInParam++, pOutParam++) {
360 
361 
362  switch (AuditParameters->Parameters[i].Type) {
363  case SeAdtParmTypeNone:
364  case SeAdtParmTypeUlong:
365  case SeAdtParmTypeHexUlong:
366  case SeAdtParmTypeLogonId:
367  case SeAdtParmTypeLuid:
368  case SeAdtParmTypeNoLogonId:
369  case SeAdtParmTypeTime:
370  case SeAdtParmTypeAccessMask:
371  case SeAdtParmTypePtr:
372  case SeAdtParmTypeDuration:
373  case SeAdtParmTypeHexInt64:
374  case SeAdtParmTypeDateTime:
375  case SeAdtParmTypeMessage:
376  {
377  //
378  // Nothing to do for this
379  //
380 
381  break;
382 
383  }
384  case SeAdtParmTypeString:
385  case SeAdtParmTypeFileSpec:
386  {
387  PUNICODE_STRING SourceString;
388  //
389  // We must copy the body of the unicode string
390  // and then copy the body of the string. Pointers
391  // must be turned into offsets.
392 
393  TargetString = (PUNICODE_STRING)Base;
394 
395  SourceString = pInParam->Address;
396 
397  *TargetString = *SourceString;
398 
399  //
400  // Reset the data pointer in the output parameters to
401  // 'point' to the new string structure.
402  //
403 
404  pOutParam->Address = Base - (ULONG_PTR)(*MarshalledAuditParameters);
405 
406  Base += sizeof( UNICODE_STRING );
407 
408  RtlCopyMemory( Base, SourceString->Buffer, SourceString->Length );
409 
410  //
411  // Make the string buffer in the target string point to where we
412  // just copied the data.
413  //
414 
415  TargetString->Buffer = (PWSTR)(Base - (ULONG_PTR)(*MarshalledAuditParameters));
416 
417  BaseIncr = PtrAlignSize(SourceString->Length);
418 
419  Base += BaseIncr;
420 
421  ASSERT( (ULONG_PTR)Base <= (ULONG_PTR)(*MarshalledAuditParameters) + TotalSize );
422  break;
423  }
424 
425  //
426  // Handle types where we simply copy the buffer.
427  //
428  case SeAdtParmTypePrivs:
429  case SeAdtParmTypeSid:
430  case SeAdtParmTypeObjectTypes:
431  case SeAdtParmTypeSockAddr:
432  case SeAdtParmTypeGuid:
433 
434  {
435  //
436  // Copy the data into the output buffer
437  //
438 
439  RtlCopyMemory( Base, pInParam->Address, pInParam->Length );
440 
441  //
442  // Reset the 'address' of the data to be its offset in the
443  // buffer.
444  //
445 
446  pOutParam->Address = Base - (ULONG_PTR)(*MarshalledAuditParameters);
447 
448  Base += PtrAlignSize( pInParam->Length );
449 
450 
451  ASSERT( (ULONG_PTR)Base <= (ULONG_PTR)(*MarshalledAuditParameters) + TotalSize );
452  break;
453  }
454 
455  default:
456  {
457  //
458  // We got passed junk, complain.
459  //
460 
461  ASSERT( FALSE );
462  break;
463  }
464  }
465  }
466 
467  return( STATUS_SUCCESS );
468 }

569 BOOLEAN
570 SepQueueWorkItem(
571  IN PSEP_LSA_WORK_ITEM LsaWorkItem,
572  IN BOOLEAN ForceQueue
573  )
574 
575 /*++
576 
577 Routine Description:
578 
579  Puts the passed work item on the queue to be passed to LSA,
580  and returns the state of the queue upon arrival.
581 
582 Arguments:
583 
584  LsaWorkItem - Pointer to the work item to be queued.
585 
586  ForceQueue - Indicate that this item is not to be discarded
587  because of a full queue.
588 
589 Return Value:
590 
591  TRUE - The item was successfully queued.
592 
593  FALSE - The item was not queued and must be discarded.
594 
595 --*/
596 
597 {
598  BOOLEAN rc = TRUE;
599  BOOLEAN StartExThread = FALSE ;
600 
601  PAGED_CODE();
602 
603  SepLockLsaQueue();
604 
605  //
606  // See if LSA has died. If it has then just return with an error.
607  //
608  if (SepAdtLsaDeadEvent != NULL) {
609  rc = FALSE;
610  goto Exit;
611  }
612 
613  if (SepAdtDiscardingAudits && !ForceQueue) {
614 
615  if (SepAdtCurrentListLength < SepAdtMinListLength) {
616 
617  //
618  // We need to generate an audit saying how many audits we've
619  // discarded.
620  //
621  // Since we have the mutex protecting the Audit queue, we don't
622  // have to worry about anyone coming along and logging an
623  // audit. But *we* can, since a mutex may be acquired recursively.
624  //
625  // Since we are so protected, turn off the SepAdtDiscardingAudits
626  // flag here so that we don't come through this path again.
627  //
628 
629  SepAdtDiscardingAudits = FALSE;
630 
631  SepAdtGenerateDiscardAudit();
632 
633  //
634  // We must assume that that worked, so clear the discard count.
635  //
636 
637  SepAdtCountEventsDiscarded = 0;
638 
639  //
640  // Our 'audits discarded' audit is now on the queue,
641  // continue logging the one we started with.
642  //
643 
644  } else {
645 
646  //
647  // We are not yet below our low water mark. Toss
648  // this audit and increment the discard count.
649  //
650 
651  SepAdtCountEventsDiscarded++;
652  rc = FALSE;
653  goto Exit;
654  }
655  }
656 
657  if (SepAdtCurrentListLength < SepAdtMaxListLength || ForceQueue) {
658 
659  InsertTailList(&SepLsaQueue, &LsaWorkItem->List);
660 
661  if (++SepAdtCurrentListLength == 1) {
662 
663  StartExThread = TRUE ;
664  }
665 
666  } else {
667 
668  //
669  // There is no room for this audit on the queue,
670  // so change our state to 'discarding' and tell
671  // the caller to toss this audit.
672  //
673 
674  SepAdtDiscardingAudits = TRUE;
675 
676  rc = FALSE;
677  }
678 
679 Exit:
680 
681  SepUnlockLsaQueue();
682 
683  if ( StartExThread )
684  {
685  ExInitializeWorkItem( &SepExWorkItem.WorkItem,
686  (PWORKER_THREAD_ROUTINE) SepRmCallLsa,
687  &SepExWorkItem
688  );
689 
690  ExQueueWorkItem( &SepExWorkItem.WorkItem, DelayedWorkQueue );
691  }
692 
693  return( rc );
694 }

703 /*++
704 
705 Routine Description:
706 
707  Removes the top element of the SepLsaQueue and returns the
708  next element if there is one, NULL otherwise.
709 
710 Arguments:
711 
712  None.
713 
714 Return Value:
715 
716  A pointer to the next SEP_LSA_WORK_ITEM, or NULL.
717 
718 --*/
719 
720 {
721  PSEP_LSA_WORK_ITEM OldWorkQueueItem;
722 
723  PAGED_CODE();
724 
725  SepLockLsaQueue();
726 
727  OldWorkQueueItem = (PSEP_LSA_WORK_ITEM)RemoveHeadList(&SepLsaQueue);
728  OldWorkQueueItem->List.Flink = NULL;
729 
730  SepAdtCurrentListLength--;
731 
732  if (IsListEmpty( &SepLsaQueue )) {

735  //

748  //
749

751 
752  ExFreePool( OldWorkQueueItem );
753 
754  return((PSEP_LSA_WORK_ITEM)(&SepLsaQueue)->Flink);
755 }
