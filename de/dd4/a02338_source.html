146 /*++
147 
148 Routine Description:
149 
150  This function processes a nonmaskable interrupt (NMI).
151 
152  N.B. This function is called from the NMI trap routine with interrupts
153  disabled.
154 
155 Arguments:
156 
157  TrapFrame - Supplies a pointer to a trap frame.
158 
159  ExceptionFrame - Supplies a pointer to an exception frame.
160 
161 Return Value:
162 
163  None.
164 
165 --*/
166 
167 {
168  static LONG64 NmiInProgress = 0;
169  ULONG Number;
170 
171  //
172  // First check for a pending freeze execution request from another
173  // processor.
174  //
175 
176 #if !defined(NT_UP)
177 
178  if (KiCheckForFreezeExecution(TrapFrame, ExceptionFrame) != FALSE) {
179  return;
180  }
181 
182 #endif
183 
184  //
185  // Return immediately if this processor is already processing an NMI.
186  //
187 
188  Number = KeGetCurrentPrcb()->Number;
189  if (InterlockedBitTestAndSet64(&NmiInProgress, Number) != FALSE) {
190  return;
191  }
192 
193  //
194  // Process NMI callback functions.
195  //
196  // If no callback function handles the NMI, then let the HAL handle it.
197  //
198 
199  if (KiHandleNmi() == FALSE) {
200  KiAcquireSpinLockCheckForFreeze(&KiNMILock, TrapFrame, ExceptionFrame);
201  KeBugCheckActive = TRUE;
202  HalHandleNMI(NULL);
203  KeBugCheckActive = FALSE;
204  KeReleaseSpinLockFromDpcLevel(&KiNMILock);
205  }
206 
207  InterlockedBitTestAndReset64(&NmiInProgress, Number);
208 
209  return;
210 }
