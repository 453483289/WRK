30  __in HANDLE ProcessHandle,
31  __in PVOID BaseAddress
32  )
33 
34 /*++
35 
36 Routine Description:
37 
38  This function unmaps a previously created view to a section.
39 
40 Arguments:
41 
42  ProcessHandle - Supplies an open handle to a process object.
43 
44  BaseAddress - Supplies the base address of the view.
45 
46 Return Value:
47 
48  NTSTATUS.
49 
50 --*/
51 
52 {
53  PEPROCESS Process;
54  KPROCESSOR_MODE PreviousMode;
55  NTSTATUS Status;
56 
57  PAGED_CODE ();
58 
59  PreviousMode = KeGetPreviousMode ();
60 
61  if ((PreviousMode == UserMode) && (BaseAddress > MM_HIGHEST_USER_ADDRESS)) {
62  return STATUS_NOT_MAPPED_VIEW;
63  }
64 
65  Status = ObReferenceObjectByHandle (ProcessHandle,
66  PROCESS_VM_OPERATION,
67  PsProcessType,
68  PreviousMode,
69  (PVOID *)&Process,
70  NULL);
71 
72  if (!NT_SUCCESS (Status)) {
73  return Status;
74  }
75 
76  Status = MiUnmapViewOfSection (Process, BaseAddress, 0);
77  ObDereferenceObject (Process);
78

82 NTSTATUS
83 MiUnmapViewOfSection (
84  IN PEPROCESS Process,
85  IN PVOID BaseAddress,
86  IN ULONG Flags
87  )
88 
89 /*++
90 
91 Routine Description:
92 
93  This function unmaps a previously created view to a section.
94 
95 Arguments:
96 
97  Process - Supplies a referenced pointer to a process object.
98 
99  BaseAddress - Supplies the base address of the view.
100 
101  Flags - Supplies a bitfield directive.
102 
103 Return Value:
104 
105  NTSTATUS.
106 
107 --*/
108 
109 {
110  PMMVAD Vad;
111  PMMVAD PreviousVad;
112  PMMVAD NextVad;
113  PETHREAD Thread;
114  PEPROCESS CurrentProcess;
115  SIZE_T RegionSize;
116  PVOID UnMapImageBase;
117  PVOID StartingVa;
118  PVOID EndingVa;
119  NTSTATUS status;
120  LOGICAL Attached;
121  KAPC_STATE ApcState;
122 
123  PAGED_CODE();
124 
125  Attached = FALSE;
126  UnMapImageBase = NULL;
127 
128  Thread = PsGetCurrentThread ();
129  CurrentProcess = PsGetCurrentProcessByThread (Thread);
130 
131  //
132  // If the specified process is not the current process, attach
133  // to the specified process.
134  //
135 
136  if (CurrentProcess != Process) {
137  KeStackAttachProcess (&Process->Pcb, &ApcState);
138  Attached = TRUE;
139  }
140 
141  //
142  // Get the address creation mutex to block multiple threads from
143  // creating or deleting address space at the same time and
144  // get the working set mutex so virtual address descriptors can
145  // be removed. Raise IRQL to block APCs.
146  //
147 
148  if ((Flags & UNMAP_ADDRESS_SPACE_HELD) == 0) {
149  LOCK_ADDRESS_SPACE (Process);
150  }
151 
152  //
153  // Make sure the address space was not deleted, if so, return an error.
154  //
155 
156  if (Process->Flags & PS_PROCESS_FLAGS_VM_DELETED) {
157  if ((Flags & UNMAP_ADDRESS_SPACE_HELD) == 0) {
158  UNLOCK_ADDRESS_SPACE (Process);
159  }
160  status = STATUS_PROCESS_IS_TERMINATING;
161  goto ErrorReturn;
162  }
163 
164  //
165  // Find the associated vad.
166  //
167 
168  Vad = MiLocateAddress (BaseAddress);
169 
170  if ((Vad == NULL) || (Vad->u.VadFlags.PrivateMemory)) {
171 
172  //
173  // No Virtual Address Descriptor located for Base Address.
174  //
175 
176  if ((Flags & UNMAP_ADDRESS_SPACE_HELD) == 0) {
177  UNLOCK_ADDRESS_SPACE (Process);
178  }
179  status = STATUS_NOT_MAPPED_VIEW;
180  goto ErrorReturn;
181  }
182 
183  StartingVa = MI_VPN_TO_VA (Vad->StartingVpn);
184  EndingVa = MI_VPN_TO_VA_ENDING (Vad->EndingVpn);
185 
186  //
187  // If this Vad is for an image section, then
188  // get the base address of the section.
189  //
190 
191  ASSERT (Process == PsGetCurrentProcess ());
192 
193  if (Vad->u.VadFlags.VadType == VadImageMap) {
194  UnMapImageBase = StartingVa;
195  }
196 
197  RegionSize = PAGE_SIZE + ((Vad->EndingVpn - Vad->StartingVpn) << PAGE_SHIFT);
198 
199  if (Vad->u.VadFlags.NoChange == 1) {
200 
201  //
202  // An attempt is being made to delete a secured VAD, check
203  // the whole VAD to see if this deletion is allowed.
204  //
205 
206  status = MiCheckSecuredVad (Vad,
207  StartingVa,
208  RegionSize,
209  MM_SECURE_DELETE_CHECK);
210 
211  if (!NT_SUCCESS (status)) {
212  if ((Flags & UNMAP_ADDRESS_SPACE_HELD) == 0) {
213  UNLOCK_ADDRESS_SPACE (Process);
214  }
215  goto ErrorReturn;
216  }
217  }
218 
219  PreviousVad = MiGetPreviousVad (Vad);
220  NextVad = MiGetNextVad (Vad);
221 
222  if (Vad->u.VadFlags.VadType == VadRotatePhysical) {
223 
224  if ((Flags & UNMAP_ROTATE_PHYSICAL_OK) == 0) {
225 
226  //
227  // This caller is treating the VA space via the section APIs,
228  // but he's not supposed to know that internally we converted
229  // the private memory request into a section. So fail this.
230  //
231 
232  if ((Flags & UNMAP_ADDRESS_SPACE_HELD) == 0) {
233  UNLOCK_ADDRESS_SPACE (Process);
234  }
235  status = STATUS_NOT_MAPPED_VIEW;
236  goto ErrorReturn;
237  }
238 
239  MiRemoveVadCharges (Vad, Process);
240 
241  LOCK_WS_UNSAFE (Thread, Process);
242 
243  MiPhysicalViewRemover (Process, Vad);
244  }
245  else {
246  MiRemoveVadCharges (Vad, Process);
247 
248  LOCK_WS_UNSAFE (Thread, Process);
249  }
250 
251  MiRemoveVad (Vad, Process);
252 
253  //
254  // NOTE: MiRemoveMappedView releases the working set pushlock
255  // before returning.
256  //
257 
258  MiRemoveMappedView (Process, Vad);
259 
260  //
261  // Return commitment for page table pages if possible.
262  //
263 
264  MiReturnPageTablePageCommitment (StartingVa,
265  EndingVa,
266  Process,
267  PreviousVad,
268  NextVad);
269 
270  //
271  // Update the current virtual size in the process header.
272  //
273 
274  Process->VirtualSize -= RegionSize;
275  if ((Flags & UNMAP_ADDRESS_SPACE_HELD) == 0) {
276  UNLOCK_ADDRESS_SPACE (Process);
277  }
278 
279  ExFreePool (Vad);
280  status = STATUS_SUCCESS;
281 
282 ErrorReturn:
283 
284  if (UnMapImageBase) {
285  DbgkUnMapViewOfSection (UnMapImageBase);
286  }
287  if (Attached == TRUE) {
288  KeUnstackDetachProcess (&ApcState);
289  }
290 
291  return status;
292 }

322 VOID
323 MiDecrementSubsections (
324  IN PSUBSECTION FirstSubsection,
325  IN PSUBSECTION LastSubsection OPTIONAL
326  )
327 /*++
328 
329 Routine Description:
330 
331  This function decrements the subsections, inserting them on the unused
332  subsection list if they qualify.
333 
334 Arguments:
335 
336  FirstSubsection - Supplies the subsection to start at.
337 
338  LastSubsection - Supplies the last subsection to insert. Supplies NULL
339  to decrement all the subsections in the chain.
340 
341 Return Value:
342 
343  None.
344 
345 Environment:
346 
347  PFN lock held.
348 
349 --*/
350 {
351  PMSUBSECTION MappedSubsection;
352 
353  ASSERT ((FirstSubsection->ControlArea->u.Flags.Image == 0) &&
354  (FirstSubsection->ControlArea->FilePointer != NULL) &&
355  (FirstSubsection->ControlArea->u.Flags.PhysicalMemory == 0));
356 
357  MM_PFN_LOCK_ASSERT();
358 
359  do {
360  MappedSubsection = (PMSUBSECTION) FirstSubsection;
361 
362  ASSERT (MappedSubsection->DereferenceList.Flink == NULL);
363 
364  ASSERT (((LONG_PTR)MappedSubsection->NumberOfMappedViews >= 1) ||
365  (MappedSubsection->u.SubsectionFlags.SubsectionStatic == 1));
366 
367  MappedSubsection->NumberOfMappedViews -= 1;
368 
369  if ((MappedSubsection->NumberOfMappedViews == 0) &&
370  (MappedSubsection->u.SubsectionFlags.SubsectionStatic == 0)) {
371 
372  //
373  // Insert this subsection into the unused subsection list.
374  //
375 
376  InsertTailList (&MmUnusedSubsectionList,
377  &MappedSubsection->DereferenceList);
378 
379  MI_UNUSED_SUBSECTIONS_COUNT_INSERT (MappedSubsection);
380  }
381 
382  if (ARGUMENT_PRESENT (LastSubsection)) {
383  if (FirstSubsection == LastSubsection) {
384  break;
385  }
386  }
387  else {
388  if (FirstSubsection->NextSubsection == NULL) {
389  break;
390  }
391  }
392 
393  FirstSubsection = FirstSubsection->NextSubsection;
394  } while (TRUE);
395 }

398 VOID
399 MiRemoveMappedView (
400  IN PEPROCESS CurrentProcess,
401  IN PMMVAD Vad
402  )
403 
404 /*++
405 
406 Routine Description:
407 
408  This function removes the mapping from the current process's
409  address space. The physical VAD may be a normal mapping (backed by
410  a control area) or it may have no control area (it was mapped by a driver).
411 
412 Arguments:
413 
414  CurrentProcess - Supplies a referenced pointer to the current process object.
415 
416  Vad - Supplies the VAD which maps the view.
417 
418 Return Value:
419 
420  None.
421 
422 Environment:
423 
424  APC level, working set mutex pushlock and address creation mutex held.
425 
426  NOTE: THE WORKING SET PUSHLOCK IS RELEASED BEFORE RETURNING !!!!
427 
428  The working set pushlock must be acquired UNSAFE by the caller.
429 
430 --*/
431 
432 {
433  KIRQL OldIrql;
434  PCONTROL_AREA ControlArea;
435  PMMPTE PointerPte;
436  PMMPTE PointerPde;
437  PMMPTE ProtoPte;
438  PMMPTE LastPte;
439  PFN_NUMBER PdePage;
440  PVOID TempVa;
441  MMPTE_FLUSH_LIST PteFlushList;
442  PVOID UsedPageTableHandle;
443  PMMPFN Pfn2;
444  PSUBSECTION FirstSubsection;
445  PSUBSECTION LastSubsection;
446  PMMPFN Pfn1;
447  PMMPFN EndPfn;
448  MMPTE TempPte;
449  PFN_NUMBER PageFrameIndex;
450  PMMPTE EndPde;
451  PETHREAD CurrentThread;
452 #if DBG
453  PMMPTE EndProto;
454 #endif
455 #if (_MI_PAGING_LEVELS >= 3)
456  PMMPTE PointerPpe;
457  PVOID UsedPageDirectoryHandle;
458 #endif
459 #if (_MI_PAGING_LEVELS >= 4)
460  PMMPTE PointerPxe;
461  PVOID UsedPageDirectoryParentHandle;
462 #endif
463 
464  ControlArea = Vad->ControlArea;
465  CurrentThread = PsGetCurrentThread ();
466 
467  if (Vad->u.VadFlags.VadType == VadDevicePhysicalMemory) {
468 
469  if (((PMMVAD_LONG)Vad)->u4.Banked != NULL) {
470  ExFreePool (((PMMVAD_LONG)Vad)->u4.Banked);
471  }
472 
473  //
474  // This is a physical memory view. The pages map physical memory
475  // and are not accounted for in the working set list or in the PFN
476  // database.
477  //
478 
479  MiPhysicalViewRemover (CurrentProcess, Vad);
480 
481  //
482  // Set count so only flush entire TB operations are performed.
483  //
484 
485  PteFlushList.Count = MM_MAXIMUM_FLUSH_COUNT;
486 
487  PointerPde = MiGetPdeAddress (MI_VPN_TO_VA (Vad->StartingVpn));
488  PointerPte = MiGetPteAddress (MI_VPN_TO_VA (Vad->StartingVpn));
489  LastPte = MiGetPteAddress (MI_VPN_TO_VA (Vad->EndingVpn));
490 
491  ASSERT (PointerPde->u.Hard.Valid == 1);
492 
493  if (MI_PDE_MAPS_LARGE_PAGE (PointerPde)) {
494 
495  MiFreeLargePages (MI_VPN_TO_VA (Vad->StartingVpn),
496  MI_VPN_TO_VA_ENDING (Vad->EndingVpn),
497  TRUE);
498 
499  UNLOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
500  LOCK_PFN (OldIrql);
501  }
502  else {
503 
504  //
505  // Remove the PTES from the address space.
506  //
507 
508  PdePage = MI_GET_PAGE_FRAME_FROM_PTE (PointerPde);
509 
510  UsedPageTableHandle = MI_GET_USED_PTES_HANDLE (MI_VPN_TO_VA (Vad->StartingVpn));
511 
512  LOCK_PFN (OldIrql);
513 
514  while (PointerPte <= LastPte) {
515 
516  if (MiIsPteOnPdeBoundary (PointerPte)) {
517 
518  PointerPde = MiGetPteAddress (PointerPte);
519  PdePage = MI_GET_PAGE_FRAME_FROM_PTE (PointerPde);
520 
521  UsedPageTableHandle = MI_GET_USED_PTES_HANDLE (MiGetVirtualAddressMappedByPte (PointerPte));
522  }
523 
524  //
525  // Decrement the count of non-zero page table entries for this
526  // page table.
527  //
528 
529  MI_DECREMENT_USED_PTES_BY_HANDLE (UsedPageTableHandle);
530 
531  MI_WRITE_ZERO_PTE (PointerPte);
532 
533  Pfn2 = MI_PFN_ELEMENT (PdePage);
534  MiDecrementShareCountInline (Pfn2, PdePage);
535 
536  //
537  // If all the entries have been eliminated from the previous
538  // page table page, delete the page table page itself. And if
539  // this results in an empty page directory page, then delete
540  // that too.
541  //
542 
543  if (MI_GET_USED_PTES_FROM_HANDLE(UsedPageTableHandle) == 0) {
544 
545  TempVa = MiGetVirtualAddressMappedByPte(PointerPde);
546 
547  PteFlushList.Count = MM_MAXIMUM_FLUSH_COUNT;
548 
549 #if (_MI_PAGING_LEVELS >= 3)
550  UsedPageDirectoryHandle = MI_GET_USED_PTES_HANDLE (PointerPte);
551 
552  MI_DECREMENT_USED_PTES_BY_HANDLE (UsedPageDirectoryHandle);
553 #endif
554 
555  MiDeletePte (PointerPde,
556  TempVa,
557  FALSE,
558  CurrentProcess,
559  NULL,
560  &PteFlushList,
561  OldIrql);
562 
563  //
564  // Add back in the private page MiDeletePte subtracted.
565  //
566 
567  CurrentProcess->NumberOfPrivatePages += 1;
568 
569 #if (_MI_PAGING_LEVELS >= 3)
570 
571  if (MI_GET_USED_PTES_FROM_HANDLE(UsedPageDirectoryHandle) == 0) {
572 
573  PointerPpe = MiGetPdeAddress (PointerPte);
574  TempVa = MiGetVirtualAddressMappedByPte (PointerPpe);
575 
576  PteFlushList.Count = MM_MAXIMUM_FLUSH_COUNT;
577 
578 #if (_MI_PAGING_LEVELS >= 4)
579  UsedPageDirectoryParentHandle = MI_GET_USED_PTES_HANDLE (PointerPde);
580 
581  MI_DECREMENT_USED_PTES_BY_HANDLE (UsedPageDirectoryParentHandle);
582 #endif
583 
584  MiDeletePte (PointerPpe,
585  TempVa,
586  FALSE,
587  CurrentProcess,
588  NULL,
589  &PteFlushList,
590  OldIrql);
591 
592  //
593  // Add back in the private page MiDeletePte subtracted.
594  //
595 
596  CurrentProcess->NumberOfPrivatePages += 1;
597 
598 #if (_MI_PAGING_LEVELS >= 4)
599 
600  if (MI_GET_USED_PTES_FROM_HANDLE(UsedPageDirectoryParentHandle) == 0) {
601 
602  PointerPxe = MiGetPpeAddress(PointerPte);
603  TempVa = MiGetVirtualAddressMappedByPte(PointerPxe);
604 
605  PteFlushList.Count = MM_MAXIMUM_FLUSH_COUNT;
606 
607  MiDeletePte (PointerPxe,
608  TempVa,
609  FALSE,
610  CurrentProcess,
611  NULL,
612  &PteFlushList,
613  OldIrql);
614 
615  //
616  // Add back in the private page MiDeletePte subtracted.
617  //
618 
619  CurrentProcess->NumberOfPrivatePages += 1;
620  }
621 #endif
622 
623  }
624 #endif
625  }
626  PointerPte += 1;
627  }
628  MI_FLUSH_PROCESS_TB (FALSE);
629  UNLOCK_PFN (OldIrql);
630  UNLOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
631  LOCK_PFN (OldIrql);
632  }
633  }
634  else {
635 
636  if (Vad->u2.VadFlags2.ExtendableFile) {
637  PMMEXTEND_INFO ExtendedInfo;
638  PMMVAD_LONG VadLong;
639 
640  //
641  // Release the working set pushlock before referencing the
642  // segment since it is pageable.
643  //
644 
645  UNLOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
646 
647  ExtendedInfo = NULL;
648  VadLong = (PMMVAD_LONG) Vad;
649 
650  KeAcquireGuardedMutexUnsafe (&MmSectionBasedMutex);
651  ASSERT (Vad->ControlArea->Segment->ExtendInfo == VadLong->u4.ExtendedInfo);
652  VadLong->u4.ExtendedInfo->ReferenceCount -= 1;
653  if (VadLong->u4.ExtendedInfo->ReferenceCount == 0) {
654  ExtendedInfo = VadLong->u4.ExtendedInfo;
655  VadLong->ControlArea->Segment->ExtendInfo = NULL;
656  }
657  KeReleaseGuardedMutexUnsafe (&MmSectionBasedMutex);
658  if (ExtendedInfo != NULL) {
659  ExFreePool (ExtendedInfo);
660  }
661 
662  LOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
663  }
664 
665  FirstSubsection = NULL;
666  LastSubsection = NULL;
667 
668  if (Vad->u.VadFlags.VadType != VadImageMap) {
669 
670  if (ControlArea->FilePointer != NULL) {
671 
672  if ((Vad->u.VadFlags.Protection == MM_READWRITE) ||
673  (Vad->u.VadFlags.Protection == MM_EXECUTE_READWRITE)) {
674 
675  //
676  // Adjust the count of writable user mappings
677  // to support transactions.
678  //
679 
680  InterlockedDecrement ((PLONG)&ControlArea->WritableUserReferences);
681  }
682 
683  FirstSubsection = MiLocateSubsection (Vad, Vad->StartingVpn);
684 
685  ASSERT (FirstSubsection != NULL);
686 
687  //
688  // Note LastSubsection may be NULL for extendable VADs when the
689  // EndingVpn is past the end of the section. In this case,
690  // all the subsections can be safely decremented.
691  //
692 
693  LastSubsection = MiLocateSubsection (Vad, Vad->EndingVpn);
694  }
695  }
696  else {
697 
698  //
699  // Handle this specially if it is an image view using large pages.
700  //
701 
702  TempVa = MI_VPN_TO_VA (Vad->StartingVpn);
703  PointerPde = MiGetPdeAddress (TempVa);
704 
705  if (
706 #if (_MI_PAGING_LEVELS>=4)
707  (MiGetPxeAddress(TempVa)->u.Hard.Valid == 1) &&
708 #endif
709 #if (_MI_PAGING_LEVELS>=3)
710  (MiGetPpeAddress(TempVa)->u.Hard.Valid == 1) &&
711 #endif
712  (PointerPde->u.Hard.Valid == 1) &&
713  (MI_PDE_MAPS_LARGE_PAGE (PointerPde))) {
714 
715  MiFreeLargePages (TempVa,
716  MI_VPN_TO_VA_ENDING (Vad->EndingVpn),
717  FALSE);
718 
719  UNLOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
720 
721  MiReleasePhysicalCharges (Vad->EndingVpn - Vad->StartingVpn + 1,
722  CurrentProcess);
723 
724  //
725  // Decrement the count of the number of views for the
726  // control area. This requires the PFN lock to be held.
727  //
728 
729  LOCK_PFN (OldIrql);
730 
731  ControlArea->NumberOfMappedViews -= 1;
732  ControlArea->NumberOfUserReferences -= 1;
733 
734  //
735  // Check to see if the control area should be deleted.
736  // This routine releases the PFN lock.
737  //
738 
739  MiCheckControlArea (ControlArea, OldIrql);
740 
741  return;
742  }
743  }
744 
745  if (Vad->u.VadFlags.VadType == VadLargePageSection) {
746 
747  PointerPde = MiGetPdeAddress (MI_VPN_TO_VA (Vad->StartingVpn));
748  EndPde = MiGetPdeAddress (MI_VPN_TO_VA_ENDING (Vad->EndingVpn));
749 
750  ControlArea = Vad->ControlArea;
751 
752 #if DBG
753  //
754  // Release the working set pushlock before referencing the
755  // segment since it is pageable.
756  //
757 
758  UNLOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
759  ASSERT (ControlArea->Segment->SegmentFlags.LargePages == 1);
760  LOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
761 #endif
762 
763  ProtoPte = Vad->FirstPrototypePte;
764 
765  do {
766  TempPte = *PointerPde;
767  ASSERT (TempPte.u.Hard.Valid == 1);
768  ASSERT (MI_PDE_MAPS_LARGE_PAGE (&TempPte));
769 
770  PageFrameIndex = MI_GET_PAGE_FRAME_FROM_PTE (&TempPte);
771  Pfn1 = MI_PFN_ELEMENT (PageFrameIndex);
772  EndPfn = Pfn1 + (MM_VA_MAPPED_BY_PDE >> PAGE_SHIFT);
773 
774  ASSERT (PageFrameIndex % (MM_VA_MAPPED_BY_PDE >> PAGE_SHIFT) == 0);
775 
776 #if DBG
777  //
778  // Release the working set pushlock before referencing the
779  // prototype PTEs since they are pageable.
780  //
781 
782  UNLOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
783 
784  ASSERT (ProtoPte->u.Hard.Valid == 1);
785  ASSERT (PageFrameIndex == MI_GET_PAGE_FRAME_FROM_PTE (ProtoPte));
786  EndProto = ProtoPte + (MM_VA_MAPPED_BY_PDE >> PAGE_SHIFT);
787

789  TempPte = *ProtoPte;
790  ASSERT (TempPte.u.Hard.Valid == 1);
791 
792  ASSERT (MI_GET_PAGE_FRAME_FROM_PTE (&TempPte) == PageFrameIndex);
793  ASSERT (MI_PFN_ELEMENT(PageFrameIndex)->PteAddress == ProtoPte);
794 
795  ProtoPte += 1;
796  PageFrameIndex += 1;
797  }
798 
799  LOCK_WS_UNSAFE (CurrentThread, CurrentProcess);
800 #endif
801 
802  LOCK_PFN (OldIrql);
803 
804  do {
805  ASSERT (Pfn1->u2.ShareCount >= 1);
806  Pfn1->u2.ShareCount -= 1;
807  Pfn1 += 1;
808  } while (Pfn1 < EndPfn);
809 
810  MI_WRITE_ZERO_PTE (PointerPde);
811 
812  MI_FLUSH_PROCESS_TB (FALSE);
813
