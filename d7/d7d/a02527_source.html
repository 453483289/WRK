32 /*++
33 
34 Routine Description:
35 
36  This routine decrements the share count within the PFN element
37  for the specified physical page. If the share count becomes
38  zero the corresponding PTE is converted to the transition state
39  and the reference count is decremented and the ValidPte count
40  of the PTEframe is decremented.
41 
42 Arguments:
43 
44  Pfn1 - Supplies the PFN database entry to decrement.
45 
46  PageFrameIndex - Supplies the physical page number of which to decrement
47  the share count.
48 
49 Return Value:
50 
51  None.
52 
53 Environment:
54 
55  Must be holding the PFN database lock with APCs disabled.
56 
57 --*/
58 
59 {
60  ULONG FreeBit;
61  MMPTE TempPte;
62  PMMPTE PointerPte;
63  PEPROCESS Process;
64 
65  ASSERT (PageFrameIndex > 0);
66  ASSERT (MI_IS_PFN (PageFrameIndex));
67  ASSERT (Pfn1 == MI_PFN_ELEMENT (PageFrameIndex));
68 
69  if (Pfn1->u3.e1.PageLocation != ActiveAndValid &&
70  Pfn1->u3.e1.PageLocation != StandbyPageList) {
71  KeBugCheckEx (PFN_LIST_CORRUPT,
72  0x99,
73  PageFrameIndex,
74  Pfn1->u3.e1.PageLocation,
75  0);
76  }
77 
78  Pfn1->u2.ShareCount -= 1;
79 
80  ASSERT (Pfn1->u2.ShareCount < 0xF000000);
81 
82  if (Pfn1->u2.ShareCount == 0) {
83 
84  if (PERFINFO_IS_GROUP_ON(PERF_MEMORY)) {
85  PERFINFO_PFN_INFORMATION PerfInfoPfn;
86 
87  PerfInfoPfn.PageFrameIndex = PageFrameIndex;
88  PerfInfoLogBytes(PERFINFO_LOG_TYPE_ZEROSHARECOUNT,
89  &PerfInfoPfn,
90  sizeof(PerfInfoPfn));
91  }
92 
93  //
94  // The share count is now zero, decrement the reference count
95  // for the PFN element and turn the referenced PTE into
96  // the transition state if it refers to a prototype PTE.
97  // PTEs which are not prototype PTEs do not need to be placed
98  // into transition as they are placed in transition when
99  // they are removed from the working set (working set free routine).
100  //
101 
102  //
103  // If the PTE referenced by this PFN element is actually
104  // a prototype PTE, it must be mapped into hyperspace and
105  // then operated on.
106  //
107 
108  if (Pfn1->u3.e1.PrototypePte == 1) {
109 
110  if (MiIsProtoAddressValid (Pfn1->PteAddress)) {
111  Process = NULL;
112  PointerPte = Pfn1->PteAddress;
113  }
114  else {
115 
116  //
117  // The address is not valid in this process, map it into
118  // hyperspace so it can be operated upon.
119  //
120 
121  Process = PsGetCurrentProcess ();
122  PointerPte = (PMMPTE) MiMapPageInHyperSpaceAtDpc(Process, Pfn1->u4.PteFrame);
123  PointerPte = (PMMPTE)((PCHAR)PointerPte +
124  MiGetByteOffset(Pfn1->PteAddress));
125  }
126 
127  TempPte = *PointerPte;
128 
129  MI_MAKE_VALID_PTE_TRANSITION (TempPte,
130  Pfn1->OriginalPte.u.Soft.Protection);
131  MI_WRITE_INVALID_PTE (PointerPte, TempPte);
132 
133  if (Process != NULL) {
134  MiUnmapPageInHyperSpaceFromDpc (Process, PointerPte);
135  }
136 
137  //
138  // There is no need to flush the translation buffer at this
139  // time as we only invalidated a prototype PTE.
140  //
141  }
142 
143  //
144  // Change the page location to inactive (from active and valid).
145  //
146 
147  Pfn1->u3.e1.PageLocation = TransitionPage;
148 
149  //
150  // Decrement the reference count as the share count is now zero.
151  //
152 
153  MM_PFN_LOCK_ASSERT();
154 
155  ASSERT (Pfn1->u3.e2.ReferenceCount != 0);
156 
157  if (Pfn1->u3.e2.ReferenceCount == 1) {
158 
159  if (MI_IS_PFN_DELETED (Pfn1)) {
160 
161  Pfn1->u3.e2.ReferenceCount = 0;
162 
163  //
164  // There is no referenced PTE for this page, delete the page
165  // file space (if any), and place the page on the free list.
166  //
167 
168  ASSERT (Pfn1->OriginalPte.u.Soft.Prototype == 0);
169 
170  FreeBit = GET_PAGING_FILE_OFFSET (Pfn1->OriginalPte);
171 
172  if ((FreeBit != 0) && (FreeBit != MI_PTE_LOOKUP_NEEDED)) {
173  MiReleaseConfirmedPageFileSpace (Pfn1->OriginalPte);
174  }
175 
176  //
177  // Temporarily mark the frame as active and valid so that
178  // MiIdentifyPfn knows it is safe to walk back through the
179  // containing frames for a more accurate identification.
180  // Note the page will be immediately re-marked as it is
181  // inserted into the freelist.
182  //
183 
184  Pfn1->u3.e1.PageLocation = ActiveAndValid;
185 
186  MiInsertPageInFreeList (PageFrameIndex);
187  }
188  else {
189  MiDecrementReferenceCount (Pfn1, PageFrameIndex);
190  }
191  }
192  else {
193  InterlockedDecrementPfn ((PSHORT)&Pfn1->u3.e2.ReferenceCount);
194  }
195  }
196 
197  return;
198 }

202 MiDecrementReferenceCount (

207 /*++
208 
209 Routine Description:
210 
211  This routine decrements the reference count for the specified page.
212  If the reference count becomes zero, the page is placed on the
213  appropriate list (free, modified, standby or bad). If the page
214  is placed on the free or standby list, the number of available
215  pages is incremented and if it transitions from zero to one, the
216  available page event is set.
217 
218 
219 Arguments:
220 
221  Pfn1 - Supplies the PFN database entry to decrement.
222 
223  PageFrameIndex - Supplies the physical page number of which to
224  decrement the reference count.
225 
226 Return Value:
227 
228  None.
229 
230 Environment:
231 
232  Must be holding the PFN database lock with APCs disabled.
233 
234 --*/
235 
236 {
237  ULONG FreeBit;
238 
239  MM_PFN_LOCK_ASSERT();
240 
241  ASSERT (MI_IS_PFN (PageFrameIndex));
242  ASSERT (Pfn1 == MI_PFN_ELEMENT (PageFrameIndex));
243  ASSERT (Pfn1->u3.e2.ReferenceCount != 0);
244 
245  InterlockedDecrementPfn ((PSHORT)&Pfn1->u3.e2.ReferenceCount);
246 
247  if (Pfn1->u3.e2.ReferenceCount != 0) {
248 
249  //
250  // The reference count is not zero, return.
251  //
252 
253  return;
254  }
255 
256  //
257  // The reference count is now zero, put the page on some list.
258  //
259 
260  if (Pfn1->u2.ShareCount != 0) {
261 
262  KeBugCheckEx (PFN_LIST_CORRUPT,
263  7,
264  PageFrameIndex,
265  Pfn1->u2.ShareCount,
266  0);
267  return;
268  }
269 
270  ASSERT (Pfn1->u3.e1.PageLocation != ActiveAndValid);
271 
272  if (MI_IS_PFN_DELETED (Pfn1)) {
273 
274  //
275  // There is no referenced PTE for this page, delete the page
276  // file space (if any), and place the page on the free list.
277  //
278 
279  if (Pfn1->OriginalPte.u.Soft.Prototype == 0) {
280 
281  FreeBit = GET_PAGING_FILE_OFFSET (Pfn1->OriginalPte);
282 
283  if ((FreeBit != 0) && (FreeBit != MI_PTE_LOOKUP_NEEDED)) {
284  MiReleaseConfirmedPageFileSpace (Pfn1->OriginalPte);
285  }
286  }
287 
288  MiInsertPageInFreeList (PageFrameIndex);
289 
290  return;
291  }
292 
293  //
294  // Place the page on the modified or standby list depending
295  // on the state of the modify bit in the PFN element.
296  //
297 
298  if (Pfn1->u3.e1.Modified == 1) {
299  MiInsertPageInList (&MmModifiedPageListHead, PageFrameIndex);
300  }
301  else {
302 
303  if (Pfn1->u3.e1.RemovalRequested == 1) {
304 
305  //
306  // The page may still be marked as on the modified list if the
307  // current thread is the modified writer completing the write.
308  // Mark it as standby so restoration of the transition PTE
309  // doesn't flag this as illegal.
310  //
311 
312  Pfn1->u3.e1.PageLocation = StandbyPageList;
313 
314  MiRestoreTransitionPte (Pfn1);
315  MiInsertPageInList (&MmBadPageListHead, PageFrameIndex);
316  return;
317  }
318 
319  if (!MmFrontOfList) {
320  MiInsertPageInList (&MmStandbyPageListHead, PageFrameIndex);
321  }
322  else {
323  MiInsertStandbyListAtFront (PageFrameIndex);
324  }
325  }
326 
327  return;
328 }
