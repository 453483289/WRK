135 //
136 // Each type of MCE has a control structure that is used to determine
137 // whether to poll or wait for an interrupt to determine when to query
138 // for the logs. This is needed since we can get a callback from the
139 // HAL at high IRQL to inform us that a MCE log is available.
140 // Additionally Ke Timer used for polling will calls us at DPC level.
141 // So in the case of an interrupt we will queue a DPC. Within the DPC
142 // routine we will queue a work item so that we can get back to
143 // passive level and be able to call the hal to get the logs (Can only
144 // call hal at passive). The DPC and work item routines are common so a
145 // MCEQUERYINFO struct is passed around so that it can operate on the
146 // correct log type. Note that this implies that there may be multiple
147 // work items querying the hal for different log types at the same
148 // time. In addition this struct also contains useful log related
149 // information including the maximum log size (as reported by the HAL),
150 // the token that must be passed to the HAL when querying for the
151 // logs and the HAL InfoClass to use when querying for the logs.
152 //
153 // PollFrequency keeps track of the number of seconds before initiating a
154 // query. If it is 0 (HAL_CPE_DISABLED / HAL_CMC_DISABLED) then no
155 // polling occurs and if it is -1 (HAL_CPE_INTERRUPTS_BASED /
156 // HAL_CMC_INTERRUPTS_BASED) then no polling occurs either. There is
157 // only one work item active for each log type and this is enforced via
158 // ItemsOutstanding in that only whenever it transitions from 0 to 1 is
159 // the work item queued.
160 //
161 #define DEFAULT_MAX_MCA_SIZE 0x1000
162 #define DEFAULT_MAX_CMC_SIZE 0x1000
163 #define DEFAULT_MAX_CPE_SIZE 0x1000
164 
165 typedef struct
166 {
167  HAL_QUERY_INFORMATION_CLASS InfoClass; // HAL Info class to use in MCE query
168  ULONG PollFrequency; // Polling Frequency in seconds
169  PVOID Token; // HAL Token to use in MCE Queries
170  LONG ItemsOutstanding; // Number of interrupts or poll requests to process
171  ULONG MaxSize; // Max size for log (as reported by HAL)
172  GUID WnodeGuid; // GUID to use for the raw data event
173  GUID SwitchToPollGuid; // GUID to use to fire event for switching to polled mode
174  NTSTATUS SwitchToPollErrorCode; // Eventlog error code that indicates a switch to polled mode
175  ULONG WorkerInProgress; // Set to 1 if worker routine is running
176  KSPIN_LOCK DpcLock;
177  KDPC DeliveryDpc; // DPC to handle delivery
178  KTIMER PollingTimer; // KTIMER used for polling
179  KDPC PollingDpc; // DPC to use for polling
180  WORK_QUEUE_ITEM WorkItem; // Work item used to query for log
181 } MCEQUERYINFO, *PMCEQUERYINFO;

540 void WmipMceWorkerRoutine(
541  IN PVOID Context // MCEQUERYINFO
542  )
543 /*++
544 
545 Routine Description:
546 
547  Worker routine that handles polling for corrected MCA, CMC and CPE
548  logs from the HAL and then firing them as WMI events.
549 
550 Arguments:
551 
552  Context is a pointer to the MCEQUERYINFO for the type of log that
553  needs to be queried.
554 
555 Return Value:
556 
557 --*/
558 {
559  PMCEQUERYINFO QueryInfo = (PMCEQUERYINFO)Context;
560  NTSTATUS Status;
561  ULONG i;
562  LONG x, Count;
563 
564  PAGED_CODE();
565 
566  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_MCA_LEVEL,
567  "WMI: WmipMceWorkerRoutine %p enter\n",
568  QueryInfo));
569 
570  //
571  // If the worker is already in progress then we just exit
572  //
573  WmipEnterSMCritSection();
574  if (QueryInfo->WorkerInProgress == 0)
575  {
576  QueryInfo->WorkerInProgress = 1;
577  WmipLeaveSMCritSection();
578  } else {
579  WmipLeaveSMCritSection();
580  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_MCA_LEVEL,
581  "WMI: WmipMceWorkerRoutine %p in progress\n",
582  QueryInfo));
583  return;
584  }
585 
586  //
587  // Check to see if access has already been disabled
588  //
589  if (QueryInfo->PollFrequency != HAL_MCE_DISABLED)
590  {
591  //
592  // We get all of the records by calling into the hal and querying
593  // for the logs until the hal returns an error or we've
594  // retrieved 256 records. We want to protect ourselves from the
595  // case where a repeated corrected error would cause the loop
596  // to be infinite.
597  //
598  i = 0;
599  do
600  {
601  //
602  // Remember how many corrected errors we have received up until
603  // this point. We guarantee that we've handled them up
604  // until this point
605  //
606  Count = QueryInfo->ItemsOutstanding;
607 
608  Status = WmipQueryLogAndFireEvent(QueryInfo);
609  } while ((NT_SUCCESS(Status) && (i++ < 256)));
610 
611  //
612  // Reset counter back to 0, but check if any errors
613  // had occured while we were processing. If so we go
614  // back and make sure they are handled. Note that this
615  // could cause a new worker thread to be created while we
616  // are still processing these, but that is ok since we only
617  // allow one worker thread to run at one time.
618  //
619  WmipEnterSMCritSection();
620  x = InterlockedExchange(&QueryInfo->ItemsOutstanding,
621  0);
622  if ((x > Count) && (i < 257))
623  {
624  //
625  // Since there are still more corrected errors to
626  // process, queue a new DPC to cause a new worker
627  // routine to be run.
628  //
629  WmipInsertQueueMCEDpc(QueryInfo);
630  }
631 
632  QueryInfo->WorkerInProgress = 0;
633  WmipLeaveSMCritSection();
634  }
635 }
