49 /*++
50 
51 Routine Description:
52 
53  This function selectively moves the contents of the specified trap frame
54  and nonvolatile context to the specified context record.
55 
56 Arguments:
57 
58  TrapFrame - Supplies the contents of a trap frame.
59 
60  ContextPointers - Supplies the address of context pointers record.
61 
62  ContextRecord - Supplies the address of a context record.
63 
64 Return Value:
65 
66  None.
67 
68 --*/
69 
70 {
71 
72  ULONG ContextFlags;
73 
74  PAGED_CODE();
75 
76  //
77  // Get control information if specified.
78  //
79 
80  ContextFlags = ContextRecord->ContextFlags;
81  if ((ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL) {
82 
83  //
84  // Set registers RIP, CS, RSP, SS, and EFlags.
85  //
86 
87  ContextRecord->Rip = TrapFrame->Rip;
88  ContextRecord->SegCs = TrapFrame->SegCs;
89  ContextRecord->SegSs = TrapFrame->SegSs;
90  ContextRecord->Rsp = TrapFrame->Rsp;
91  ContextRecord->EFlags = TrapFrame->EFlags;
92  }
93 
94  //
95  // Get segment register contents if specified.
96  //
97 
98  if ((ContextFlags & CONTEXT_SEGMENTS) == CONTEXT_SEGMENTS) {
99 
100  //
101  // Set segment registers GS, FS, ES, DS.
102  //
103 
104  ContextRecord->SegDs = KGDT64_R3_DATA | RPL_MASK;
105  ContextRecord->SegEs = KGDT64_R3_DATA | RPL_MASK;
106  ContextRecord->SegFs = KGDT64_R3_CMTEB | RPL_MASK;
107  ContextRecord->SegGs = KGDT64_R3_DATA | RPL_MASK;
108  }
109 
110  //
111  // Get integer register contents if specified.
112  //
113 
114  if ((ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER) {
115 
116  //
117  // Set integer registers RAX, RCX, RDX, RSI, RDI, R8, R9, R10, RBX,
118  // RBP, R11, R12, R13, R14, and R15.
119  //
120 
121  ContextRecord->Rax = TrapFrame->Rax;
122  ContextRecord->Rcx = TrapFrame->Rcx;
123  ContextRecord->Rdx = TrapFrame->Rdx;
124  ContextRecord->R8 = TrapFrame->R8;
125  ContextRecord->R9 = TrapFrame->R9;
126  ContextRecord->R10 = TrapFrame->R10;
127  ContextRecord->R11 = TrapFrame->R11;
128 
129  ContextRecord->Rbx = *ContextPointers->Rbx;
130  ContextRecord->Rbp = *ContextPointers->Rbp;
131  ContextRecord->Rsi = *ContextPointers->Rsi;
132  ContextRecord->Rdi = *ContextPointers->Rdi;
133  ContextRecord->R12 = *ContextPointers->R12;
134  ContextRecord->R13 = *ContextPointers->R13;
135  ContextRecord->R14 = *ContextPointers->R14;
136  ContextRecord->R15 = *ContextPointers->R15;
137  }
138 
139  //
140  // Get floating point context if specified.
141  //
142 
143  if ((ContextFlags & CONTEXT_FLOATING_POINT) == CONTEXT_FLOATING_POINT) {
144 
145  //
146  // Set XMM registers Xmm0-Xmm15 and the XMM CSR contents.
147  //
148  // N.B. The legacy floating state is handled separately.
149  //
150 
151  ContextRecord->Xmm0 = TrapFrame->Xmm0;
152  ContextRecord->Xmm1 = TrapFrame->Xmm1;
153  ContextRecord->Xmm2 = TrapFrame->Xmm2;
154  ContextRecord->Xmm3 = TrapFrame->Xmm3;
155  ContextRecord->Xmm4 = TrapFrame->Xmm4;
156  ContextRecord->Xmm5 = TrapFrame->Xmm5;
157 
158  ContextRecord->Xmm6 = *ContextPointers->Xmm6;
159  ContextRecord->Xmm7 = *ContextPointers->Xmm7;
160  ContextRecord->Xmm8 = *ContextPointers->Xmm8;
161  ContextRecord->Xmm9 = *ContextPointers->Xmm9;
162  ContextRecord->Xmm10 = *ContextPointers->Xmm10;
163  ContextRecord->Xmm11 = *ContextPointers->Xmm11;
164  ContextRecord->Xmm12 = *ContextPointers->Xmm12;
165  ContextRecord->Xmm13 = *ContextPointers->Xmm13;
166  ContextRecord->Xmm14 = *ContextPointers->Xmm14;
167  ContextRecord->Xmm15 = *ContextPointers->Xmm15;
168 
169  ContextRecord->MxCsr = TrapFrame->MxCsr;
170  }
171 
172  //
173  //
174  // Get debug register contents if requested.
175  //
176 
177  if ((ContextFlags & CONTEXT_DEBUG_REGISTERS) == CONTEXT_DEBUG_REGISTERS) {
178 
179  //
180  // Set the debug registers DR0, DR1, DR2, DR3, DR6, and DR7.
181  //
182 
183  if ((TrapFrame->Dr7 & DR7_ACTIVE) != 0) {
184  ContextRecord->Dr0 = TrapFrame->Dr0;
185  ContextRecord->Dr1 = TrapFrame->Dr1;
186  ContextRecord->Dr2 = TrapFrame->Dr2;
187  ContextRecord->Dr3 = TrapFrame->Dr3;
188  ContextRecord->Dr6 = TrapFrame->Dr6;
189  ContextRecord->Dr7 = TrapFrame->Dr7;
190  if ((TrapFrame->Dr7 & DR7_LAST_BRANCH) != 0) {
191  ContextRecord->LastBranchToRip = TrapFrame->LastBranchToRip;
192  ContextRecord->LastBranchFromRip = TrapFrame->LastBranchFromRip;
193  ContextRecord->LastExceptionToRip = TrapFrame->LastExceptionToRip;
194  ContextRecord->LastExceptionFromRip = TrapFrame->LastExceptionFromRip;
195 
196  } else {
197  ContextRecord->LastBranchToRip = 0;
198  ContextRecord->LastBranchFromRip = 0;
199  ContextRecord->LastExceptionToRip = 0;
200  ContextRecord->LastExceptionFromRip = 0;
201  }
202 
203  } else {
204  ContextRecord->Dr0 = 0;
205  ContextRecord->Dr1 = 0;
206  ContextRecord->Dr2 = 0;
207  ContextRecord->Dr3 = 0;
208  ContextRecord->Dr6 = 0;
209  ContextRecord->Dr7 = 0;
210  ContextRecord->LastBranchToRip = 0;
211  ContextRecord->LastBranchFromRip = 0;
212  ContextRecord->LastExceptionToRip = 0;
213  ContextRecord->LastExceptionFromRip = 0;
214  }
215  }
216 
217  //
218  // Get exception reporting information if requested.
219  //
220 
221  if ((ContextFlags & CONTEXT_EXCEPTION_REQUEST) != 0) {
222  ContextRecord->ContextFlags &= ~CONTEXT_EXCEPTION_FLAGS;
223  ContextRecord->ContextFlags |= CONTEXT_EXCEPTION_REPORTING;
224  if (TrapFrame->ExceptionActive == 1) {
225  ContextRecord->ContextFlags |= CONTEXT_EXCEPTION_ACTIVE;
226 
227  } else if (TrapFrame->ExceptionActive == 2) {
228  ContextRecord->ContextFlags |= CONTEXT_SERVICE_ACTIVE;
229  }
230  }
231 
232  return;
233 }

235 VOID
236 PspSetContext (
237  OUT PKTRAP_FRAME TrapFrame,
238  OUT PKNONVOLATILE_CONTEXT_POINTERS ContextPointers,
239  IN PCONTEXT ContextRecord,
240  KPROCESSOR_MODE PreviousMode
241  )
242 
243 /*++
244 
245 Routine Description:
246 
247  This function selectively moves the contents of the specified context
248  record to the specified trap frame and nonvolatile context.
249 
250 Arguments:
251 
252  TrapFrame - Supplies the address of a trap frame.
253 
254  ContextPointers - Supplies the address of a context pointers record.
255 
256  ContextRecord - Supplies the address of a context record.
257 
258  ProcessorMode - Supplies the processor mode to use when sanitizing
259  the PSR and FSR.
260 
261 Return Value:
262 
263  None.
264 
265 --*/
266 
267 {
268 
269  ULONG ContextFlags;
270 
271  PAGED_CODE();
272 
273  //
274  // Set control information if specified.
275  //
276 
277  ContextFlags = ContextRecord->ContextFlags;
278  if ((ContextFlags & CONTEXT_CONTROL) == CONTEXT_CONTROL) {
279  TrapFrame->EFlags = SANITIZE_EFLAGS(ContextRecord->EFlags, PreviousMode);
280  TrapFrame->Rip = ContextRecord->Rip;
281  TrapFrame->Rsp = ContextRecord->Rsp;
282  }
283 
284  //
285  // The segment registers DS, ES, FS, and GS are never restored from saved
286  // data. However, SS and CS are restored from the trap frame. Make sure
287  // that these segment registers have the proper values.
288  //
289 
290  if (PreviousMode == UserMode) {
291  TrapFrame->SegSs = KGDT64_R3_DATA | RPL_MASK;
292  if (ContextRecord->SegCs != (KGDT64_R3_CODE | RPL_MASK)) {
293  TrapFrame->SegCs = KGDT64_R3_CMCODE | RPL_MASK;
294 
295  } else {
296  TrapFrame->SegCs = KGDT64_R3_CODE | RPL_MASK;
297  }
298 
299  } else {
300  TrapFrame->SegCs = KGDT64_R0_CODE;
301  TrapFrame->SegSs = KGDT64_NULL;
302  }
303 
304  TrapFrame->Rip = SANITIZE_VA(TrapFrame->Rip, TrapFrame->SegCs, PreviousMode);
305 
306  //
307  // Set integer registers contents if specified.
308  //
309 
310  if ((ContextFlags & CONTEXT_INTEGER) == CONTEXT_INTEGER) {
311 
312  //
313  // Set integer registers RAX, RCX, RDX, RSI, RDI, R8, R9, R10, RBX,
314  // RBP, R11, R12, R13, R14, and R15.
315  //
316 
317  TrapFrame->Rax = ContextRecord->Rax;
318  TrapFrame->Rcx = ContextRecord->Rcx;
319  TrapFrame->Rdx = ContextRecord->Rdx;
320  TrapFrame->R8 = ContextRecord->R8;
321  TrapFrame->R9 = ContextRecord->R9;
322  TrapFrame->R10 = ContextRecord->R10;
323  TrapFrame->R11 = ContextRecord->R11;
324 
325  *ContextPointers->Rbx = ContextRecord->Rbx;
326  *ContextPointers->Rbp = ContextRecord->Rbp;
327  *ContextPointers->Rsi = ContextRecord->Rsi;
328  *ContextPointers->Rdi = ContextRecord->Rdi;
329  *ContextPointers->R12 = ContextRecord->R12;
330  *ContextPointers->R13 = ContextRecord->R13;
331  *ContextPointers->R14 = ContextRecord->R14;
332  *ContextPointers->R15 = ContextRecord->R15;
333  }
334 
335  //
336  // Set floating register contents if requested.
337  //
338 
339  if ((ContextFlags & CONTEXT_FLOATING_POINT) == CONTEXT_FLOATING_POINT) {
340 
341  //
342  // Set XMM registers Xmm0-Xmm15 and the XMM CSR contents.
343  //
344  // N.B. The legacy floating state is handled separately.
345  //
346 
347  TrapFrame->Xmm0 = ContextRecord->Xmm0;
348  TrapFrame->Xmm1 = ContextRecord->Xmm1;
349  TrapFrame->Xmm2 = ContextRecord->Xmm2;
350  TrapFrame->Xmm3 = ContextRecord->Xmm3;
351  TrapFrame->Xmm4 = ContextRecord->Xmm4;
352  TrapFrame->Xmm5 = ContextRecord->Xmm5;
353 
354  *ContextPointers->Xmm6 = ContextRecord->Xmm6;
355  *ContextPointers->Xmm7 = ContextRecord->Xmm7;
356  *ContextPointers->Xmm8 = ContextRecord->Xmm8;
357  *ContextPointers->Xmm9 = ContextRecord->Xmm9;
358  *ContextPointers->Xmm10 = ContextRecord->Xmm10;
359  *ContextPointers->Xmm11 = ContextRecord->Xmm11;
360  *ContextPointers->Xmm12 = ContextRecord->Xmm12;
361  *ContextPointers->Xmm13 = ContextRecord->Xmm13;
362  *ContextPointers->Xmm14 = ContextRecord->Xmm14;
363  *ContextPointers->Xmm15 = ContextRecord->Xmm15;
364 
365  //
366  // Clear all reserved bits in MXCSR.
367  //
368 
369  TrapFrame->MxCsr = SANITIZE_MXCSR(ContextRecord->MxCsr);
370 
371  //
372  // Clear all reserved bits in legacy floating state.
373  //
374  // N.B. The legacy floating state is restored if and only if the
375  // request mode is user.
376  //
377  // N.B. The current MXCSR value is placed in the legacy floating
378  // state so it will get restored if the legacy state is
379  // restored.
380  //
381 
382  ContextRecord->FltSave.MxCsr = ReadMxCsr();
383  ContextRecord->FltSave.ControlWord =
384  SANITIZE_FCW(ContextRecord->FltSave.ControlWord);
385  }
386 
387  //
388  // Set debug register state if specified.
389  //
390 
391  if ((ContextFlags & CONTEXT_DEBUG_REGISTERS) == CONTEXT_DEBUG_REGISTERS) {
392 
393  //
394  // Set the debug registers DR0, DR1, DR2, DR3, DR6, and DR7.
395  //
396 
397  TrapFrame->Dr0 = SANITIZE_DRADDR(ContextRecord->Dr0, PreviousMode);
398  TrapFrame->Dr1 = SANITIZE_DRADDR(ContextRecord->Dr1, PreviousMode);
399  TrapFrame->Dr2 = SANITIZE_DRADDR(ContextRecord->Dr2, PreviousMode);
400  TrapFrame->Dr3 = SANITIZE_DRADDR(ContextRecord->Dr3, PreviousMode);
401  TrapFrame->Dr6 = 0;
402  TrapFrame->Dr7 = SANITIZE_DR7(ContextRecord->Dr7, PreviousMode);
403  if (PreviousMode != KernelMode) {
404  KeGetCurrentThread()->Header.DebugActive =
405  (BOOLEAN)((TrapFrame->Dr7 & DR7_ACTIVE) != 0);
406  }
407  }
408 
409  return;
410 }

413 PspGetSetContextInternal (
414  IN PKAPC Apc,
415  IN PVOID OperationType,
416  OUT PKEVENT *Event
417  )
418 
419 /*++
420 
421 Routine Description:
422 
423  This function either captures the state of the current thread, or sets
424  the state of the current thread.
425 
426 Arguments:
427 
428  Apc - Supplies a pointer to an APC object.
429 
430  OperationType - Supplies the type of context operation to be performed.
431  A value of NULL specifies a get context operation and a nonNULL value
432  a set context operation.
433 
434  Event - Supplies a pointer to a variable that receives the completion
435  event address.
436 
437 Return Value:
438 
439  If the context operation is a set context and the legacy floating state is
440  switched for the current thread, then the address of the legacy floating
441  save area is returned as the function value. Otherwise, NULL is returned.
442 
443 --*/
444 
445 {
446 
447  PGETSETCONTEXT ContextBlock;
448  ULONG ContextFlags;
449  PKNONVOLATILE_CONTEXT_POINTERS ContextPointers;
450  CONTEXT ContextRecord;
451  ULONG64 ControlPc;
452  ULONG64 EstablisherFrame;
453  PRUNTIME_FUNCTION FunctionEntry;
454  PVOID HandlerData;
455  ULONG64 ImageBase;
456  PETHREAD Thread;
457  ULONG64 TrapFrame;
458  PXMM_SAVE_AREA32 XmmSaveArea;
459 
460  PAGED_CODE();
461 
462  //
463  // Get the address of the context block and compute the address of the
464  // system entry trap frame.
465  //
466 
467  ContextBlock = CONTAINING_RECORD(Apc, GETSETCONTEXT, Apc);
468  ContextPointers = &ContextBlock->NonVolatileContext;
469  EstablisherFrame = 0;
470  Thread = PsGetCurrentThread();
471  TrapFrame = 0;
472  if (ContextBlock->Mode == KernelMode) {
473  TrapFrame = (ULONG64)Thread->Tcb.TrapFrame;
474  }
475 
476  if (TrapFrame == 0) {
477  TrapFrame = (ULONG64)PspGetBaseTrapFrame(Thread);
478  }
479 
480  //
481  // Capture the current thread context and set the initial control PC
482  // value.
483  //
484 
485  RtlCaptureContext(&ContextRecord);
486 
487  //
488  // Initialize context pointers for the nonvolatile integer and floating
489  // registers.
490  //
491 
492 #if DBG
493 
494  RtlZeroMemory(ContextPointers, sizeof(KNONVOLATILE_CONTEXT_POINTERS));
495 
496 #endif
497 
498  ContextPointers->Rbx = &ContextRecord.Rbx;
499  ContextPointers->Rsp = &ContextRecord.Rsp;
500  ContextPointers->Rbp = &ContextRecord.Rbp;
501  ContextPointers->Rsi = &ContextRecord.Rsi;
502  ContextPointers->Rdi = &ContextRecord.Rdi;
503  ContextPointers->R12 = &ContextRecord.R12;
504  ContextPointers->R13 = &ContextRecord.R13;
505  ContextPointers->R14 = &ContextRecord.R14;
506  ContextPointers->R15 = &ContextRecord.R15;
507 
508  ContextPointers->Xmm6 = &ContextRecord.Xmm6;
509  ContextPointers->Xmm7 = &ContextRecord.Xmm7;
510  ContextPointers->Xmm8 = &ContextRecord.Xmm8;
511  ContextPointers->Xmm9 = &ContextRecord.Xmm9;
512  ContextPointers->Xmm10 = &ContextRecord.Xmm10;
513  ContextPointers->Xmm11 = &ContextRecord.Xmm11;
514  ContextPointers->Xmm12 = &ContextRecord.Xmm12;
515  ContextPointers->Xmm13 = &ContextRecord.Xmm13;
516  ContextPointers->Xmm14 = &ContextRecord.Xmm14;
517  ContextPointers->Xmm15 = &ContextRecord.Xmm15;
518 
519  //
520  // Start with the frame specified by the context record and virtually
521  // unwind call frames until the system entry trap frame is encountered.
522  //
523 
524  do {
525 
526  //
527  // Lookup the function table entry using the point at which control
528  // left the function.
529  //
530 
531  ControlPc = ContextRecord.Rip;
532  FunctionEntry = RtlLookupFunctionEntry(ControlPc, &ImageBase, NULL);
533 
534  //
535  // If there is a function table entry for the routine, then virtually
536  // unwind to the caller of the current routine to obtain the address
537  // where control left the caller. Otherwise, the function is a leaf
538  // function and the return address register contains the address of
539  // where control left the caller.
540  //
541 
542  if (FunctionEntry != NULL) {
543  RtlVirtualUnwind(UNW_FLAG_EHANDLER,
544  ImageBase,
545  ControlPc,
546  FunctionEntry,
547  &ContextRecord,
548  &HandlerData,
549  &EstablisherFrame,
550  ContextPointers);
551 
552  } else {
553  ContextRecord.Rip = *(PULONG64)(ContextRecord.Rsp);
554  ContextRecord.Rsp += 8;
555  }
556 
557  } while (EstablisherFrame != TrapFrame);
558 
559  //
560  // If system argument one is nonzero, then set the context of the current
561  // thread. Otherwise, get the context of the current thread.
562  //
563 
564  XmmSaveArea = NULL;
565  if (OperationType != NULL) {
566 
567  //
568  // Set context.
569  //
570  // If the context mode is user and floating state is being set, then
571  // set the address of the legacy floating information.
572  //
573 
574  ContextFlags = ContextBlock->Context.ContextFlags;
575  if ((ContextBlock->Mode == UserMode) &&
576  ((ContextFlags & CONTEXT_FLOATING_POINT) == CONTEXT_FLOATING_POINT)) {
577 
578  XmmSaveArea = &ContextBlock->Context.FltSave;
579  }
580 
581  //
582  // Set context.
583  //
584 
585  PspSetContext((PKTRAP_FRAME)TrapFrame,
586  ContextPointers,
587  &ContextBlock->Context,
588  ContextBlock->Mode);
589 
590  } else {
591 
592  //
593  // Get context.
594  //
595  // If the context mode is user, then save the legacy floating state.
596  //
597 
598  if (ContextBlock->Mode == UserMode) {
599  KeSaveLegacyFloatingPointState(&ContextBlock->Context.FltSave);
600  }
601 
602  PspGetContext((PKTRAP_FRAME)TrapFrame,
603  ContextPointers,
604  &ContextBlock->Context);
605  }
606 
607  *Event = &ContextBlock->OperationComplete;
608  return XmmSaveArea;
609 }
