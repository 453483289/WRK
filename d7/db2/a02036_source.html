40 /*++
41 
42 Routine Description:
43 
44  This routine supports the setting of disable read ahead or disable write
45  behind flags to control Cache Manager operation. This routine may be
46  called any time after calling CcInitializeCacheMap. Initially both
47  read ahead and write behind are enabled. Note that the state of both
48  of these flags must be specified on each call to this routine.
49 
50 Arguments:
51 
52  FileObject - File object for which the respective flags are to be set.
53 
54  DisableReadAhead - FALSE to enable read ahead, TRUE to disable it.
55 
56  DisableWriteBehind - FALSE to enable write behind, TRUE to disable it.
57 
58 Return Value:
59 
60  None.
61 
62 --*/
63 
64 {
65  PSHARED_CACHE_MAP SharedCacheMap;
66  KIRQL OldIrql;
67 
68  //
69  // Get pointer to SharedCacheMap.
70  //
71 
72  SharedCacheMap = FileObject->SectionObjectPointer->SharedCacheMap;
73 
74  //
75  // Now set the flags and return.
76  //
77 
78  CcAcquireMasterLock( &OldIrql );
79  if (DisableReadAhead) {
80  SetFlag(SharedCacheMap->Flags, DISABLE_READ_AHEAD);
81  } else {
82  ClearFlag(SharedCacheMap->Flags, DISABLE_READ_AHEAD);
83  }
84  if (DisableWriteBehind) {
85  SetFlag(SharedCacheMap->Flags, DISABLE_WRITE_BEHIND | MODIFIED_WRITE_DISABLED);
86  } else {
87  ClearFlag(SharedCacheMap->Flags, DISABLE_WRITE_BEHIND);
88  }
89  CcReleaseMasterLock( OldIrql );
90 }

95 CcSetPrivateWriteFile(
96  PFILE_OBJECT FileObject
97  )
98 
99 /*++
100 
101 Routine Description:
102 
103  This routine will instruct the cache manager to treat the file as
104  a private-write stream, so that a caller can implement a private
105  logging mechanism for it. We will turn on both Mm's modify-no-write
106  and our disable-write-behind, and disallow non-aware flush/purge for
107  the file.
108 
109  Caching must already be initiated on the file.
110 
111  This routine is only exported to the kernel.
112 
113 Arguments:
114 
115  FileObject - File to make private-write.
116 
117 Return Value:
118 
119  None.
120 
121 --*/
122 
123 {
124  PSHARED_CACHE_MAP SharedCacheMap;
125  BOOLEAN Disabled;
126  KIRQL OldIrql;
127  PVACB Vacb;
128  ULONG ActivePage;
129  ULONG PageIsDirty;
130 
131  //
132  // Pick up the file exclusive to synchronize against readahead and
133  // other purge/map activity.
134  //
135 
136  FsRtlAcquireFileExclusive( FileObject );
137 
138  //
139  // Get a pointer to the SharedCacheMap. Be sure to release the FileObject
140  // in case an error condition forces a premature exit.
141  //
142 
143  if ((FileObject->SectionObjectPointer == NULL) ||
144  ((SharedCacheMap = FileObject->SectionObjectPointer->SharedCacheMap) == NULL)){
145  FsRtlReleaseFile( FileObject );
146  return FALSE;
147  }
148 
149  //
150  // Unmap all the views in preparation for making the disable mw call.
151  //
152 
153  //
154  // We still need to wait for any dangling cache read or writes.
155  //
156  // In fact we have to loop and wait because the lazy writer can
157  // sneak in and do an CcGetVirtualAddressIfMapped, and we are not
158  // synchronized.
159  //
160  // This is the same bit of code that our purge will do. We assume
161  // that a private writer has succesfully blocked out other activity.
162  //
163 
164  //
165  // If there is an active Vacb, then delete it now (before waiting!).
166  //
167 
168  CcAcquireMasterLock( &OldIrql );
169  GetActiveVacbAtDpcLevel( SharedCacheMap, Vacb, ActivePage, PageIsDirty );
170  CcReleaseMasterLock( OldIrql );
171 
172  if (Vacb != NULL) {
173 
174  CcFreeActiveVacb( SharedCacheMap, Vacb, ActivePage, PageIsDirty );
175  }
176 
177  while ((SharedCacheMap->Vacbs != NULL) &&
178  !CcUnmapVacbArray( SharedCacheMap, NULL, 0, FALSE )) {
179 
180  CcWaitOnActiveCount( SharedCacheMap );
181  }
182 
183  //
184  // Knock the file down.
185  //
186 
187  CcFlushCache( FileObject->SectionObjectPointer, NULL, 0, NULL );
188 
189  //
190  // Now the file is clean and unmapped. We can still have a racing
191  // lazy writer, though.
192  //
193  // We just wait for the lazy writer queue to drain before disabling
194  // modified write. There may be a better way to do this by having
195  // an event for the WRITE_QUEUED flag. ? This would also let us
196  // dispense with the pagingio pick/drop in the FS cache coherency
197  // paths, but there could be reasons why CcFlushCache shouldn't
198  // always do such a block. Investigate this.
199  //
200  // This wait takes on the order of ~.5s avg. case.
201  //
202 
203  CcAcquireMasterLock( &OldIrql );
204 
205  if (FlagOn( SharedCacheMap->Flags, WRITE_QUEUED ) ||
206  FlagOn( SharedCacheMap->Flags, READ_AHEAD_QUEUED )) {
207 
208  CcReleaseMasterLock( OldIrql );
209  FsRtlReleaseFile( FileObject );
210  CcWaitForCurrentLazyWriterActivity();
211  FsRtlAcquireFileExclusive( FileObject );
212 
213  } else {
214 
215  CcReleaseMasterLock( OldIrql );
216  }
217 
218  //
219  // Now set the flags and return. We do not set our MODIFIED_WRITE_DISABLED
220  // since we don't want to fully promote this cache map. Future?
221  //
222 
223  Disabled = MmDisableModifiedWriteOfSection( FileObject->SectionObjectPointer );
224 
225  if (Disabled) {
226  CcAcquireMasterLock( &OldIrql );
227  SetFlag(SharedCacheMap->Flags, DISABLE_WRITE_BEHIND | PRIVATE_WRITE);
228  CcReleaseMasterLock( OldIrql );
229  }
230 
231  //
232  // Now release the file for regular operation.
233  //
234 
235  FsRtlReleaseFile( FileObject );
236 
237  return Disabled;
238 }

291 LARGE_INTEGER
292 CcGetDirtyPages (
293  __in PVOID LogHandle,
294  __in PDIRTY_PAGE_ROUTINE DirtyPageRoutine,
295  __in PVOID Context1,
296  __in PVOID Context2
297  )
298 
299 /*++
300 
301 Routine Description:
302 
303  This routine may be called to return all of the dirty pages in all files
304  for a given log handle. Each page is returned by an individual call to
305  the Dirty Page Routine. The Dirty Page Routine is defined by a prototype
306  in ntos\inc\cache.h.
307 
308 Arguments:
309 
310  LogHandle - Log Handle which must match the log handle previously stored
311  for all files which are to be returned.
312 
313  DirtyPageRoutine -- The routine to call as each dirty page for this log
314  handle is found.
315 
316  Context1 - First context parameter to be passed to the Dirty Page Routine.
317 
318  Context2 - First context parameter to be passed to the Dirty Page Routine.
319 
320 Return Value:
321 
322  LARGE_INTEGER - Oldest Lsn found of all the dirty pages, or 0 if no dirty pages
323 
324 --*/
325 
326 {
327  PSHARED_CACHE_MAP SharedCacheMap;
328  PBCB Bcb, BcbToUnpin = NULL;
329  KLOCK_QUEUE_HANDLE LockHandle;
330  LARGE_INTEGER SavedFileOffset, SavedOldestLsn, SavedNewestLsn;
331  ULONG SavedByteLength;
332  LARGE_INTEGER OldestLsn = {0,0};
333 
334  //
335  // Synchronize with changes to the SharedCacheMap list.
336  //
337 
338  CcAcquireMasterLock( &LockHandle.OldIrql );
339 
340  SharedCacheMap = CONTAINING_RECORD( CcDirtySharedCacheMapList.SharedCacheMapLinks.Flink,
341  SHARED_CACHE_MAP,
342  SharedCacheMapLinks );
343 
344  //
345  // Use try/finally for cleanup. The only spot where we can raise is out of the
346  // filesystem callback, but we have the exception handler out here so we aren't
347  // constantly setting/unsetting it.
348  //
349 
350  try {
351 
352  while (&SharedCacheMap->SharedCacheMapLinks != &CcDirtySharedCacheMapList.SharedCacheMapLinks) {
353 
354  //
355  // Skip over cursors, SharedCacheMaps for other LogHandles, and ones with
356  // no dirty pages
357  //
358 
359  if (!FlagOn(SharedCacheMap->Flags, IS_CURSOR) && (SharedCacheMap->LogHandle == LogHandle) &&
360  (SharedCacheMap->DirtyPages != 0)) {
361 
362  //
363  // This SharedCacheMap should stick around for a while in the dirty list.
364  //
365 
366  CcIncrementOpenCount( SharedCacheMap, 'pdGS' );
367  SharedCacheMap->DirtyPages += 1;
368  CcReleaseMasterLock( LockHandle.OldIrql );
369 
370  //
371  // Set our initial resume point and point to first Bcb in List.
372  //
373 
374  KeAcquireInStackQueuedSpinLock( &SharedCacheMap->BcbSpinLock, &LockHandle );
375  Bcb = CONTAINING_RECORD( SharedCacheMap->BcbList.Flink, BCB, BcbLinks );
376 
377  //
378  // Scan to the end of the Bcb list.
379  //
380 
381  while (&Bcb->BcbLinks != &SharedCacheMap->BcbList) {
382 
383  //
384  // If the Bcb is dirty, then capture the inputs for the
385  // callback routine so we can call without holding a spinlock.
386  //
387 
388  if ((Bcb->NodeTypeCode == CACHE_NTC_BCB) && Bcb->Dirty) {
389 
390  SavedFileOffset = Bcb->FileOffset;
391  SavedByteLength = Bcb->ByteLength;
392  SavedOldestLsn = Bcb->OldestLsn;
393  SavedNewestLsn = Bcb->NewestLsn;
394 
395  //
396  // Increment PinCount so the Bcb sticks around
397  //
398 
399  Bcb->PinCount += 1;
400 
401  KeReleaseInStackQueuedSpinLock( &LockHandle );
402 
403  //
404  // Any Bcb to unref from a previous loop?
405  //
406 
407  if (BcbToUnpin != NULL) {
408  CcUnpinFileData( BcbToUnpin, TRUE, UNREF );
409  BcbToUnpin = NULL;
410  }
411 
412  //
413  // Call the file system. This callback may raise status.
414  //
415 
416  (*DirtyPageRoutine)( SharedCacheMap->FileObject,
417  &SavedFileOffset,
418  SavedByteLength,
419  &SavedOldestLsn,
420  &SavedNewestLsn,
421  Context1,
422  Context2 );
423 
424  //
425  // Possibly update OldestLsn
426  //
427 
428  if ((SavedOldestLsn.QuadPart != 0) &&
429  ((OldestLsn.QuadPart == 0) || (SavedOldestLsn.QuadPart < OldestLsn.QuadPart ))) {
430  OldestLsn = SavedOldestLsn;
431  }
432 
433  //
434  // Now reacquire the spinlock and scan from the resume point
435  // point to the next Bcb to return in the descending list.
436  //
437 
438  KeAcquireInStackQueuedSpinLock( &SharedCacheMap->BcbSpinLock, &LockHandle );
439 
440  //
441  // Normally the Bcb can stay around a while, but if not,
442  // we will just remember it for the next time we do not
443  // have the spin lock. We cannot unpin it now, because
444  // we would lose our place in the list.
445  //
446  // This is cheating, but it works and is sane since we're
447  // already traversing the bcb list - dropping the bcb count
448  // is OK, as long as we don't hit zero. Zero requires a
449  // slight bit more attention that shouldn't be replicated.
450  // (unmapping the view)
451  //
452 
453  if (Bcb->PinCount > 1) {
454  Bcb->PinCount -= 1;
455  } else {
456  BcbToUnpin = Bcb;
457  }
458  }
459 
460  Bcb = CONTAINING_RECORD( Bcb->BcbLinks.Flink, BCB, BcbLinks );
461  }
462  KeReleaseInStackQueuedSpinLock( &LockHandle );
463 
464  //
465  // We need to unref any Bcb we are holding before moving on to
466  // the next SharedCacheMap, or else CcDeleteSharedCacheMap will
467  // also delete this Bcb.
468  //
469 
470  if (BcbToUnpin != NULL) {
471 
472  CcUnpinFileData( BcbToUnpin, TRUE, UNREF );
473  BcbToUnpin = NULL;
474  }
475 
476  CcAcquireMasterLock( &LockHandle.OldIrql );
477 
478  //
479  // Now release the SharedCacheMap, leaving it in the dirty list.
480  //
481 
482  CcDecrementOpenCount( SharedCacheMap, 'pdGF' );
483  SharedCacheMap->DirtyPages -= 1;
484  }
485 
486  //
487  // Now loop back for the next cache map.
488  //
489 
490  SharedCacheMap =
491  CONTAINING_RECORD( SharedCacheMap->SharedCacheMapLinks.Flink,
492  SHARED_CACHE_MAP,
493  SharedCacheMapLinks );
494  }
495 
496  CcReleaseMasterLock( LockHandle.OldIrql );
497 
498  } finally {
499 
500  //
501  // Drop the Bcb if we are being ejected. We are guaranteed that the
502  // only raise is from the callback, at which point we have an incremented
503  // pincount.
504  //
505 
506  if (AbnormalTermination()) {
507 
508  CcUnpinFileData( Bcb, TRUE, UNPIN );
509  }
510  }
511 
512  return OldestLsn;
513 }

516 BOOLEAN
517 CcIsThereDirtyData (
518  __in PVPB Vpb
519  )
520 
521 /*++
522 
523 Routine Description:
524 
525  This routine returns TRUE if the specified Vcb has any unwritten dirty
526  data in the cache.
527 
528 Arguments:
529 
530  Vpb - specifies Vpb to check for
531 
532 Return Value:
533 
534  FALSE - if the Vpb has no dirty data
535  TRUE - if the Vpb has dirty data
536 
537 --*/
538 
539 {
540  PSHARED_CACHE_MAP SharedCacheMap;
541  KIRQL OldIrql;
542  ULONG LoopsWithLockHeld = 0;
543 
544  //
545  // Synchronize with changes to the SharedCacheMap list.
546  //
547 
548  CcAcquireMasterLock( &OldIrql );
549 
550  SharedCacheMap = CONTAINING_RECORD( CcDirtySharedCacheMapList.SharedCacheMapLinks.Flink,
551  SHARED_CACHE_MAP,
552  SharedCacheMapLinks );
553 
554  while (&SharedCacheMap->SharedCacheMapLinks != &CcDirtySharedCacheMapList.SharedCacheMapLinks) {
555 
556  //
557  // Look at this one if the Vpb matches and if there is dirty data.
558  // For what it's worth, don't worry about dirty data in temporary files,
559  // as that should not concern the caller if it wants to dismount.
560  //
561 
562  if (!FlagOn(SharedCacheMap->Flags, IS_CURSOR) &&
563  (SharedCacheMap->FileObject->Vpb == Vpb) &&
564  (SharedCacheMap->DirtyPages != 0) &&
565  !FlagOn(SharedCacheMap->FileObject->Flags, FO_TEMPORARY_FILE)) {
566 
567  CcReleaseMasterLock( OldIrql );
568  return TRUE;
569  }
570 
571  //
572  // Make sure we occasionally drop the lock. Set WRITE_QUEUED
573  // to keep the guy from going away, and increment DirtyPages to
574  // keep it in this list.
575  //
576 
577  if ((++LoopsWithLockHeld >= 20) &&
578  !FlagOn(SharedCacheMap->Flags, WRITE_QUEUED | IS_CURSOR)) {
579 
580  SetFlag( *((ULONG volatile *)&SharedCacheMap->Flags), WRITE_QUEUED);
581  *((ULONG volatile *)&SharedCacheMap->DirtyPages) += 1;
582  CcReleaseMasterLock( OldIrql );
583  LoopsWithLockHeld = 0;
584  CcAcquireMasterLock( &OldIrql );
585  ClearFlag( *((ULONG volatile *)&SharedCacheMap->Flags), WRITE_QUEUED);
586  *((ULONG volatile *)&SharedCacheMap->DirtyPages) -= 1;
587  }
588 
589  //
590  // Now loop back for the next cache map.
591  //
592 
593  SharedCacheMap =
594  CONTAINING_RECORD( SharedCacheMap->SharedCacheMapLinks.Flink,
595  SHARED_CACHE_MAP,
596  SharedCacheMapLinks );
597  }
598 
599  CcReleaseMasterLock( OldIrql );
600 
601  return FALSE;
602 }

604 LARGE_INTEGER
605 CcGetLsnForFileObject(
606  __in PFILE_OBJECT FileObject,
607  __out_opt PLARGE_INTEGER OldestLsn
608  )
609 
610 /*++
611 
612 Routine Description:
613 
614  This routine returns the oldest and newest LSNs for a file object.
615 
616 Arguments:
617 
618  FileObject - File for which the log handle should be stored.
619 
620  OldestLsn - pointer to location to store oldest LSN for file object.
621 
622 Return Value:
623 
624  The newest LSN for the file object.
625 
626 --*/
627 
628 {
629  PBCB Bcb;
630  KLOCK_QUEUE_HANDLE LockHandle;
631  LARGE_INTEGER Oldest, Newest;
632  PSHARED_CACHE_MAP SharedCacheMap = FileObject->SectionObjectPointer->SharedCacheMap;
633 
634  //
635  // initialize lsn variables
636  //
637 
638  Oldest.LowPart = 0;
639  Oldest.HighPart = 0;
640  Newest.LowPart = 0;
641  Newest.HighPart = 0;
642 
643  if(SharedCacheMap == NULL) {
644  return Oldest;
645  }
646 
647  KeAcquireInStackQueuedSpinLock(&SharedCacheMap->BcbSpinLock, &LockHandle);
648 
649  //
650  // Now point to first Bcb in List, and loop through it.
651  //
652 
653  Bcb = CONTAINING_RECORD( SharedCacheMap->BcbList.Flink, BCB, BcbLinks );
654 
655  while (&Bcb->BcbLinks != &SharedCacheMap->BcbList) {
656 
657  //
658  // If the Bcb is dirty then capture the oldest and newest lsn
659  //
660 
661 
662  if ((Bcb->NodeTypeCode == CACHE_NTC_BCB) && Bcb->Dirty) {
663 
664  LARGE_INTEGER BcbLsn, BcbNewest;
665 
666  BcbLsn = Bcb->OldestLsn;
667  BcbNewest = Bcb->NewestLsn;
668 
669  if ((BcbLsn.QuadPart != 0) &&
670  ((Oldest.QuadPart == 0) ||
671  (BcbLsn.QuadPart < Oldest.QuadPart))) {
672 
673  Oldest = BcbLsn;
674  }
675 
676  if ((BcbLsn.QuadPart != 0) && (BcbNewest.QuadPart > Newest.QuadPart)) {
677 
678  Newest = BcbNewest;
679  }
680  }
681 
682 
683  Bcb = CONTAINING_RECORD( Bcb->BcbLinks.Flink, BCB, BcbLinks );
684  }
685 
686  //
687  // Now release the spin lock for this Bcb list and generate a callback
688  // if we got something.
689  //
690 
691  KeReleaseInStackQueuedSpinLock( &LockHandle );
692 
693  if (ARGUMENT_PRESENT(OldestLsn)) {
694 
695  *OldestLsn = Oldest;
696  }
697 
698  return Newest;
699 }
