97 /*++
98 
99 Routine Description:
100 
101  This function initializes a kernel interrupt object. The service routine,
102  service context, spin lock, vector, IRQL, SynchronizeIrql, and floating
103  context save flag are initialized.
104 
105 Arguments:
106 
107  Interrupt - Supplies a pointer to a control object of type interrupt.
108 
109  ServiceRoutine - Supplies a pointer to a function that is to be
110  executed when an interrupt occurs via the specified interrupt
111  vector.
112 
113  ServiceContext - Supplies a pointer to an arbitrary data structure which is
114  to be passed to the function specified by the ServiceRoutine parameter.
115 
116  SpinLock - Supplies a pointer to an executive spin lock.
117 
118  Vector - Supplies the index of the entry in the Interrupt Dispatch Table
119  that is to be associated with the ServiceRoutine function.
120 
121  Irql - Supplies the request priority of the interrupting source.
122 
123  SynchronizeIrql - The request priority that the interrupt should be
124  synchronized with.
125 
126  InterruptMode - Supplies the mode of the interrupt; LevelSensitive or
127 
128  ShareVector - Supplies a boolean value that specifies whether the
129  vector can be shared with other interrupt objects or not. If FALSE
130  then the vector may not be shared, if TRUE it may be.
131  Latched.
132 
133  ProcessorNumber - Supplies the number of the processor to which the
134  interrupt will be connected.
135 
136  FloatingSave - Supplies a boolean value that determines whether the
137  floating point registers and pipe line are to be saved before calling
138  the ServiceRoutine function.
139 
140 Return Value:
141 
142  None.
143 
144 --*/
145 
146 {
147 
148  LONG Index;
149  PULONG pl;
150  PULONG NormalDispatchCode;
151 
152  //
153  // Initialize standard control object header.
154  //
155 
156  Interrupt->Type = InterruptObject;
157  Interrupt->Size = sizeof(KINTERRUPT);
158 
159  //
160  // Initialize the address of the service routine,
161  // the service context, the address of the spin lock, the vector
162  // number, the IRQL of the interrupting source, the Irql used for
163  // synchronize execution, the interrupt mode, the processor
164  // number, and the floating context save flag.
165  //
166 
167  Interrupt->ServiceRoutine = ServiceRoutine;
168  Interrupt->ServiceContext = ServiceContext;
169 
170  if (ARGUMENT_PRESENT(SpinLock)) {
171  Interrupt->ActualLock = SpinLock;
172  } else {
173  KeInitializeSpinLock (&Interrupt->SpinLock);
174  Interrupt->ActualLock = &Interrupt->SpinLock;
175  }
176 
177  Interrupt->Vector = Vector;
178  Interrupt->Irql = Irql;
179  Interrupt->SynchronizeIrql = SynchronizeIrql;
180  Interrupt->Mode = InterruptMode;
181  Interrupt->ShareVector = ShareVector;
182  Interrupt->Number = ProcessorNumber;
183  Interrupt->FloatingSave = FloatingSave;
184 
185  //
186  // Initialize fields for the interrupt storm detection. Set these
187  // to -1 so that the first time through the interrupt dispatch they
188  // will be reset correctly.
189  //
190  Interrupt->TickCount = (ULONG)-1;
191  Interrupt->DispatchCount = (ULONG)-1;
192 
193  //
194  // Copy the interrupt dispatch code template into the interrupt object
195  // and edit the machine code stored in the structure (please see
196  // _KiInterruptTemplate in intsup.asm.) Finally, flush the dcache
197  // on all processors that the current thread can
198  // run on to ensure that the code is actually in memory.
199  //
200 
201  NormalDispatchCode = &(Interrupt->DispatchCode[0]);
202 
203  pl = NormalDispatchCode;
204 
205  for (Index = 0; Index < NORMAL_DISPATCH_LENGTH; Index += 1) {
206  *NormalDispatchCode++ = KiInterruptTemplate[Index];
207  }
208 
209  //
210  // The following two instructions set the address of current interrupt
211  // object the the NORMAL dispatching code.
212  //
213 
214  pl = (PULONG)((PUCHAR)pl + ((PUCHAR)&KiInterruptTemplateObject -
215  (PUCHAR)KiInterruptTemplate) -4);
216  *pl = (ULONG)Interrupt;
217 
218  KeSweepDcache(FALSE);
219 
220  //
221  // Set the connected state of the interrupt object to FALSE.
222  //
223 
224  Interrupt->Connected = FALSE;
225  return;
226 }

228 BOOLEAN
229 KeConnectInterrupt (
230  __inout PKINTERRUPT Interrupt
231  )
232 
233 /*++
234 
235 Routine Description:
236 
237  This function connects an interrupt object to the interrupt vector
238  specified by the interrupt object. If the interrupt object is already
239  connected, or an attempt is made to connect to an interrupt that cannot
240  be connected, then a value of FALSE is returned. Else the specified
241  interrupt object is connected to the interrupt vector, the connected
242  state is set to TRUE, and TRUE is returned as the function value.
243 
244 Arguments:
245 
246  Interrupt - Supplies a pointer to a control object of type interrupt.
247 
248 Return Value:
249 
250  If the interrupt object is already connected or an attempt is made to
251  connect to an interrupt vector that cannot be connected, then a value
252  of FALSE is returned. Else a value of TRUE is returned.
253 
254 --*/
255 
256 {
257  DISPATCH_INFO DispatchInfo;
258  BOOLEAN Connected;
259  BOOLEAN ConnectError;
260  BOOLEAN Enabled;
261  KIRQL Irql;
262  CCHAR Number;
263  KIRQL OldIrql;
264  ULONG Vector;
265 
266  //
267  // If the interrupt object is already connected, the interrupt vector
268  // number is invalid, an attempt is being made to connect to a vector
269  // that cannot be connected, the interrupt request level is invalid, or
270  // the processor number is invalid, then do not connect the interrupt
271  // object. Else connect interrupt object to the specified vector and
272  // establish the proper interrupt dispatcher.
273  //
274 
275  Connected = FALSE;
276  ConnectError = FALSE;
277  Irql = Interrupt->Irql;
278  Number = Interrupt->Number;
279  Vector = Interrupt->Vector;
280  if ( !((Irql > HIGH_LEVEL) ||
281  (Number >= KeNumberProcessors) ||
282  (Interrupt->SynchronizeIrql < Irql) ||
283  (Interrupt->FloatingSave) // R0 x87 usage not supported on x86
284  )
285  ) {
286 
287  //
288  //
289  // Set system affinity to the specified processor.
290  //
291 
292  KeSetSystemAffinityThread((KAFFINITY)(1<<Number));
293 
294  //
295  // Raise IRQL to dispatcher level and lock dispatcher database.
296  //
297 
298  KiLockDispatcherDatabase(&OldIrql);
299 
300  //
301  // Is interrupt object already connected?
302  //
303 
304  if (!Interrupt->Connected) {
305 
306  //
307  // Determine interrupt dispatch vector
308  //
309 
310  KiGetVectorInfo (
311  Vector,
312  &DispatchInfo
313  );
314 
315  //
316  // If dispatch vector is not connected, then connect it
317  //
318 
319  if (DispatchInfo.Type == NoConnect) {
320  Connected = TRUE;
321  Interrupt->Connected = TRUE;
322 
323  //
324  // Connect interrupt dispatch to interrupt object dispatch code
325  //
326 
327  InitializeListHead(&Interrupt->InterruptListEntry);
328  KiConnectVectorAndInterruptObject (Interrupt, NormalConnect);
329 
330  //
331  // Enabled system vector
332  //
333 
334  Enabled = HalEnableSystemInterrupt(Vector, Irql, Interrupt->Mode);
335  if (!Enabled) {
336  ConnectError = TRUE;
337  }
338 
339 
340  } else if (DispatchInfo.Type != UnknownConnect &&
341  Interrupt->ShareVector &&
342  DispatchInfo.Interrupt->ShareVector &&
343  DispatchInfo.Interrupt->Mode == Interrupt->Mode) {
344 
345  //
346  // Vector is already connected as sharable. New vector is sharable
347  // and modes match. Chain new vector.
348  //
349 
350  Connected = TRUE;
351  Interrupt->Connected = TRUE;
352 
353  ASSERT (Irql <= SYNCH_LEVEL);
354 
355  //
356  // If not already using chained dispatch handler, set it up
357  //
358 
359  if (DispatchInfo.Type != ChainConnect) {
360  KiConnectVectorAndInterruptObject (DispatchInfo.Interrupt, ChainConnect);
361  }
362 
363  //
364  // Add to tail of chained dispatch
365  //
366 
367  InsertTailList(
368  &DispatchInfo.Interrupt->InterruptListEntry,
369  &Interrupt->InterruptListEntry
370  );
371 
372  }
373  }
374 
375  //
376  // Unlock dispatcher database and lower IRQL to its previous value.
377  //
378 
379  KiUnlockDispatcherDatabase(OldIrql);
380 
381  //
382  // Set system affinity back to the original value.
383  //
384 
385  KeRevertToUserAffinityThread();
386  }
387 
388  if (Connected && ConnectError) {
389 #if DBG
390  DbgPrint ("HalEnableSystemInterrupt failed\n");
391 #endif
392  KeDisconnectInterrupt (Interrupt);
393  Connected = FALSE;
394  }
395 
396  //
397  // Return whether interrupt was connected to the specified vector.
398  //
399 
400  return Connected;
401 }

404 KeDisconnectInterrupt (
405  __inout PKINTERRUPT Interrupt

407 
408 /*++
409 
410 Routine Description:
411 
412  This function disconnects an interrupt object from the interrupt vector
413  specified by the interrupt object. If the interrupt object is not
414  connected, then a value of FALSE is returned. Else the specified interrupt
415  object is disconnected from the interrupt vector, the connected state is
416  set to FALSE, and TRUE is returned as the function value.
417 
418 Arguments:
419 
420  Interrupt - Supplies a pointer to a control object of type interrupt.
421 
422 Return Value:
423 
424  If the interrupt object is not connected, then a value of FALSE is
425  returned. Else a value of TRUE is returned.
426 
427 --*/
428 
429 {
430 
431  DISPATCH_INFO DispatchInfo;
432  BOOLEAN Connected;
433  PKINTERRUPT Interrupty;
434  KIRQL Irql;
435  KIRQL OldIrql;
436  ULONG Vector;
437 
438  //
439  // Set system affinity to the specified processor.
440  //
441 
442  KeSetSystemAffinityThread((KAFFINITY)(1<<Interrupt->Number));
443 
444  //
445  // Raise IRQL to dispatcher level and lock dispatcher database.
446  //
447 
448  KiLockDispatcherDatabase(&OldIrql);
449 
450  //
451  // If the interrupt object is connected, then disconnect it from the
452  // specified vector.
453  //
454 
455  Connected = Interrupt->Connected;
456  if (Connected) {
457  Irql = Interrupt->Irql;
458  Vector = Interrupt->Vector;
459 
460  //
461  // If the specified interrupt vector is not connected to the chained
462  // interrupt dispatcher, then disconnect it by setting its dispatch
463  // address to the unexpected interrupt routine. Else remove the
464  // interrupt object from the interrupt chain. If there is only
465  // one entry remaining in the list, then reestablish the dispatch
466  // address.
467  //
468 
469  //
470  // Determine interrupt dispatch vector
471  //
472 
473  KiGetVectorInfo (
474  Vector,
475  &DispatchInfo
476  );
477 
478 
479  //
480  // Is dispatch a chained handler?
481  //
482 
483  if (DispatchInfo.Type == ChainConnect) {
484 
485  ASSERT (Irql <= SYNCH_LEVEL);
486 
487  //
488  // Is interrupt being removed from head?
489  //
490 
491  if (Interrupt == DispatchInfo.Interrupt) {
492 
493  //
494  // Update next interrupt object to be head
495  //
496 
497  DispatchInfo.Interrupt = CONTAINING_RECORD(
498  DispatchInfo.Interrupt->InterruptListEntry.Flink,
499  KINTERRUPT,
500  InterruptListEntry
501  );
502 
503  KiConnectVectorAndInterruptObject (DispatchInfo.Interrupt, ChainConnect);
504  }
505 
506  //
507  // Remove interrupt object
508  //
509 
510  RemoveEntryList(&Interrupt->InterruptListEntry);
511 
512  //
513  // If there's only one interrupt object left on this vector,
514  // determine proper interrupt dispatcher
515  //
516 
517  Interrupty = CONTAINING_RECORD(
518  DispatchInfo.Interrupt->InterruptListEntry.Flink,
519  KINTERRUPT,
520  InterruptListEntry
521  );
522 
523  if (DispatchInfo.Interrupt == Interrupty) {
524  KiConnectVectorAndInterruptObject (Interrupty, NormalConnect);
525  }
526 
527  } else {
528 
529  //
530  // Removing last interrupt object from the vector. Disable the
531  // vector, and set it to unconnected
532  //
533 
534  HalDisableSystemInterrupt(Interrupt->Vector, Irql);
535  KiConnectVectorAndInterruptObject (Interrupt, NoConnect);
536  }
537 
538 
539  KeSweepIcache(TRUE);
540  Interrupt->Connected = FALSE;
541  }
542 
543  //
544  // Unlock dispatcher database and lower IRQL to its previous value.
545  //
546 
547  KiUnlockDispatcherDatabase(OldIrql);
548 
549  //
550  // Set system affinity back to the original value.
551  //
552 
553  KeRevertToUserAffinityThread();
554 
555  //
556  // Return whether interrupt was disconnected from the specified vector.
557  //
558 
559  return Connected;
560 }

562 VOID
563 KiGetVectorInfo (
564  IN ULONG Vector,
565  OUT PDISPATCH_INFO DispatchInfo
566  )
567 {
568  PKINTERRUPT_ROUTINE Dispatch;
569  ULONG CurrentDispatch;
570  ULONG DispatchType;
571  UCHAR IDTEntry;
572 
573  //
574  // Get second level dispatch point
575  //
576 
577 
578  DispatchType = HalSystemVectorDispatchEntry (
579  Vector,
580  &DispatchInfo->FlatDispatch,
581  &DispatchInfo->NoDispatch
582  );
583 
584  //
585  // Get vector info
586  //
587 
588  switch (DispatchType) {
589  case 0:
590  //
591  // Primary dispatch
592  //
593 
594  IDTEntry = HalVectorToIDTEntry(Vector);
595  DispatchInfo->NoDispatch = (PKINTERRUPT_ROUTINE) (((ULONG) &KiStartUnexpectedRange) +
596  (IDTEntry - PRIMARY_VECTOR_BASE) * KiUnexpectedEntrySize);
597 
598  DispatchInfo->InterruptDispatch = KiInterruptDispatch;
599  DispatchInfo->FloatingDispatch = KiFloatingDispatch;
600  DispatchInfo->ChainedDispatch = KiChainedDispatch;
601  DispatchInfo->FlatDispatch = NULL;
602 
603  CurrentDispatch = (ULONG) KiReturnHandlerAddressFromIDT(Vector);
604  DispatchInfo->Interrupt = CONTAINING_RECORD (
605  CurrentDispatch,
606  KINTERRUPT,
607  DispatchCode
608  );
609  break;
610 
611  case 1:
612  //
613  // Secondary dispatch.
614  //
615 
616  DispatchInfo->InterruptDispatch = KiInterruptDispatch2ndLvl;
617  DispatchInfo->FloatingDispatch = KiInterruptDispatch2ndLvl;
618  DispatchInfo->ChainedDispatch = KiChainedDispatch2ndLvl;
619 
620  CurrentDispatch = (ULONG) *DispatchInfo->FlatDispatch;
621  DispatchInfo->Interrupt = (PKINTERRUPT) ( (PUCHAR) CurrentDispatch -
622  (PUCHAR) KiInterruptTemplate +
623  (PUCHAR) &KiInterruptTemplate2ndDispatch
624  );
625  break;
626 
627  default:
628  // Other values reserved
629  KeBugCheck (MISMATCHED_HAL);
630  }
631 
632 
633  //
634  // Determine dispatch type
635  //
636 
637  if (((PKINTERRUPT_ROUTINE) CurrentDispatch) == DispatchInfo->NoDispatch) {
638 
639  //
640  // Is connected to the NoDispatch function
641  //
642 
643  DispatchInfo->Type = NoConnect;
644 
645  } else {
646  Dispatch = DispatchInfo->Interrupt->DispatchAddress;
647 
648  if (Dispatch == DispatchInfo->ChainedDispatch) {
649  //
650  // Is connected to the chained handler
651  //
652 
653  DispatchInfo->Type = ChainConnect;
654 
655  } else if (Dispatch == DispatchInfo->InterruptDispatch ||
656  Dispatch == DispatchInfo->FloatingDispatch) {
657  //
658  // If connection to the non-chained handler
659  //
660 
661  DispatchInfo->Type = NormalConnect;
662 
663  } else {
664 
665  //
666  // Unknown connection
667  //
668 
669  DispatchInfo->Type = UnknownConnect;
670 #if DBG
671  DbgPrint ("KiGetVectorInfo not understood\n");
672 #endif
673  }
674  }
675 }

678 KiConnectVectorAndInterruptObject (
679  IN PKINTERRUPT Interrupt,
680  IN CONNECT_TYPE Type
681  )
682 {
683  PKINTERRUPT_ROUTINE DispatchAddress;
684  DISPATCH_INFO DispatchInfo;
685  PULONG pl;
686 
687  //
688  // Get current connect info
689  //
690 
691  KiGetVectorInfo (
692  Interrupt->Vector,
693  &DispatchInfo
694  );
695 
696  //
697  // If disconnecting, set vector to NoDispatch
698  //
699 
700  if (Type == NoConnect) {
701 
702  DispatchAddress = DispatchInfo.NoDispatch;
703 
704  } else {
705 
706  //
707  // Set interrupt objects dispatch for new type
708  //
709 
710  DispatchAddress = DispatchInfo.ChainedDispatch;
711 
712  if (Type == NormalConnect) {
713  DispatchAddress = DispatchInfo.InterruptDispatch;
714  if (Interrupt->FloatingSave) {
715  DispatchAddress = DispatchInfo.FloatingDispatch;
716  }
717  }
718 
719  Interrupt->DispatchAddress = DispatchAddress;
720 
721  //
722  // Set interrupt objects dispatch code to kernel dispatcher
723  //
724 
725  pl = &(Interrupt->DispatchCode[0]);
726  pl = (PULONG)((PUCHAR)pl +
727  ((PUCHAR)&KiInterruptTemplateDispatch -
728  (PUCHAR)KiInterruptTemplate) -4);
729 
730  *pl = (ULONG)DispatchAddress-(ULONG)((PUCHAR)pl+4);
731 
732  //
733  // Set dispatch vector to proper address dispatch code location
734  //
735 
736  if (DispatchInfo.FlatDispatch) {
737 
738  //
739  // Connect to flat dispatch
740  //
741 
742  DispatchAddress = (PKINTERRUPT_ROUTINE) (ULONG_PTR)
743  ((PUCHAR) &(Interrupt->DispatchCode[0]) +
744  ((PUCHAR) &KiInterruptTemplate2ndDispatch -
745  (PUCHAR) KiInterruptTemplate));
746 
747  } else {
748 
749  //
750  // Connect to enter_all dispatch
751  //
752 
753  DispatchAddress = (PKINTERRUPT_ROUTINE) (ULONG_PTR) &Interrupt->DispatchCode;
754  }
755  }
756 
757 
758  if (DispatchInfo.FlatDispatch) {
759 
760  //
761  // Connect to flat dispatch
762  //
763 
764  *DispatchInfo.FlatDispatch = DispatchAddress;
765 
766  } else {
767 
768  //
769  // Connect to IDT
770  //
771 
772  KiSetHandlerAddressToIDT (Interrupt->Vector, DispatchAddress);
773  }
774 }

776 VOID
777 FASTCALL
778 KiTimedChainedDispatch2ndLvl(
779  PKINTERRUPT Interrupt
780  )
781 
782 /*++
783 
784 Routine Description:
785 
786  This function performs the same function as KiChainedDispatch2ndLvl
787  except that it is written in C instead of assembly code and includes
788  code for timing ISRs.
789 
790  I'd be interested in seeing some benchmarks to show if the assembly
791  code is actually faster. The Acquire/Release spinlock could be
792  inlined fairly easily.
793 
794 Arguments:
795 
796  Interrupt - Supplies a pointer to a control object of type interrupt.
797 
798 Return Value:
799 
800  None.
801 
802 --*/
803 
804 {
805  BOOLEAN Handled = FALSE;
806  PVOID ListEnd = &Interrupt->InterruptListEntry.Flink;
807  //
808  //BEGINTIMING
809 
810  PKPRCB Prcb = KeGetCurrentPrcb();
811  ULONGLONG StartTimeHigher;
812  ULONGLONG StartTime;
813  ULONGLONG TimeHigher;
814  ULONGLONG ElapsedTime;
815 
816  //BEGINTIMINGend
817 
818 
819  //
820  // For each interrupt on this chain.
821  //
822 
823  do {
824 
825  //
826  // If the current IRQL (IRQL raised to by nature of taking this
827  // interrupt) is not equal to the Synchronization IRQL required
828  // for this interrupt, raise to the appropriate level.
829  //
830 
831  if (Interrupt->Irql != Interrupt->SynchronizeIrql) {
832  KfRaiseIrql(Interrupt->SynchronizeIrql);
833  }
834 
835  //BEGINTIMING
836 
837  StartTimeHigher = Prcb->IsrTime;
838  StartTime = RDTSC();
839 
840  //BEGINTIMINGend
841 
842  //
843  // Acquire the interrupt lock.
844  //
845 
846  KiAcquireSpinLock(Interrupt->ActualLock);
847 
848  //
849  // Call the Interrupt Service Routine.
850  //
851 
852  Handled |= Interrupt->ServiceRoutine(Interrupt,
853  Interrupt->ServiceContext);
854 
855  //
856  // Release the interrupt lock.
857  //
858 
859  KiReleaseSpinLock(Interrupt->ActualLock);
860 
861  //ENDTIMING
862 
863  //
864  // ElapsedTime is time since we started looking at this element
865  // on the chain. (ie the current interrupt object).
866  //
867 
868  ElapsedTime = RDTSC() - StartTime;
869 
870  //
871  // TimeHigher is the amount Prcb->IsrTime has increased since we
872  // begin servicing this interrupt object, ie the amount of time
873  // spent in higher level ISRs.
874  //
875 
876  TimeHigher = Prcb->IsrTime - StartTimeHigher;
877 
878  //
879  // Adjust ElapsedTime to time spent on this interrupt object, excluding
880  // higher level ISRs.
881  //
882 
883  ElapsedTime -= TimeHigher;
884  if (ElapsedTime > KiIsrTscLimit) {
885 
886  //
887  // If there is a debugger attached, breakin. Otherwise do nothing.
888  // N.B. bugchecking is another possibility.
889  //
890 
891  if (KdDebuggerEnabled) {
892  DbgPrint("KE; ISR time limit exceeded (intobj %p)\n",
893  Interrupt);
894  DbgBreakPoint();
895  }
896  }
897 
898  //
899  // Update time spent processing interrupts. This doesn't need
900  // to be atomic as it doesn't matter if it's a little bit lossy.
901  // (Though a simple atomic add would do, it's per processor and
902  // at IRQL > DISPATCH_LEVEL so it doesn't need to be locked).
903  //
904 
905  Prcb->IsrTime += ElapsedTime;
906 
907  //ENDTIMINGend
908 
909  //
910  // If IRQL was raised, lower to the previous level.
911  //
912
