63  //
64  // o Specifies the device itself (file name == ""),
65  // o specifies that this is an OPEN operation,
66  // o and does not ask to create a directory.
67  //
68

70  Executive,
71  KernelMode,
72  FALSE,
73  (PLARGE_INTEGER) NULL );
74  ASSERT( NT_SUCCESS( Status ) );
75

77  // Don't allow any relative opens as well as opens with a filename. These opens have
78  // only been checked for traverse access by the I/O manager.
79  //
80  if (((IrpSp->FileObject == NULL) || ((IrpSp->FileObject->FileName.Length == 0) &&
81  IrpSp->FileObject->RelatedFileObject == NULL)) &&
82  ((IrpSp->Parameters.Create.Options >> 24) == FILE_OPEN) &&
83  ((IrpSp->Parameters.Create.Options & FILE_DIRECTORY_FILE) == 0)) {
84 
85  //
86  // If the volume is locked or dismounted we cannot open it again.
87  //

104  //
