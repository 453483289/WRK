153 NTSTATUS
154 HvpAdjustBitmap(
155  IN PHHIVE Hive,
156  IN ULONG HiveLength,
157  IN OUT PFREE_DISPLAY FreeDisplay
158  )
159 /*++
160 
161 Routine Description:
162 
163  When the length of the hive grows/shrinks, adjust the bitmap accordingly.
164  - allocates a bitmap buffer large enough.
165  - copies the relevant information from the old bitmap.
166 
167 Arguments:
168 
169  Hive - used for quota tracking.
170 
171  HiveLength - the new length of the hive.
172 
173  Bitmap - bitmap to operate on.
174 
175 Return Value:
176 
177  NTSTATUS code.
178 
179 --*/
180 {
181  ULONG VectorSize;
182  ULONG NewBufferSize;
183  ULONG OldBufferSize;
184  PULONG Vector;
185  PULONG OldVector;
186  ULONG OldVectorSize;
187  PRTL_BITMAP Bitmap;
188 
189  CM_PAGED_CODE();
190 
191  Bitmap = &(FreeDisplay->Display);
192 
193  VectorSize = HiveLength / HBLOCK_SIZE; // Vector size == bits
194 
195  NewBufferSize = ROUND_UP_NOZERO( (VectorSize + 7) / 8,ROUND_INCREMENTS); // BufferSize == Bytes
196 
197  if( Bitmap->SizeOfBitMap == 0 ) {
198  OldBufferSize = 0;
199  } else {
200  OldBufferSize = ROUND_UP_NOZERO( (Bitmap->SizeOfBitMap + 7) / 8, ROUND_INCREMENTS);
201  }
202 
203  if( NewBufferSize <= FreeDisplay->RealVectorSize ) {
204  //
205  // We don't shrink the vector; next time we grow, we'll perform
206  // the adjustments
207  //
208 
209 
210  //
211  // Clear all the unused bits and return;
212  //
213  // we don't really need to do this as nobody will write in here
214  // we'll drop it in the final implementation
215  //
216  OldVectorSize = Bitmap->SizeOfBitMap;
217  //
218  // set the new size
219  //
220  RtlInitializeBitMap(Bitmap,Bitmap->Buffer,VectorSize);
221  if( OldVectorSize < VectorSize ) {
222  RtlClearBits (Bitmap,OldVectorSize,VectorSize - OldVectorSize);
223  }
224  return STATUS_SUCCESS;
225  }
226  //
227  // else, the bitmap has enlarged. Allocate a new buffer and copy the bits already set.
228  //
229  Vector = (PULONG)((Hive->Allocate)(NewBufferSize, TRUE,CM_FIND_LEAK_TAG39));
230  if (Vector == NULL) {
231  return STATUS_INSUFFICIENT_RESOURCES;
232  }
233 
234  FreeDisplay->RealVectorSize = NewBufferSize;
235 
236  OldVector = Bitmap->Buffer;
237  RtlZeroMemory(Vector,NewBufferSize);
238  RtlInitializeBitMap(Bitmap, Vector, VectorSize);
239 
240  if( OldVector != NULL ) {
241  //
242  // copy the already set bits
243  //
244  RtlCopyMemory (Vector,OldVector,OldBufferSize);
245 
246  //
247  // Free the old vector
248  //
249  (Hive->Free)(OldVector, OldBufferSize);
250  }
251 
252  return STATUS_SUCCESS;
253 }

307 VOID
308 HvpRemoveFreeCellHint(
309  PHHIVE Hive,
310  HCELL_INDEX Cell,
311  ULONG Index,
312  HSTORAGE_TYPE Type
313  )
314 /*++
315 
316 Routine Description:
317 
318  Clears the corresponding bit in the bitmap
319 
320 Arguments:
321 
322  Hive - hive operating on
323 
324  Cell - free cell
325 
326  Index - index in FreeDisplay (based on the free cell size)
327 
328  Type - storage type (Stable or Volatile)
329 
330 Return Value:
331 
332  VOID
333 
334 --*/
335 {
336  ULONG BinIndex;
337  ULONG TempIndex;
338  PHMAP_ENTRY Me;
339  PHBIN Bin;
340  ULONG CellOffset;
341  ULONG Size;
342  PHCELL p;
343  BOOLEAN CellFound = FALSE;
344 
345  CM_PAGED_CODE();
346 
347  ASSERT_HIVE_WRITER_LOCK_OWNED((PCMHIVE)Hive);
348  ASSERT_HIVE_FLUSHER_LOCKED((PCMHIVE)Hive);
349 
350  Me = HvpGetCellMap(Hive, Cell);
351  VALIDATE_CELL_MAP(__LINE__,Me,Hive,Cell);
352 
353  Bin = (PHBIN)HBIN_BASE(Me->BinAddress);
354 
355  CellOffset = Bin->FileOffset + sizeof(HBIN);
356 
357 
358  //
359  // we ned to be protected against exception raised by the FS while faulting in data
360  //
361  try {
362 
363  //
364  // There is a chance we can find a suitable free cell
365  //
366 
367  p = (PHCELL)((PUCHAR)Bin + sizeof(HBIN));
368 
369  while (p < (PHCELL)((PUCHAR)Bin + Bin->Size)) {
370 
371  //
372  // if free cell, check it out, add it to free list for hive
373  //
374  if (p->Size >= 0) {
375 
376  Size = (ULONG)p->Size;
377 
378  HvpComputeIndex(TempIndex, Size);
379  if ((Index == TempIndex) && (CellOffset != (Cell&(~HCELL_TYPE_MASK)) )) {
380  //
381  // there is at least one free cell of this size (this one)
382  // different than the one being delisted
383  //
384  CellFound = TRUE;
385  break;
386  }
387 
388  } else {
389  //
390  // used cell
391  //
392  Size = (ULONG)(p->Size * -1);
393 
394  }
395 
396  ASSERT( ((LONG)Size) >= 0);
397  p = (PHCELL)((PUCHAR)p + Size);
398  CellOffset += Size;
399  }
400 
401  } except (EXCEPTION_EXECUTE_HANDLER) {
402  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"HvpRemoveFreeCellHint: exception thrown ehile faulting in data, code:%08lx\n", GetExceptionCode()));
403  //
404  // better not use cells in this range rather than leaving false hints
405  //
406  CellFound = FALSE;
407  }
408 
409  if( CellFound == FALSE ) {
410  //
411  // no cell with this index was found
412  // compute the bin index and for the beginning of the bin
413  //
414  BinIndex = Bin->FileOffset / HBLOCK_SIZE;
415 
416  RtlClearBits (&(Hive->Storage[Type].FreeDisplay[Index].Display), BinIndex, Bin->Size / HBLOCK_SIZE);
417  }
418 
419  if( RtlNumberOfSetBits(&(Hive->Storage[Type].FreeDisplay[Index].Display) ) != 0 ) {
420  //
421  // there are still some other free cells of this size
422  //
423  Hive->Storage[Type].FreeSummary |= (1 << Index);
424  } else {
425  //
426  // entire bitmap is 0 (i.e. no other free cells of this size)
427  //
428  Hive->Storage[Type].FreeSummary &= (~(1 << Index));
429  }
430 }

433 HvpFindFreeCellInBin(
434  PHHIVE Hive,
435  ULONG Index,
436  ULONG NewSize,
437  HSTORAGE_TYPE Type,
438  PHBIN Bin
439  )
440 /*++
441 
442 Routine Description:
443 
444  Lookup for a free cell with the size NewSize in this particular bin
445 
446 Arguments:
447 
448  Hive - target hive.
449 
450  Index - index in FreeDisplay (based on the free cell size)
451 
452  NewSize - desired size
453 
454  Type - storage type (Stable or Volatile)
455 
456  Bin - Bin in question
457 
458 Return Value:
459 
460  A free cellindex with a size bigger than NewSize, or HCELL_NIL
461 
462 --*/
463 {
464 
465  ULONG BinIndex;
466  ULONG CellOffset;
467  PHCELL p;
468  ULONG BinOffset;
469  ULONG Size;
470  HCELL_INDEX cellindex;
471  ULONG FoundCellIndex;
472 
473  CM_PAGED_CODE();
474 
475  ASSERT_HIVE_WRITER_LOCK_OWNED((PCMHIVE)Hive);
476 
477  BinOffset = Bin->FileOffset;
478  BinIndex = BinOffset/HBLOCK_SIZE;
479 
480  if( RtlCheckBit(&(Hive->Storage[Type].FreeDisplay[Index].Display), BinIndex) == 0 ) {
481  //
482  // no hint for this bin
483  //
484  return HCELL_NIL;
485  }
486 
487  CellOffset = sizeof(HBIN);
488 
489  //
490  // we ned to be protected against exception raised by the FS while faulting in data
491  //
492  try {
493 
494  //
495  // There is a chance we can find a suitable free cell
496  //
497  p = (PHCELL)((PUCHAR)Bin + sizeof(HBIN));
498 
499  while (p < (PHCELL)((PUCHAR)Bin + Bin->Size)) {
500 
501  //
502  // if free cell, check it out, add it to free list for hive
503  //
504  if (p->Size >= 0) {
505 
506  Size = (ULONG)p->Size;
507 
508  //
509  // cell is free, and is not obviously corrupt, add to free list
510  //
511  CellOffset = (ULONG)((PUCHAR)p - (PUCHAR)Bin);
512  cellindex = BinOffset + CellOffset + (Type*HCELL_TYPE_MASK);
513 
514  if (NewSize <= (ULONG)Size) {
515  //
516  // Found a big enough cell.
517  //
518  HvpComputeIndex(FoundCellIndex, Size);
519  if( Index == FoundCellIndex ) {
520  //
521  // and enlisted at the same index (we want to avoid fragmentation if possible!)
522  //
523 
524  if (! HvMarkCellDirty(Hive, cellindex,TRUE)) {
525  return HCELL_NIL;
526  }
527 
528  HvpDelistFreeCell(Hive, cellindex, Type);
529 
530  ASSERT(p->Size > 0);
531  ASSERT(NewSize <= (ULONG)p->Size);
532  return cellindex;
533  }
534  }
535 
536  } else {
537  //
538  // used cell
539  //
540  Size = (ULONG)(p->Size * -1);
541 
542  }
543 
544  ASSERT( ((LONG)Size) >= 0);
545  p = (PHCELL)((PUCHAR)p + Size);
546  }
547  } except (EXCEPTION_EXECUTE_HANDLER) {
548  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"HvpFindFreeCellInBin: exception thrown ehile faulting in data, code:%08lx\n", GetExceptionCode()));
549  return HCELL_NIL;
550  }
551 
552  //
553  // no free cell matching this size on this bin ; We did all this work for nothing!
554  //
555  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"[HvpFindFreeCellInBin] (Offset,Size) = (%lx,%lx) ==> No Match\n",BinOffset,Bin->Size));
556  return HCELL_NIL;
557 }

560 HvpScanForFreeCellInViewWindow(
561  PHHIVE Hive,
562  ULONG Index,
563  ULONG NewSize,
564  HSTORAGE_TYPE Type,
565  HCELL_INDEX FileOffsetStart
566  )
567 /*++
568 
569 Routine Description:
570 
571  Lookup for a free cell with the size NewSize in the CM_VIEW_SIZE window defined by
572  Vicinity.
573 
574  If it doesn't find a free cell for the specifed index, tries with the
575 
576 
577 Arguments:
578 
579  Hive - target hive.
580 
581  Index - index in FreeDisplay (based on the free cell size)
582 
583  NewSize - desired size
584 
585  Type - storage type (Stable or Volatile)
586 
587  Vicinity - defines the window; it is never HCELL_NIL !!!
588 
589 Return Value:
590 
591  A free cellindex with a size bigger than NewSize, or HCELL_NIL
592 
593 Note:
594 
595  Vicinity is a physical file offset at this point. we need to
596  convert it to a logical one prior to accessing the map
597 --*/
598 {
599  ULONG FileOffsetEnd;
600  HCELL_INDEX Cell;
601  PHMAP_ENTRY Me;
602  PHBIN Bin;
603  PFREE_HBIN FreeBin;
604  ULONG BinFileOffset;
605  ULONG BinSize;
606  PCM_VIEW_OF_FILE CmView;
607 
608  CM_PAGED_CODE();
609 
610  FileOffsetEnd = FileOffsetStart + CM_VIEW_SIZE;
611  FileOffsetEnd -= HBLOCK_SIZE;
612  if( FileOffsetStart != 0 ) {
613  FileOffsetStart -= HBLOCK_SIZE;
614  }
615  if( FileOffsetEnd > Hive->Storage[Type].Length ) {
616  FileOffsetEnd = Hive->Storage[Type].Length;
617  }
618 
619  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"\t[HvpScanForFreeCellInViewWindow] (Start,End) = (%lx,%lx) Size = %lx\n",FileOffsetStart,FileOffsetEnd,Hive->Storage[Type].Length));
620 
621  //
622  // sanity ASSERT
623  //
624  ASSERT( FileOffsetStart < FileOffsetEnd );
625 
626 
627  //
628  // the caller already checked for this; remember, hints are for real!
629  //
630  ASSERT( !RtlAreBitsClear( &(Hive->Storage[Type].FreeDisplay[Index].Display),
631  FileOffsetStart/HBLOCK_SIZE,(FileOffsetEnd - FileOffsetStart) / HBLOCK_SIZE) );
632 
633  while( FileOffsetStart < FileOffsetEnd ) {
634  Cell = FileOffsetStart + (Type*HCELL_TYPE_MASK);
635  Me = HvpGetCellMap(Hive, Cell);
636  VALIDATE_CELL_MAP(__LINE__,Me,Hive,Cell);
637  CmView = NULL;
638 
639  //
640  // skip discarded bins
641  //
642  if(Me->BinAddress & HMAP_DISCARDABLE) {
643  FreeBin = (PFREE_HBIN)Me->BlockAddress;
644  if( FreeBin->FileOffset == FileOffsetStart ) {
645  FileOffsetStart += FreeBin->Size;
646  } else {
647  //
648  // the bin does not start in this window;
649  // skip to the next bin in this window
650  //
651  FileOffsetStart = FreeBin->FileOffset + FreeBin->Size;
652  }
653  continue;
654  }
655 
656  //
657  // this will add a reference on the view so it doesn't get away from under us.
658  //
659  CmLockHiveViews(Hive);
660  if((Me->BinAddress & (HMAP_INVIEW|HMAP_INPAGEDPOOL)) == 0) {
661  //
662  // bin is not mapped, map it now!!!
663  // do not touch the view as we may iterate through
664  // the hole hive; this will keep the view for this window
665  // mapped, as we hold the registry lock exclusive
666  //
667  if( !NT_SUCCESS(CmpMapThisBin((PCMHIVE)Hive,Cell,FALSE)) ) {
668  //
669  // cannot map bin due to insufficient resources
670  //
671  CmUnlockHiveViews(Hive);
672  return HCELL_NIL;
673  }
674  ASSERT( Me->BinAddress & HMAP_INVIEW );
675  ASSERT( Me->CmView != NULL );
676  //
677  // this will prevent the view from going away while we are working on the bin
678  //
679  CmpReferenceHiveView((PCMHIVE)Hive,(CmView = Me->CmView));
680  }
681  CmUnlockHiveViews(Hive);
682 
683  Bin = (PHBIN)HBIN_BASE(Me->BinAddress);
684 
685  //
686  // we need to protect against in-page-errors thrown by mm while faulting in data
687  //
688  try {
689  BinFileOffset = Bin->FileOffset;
690  BinSize = Bin->Size;
691  } except (EXCEPTION_EXECUTE_HANDLER) {
692  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"HvpScanForFreeCellInViewWindow: exception thrown while faulting in data, code:%08lx\n", GetExceptionCode()));
693  CmpDereferenceHiveViewWithLock((PCMHIVE)Hive,CmView);
694  return HCELL_NIL;
695  }
696 
697  if( BinFileOffset == FileOffsetStart ) {
698 
699  Cell = HvpFindFreeCellInBin(Hive,Index,NewSize,Type,Bin);
700  if( Cell != HCELL_NIL ) {
701  //found it!
702  CmpDereferenceHiveViewWithLock((PCMHIVE)Hive,CmView);
703  return Cell;
704  }
705 
706  FileOffsetStart += BinSize;
707  } else {
708  //
709  // bin does not start in this CM_VIEW_SIZE window; skip to the next bin in this window
710  //
711  FileOffsetStart = BinFileOffset + BinSize;
712  }
713  CmpDereferenceHiveViewWithLock((PCMHIVE)Hive,CmView);
714  }
715 
716  //
717  // no free cell matching this size on the CM_VIEW_SIZE window ; We did all this work for nothing!
718  //
719  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"[HvpScanForFreeCellInViewWindow] (Start,End) = (%lx,%lx) ==> No Match\n",FileOffsetStart,FileOffsetEnd));
720  return HCELL_NIL;
721 }

731 /*++
732 
733 Routine Description:
734 
735  Lookup for a free cell with the size NewSize in the window defined by
736  Vicinity.
737 
738  If it doesn't find a free cell for the specifed index, tries with the
739 
740 
741 Arguments:
742 
743  Hive - target hive.
744 
745  Index - index in FreeDisplay (based on the free cell size)
746 
747  NewSize - desired size
748 
749  Type - storage type (Stable or Volatile)
750 
751  Vicinity - defines the window; it is never HCELL_NIL !!!
752 
753 Return Value:
754 
755  A free cellindex with a size bigger than NewSize, or HCELL_NIL
756 
757 Note:
758 
759  Vicinity is a logical file offset at this point. This function
760  converts it to a physical one, and HvpScanForFindFreeCellInViewWindow
761  converts it back to logical prior to getting the cell map.
762 
763 --*/
764 {
765  HCELL_INDEX Cell;
766  ULONG FileOffsetStart;
767  ULONG FileOffsetEnd;
768  ULONG VicinityViewOffset;
769  ULONG Summary;
770  ULONG Offset;
771  ULONG RunLength;
772 
773  CM_PAGED_CODE();
774 
775  ASSERT( Vicinity != HCELL_NIL );
776 
777  VicinityViewOffset = ((Vicinity&(~HCELL_TYPE_MASK)) + HBLOCK_SIZE) & (~(CM_VIEW_SIZE - 1));
778  FileOffsetStart = VicinityViewOffset & (~(CM_VIEW_SIZE - 1));
779 
780  FileOffsetEnd = FileOffsetStart + CM_VIEW_SIZE;
781  if( FileOffsetEnd > (Hive->Storage[Type].Length + HBLOCK_SIZE) ) {
782  FileOffsetEnd = Hive->Storage[Type].Length + HBLOCK_SIZE;
783  }
784 
785  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"[HvpFindFreeCellInThisViewWindow] Vicinity = %lx (Start,End) = (%lx,%lx) Size = %lx\n",Vicinity,FileOffsetStart,FileOffsetEnd,Hive->Storage[Type].Length));
786 
787  //
788  // sanity assert
789  //
790  ASSERT( FileOffsetStart < FileOffsetEnd );
791 
792  //
793  // at this point the offset is physical (file-oriented, i.e. it is
794  // translated with HBLOCK_SIZE; HvpScanForFreeCellInViewWindow will do the
795  // reverse computation to adjust the offset)
796  //
797 
798  //
799  // Compute Summary vector of Display entries that are non null
800  //
801  Summary = Hive->Storage[Type].FreeSummary;
802  Summary = Summary & ~((1 << Index) - 1);
803  //
804  // We now have a summary of lists that are non-null and may
805  // contain entries large enough to satisfy the request.
806  // Iterate through the list and pull the first cell that is
807  // big enough. If no cells are big enough, advance to the
808  // next non-null list.
809  //
810  ASSERT(HHIVE_FREE_DISPLAY_SIZE == 24);
811 
812  Offset = FileOffsetStart?(FileOffsetStart-HBLOCK_SIZE):0;
813  RunLength = FileOffsetEnd - FileOffsetStart;
814  if( FileOffsetStart == 0 ) {
815  //
816  // first run is one block shorter !
817  //
818  RunLength -= HBLOCK_SIZE;
819  }
820  Offset /= HBLOCK_SIZE;
821  RunLength /= HBLOCK_SIZE;
822 
823  while (Summary != 0) {
824  if (Summary & 0xff) {
825  Index = CmpFindFirstSetRight[Summary & 0xff];
826  } else if (Summary & 0xff00) {
827  Index = CmpFindFirstSetRight[(Summary & 0xff00) >> 8] + 8;
828  } else {
829  ASSERT(Summary & 0xff0000);
830  Index = CmpFindFirstSetRight[(Summary & 0xff0000) >> 16] + 16;
831  }
832 
833  //
834  // we go down this path only if we have any hints
835  //
836  if( !RtlAreBitsClear( &(Hive->Storage[Type].FreeDisplay[Index].Display),Offset,RunLength) ) {
837 
838  //
839  // we have a reason to scan this view
840  //
841  Cell = HvpScanForFreeCellInViewWindow(Hive,Index,NewSize,Type,VicinityViewOffset);
842  if( Cell != HCELL_NIL ) {
843  // found it
844  return Cell;
845  }
846 
847  //
848  // if we got here, the hints are invalid
849  //
850  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"[HvpFindFreeCellInThisViewWindow] (Start,End) = (%lx,%lx) Offset = %lx RunLength = %lx\n",FileOffsetStart,FileOffsetEnd,Offset,RunLength));
851 
852  }
853  //
854  // No suitable cell was found of this size.
855  // Clear the bit in the summary and try the
856  // next biggest size
857  //
858  ASSERT(Summary & (1 << Index));
859  Summary = Summary & ~(1 << Index);
860  }
861 
862  return HCELL_NIL;
863 }

866 HvpFindFreeCell(
867  PHHIVE Hive,
868  ULONG Index,
869  ULONG NewSize,
870  HSTORAGE_TYPE Type,
871  HCELL_INDEX Vicinity
872  )
873 /*++
874 
875 Routine Description:
876 
877  Lookup for a free cell. First try is in the CM_VIEW_SIZE window defined
878  by Vicinity. If no free cell is found in this window (or vicinity
879  if NIL), entire hive is searched (window by window).
880 
881 Arguments:
882 
883  Hive - target hive.
884 
885  Index - index in FreeDisplay (based on the free cell size)
886 
887  NewSize - desired size
888 
889  Type - storage type (Stable or Volatile)
890 
891  Vicinity - defines the window.
892 
893 Return Value:
894 
895  A free cellindex with a size bigger than NewSize, or HCELL_NIL
896 
897 Optimization:
898 
899  When Vicinity is HCELL_NIL or if a cell is not found in the same window
900  as the vicinity, we don't really care where the cell gets allocated.
901  So, rather than iterating the whole hive, is a good idea to search first
902  in the pinned view list, then in the mapped view list, and at the end
903  in the rest of unmapped views.
904 
905 --*/
906 {
907  HCELL_INDEX Cell = HCELL_NIL;
908  ULONG FileOffset = 0;
909  PCMHIVE CmHive;
910 
911  CM_PAGED_CODE();
912 
913  CmHive = (PCMHIVE)CONTAINING_RECORD(Hive, CMHIVE, Hive);
914 
915  ASSERT_HIVE_WRITER_LOCK_OWNED(CmHive);
916  ASSERT_HIVE_FLUSHER_LOCKED(CmHive);
917 #if DBG
918  {
919  UNICODE_STRING HiveName;
920  RtlInitUnicodeString(&HiveName, (PCWSTR)Hive->BaseBlock->FileName);
921  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"[HvpFindFreeCell] CellSize = %lu Vicinity = %lx :: Hive (%p) (%.*S) ...",NewSize,Vicinity,Hive,HiveName.Length / sizeof(WCHAR),HiveName.Buffer));
922  }
923 #endif
924 
925  //
926  // Vicinity should have the same storage as the new cell !
927  //
928  ASSERT( (Vicinity == HCELL_NIL) || (HvGetCellType(Vicinity) == (ULONG)Type) );
929 
930  //
931  // we have the lock exclusive or nobody is operating inside this hive
932  //
933  ASSERT_HIVE_WRITER_LOCK_OWNED((PCMHIVE)Hive);
934 
935  if( (Vicinity != HCELL_NIL) && (CmHive->GrowOnlyMode == FALSE) ) {
936  //
937  // try first in this window
938  //
939  Cell = HvpFindFreeCellInThisViewWindow(Hive,Index,NewSize,Type,Vicinity);
940  }
941 
942  if( Cell != HCELL_NIL ) {
943  //
944  // found it!!!
945  //
946  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"found cell %lx \n",Cell));
947  return Cell;
948  }
949 
950  //
951  // We have to search the entire hive
952  //
953 
954  while( FileOffset < Hive->Storage[Type].Length ) {
955  //
956  // don't search again in the vicinity window
957  // we already did it once
958  //
959  if( ( ((CmHive->GrowOnlyMode == FALSE) || (Type == Volatile)) &&
960  ((Vicinity == HCELL_NIL) || (HvpCheckViewBoundary(FileOffset,Vicinity&(~HCELL_TYPE_MASK)) == FALSE)) ) ||
961  ( (CmHive->GrowOnlyMode == TRUE) && (FileOffset >= CmHive->GrowOffset) )
962  ) {
963  //
964  // search in this window
965  //
966  Cell = FileOffset + (Type*HCELL_TYPE_MASK);
967  Cell = HvpFindFreeCellInThisViewWindow(Hive,Index,NewSize,Type,Cell);
968  if( Cell != HCELL_NIL ) {
969  //
970  // found it!
971  //
972  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"found cell %lx \n",Cell));
973  return Cell;
974  }
975  }
976 
977  //
978  // advance to the new window
979  //
980  if( FileOffset == 0) {
981  // account for the base block
982  FileOffset += (CM_VIEW_SIZE - HBLOCK_SIZE);
983  } else {
984  FileOffset += CM_VIEW_SIZE;
985  }
986  }
987 
988  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_FREECELL,"No cell found.\n"));
989 
990  return HCELL_NIL;
991 }
