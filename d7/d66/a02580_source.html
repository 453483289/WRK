69 /*++
70 
71 Routine Description:
72 
73  This function is used to queue a user-mode APC to the specified thread. The APC
74  will fire when the specified thread does an alertable wait
75 
76 Arguments:
77 
78  ThreadHandle - Supplies a handle to a thread object. The caller
79  must have THREAD_SET_CONTEXT access to the thread.
80 
81  ApcRoutine - Supplies the address of the APC routine to execute when the
82  APC fires.
83 
84  ApcArgument1 - Supplies the first PVOID passed to the APC
85 
86  ApcArgument2 - Supplies the second PVOID passed to the APC
87 
88  ApcArgument3 - Supplies the third PVOID passed to the APC
89 
90 Return Value:
91 
92  Returns an NT Status code indicating success or failure of the API
93 
94 --*/
95 
96 {
97  PETHREAD Thread;
98  NTSTATUS st;
99  KPROCESSOR_MODE Mode;
100  PKAPC Apc;
101 
102  PAGED_CODE();
103 
104  Mode = KeGetPreviousMode ();
105 
106  st = ObReferenceObjectByHandle (ThreadHandle,
107  THREAD_SET_CONTEXT,
108  PsThreadType,
109  Mode,
110  &Thread,
111  NULL);
112  if (NT_SUCCESS (st)) {
113  st = STATUS_SUCCESS;
114  if (IS_SYSTEM_THREAD (Thread)) {
115  st = STATUS_INVALID_HANDLE;
116  } else {
117  Apc = ExAllocatePoolWithQuotaTag (NonPagedPool | POOL_QUOTA_FAIL_INSTEAD_OF_RAISE,
118  sizeof(*Apc),
119  'pasP');
120 
121  if (Apc == NULL) {
122  st = STATUS_NO_MEMORY;
123  } else {
124  KeInitializeApc (Apc,
125  &Thread->Tcb,
126  OriginalApcEnvironment,
127  PspQueueApcSpecialApc,
128  NULL,
129  (PKNORMAL_ROUTINE)ApcRoutine,
130  UserMode,
131  ApcArgument1);
132 
133  if (!KeInsertQueueApc (Apc, ApcArgument2, ApcArgument3, 0)) {
134  ExFreePool (Apc);
135  st = STATUS_UNSUCCESSFUL;
136  }
137  }
138  }
139  ObDereferenceObject (Thread);
140  }
141 
142  return st;
143 }

147 PsGetContextThread(
148  __in PETHREAD Thread,
149  __inout PCONTEXT ThreadContext,
150  __in KPROCESSOR_MODE Mode

153 /*++
154 
155 Routine Description:
156 
157  This function returns the usermode context of the specified thread. This
158  function will fail if the specified thread is a system thread. It will
159  return the wrong answer if the thread is a non-system thread that does
160  not execute in user-mode.
161 
162 Arguments:
163 
164  Thread - Supplies a pointer to the thread object from
165  which to retrieve context information.
166 
167  ThreadContext - Supplies the address of a buffer that will receive
168  the context of the specified thread.
169 
170  Mode - Mode to use for validation checks.
171 
172 Return Value:
173 
174  None.
175 
176 --*/
177 
178 {
179 
180  ULONG ContextFlags=0;
181  GETSETCONTEXT ContextFrame = {0};
182  ULONG ContextLength=0;
183  NTSTATUS Status;
184  PETHREAD CurrentThread;
185 
186  PAGED_CODE();
187 
188  Status = STATUS_SUCCESS;
189 
190  //
191  // Get previous mode and reference specified thread.
192  //
193 
194  CurrentThread = PsGetCurrentThread ();
195 
196  //
197  // Attempt to get the context of the specified thread.
198  //
199 
200  try {
201 
202  //
203  // Set the default alignment, capture the context flags,
204  // and set the default size of the context record.
205  //
206 
207  if (Mode != KernelMode) {
208  ProbeForReadSmallStructure (ThreadContext,
209  FIELD_OFFSET (CONTEXT, ContextFlags) + sizeof (ThreadContext->ContextFlags),
210  CONTEXT_ALIGN);
211  }
212 
213  ContextFlags = ThreadContext->ContextFlags;
214 
215  //
216  // We don't need to re-probe here so long as the structure is smaller
217  // than the guard region
218  //
219  ContextLength = sizeof(CONTEXT);
220  ASSERT (ContextLength < 0x10000);
221 
222 #if defined(_X86_)
223  //
224  // CONTEXT_EXTENDED_REGISTERS is SET, then we want sizeof(CONTEXT) set above
225  // otherwise (not set) we only want the old part of the context record.
226  //
227  if ((ContextFlags & CONTEXT_EXTENDED_REGISTERS) != CONTEXT_EXTENDED_REGISTERS) {
228  ContextLength = FIELD_OFFSET(CONTEXT, ExtendedRegisters);
229  }
230 #endif
231 
232  } except (EXCEPTION_EXECUTE_HANDLER) {
233  return GetExceptionCode ();
234  }
235 
236  KeInitializeEvent (&ContextFrame.OperationComplete,
237  NotificationEvent,
238  FALSE);
239 
240  ContextFrame.Context.ContextFlags = ContextFlags;
241 
242  ContextFrame.Mode = Mode;
243  if (Thread == CurrentThread) {
244  ContextFrame.Apc.SystemArgument1 = NULL;
245  ContextFrame.Apc.SystemArgument2 = Thread;
246  KeEnterGuardedRegionThread (&CurrentThread->Tcb);
247  PspGetSetContextSpecialApc (&ContextFrame.Apc,
248  NULL,
249  NULL,
250  &ContextFrame.Apc.SystemArgument1,
251  &ContextFrame.Apc.SystemArgument2);
252 
253  KeLeaveGuardedRegionThread (&CurrentThread->Tcb);
254 
255  //
256  // Move context to specfied context record. If an exception
257  // occurs, then return the error.
258  //
259 
260  try {
261  RtlCopyMemory (ThreadContext,
262  &ContextFrame.Context,
263  ContextLength);
264 
265  } except (EXCEPTION_EXECUTE_HANDLER) {
266  Status = GetExceptionCode ();
267  }
268 
269  } else {
270  KeInitializeApc (&ContextFrame.Apc,
271  &Thread->Tcb,
272  OriginalApcEnvironment,
273  PspGetSetContextSpecialApc,
274  NULL,
275  NULL,
276  KernelMode,
277  NULL);
278 
279  if (!KeInsertQueueApc (&ContextFrame.Apc, NULL, Thread, 2)) {
280  Status = STATUS_UNSUCCESSFUL;
281 
282  } else {
283  KeWaitForSingleObject (&ContextFrame.OperationComplete,
284  Executive,
285  KernelMode,
286  FALSE,
287  NULL);
288  //
289  // Move context to specfied context record. If an
290  // exception occurs, then silently handle it and
291  // return success.
292  //
293 
294  try {
295  RtlCopyMemory (ThreadContext,
296  &ContextFrame.Context,
297  ContextLength);
298 
299  } except (EXCEPTION_EXECUTE_HANDLER) {
300  Status = GetExceptionCode ();
301  }
302  }
303  }
304 
305  return Status;
306 }

316 Routine Description:
317 
318  This function returns the usermode context of the specified thread. This

320  return the wrong answer if the thread is a non-system thread that does

322

324

334  None.
335 
336 --*/
337 
338 {

340  KPROCESSOR_MODE Mode;
341  NTSTATUS Status;
342  PETHREAD Thread;
343  PETHREAD CurrentThread;
344 
345  PAGED_CODE();
346 
347  //
348  // Get previous mode and reference specified thread.
349  //
350 
351  CurrentThread = PsGetCurrentThread ();
352  Mode = KeGetPreviousModeByThread (&CurrentThread->Tcb);
353 
354  Status = ObReferenceObjectByHandle (ThreadHandle,
355  THREAD_GET_CONTEXT,
356  PsThreadType,
357  Mode,
358  &Thread,
359  NULL);
360

363  // is a system thread.
364  //

611 /*++
612 
613 Routine Description:
614 
615  This routine is used by kernel mode callers to queue APCs to a 32-bit thread
616  running inside a Wow64 process. It wraps the original APC routine
617  with a wrapper routine inside Wow64. The target Apc routine must be
618  inside 32-bit code. This routine must be executed in the context of the
619  Wow64 thread where the target APC is going to run in. The resulting ApcContext
620  and ApcRoutine from this routine are used when initializing, using KeInitializeApc,
621  to be queued later to a Wow64 thread.
622 
623  The API does nothing and succeeds on :
624  - 32-bit systems.
625  - native 64-bit processes on 64-bit systems.
626 
627 Environment:
628 
629  Kernel mode only.
630 
631 Arguments:
632 
633  ApcContext - Pointer to the original ApcContext parameter.
634 
635  ApcRoutine - Pointer to the original ApcRoutine that is targeted to run
636  32-bit code.
637 
638 Return:
639 
640  NTSTATUS.
641 
642 --*/
643 
644 {
645 
646  NTSTATUS NtStatus;
647 
648  NtStatus = STATUS_SUCCESS;
649 
650 #if defined(_WIN64)
651 
652  if (PsGetCurrentProcess ()->Wow64Process != NULL) {
653 
654  try {
655  Wow64KiWrapApcProc (ApcContext, ApcRoutine);
656 
657  } except (EXCEPTION_EXECUTE_HANDLER) {
658  NtStatus = GetExceptionCode ();
659  }
660  }
661 
662 #else
663 
664  UNREFERENCED_PARAMETER(ApcContext);
665  UNREFERENCED_PARAMETER(ApcRoutine);
666 
667 #endif
668 
669  return NtStatus;
670 }
