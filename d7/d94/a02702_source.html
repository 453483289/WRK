47 /*++
48 
49 Routine Description:
50 
51  This routine takes a snapshot of the calling thread's security
52  context (locking tokens as necessary to do so). This function
53  is intended to support the object manager and other components
54  that utilize the reference monitor's access validation,
55  privilege test, and audit generation services.
56 
57  A subject's security context should be captured before initiating
58  access validation and should be released after audit messages
59  are generated. This is necessary to provide a consistent security
60  context to all those services.
61 
62  After calling access validation, privilege test, and audit generation
63  services, the captured context should be released as soon as possible
64  using the SeReleaseSubjectContext() service.
65 
66 Arguments:
67 
68  SubjectContext - Points to a SECURITY_SUBJECT_CONTEXT data structure
69  to be filled in with a snapshot of the calling thread's security
70  profile.
71 
72 Return Value:
73 
74  none.
75 
76 --*/
77 
78 {
79  SeCaptureSubjectContextEx (PsGetCurrentThread (),
80  PsGetCurrentProcess (),
81  SubjectContext);
82 }

99  that utilize the reference monitor's access validation,
100  privilege test, and audit generation services.
101 
102  A subject's security context should be captured before initiating
103  access validation and should be released after audit messages
104  are generated. This is necessary to provide a consistent security
105  context to all those services.
106 
107  After calling access validation, privilege test, and audit generation
108  services, the captured context should be released as soon as possible
109  using the SeReleaseSubjectContext() service.
110 
111 Arguments:
112 
113  Thread - Thread to capture the thread token from. If NULL we don't capture
114  an impersonation token.
115 
116  Process - Process to capture primary token from.
117

121

123 
124  none.

129  BOOLEAN IgnoreCopyOnOpen;
130  BOOLEAN IgnoreEffectiveOnly;
131 
132  PAGED_CODE();
133 
134  SubjectContext->ProcessAuditId = PsProcessAuditId( Process );
135 
136  //
137  // Get pointers to primary and impersonation tokens
138  //
139 
140  if (Thread == NULL) {
141  SubjectContext->ClientToken = NULL;
142  } else {
143  SubjectContext->ClientToken = PsReferenceImpersonationToken(
144  Thread,
145  &IgnoreCopyOnOpen,
146  &IgnoreEffectiveOnly,
147  &(SubjectContext->ImpersonationLevel)
148  );
149  }

161  }
162

164  InterlockedIncrement(&((PTOKEN)(SubjectContext->ClientToken))->CaptureCount);
165  if (SubjectContext->ClientToken == SepTokenLeakToken)

320 SepGetDefaultsSubjectContext(
321  IN PSECURITY_SUBJECT_CONTEXT SubjectContext,
322  OUT PSID *Owner,
323  OUT PSID *Group,
324  OUT PSID *ServerOwner,
325  OUT PSID *ServerGroup,
326  OUT PACL *Dacl
327  )
328 /*++
329 
330 Routine Description:
331 
332  This routine retrieves pointers to the default owner, primary group,
333  and, if present, discretionary ACL of the provided subject security
334  context.
335 
336 Arguments:
337 
338  SubjectContext - Points to the subject security context whose default
339  values are to be retrieved.
340 
341  Owner - Receives a pointer to the subject's default owner SID. This
342  value will always be returned as a non-zero pointer. That is,
343  a subject's security context must contain a owner SID.
344 
345  Group - Receives a pointer to the subject's default primary group SID.
346  This value will always be returned as a non-zero pointer. That is,
347  a subject's security context must contain a primary group.
348 
349  Dacl - Receives a pointer to the subject's default discretionary ACL,
350  if one is define for the subject. Note that a subject security context
351  does not have to include a default discretionary ACL. In this case,
352  this value will be returned as NULL.
353 
354 
355 
356 
357 Return Value:
358 
359  none.
360 
361 --*/
362 
363 {
364  PTOKEN EffectiveToken;
365  PTOKEN PrimaryToken;
366 
367  PAGED_CODE();
368 
369  if (ARGUMENT_PRESENT(SubjectContext->ClientToken)) {
370  EffectiveToken = (PTOKEN)SubjectContext->ClientToken;
371  } else {
372  EffectiveToken = (PTOKEN)SubjectContext->PrimaryToken;
373  }
374 
375  (*Owner) = EffectiveToken->UserAndGroups[EffectiveToken->DefaultOwnerIndex].Sid;
376 
377  (*Group) = EffectiveToken->PrimaryGroup;
378 
379  (*Dacl) = EffectiveToken->DefaultDacl;
380 
381  PrimaryToken = (PTOKEN)SubjectContext->PrimaryToken;
382 
383  *ServerOwner = PrimaryToken->UserAndGroups[PrimaryToken->DefaultOwnerIndex].Sid;
384 
385  *ServerGroup = PrimaryToken->PrimaryGroup;
386 
387  return;
388 }

392 SepIdAssignableAsGroup(
393  IN PACCESS_TOKEN AToken,
394  IN PSID Group
395  )
396 /*++
397 
398 Routine Description:
399 
400  This routine checks to see whether the provided SID is one that
401  may be assigned to be the default primary group in a token.
402 
403  The current criteria is that the passed SID be a group in the
404  token, with no other restrictions.
405 
406 Arguments:
407 
408  Token - Points to the token to be examined.
409 
410  Group - Points to the SID to be checked.
411 
412 Return Value:
413 
414  TRUE - SID passed by be assigned as the default primary group in a token.
415 
416  FALSE - Passed SID may not be so assigned.
417 
418 --*/
419 
420 {
421  ULONG Index;
422  BOOLEAN Found = FALSE;
423  PTOKEN Token;
424 
425  PAGED_CODE();
426 
427  Token = (PTOKEN)AToken;
428 
429  //
430  // Let's make it invalid to assign a NULL primary group,
431  // but we may need to revisit this.
432  //
433 
434  if (Group == NULL) {
435  return( FALSE );
436  }
437  Index = 0;
438 
439  SepAcquireTokenReadLock( Token );
440 
441  //
442  // Walk through the list of user and group IDs looking
443  // for a match to the specified SID.
444  //
445 
446  while (Index < Token->UserAndGroupCount) {
447 
448  Found = RtlEqualSid(
449  Group,
450  Token->UserAndGroups[Index].Sid
451  );
452 
453  if ( Found ) {
454  break;
455  }
456 
457  Index += 1;
458  }
459 
460  SepReleaseTokenReadLock( Token );
461 
462  return Found;
463 }

466 BOOLEAN
467 SepValidOwnerSubjectContext(
468  IN PSECURITY_SUBJECT_CONTEXT SubjectContext,
469  IN PSID Owner,
470  IN BOOLEAN ServerObject
471  )
472 /*++
473 
474 Routine Description:
475 
476  This routine checks to see whether the provided SID is one the subject
477  is authorized to assign as the owner of objects. It will also check to
478  see if the caller has SeRestorePrivilege, if so, the request is granted.
479 
480 Arguments:
481 
482  SubjectContext - Points to the subject's security context.
483 
484  Owner - Points to the SID to be checked.
485 
486 
487 
488 Return Value:
489 
490  none.
491 
492 --*/
493 
494 {
495 
496  ULONG Index;
497  BOOLEAN Found;
498  PTOKEN EffectiveToken;
499  BOOLEAN Rc = FALSE;
500 
501  PAGED_CODE();
502 
503  //
504  // It is invalid to assign a NULL owner, regardless of
505  // whether you have SeRestorePrivilege or not.
506  //
507 
508  if (Owner == NULL) {
509  return( FALSE );
510  }
511 
512  //
513  // Allowable owners come from the primary if it's a server object.
514  //
515 
516  if (!ServerObject && ARGUMENT_PRESENT(SubjectContext->ClientToken)) {
517  EffectiveToken = (PTOKEN)SubjectContext->ClientToken;
518  } else {
519  EffectiveToken = (PTOKEN)SubjectContext->PrimaryToken;
520  }
521 
522 
523  //
524  // If we're impersonating, make sure we're at TokenImpersonation
525  // or above. This prevents someone from setting the owner of an
526  // object when impersonating at less Identify or Anonymous.
527  //
528 
529  if (EffectiveToken->TokenType == TokenImpersonation) {
530 
531  if (EffectiveToken->ImpersonationLevel < SecurityImpersonation) {
532 
533  return( FALSE );
534 
535  }
536  }
537 
538  Index = 0;
539 
540  SepAcquireTokenReadLock( EffectiveToken );
541 
542  //
543  // Walk through the list of user and group IDs looking
544  // for a match to the specified SID. If one is found,
545  // make sure it may be assigned as an owner.
546  //
547  // This code is similar to that performed to set the default
548  // owner of a token (NtSetInformationToken).
549  //
550 
551  while (Index < EffectiveToken->UserAndGroupCount) {
552 
553 
554  Found = RtlEqualSid(
555  Owner,
556  EffectiveToken->UserAndGroups[Index].Sid
557  );
558 
559  if ( Found ) {
560 
561  //
562  // We may return success if the Sid is one that may be assigned
563  // as an owner, or if the caller has SeRestorePrivilege
564  //
565 
566  if ( SepIdAssignableAsOwner(EffectiveToken,Index) ) {
567 
568  SepReleaseTokenReadLock( EffectiveToken );
569  Rc = TRUE;
570  goto exit;
571 
572  } else {
573 
574  //
575  // Rc is already set to FALSE, just exit.
576  //
577 
578  SepReleaseTokenReadLock( EffectiveToken );
579  goto exit;
580 
581  } //endif assignable
582 
583 
584  } //endif Found
585 
586 
587  Index += 1;
588 
589  } //endwhile
590 
591 
592  SepReleaseTokenReadLock( EffectiveToken );
593 
594 exit:
595 
596  //
597  // If we are going to fail this call, check for Restore privilege,
598  // and succeed if he has it.
599  //
600 
601  //
602  // We should really have gotten PreviousMode from the caller, but we
603  // didn't, so hard wire it to be user-mode here.
604  //
605 
606  if ( Rc == FALSE ) {
607  Rc = SeSinglePrivilegeCheck( SeRestorePrivilege, UserMode );
608  }
609 
610  return Rc;
611 }
