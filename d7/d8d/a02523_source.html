58 /*++
59 
60 Routine Description:
61 
62  This routine checks to ensure the system has sufficient page file
63  space remaining.
64 
65  Since this routine is generally used to charge commitment on behalf of
66  usermode or other optional actions, this routine does not allow the
67  caller to use up the very last morsels of commit on the premise that
68  the operating system and drivers can put those to better use than any
69  application in order to prevent the appearance of system hangs.
70 
71 Arguments:
72 
73  QuotaCharge - Supplies the quota amount to charge.
74 
75  Process - Optionally supplies the current process IF AND ONLY IF
76  the working set mutex is held. If the paging file
77  is being extended, the working set mutex is released if
78  this is non-null.
79 
80 Return Value:
81 
82  TRUE if there is sufficient space, FALSE if not.
83 
84 Environment:
85 
86  Kernel mode, APCs disabled, WorkingSetLock and AddressCreation mutexes
87  *MAY* be held.
88 
89 --*/
90 
91 {
92  SIZE_T OldCommitValue;
93  SIZE_T NewCommitValue;
94  SIZE_T CommitLimit;
95  PETHREAD Thread;
96  MMPAGE_FILE_EXPANSION PageExtend;

98  LOGICAL WsHeldShared;

102 #if DBG
103  if (InitializationPhase > 1) {
104  ULONG i;
105 
106  Thread = PsGetCurrentThread ();
107 
108  for (i = 0; i < (ULONG)KeNumberProcessors; i += 1) {
109  ASSERT (KiProcessorBlock[i]->IdleThread != &Thread->Tcb);
110  }
111  }
112 #endif

116  // the compiler cannot compile this code W4 to check for use of
117  // uninitialized variables.
118  //

121  WsHeldShared = FALSE;

191  }
192 
193  if (Process != NULL) {
194  LOCK_WS_REGARDLESS (Thread, Process, WsHeldSafe, WsHeldShared);
195  }
196 
197  OldCommitValue = MmTotalCommittedPages;
198 
199  NewCommitValue = OldCommitValue + QuotaCharge;

201 
202 #if defined(_WIN64)

204  (PLONGLONG) &MmTotalCommittedPages,
205  (LONGLONG) NewCommitValue,
206  (LONGLONG) OldCommitValue);
207 #else
208  NewCommitValue = InterlockedCompareExchange (
209  (PLONG) &MmTotalCommittedPages,
210  (LONG) NewCommitValue,
211  (LONG) OldCommitValue);
212 #endif
213 
214  } while (NewCommitValue != OldCommitValue);
215 
216  //
217  // Success.
218  //
219 
220  MI_LOG_COMMIT_CHANGE (NewCommitValue + QuotaCharge, QuotaCharge);
221

225  MmPeakCommitment = MmTotalCommittedPages;
226  }
227

231  //

235

243  //

253  // bother trying to extend them, but do trim the cache.
254  //
255 
256  if (MmTotalCommitLimit + 100 >= MmTotalCommitLimitMaximum) {
257  MiTrimSegmentCache ();
258  }
259  }
260  }
261 
262  return TRUE;
263 }

272 /*++
273 
274 Routine Description:
275 
276  This routine charges the specified commitment without attempting
277  to expand paging files and waiting for the expansion. The routine
278  determines if the paging file space is exhausted, and if so,
279  it attempts to ascertain if the paging file space could be expanded.
280

282

284

288

291 Environment:
292 
293  Kernel mode, APCs disabled.
294 
295 --*/

298  SIZE_T CommitLimit;
299  SIZE_T ExtendAmount;
300  SIZE_T OldCommitValue;
301  SIZE_T NewCommitValue;
302 
303  ASSERT ((SSIZE_T)QuotaCharge > 0);
304 
305  ASSERT32 ((QuotaCharge < 0x100000) || (QuotaCharge < MmTotalCommitLimit));
306 
307  do {
308 
309  OldCommitValue = ReadForWriteAccess (&MmTotalCommittedPages);
310 
311  NewCommitValue = OldCommitValue + QuotaCharge;
312 
313  if ((NewCommitValue > MmTotalCommitLimit) && (!MustSucceed)) {
314 
315  if ((NewCommitValue < MmTotalCommittedPages) ||
316  (MmTotalCommitLimit + 100 >= MmTotalCommitLimitMaximum)) {
317 
318  MiChargeCommitmentFailures[1] += 1;
319 
320  MI_LOG_COMMIT_FAILURE (NewCommitValue, QuotaCharge);
321 
322  return FALSE;
323  }
324 
325  //
326  // Attempt to expand the paging file, but don't wait
327  // to see if it succeeds.
328  //
329 
330  MiChargeCommitmentFailures[0] += 1;
331  MiIssuePageExtendRequestNoWait (MM_MINIMAL_COMMIT_INCREASE);
332  return FALSE;
333  }
334 
335 #if defined(_WIN64)
336  NewCommitValue = InterlockedCompareExchange64 (
337  (PLONGLONG) &MmTotalCommittedPages,
338  (LONGLONG) NewCommitValue,
339  (LONGLONG) OldCommitValue);
340 #else
341  NewCommitValue = InterlockedCompareExchange (
342  (PLONG) &MmTotalCommittedPages,
343  (LONG) NewCommitValue,
344  (LONG) OldCommitValue);
345 #endif
346 
347  } while (NewCommitValue != OldCommitValue);
348 
349  MI_LOG_COMMIT_CHANGE (NewCommitValue + QuotaCharge, QuotaCharge);
350 
351  MM_TRACK_COMMIT (MM_DBG_COMMIT_CHARGE_CANT_EXPAND, QuotaCharge);
352 
353  //
354  // Success. If system commit exceeds 90%, attempt a preemptive pagefile
355  // increase anyway.
356  //
357 
358  NewCommitValue = MmTotalCommittedPages;
359  CommitLimit = MmTotalCommitLimit;
360 
361  if ((NewCommitValue > ((CommitLimit/10)*9)) &&
362  (CommitLimit < MmTotalCommitLimitMaximum)) {
363 
364  //
365  // Attempt to expand the paging file, but don't wait
366  // to see if it succeeds.
367  //
368  // Queue a message to the segment dereferencing / pagefile extending
369  // thread to see if the page file can be extended. This is done
370  // in the context of a system thread due to mutexes which may
371  // currently be held.
372  //
373 
374  ExtendAmount = NewCommitValue - ((CommitLimit/100)*85);
375 
376  if (QuotaCharge > ExtendAmount) {
377  ExtendAmount = QuotaCharge;
378  }
379 
380  MiIssuePageExtendRequestNoWait (ExtendAmount);
381  }
382 
383  return TRUE;
384 }

471
