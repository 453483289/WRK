9 Module Name:
10 
11  FiltrCtx.c
12 
13 Abstract:
14 
15  This module provides three routines that allow filesystem filter drivers
16  to associate state with FILE_OBJECTs -- for filesystems which support
17  an extended FSRTL_COMMON_HEADER with FsContext.
18 
19  These routines depend on fields (FastMutext and FilterContexts)
20  added at the end of FSRTL_COMMON_HEADER in NT 5.0.
21 
22  Filesystems should set FSRTL_FLAG2_SUPPORTS_FILTER_CONTEXTS if
23  these new fields are supported. They must also initialize the mutex
24  and list head.
25 
26  Filter drivers must use a common header for the context they wish to
27  associate with a file object:
28 
29  FSRTL_FILTER_CONTEXT:
30  LIST_ENTRY Links;
31  PVOID OwnerId;
32  PVOID InstanceId;
33 
34  The OwnerId is a bit pattern unique to each filter driver
35  (e.g. the device object).
36 
37  The InstanceId is used to specify a particular instance of the context
38  data owned by a filter driver (e.g. the file object).
39 
40 --*/

125 
126 Routine Description:
127 
128  This routine lookups filter driver context associated with a stream.
129 
130  The macro FsRtlLookupFilterContext should be used instead of calling
131  this routine directly. The macro optimizes for the common case
132  of an empty list.
133 
134 Arguments:
135 
136  AdvFcbHeader - Advanced FCB Header for stream of interest.
137 
138  OwnerId - Used to identify context information belonging to a particular
139  filter driver.
140 
141  InstanceId - Used to search for a particular instance of a filter driver
142  context. If not provided, any of the contexts owned by the filter
143  driver is returned.
144 
145  If neither the OwnerId nor the InstanceId is provided, any associated
146  filter context will be returned.
147 
148 Return Value:
149 
150  A pointer to the filter context, or NULL if no match found.
151 
152 --*/
153 
154 {
155  PFSRTL_PER_STREAM_CONTEXT ctx;
156  PFSRTL_PER_STREAM_CONTEXT rtnCtx;
157  PLIST_ENTRY list;
158 
159  ASSERT(AdvFcbHeader);
160  ASSERT(FlagOn(AdvFcbHeader->Flags2,FSRTL_FLAG2_SUPPORTS_FILTER_CONTEXTS));
161 
162  ExAcquireFastMutex(AdvFcbHeader->FastMutex);
163  rtnCtx = NULL;
164 
165  //
166  // Use different loops depending on whether we are comparing both Ids or not.
167  //
168 
169  if ( ARGUMENT_PRESENT(InstanceId) ) {
170 
171  MySearchList (&AdvFcbHeader->FilterContexts, list) {
172 
173  ctx = CONTAINING_RECORD(list, FSRTL_PER_STREAM_CONTEXT, Links);
174  if (ctx->OwnerId == OwnerId && ctx->InstanceId == InstanceId) {
175 
176  rtnCtx = ctx;
177  break;
178  }
179  }
180 
181  } else if ( ARGUMENT_PRESENT(OwnerId) ) {
182 
183  MySearchList (&AdvFcbHeader->FilterContexts, list) {
184 
185  ctx = CONTAINING_RECORD(list, FSRTL_PER_STREAM_CONTEXT, Links);
186  if (ctx->OwnerId == OwnerId) {
187 
188  rtnCtx = ctx;
189  break;
190  }
191  }
192 
193  } else if (!IsListEmpty(&AdvFcbHeader->FilterContexts)) {
194 
195  rtnCtx = (PFSRTL_PER_STREAM_CONTEXT)AdvFcbHeader->FilterContexts.Flink;
196  }
197 
198  ExReleaseFastMutex(AdvFcbHeader->FastMutex);
199  return rtnCtx;
200 }

293 NTKERNELAPI
294 VOID
295 FsRtlTeardownPerStreamContexts (
296  __in PFSRTL_ADVANCED_FCB_HEADER AdvFcbHeader
297  )
298 /*++
299 
300 Routine Description:
301 
302  This routine is called by filesystems to free the filter contexts
303  associated with an FSRTL_COMMON_FCB_HEADER by calling the FreeCallback
304  routine for each FilterContext.
305 
306 Arguments:
307 
308  FilterContexts - the address of the FilterContexts field within
309  the FSRTL_COMMON_FCB_HEADER of the structure being torn down
310  by the filesystem.
311 
312 Return Value:
313 
314  None.
315 
316 --*/
317 
318 {
319  PFSRTL_PER_STREAM_CONTEXT ctx;
320  PLIST_ENTRY ptr;
321  BOOLEAN lockHeld;
322 
323  //
324  // Acquire the lock because someone could be trying to free this
325  // entry while we are trying to free it.
326  //
327 
328  ExAcquireFastMutex( AdvFcbHeader->FastMutex );
329  lockHeld = TRUE;
330 
331  try {
332 
333  while (!IsListEmpty( &AdvFcbHeader->FilterContexts )) {
334 
335  //
336  // Unlink the top entry then release the lock. We must
337  // release the lock before calling the use or their could
338  // be potential locking order deadlocks.
339  //
340 
341  ptr = RemoveHeadList( &AdvFcbHeader->FilterContexts );
342 
343  ExReleaseFastMutex(AdvFcbHeader->FastMutex);
344  lockHeld = FALSE;
345 
346  //
347  // Call filter to free this entry
348  //
349 
350  ctx = CONTAINING_RECORD( ptr, FSRTL_PER_STREAM_CONTEXT, Links );
351  ASSERT(ctx->FreeCallback);
352 
353  (*ctx->FreeCallback)( ctx );
354 
355  //
356  // re-get the lock
357  //
358 
359  ExAcquireFastMutex( AdvFcbHeader->FastMutex );
360  lockHeld = TRUE;
361  }
362 
363  } finally {
364 
365  if (lockHeld) {
366 
367  ExReleaseFastMutex( AdvFcbHeader->FastMutex );
368  }
369  }
370 }

402 NTKERNELAPI
403 NTSTATUS
404 FsRtlInsertPerFileObjectContext (
405  __in PFILE_OBJECT FileObject,
406  __in PFSRTL_PER_FILEOBJECT_CONTEXT Ptr
407  )
408 /*++
409 
410 Routine Description:
411 
412  This routine associates a context with a file object.
413 
414 Arguments:
415 
416  FileObject - Specifies the file object of interest.
417 
418  Ptr - Pointer to the filter-specific context structure.
419  The common header fields OwnerId and InstanceId should
420  be filled in by the filter driver before calling.
421 
422 Return Value:
423 
424  STATUS_SUCCESS - operation succeeded.
425 
426  STATUS_INVALID_DEVICE_REQUEST - underlying filesystem does not support
427  filter contexts.
428 
429 --*/
430 
431 {
432  PPER_FILEOBJECT_CTXCTRL ctxCtrl;
433  NTSTATUS status;
434 
435  //
436  // Return if no file object
437  //
438 
439  if (NULL == FileObject) {
440 
441  return STATUS_INVALID_PARAMETER;
442  }
443 
444  if (!FsRtlSupportsPerFileObjectContexts(FileObject)) {
445 
446  return STATUS_INVALID_DEVICE_REQUEST;
447  }
448 
449  //
450  // Get the context control structure out of the file object extension
451  //
452 
453  ctxCtrl = IoGetFileObjectFilterContext( FileObject );
454 
455  if (NULL == ctxCtrl) {
456 
457  //
458  // There is not a control structure, allocate and initialize one
459  //
460 
461  ctxCtrl = ExAllocatePoolWithTag( NonPagedPool,
462  sizeof(PER_FILEOBJECT_CTXCTRL),
463  'XCOF' );
464  if (NULL == ctxCtrl) {
465 
466  return STATUS_INSUFFICIENT_RESOURCES;
467  }
468 
469  ExInitializeFastMutex( &ctxCtrl->FastMutex );
470  InitializeListHead( &ctxCtrl->FilterContexts );
471 
472  //
473  // Insert into the file object extension
474  //
475 
476  status = IoChangeFileObjectFilterContext( FileObject,
477  ctxCtrl,
478  TRUE );
479 
480  if (!NT_SUCCESS(status)) {
481 
482  //
483  // If this operation fails it is because someone else inserted the
484  // entry at the same time. In this case free the memory we
485  // allocated and re-get the current value.
486  //
487 
488  ExFreePool( ctxCtrl );
489 
490  ctxCtrl = IoGetFileObjectFilterContext( FileObject );
491 
492  if (NULL == ctxCtrl) {
493 
494  //
495  // This should never actually happen. If it does it means
496  // someone allocated and then freed a context very quickly.
497  //
498 
499  ASSERT(!"This operation should not have failed");
500  return STATUS_UNSUCCESSFUL;
501  }
502  }
503  }
504 
505  ExAcquireFastMutex( &ctxCtrl->FastMutex );
506 
507  InsertHeadList( &ctxCtrl->FilterContexts, &Ptr->Links );
508 
509  ExReleaseFastMutex( &ctxCtrl->FastMutex );
510 
511  return STATUS_SUCCESS;
512 }
