65 
66 Routine Description:
67 
68  - All hives are freed
69  - KCB table is freed
70  - Name hash table is freed
71  - delay close table is freed - question: We need to clean/free all delayed close KCBs

73

75

77 
78

100  //

102  //
103  // Remove the hive from the list
104  //

137  //
138  // walk through the list and spew the keynames and postblock types.
139  //
140  while ( PostBlock != (PCM_POST_BLOCK)AnchorAddr ) {
141  PostBlock = CONTAINING_RECORD(PostBlock,
142  CM_POST_BLOCK,
143  NotifyList);
144 
145  if( PostBlock->PostKeyBody ) {
146  if( MessageDisplayed == FALSE ){
147  MessageDisplayed = TRUE;
148  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"Dumping untriggered notifications for hive (%lx) (%.*S) \n\n",CmHive,
149  HBASE_NAME_ALLOC / sizeof(WCHAR),CmHive->Hive.BaseBlock->FileName);
150  }
151  switch (PostBlockType(PostBlock)) {
152  case PostSynchronous:
153  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"Synchronous ");
154  break;
155  case PostAsyncUser:
156  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"AsyncUser ");
157  break;
158  case PostAsyncKernel:
159  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"AsyncKernel ");
160  break;
161  }

163  PostBlock->PostKeyBody->KeyBody->KeyControlBlock);
164  }
165 
166 
167  //
168  // skip to the next element
169  //
170  PostBlock = (PCM_POST_BLOCK)(PostBlock->NotifyList.Flink);
171 
172  }
173  NotifyPtr = NotifyPtr->Flink;
174  }
175 
176  //
177  // free security cache
178  //
179  CmpDestroySecurityCache (CmHive);
180 
181  //
182  // free the hv level structure
183  //
184  HvFreeHive(&(CmHive->Hive));
185 
186  //
187  // free the cm level structure
188  //
189  CmpFree(CmHive, sizeof(CMHIVE));
190 
191  }
192 
193  //
194  // Now free the CM globals
195  //
196 
197  // the stash buffer
198  if( CmpStashBuffer != NULL ) {
199  ExFreePool( CmpStashBuffer );
200  }
201 
202  //
203  // Spew open handles and associated processes
204  //
205  Count = 0;
206  MessageDisplayed = FALSE;
207  for (i=0; i<(LONG)CmpHashTableSize; i++) {
208  Current = CmpCacheTable[i].Entry;
209  while (Current) {
210  KeyControlBlock = CONTAINING_RECORD(Current, CM_KEY_CONTROL_BLOCK, KeyHash);
211  if( MessageDisplayed == FALSE ){
212  MessageDisplayed = TRUE;
213  DbgPrintEx(DPFLTR_CONFIG_ID,DPFLTR_ERROR_LEVEL,"\nDumping open handles : \n\n");
214  }
215  CmpDumpKeyBodyList(KeyControlBlock,&Count,NULL);
216  Current = Current->NextHash;
217  }
218  }
219 
220  if( Count != 0 ) {
221  //
222  // there some open handles; bugcheck
223  //
224  CM_BUGCHECK( REGISTRY_ERROR,HANDLES_STILL_OPEN_AT_SHUTDOWN,1,Count,0);
225  }
226 
227  //
228  // in case of private alloc, free pages
229  //
230  CmpDestroyCmPrivateAlloc();
231  CmpDestroyCmPrivateDelayAlloc();
232  //
233  // For the 3 tables below, the objects actually pointed from inside
234  // should be cleaned up (freed) at the last handle closure time
235  // the related handles are closed
236  //
237  // KCB cache table
238  ASSERT( CmpCacheTable != NULL );
239  ExFreePool(CmpCacheTable);
240 
241  // NameCacheTable
242  ASSERT( CmpNameCacheTable != NULL );
243  ExFreePool( CmpNameCacheTable );
244

252 /*++
253 
254 Routine Description:
255 
256  Shuts down the registry.
257 
258 Arguments:
259 
260  NONE
261 
262 Return Value:
263 
264  NONE
265 
266 --*/
267 {
268 
269  PLIST_ENTRY p;
270  PCMHIVE CmHive;
271  NTSTATUS Status;
272  PVOID RegistryRoot;
273 
274  CM_PAGED_CODE();
275 
276  if (CmpRegistryRootHandle) {
277  Status = ObReferenceObjectByHandle(CmpRegistryRootHandle,
278  KEY_READ,
279  NULL,
280  KernelMode,
281  &RegistryRoot,
282  NULL);
283 
284  if (NT_SUCCESS(Status)) {
285  // We want to dereference the object twice -- once for the
286  // reference we just made, and once for the reference
287  // fromCmpCreateRegistryRoot.
288  ObDereferenceObject(RegistryRoot);
289  ObDereferenceObject(RegistryRoot);
290  }
291 
292  ObCloseHandle(CmpRegistryRootHandle, KernelMode);
293  }
294 
295  CmpLockRegistryExclusive();
296 
297  //
298  // Stop the workers; only if registry has been initializeed
299  //
300  if( CmFirstTime == FALSE ) {
301  CmpShutdownWorkers();
302  }
303 
304  //
305  // shut down the registry
306  //
307  CmpDoFlushAll(TRUE);
308 
309  //
310  // try to compress the system hive
311  //
312  CmCompressKey( &(CmpMachineHiveList[SYSTEM_HIVE_INDEX].CmHive->Hive) );
313 
314  //
315  // close all the hive files
316  //
317  p = CmpHiveListHead.Flink;
318  while(p != &CmpHiveListHead) {
319  CmHive = CONTAINING_RECORD(p, CMHIVE, HiveList);
320  //
321  // we need to unmap all views mapped for this hive first
322  //
323  CmpDestroyHiveViewList(CmHive);
324  CmpUnJoinClassOfTrust(CmHive);
325  //
326  // dereference the fileobject (if any).
327  //
328  CmpDropFileObjectForHive(CmHive);
329 
330  //
331  // now we can safely close all the handles
332  //
333  CmpCmdHiveClose(CmHive);
334 
335  p=p->Flink;
336  }
337 
338  HvShutdownComplete = TRUE; // Tell HvSyncHive to ignore all further requests
339 
340  if((PoCleanShutdownEnabled() & PO_CLEAN_SHUTDOWN_REGISTRY) && (CmFirstTime == FALSE)){
341  //
342  // Free aux memory used internally by CM
343  //
344  CmpFreeAllMemory();
345  }
346 
347  CmpUnlockRegistry();
