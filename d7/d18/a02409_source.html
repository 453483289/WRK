28 /*++
29 
30 Routine Description:
31 
32  This function checks to detemine if a kernel APC can be delivered
33  immediately to the current thread or a kernel APC interrupt should
34  be requested. On entry to this routine the following conditions are
35  true:
36 
37  1. Special kernel APCs are enabled for the current thread.
38 
39  2. Normal kernel APCs may also be enabled for the current thread.
40 
41  3. The kernel APC queue is not empty.
42 
43  N.B. This routine is only called by kernel code that leaves a guarded
44  or critcial region.
45 
46 Arguments:
47 
48  None.
49 
50 Return Value:
51 
52  None.
53 
54 --*/
55 
56 {
57 
58  //
59  // If the current IRQL is passive level, then a kernel APC can be
60  // delivered immediately. Otherwise, an APC interrupt must be
61  // requested.
62  //
63 
64  if (KeGetCurrentIrql() == PASSIVE_LEVEL) {
65  KfRaiseIrql(APC_LEVEL);
66  KiDeliverApc(KernelMode, NULL, NULL);
67  KeLowerIrql(PASSIVE_LEVEL);
68 
69  } else {
70  KeGetCurrentThread()->ApcState.KernelApcPending = TRUE;
71  KiRequestSoftwareInterrupt(APC_LEVEL);
72  }
73 
74  return;
75 }

84 /*++
85 
86 Routine Description:
87 
88  This function is called from the APC interrupt code and when one or
89  more of the APC pending flags are set at system exit and the previous
90  IRQL is zero. All special kernel APC's are delivered first, followed
91  by normal kernel APC's if one is not already in progress, and finally
92  if the user APC queue is not empty, the user APC pending flag is set,
93  and the previous mode is user, then a user APC is delivered. On entry
94  to this routine IRQL is set to APC_LEVEL.
95 
96  N.B. The exception frame and trap frame addresses are only guaranteed
97  to be valid if, and only if, the previous mode is user.
98 
99 Arguments:
100 
101  PreviousMode - Supplies the previous processor mode.
102 
103  ExceptionFrame - Supplies a pointer to an exception frame.
104 
105  TrapFrame - Supplies a pointer to a trap frame.
106 
107 Return Value:
108 
109  None.
110 
111 --*/
112 
113 {
114 
115  PKAPC Apc;
116  PKKERNEL_ROUTINE KernelRoutine;
117  KLOCK_QUEUE_HANDLE LockHandle;
118  PLIST_ENTRY NextEntry;
119  PVOID NormalContext;
120  PKNORMAL_ROUTINE NormalRoutine;
121  PKTRAP_FRAME OldTrapFrame;
122  PKPROCESS Process;
123  PVOID SystemArgument1;
124  PVOID SystemArgument2;
125  PKTHREAD Thread;
126 
127  //
128  // If the thread was interrupted in the middle of the SLIST pop code,
129  // then back up the PC to the start of the SLIST pop.
130  //
131 
132  if (TrapFrame != NULL) {
133  KiCheckForSListAddress(TrapFrame);
134  }
135 
136  //
137  // Save the current thread trap frame address and set the thread trap
138  // frame address to the new trap frame. This will prevent a user mode
139  // exception from being raised within an APC routine.
140  //
141 
142  Thread = KeGetCurrentThread();
143  OldTrapFrame = Thread->TrapFrame;
144  Thread->TrapFrame = TrapFrame;
145 
146  //
147  // If special APC are not disabled, then attempt to deliver one or more
148  // APCs.
149  //
150 
151  Process = Thread->ApcState.Process;
152  Thread->ApcState.KernelApcPending = FALSE;
153  if (Thread->SpecialApcDisable == 0) {
154 
155  //
156  // If the kernel APC queue is not empty, then attempt to deliver a
157  // kernel APC.
158  //
159  // N.B. The following test is not synchronized with the APC insertion
160  // code. However, when an APC is inserted in the kernel queue of
161  // a running thread an APC interrupt is requested. Therefore, if
162  // the following test were to falsely return that the kernel APC
163  // queue was empty, an APC interrupt would immediately cause this
164  // code to be executed a second time in which case the kernel APC
165  // queue would found to contain an entry.
166  //
167 
168  KeMemoryBarrier();
169  while (IsListEmpty(&Thread->ApcState.ApcListHead[KernelMode]) == FALSE) {
170 
171  //
172  // Raise IRQL to dispatcher level, lock the APC queue, and check
173  // if any kernel mode APC's can be delivered.
174  //
175  // If the kernel APC queue is now empty because of the removal of
176  // one or more entries, then release the APC lock, and attempt to
177  // deliver a user APC.
178  //
179 
180  KeAcquireInStackQueuedSpinLock(&Thread->ApcQueueLock, &LockHandle);
181  NextEntry = Thread->ApcState.ApcListHead[KernelMode].Flink;
182  if (NextEntry == &Thread->ApcState.ApcListHead[KernelMode]) {
183  KeReleaseInStackQueuedSpinLock(&LockHandle);
184  break;
185  }
186 
187  //
188  // Clear kernel APC pending, get the address of the APC object,
189  // and determine the type of APC.
190  //
191  // N.B. Kernel APC pending must be cleared each time the kernel
192  // APC queue is found to be non-empty.
193  //
194 
195  Thread->ApcState.KernelApcPending = FALSE;
196  Apc = CONTAINING_RECORD(NextEntry, KAPC, ApcListEntry);
197  ReadForWriteAccess(Apc);
198  KernelRoutine = Apc->KernelRoutine;
199  NormalRoutine = Apc->NormalRoutine;
200  NormalContext = Apc->NormalContext;
201  SystemArgument1 = Apc->SystemArgument1;
202  SystemArgument2 = Apc->SystemArgument2;
203  if (NormalRoutine == (PKNORMAL_ROUTINE)NULL) {
204 
205  //
206  // First entry in the kernel APC queue is a special kernel APC.
207  // Remove the entry from the APC queue, set its inserted state
208  // to FALSE, release dispatcher database lock, and call the kernel
209  // routine. On return raise IRQL to dispatcher level and lock
210  // dispatcher database lock.
211  //
212 
213  RemoveEntryList(NextEntry);
214  Apc->Inserted = FALSE;
215  KeReleaseInStackQueuedSpinLock(&LockHandle);
216  (KernelRoutine)(Apc,
217  &NormalRoutine,
218  &NormalContext,
219  &SystemArgument1,
220  &SystemArgument2);
221 
222 #if DBG
223 
224  if (KeGetCurrentIrql() != LockHandle.OldIrql) {
225  KeBugCheckEx(IRQL_UNEXPECTED_VALUE,
226  KeGetCurrentIrql() << 16 | LockHandle.OldIrql << 8,
227  (ULONG_PTR)KernelRoutine,
228  (ULONG_PTR)Apc,
229  (ULONG_PTR)NormalRoutine);
230  }
231 
232 #endif
233 
234  } else {
235 
236  //
237  // First entry in the kernel APC queue is a normal kernel APC.
238  // If there is not a normal kernel APC in progress and kernel
239  // APC's are not disabled, then remove the entry from the APC
240  // queue, set its inserted state to FALSE, release the APC queue
241  // lock, call the specified kernel routine, set kernel APC in
242  // progress, lower the IRQL to zero, and call the normal kernel
243  // APC routine. On return raise IRQL to dispatcher level, lock
244  // the APC queue, and clear kernel APC in progress.
245  //
246 
247  if ((Thread->ApcState.KernelApcInProgress == FALSE) &&
248  (Thread->KernelApcDisable == 0)) {
249 
250  RemoveEntryList(NextEntry);
251  Apc->Inserted = FALSE;
252  KeReleaseInStackQueuedSpinLock(&LockHandle);
253  (KernelRoutine)(Apc,
254  &NormalRoutine,
255  &NormalContext,
256  &SystemArgument1,
257  &SystemArgument2);
258 
259 #if DBG
260 
261  if (KeGetCurrentIrql() != LockHandle.OldIrql) {
262  KeBugCheckEx(IRQL_UNEXPECTED_VALUE,
263  KeGetCurrentIrql() << 16 | LockHandle.OldIrql << 8 | 1,
264  (ULONG_PTR)KernelRoutine,
265  (ULONG_PTR)Apc,
266  (ULONG_PTR)NormalRoutine);
267  }
268 
269 #endif
270 
271  if (NormalRoutine != (PKNORMAL_ROUTINE)NULL) {
272  Thread->ApcState.KernelApcInProgress = TRUE;
273  KeLowerIrql(0);
274  (NormalRoutine)(NormalContext,
275  SystemArgument1,
276  SystemArgument2);
277 
278  KeRaiseIrql(APC_LEVEL, &LockHandle.OldIrql);
279  }
280 
281  Thread->ApcState.KernelApcInProgress = FALSE;
282 
283  } else {
284  KeReleaseInStackQueuedSpinLock(&LockHandle);
285  goto CheckProcess;
286  }
287  }
288  }
289 
290  //
291  // Kernel APC queue is empty. If the previous mode is user, user APC
292  // pending is set, and the user APC queue is not empty, then remove
293  // the first entry from the user APC queue, set its inserted state to
294  // FALSE, clear user APC pending, release the dispatcher database lock,
295  // and call the specified kernel routine. If the normal routine address
296  // is not NULL on return from the kernel routine, then initialize the
297  // user mode APC context and return. Otherwise, check to determine if
298  // another user mode APC can be processed.
299  //
300  // N.B. There is no race condition associated with checking the APC
301  // queue outside the APC lock. User APCs are always delivered at
302  // system exit and never interrupt the execution of the thread
303  // in the kernel.
304  //
305 
306  if ((PreviousMode == UserMode) &&
307  (IsListEmpty(&Thread->ApcState.ApcListHead[UserMode]) == FALSE) &&
308  (Thread->ApcState.UserApcPending != FALSE)) {
309 
310  //
311  // Raise IRQL to dispatcher level, lock the APC queue, and deliver
312  // a user mode APC.
313  //
314 
315  KeAcquireInStackQueuedSpinLock(&Thread->ApcQueueLock, &LockHandle);
316 
317  //
318  // If the user APC queue is now empty because of the removal of
319  // one or more entries, then release the APC lock and exit.
320  //
321 
322  Thread->ApcState.UserApcPending = FALSE;
323  NextEntry = Thread->ApcState.ApcListHead[UserMode].Flink;
324  if (NextEntry == &Thread->ApcState.ApcListHead[UserMode]) {
325  KeReleaseInStackQueuedSpinLock(&LockHandle);
326  goto CheckProcess;
327  }
328 
329  Apc = CONTAINING_RECORD(NextEntry, KAPC, ApcListEntry);
330  ReadForWriteAccess(Apc);
331  KernelRoutine = Apc->KernelRoutine;
332  NormalRoutine = Apc->NormalRoutine;
333  NormalContext = Apc->NormalContext;
334  SystemArgument1 = Apc->SystemArgument1;
335  SystemArgument2 = Apc->SystemArgument2;
336  RemoveEntryList(NextEntry);
337  Apc->Inserted = FALSE;
338  KeReleaseInStackQueuedSpinLock(&LockHandle);
339  (KernelRoutine)(Apc,
340  &NormalRoutine,
341  &NormalContext,
342  &SystemArgument1,
343  &SystemArgument2);
344 
345  if (NormalRoutine == (PKNORMAL_ROUTINE)NULL) {
346  KeTestAlertThread(UserMode);
347 
348  } else {
349  KiInitializeUserApc(ExceptionFrame,
350  TrapFrame,
351  NormalRoutine,
352  NormalContext,
353  SystemArgument1,
354  SystemArgument2);
355  }
356  }
357  }
358 
359  //
360  // Check if process was attached during the APC routine.
361  //
362 
363 CheckProcess:
364  if (Thread->ApcState.Process != Process) {
365  KeBugCheckEx(INVALID_PROCESS_ATTACH_ATTEMPT,
366  (ULONG_PTR)Process,
367  (ULONG_PTR)Thread->ApcState.Process,
368  (ULONG)Thread->ApcStateIndex,
369  (ULONG)KeIsExecutingDpc());
370  }
371 
372  //
373  // Restore the previous thread trap frame address.
374  //
375 
376  Thread->TrapFrame = OldTrapFrame;
377  return;
378 }

382 KiInsertQueueApc (
383  IN PKAPC Apc,
384  IN KPRIORITY Increment
385  )
386 
387 /*++
388 
389 Routine Description:
390 
391  This function inserts an APC object into a thread's APC queue. The address
392  of the thread object, the APC queue, and the type of APC are all derived
393  from the APC object. If the APC object is already in an APC queue, then
394  no operation is performed and a function value of FALSE is returned. Else
395  the APC is inserted in the specified APC queue, its inserted state is set
396  to TRUE, and a function value of TRUE is returned. The APC will actually
397  be delivered when proper enabling conditions exist.
398 
399  N.B. The thread APC queue lock must be held when this routine is called.
400 
401  N.B. It is the responsibility of the caller to ensure that the APC is not
402  already inserted in an APC queue and to set the Inserted field of
403  the APC.
404 
405 Arguments:
406 
407  Apc - Supplies a pointer to a control object of type APC.
408 
409  Increment - Supplies the priority increment that is to be applied if
410  queuing the APC causes a thread wait to be satisfied.
411 
412 Return Value:
413 
414  None.
415 
416 --*/
417 
418 {
419 
420  KPROCESSOR_MODE ApcMode;
421  PKAPC ApcEntry;
422  PKAPC_STATE ApcState;
423  PKGATE GateObject;
424  PLIST_ENTRY ListEntry;
425  PKQUEUE Queue;
426  BOOLEAN RequestInterrupt;
427  PKTHREAD Thread;
428  KTHREAD_STATE ThreadState;
429 
430  //
431  // Insert the APC object in the specified APC queue, set the APC inserted
432  // state to TRUE, and check to determine if the APC should be delivered
433  // immediately.
434  //
435  // For multiprocessor performance, the following code utilizes the fact
436  // that kernel APC disable count is incremented before checking whether
437  // the kernel APC queue is nonempty.
438  //
439  // See KeLeaveCriticalRegion().
440  //
441 
442  Thread = Apc->Thread;
443  if (Apc->ApcStateIndex == InsertApcEnvironment) {
444  Apc->ApcStateIndex = Thread->ApcStateIndex;
445  }
446 
447  ApcState = Thread->ApcStatePointer[Apc->ApcStateIndex];
448 
449  //
450  // Insert the APC after all other special APC entries selected by
451  // the processor mode if the normal routine value is NULL. Else
452  // insert the APC object at the tail of the APC queue selected by
453  // the processor mode unless the APC mode is user and the address
454  // of the special APC routine is exit thread, in which case insert
455  // the APC at the front of the list and set user APC pending.
456  //
457 
458  ApcMode = Apc->ApcMode;
459 
460  ASSERT (Apc->Inserted == TRUE);
461 
462  if (Apc->NormalRoutine != NULL) {
463  if ((ApcMode != KernelMode) && (Apc->KernelRoutine == PsExitSpecialApc)) {
464  Thread->ApcState.UserApcPending = TRUE;
465  InsertHeadList(&ApcState->ApcListHead[ApcMode],
466  &Apc->ApcListEntry);
467 
468  } else {
469  InsertTailList(&ApcState->ApcListHead[ApcMode],
470  &Apc->ApcListEntry);
471  }
472 
473  } else {
474  ListEntry = ApcState->ApcListHead[ApcMode].Blink;
475  while (ListEntry != &ApcState->ApcListHead[ApcMode]) {
476  ApcEntry = CONTAINING_RECORD(ListEntry, KAPC, ApcListEntry);
477  if (ApcEntry->NormalRoutine == NULL) {
478  break;
479  }
480 
481  ListEntry = ListEntry->Blink;
482  }
483 
484  InsertHeadList(ListEntry, &Apc->ApcListEntry);
485  }
486 
487  //
488  // If the APC index from the APC object matches the APC Index of
489  // the thread, then check to determine if the APC should interrupt
490  // thread execution or sequence the thread out of a wait state.
491  //
492 
493  if (Apc->ApcStateIndex == Thread->ApcStateIndex) {
494 
495  //
496  // If the target thread is the current thread, then the thread state
497  // is running and cannot change.
498  //
499 
500  if (Thread == KeGetCurrentThread()) {
501 
502  ASSERT(Thread->State == Running);
503 
504  //
505  // If the APC mode is kernel, then set kernel APC pending and
506  // request an APC interrupt if special APC's are not disabled.
507  //
508 
509  if (ApcMode == KernelMode) {
510  Thread->ApcState.KernelApcPending = TRUE;
511  if (Thread->SpecialApcDisable == 0) {
512  KiRequestSoftwareInterrupt(APC_LEVEL);
513  }
514  }
515 
516  return;
517  }
518 
519  //
520  // Lock the dispatcher database and test the processor mode.
521  //
522  // If the processor mode of the APC is kernel, then check if
523  // the APC should either interrupt the thread or sequence the
524  // thread out of a Waiting state. Else check if the APC should
525  // sequence the thread out of an alertable Waiting state.
526  //
527 
528  RequestInterrupt = FALSE;
529  KiLockDispatcherDatabaseAtSynchLevel();
530  if (ApcMode == KernelMode) {
531 
532  //
533  // Thread transitions from the standby state to the running
534  // state can occur from the idle thread without holding the
535  // dispatcher lock. Reading the thread state after setting
536  // the kernel APC pending flag prevents the code from not
537  // delivering the APC interrupt in this case.
538  //
539  // N.B. Transitions from gate wait to running are synchronized
540  // using the thread lock. Transitions from running to gate
541  // wait are synchronized using the APC queue lock.
542  //
543  // N.B. If the target thread is found to be in the running state,
544  // then the APC interrupt request can be safely deferred to
545  // after the dispatcher lock is released even if the thread
546  // were to be switched to another processor, i.e., the APC
547  // would be delivered by the context switch code.
548  //
549 
550  Thread->ApcState.KernelApcPending = TRUE;
551  KeMemoryBarrier();
552  ThreadState = Thread->State;
553  if (ThreadState == Running) {
554  RequestInterrupt = TRUE;
555 
556  } else if ((ThreadState == Waiting) &&
557  (Thread->WaitIrql == 0) &&
558  (Thread->SpecialApcDisable == 0) &&
559  ((Apc->NormalRoutine == NULL) ||
560  ((Thread->KernelApcDisable == 0) &&
561  (Thread->ApcState.KernelApcInProgress == FALSE)))) {
562 
563  KiUnwaitThread(Thread, STATUS_KERNEL_APC, Increment);
564 
565  } else if (Thread->State == GateWait) {
566  KiAcquireThreadLock(Thread);
567  if ((Thread->State == GateWait) &&
568  (Thread->WaitIrql == 0) &&
569  (Thread->SpecialApcDisable == 0) &&
570  ((Apc->NormalRoutine == NULL) ||
571  ((Thread->KernelApcDisable == 0) &&
572  (Thread->ApcState.KernelApcInProgress == FALSE)))) {
573 
574  GateObject = Thread->GateObject;
575  KiAcquireKobjectLock(GateObject);
576  RemoveEntryList(&Thread->WaitBlock[0].WaitListEntry);
577  KiReleaseKobjectLock(GateObject);
578  if ((Queue = Thread->Queue) != NULL) {
579  Queue->CurrentCount += 1;
580  }
581 
582  Thread->WaitStatus = STATUS_KERNEL_APC;
583  KiInsertDeferredReadyList(Thread);
584  }
585 
586  KiReleaseThreadLock(Thread);
587  }
588 
589  } else if ((Thread->State == Waiting) &&
590  (Thread->WaitMode == UserMode) &&
591  (Thread->Alertable || Thread->ApcState.UserApcPending)) {
592 
593  Thread->ApcState.UserApcPending = TRUE;
594  KiUnwaitThread(Thread, STATUS_USER_APC, Increment);
595  }
596 
597  //
598  // Unlock the dispatcher database and request an APC interrupt if
599  // required.
600  //
601 
602  KiUnlockDispatcherDatabaseFromSynchLevel();
603  if (RequestInterrupt == TRUE) {
604  KiRequestApcInterrupt(Thread->NextProcessor);
605  }
606  }
607 
608  return;
609 }
