39 NTSTATUS
40 NtCancelIoFile (
41  __in HANDLE FileHandle,
42  __out PIO_STATUS_BLOCK IoStatusBlock
43  )
44 
45 /*++
46 
47 Routine Description:
48 
49  This service causes all pending I/O operations for the specified file to be
50  marked as canceled. Most types of operations can be canceled immediately,
51  while others may continue toward completion before they are actually
52  canceled and the caller is notified.
53 
54  Only those pending operations that were issued by the current thread using
55  the specified handle are canceled. Any operations issued for the file by
56  any other thread or any other process continues normally.
57 
58 Arguments:
59 
60  FileHandle - Supplies a handle to the file whose operations are to be
61  canceled.
62 
63  IoStatusBlock - Address of the caller's I/O status block.
64 
65 Return Value:
66 
67  The status returned is the final completion status of the operation.
68 
69 --*/
70 
71 {
72  PIRP irp;
73  NTSTATUS status;
74  PFILE_OBJECT fileObject;
75  KPROCESSOR_MODE requestorMode;
76  PETHREAD thread;
77  BOOLEAN found = FALSE;
78  PLIST_ENTRY header;
79  PLIST_ENTRY entry;
80  KIRQL irql;
81 
82  PAGED_CODE();
83 
84 
85  //
86  // Get the address of the current thread. The thread contains a list of
87  // the pending operations for this file.
88  //
89 
90  thread = PsGetCurrentThread();
91 
92  //
93  // Get the previous mode; i.e., the mode of the caller.
94  //
95 
96  requestorMode = KeGetPreviousModeByThread(&thread->Tcb);
97 
98  if (requestorMode != KernelMode) {
99 
100  //
101  // The caller's access mode is user, so probe each of the arguments
102  // and capture them as necessary. If any failures occur, the condition
103  // handler will be invoked to handle them. It will simply cleanup and
104  // return an access violation status code back to the system service
105  // dispatcher.
106  //
107 
108  try {
109 
110  //
111  // The IoStatusBlock parameter must be writeable by the caller.
112  //
113 
114  ProbeForWriteIoStatus( IoStatusBlock );
115 
116  } except(EXCEPTION_EXECUTE_HANDLER) {
117 
118  //
119  // An exception was incurred attempting to probe the caller'
120  // I/O status block. Simply return an appropriate error status
121  // code.
122  //
123 
124  return GetExceptionCode();
125  }
126  }
127 
128  //
129  // There were no blatant errors so far, so reference the file object so
130  // the target device object can be found. Note that if the handle does
131  // not refer to a file object, or if the caller does not have the required
132  // access to the file, then it will fail.
133  //
134 
135  status = ObReferenceObjectByHandle( FileHandle,
136  0,
137  IoFileObjectType,
138  requestorMode,
139  (PVOID *) &fileObject,
140  NULL );
141  if (!NT_SUCCESS( status )) {
142  return(status);
143  }
144 
145  //
146  // Note that here the I/O system would normally make a check to determine
147  // whether or not the file was opened for synchronous I/O. If it was, then
148  // it would attempt to exclusively acquire the file object lock. However,
149  // since this service is attempting to cancel all of the I/O for the file,
150  // it does not make much sense to wait until it has all completed before
151  // attempting to cancel it.
152  //
153 
154 
155  //
156  // Update the operation count statistic for the current process for
157  // operations other than read and write.
158  //
159 
160  IopUpdateOtherOperationCount();
161 
162  //
163  // Walk the list of IRPs on the thread's pending I/O queue looking for IRPs
164  // which specify the same file as the FileHandle refers to. For each IRP
165  // found, set its cancel flag. If no IRPs are found, simply complete the
166  // I/O here. The only synchronization needed here is to block out all APCs
167  // for this thread so that no I/O can complete and remove packets from the
168  // queue. No considerations need be made for multi-processing since this
169  // thread can only be running on one processor at a time and this routine
170  // has control of the thread for now.
171  //
172 
173  KeRaiseIrql( APC_LEVEL, &irql );
174 
175  header = &thread->IrpList;
176  entry = thread->IrpList.Flink;
177 
178  while (header != entry) {
179 
180  //
181  // An IRP has been found for this thread. If the IRP refers to the
182  // appropriate file object, set its cancel flag and remember that it
183  // was found; otherwise, simply continue the loop.
184  //
185 
186  irp = CONTAINING_RECORD( entry, IRP, ThreadListEntry );
187  if (irp->Tail.Overlay.OriginalFileObject == fileObject) {
188  found = TRUE;
189  IoCancelIrp( irp );
190  }
191 
192  entry = entry->Flink;
193  }
194 
195  //
196  // Lower the IRQL back down to what it was on entry to this procedure.
197  //
198 
199  KeLowerIrql( irql );
200 
201  if (found) {
202 
203  LARGE_INTEGER interval;
204 
205  //
206  // Delay execution for a time and let the request
207  // finish. The delay time is 10ms.
208  //
209 
210  interval.QuadPart = -10 * 1000 * 10;
211 
212  //
213  // Wait for a while so the canceled requests can complete.
214  //
215 
216  while (found) {
217 
218  (VOID) KeDelayExecutionThread( KernelMode, FALSE, &interval );
219 
220  found = FALSE;
221 
222  //
223  // Raise the IRQL to prevent modification to the IRP list by the
224  // thread's APC routine.
225  //
226 
227  KeRaiseIrql( APC_LEVEL, &irql );
228 
229  //
230  // Check the IRP list for requests which refer to the specified
231  // file object.
232  //
233 
234  entry = thread->IrpList.Flink;
235 
236  while (header != entry) {
237 
238  //
239  // An IRP has been found for this thread. If the IRP refers
240  // to the appropriate file object, remember that it
241  // was found; otherwise, simply continue the loop.
242  //
243 
244  irp = CONTAINING_RECORD( entry, IRP, ThreadListEntry );
245  if (irp->Tail.Overlay.OriginalFileObject == fileObject) {
246  found = TRUE;
247  break;
248  }
249 
250  entry = entry->Flink;
251  }
252 
253  //
254  // Lower the IRQL back down to what it was on entry to this procedure.
255  //
256 
257  KeLowerIrql( irql );
258 
259  }
260  }
261 
262  try {
263 
264  //
265  // Write the status back to the user.
266  //
267 
268  IoStatusBlock->Status = STATUS_SUCCESS;
269  IoStatusBlock->Information = 0L;
270 
271  } except(EXCEPTION_EXECUTE_HANDLER) {
272 
273  //
274  // An exception was incurred attempting to write the caller's
275  // I/O status block; however, the service completed successfully so
276  // just return success.
277  //
278 
279  }
280 
281  //
282  // Dereference the file object.
283  //
284 
285  ObDereferenceObject( fileObject );
286 
287  return STATUS_SUCCESS;
288 }

290 NTSTATUS
291 NtDeleteFile (
292  __in POBJECT_ATTRIBUTES ObjectAttributes
293  )
294 
295 /*++
296 
297 Routine Description:
298 
299  This service deletes the specified file.
300 
301 Arguments:
302 
303  ObjectAttributes - Supplies the attributes to be used for file object (name,
304  SECURITY_DESCRIPTOR, etc.)
305 
306 Return Value:
307 
308  The status returned is the final completion status of the operation.
309 
310 --*/
311 
312 {
313  KPROCESSOR_MODE requestorMode;
314  NTSTATUS status;
315  OPEN_PACKET openPacket;
316  DUMMY_FILE_OBJECT localFileObject;
317  HANDLE handle;
318 
319  PAGED_CODE();
320 
321  //
322  // Get the previous mode; i.e., the mode of the caller.
323  //
324 
325  requestorMode = KeGetPreviousMode();
326 
327  //
328  // Build a parse open packet that tells the parse method to open the file
329  // for open for delete access w/the delete bit set, and then close it.
330  //
331 
332  RtlZeroMemory( &openPacket, sizeof( OPEN_PACKET ) );
333 
334  openPacket.Type = IO_TYPE_OPEN_PACKET;
335  openPacket.Size = sizeof( OPEN_PACKET );
336  openPacket.CreateOptions = FILE_DELETE_ON_CLOSE;
337  openPacket.ShareAccess = (USHORT) FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE;
338  openPacket.Disposition = FILE_OPEN;
339  openPacket.DeleteOnly = TRUE;
340  openPacket.TraversedMountPoint = FALSE;
341  openPacket.LocalFileObject = &localFileObject;
342 
343  //
344  // Update the open count for this process.
345  //
346 
347  IopUpdateOtherOperationCount();
348 
349  //
350  // Open the object by its name. Because of the special DeleteOnly flag
351  // set in the open packet, the parse routine will open the file, and
352  // then realize that it is only deleting the file, and will therefore
353  // immediately dereference the file. This will cause the cleanup and
354  // the close to be sent to the file system, thus causing the file to
355  // be deleted.
356  //
357 
358  status = ObOpenObjectByName( ObjectAttributes,
359  (POBJECT_TYPE) NULL,
360  requestorMode,
361  NULL,
362  DELETE,
363  &openPacket,
364  &handle );
365 
366  //
367  // The operation is successful if the parse check field of the open packet
368  // indicates that the parse routine was actually invoked, and the final
369  // status field of the packet is set to success.
370  //
371 
372  if (openPacket.ParseCheck != OPEN_PACKET_PATTERN) {
373  return status;
374  } else {
375  return openPacket.FinalStatus;
376  }
377 }

379 NTSTATUS
380 NtFlushBuffersFile (
381  __in HANDLE FileHandle,
382  __out PIO_STATUS_BLOCK IoStatusBlock
383  )
384 
385 /*++
386 
387 Routine Description:
388 
389  This service causes all buffered data to the file to be written.
390 
391 Arguments:
392 
393  FileHandle - Supplies a handle to the file whose buffers should be flushed.
394 
395  IoStatusBlock - Address of the caller's I/O status block.
396 
397 Return Value:
398 
399  The status returned is the final completion status of the operation.
400 
401 --*/
402 
403 {
404  PIRP irp;
405  NTSTATUS status;
406  PFILE_OBJECT fileObject;
407  PDEVICE_OBJECT deviceObject;
408  PKEVENT event;
409  KPROCESSOR_MODE requestorMode;
410  PIO_STACK_LOCATION irpSp;
411  IO_STATUS_BLOCK localIoStatus;
412  OBJECT_HANDLE_INFORMATION objectHandleInformation;
413  BOOLEAN synchronousIo;
414  PETHREAD CurrentThread;
415 
416  PAGED_CODE();
417 
418  //
419  // Get the previous mode; i.e., the mode of the caller.
420  //
421 
422  CurrentThread = PsGetCurrentThread ();
423  requestorMode = KeGetPreviousModeByThread(&CurrentThread->Tcb);
424 
425  if (requestorMode != KernelMode) {
426 
427  //
428  // The caller's access mode is not kernel so probe each of the arguments
429  // and capture them as necessary. If any failures occur, the condition
430  // handler will be invoked to handle them. It will simply cleanup and
431  // return an access violation status code back to the system service
432  // dispatcher.
433  //
434 
435  try {
436 
437  //
438  // The IoStatusBlock parameter must be writeable by the caller.
439  //
440 
441  ProbeForWriteIoStatus( IoStatusBlock );
442 
443  } except(EXCEPTION_EXECUTE_HANDLER) {
444 
445  //
446  // An exception was incurred attempting to probe the caller's
447  // I/O status block. Simply return an appropriate error status
448  // code.
449  //
450 
451  return GetExceptionCode();
452 
453  }
454  }
455 
456  //
457  // There were no blatant errors so far, so reference the file object so
458  // the target device object can be found. Note that if the handle does
459  // not refer to a file object, or if the caller does not have the required
460  // access to the file, then it will fail.
461  //
462 
463  status = ObReferenceObjectByHandle( FileHandle,
464  0,
465  IoFileObjectType,
466  requestorMode,
467  (PVOID *) &fileObject,
468  &objectHandleInformation );
469  if (!NT_SUCCESS( status )) {
470  return status;
471  }
472 
473  //
474  // Ensure that the caller has either WRITE or APPEND access to the file
475  // before allowing this call to continue. This is especially important
476  // if the caller opened a volume, where a flush operation may flush more
477  // than what this opener has written to buffers. Note however that if
478  // this is a pipe, then the APPEND access cannot be made since this
479  // access code is overlaid with the CREATE_PIPE_INSTANCE access.
480  //
481 
482  if (SeComputeGrantedAccesses( objectHandleInformation.GrantedAccess,
483  (!(fileObject->Flags & FO_NAMED_PIPE) ?
484  FILE_APPEND_DATA : 0) |
485  FILE_WRITE_DATA ) == 0) {
486  ObDereferenceObject( fileObject );
487  return STATUS_ACCESS_DENIED;
488  }
489 
490  //
491  // Make a special check here to determine whether this is a synchronous
492  // I/O operation. If it is, then wait here until the file is owned by
493  // the current thread. If this is not a (serialized) synchronous I/O
494  // operation, then allocate and initialize the local event.
495  //
496 
497  if (fileObject->Flags & FO_SYNCHRONOUS_IO) {
498 
499  BOOLEAN interrupted;
500 
501  if (!IopAcquireFastLock( fileObject )) {
502  status = IopAcquireFileObjectLock( fileObject,
503  requestorMode,
504  (BOOLEAN) ((fileObject->Flags & FO_ALERTABLE_IO) != 0),
505  &interrupted );
506  if (interrupted) {
507  ObDereferenceObject( fileObject );
508  return status;
509  }
510  }
511  synchronousIo = TRUE;
512  event = NULL;
513  } else {
514 
515  //
516  // This is a synchronous API being invoked for a file that is opened
517  // for asynchronous I/O. This means that this system service is
518  // to synchronize the completion of the operation before returning
519  // to the caller. A local event is used to do this.
520  //
521 
522  event = ExAllocatePool( NonPagedPool, sizeof( KEVENT ) );
523  if (event == NULL) {
524  ObDereferenceObject( fileObject );
525  return STATUS_INSUFFICIENT_RESOURCES;
526  }
527  KeInitializeEvent( event, SynchronizationEvent, FALSE );
528  synchronousIo = FALSE;
529  }
530 
531  //
532  // Set the file object to the Not-Signaled state.
533  //
534 
535  KeClearEvent( &fileObject->Event );
536 
537  //
538  // Get the address of the target device object.
539  //
540 
541  deviceObject = IoGetRelatedDeviceObject( fileObject );
542 
543  //
544  // Allocate and initialize the I/O Request Packet (IRP) for this operation.
545  //
546 
547  irp = IoAllocateIrp( deviceObject->StackSize, FALSE );
548  if (!irp) {
549 
550  //
551  // An exception was incurred while attempting to allocate the IRP.
552  // Cleanup and return an appropriate error status code.
553  //
554 
555  if (event) {
556  ExFreePool( event );
557  }
558 
559  IopAllocateIrpCleanup( fileObject, (PKEVENT) NULL );
560 
561  return STATUS_INSUFFICIENT_RESOURCES;
562  }
563  irp->Tail.Overlay.OriginalFileObject = fileObject;
564  irp->Tail.Overlay.Thread = CurrentThread;
565  irp->RequestorMode = requestorMode;
566 
567  //
568  // Fill in the service independent parameters in the IRP.
569  //
570 
571  if (synchronousIo) {
572  irp->UserEvent = (PKEVENT) NULL;
573  irp->UserIosb = IoStatusBlock;
574  } else {
575  irp->UserEvent = event;
576  irp->UserIosb = &localIoStatus;
577  irp->Flags = IRP_SYNCHRONOUS_API;
578  }
579  irp->Overlay.AsynchronousParameters.UserApcRoutine = (PIO_APC_ROUTINE) NULL;
580 
581  //
582  // Get a pointer to the stack location for the first driver. This is used
583  // to pass the original function codes and parameters.
584  //
585 
586  irpSp = IoGetNextIrpStackLocation( irp );
587  irpSp->MajorFunction = IRP_MJ_FLUSH_BUFFERS;
588  irpSp->FileObject = fileObject;
589 
590  //
591  // Queue the packet, call the driver, and synchronize appropriately with
592  // I/O completion.
593  //
594 
595  status = IopSynchronousServiceTail( deviceObject,
596  irp,
597  fileObject,
598  FALSE,
599  requestorMode,
600  synchronousIo,
601  OtherTransfer );
602 
603  //
604  // If the file for this operation was not opened for synchronous I/O, then
605  // synchronization of completion of the I/O operation has not yet occurred
606  // since the allocated event must be used for synchronous APIs on files
607  // opened for asynchronous I/O. Synchronize the completion of the I/O
608  // operation now.
609  //
610 
611  if (!synchronousIo) {
612 
613  status = IopSynchronousApiServiceTail( status,
614  event,
615  irp,
616  requestorMode,
617  &localIoStatus,
618  IoStatusBlock );
619  }
620 
621  return status;
622 }

625 NtQueryAttributesFile (
626  __in POBJECT_ATTRIBUTES ObjectAttributes,
627  __out PFILE_BASIC_INFORMATION FileInformation
628  )
629 
630 /*++
631 
632 Routine Description:
633 
634  This service queries the basic attributes information for a specified file.
635 
636 Arguments:
637 
638  ObjectAttributes - Supplies the attributes to be used for file object (name,
639  SECURITY_DESCRIPTOR, etc.)
640 
641  FileInformation - Supplies an output buffer to receive the returned file
642  attributes information.
643 
644 Return Value:
645 
646  The status returned is the final completion status of the operation.
647 
648 --*/
649 
650 {
651  KPROCESSOR_MODE requestorMode;
652  NTSTATUS status;
653  OPEN_PACKET openPacket;
654  DUMMY_FILE_OBJECT localFileObject;
655  FILE_NETWORK_OPEN_INFORMATION networkInformation;
656  HANDLE handle;
657 
658  PAGED_CODE();
659 
660  //
661  // Get the previous mode; i.e., the mode of the caller.
662  //
663 
664  requestorMode = KeGetPreviousMode();
665 
666  if (requestorMode != KernelMode) {
667 
668  try {
669 
670  //
671  // The caller's mode is not kernel, so probe the output buffer.
672  //
673 
674  ProbeForWriteSmallStructure( FileInformation,
675  sizeof( FILE_BASIC_INFORMATION ),
676  sizeof( ULONG_PTR ));
677 
678  } except(EXCEPTION_EXECUTE_HANDLER) {
679 
680  //
681  // An exception was incurred while probing the caller's parameters.
682  // Simply return an appropriate error status code.
683  //
684 
685  return GetExceptionCode();
686  }
687  }
688 
689  //
690  // Build a parse open packet that tells the parse method to open the file,
691  // query the file's basic attributes, and close the file.
692  //
693 
694  RtlZeroMemory( &openPacket, sizeof( OPEN_PACKET ) );
695 
696  openPacket.Type = IO_TYPE_OPEN_PACKET;
697  openPacket.Size = sizeof( OPEN_PACKET );
698  openPacket.ShareAccess = (USHORT) FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE;
699  openPacket.Disposition = FILE_OPEN;
700  openPacket.CreateOptions = FILE_OPEN_REPARSE_POINT|FILE_OPEN_FOR_BACKUP_INTENT;
701  openPacket.BasicInformation = FileInformation;
702  openPacket.NetworkInformation = &networkInformation;
703  openPacket.QueryOnly = TRUE;
704  openPacket.TraversedMountPoint = FALSE;
705  openPacket.LocalFileObject = &localFileObject;
706 
707  //
708  // Update the open count for this process.
709  //
710 
711  IopUpdateOtherOperationCount();
712 
713  //
714  // Open the object by its name. Because of the special QueryOnly flag set
715  // in the open packet, the parse routine will open the file, and then
716  // realize that it is only performing a query. It will therefore perform
717  // the query, and immediately close the file.
718  //
719 
720  status = ObOpenObjectByName( ObjectAttributes,
721  (POBJECT_TYPE) NULL,
722  requestorMode,
723  NULL,
724  FILE_READ_ATTRIBUTES,
725  &openPacket,
726  &handle );
727 
728  //
729  // The operation is successful if the parse check field of the open packet
730  // indicates that the parse routine was actually invoked, and the final
731  // status field of the packet is set to success.
732  //
733 
734  if (openPacket.ParseCheck != OPEN_PACKET_PATTERN) {
735  if (NT_SUCCESS(status)) {
736  ZwClose(handle);
737  status = STATUS_OBJECT_TYPE_MISMATCH;
738  }
739  return status;
740  } else {
741  return openPacket.FinalStatus;
742  }
743 }
