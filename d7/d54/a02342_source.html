55 NTSTATUS
56 IopCheckDeviceAndDriver(
57  IN POPEN_PACKET op,
58  IN PDEVICE_OBJECT parseDeviceObject
59  )
60 {
61  NTSTATUS status;
62  KIRQL irql;
63 
64  //
65  // Make sure that the device and its driver are really there and they are
66  // going to stay there. The object itself cannot go away just yet because
67  // the object management system has performed a reference which bumps the
68  // count of the number of reasons why the object must stick around.
69  // However, the driver could be attempting to unload itself, so perform
70  // this check. If the driver is being unloaded, then set the final status
71  // of the operation to "No such device" and return with a NULL file object
72  // pointer.
73  //
74  // Note that it is possible to "open" an exclusive device more than once
75  // provided that the caller is performing a relative open. This feature
76  // is how users "allocate" a device, and then use it to perform operations.
77  //
78 
79  irql = KeAcquireQueuedSpinLock( LockQueueIoDatabaseLock );
80 
81  if (parseDeviceObject->DeviceObjectExtension->ExtensionFlags &
82  (DOE_UNLOAD_PENDING | DOE_DELETE_PENDING | DOE_REMOVE_PENDING | DOE_REMOVE_PROCESSED | DOE_START_PENDING) ||
83  parseDeviceObject->Flags & DO_DEVICE_INITIALIZING) {
84 
85  status = STATUS_NO_SUCH_DEVICE;
86 
87  } else if (parseDeviceObject->Flags & DO_EXCLUSIVE &&
88  parseDeviceObject->ReferenceCount != 0 &&
89  op->RelatedFileObject == NULL &&
90  !(op->Options & IO_ATTACH_DEVICE)) {
91 
92  status = STATUS_ACCESS_DENIED;
93 
94  } else {
95 
96  parseDeviceObject->ReferenceCount++;
97  status = STATUS_SUCCESS;
98 
99  }
100 
101  KeReleaseQueuedSpinLock( LockQueueIoDatabaseLock, irql );
102 
103  return status;
104 }

106 PVPB
107 IopCheckVpbMounted(
108  IN POPEN_PACKET op,
109  IN PDEVICE_OBJECT parseDeviceObject,
110  IN OUT PUNICODE_STRING RemainingName,
111  OUT PNTSTATUS status
112  )
113 {
114  PVPB vpb;
115  PVPB mountVpb;
116  KIRQL irql;
117  BOOLEAN alertable;
118 
119  //
120  // Loop here until the VPB_MOUNTED test can be passed while holding the
121  // VPB spinlock. After the mount succeeds, it is still necessary to acquire
122  // the spinlock to check that the VPB (which may be different from the one
123  // before the mount) is still mounted. If it is, then its reference count
124  // is incremented before releasing the spinlock.
125  //
126 
127  irql = KeAcquireQueuedSpinLock( LockQueueIoVpbLock );
128 
129  while (!(parseDeviceObject->Vpb->Flags & VPB_MOUNTED)) {
130 
131  KeReleaseQueuedSpinLock( LockQueueIoVpbLock, irql );
132 
133  alertable = (op->CreateOptions & FILE_SYNCHRONOUS_IO_ALERT) ? TRUE : FALSE;
134 
135  //
136  // Try to mount the volume, allowing only RAW to perform the mount if
137  // this is a DASD open.
138  //
139 
140  mountVpb = NULL;
141  *status = IopMountVolume( parseDeviceObject,
142  (BOOLEAN) (!RemainingName->Length && !op->RelatedFileObject),
143  FALSE,
144  alertable,
145  &mountVpb );
146  //
147  // If the mount operation was unsuccessful, adjust the reference
148  // count for the device and return now.
149  //
150 
151  if (!NT_SUCCESS( *status ) || *status == STATUS_USER_APC || *status == STATUS_ALERTED) {
152 
153  IopDecrementDeviceObjectRef( parseDeviceObject, FALSE, FALSE );
154 
155  if (!NT_SUCCESS( *status )) {
156  return NULL;
157  } else {
158  *status = STATUS_WRONG_VOLUME;
159  return NULL;
160  }
161  } else {
162 
163  //
164  // In this case IopMountVolume did the synchronization already.
165  //
166 
167  if (mountVpb) {
168  return mountVpb;
169  }
170  }
171 
172  irql = KeAcquireQueuedSpinLock( LockQueueIoVpbLock );
173  }
174 
175  //
176  // Synchronize here with the file system to make sure that volumes do not
177  // go away while en route to the FS.
178  //
179 
180  vpb = parseDeviceObject->Vpb;
181 
182  //
183  // Check here that the VPB is not locked.
184  //
185 
186  if (vpb->Flags & VPB_LOCKED) {
187 
188  *status = STATUS_ACCESS_DENIED;
189  vpb = NULL;
190 
191  } else {
192 
193  vpb->ReferenceCount += 1;
194  }
195 
196  KeReleaseQueuedSpinLock( LockQueueIoVpbLock, irql );
197 
198  //
199  // This is because VPB is locked.
200  // Do the decrement outside the VPB lock because of possible deadlocks.
201  //
202 
203  if (!vpb) {
204  IopDecrementDeviceObjectRef( parseDeviceObject, FALSE, FALSE );
205  }
206 
207  return vpb;
208 }

234 IopCheckTopDeviceHint(
235  IN OUT PDEVICE_OBJECT *TargetDeviceObject,
236  IN POPEN_PACKET Op,
237  IN BOOLEAN DirectDeviceOpen
238 )
239 /*++
240 
241 Routine Description:
242 
243  Check a device object hint specified in an open packet for validity.
244 
245 Arguments:
246 
247  TargetDeviceObject - Device object whose stack the hint must be on
248  for it to be valid; holds the hint on return if it is valid.
249 
250  Op - Open packet that contains the device object hint.
251 
252  DirectDeviceOpen - Specifies whether the open is a direct device open.
253 
254 Return Value:
255 
256  STATUS_INVALID_PARAMETER - The hint is invalid for this open.
257 
258  STATUS_MOUNT_POINT_NOT_RESOLVED - The target device object and the hint
259  are on different stacks and a mount point was previously traversed.
260 
261  STATUS_INVALID_DEVICE_OBJECT_PARAMETER - The target device object and the
262  hint are on different stacks.
263 
264  STATUS_SUCCESS otherwise.
265 
266 --*/
267 {
268  PDEVICE_OBJECT deviceObject = *TargetDeviceObject;
269 
270  //
271  // You cannot use the device object hint if you are trying to
272  // open the device directly or if you are dealing with a device
273  // that is not a file system. In these cases, return an error.
274  //
275 
276  if (DirectDeviceOpen ||
277  (deviceObject->DeviceType != FILE_DEVICE_DISK_FILE_SYSTEM &&
278  deviceObject->DeviceType != FILE_DEVICE_CD_ROM_FILE_SYSTEM &&
279  deviceObject->DeviceType != FILE_DEVICE_TAPE_FILE_SYSTEM &&
280  deviceObject->DeviceType != FILE_DEVICE_NETWORK_FILE_SYSTEM &&
281  deviceObject->DeviceType != FILE_DEVICE_DFS_FILE_SYSTEM)) {
282 
283  return STATUS_INVALID_PARAMETER;
284  }
285 
286  if (IopVerifyDeviceObjectOnStack(deviceObject, Op->TopDeviceObjectHint)) {
287 
288  *TargetDeviceObject = Op->TopDeviceObjectHint;
289 
290  } else {
291 
292  if (Op->TraversedMountPoint) {
293  Op->TraversedMountPoint = FALSE;
294  return STATUS_MOUNT_POINT_NOT_RESOLVED;
295  } else {
296  return STATUS_INVALID_DEVICE_OBJECT_PARAMETER;
297  }
298  }
299 
300  return STATUS_SUCCESS;
301 }

304 IopParseDevice(
305  IN PVOID ParseObject,
306  IN PVOID ObjectType,
307  IN PACCESS_STATE AccessState,
308  IN KPROCESSOR_MODE AccessMode,
309  IN ULONG Attributes,
310  IN OUT PUNICODE_STRING CompleteName,
311  IN OUT PUNICODE_STRING RemainingName,
312  IN OUT PVOID Context OPTIONAL,
313  IN PSECURITY_QUALITY_OF_SERVICE SecurityQos OPTIONAL,
314  OUT PVOID *Object
315  )
316 
317 /*++
318 
319 Routine Description:
320 
321  This routine interfaces to the NT Object Manager. It is invoked when
322  the object system is given the name of an entity to create or open and the
323  name translates to a device object. This routine is specified as the parse
324  routine for all device objects.
325 
326  In the normal case of an NtCreateFile, the user specifies either the name
327  of a device or of a file. In the former situation, this routine is invoked
328  with a pointer to the device and a null ("") string. For this case, the
329  routine simply allocates an IRP, fills it in, and passes it to the driver
330  for the device. The driver will then perform whatever rudimentary functions
331  are necessary and will return a status code indicating whether an error was
332  incurred. This status code is remembered in the Open Packet (OP).
333 
334  In the latter situation, the name string to be opened/created is non-null.
335  That is, it contains the remainder of the pathname to the file that is to
336  be opened or created. For this case, the routine allocates an IRP, fills
337  it in, and passes it to the driver for the device. The driver may then
338  need to take further action or it may complete the request immediately. If
339  it needs to perform some work asynchronously, then it can queue the request
340  and return a status of STATUS_PENDING. This allows this routine and its
341  caller to return to the user so that he can continue. Otherwise, the open/
342  create is basically finished.
343 
344  If the driver supports symbolic links, then it is also possible for the
345  driver to return a new name. This name will be returned to the Object
346  Manager as a new name to look up. The parsing will then begin again from
347  the start.
348 
349  It is also the responsibility of this routine to create a file object for
350  the file, if the name specifies a file. The file object's address is
351  returned to the NtCreateFile service through the OP.
352 
353 Arguments:
354 
355  ParseObject - Pointer to the device object the name translated into.
356 
357  ObjectType - Type of the object being opened.
358 
359  AccessState - Running security access state information for operation.
360 
361  AccessMode - Access mode of the original caller.
362 
363  Attributes - Attributes to be applied to the object.
364 
365  CompleteName - Complete name of the object.
366 
367  RemainingName - Remaining name of the object.
368 
369  Context - Pointer to an Open Packet (OP) from NtCreateFile service.
370 
371  SecurityQos - Optional security quality of service indicator.
372 
373  Object - The address of a variable to receive the created file object, if
374  any.
375 
376 Return Value:
377 
378  The function return value is one of the following:
379 
380  a) Success - This indicates that the function succeeded and the object
381  parameter contains the address of the created file object.
382 
383  b) Error - This indicates that the file was not found or created and
384  no file object was created.
385 
386  c) Reparse - This indicates that the remaining name string has been
387  replaced by a new name that is to be parsed.
388 
389 --*/
390

392

395  PIO_STACK_LOCATION irpSp;
396  POPEN_PACKET op;
397  PFILE_OBJECT fileObject;
398  NTSTATUS status;

403  BOOLEAN checkDeviceHint = FALSE;

406  ACCESS_MASK desiredAccess;
407  PDUMMY_FILE_OBJECT localFileObject;
408  LOGICAL realFileObjectRequired;
409  KPROCESSOR_MODE modeForPrivilegeCheck;
410  ULONG retryCount = 0;
411  BOOLEAN relativeVolumeOpen = FALSE; // True if opening a filesystem volume
412  PETHREAD CurrentThread;
413  ULONG returnedLength;
414 
415  PAGED_CODE();

417  UNREFERENCED_PARAMETER (ObjectType);
418 
419  CurrentThread = PsGetCurrentThread ();

421 reparse_loop:
422 
423  //
424  // Assume failure by setting the returned object pointer to NULL.
425  //
426 
427  *Object = (PVOID) NULL;
428 
429  //
430  // Get the address of the Open Packet (OP).
431  //
432 
433  op = Context;
434 
435  //
436  // Ensure that this routine is actually being invoked because someone is
437  // attempting to open a device or a file through NtCreateFile. This code
438  // must be invoked from there (as opposed to some other random object
439  // create or open routine).
440  //
441 
442  if (op == NULL ||
443  op->Type != IO_TYPE_OPEN_PACKET ||
444  op->Size != sizeof( OPEN_PACKET )) {
445 
446  return STATUS_OBJECT_TYPE_MISMATCH;
447  }
448 
449  //
450  // Obtain a pointer to the parse object as a device object, which is the
451  // actual type of the object anyway.
452  //
453 
454  parseDeviceObject = (PDEVICE_OBJECT) ParseObject;
455 
456  //
457  // If we passed through a mountpoint do an extra set of validation checks
458  // that we don't go to a remote device. We really have to let the object manager
459  // open the new path. If we fail it in the path that obtains the reparse point and
460  // validates the name (using IoIsValidNameGraftingBuffer) its not sufficient. This is because
461  // the path may be valid at that time and change before OB does the reparse.
462  //
463 
464  if (op->TraversedMountPoint) {
465 
466  ASSERT (op->Information == IO_REPARSE_TAG_MOUNT_POINT);
467 
468  if ((parseDeviceObject->DeviceType != FILE_DEVICE_DISK) &&
469  (parseDeviceObject->DeviceType != FILE_DEVICE_CD_ROM) &&
470  (parseDeviceObject->DeviceType != FILE_DEVICE_VIRTUAL_DISK) &&
471  (parseDeviceObject->DeviceType != FILE_DEVICE_TAPE)) {
472 
473  status = STATUS_IO_REPARSE_DATA_INVALID;
474  return op->FinalStatus = status;
475  }
476  }
477 
478  //
479  // If this is a relative open, then get the device on which the file
480  // is really being opened from the related file object and use that for
481  // the remainder of this function and for all operations performed on
482  // the file object that is about to be created.
483  //
484 
485  if (op->RelatedFileObject) {
486  parseDeviceObject = op->RelatedFileObject->DeviceObject;
487  }
488 
489  //
490  // Make sure that the device and its driver are really there and they are
491  // going to stay there. The object itself cannot go away just yet because
492  // the object management system has performed a reference which bumps the
493  // count of the number of reasons why the object must stick around.
494  // However, the driver could be attempting to unload itself, so perform
495  // this check. If the driver is being unloaded, then set the final status
496  // of the operation to "No such device" and return with a NULL file object
497  // pointer.
498  //
499  // Note that it is possible to "open" an exclusive device more than once
500  // provided that the caller is performing a relative open. This feature
501  // is how users "allocate" a device, and then use it to perform operations.
502  //
503 
504  status = IopCheckDeviceAndDriver( op, parseDeviceObject );
505 
506  if (!NT_SUCCESS(status)) {
507  return op->FinalStatus = status;
508  }
509 
510  //
511  // Since ObOpenObjectByName is called without being passed
512  // any object type information, we need to map the generic
513  // bits in the DesiredAccess mask here. We also need to save
514  // the object's generic mapping in the access state structure
515  // here, because this is the earliest opportunity we have
516  // to do so.
517  //
518 
519  RtlMapGenericMask( &AccessState->RemainingDesiredAccess,
520  &IoFileObjectType->TypeInfo.GenericMapping );
521 
522  RtlMapGenericMask( &AccessState->OriginalDesiredAccess,
523  &IoFileObjectType->TypeInfo.GenericMapping );
524 
525  SeSetAccessStateGenericMapping( AccessState, &IoFileObjectType->TypeInfo.GenericMapping );
526 
527  desiredAccess = AccessState->RemainingDesiredAccess;
528 
529  //
530  // Compute the previous mode to be passed in to the privilege check
531  //
532 
533  if (AccessMode != KernelMode || op->Options & IO_FORCE_ACCESS_CHECK) {
534  modeForPrivilegeCheck = UserMode;
535  } else {
536  modeForPrivilegeCheck = KernelMode;
537  }
538 
539  IopCheckBackupRestorePrivilege( AccessState,
540  &op->CreateOptions,
541  modeForPrivilegeCheck,
542  op->Disposition
543  );
544 
545  //
546  // If this is not the first time through here for this object, and the
547  // object itself is being opened, then the desired access must also
548  // include the previously granted access from the last pass. Likewise,
549  // if the privileges have been checked already, then this is another
550  // pass through for a file, so add in the previously granted access.
551  //
552 
553  if ((op->Override && !RemainingName->Length) ||
554  AccessState->Flags & SE_BACKUP_PRIVILEGES_CHECKED) {
555  desiredAccess |= AccessState->PreviouslyGrantedAccess;
556  }
557 
558  //
559  // If its a filesystem volume open and we are doing a relative open to it
560  // then do the access check. Note that relative opens can be nested and we propagate
561  // the fact that the relative open is for a volume using the FO_VOLUME_OPEN flag.
562  //
563 
564  if (op->RelatedFileObject) {
565  if ((op->RelatedFileObject->Flags & FO_VOLUME_OPEN) && RemainingName->Length == 0) {
566  relativeVolumeOpen = TRUE;
567  }
568  }
569 
570  //
571  // Now determine what type of security check should be made. This is
572  // based on whether the remaining name string is null. If it is null,
573  // then the device itself is being opened, so a full security check is
574  // performed. Otherwise, only a check to ensure that the caller can
575  // traverse the device object is made. Note that these checks are only
576  // made if the caller's mode is user, or if access checking is being
577  // forced. Note also that if an access check was already made on the
578  // device itself, and this code is being executed again because of a
579  // reparse, then the access check need not be made the second time
580  // around.
581  //
582 
583 
584  if ((AccessMode != KernelMode || op->Options & IO_FORCE_ACCESS_CHECK) &&
585  (!op->RelatedFileObject || relativeVolumeOpen) &&
586  !op->Override) {
587 
588  BOOLEAN subjectContextLocked = FALSE;
589  BOOLEAN accessGranted;
590  ACCESS_MASK grantedAccess;
591 
592  //
593  // The caller's mode is either user or access checking is being
594  // forced. Perform the appropriate access check on the device
595  // object.
596  //
597 
598  if (!RemainingName->Length) {
599 
600  UNICODE_STRING nameString;
601  PPRIVILEGE_SET privileges = NULL;
602 
603  //
604  // The device itself is being opened. Make a full security check
605  // to ensure that the caller has the appropriate access.
606  //
607 
608  KeEnterCriticalRegionThread( &CurrentThread->Tcb );
609  ExAcquireResourceSharedLite( &IopSecurityResource, TRUE );
610 
611  SeLockSubjectContext( &AccessState->SubjectSecurityContext );
612  subjectContextLocked = TRUE;
613 
614  accessGranted = SeAccessCheck( parseDeviceObject->SecurityDescriptor,
615  &AccessState->SubjectSecurityContext,
616  subjectContextLocked,
617  desiredAccess,
618  0,
619  &privileges,
620  &IoFileObjectType->TypeInfo.GenericMapping,
621  UserMode,
622  &grantedAccess,
623  &status );
624 
625  if (privileges) {
626  (VOID) SeAppendPrivileges( AccessState,
627  privileges );
628  SeFreePrivileges( privileges );
629  }
630 
631  if (accessGranted) {
632  AccessState->PreviouslyGrantedAccess |= grantedAccess;
633  AccessState->RemainingDesiredAccess &= ~( grantedAccess | MAXIMUM_ALLOWED );
634  op->Override = TRUE;
635  }
636 
637  nameString.Length = 8;
638  nameString.MaximumLength = 8;
639  nameString.Buffer = L"File";
640 
641  SeOpenObjectAuditAlarm( &nameString,
642  parseDeviceObject,
643  CompleteName,
644  parseDeviceObject->SecurityDescriptor,
645  AccessState,
646  FALSE,
647  accessGranted,
648  UserMode,
649  &AccessState->GenerateOnClose );
650 
651  ExReleaseResourceLite( &IopSecurityResource );
652  KeLeaveCriticalRegionThread( &CurrentThread->Tcb );
653 
654  } else {
655 
656  //
657  // The device is not being opened, rather, a file on the device
658  // is being opened or created. Therefore, only perform a check
659  // here for traverse access to the device.
660  //
661 
662  //
663  // First determine if we have to perform traverse checking at all.
664  // Traverse checking only needs to be done if the device being
665  // traversed is a disk, or if the caller does not already have
666  // traverse checking privilege. Note that the former case is so
667  // that an administrator can turn off access to the "system
668  // partition", or someone would be able to install a trojan horse
669  // into the system by simply replacing one of the files there with
670  // something of their own.
671  //
672 
673  if (!(AccessState->Flags & TOKEN_HAS_TRAVERSE_PRIVILEGE) ||
674  parseDeviceObject->DeviceType == FILE_DEVICE_DISK ||
675  parseDeviceObject->DeviceType == FILE_DEVICE_CD_ROM ) {
676 
677  KeEnterCriticalRegionThread( &CurrentThread->Tcb );
678  ExAcquireResourceSharedLite( &IopSecurityResource, TRUE );
679 
680  accessGranted = SeFastTraverseCheck( parseDeviceObject->SecurityDescriptor,
681  AccessState,
682  FILE_TRAVERSE,
683  UserMode );
684 
685  if (!accessGranted) {
686 
687  PPRIVILEGE_SET privileges = NULL;
688 
689  //
690  // The caller was not granted traverse access through the
691  // normal fast path lookup. Perform a full-blown access
692  // check to determine whether some other ACE allows traverse
693  // access.
694  //
695 
696  SeLockSubjectContext( &AccessState->SubjectSecurityContext );
697 
698  subjectContextLocked = TRUE;
699 
700  accessGranted = SeAccessCheck( parseDeviceObject->SecurityDescriptor,
701  &AccessState->SubjectSecurityContext,
702  subjectContextLocked,
703  FILE_TRAVERSE,
704  0,
705  &privileges,
706  &IoFileObjectType->TypeInfo.GenericMapping,
707  UserMode,
708  &grantedAccess,
709  &status );
710 
711  if (privileges) {
712 
713  (VOID) SeAppendPrivileges( AccessState,
714  privileges );
715  SeFreePrivileges( privileges );
716  }
717 
718  }
719 
720  ExReleaseResourceLite( &IopSecurityResource );
721  KeLeaveCriticalRegionThread( &CurrentThread->Tcb );
722 
723  } else {
724 
725  accessGranted = TRUE;
726  }
727  }
728 
729  //
730  // Unlock the subject's security context so that it can be changed,
731  // if it was locked.
732  //
733 
734  if (subjectContextLocked) {
735  SeUnlockSubjectContext( &AccessState->SubjectSecurityContext );
736  }
737 
738  //
739  // Finally, determine whether or not access was granted to the device.
740  // If not, clean everything up and get out now without even invoking
741  // the device driver.
742  //
743 
744  if (!accessGranted) {
745 
746  IopDecrementDeviceObjectRef( parseDeviceObject, FALSE, FALSE );
747  return STATUS_ACCESS_DENIED;
748  }
749 
750  }
751 
752  realFileObjectRequired = !(op->QueryOnly || op->DeleteOnly);
753 
754  if (RemainingName->Length == 0 &&
755  op->RelatedFileObject == NULL &&
756  ((desiredAccess & ~(SYNCHRONIZE |
757  FILE_READ_ATTRIBUTES |
