34 /*++
35 
36 Routine Description:
37 
38  This routine initializes and inserts a new Vcb record into the in-memory
39  data structure. The Vcb record "hangs" off the end of the Volume device
40  object and must be allocated by our caller.
41 
42 Arguments:
43 
44  Vcb - Supplies the address of the Vcb record being initialized.
45 
46  TargetDeviceObject - Supplies the address of the target device object to
47  associate with the Vcb record.
48 
49  Vpb - Supplies the address of the Vpb to associate with the Vcb record.
50 
51 Return Value:
52 
53  NTSTATUS for any errors
54 
55 --*/
56 
57 {
58  NTSTATUS Status = STATUS_SUCCESS;
59 
60  PAGED_CODE();
61 
62  //
63  // We start by first zeroing out all of the VCB, this will guarantee
64  // that any stale data is wiped clean
65  //
66 
67  RtlZeroMemory( Vcb, sizeof(VCB) );
68 
69  //
70  // Set the proper node type code and node byte size
71  //
72 
73  Vcb->NodeTypeCode = RAW_NTC_VCB;
74  Vcb->NodeByteSize = sizeof(VCB);
75 
76  //
77  // Set the Target Device Object, Vpb, and Vcb State fields
78  //
79 
80  //
81  // No need to take a extra reference on the Target Device object as
82  // IopMountVolume already has taken a reference.
83  //
84 
85  Vcb->TargetDeviceObject = TargetDeviceObject;
86  Vcb->Vpb = Vpb;
87 
88  //
89  // Initialize the Mutex.
90  //
91 
92  KeInitializeMutex( &Vcb->Mutex, MUTEX_LEVEL_FILESYSTEM_RAW_VCB );
93 
94  //
95  // allocate the spare vpb for forced dismount
96  //
97 
98  Vcb->SpareVpb = ExAllocatePoolWithTag( NonPagedPool, sizeof( VPB ), 'Raw ');
99  if (Vcb->SpareVpb == NULL) {
100  Status = STATUS_INSUFFICIENT_RESOURCES;
101  }
102 
103  //
104  // and return to our caller
105  //
106 
107  return Status;
108 }

110 BOOLEAN
111 RawCheckForDismount (
112  PVCB Vcb,
113  BOOLEAN CalledFromCreate
114  )
115 
116 /*++
117 
118 Routine Description:
119 
120  This routine determines if a volume is ready for deletion. It
121  correctly synchronizes with creates en-route to the file system.
122  On exit if the vcb is deleted the mutex is released
123 
124 Arguments:
125 
126  Vcb - Supplies the value to examine
127 
128  CalledFromCreate - Tells us if we should allow 0 or 1 in VpbRefCount
129 
130 Return Value:
131 
132  BOOLEAN - TRUE if the volume was deleted, FALSE otherwise.
133 
134 --*/
135 
136 {
137 
138  KIRQL SavedIrql;
139  ULONG ReferenceCount = 0;
140  BOOLEAN DeleteVolume = FALSE;
141 
142  //
143  // We must enter with the vcb mutex acquired
144  //
145 
146  ASSERT( KeReadStateMutant( &Vcb->Mutex ) == 0 );
147 
148  IoAcquireVpbSpinLock( &SavedIrql );
149 
150  ReferenceCount = Vcb->Vpb->ReferenceCount;
151 
152  {
153  PVPB Vpb;
154 
155  Vpb = Vcb->Vpb;
156 
157  //
158  // If a create is in progress on this volume, don't
159  // delete it.
160  //
161 
162  if ( ReferenceCount != (ULONG)(CalledFromCreate ? 1 : 0) ) {
163 
164  //
165  // Cleanup the vpb on a forced dismount even if we can't delete the vcb if
166  // we haven't already done so
167  //
168 
169  if ((Vcb->SpareVpb != NULL) &&
170  FlagOn( Vcb->VcbState, VCB_STATE_FLAG_DISMOUNTED )) {
171 
172  //
173  // Setup the spare vpb and put it on the real device
174  //
175 
176  RtlZeroMemory( Vcb->SpareVpb, sizeof( VPB ) );
177 
178  Vcb->SpareVpb->Type = IO_TYPE_VPB;
179  Vcb->SpareVpb->Size = sizeof( VPB );
180  Vcb->SpareVpb->RealDevice = Vcb->Vpb->RealDevice;
181  Vcb->SpareVpb->DeviceObject = NULL;
182  Vcb->SpareVpb->Flags = FlagOn( Vcb->Vpb->Flags, VPB_REMOVE_PENDING );
183 
184  Vcb->Vpb->RealDevice->Vpb = Vcb->SpareVpb;
185 
186  //
187  // The spare vpb now belongs to the iosubsys and we own the original one
188  //
189 
190  Vcb->SpareVpb = NULL;
191  Vcb->Vpb->Flags |= VPB_PERSISTENT;
192 
193  }
194 
195  DeleteVolume = FALSE;
196 
197  } else {
198 
199  DeleteVolume = TRUE;
200 
201  if ( Vpb->RealDevice->Vpb == Vpb ) {
202 
203  Vpb->DeviceObject = NULL;
204 
205  Vpb->Flags &= ~VPB_MOUNTED;
206  }
207  }
208  }
209  IoReleaseVpbSpinLock( SavedIrql );
210 
211  if (DeleteVolume) {
212 
213  (VOID)KeReleaseMutex( &Vcb->Mutex, FALSE );
214 
215  //
216  // Free the spare vpb if we didn't use it or the original one if
217  // we did use it and there are no more reference counts. Otherwise i/o
218  // subsystem still has a ref and will free the vpb itself
219  //
220 
221  if (Vcb->SpareVpb) {
222  ExFreePool( Vcb->SpareVpb );
223  } else if (ReferenceCount == 0) {
224  ExFreePool( Vcb->Vpb );
225  }
226 
227  ObDereferenceObject( Vcb->TargetDeviceObject );
228  IoDeleteDevice( (PDEVICE_OBJECT)CONTAINING_RECORD( Vcb,
229  VOLUME_DEVICE_OBJECT,
230  Vcb));
231  }
232 
233  return DeleteVolume;
234 }
