93 //
94 // Macro definition for defining IOCTL and FSCTL function control codes. Note
95 // that function codes 0-2047 are reserved for Microsoft Corporation, and
96 // 2048-4095 are reserved for customers.
97 //
98 
99 #define CTL_CODE( DeviceType, Function, Method, Access ) ( \
100  ((DeviceType) << 16) | ((Access) << 14) | ((Function) << 2) | (Method) \
101 )
