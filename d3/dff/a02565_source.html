52 /*++
53 
54 Routine Description:
55 
56  This function atomically signals the specified signal object and then
57  waits until the specified wait object attains a state of Signaled. An
58  optional timeout can also be specified. If a timeout is not specified,
59  then the wait will not be satisfied until the wait object attains a
60  state of Signaled. If a timeout is specified, and the wait object has
61  not attained a state of Signaled when the timeout expires, then the
62  wait is automatically satisfied. If an explicit timeout value of zero
63  is specified, then no wait will occur if the wait cannot be satisfied
64  immediately. The wait can also be specified as alertable.
65 
66 Arguments:
67 
68  SignalHandle - Supplies the handle of the signal object.
69 
70  WaitHandle - Supplies the handle for the wait object.
71 
72  Alertable - Supplies a boolean value that specifies whether the wait
73  is alertable.
74 
75  Timeout - Supplies an pointer to an absolute or relative time over
76  which the wait is to occur.
77 
78 Return Value:
79 
80  The wait completion status. A value of STATUS_TIMEOUT is returned if a
81  timeout occurred. A value of STATUS_SUCCESS is returned if the specified
82  object satisfied the wait. A value of STATUS_ALERTED is returned if the
83  wait was aborted to deliver an alert to the current thread. A value of
84  STATUS_USER_APC is returned if the wait was aborted to deliver a user
85  APC to the current thread.
86 
87 --*/
88 
89 {
90  OBJECT_HANDLE_INFORMATION HandleInformation;
91  KPROCESSOR_MODE PreviousMode;
92  PVOID RealObject;
93  PVOID SignalObject;
94  POBJECT_HEADER SignalObjectHeader;
95  NTSTATUS Status;
96  LARGE_INTEGER TimeoutValue;
97  PVOID WaitObject;
98  POBJECT_HEADER WaitObjectHeader;
99 
100  //
101  // Establish an exception handler and probe the specified timeout value
102  // if necessary. If the probe fails, then return the exception code as
103  // the service status.
104  //
105 
106  PreviousMode = KeGetPreviousMode();
107 
108  if ((ARGUMENT_PRESENT(Timeout)) && (PreviousMode != KernelMode)) {
109 
110  try {
111 
112  TimeoutValue = ProbeAndReadLargeInteger(Timeout);
113  Timeout = &TimeoutValue;
114 
115  } except(EXCEPTION_EXECUTE_HANDLER) {
116 
117  return GetExceptionCode();
118  }
119  }
120 
121  //
122  // Reference the signal object by handle.
123  //
124 
125  Status = ObReferenceObjectByHandle( SignalHandle,
126  0,
127  NULL,
128  PreviousMode,
129  &SignalObject,
130  &HandleInformation );
131 
132  //
133  // If the reference was successful, then reference the wait object by
134  // handle.
135  //
136 
137  if (NT_SUCCESS(Status)) {
138 
139  Status = ObReferenceObjectByHandle( WaitHandle,
140  SYNCHRONIZE,
141  NULL,
142  PreviousMode,
143  &WaitObject,
144  NULL );
145 
146  //
147  // If the reference was successful, then determine the real wait
148  // object, check the signal object access, signal the signal object,
149  // and wait for the real wait object.
150  //
151 
152  if (NT_SUCCESS(Status)) {
153 
154  WaitObjectHeader = OBJECT_TO_OBJECT_HEADER(WaitObject);
155  RealObject = WaitObjectHeader->Type->DefaultObject;
156 
157  if ((LONG_PTR)RealObject >= 0) {
158 
159  RealObject = (PVOID)((PCHAR)WaitObject + (ULONG_PTR)RealObject);
160  }
161 
162  //
163  // If the signal object is an event, then check for modify access
164  // and set the event. Otherwise, if the signal object is a
165  // mutant, then attempt to release ownership of the mutant.
166  // Otherwise, if the object is a semaphore, then check for modify
167  // access and release the semaphore. Otherwise, the signal objet
168  // is invalid.
169  //
170 
171  SignalObjectHeader = OBJECT_TO_OBJECT_HEADER(SignalObject);
172  Status = STATUS_ACCESS_DENIED;
173 
174  if (SignalObjectHeader->Type == ExEventObjectType) {
175 
176  //
177  // Check for access to the specified event object,
178  //
179 
180  if ((PreviousMode != KernelMode) &&
181  (SeComputeDeniedAccesses( HandleInformation.GrantedAccess,
182  EVENT_MODIFY_STATE) != 0 )) {
183 
184  goto WaitExit;
185  }
186 
187  //
188  // Set the specified event and wait atomically.
189  //
190  // N.B. This returns with the dispatcher lock held !!!
191  //
192 
193  KeSetEvent((PKEVENT)SignalObject, EVENT_INCREMENT, TRUE);
194 
195  } else if (SignalObjectHeader->Type == ExMutantObjectType) {
196 
197  //
198  // Release the specified mutant and wait atomically.
199  //
200  // N.B. The release will only be successful if the current
201  // thread is the owner of the mutant.
202  //
203 
204  try {
205 
206  KeReleaseMutant( (PKMUTANT)SignalObject,
207  MUTANT_INCREMENT,
208  FALSE,
209  TRUE );
210 
211  } except((GetExceptionCode () == STATUS_ABANDONED ||
212  GetExceptionCode () == STATUS_MUTANT_NOT_OWNED)?
213  EXCEPTION_EXECUTE_HANDLER :
214  EXCEPTION_CONTINUE_SEARCH) {
215  Status = GetExceptionCode();
216 
217  goto WaitExit;
218  }
219 
220  } else if (SignalObjectHeader->Type == ExSemaphoreObjectType) {
221 
222  //
223  // Check for access to the specified semaphore object,
224  //
225 
226  if ((PreviousMode != KernelMode) &&
227  (SeComputeDeniedAccesses( HandleInformation.GrantedAccess,
228  SEMAPHORE_MODIFY_STATE) != 0 )) {
229 
230  goto WaitExit;
231  }
232 
233  //
234  // Release the specified semaphore and wait atomically.
235  //
236 
237  try {
238 
239  //
240  // Release the specified semaphore and wait atomically.
241  //
242 
243  KeReleaseSemaphore( (PKSEMAPHORE)SignalObject,
244  SEMAPHORE_INCREMENT,
245  1,
246  TRUE );
247 
248  } except((GetExceptionCode () == STATUS_SEMAPHORE_LIMIT_EXCEEDED)?
249  EXCEPTION_EXECUTE_HANDLER :
250  EXCEPTION_CONTINUE_SEARCH) {
251 
252  Status = GetExceptionCode();
253 
254  goto WaitExit;
255  }
256 
257  } else {
258 
259  Status = STATUS_OBJECT_TYPE_MISMATCH;
260 
261  goto WaitExit;
262  }
263 
264  //
265  // Protect the wait call just in case KeWait decides to raise
266  // For example, a mutant level is exceeded
267  //
268 
269  try {
270 
271  Status = KeWaitForSingleObject( RealObject,
272  UserRequest,
273  PreviousMode,
274  Alertable,
275  Timeout );
276 
277  } except((GetExceptionCode () == STATUS_MUTANT_LIMIT_EXCEEDED)?
278  EXCEPTION_EXECUTE_HANDLER :
279  EXCEPTION_CONTINUE_SEARCH) {
280 
281  Status = GetExceptionCode();
282  }
283 
284 WaitExit:
285 
286  ObDereferenceObject(WaitObject);
287  }
288 
289  ObDereferenceObject(SignalObject);
290  }
291 
292  return Status;
293 }

302 /*++
303 
304 Routine Description:
305 
306  This function waits until the specified object attains a state of
307  Signaled. An optional timeout can also be specified. If a timeout
308  is not specified, then the wait will not be satisfied until the
309  object attains a state of Signaled. If a timeout is specified, and
310  the object has not attained a state of Signaled when the timeout
311  expires, then the wait is automatically satisfied. If an explicit
312  timeout value of zero is specified, then no wait will occur if the
313  wait cannot be satisfied immediately. The wait can also be specified
314  as alertable.
315 
316 Arguments:
317 
318  Handle - Supplies the handle for the wait object.
319 
320  Alertable - Supplies a boolean value that specifies whether the wait
321  is alertable.
322 
323  Timeout - Supplies an pointer to an absolute or relative time over
324  which the wait is to occur.
325 
326 Return Value:
327 
328  The wait completion status. A value of STATUS_TIMEOUT is returned if a
329  timeout occurred. A value of STATUS_SUCCESS is returned if the specified
330  object satisfied the wait. A value of STATUS_ALERTED is returned if the
331  wait was aborted to deliver an alert to the current thread. A value of
332  STATUS_USER_APC is returned if the wait was aborted to deliver a user
333  APC to the current thread.
334 
335 --*/
336 
337 {
338  PVOID Object;
339  POBJECT_HEADER ObjectHeader;
340  KPROCESSOR_MODE PreviousMode;
341  NTSTATUS Status;
342  LARGE_INTEGER TimeoutValue;
343  PVOID WaitObject;
344 
345  PAGED_CODE();
346 
347  //
348  // Get previous processor mode and probe and capture timeout argument
349  // if necessary.
350  //
351 
352  PreviousMode = KeGetPreviousMode();
353 
354  if ((ARGUMENT_PRESENT(Timeout)) && (PreviousMode != KernelMode)) {
355 
356  try {
357 
358  TimeoutValue = ProbeAndReadLargeInteger(Timeout);
359  Timeout = &TimeoutValue;
360 
361  } except(EXCEPTION_EXECUTE_HANDLER) {
362 
363  return GetExceptionCode();
364  }
365  }
366 
367  //
368  // Get a referenced pointer to the specified object with synchronize
369  // access.
370  //
371 
372  Status = ObReferenceObjectByHandle( Handle,
373  SYNCHRONIZE,
374  NULL,
375  PreviousMode,
376  &Object,
377  NULL );
378 
379  //
380  // If access is granted, then check to determine if the specified object
381  // can be waited on.
382  //
383 
384  if (NT_SUCCESS(Status)) {
385 
386  ObjectHeader = OBJECT_TO_OBJECT_HEADER( Object );
387  WaitObject = ObjectHeader->Type->DefaultObject;
388 
389  if ((LONG_PTR)WaitObject >= 0) {
390 
391  WaitObject = (PVOID)((PCHAR)Object + (ULONG_PTR)WaitObject);
392  }
393 
394  //
395  // Protect the wait call just in case KeWait decides to raise
396  // For example, a mutant level is exceeded
397  //
398 
399  try {
400  PERFINFO_DECLARE_OBJECT(Object);
401 
402  Status = KeWaitForSingleObject( WaitObject,
403  UserRequest,
404  PreviousMode,
405  Alertable,
406  Timeout );
407 
408  } except((GetExceptionCode () == STATUS_MUTANT_LIMIT_EXCEEDED)?
409  EXCEPTION_EXECUTE_HANDLER :
410  EXCEPTION_CONTINUE_SEARCH) {
411 
412  Status = GetExceptionCode();
413  }
414 
415  ObDereferenceObject(Object);
416  }
417 
418  return Status;

422 NtWaitForMultipleObjects (
423  __in ULONG Count,
424  __in_ecount(Count) HANDLE Handles[],

426  __in BOOLEAN Alertable,
427  __in_opt PLARGE_INTEGER Timeout
428  )

430 /*++
431 
432 Routine Description:
433 
434  This function waits until the specified objects attain a state of
435  Signaled.
436 
437 Arguments:
438 
439  Count - Supplies a count of the number of objects that are to be waited
440  on.
441 
442  Handles[] - Supplies an array of handles to wait objects.
443 
444  WaitType - Supplies the type of wait to perform (WaitAll, WaitAny).
445 
446  Alertable - Supplies a boolean value that specifies whether the wait is
447  alertable.
448 
449  Timeout - Supplies a pointer to an optional absolute of relative time over
450  which the wait is to occur.
451 
452 Return Value:
453 
454  The wait completion status. A value of STATUS_TIMEOUT is returned if a
455  timeout occurred. The index of the object (zero based) in the object
456  pointer array is returned if an object satisfied the wait. A value of
457  STATUS_ALERTED is returned if the wait was aborted to deliver an alert
458  to the current thread. A value of STATUS_USER_APC is returned if the
459  wait was aborted to deliver a user APC to the current thread.
460 
461 --*/
462 
463 {
464 
465  HANDLE CapturedHandles[MAXIMUM_WAIT_OBJECTS];
466  KPROCESSOR_MODE PreviousMode;
467  LARGE_INTEGER TimeoutValue;
468 
469  PAGED_CODE();
470 
471  //
472  // If the number of objects is zero or greater than the largest number
473  // that can be waited on, then return and invalid parameter status.
474  //
475 
476  if ((Count == 0) || (Count > MAXIMUM_WAIT_OBJECTS)) {
477  return STATUS_INVALID_PARAMETER_1;
478  }
479 
480  //
481  // If the wait type is not wait any or wait all, then return an invalid
482  // parameter status.
483  //
484 
485  if ((WaitType != WaitAny) && (WaitType != WaitAll)) {
486  return STATUS_INVALID_PARAMETER_3;
487  }
488 
489  //
490  // Get previous processor mode and probe and capture input arguments if
491  // necessary.
492  //
493 
494  PreviousMode = KeGetPreviousMode();
495  try {
496  if (PreviousMode != KernelMode) {
497  if (ARGUMENT_PRESENT(Timeout)) {
498  TimeoutValue = ProbeAndReadLargeInteger(Timeout);
499  Timeout = &TimeoutValue;
500  }
501 
502  ProbeForRead( Handles, Count * sizeof(HANDLE), sizeof(HANDLE) );
503  }
504 
505  RtlCopyMemory (CapturedHandles, Handles, Count * sizeof(HANDLE));
506 
507  } except(EXCEPTION_EXECUTE_HANDLER) {
508  return GetExceptionCode();
509  }
510 
511  return ObpWaitForMultipleObjects(Count,
512  CapturedHandles,
513  WaitType,
514  Alertable,
515  Timeout);
516 }

527 /*++
528 
529 Routine Description:
530 
531  This function waits until the specified objects attain a state of
532  Signaled. This function is provided to avoid thunking a 32-bit handle
533  table to a 64-bit handle table in wow64.
534 
535 Arguments:
536 
537  Count - Supplies a count of the number of objects that are to be waited
538  on.
539 
540  Handles[] - Supplies an array of 32-bit handles to wait objects.
541 
542  WaitType - Supplies the type of wait to perform (WaitAll, WaitAny).
543 
544  Alertable - Supplies a boolean value that specifies whether the wait is
545  alertable.
546 
547  Timeout - Supplies a pointer to an optional absolute of relative time over
548  which the wait is to occur.
549 
550 Return Value:
551 
552  The wait completion status. A value of STATUS_TIMEOUT is returned if a
553  timeout occurred. The index of the object (zero based) in the object
554  pointer array is returned if an object satisfied the wait. A value of
555  STATUS_ALERTED is returned if the wait was aborted to deliver an alert
556  to the current thread. A value of STATUS_USER_APC is returned if the
557  wait was aborted to deliver a user APC to the current thread.
558 
559 --*/

561 {
562 
563  HANDLE CapturedHandles[MAXIMUM_WAIT_OBJECTS];
564  ULONG Index;
565  KPROCESSOR_MODE PreviousMode;
566  LARGE_INTEGER TimeoutValue;
567 
568  PAGED_CODE();
569 
570  //
571  // If the number of objects is zero or greater than the largest number
572  // that can be waited on, then return and invalid parameter status.
573  //
574 
575  if ((Count == 0) || (Count > MAXIMUM_WAIT_OBJECTS)) {
576  return STATUS_INVALID_PARAMETER_1;
577  }
578 
579  //
580  // Get previous processor mode and probe and capture input arguments if
581  // necessary.
582  //
583 
584  PreviousMode = KeGetPreviousMode();
585  try {
586  if (PreviousMode != KernelMode) {

614 NTSTATUS
615 ObpWaitForMultipleObjects (
616  IN ULONG Count,
617  IN HANDLE CapturedHandles[],
618  IN WAIT_TYPE WaitType,
619  IN BOOLEAN Alertable,
620  IN PLARGE_INTEGER Timeout OPTIONAL
621  )
622 
623 /*++
624 
625 Routine Description:
626 
627  This function waits until the specified objects attain a state of
628  Signaled. The wait can be specified to wait until all of the objects
629  attain a state of Signaled or until one of the objects attains a state
630  of Signaled. An optional timeout can also be specified. If a timeout
631  is not specified, then the wait will not be satisfied until the objects
632  attain a state of Signaled. If a timeout is specified, and the objects
633  have not attained a state of Signaled when the timeout expires, then
634  the wait is automatically satisfied. If an explicit timeout value of
635  zero is specified, then no wait will occur if the wait cannot be satisfied
636  immediately. The wait can also be specified as alertable.
637 
638 Arguments:
639 
640  Count - Supplies a count of the number of objects that are to be waited
641  on.
642 
643  CapturedHandles[] - Supplies an array of handles to wait objects.
644 
645  WaitType - Supplies the type of wait to perform (WaitAll, WaitAny).
646 
647  Alertable - Supplies a boolean value that specifies whether the wait is
648  alertable.
649 
650  Timeout - Supplies a pointer to an optional absolute of relative time over
651  which the wait is to occur.
652 
653 Return Value:
654 
655  The wait completion status. A value of STATUS_TIMEOUT is returned if a
656  timeout occurred. The index of the object (zero based) in the object
657  pointer array is returned if an object satisfied the wait. A value of
658  STATUS_ALERTED is returned if the wait was aborted to deliver an alert
659  to the current thread. A value of STATUS_USER_APC is returned if the
660  wait was aborted to deliver a user APC to the current thread.
661 
662 --*/
663 
664 {
665 
666  ULONG i;
667  ULONG j;
668  POBJECT_HEADER ObjectHeader;
669  PVOID Objects[MAXIMUM_WAIT_OBJECTS];
670  KPROCESSOR_MODE PreviousMode;
671  ULONG RefCount;
672  ULONG Size;
673  NTSTATUS Status;
674  PKWAIT_BLOCK WaitBlockArray;
675  ACCESS_MASK GrantedAccess;
676  PVOID WaitObjects[MAXIMUM_WAIT_OBJECTS];
677  PHANDLE_TABLE HandleTable;
678  PHANDLE_TABLE_ENTRY HandleEntry;
679  BOOLEAN InCriticalRegion = FALSE;
680  PETHREAD CurrentThread;
681 
682  PAGED_CODE();
683 
684  //
685  // If the number of objects to be waited on is greater than the number
686  // of builtin wait blocks, then allocate an array of wait blocks from
687  // nonpaged pool. If the wait block array cannot be allocated, then
688  // return insufficient resources.
689  //
690 
691  PreviousMode = KeGetPreviousMode();
692 
693  WaitBlockArray = NULL;
694 
695  if (Count > THREAD_WAIT_OBJECTS) {
696 
697  Size = Count * sizeof( KWAIT_BLOCK );
698  WaitBlockArray = ExAllocatePoolWithTag(NonPagedPool, Size, 'tiaW');
699 
700  if (WaitBlockArray == NULL) {
701 
702  return STATUS_INSUFFICIENT_RESOURCES;
703  }
704  }
705 
706  //
707  // Loop through the array of handles and get a referenced pointer to
708  // each object.
709  //
710 
711  i = 0;
712  RefCount = 0;
713 
714  Status = STATUS_SUCCESS;
715 
716  //
717  // Protect ourselves from being interrupted while we hold a handle table
718  // entry lock
719  //
720 
721  CurrentThread = PsGetCurrentThread ();
722  KeEnterCriticalRegionThread(&CurrentThread->Tcb);
723  InCriticalRegion = TRUE;
724 
725  do {
726 
727  //
728  // Get a pointer to the object table entry. Check if this is a kernel
729  // handle and if so then use the kernel handle table otherwise use the
730  // processes handle table. If we are going for a kernel handle we'll
731  // need to attach to the kernel process otherwise we need to ensure
732  // that we aren't attached.
733  //
734 
735  if (IsKernelHandle( CapturedHandles[i], PreviousMode )) {
736 
737  HANDLE KernelHandle;
738 
739  //
740  // Decode the user supplied handle into a regular handle value
741  // and get its handle table entry
742  //
743 
744  KernelHandle = DecodeKernelHandle( CapturedHandles[i] );
745 
746  HandleTable = ObpKernelHandleTable;
747  HandleEntry = ExMapHandleToPointerEx ( HandleTable, KernelHandle, PreviousMode );
748 
749  } else {
750 
751  //
752  // Get the handle table entry
753  //
754 
755  HandleTable = PsGetCurrentProcessByThread (CurrentThread)->ObjectTable;
756  HandleEntry = ExMapHandleToPointerEx ( HandleTable, CapturedHandles[ i ], PreviousMode );
757  }
758 
759  //
760  // Make sure the handle really did translate to a valid
761  // entry
762  //
763 
764  if (HandleEntry != NULL) {
765 
766  //
767  // Get the granted access for the handle
768  //
769 
770 #if i386
771 
772  if (NtGlobalFlag & FLG_KERNEL_STACK_TRACE_DB) {
773 
774  GrantedAccess = ObpTranslateGrantedAccessIndex( HandleEntry->GrantedAccessIndex );
775 
776  } else {
777 
778  GrantedAccess = ObpDecodeGrantedAccess(HandleEntry->GrantedAccess);
779  }
780 
781 #else
782  GrantedAccess = ObpDecodeGrantedAccess(HandleEntry->GrantedAccess);
783 
784 #endif // i386
785 
786  //
787  // Make sure the handle as synchronize access to the
788  // object
789  //
790 
791  if ((PreviousMode != KernelMode) &&
792  (SeComputeDeniedAccesses( GrantedAccess, SYNCHRONIZE ) != 0)) {
793 
794  Status = STATUS_ACCESS_DENIED;
795 
796  ExUnlockHandleTableEntry( HandleTable, HandleEntry );
797 
798  goto ServiceFailed;
799 
800  } else {
801 
802  //
803  // We have a object with proper access so get the header
804  // and if the default objects points to a real object
805  // then that is the one we're going to wait on.
806  // Otherwise we'll find the kernel wait object at an
807  // offset into the object body
808  //
809 
810  ObjectHeader = (POBJECT_HEADER)(((ULONG_PTR)(HandleEntry->Object)) & ~OBJ_HANDLE_ATTRIBUTES);
811 
812  if ((LONG_PTR)ObjectHeader->Type->DefaultObject < 0) {
813 
814  RefCount += 1;
815  Objects[i] = NULL;
816  WaitObjects[i] = ObjectHeader->Type->DefaultObject;
817 
818  } else {
819 
820  ObpIncrPointerCount( ObjectHeader );
821  RefCount += 1;
822  Objects[i] = &ObjectHeader->Body;
823 
824  PERFINFO_DECLARE_OBJECT(Objects[i]);
825 
826  //
827  // Compute the address of the kernel wait object.
828  //
829 
830  WaitObjects[i] = (PVOID)((PCHAR)&ObjectHeader->Body +
831  (ULONG_PTR)ObjectHeader->Type->DefaultObject);
832  }
833  }
834 
835  ExUnlockHandleTableEntry( HandleTable, HandleEntry );
836 
837  } else {
838 
839  //
840  // The entry in the handle table isn't in use
841  //
842 
843  Status = STATUS_INVALID_HANDLE;
