618 BOOLEAN
619 ExAcquireResourceExclusiveLite(
620  __inout PERESOURCE Resource,
621  __in BOOLEAN Wait
622  )
623 
624 /*++
625 
626 Routine Description:
627 
628  The routine acquires the specified resource for exclusive access.
629 
630 Arguments:
631 
632  Resource - Supplies a pointer to the resource that is acquired
633  for exclusive access.
634 
635  Wait - A boolean value that specifies whether to wait for the
636  resource to become available if access cannot be granted
637  immediately.
638 
639 Return Value:
640 
641  BOOLEAN - TRUE if the resource is acquired and FALSE otherwise.
642 
643 --*/
644 
645 {
646 
647  ERESOURCE_THREAD CurrentThread;
648  EXP_LOCK_HANDLE LockHandle;
649  BOOLEAN Result;
650 
651  ASSERT((Resource->Flag & ResourceNeverExclusive) == 0);
652 
653  //
654  // Acquire exclusive access to the specified resource.
655  //
656 
657  CurrentThread = (ERESOURCE_THREAD)PsGetCurrentThread();
658  ASSERT(KeIsExecutingDpc() == FALSE);
659  ASSERT_RESOURCE(Resource);
660 
661  EXP_LOCK_RESOURCE(Resource, &LockHandle);
662 
663  //
664  // Resource acquisition must be protected from thread suspends.
665  //
666 
667  EX_ENSURE_APCS_DISABLED (LockHandle.OldIrql,
668  Resource,
669  KeGetCurrentThread());
670 
671  ExpIncrementCounter(ExclusiveAcquire);
672 
673  //
674  // If the active count of the resource is zero, then there is neither
675  // an exclusive owner nor a shared owner and access to the resource can
676  // be immediately granted. Otherwise, there is either a shared owner or
677  // an exclusive owner.
678  //
679 
680 retry:
681  if (Resource->ActiveCount != 0) {
682 
683  //
684  // The resource is either owned exclusive or shared.
685  //
686  // If the resource is owned exclusive and the current thread is the
687  // owner, then increment the recursion count.
688  //
689 
690  if (IsOwnedExclusive(Resource) &&
691  (Resource->OwnerThreads[0].OwnerThread == CurrentThread)) {
692  Resource->OwnerThreads[0].OwnerCount += 1;
693  Result = TRUE;
694 
695  } else {
696 
697  //
698  // The resource is either owned exclusive by some other thread,
699  // or owned shared.
700  //
701  // If wait is not specified, then return that the resource was
702  // not acquired. Otherwise, wait for exclusive access to the
703  // resource to be granted.
704  //
705 
706  if (Wait == FALSE) {
707  Result = FALSE;
708 
709  } else {
710 
711  //
712  // If the exclusive wait event has not yet been allocated,
713  // then the long path code must be taken.
714  //
715 
716  if (Resource->ExclusiveWaiters == NULL) {
717  ExpAllocateExclusiveWaiterEvent(Resource, &LockHandle);
718  goto retry;
719  }
720 
721  //
722  // Wait for exclusive access to the resource to be granted
723  // and set the owner thread.
724  //
725 
726  Resource->NumberOfExclusiveWaiters += 1;
727  EXP_UNLOCK_RESOURCE(Resource, &LockHandle);
728  ExpWaitForResource(Resource, Resource->ExclusiveWaiters);
729 
730  //
731  // N.B. It is "safe" to store the owner thread without
732  // obtaining any locks since the thread has already
733  // been granted exclusive ownership.
734  //
735 
736  Resource->OwnerThreads[0].OwnerThread = (ERESOURCE_THREAD)PsGetCurrentThread();
737  return TRUE;
738  }
739  }
740 
741  } else {
742 
743  //
744  // The resource is not owned.
745  //
746 
747  Resource->Flag |= ResourceOwnedExclusive;
748  Resource->OwnerThreads[0].OwnerThread = CurrentThread;
749  Resource->OwnerThreads[0].OwnerCount = 1;
750  Resource->ActiveCount = 1;
751  Result = TRUE;
752  }
753 
754  EXP_UNLOCK_RESOURCE(Resource, &LockHandle);
755  return Result;
756 }
