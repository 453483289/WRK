264 /*++
265 
266 Routine Description:
267 
268  This function removes the next entry from the Queue object entry
269  list. If no list entry is available, then the calling thread is
270  put in a wait state.
271 
272  N.B. The wait discipline for Queue object LIFO.
273 
274 Arguments:
275 
276  Queue - Supplies a pointer to a dispatcher object of type Queue.
277 
278  WaitMode - Supplies the processor mode in which the wait is to occur.
279 
280  Timeout - Supplies a pointer to an optional absolute of relative time over
281  which the wait is to occur.
282 
283 Return Value:
284 
285  The address of the entry removed from the Queue object entry list or
286  STATUS_TIMEOUT.
287 
288  N.B. These values can easily be distinguished by the fact that all
289  addresses in kernel mode have the high order bit set.
290 
291 --*/
292 
293 {
294 
295  PKPRCB CurrentPrcb;
296  LARGE_INTEGER DueTime;
297  PLIST_ENTRY Entry;
298  ULONG Hand;
299  LARGE_INTEGER NewTime;
300  PRKQUEUE OldQueue;
301  PLARGE_INTEGER OriginalTime;
302  LOGICAL StackSwappable;
303  PRKTHREAD Thread;
304  PRKTIMER Timer;
305  PRKWAIT_BLOCK WaitBlock;
306  LONG_PTR WaitStatus;
307  PRKWAIT_BLOCK WaitTimer;
308 
309  ASSERT_QUEUE(Queue);
310 
311  //
312  // Set constant variables.
313  //
314 
315  Hand = 0;
316  OriginalTime = Timeout;
317  Thread = KeGetCurrentThread();
318  Timer = &Thread->Timer;
319  WaitBlock = &Thread->WaitBlock[0];
320  WaitTimer = &Thread->WaitBlock[TIMER_WAIT_BLOCK];
321 
322  //
323  // If the dispatcher database lock is already held, then initialize the
324  // local variables. Otherwise, raise IRQL to SYNCH_LEVEL, initialize the
325  // thread local variables, and lock the dispatcher database.
326  //
327 
328  if (Thread->WaitNext) {
329  Thread->WaitNext = FALSE;
330  InitializeRemoveQueue();
331 
332  } else {
333  Thread->WaitIrql = KeRaiseIrqlToSynchLevel();
334  InitializeRemoveQueue();
335  KiLockDispatcherDatabaseAtSynchLevel();
336  }
337 
338  //
339  // Check if the thread is currently processing a queue entry and whether
340  // the new queue is the same as the old queue.
341  //
342 
343  OldQueue = Thread->Queue;
344  Thread->Queue = Queue;
345  if (Queue != OldQueue) {
346 
347  //
348  // If the thread was previously associated with a queue, then remove
349  // the thread from the old queue object thread list and attempt to
350  // activate another thread.
351  //
352 
353  Entry = &Thread->QueueListEntry;
354  if (OldQueue != NULL) {
355  RemoveEntryList(Entry);
356  KiActivateWaiterQueue(OldQueue);
357  }
358 
359  //
360  // Insert thread in the thread list of the new queue that the thread
361  // will be associate with.
362  //
363 
364  InsertTailList(&Queue->ThreadListHead, Entry);
365 
366  } else {
367

369  // The previous and current queue are the same queue - decrement the
370  // current number of threads.
371  //
372 
373  Queue->CurrentCount -= 1;
374  }
375 
376  //
377  // Start of wait loop.
378  //
379  // Note this loop is repeated if a kernel APC is delivered in the
380  // middle of the wait or a kernel APC is pending on the first attempt
381  // through the loop.
382  //
383  // If the Queue object entry list is not empty, then remove the next
384  // entry from the Queue object entry list. Otherwise, wait for an entry
385  // to be inserted in the queue.
386  //
387

389 
390  //
391  // Check if there is a queue entry available and the current
392  // number of active threads is less than target maximum number
393  // of threads.
394  //
395

397  if ((Entry != &Queue->EntryListHead) &&
398  (Queue->CurrentCount < Queue->MaximumCount)) {
399 
400  //
401  // Decrement the number of entries in the queue object entry list,
402  // increment the number of active threads, remove the next entry
403  // from the list, and set the forward link to NULL.
404  //
405 
406  Queue->Header.SignalState -= 1;
407  Queue->CurrentCount += 1;
408  if ((Entry->Flink == NULL) || (Entry->Blink == NULL)) {
409  KeBugCheckEx(INVALID_WORK_QUEUE_ITEM,
410  (ULONG_PTR)Entry,
411  (ULONG_PTR)Queue,
412  (ULONG_PTR)&ExWorkerQueue[0],
413  (ULONG_PTR)((PWORK_QUEUE_ITEM)Entry)->WorkerRoutine);
414  }
415 
416  RemoveEntryList(Entry);
417  Entry->Flink = NULL;
418  break;
419 
420  } else {
421 
422  //
423  // Test to determine if a kernel APC is pending.
424  //
425  // If a kernel APC is pending, the special APC disable count is
426  // zero, and the previous IRQL was less than APC_LEVEL, then a
427  // kernel APC was queued by another processor just after IRQL was
428  // raised to DISPATCH_LEVEL, but before the dispatcher database
429  // was locked.

432  //

436  (Thread->WaitIrql < APC_LEVEL)) {
437 
438  //
439  // Increment the current thread count, unlock the dispatcher
440  // database, and exit the dispatcher. An APC interrupt will
441  // immediately occur which will result in the delivery of the
442  // kernel APC, if possible.
443  //
444

446  KiUnlockDispatcherDatabaseFromSynchLevel();

448 
449  } else {
450 
451  //
452  // Test if a user APC is pending.
453  //
454 
455  if ((WaitMode != KernelMode) && (Thread->ApcState.UserApcPending)) {
456  Entry = (PLIST_ENTRY)ULongToPtr(STATUS_USER_APC);
457  Queue->CurrentCount += 1;
458  break;
459  }
460

462  // Check to determine if a timeout value is specified.
463  //
464 
465  if (ARGUMENT_PRESENT(Timeout)) {
466 
467  //
468  // Check if the timer has already expired.
469  //
470  // N.B. The constant fields of the timer wait block are
471  // initialized when the thread is initialized. The
472  // constant fields include the wait object, wait key,
473  // wait type, and the wait list entry link pointers.
474  //
475 
476  if (KiCheckDueTime(Timer) == FALSE) {
477  Entry = (PLIST_ENTRY)ULongToPtr(STATUS_TIMEOUT);
478  Queue->CurrentCount += 1;
479  break;
480  }
481  }
482

484  // Insert wait block in object wait list.
485  //
486 
487  InsertTailList(&Queue->Header.WaitListHead, &WaitBlock->WaitListEntry);
488 
489  //
490  // Set the thread wait parameters, set the thread dispatcher
491  // state to Waiting, and insert the thread in the wait list.
492  //
493 
494  CurrentPrcb = KeGetCurrentPrcb();
495  Thread->State = Waiting;
496  if (StackSwappable != FALSE) {
497  InsertTailList(&CurrentPrcb->WaitListHead, &Thread->WaitListEntry);
498  }
499 
500  //
501  // Set swap busy for the current thread, unlock the dispatcher
502  // database, and switch to a new thread.
503  //
504  // Control is returned at the original IRQL.
505  //
506 
507  ASSERT(Thread->WaitIrql <= DISPATCH_LEVEL);
508 
509  KiSetContextSwapBusy(Thread);
510 
511  if (ARGUMENT_PRESENT(Timeout)) {
512  KiInsertOrSignalTimer(Timer, Hand);
513 
514  } else {
515  KiUnlockDispatcherDatabaseFromSynchLevel();
516  }
517 
518  WaitStatus = KiSwapThread(Thread, CurrentPrcb);
519 
520  //
521  // If the thread was not awakened to deliver a kernel mode APC,
522  // then return wait status.
523  //
524 
525  Thread->WaitReason = 0;
526  if (WaitStatus != STATUS_KERNEL_APC) {
527  return (PLIST_ENTRY)WaitStatus;
528  }
529 
530  if (ARGUMENT_PRESENT(Timeout)) {
531 
532  //
533  // Reduce the amount of time remaining before timeout occurs.
534  //
535 
536  Timeout = KiComputeWaitInterval(OriginalTime,
537  &DueTime,
538  &NewTime);
539  }
540  }
541 
542  //
543  // Raise IRQL to SYNCH level, initialize the local variables,
544  // lock the dispatcher database, and decrement the count of
545  // active threads.
546  //
547 
548  Thread->WaitIrql = KeRaiseIrqlToSynchLevel();
549  InitializeRemoveQueue();
550  KiLockDispatcherDatabaseAtSynchLevel();
551  Queue->CurrentCount -= 1;
552  }
553 
554  } while (TRUE);
555 
556  //
557  // Unlock the dispatcher database, exit the dispatcher, and return the
558  // list entry address or a status of timeout.
559  //
560 
561  KiUnlockDispatcherDatabaseFromSynchLevel();
562  KiExitDispatcher(Thread->WaitIrql);
563  return Entry;
564 }

571 /*++
572 
573 Routine Description:
574 
575  This function runs down the specified queue by removing the listhead
576  from the queue list, removing any associated threads from the thread
577  list, and returning the address of the first entry.
578 
579 
580 Arguments:
581 
582  Queue - Supplies a pointer to a dispatcher object of type Queue.
583 
584 Return Value:
585 
586  If the queue list is not empty, then the address of the first entry in
587  the queue is returned as the function value. Otherwise, a value of NULL
588  is returned.
589 
590 --*/
591 
592 {
593 
594  PLIST_ENTRY Entry;
595  PLIST_ENTRY FirstEntry;
596  KIRQL OldIrql;
597  PKTHREAD Thread;
598 
599  ASSERT_QUEUE(Queue);
600  ASSERT(KeGetCurrentIrql() <= DISPATCH_LEVEL);
601  ASSERT(IsListEmpty(&Queue->Header.WaitListHead));
602 
603  //
604  // Raise IRQL to SYNCH level and lock the dispatcher database.
605  //
606 
607  KiLockDispatcherDatabase(&OldIrql);
608 
609  //
610  // Get the address of the first entry in the queue and check if the
611  // list is empty or contains entries that should be flushed. If there
612  // are no entries in the list, then set the return value to NULL.
613  // Otherwise, set the return value to the address of the first list
614  // entry and remove the listhead from the list.
615  //
616 
617  FirstEntry = Queue->EntryListHead.Flink;
618  if (FirstEntry == &Queue->EntryListHead) {
619  FirstEntry = NULL;
620 
621  } else {
622  RemoveEntryList(&Queue->EntryListHead);
623  }
624 
625  //
626  // Remove all associated threads from the thread list of the queue.
627  //
628 
629  while (Queue->ThreadListHead.Flink != &Queue->ThreadListHead) {
630  Entry = Queue->ThreadListHead.Flink;
631  Thread = CONTAINING_RECORD(Entry, KTHREAD, QueueListEntry);
632  Thread->Queue = NULL;
633  RemoveEntryList(Entry);
634  }
635 
636 #if DBG
637 
638  Queue->EntryListHead.Flink = Queue->EntryListHead.Blink = NULL;
639  Queue->ThreadListHead.Flink = Queue->ThreadListHead.Blink = NULL;
640  Queue->Header.WaitListHead.Flink = Queue->Header.WaitListHead.Blink = NULL;
641 
642 #endif

645  // Unlock the dispatcher database, exit the dispatcher, and return the
646  // function value.
647  //

649  KiUnlockDispatcherDatabaseFromSynchLevel();
650  KiExitDispatcher(OldIrql);
651  return FirstEntry;
652 }

662 /*++
663 
664 Routine Description:
665 
666  This function inserts the specified entry in the queue object entry
667  list and attempts to satisfy the wait of a single waiter.
668 
669  N.B. The wait discipline for Queue object is LIFO.
670 
671 Arguments:
672 
673  Queue - Supplies a pointer to a dispatcher object of type Queue.
674 
675  Entry - Supplies a pointer to a list entry that is inserted in the
676  queue object entry list.
677 
678  Head - Supplies a boolean value that determines whether the queue
679  entry is inserted at the head or tail of the queue if it can
680  not be immediately dispatched.
681 
682 Return Value:
683 
684  The previous signal state of the Queue object.
685 
686 --*/
687 
688 {
689 
690  LONG OldState;
691  PRKTHREAD Thread;
692  PKTIMER Timer;
693  PKWAIT_BLOCK WaitBlock;
694  PLIST_ENTRY WaitEntry;
695 
696  ASSERT_QUEUE(Queue);
697 
698  //
699  // Capture the current signal state of queue object and check if there
700  // is a thread waiting on the queue object, the current number of active
701  // threads is less than the target number of threads, and the wait reason
702  // of the current thread is not queue wait or the wait queue is not the
703  // same queue as the insertion queue. If these conditions are satisfied,
704  // then satisfy the thread wait and pass the thread the address of the
705  // queue entry as the wait status. Otherwise, set the state of the queue
706  // object to signaled and insert the specified entry in the queue object
707  // entry list.
708  //
709 
710  OldState = Queue->Header.SignalState;
711  Thread = KeGetCurrentThread();
712  WaitEntry = Queue->Header.WaitListHead.Blink;
713  if ((WaitEntry != &Queue->Header.WaitListHead) &&
714  (Queue->CurrentCount < Queue->MaximumCount) &&
715  ((Thread->Queue != Queue) ||
716  (Thread->WaitReason != WrQueue))) {
717 
718  //
719  // Remove the last wait block from the wait list and get the address
720  // of the waiting thread object.
721  //
722 
723  RemoveEntryList(WaitEntry);
724  WaitBlock = CONTAINING_RECORD(WaitEntry, KWAIT_BLOCK, WaitListEntry);
725  Thread = WaitBlock->Thread;
726 
727  //
728  // Set the wait completion status, remove the thread from its wait
729  // list, increment the number of active threads, and clear the wait
730  // reason.
731  //
732 
733  Thread->WaitStatus = (LONG_PTR)Entry;
734  if (Thread->WaitListEntry.Flink != NULL) {
735  RemoveEntryList(&Thread->WaitListEntry);
736  }
737 
738  Queue->CurrentCount += 1;
739  Thread->WaitReason = 0;
740 
741  //
742  // If thread timer is still active, then cancel thread timer.
743  //
744 
745  Timer = &Thread->Timer;
746  if (Timer->Header.Inserted == TRUE) {
747  KiRemoveTreeTimer(Timer);
748  }
749 
750  //
751  // Ready the thread for execution.
752  //
753 
754  KiReadyThread(Thread);
755 
756  } else {
757  Queue->Header.SignalState += 1;
758  if (Head != FALSE) {
759  InsertHeadList(&Queue->EntryListHead, Entry);
760 
761  } else {
762  InsertTailList(&Queue->EntryListHead, Entry);
763  }
764  }
765 
766  return OldState;
767 }
