91 Ki386CheckDivideByZeroTrap (
92  IN PKTRAP_FRAME UserFrame
93  )
94 
95 /*++
96 
97 Routine Description:
98 
99  This function gains control when the x86 processor generates a
100  divide by zero trap. The x86 design generates such a trap on
101  divide by zero and on division overflows. In order to determine
102  which exception code to dispatch, the divisor of the "div" or "idiv"
103  instruction needs to be inspected.
104 
105 Arguments:
106 
107  UserFrame - Trap frame of the divide by zero trap
108 
109 Return Value:
110 
111  exception code dispatch
112 
113 --*/
114 
115 {
116 
117  ULONG operandsize, operandmask, i, accum;
118  PUCHAR istream, pRM;
119  UCHAR ibyte, rm;
120  PKMOD Mod;
121  BOOLEAN fPrefix;
122  NTSTATUS status;
123  KIRQL OldIrql;
124 
125  status = STATUS_INTEGER_DIVIDE_BY_ZERO;
126 
127  //
128  // Raise IRQL to synchronize access to the trap frame
129  //
130  OldIrql = KeGetCurrentIrql();
131  if (OldIrql < APC_LEVEL) {
132  KeRaiseIrql(APC_LEVEL, &OldIrql);
133  }
134 
135  if (UserFrame->SegCs == KGDT_R0_CODE && (UserFrame->EFlags & EFLAGS_V86_MASK) == 0) {
136 
137  //
138  // Divide by zero exception from Kernel Mode?
139  // Likely bad hardware interrupt and the device or vector table
140  // is corrupt. Bugcheck NOW so we can figure out what went wrong.
141  // If we try and proceed, then we'll likely fault in reading the
142  // top of user space, and then double fault (page fault in the
143  // div zero handler.) -- This is a debugging consideration.
144  // You can't put breakpoints on the trap labels so this is hard
145  // to debug.
146  //
147 
148  KeBugCheck (UNEXPECTED_KERNEL_MODE_TRAP);
149  }
150 
151  //
152  // read instruction prefixes
153  //
154 
155  fPrefix = TRUE;
156  pRM = RM32;
157  operandsize = 4;
158  operandmask = 0xffffffff;
159  ibyte = 0;
160  istream = (PUCHAR) UserFrame->Eip;
161 
162  try {
163 
164  while (fPrefix) {
165  ibyte = ProbeAndReadUchar(istream);
166  istream++;
167  switch (ibyte) {
168  case 0x2e: // cs override
169  case 0x36: // ss override
170  case 0x3e: // ds override
171  case 0x26: // es override
172  case 0x64: // fs override
173  case 0x65: // gs override
174  case 0xF3: // rep
175  case 0xF2: // rep
176  case 0xF0: // lock
177  break;
178 
179  case 0x66:
180  // 16 bit operand override
181  operandsize = 2;
182  operandmask = 0xffff;
183  break;
184 
185  case 0x67:
186  // 16 bit address size override
187  // this is some non-flat code
188  goto try_exit;
189 
190  default:
191  fPrefix = FALSE;
192  break;
193  }
194  }
195 
196  //
197  // Check instruction opcode
198  //
199 
200  if (ibyte != 0xf7 && ibyte != 0xf6) {
201  // this is not a DIV or IDIV opcode
202  goto try_exit;
203  }
204 
205  if (ibyte == 0xf6) {
206  // this is a byte div or idiv
207  operandsize = 1;
208  operandmask = 0xff;
209  }
210 
211  //
212  // Get Mod R/M
213  //
214 
215  ibyte = ProbeAndReadUchar (istream);
216  istream++;
217  Mod = MOD32 + (ibyte >> 6);
218  rm = ibyte & 7;
219 
220  //
221  // put register values into accum
222  //
223 
224  if (operandsize == 1 && (ibyte & 0xc0) == 0xc0) {
225  pRM = RM8;
226  }
227 
228  accum = 0;
229  if (rm != Mod->RmDisplaceOnly) {
230  if (rm == Mod->RmSib) {
231  // get SIB
232  ibyte = ProbeAndReadUchar(istream);
233  istream++;
234  i = (ibyte >> 3) & 7;
235  if (i != 4) {
236  accum = GETREG(UserFrame, RM32[i]);
237  accum = accum << (ibyte >> 6); // apply scaler
238  }
239  i = ibyte & 7;
240  accum = accum + GETREG(UserFrame, RM32[i]);
241  } else {
242  // get register's value
243  accum = GETREG(UserFrame, pRM[rm]);
244  }
245  }
246 
247  //
248  // apply displacement to accum
249  //
250 
251  if (Mod->RmDisplace & (1 << rm)) {
252  if (Mod->Disp == 4) {
253  i = ProbeAndReadUlong ((PULONG) istream);
254  } else {
255  ibyte = ProbeAndReadChar ((PCHAR)istream);
256  i = (signed long) ((signed char) ibyte); // sign extend
257  }
258  accum += i;
259  }
260 
261  //
262  // if this is an effective address, go get the data value
263  //
264 
265  if (Mod->Disp && accum) {
266  switch (operandsize) {
267  case 1: accum = ProbeAndReadUchar((PUCHAR) accum); break;
268  case 2: accum = ProbeAndReadUshort((PUSHORT) accum); break;
269  case 4: accum = ProbeAndReadUlong((PULONG) accum); break;
270  }
271  }
272 
273  //
274  // accum now contains the instruction operand, see if the
275  // operand was really a zero
276  //
277 
278  if (accum & operandmask) {
279  // operand was non-zero, must be an overflow
280  status = STATUS_INTEGER_OVERFLOW;
281  }
282 
283 try_exit: ;
284  } except (EXCEPTION_EXECUTE_HANDLER) {
285  NOTHING;
286  }
287 
288  if (OldIrql < APC_LEVEL) {
289  KeLowerIrql (OldIrql);
290  }
291 
292  return status;
293 }

328 Ki386CheckDelayedNpxTrap (
329  IN PKTRAP_FRAME UserFrame,
330  IN PFX_SAVE_AREA NpxFrame
331  )
332 
333 /*++
334 
335 Routine Description:
336 
337  This function gains control from the Trap07 handler. It examines
338  the user mode instruction to see if it's a NoWait NPX instruction.
339  Such instructions do not generate floating point exceptions - this
340  check needs to be done due to the way 80386/80387 systems are
341  implemented. Such machines will generate a floating point exception
342  interrupt when the kernel performs an FRSTOR to reload the thread's
343  NPX context. If the thread's next instruction is a NoWait style
344  instruction, then we clear the exception or emulate the instruction.
345 
346  AND... due to a different 80386/80387 "feature" the kernel needs
347  to use FWAIT at times which can causes 80487's to generate delayed
348  exceptions that can lead to the same problem described above.
349 
350  Note: If the CR0_NE = 0 NPX mode is deprecated, this routine may be unnecessary,
351  as the no-wait style instructions should not generate a pending FP exception.
352 
353 Arguments:
354 
355  UserFrame - Trap frame of the exception
356  NpxFrame - Thread's NpxFrame (WARNING: may not have NpxState)
357 
358  Interrupts are disabled.
359 
360 Return Value:
361 
362  FALSE - Dispatch NPX exception to user mode
363  TRUE - Exception handled, continue
364 
365  Note that this function toggles interrupts on/off, possibly affecting the
366  caller's NPX state. Interrupts remain disabled on exit.
367 
368 --*/
369 
370 {
371 
372  EXCEPTION_RECORD ExceptionRecord;
373  UCHAR ibyte1, ibyte2 = 0, inmodrm, status;
374  USHORT StatusWord, ControlWord, UsersWord;
375  PUCHAR istream;
376  BOOLEAN fPrefix;
377  UCHAR rm;
378  PKMOD Mod;
379  ULONG accum, i, NpxState;
380  KIRQL OldIrql;
381 
382  status = 0;
383 
384  //
385  // read instruction prefixes
386  //
387 
388  fPrefix = TRUE;
389  istream = (PUCHAR) UserFrame->Eip;
390 
391  OldIrql = KeGetCurrentIrql();
392  if (OldIrql < APC_LEVEL) {
393  KeRaiseIrql(APC_LEVEL, &OldIrql);
394  }
395 
396  // Enable interrupts since the code will touch the instruction stream. Note that the trap frame
397  // integrity is protected by the elevated IRQL.
398  _asm {
399  sti
400  }
401 
402  try {
403 
404  do {
405  ibyte1 = KiNextIStreamByte (UserFrame, &istream);
406  switch (ibyte1) {
407  case 0x2e: // cs override
408  case 0x36: // ss override
409  case 0x3e: // ds override
410  case 0x26: // es override
411  case 0x64: // fs override
412  case 0x65: // gs override
413  break;
414 
415  default:
416  fPrefix = FALSE;
417  break;
418  }
419  } while (fPrefix);
420 
421  //
422  // Check for coprocessor NoWait NPX instruction
423  //
424 
425  ibyte2 = KiNextIStreamByte (UserFrame, &istream);
426  inmodrm = (ibyte2 >> 3) & 0x7;
427 
428  for (i=0; NoWaitNpxInstructions[i].Opcode1; i++) {
429 
430  if (NoWaitNpxInstructions[i].Opcode1 == ibyte1) {
431 
432  //
433  // first opcode byte matched - check second part of opcode
434  //
435 
436  if (NoWaitNpxInstructions[i].ModRm) {
437 
438  //
439  // modrm only applies for opcode in range 0-0xbf
440  //
441 
442  if (((ibyte2 & 0xc0) != 0xc0) &&
443  (NoWaitNpxInstructions[i].Opcode2 == inmodrm)) {
444 
445  //
446  // This is a no-wait NPX instruction
447  //
448 
449  status = NoWaitNpxInstructions[i].type;
450  break;
451  }
452 
453  } else {
454  if (NoWaitNpxInstructions[i].Opcode2 == ibyte2) {
455 
456  //
457  // This is a no-wait NPX instruction
458  //
459 
460  status = NoWaitNpxInstructions[i].type;
461  break;
462  }
463  }
464  }
465  }
466 
467  } except (EXCEPTION_EXECUTE_HANDLER) {
468  NOTHING;
469  }
470 
471  // Disable interrupts as we are done touching the instruction stream, and may return
472  // to the caller.
473  _asm {
474  cli
475  }
476 
477  if (status == 0) {
478  //
479  // Dispatch coprocessor exception to user mode
480  //
481  if (OldIrql < APC_LEVEL) {
482  KeLowerIrql (OldIrql);
483  }
484  return FALSE;
485  }
486 
487  if (status == 1) {
488  //
489  // Ignore pending exception, user mode instruction does not trap
490  // on pending execptions and it will clear/mask the pending exceptions
491  //
492 
493  // The thread's NPX state may no longer be resident. Avoid executing with
494  // another thread's NPX state.
495  NpxState = KeGetCurrentThread()->NpxState;
496 
497  _asm {
498  mov eax, cr0
499  and eax, NOT (CR0_MP+CR0_EM+CR0_TS)
500  or eax, NpxState
501  mov cr0, eax
502  }
503 
504  NpxFrame->Cr0NpxState &= ~CR0_TS;
505  if (OldIrql < APC_LEVEL) {
506  KeLowerIrql (OldIrql);
507  }
508  return TRUE;
509  }
510 
511  //
512  // This is either FNSTSW or FNSTCW. Both of these instructions get
513  // a value from the coprocessor without effecting the pending exception
514  // state. To do this we emulate the instructions.
515  //
516 
517  //
518  // Read the coprocessors Status & Control word state, then re-enable
519  // interrupts.
520  //
521 
522  //
523  // NOTE: The new compiler is generating a FWAIT at the
524  // entry to the try/except block if it sees inline
525  // fp instructions, even if they are only control word accesses.
526  // put this stuff in another function to fool it.
527  //
528 
529  // Since we toggled the interrupt state, our NPX state might not be resident.
530 
531  if (KeGetCurrentThread()->NpxState == NPX_STATE_NOT_LOADED) {
532  if (KeI386FxsrPresent) {
533  ControlWord = NpxFrame->U.FxArea.ControlWord;
534  StatusWord = NpxFrame->U.FxArea.StatusWord;
535  } else {
536  ControlWord = (USHORT) NpxFrame->U.FnArea.ControlWord;
537  StatusWord = (USHORT) NpxFrame->U.FnArea.StatusWord;
538  }
539  } else {
540  KipWorkAroundCompiler (&StatusWord, &ControlWord);
541  }
542 
543  if (status == 4) {
544  //
545  // Emulate FNSTSW AX
546  //
547 
548  UserFrame->Eip = (ULONG)istream;
549  UserFrame->Eax = (UserFrame->Eax & 0xFFFF0000) | StatusWord;
550  if (OldIrql < APC_LEVEL) {
551  KeLowerIrql (OldIrql);
552  }
553  return TRUE;
554  }
555 
556  // Since the following code examines the instruction stream, re-enable interrupts.
557  _asm {
558  sti
559  }
560 
561  if (status == 2) {
562  UsersWord = ControlWord;
563  } else {
564  UsersWord = StatusWord;
565  }
566 
567  try {
568 
569  //
570  // decode Mod/RM byte
571  //
572 
573  Mod = MOD32 + (ibyte2 >> 6);
574  rm = ibyte2 & 7;
575 
576  //
577  // Decode the instruction's word pointer into accum
578  //
579 
580  accum = 0;
581  if (rm != Mod->RmDisplaceOnly) {
582  if (rm == Mod->RmSib) {
583  // get SIB
584  ibyte1 = KiNextIStreamByte (UserFrame, &istream);
585  i = (ibyte1 >> 3) & 7;
586  if (i != 4) {
587  accum = GETREG(UserFrame, RM32[i]);
588  accum = accum << (ibyte1 >> 6); // apply scaler
589  }
590  i = ibyte1 & 7;
591  accum = accum + GETREG(UserFrame, RM32[i]);
592  } else {
593  // get register's value
594  accum = GETREG(UserFrame, RM32[rm]);
595  }
596  }
597 
598  //
599  // apply displacement to accum
600  //
601 
602  if (Mod->RmDisplace & (1 << rm)) {
603  if (Mod->Disp == 4) {
604  i = (KiNextIStreamByte (UserFrame, &istream) << 0) |
605  (KiNextIStreamByte (UserFrame, &istream) << 8) |
606  (KiNextIStreamByte (UserFrame, &istream) << 16) |
607  (KiNextIStreamByte (UserFrame, &istream) << 24);
608  } else {
609  ibyte1 = KiNextIStreamByte (UserFrame, &istream);
610  i = (signed long) ((signed char) ibyte1); // sign extend
611  }
612  accum += i;
613  }
614 
615  //
616  // Set the word pointer
617  //
618 
619  if (UserFrame->SegCs == KGDT_R0_CODE && (UserFrame->EFlags & EFLAGS_V86_MASK) == 0) {
620  *((PUSHORT) accum) = UsersWord;
621  } else {
622  ProbeAndWriteUshort ((PUSHORT) accum, UsersWord);
623  }
624  UserFrame->Eip = (ULONG)istream;
625 
626  } except (KiCopyInformation(&ExceptionRecord,
627  (GetExceptionInformation())->ExceptionRecord)) {
628 
629  //
630  // Faulted addressing user's memory.
631  // Set the address of the exception to the current program address
632  // and raise the exception by calling the exception dispatcher.
633  //
634  ExceptionRecord.ExceptionAddress = (PVOID)(UserFrame->Eip);
635  if (OldIrql < APC_LEVEL) {
636  KeLowerIrql (OldIrql);
637  }
638  // Note: Interrupts are still enabled along this path.
639  KiDispatchException(
640  &ExceptionRecord,
641  NULL, // ExceptionFrame
642  UserFrame,
643  UserMode,
644  TRUE
645  );
646  }
647 
648  // Lower IRQL and disable interrutps (caller expects them disabled upon return).
649  if (OldIrql < APC_LEVEL) {
650  KeLowerIrql (OldIrql);
651  }
652  _asm {
653  cli
654  }
655 
656  return TRUE;
657 }

678  USHORT cw;
679

685  mov ecx, eax
686  and eax, NOT (CR0_MP+CR0_EM+CR0_TS)

706 
707 Routine Description:
708 
709  This function is called from the APC and DPC interrupt code to check if
710  the specified EIP lies within the SLIST pop code. If the specified EIP
711  lies within the SLIST code, then RIP is reset to the SLIST resume address.
712 
713 Arguments:
714 
715  TrapFrame - Supplies the address of a trap frame.
716 
717 Return Value:
718 
719  None.
720 
721 --*/

724 
725  USHORT SegCs;
726  ULONG Eip;
727 
728  //
729  // Check for kernel mode and user mode execution.
730  //
731 
732  SegCs = (USHORT)TrapFrame->SegCs;
733  Eip = TrapFrame->Eip;
734  if (SegCs == KGDT_R0_CODE) {
735  if ((Eip >= (ULONG)&ExpInterlockedPopEntrySListResume) &&
736  (Eip <= (ULONG)&ExpInterlockedPopEntrySListEnd)) {
737 
738  TrapFrame->Eip = (ULONG)&ExpInterlockedPopEntrySListResume;
739  }
740 
741  } else if (SegCs == (KGDT_R3_CODE | RPL_MASK)) {
742  if ((Eip >= (ULONG)KeUserPopEntrySListResume) &&
743  (Eip <= (ULONG)KeUserPopEntrySListEnd)) {
744 
745  TrapFrame->Eip = (ULONG)KeUserPopEntrySListResume;
746  }
747  }
748

754 KeInvalidAccessAllowed (
755  __in_opt PVOID TrapInformation
756  )
757 
758 /*++
759 
760 Routine Description:
761 
762  Mm will pass a pointer to a trap frame prior to issuing a bugcheck on
763  a pagefault. This routine lets Mm know if it is ok to bugcheck. The
764  specific case we must protect are the interlocked pop sequences which can
765  blindly access memory that may have been freed and/or reused prior to the
766  access. We don't want to bugcheck the system in these cases, so we check
767  the instruction pointer here.
768 
769  For a usermode fault, Mm uses this routine for similar reasons, to determine
770  whether a guard page fault should be ignored.
771 
772 Arguments:
773 
774  TrapInformation - Supplies a trap frame pointer. NULL means return False.
775 
776 Return Value:
777 
778  True if the invalid access should be ignored, False otherwise.
779 
780 --*/
781 
782 {
783 
784  ULONG slistFaultIP;
785  PKTRAP_FRAME trapFrame;
786 
787  if (ARGUMENT_PRESENT(TrapInformation) == FALSE) {
788  return FALSE;
789  }
790 
791  trapFrame = TrapInformation;
792  switch (trapFrame->SegCs) {
793 
794  case KGDT_R0_CODE:
795 
796  //
797  // Fault occured in kernel mode
798  //
799 
800  slistFaultIP = (ULONG)&ExpInterlockedPopEntrySListFault;
801  break;
802 
803  case KGDT_R3_CODE | RPL_MASK:
804 
805  //
806  // Fault occured in native usermode
807  //
808 
809  slistFaultIP = (ULONG)KeUserPopEntrySListFault;
810  break;
811 
812  default:
813  return FALSE;
814  }
815 
816  if (trapFrame->Eip == slistFaultIP) {
817  return TRUE;
818 
819  } else {
820  return FALSE;
