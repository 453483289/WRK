365 #define HvpComputeIndex(Index, Size) \
366  { \
367  Index = (Size >> HHIVE_FREE_DISPLAY_SHIFT) - 1; \
368  if (Index >= HHIVE_LINEAR_INDEX ) { \
369  \
370  /* \
371  ** Too big for the linear lists, compute the exponential \
372  ** list. Shift the index to make sure we cover the whole \
373  ** range. \
374  */ \
375  Index >>= 4; \
376  \
377  if (Index > 255) { \
378  /* \
379  ** Too big for all the lists, use the last index. \
380  */ \
381  Index = HHIVE_FREE_DISPLAY_SIZE-1; \
382  } else { \
383  Index = CmpFindFirstSetLeft[Index] + \
384  HHIVE_LINEAR_INDEX; \
385  } \
386  } \
387  }
