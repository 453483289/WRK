467 //
468 // ACCESS_MASK
469 // SeComputeDeniedAccesses(
470 // IN ACCESS_MASK GrantedAccess,
471 // IN ACCESS_MASK DesiredAccess
472 // );
473 //
474 // Routine Description:
475 //
476 // This routine generates an access mask containing those accesses
477 // requested by DesiredAccess that aren't granted by GrantedAccess.
478 // The result of this routine may be compared to 0 to determine
479 // if a DesiredAccess mask contains any accesses that have not
480 // been granted.
481 //
482 // If the result IS ZERO, then all desired accesses have been granted.
483 //
484 // Arguments:
485 //
486 // GrantedAccess - Specifies the granted access mask.
487 //
488 // DesiredAccess - Specifies the desired access mask.
489 //
490 // Return Value:
491 //
492 // An ACCESS_MASK containing the desired accesses that have
493 // not been granted.
494 //
495 //--
496 
497 #define SeComputeDeniedAccesses( GrantedAccess, DesiredAccess ) \
498  ((~(GrantedAccess)) & (DesiredAccess) )
499

503 // BOOLEAN
504 // SeComputeGrantedAccesses(
505 // IN ACCESS_MASK GrantedAccess,
506 // IN ACCESS_MASK DesiredAccess
507 // );
508 //
509 // Routine Description:
510 //
511 // This routine generates an access mask containing accesses
512 // requested by DesiredAccess that are granted by GrantedAccess.
513 // The result of this routine may be compared to 0 to determine
514 // if any desired accesses have been granted.
515 //
516 // If the result IS NON-ZERO, then at least one desired accesses
517 // has been granted.
518 //
519 // Arguments:
520 //
521 // GrantedAccess - Specifies the granted access mask.
522 //
523 // DesiredAccess - Specifies the desired access mask.
524 //
525 // Return Value:
526 //
527 // This routine returns TRUE if the DesiredAccess mask does specifies
528 // any bits that are set in the GrantedAccess mask.
529 //
530 //--

533  ((GrantedAccess) & (DesiredAccess) )
