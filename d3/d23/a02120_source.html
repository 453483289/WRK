383 NtReleaseKeyedEvent (
384  __in HANDLE KeyedEventHandle,
385  __in PVOID KeyValue,
386  __in BOOLEAN Alertable,
387  __in_opt PLARGE_INTEGER Timeout
388  )
389 
390 /*++
391 
392 Routine Description:
393 
394  Release a previous or soon to be waiter with a matching key
395 
396 Arguments:
397 
398  KeyedEventHandle - Handle to a keyed event
399 
400  KeyValue - Value to be used to match the waiter against
401 
402  Alertable - Should the wait be alertable, we rarely should have to wait
403 
404  Timeout - Timeout value for the wait, waits should be rare
405 
406 Return Value:
407 
408  NTSTATUS - Status of call
409 
410 --*/
411 
412 {
413  NTSTATUS Status;
414  KPROCESSOR_MODE PreviousMode;
415  PKEYED_EVENT_OBJECT KeyedEventObject;
416  PETHREAD CurrentThread, TargetThread;
417  PEPROCESS CurrentProcess;
418  PLIST_ENTRY ListHead, ListEntry;
419  LARGE_INTEGER TimeoutValue;
420  PVOID OldKeyValue = NULL;
421 
422  if ((((ULONG_PTR)KeyValue) & KEYVALUE_RELEASE) != 0) {
423  return STATUS_INVALID_PARAMETER_1;
424  }
425 
426  CurrentThread = PsGetCurrentThread ();
427  PreviousMode = KeGetPreviousModeByThread (&CurrentThread->Tcb);
428 
429  if (Timeout != NULL) {
430  try {
431  if (PreviousMode != KernelMode) {
432  ProbeForRead (Timeout, sizeof (*Timeout), sizeof (UCHAR));
433  }
434  TimeoutValue = *Timeout;
435  Timeout = &TimeoutValue;
436  } except(ExSystemExceptionFilter ()) {
437  return GetExceptionCode ();
438  }
439  }
440 
441  Status = ObReferenceObjectByHandle (KeyedEventHandle,
442  KEYEDEVENT_WAKE,
443  ExpKeyedEventObjectType,
444  PreviousMode,
445  &KeyedEventObject,
446  NULL);
447 
448  if (!NT_SUCCESS (Status)) {
449  return Status;
450  }
451 
452  ASSERT (CurrentThread->KeyedEventInUse == 0);
453  ASSERT (KeGetCurrentIrql () == PASSIVE_LEVEL);
454  CurrentThread->KeyedEventInUse = 1;
455 
456  CurrentProcess = PsGetCurrentProcessByThread (CurrentThread);
457 
458  ListHead = &KeyedEventObject->WaitQueue;
459 
460  LOCK_KEYED_EVENT_EXCLUSIVE (KeyedEventObject, CurrentThread);
461 
462  ListEntry = ListHead->Flink;
463  while (1) {
464  if (ListEntry == ListHead) {
465  //
466  // We could not find a key matching ours in the list.
467  // Either somebody called us with wrong values or the waiter
468  // has not managed to get queued yet. We wait ourselves
469  // to be released by the waiter.
470  //
471  OldKeyValue = CurrentThread->KeyedWaitValue;
472  CurrentThread->KeyedWaitValue = (PVOID) (((ULONG_PTR)KeyValue)|KEYVALUE_RELEASE);
473  //
474  // Insert the thread at the head of the list. We establish an invariant
475  // were release waiters are always at the front of the queue to improve
476  // the wait code since it only has to search as far as the first non-release
477  // waiter.
478  //
479  InsertHeadList (ListHead, &CurrentThread->KeyedWaitChain);
480  TargetThread = NULL;
481  break;
482  } else {
483  TargetThread = CONTAINING_RECORD (ListEntry, ETHREAD, KeyedWaitChain);
484  if (TargetThread->KeyedWaitValue == KeyValue &&
485  THREAD_TO_PROCESS (TargetThread) == CurrentProcess) {
486  RemoveEntryList (ListEntry);
487  InitializeListHead (ListEntry);
488  break;
489  }
490  }
491  ListEntry = ListEntry->Flink;
492  }
493 
494  //
495  // Release the lock but leave APC's disabled.
496  // This prevents us from being suspended and holding up the target.
497  //
498  UNLOCK_KEYED_EVENT_EXCLUSIVE_UNSAFE (KeyedEventObject);
499 
500  if (TargetThread != NULL) {
501  KeReleaseSemaphore (&TargetThread->KeyedWaitSemaphore,
502  SEMAPHORE_INCREMENT,
503  1,
504  FALSE);
505  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
506  } else {
507  KeLeaveCriticalRegionThread (&CurrentThread->Tcb);
508  Status = KeWaitForSingleObject (&CurrentThread->KeyedWaitSemaphore,
509  Executive,
510  PreviousMode,
511  Alertable,
512  Timeout);
513 
514  //
515  // If we were woken by termination then we must manually remove
516  // ourselves from the queue
517  //
518  if (Status != STATUS_SUCCESS) {
519  BOOLEAN Wait = TRUE;
520 
521  LOCK_KEYED_EVENT_EXCLUSIVE (KeyedEventObject, CurrentThread);
522  if (!IsListEmpty (&CurrentThread->KeyedWaitChain)) {
523  RemoveEntryList (&CurrentThread->KeyedWaitChain);
524  InitializeListHead (&CurrentThread->KeyedWaitChain);
525  Wait = FALSE;
526  }
527  UNLOCK_KEYED_EVENT_EXCLUSIVE (KeyedEventObject, CurrentThread);
528  //
529  // If this thread was no longer in the queue then another thread
530  // must be about to wake us up. Wait for that wake.
531  //
532  if (Wait) {
533  KeWaitForSingleObject (&CurrentThread->KeyedWaitSemaphore,
534  Executive,
535  KernelMode,
536  FALSE,
537  NULL);
538  }
539  }
540  CurrentThread->KeyedWaitValue = OldKeyValue;
541  }
542 
543  ASSERT (KeGetCurrentIrql () == PASSIVE_LEVEL);
544  CurrentThread->KeyedEventInUse = 0;
545 
546  ObDereferenceObject (KeyedEventObject);
547 
548  return Status;
549 }
