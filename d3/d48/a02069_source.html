205 NTSTATUS
206 CmpSecurityMethod (
207  IN PVOID Object,
208  IN SECURITY_OPERATION_CODE OperationCode,
209  IN PSECURITY_INFORMATION SecurityInformation,
210  IN OUT PSECURITY_DESCRIPTOR SecurityDescriptor,
211  IN OUT PULONG CapturedLength,
212  IN OUT PSECURITY_DESCRIPTOR *ObjectsSecurityDescriptor,
213  IN POOL_TYPE PoolType,
214  IN PGENERIC_MAPPING GenericMapping
215  )
216 
217 /*++
218 
219 Routine Description:
220 
221  This is the security method for registry objects. It is responsible for
222  retrieving, setting, and deleting the security descriptor of a registry
223  object. It is not used to assign the original security descriptor to an
224  object (use SeAssignSecurity for that purpose).
225 
226 
227  IT IS ASSUMED THAT THE OBJECT MANAGER HAS ALREADY DONE THE ACCESS
228  VALIDATIONS NECESSARY TO ALLOW THE REQUESTED OPERATIONS TO BE PERFORMED.
229 
230 Arguments:
231 
232  Object - Supplies a pointer to the object being used.
233 
234  OperationCode - Indicates if the operation is for setting, querying, or
235  deleting the object's security descriptor.
236 
237  SecurityInformation - Indicates which security information is being
238  queried or set. This argument is ignored for the delete operation.
239 
240  SecurityDescriptor - The meaning of this parameter depends on the
241  OperationCode:
242 
243  QuerySecurityDescriptor - For the query operation this supplies the
244  buffer to copy the descriptor into. The security descriptor is
245  assumed to have been probed up to the size passed in in Length.
246  Since it still points into user space, it must always be
247  accessed in a try clause in case it should suddenly disappear.
248 
249  SetSecurityDescriptor - For a set operation this supplies the
250  security descriptor to copy into the object. The security
251  descriptor must be captured before this routine is called.
252 
253  DeleteSecurityDescriptor - It is ignored when deleting a security
254  descriptor.
255 
256  AssignSecurityDescriptor - For assign operations this is the
257  security descriptor that will be assigned to the object.
258  It is assumed to be in kernel space, and is therefore not
259  probed or captured.
260 
261  CapturedLength - For the query operation this specifies the length, in
262  bytes, of the security descriptor buffer, and upon return contains
263  the number of bytes needed to store the descriptor. If the length
264  needed is greater than the length supplied the operation will fail.
265  It is ignored in the set and delete operation.
266 
267  This parameter is assumed to be captured and probed as appropriate.
268 
269  ObjectsSecurityDescriptor - For the Set operation this supplies the address
270  of a pointer to the object's current security descriptor. This routine
271  will either modify the security descriptor in place or deallocate/
272  allocate a new security descriptor and use this variable to indicate
273  its new location. For the query operation it simply supplies
274  the security descriptor being queried.
275 
276  PoolType - For the set operation this specifies the pool type to use if
277  a new security descriptor needs to be allocated. It is ignored
278  in the query and delete operation.
279 
280  GenericMapping - Passed only for the set operation, this argument provides
281  the mapping of generic to specific/standard access types for the object
282  being accessed. This mapping structure is expected to be safe to
283  access (i.e., captured if necessary) prior to be passed to this routine.
284 
285 Return Value:
286 
287  NTSTATUS - STATUS_SUCCESS if the operation is successful and an
288  appropriate error status otherwise.
289 
290 --*/
291 
292 {
293  PCM_KEY_CONTROL_BLOCK kcb;
294  NTSTATUS Status = STATUS_UNSUCCESSFUL;
295  BOOLEAN UnlockKcb = TRUE;
296  BOOLEAN UnlockSecurity = FALSE;
297 
298  //
299  // Make sure the common parts of our input are proper
300  //
301 
302  CM_PAGED_CODE();
303  ASSERT_KEY_OBJECT(Object);
304 
305  ASSERT( (OperationCode == SetSecurityDescriptor) ||
306  (OperationCode == QuerySecurityDescriptor) ||
307  (OperationCode == AssignSecurityDescriptor) ||
308  (OperationCode == DeleteSecurityDescriptor) );
309 
310  kcb = ((PCM_KEY_BODY)Object)->KeyControlBlock;
311  //
312  // Lock hive for shared or exclusive, depending on what we need
313  // to do.
314  //
315  CmpLockRegistry();
316  if (OperationCode == QuerySecurityDescriptor) {
317  //
318  // trick to avoid recursive acquires
319  //
320  if( (ULONG_PTR)kcb & 1 ) {
321  kcb = (PCM_KEY_CONTROL_BLOCK)((ULONG_PTR)kcb ^ 1);
322  ASSERT_KCB_LOCKED(kcb);
323  UnlockKcb = FALSE;
324  } else {
325  //
326  // serialize access to this key.
327  //
328  CmpLockKCBShared(kcb);
329  }
330  } else {
331  //
332  // serialize access to this key.
333  //
334  ASSERT( ((ULONG_PTR)kcb & 1) == 0 );
335  CmpLockKCBExclusive(kcb);
336  }
337 
338  if(kcb->Delete) {
339  //
340  // Key has been deleted, performing security operations on
341  // it is Not Allowed.
342  //
343  if( UnlockKcb ) {
344  CmpUnlockKCB(kcb);
345  }
346  CmpUnlockRegistry();
347  return(STATUS_KEY_DELETED);
348  }
349 
350  if (OperationCode != QuerySecurityDescriptor) {
351  //
352  // no flush from this point on
353  //
354  CmpLockHiveFlusherShared((PCMHIVE)kcb->KeyHive);
355  //
356  // we will be changing the security for this hive.
357  //
358  CmLockHiveSecurityExclusive((PCMHIVE)kcb->KeyHive);
359  UnlockSecurity = TRUE;
360  }
361 
362  try {
363 
364  //
365  // This routine simply cases off of the operation code to decide
366  // which support routine to call
367  //
368 
369  switch (OperationCode) {
370 
371  case SetSecurityDescriptor:
372 
373  //
374  // check the rest of our input and call the set security
375  // method
376  //
377  ASSERT( (PoolType == PagedPool) || (PoolType == NonPagedPool) );
378 
379  ASSERT_KCB_LOCKED(kcb);
380  Status = CmpSetSecurityDescriptorInfo( kcb,
381  SecurityInformation,
382  SecurityDescriptor,
383  ObjectsSecurityDescriptor,
384  PoolType,
385  GenericMapping );
386 
387  //
388  // this is the one and only path on which a user could change
389  // a security descriptor, therefore, report such changes for
390  // notification here.
391  //
392  if (NT_SUCCESS(Status)) {
393  ASSERT( UnlockSecurity );
394  CmUnlockHiveSecurity((PCMHIVE)kcb->KeyHive);
395  UnlockSecurity = FALSE;
396 
397  CmpReportNotify(kcb,
398  kcb->KeyHive,
399  kcb->KeyCell,
400  REG_NOTIFY_CHANGE_ATTRIBUTES | REG_NOTIFY_CHANGE_SECURITY);
401 
402  }
403 
404  break;
405 
406  case QuerySecurityDescriptor:
407 
408  //
409  // check the rest of our input and call the default query security
410  // method
411  //
412  ASSERT( CapturedLength != NULL );
413  Status = CmpQuerySecurityDescriptorInfo( kcb,
414  SecurityInformation,
415  SecurityDescriptor,
416  CapturedLength,
417  ObjectsSecurityDescriptor );
418  break;
419 
420  case DeleteSecurityDescriptor:
421 
422  //
423  // Nobody should ever call the delete method. When the key is
424  // freed, the security descriptor associated with it is
425  // explicitly freed (CmpFreeSecurityDescriptor)
426  //
427  ASSERT(FALSE);
428 
429  break;
430 
431  case AssignSecurityDescriptor:
432 
433  //
434  // Set the SecurityDescriptor field in the object's header to
435  // NULL. This indicates that our security method needs to be
436  // called for any security descriptor operations.
437  //
438  Status = CmpAssignSecurityDescriptorWrapper(Object,SecurityDescriptor);
439  break;
440 
441  default:
442 
443  //
444  // Bugcheck on any other operation code, We won't get here if
445  // the earlier asserts are still checked.
446  //
447  CM_BUGCHECK( REGISTRY_ERROR,BAD_SECURITY_METHOD,1,kcb,OperationCode);
448 
449  }
450 
451  } except (CmpSecurityExceptionFilter(GetExceptionInformation())) {
452  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_EXCEPTION,"!!CmpSecurityMethod: code:%08lx\n", GetExceptionCode()));
453  Status = GetExceptionCode();
454  }
455 
456  if (OperationCode != QuerySecurityDescriptor) {
457  CmpUnlockHiveFlusher((PCMHIVE)kcb->KeyHive);
458  if( UnlockSecurity ) {
459  CmUnlockHiveSecurity((PCMHIVE)kcb->KeyHive);
460  }
461  }
462  if( UnlockKcb ) {
463  CmpUnlockKCB(kcb);
464  }
465  CmpUnlockRegistry();
466  return(Status);
467 
468 }

470 NTSTATUS
471 CmpSetSecurityDescriptorInfo(
472  IN PCM_KEY_CONTROL_BLOCK Key,
473  IN PSECURITY_INFORMATION SecurityInformation,
474  IN PSECURITY_DESCRIPTOR ModificationDescriptor,
475  IN OUT PSECURITY_DESCRIPTOR *ObjectsSecurityDescriptor,
476  IN POOL_TYPE PoolType,
477  IN PGENERIC_MAPPING GenericMapping
478  )
479 /*++
480 
481 Routine Description:
482 
483  This routine will set a node's security descriptor. The input
484  security descriptor must be previously captured.
485 
486 Arguments:
487 
488  Key - Supplies a pointer to the KEY_CONTROL_BLOCK for the node whose
489  security descriptor will be set.
490 
491  SecurityInformation - Indicates which security information is
492  to be applied to the object. The value(s) to be assigned are
493  passed in the SecurityDescriptor parameter.
494 
495  ModificationDescriptor - Supplies the input security descriptor to be
496  applied to the object. The caller of this routine is expected
497  to probe and capture the passed security descriptor before calling
498  and release it after calling.
499 
500  ObjectsSecurityDescriptor - Supplies the address of a pointer to
501  the objects security descriptor that is going to be altered by
502  this procedure
503 
504  PoolType - Specifies the type of pool to allocate for the objects
505  security descriptor.
506 
507  GenericMapping - This argument provides the mapping of generic to
508  specific/standard access types for the object being accessed.
509  This mapping structure is expected to be safe to access
510  (i.e., captured if necessary) prior to be passed to this routine.
511 
512 Return Value:
513 
514  NTSTATUS - STATUS_SUCCESS if successful and an appropriate error
515  value otherwise
516 
517 --*/
518 
519 {
520  NTSTATUS Status;
521  HCELL_INDEX SecurityCell;
522  HCELL_INDEX MatchSecurityCell;
523  HCELL_INDEX NewCell;
524  HCELL_INDEX OldCell;
525  PCM_KEY_SECURITY Security;
526  PCM_KEY_SECURITY NewSecurity;
527  PCM_KEY_SECURITY FlinkSecurity;
528  PCM_KEY_SECURITY BlinkSecurity;
529  PCM_KEY_NODE Node;
530  ULONG DescriptorLength;
531  PSECURITY_DESCRIPTOR DescriptorCopy;
532  ULONG Type;
533  LARGE_INTEGER SystemTime;
534  PHHIVE Hive;
535  PCM_KEY_SECURITY_CACHE CachedSecurity;
536  HV_TRACK_CELL_REF CellRef = {0};
537 
538  CM_PAGED_CODE();
539 
540  UNREFERENCED_PARAMETER (ObjectsSecurityDescriptor);
541 
542  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_SEC,"CmpSetSecurityDescriptorInfo:\n"));
543 
544  ASSERT_KCB_LOCKED_EXCLUSIVE(Key);
545  ASSERT_HIVE_SECURITY_LOCK_OWNED((PCMHIVE)Key->KeyHive);
546 
547  Node = (PCM_KEY_NODE)HvGetCell(Key->KeyHive, Key->KeyCell);
548  if( Node == NULL ) {
549  //
550  // we couldn't map the bin containing this cell;
551  // this shouldn't happen as we are about to modify the cell
552  // (i.e. it should be dirty/pinned by this time)
553  //
554  return STATUS_INSUFFICIENT_RESOURCES;
555  }
556 
557  if( !HvTrackCellRef(&CellRef,Key->KeyHive, Key->KeyCell) ) {
558  return STATUS_INSUFFICIENT_RESOURCES;
559  }
560 
561  //
562  // Map in the hive cell for the security descriptor before we make
563  // the call to SeSetSecurityDescriptorInfo. This prevents us from
564  // changing its security descriptor and then being unable to bring
565  // the hive cell into memory for updating.
566  //
567  Security = CmpGetKeySecurity(Key->KeyHive,
568  Node,
569  &SecurityCell);
570  if( Security == NULL ) {
571  //
572  // couldn't map view inside
573  //
574  Status = STATUS_INSUFFICIENT_RESOURCES;
575  goto ErrorExit;
576  }
577 
578  if( !HvTrackCellRef(&CellRef,Key->KeyHive, SecurityCell) ) {
579  Status = STATUS_INSUFFICIENT_RESOURCES;
580  goto ErrorExit;
581  }
582 
583  //
584  // SeSetSecurityDescriptorInfo takes a pointer to the original
585  // descriptor. This pointer is not freed, but a new pointer will
586  // be returned.
587  //
588  DescriptorCopy = &Security->Descriptor;
589  Status = SeSetSecurityDescriptorInfo( NULL,
590  SecurityInformation,
591  ModificationDescriptor,
592  &DescriptorCopy,
593  PoolType,
594  GenericMapping );
595 
596  if (!NT_SUCCESS(Status)) {
597  goto ErrorExit;
598  }
599 
600  //
601  // Set Security operation succeeded, so we update the security
602  // descriptor in the hive.
603  //
604  DescriptorLength = RtlLengthSecurityDescriptor(DescriptorCopy);
605  Type = HvGetCellType(Key->KeyCell);
606  Hive = Key->KeyHive;
607 
608  if (! (HvMarkCellDirty(Hive, Key->KeyCell,FALSE) && HvMarkCellDirty(Hive, SecurityCell,FALSE)) ) {
609  ExFreePool(DescriptorCopy);
610  Status = STATUS_NO_LOG_SPACE;
611  goto ErrorExit;
612  }
613 
614  //
615  // Try to find an existing security descriptor that we can share.
616  //
617  if (CmpFindMatchingDescriptorCell((PCMHIVE)Hive, DescriptorCopy, Type, &MatchSecurityCell,&CachedSecurity)) {
618  //
619  // A match was found.
620  //
621  if( MatchSecurityCell == SecurityCell ) {
622  //
623  // what we want to set is already here, so bail out
624  //
625  ExFreePool(DescriptorCopy);
626 
627  //
628  // Update the LastWriteTime of the key.
629  //
630  KeQuerySystemTime(&SystemTime);
631  Node->LastWriteTime = SystemTime;
632  // update the time in kcb too, to keep the cache in sync
633  Key->KcbLastWriteTime = SystemTime;
634 
635  HvReleaseFreeCellRefArray(&CellRef);
636  return STATUS_SUCCESS;
637  } else {
638  if (!HvMarkCellDirty(Hive, MatchSecurityCell,FALSE)) {
639  ExFreePool(DescriptorCopy);
640  Status = STATUS_NO_LOG_SPACE;
641  goto ErrorExit;
642  }
643  if (Security->ReferenceCount == 1) {
644  //
645  // No more references to the old security cell, so we can free it now.
646  //
647  if (! (HvMarkCellDirty(Hive, Security->Flink,FALSE) &&
648  HvMarkCellDirty(Hive, Security->Blink,FALSE))) {
649  ExFreePool(DescriptorCopy);
650  Status = STATUS_NO_LOG_SPACE;
651  goto ErrorExit;
652  }
653  CmpRemoveSecurityCellList(Hive, SecurityCell);
654  HvFreeCell(Hive, SecurityCell);
655  } else {
656 
657  //
658  // Just decrement the count on the old security cell
659  //
660  Security->ReferenceCount -= 1;
661  }
662 
663  //
664  // Set the node to point at the matching security cell.
665  //
666  Security = (PCM_KEY_SECURITY)HvGetCell(Hive, MatchSecurityCell);
667  if( Security == NULL ) {
668  //
669  // we couldn't map the bin containing this cell
670  // this should not happen as we just marked the cell dirty
671  //
672  ASSERT( FALSE );
673  ExFreePool(DescriptorCopy);
674  Status = STATUS_INSUFFICIENT_RESOURCES;
675  goto ErrorExit;
676  }
677 
678  if( !HvTrackCellRef(&CellRef,Hive, MatchSecurityCell) ) {
679  Status = STATUS_INSUFFICIENT_RESOURCES;
680  goto ErrorExit;
681  }
682 
683  Security->ReferenceCount += 1;
684  Node->Security = MatchSecurityCell;
685  }
686  } else {
687 
688  //
689  // No match was found, we need to create a new cell.
690  //
691  if (Security->ReferenceCount > 1) {
692 
693  //
694  // We can't change the existing security cell, since it is shared
695  // by multiple keys. Allocate a new cell and decrement the existing
696  // one's reference count.
697  //
698  NewCell = HvAllocateCell(Key->KeyHive,
699  SECURITY_CELL_LENGTH(DescriptorCopy),
700  Type,
701  HCELL_NIL);
702  if (NewCell == HCELL_NIL) {
703  ExFreePool(DescriptorCopy);
704  Status = STATUS_INSUFFICIENT_RESOURCES;
705  goto ErrorExit;
706  }
707 
708  if (! HvMarkCellDirty(Key->KeyHive, Security->Flink,FALSE)) {
709  ExFreePool(DescriptorCopy);
710  Status = STATUS_NO_LOG_SPACE;
711  goto ErrorExit;
712  }
713 
714  Security->ReferenceCount -= 1;
715 
716  //
717  // Map in the new cell and insert it into the linked list.
718  //
719  NewSecurity = (PCM_KEY_SECURITY) HvGetCell(Key->KeyHive, NewCell);
720  if( NewSecurity == NULL ) {
721  //
722  // we couldn't map the bin containing this cell
723  //
724  ExFreePool(DescriptorCopy);
725  Status = STATUS_INSUFFICIENT_RESOURCES;
726  goto ErrorExit;
727  }
728 
729  if( !HvTrackCellRef(&CellRef,Key->KeyHive, NewCell) ) {
730  Status = STATUS_INSUFFICIENT_RESOURCES;
731  goto ErrorExit;
732  }
733 
734  NewSecurity->Blink = SecurityCell;
735  NewSecurity->Flink = Security->Flink;
736  FlinkSecurity = (PCM_KEY_SECURITY) HvGetCell(Key->KeyHive, Security->Flink);
737  if( FlinkSecurity == NULL ) {
738  //
739  // we couldn't map the bin containing this cell
740  //
741  ExFreePool(DescriptorCopy);
742  Status = STATUS_INSUFFICIENT_RESOURCES;
743  goto ErrorExit;
744  }
745 
746  if( !HvTrackCellRef(&CellRef,Key->KeyHive, Security->Flink) ) {
747  Status = STATUS_INSUFFICIENT_RESOURCES;
748  goto ErrorExit;
749  }
750 
751  Security->Flink = FlinkSecurity->Blink = NewCell;
752 
753  //
754  // initialize new cell
755  //
756  NewSecurity->Signature = CM_KEY_SECURITY_SIGNATURE;
757  NewSecurity->ReferenceCount = 1;
758  NewSecurity->DescriptorLength = DescriptorLength;
759  Security=NewSecurity;
760 
761  //
762  // copy the descriptor
763  //
764  RtlCopyMemory( &(Security->Descriptor),
765  DescriptorCopy,
766  DescriptorLength );
767 
768  //
769  // Add the new created security cell to the cache
770  //
771  if( !NT_SUCCESS(CmpAddSecurityCellToCache( (PCMHIVE)Key->KeyHive,NewCell,FALSE,NULL)) ) {
772  //
773  // we couldn't map the bin containing this cell
774  // this shouldn't happen as we just allocated (marked dirty) the cell
