61 /*++
62 
63 Routine description:
64 
65  This routine is used to query information about a given object
66 
67 Arguments:
68 
69  Handle - Supplies a handle to the object being queried. This value
70  is ignored if the requested information class is for type
71  information.
72 
73  ObjectInformationClass - Specifies the type of information to return
74 
75  ObjectInformation - Supplies an output buffer for the information being
76  returned
77 
78  ObjectInformationLength - Specifies, in bytes, the length of the
79  preceding object information buffer
80 
81  ReturnLength - Optionally receives the length, in bytes, used to store
82  the object information
83 
84 Return Value:
85 
86  An appropriate status value
87 
88 --*/
89

91  KPROCESSOR_MODE PreviousMode;
92  NTSTATUS Status;
93  PVOID Object;
94  POBJECT_HEADER ObjectHeader;
95  POBJECT_HEADER_QUOTA_INFO QuotaInfo;
96  POBJECT_HEADER_NAME_INFO NameInfo;
97  POBJECT_TYPE ObjectType;
98  POBJECT_HEADER ObjectDirectoryHeader;
99  POBJECT_DIRECTORY ObjectDirectory;
100  ACCESS_MASK GrantedAccess;
101  POBJECT_HANDLE_FLAG_INFORMATION HandleFlags;
102  OBJECT_HANDLE_INFORMATION HandleInformation = {0};
103  ULONG NameInfoSize;
104  ULONG SecurityDescriptorSize;
105  ULONG TempReturnLength;
106  OBJECT_BASIC_INFORMATION ObjectBasicInfo;
107  POBJECT_TYPES_INFORMATION TypesInformation;
108  POBJECT_TYPE_INFORMATION TypeInfo;
109  ULONG i;
110

112 
113  //
114  // Initialize our local variables
115  //
116 
117  TempReturnLength = 0;
118 
119  //
120  // Get previous processor mode and probe output argument if necessary.
121  //
122 
123  PreviousMode = KeGetPreviousMode();
124 
125  if (PreviousMode != KernelMode) {
126 
127  try {
128 
129  if (ObjectInformationClass != ObjectHandleFlagInformation) {
130 
131  ProbeForWrite( ObjectInformation,
132  ObjectInformationLength,
133  sizeof( ULONG ));
134 
135  } else {
136 
137  ProbeForWrite( ObjectInformation,
138  ObjectInformationLength,
139  1 );
140  }
141 
142  //
143  // We'll use a local temp return length variable to pass
144  // through to the later ob query calls which will increment
145  // its value. We can't pass the users return length directly
146  // because the user might also be altering its value behind
147  // our back.
148  //
149 
150  if (ARGUMENT_PRESENT( ReturnLength )) {
151 
152  ProbeForWriteUlong( ReturnLength );
153  }
154 
155  } except( EXCEPTION_EXECUTE_HANDLER ) {
156 
157  return( GetExceptionCode() );
158  }
159  }
160 
161  //
162  // If the query is not for types information then we
163  // will have to get the object in question. Otherwise
164  // for types information there really isn't an object
165  // to grab.
166  //
167 
168  if (ObjectInformationClass != ObjectTypesInformation) {
169 
170  Status = ObReferenceObjectByHandle( Handle,
171  0,
172  NULL,
173  PreviousMode,
174  &Object,
175  &HandleInformation );
176 
177  if (!NT_SUCCESS( Status )) {
178 
179  return( Status );
180  }
181 
182  GrantedAccess = HandleInformation.GrantedAccess;
183 
184  ObjectHeader = OBJECT_TO_OBJECT_HEADER( Object );
185  ObjectType = ObjectHeader->Type;
186 
187  } else {
188 
189  GrantedAccess = 0;
190  Object = NULL;
191  ObjectHeader = NULL;
192  ObjectType = NULL;
193  Status = STATUS_SUCCESS;
194  }
195 
196  //
197  // Now process the particular information class being
198  // requested
199  //
200 
201  switch( ObjectInformationClass ) {
202 
203  case ObjectBasicInformation:
204 
205  //
206  // Make sure the output buffer is long enough and then
207  // fill in the appropriate fields into our local copy
208  // of basic information.
209  //
210 
211  if (ObjectInformationLength != sizeof( OBJECT_BASIC_INFORMATION )) {
212 
213  ObDereferenceObject( Object );
214 
215  return( STATUS_INFO_LENGTH_MISMATCH );
216  }
217 
218  ObjectBasicInfo.Attributes = HandleInformation.HandleAttributes;
219 
220  if (ObjectHeader->Flags & OB_FLAG_PERMANENT_OBJECT) {
221 
222  ObjectBasicInfo.Attributes |= OBJ_PERMANENT;
223  }
224 
225  if (ObjectHeader->Flags & OB_FLAG_EXCLUSIVE_OBJECT) {
226 
227  ObjectBasicInfo.Attributes |= OBJ_EXCLUSIVE;
228  }
229 
230  ObjectBasicInfo.GrantedAccess = GrantedAccess;
231  ObjectBasicInfo.HandleCount = (ULONG)ObjectHeader->HandleCount;
232  ObjectBasicInfo.PointerCount = (ULONG)ObjectHeader->PointerCount;
233 
234  QuotaInfo = OBJECT_HEADER_TO_QUOTA_INFO( ObjectHeader );
235 
236  if (QuotaInfo != NULL) {
237 
238  ObjectBasicInfo.PagedPoolCharge = QuotaInfo->PagedPoolCharge;
239  ObjectBasicInfo.NonPagedPoolCharge = QuotaInfo->NonPagedPoolCharge;
240 
241  } else {
242 
243  ObjectBasicInfo.PagedPoolCharge = 0;
244  ObjectBasicInfo.NonPagedPoolCharge = 0;
245  }
246 
247  if (ObjectType == ObpSymbolicLinkObjectType) {
248 
249  ObjectBasicInfo.CreationTime = ((POBJECT_SYMBOLIC_LINK)Object)->CreationTime;
250 
251  } else {
252 
253  RtlZeroMemory( &ObjectBasicInfo.CreationTime,
254  sizeof( ObjectBasicInfo.CreationTime ));
255  }
256 
257  //
258  // Compute the size of the object name string by taking its name plus
259  // separators and traversing up to the root adding each directories
260  // name length plus separators
261  //
262 
263  NameInfo = ObpReferenceNameInfo( ObjectHeader );
264 
265  if ((NameInfo != NULL) && (NameInfo->Directory != NULL)) {
266 
267  PVOID ReferencedDirectory = NULL;
268 
269  //
270  // We grab the root directory lock and test again the directory
271  //
272 
273  ObjectDirectory = NameInfo->Directory;
274 
275  ASSERT (ObjectDirectory);
276 
277  ObfReferenceObject( ObjectDirectory );
278  ReferencedDirectory = ObjectDirectory;
279 
280  NameInfoSize = sizeof( OBJ_NAME_PATH_SEPARATOR ) + NameInfo->Name.Length;
281 
282  ObpDereferenceNameInfo( NameInfo );
283  NameInfo = NULL;
284 
285  while (ObjectDirectory) {
286 
287  ObjectDirectoryHeader = OBJECT_TO_OBJECT_HEADER( ObjectDirectory );
288  NameInfo = ObpReferenceNameInfo( ObjectDirectoryHeader );
289 
290  if ((NameInfo != NULL) && (NameInfo->Directory != NULL)) {
291 
292  NameInfoSize += sizeof( OBJ_NAME_PATH_SEPARATOR ) + NameInfo->Name.Length;
293 
294  ObjectDirectory = NameInfo->Directory;
295 
296  ObfReferenceObject( ObjectDirectory );
297 
298  ObpDereferenceNameInfo( NameInfo );
299  NameInfo = NULL;
300  ObDereferenceObject( ReferencedDirectory );
301  ReferencedDirectory = ObjectDirectory;
302 
303  } else {
304 
305  break;

308

310 
311  ObDereferenceObject( ReferencedDirectory );
312  }
313

316  } else {
317 
318  NameInfoSize = 0;

321  ObpDereferenceNameInfo( NameInfo );

336

339  DACL_SECURITY_INFORMATION |
340  SACL_SECURITY_INFORMATION;

345  QuerySecurityDescriptor,
346  &SecurityInformation,
347  NULL,
348  &SecurityDescriptorSize,
349  &ObjectHeader->SecurityDescriptor,
350  ObjectType->TypeInfo.PoolType,
351  &ObjectType->TypeInfo.GenericMapping );
352 
353  } else {
354 
355  SecurityDescriptorSize = 0;
356  }
357 
358  ObjectBasicInfo.SecurityDescriptorSize = SecurityDescriptorSize;
359

361  // Now that we've packaged up our local copy of basic info we need
362  // to copy it into the output buffer and set the return
363  // length
364  //
365 
366  try {
367 
368  *(POBJECT_BASIC_INFORMATION) ObjectInformation = ObjectBasicInfo;
369 
370  TempReturnLength = ObjectInformationLength;
371 
372  } except( EXCEPTION_EXECUTE_HANDLER ) {
373 
374  //
375  // Fall through, since we cannot undo what we have done.
376  //
377  }
378 
379  break;
380 
381  case ObjectNameInformation:
382 
383  //
384  // Call a local worker routine
385  //
386 
387  Status = ObpQueryNameString( Object,
388  (POBJECT_NAME_INFORMATION)ObjectInformation,
389  ObjectInformationLength,
390  &TempReturnLength,
391  PreviousMode );
392  break;
393 
394  case ObjectTypeInformation:
395 
396  //
397  // Call a local worker routine
398  //
399 
400  Status = ObQueryTypeInfo( ObjectType,
401  (POBJECT_TYPE_INFORMATION)ObjectInformation,
402  ObjectInformationLength,
403  &TempReturnLength );
404  break;
405 
406  case ObjectTypesInformation:
407 
408  try {
409 
410  //
411  // The first thing we do is set the return length to cover the
412  // types info record. Later in each call to query type info
413  // this value will be updated as necessary
414  //
415 
416  TempReturnLength = sizeof( OBJECT_TYPES_INFORMATION );
417 
418  //
419  // Make sure there is enough room to hold the types info record
420  // and if so then compute the number of defined types there are
421  //
422 
423  TypesInformation = (POBJECT_TYPES_INFORMATION)ObjectInformation;
424 
425  if (ObjectInformationLength < sizeof( OBJECT_TYPES_INFORMATION ) ) {
426 
427  Status = STATUS_INFO_LENGTH_MISMATCH;
428 
429  } else {
430 
431  TypesInformation->NumberOfTypes = 0;
432 
433  for (i=0; i<OBP_MAX_DEFINED_OBJECT_TYPES; i++) {
434 
435  ObjectType = ObpObjectTypes[ i ];
436 
437  if (ObjectType == NULL) {
438 
439  break;
440  }
441 
442  TypesInformation->NumberOfTypes += 1;
443  }
444  }
445 
446  //
447  // For each defined type we will query the type info for the
448  // object type and adjust the TypeInfo pointer to the next
449  // free spot
450  //
451 
452  TypeInfo = (POBJECT_TYPE_INFORMATION)(((PUCHAR)TypesInformation) + ALIGN_UP( sizeof(*TypesInformation), ULONG_PTR ));
453 
454  for (i=0; i<OBP_MAX_DEFINED_OBJECT_TYPES; i++) {
455 
456  ObjectType = ObpObjectTypes[ i ];
457 
458  if (ObjectType == NULL) {
459 
460  break;
461  }
462 
463  Status = ObQueryTypeInfo( ObjectType,
464  TypeInfo,
465  ObjectInformationLength,
466  &TempReturnLength );
467 
468  if (NT_SUCCESS( Status )) {
469 
470  TypeInfo = (POBJECT_TYPE_INFORMATION)
471  ((PCHAR)(TypeInfo+1) + ALIGN_UP( TypeInfo->TypeName.MaximumLength, ULONG_PTR ));
472  }
473  }
474 
475  } except( EXCEPTION_EXECUTE_HANDLER ) {
476 
477  Status = GetExceptionCode();
478  }
479 
480  break;
481 
482  case ObjectHandleFlagInformation:
483 
484  try {
485 
486  //
487  // Set the amount of data we are going to return
488  //
489 
490  TempReturnLength = sizeof(OBJECT_HANDLE_FLAG_INFORMATION);
491 
492  HandleFlags = (POBJECT_HANDLE_FLAG_INFORMATION)ObjectInformation;
493 
494  //
495  // Make sure we have enough room for the query, and if so we'll
496  // set the output based on the flags stored in the handle
497  //
498 
499  if (ObjectInformationLength < sizeof( OBJECT_HANDLE_FLAG_INFORMATION)) {
500 
501  Status = STATUS_INFO_LENGTH_MISMATCH;
502 
503  } else {
504 
505  HandleFlags->Inherit = FALSE;
506 
507  if (HandleInformation.HandleAttributes & OBJ_INHERIT) {
508 
509  HandleFlags->Inherit = TRUE;
510  }
511 
512  HandleFlags->ProtectFromClose = FALSE;
513 
514  if (HandleInformation.HandleAttributes & OBJ_PROTECT_CLOSE) {
515 
516  HandleFlags->ProtectFromClose = TRUE;
517  }
518  }
519 
520  } except( EXCEPTION_EXECUTE_HANDLER ) {
521 
522  Status = GetExceptionCode();
523  }
524 
525  break;
526 
527  default:
528 
529  //
530  // To get to this point we must have had an object and the
531  // information class is not defined, so we should dereference the
532  // object and return to our user the bad status
533  //
534 
535  ObDereferenceObject( Object );
536 
537  return( STATUS_INVALID_INFO_CLASS );
538  }
539 
540  //
541  // Now if the caller asked for a return length we'll set it from
542  // our local copy
543  //
544 
545  try {
546 
547  if (ARGUMENT_PRESENT( ReturnLength ) ) {
548 
549  *ReturnLength = TempReturnLength;
550  }
551 
552  } except( EXCEPTION_EXECUTE_HANDLER ) {
553 
554  //
555  // Fall through, since we cannot undo what we have done.
556  //
557  }
558 
559  //
560  // In the end we can free the object if there was one and return
561  // to our caller
562  //
563 
564  if (Object != NULL) {
565 
566  ObDereferenceObject( Object );
567  }
568 
569  return( Status );
570 }

572 NTSTATUS
573 ObSetHandleAttributes (
574  __in HANDLE Handle,
575  __in POBJECT_HANDLE_FLAG_INFORMATION HandleFlags,
576  __in KPROCESSOR_MODE PreviousMode
577  )
578 {
579  BOOLEAN AttachedToProcess = FALSE;
580  KAPC_STATE ApcState;
581  OBP_SET_HANDLE_ATTRIBUTES CapturedInformation;
582  PVOID ObjectTable;
583  HANDLE ObjectHandle;
584  NTSTATUS Status;
585 
586  PAGED_CODE();
587 
588  CapturedInformation.PreviousMode = PreviousMode;
589  CapturedInformation.ObjectInformation = *HandleFlags;
590 
591  //
592  // Get the address of the object table for the current process. Or
593  // get the system handle table if this is a kernel handle and we are
594  // in kernel mode
595  //
596 
597  if (IsKernelHandle( Handle, PreviousMode )) {
598 
599  //
600  // Make the handle look like a regular handle
601  //
602 
603  ObjectHandle = DecodeKernelHandle( Handle );
604 
605  //
606  // The global kernel handle table
607  //
608 
609  ObjectTable = ObpKernelHandleTable;
610 
611  //
612  // Go to the system process
613  //
614 
615  if (PsGetCurrentProcess() != PsInitialSystemProcess) {
616  KeStackAttachProcess (&PsInitialSystemProcess->Pcb, &ApcState);
617  AttachedToProcess = TRUE;
618  }
619 
620  } else {
621 
622  ObjectTable = ObpGetObjectTable();
623  ObjectHandle = Handle;
624  }
625 
626  //
627  // Make the change to the handle table entry. The callback
628  // routine will do the actual change
629  //
630 
631  if (ExChangeHandle( ObjectTable,
632  ObjectHandle,
633  ObpSetHandleAttributes,
634  (ULONG_PTR)&CapturedInformation) ) {
635 
636  Status = STATUS_SUCCESS;
637 
638  } else {
639 
640  Status = STATUS_ACCESS_DENIED;
641  }
642 
643  //
644  // If we are attached to the system process then return
645  // back to our caller
646  //
647 
648  if (AttachedToProcess) {
649  KeUnstackDetachProcess(&ApcState);
650  AttachedToProcess = FALSE;
651  }
652  return Status;
653 }

665 
666 Routine description:
667 
668  This routine is used to set handle information about a specified
669  handle
670 
671 Arguments:
672 
673  Handle - Supplies the handle being modified
674 
675  ObjectInformationClass - Specifies the class of information being
676  modified. The only accepted value is ObjectHandleFlagInformation
677 
678  ObjectInformation - Supplies the buffer containing the handle
679  flag information structure
680 
681  ObjectInformationLength - Specifies the length, in bytes, of the
682  object information buffer
683 
684 Return Value:
685 
686  An appropriate status value
687 
688 --*/

691  NTSTATUS Status;
692  OBJECT_HANDLE_FLAG_INFORMATION CapturedFlags;
693  KPROCESSOR_MODE PreviousMode;
694 
695  PAGED_CODE();
696 
697 
698  Status = STATUS_INVALID_INFO_CLASS;
699 
700  switch (ObjectInformationClass) {
701 
702  case ObjectHandleFlagInformation:
703  {
704  if (ObjectInformationLength != sizeof(OBJECT_HANDLE_FLAG_INFORMATION)) {
705 
706  return STATUS_INFO_LENGTH_MISMATCH;
707  }
708 
709  //
710  // Get previous processor mode and probe and capture the input
711  // buffer
712  //
713 
714  PreviousMode = KeGetPreviousMode();
715 
716  try {
717 
718  if (PreviousMode != KernelMode) {
719 
720  ProbeForRead(ObjectInformation, ObjectInformationLength, 1);
721  }
722 
723  CapturedFlags = *(POBJECT_HANDLE_FLAG_INFORMATION)ObjectInformation;
724 
725  } except(ExSystemExceptionFilter()) {
726 
727  return GetExceptionCode();
728  }
729 
730  Status = ObSetHandleAttributes (Handle,
731  &CapturedFlags,
732  PreviousMode);
733 
734  }
735 
736  break;
737 
738  case ObjectSessionInformation:
739  {
740  PreviousMode = KeGetPreviousMode();
741 
742  if (!SeSinglePrivilegeCheck( SeTcbPrivilege,
743  PreviousMode)) {
744 
745  Status = STATUS_PRIVILEGE_NOT_HELD;
746 
747  } else {
748 
749  PVOID Object;
750  OBJECT_HANDLE_INFORMATION HandleInformation;
751 
752  Status = ObReferenceObjectByHandle(Handle,
753  0,
754  ObpDirectoryObjectType,
755  PreviousMode,
756  &Object,
757  &HandleInformation
758  );
759 
760  if (NT_SUCCESS(Status)) {
761 
762  POBJECT_DIRECTORY Directory;
763  OBP_LOOKUP_CONTEXT LockContext;
764  Directory = (POBJECT_DIRECTORY)Object;
765 
766  ObpInitializeLookupContext( &LockContext );

769 
770  Directory->SessionId = PsGetCurrentProcessSessionId();
771 
772  ObpUnlockDirectory(Directory, &LockContext);
773 
774  ObDereferenceObject(Object);
775  }
776  }
777  }
778 
779  break;
780  }
781 
782  //
783  // And return to our caller
784  //
785 
786  return Status;
787 }

879 {
880  NTSTATUS Status = STATUS_UNSUCCESSFUL;
881  POBJECT_HEADER ObjectHeader;
882  POBJECT_HEADER_NAME_INFO NameInfo;
883  POBJECT_HEADER ObjectDirectoryHeader;
884  POBJECT_DIRECTORY ObjectDirectory;
