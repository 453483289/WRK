284 //
285 // The following macros are used to establish the semantics needed
286 // to do a return from within a try-finally clause. As a rule every
287 // try clause must end with a label call try_exit. For example,
288 //
289 // try {
290 // :
291 // :
292 //
293 // try_exit: NOTHING;
294 // } finally {
295 //
296 // :
297 // :
298 // }
299 //
300 // Every return statement executed inside of a try clause should use the
301 // try_return macro. If the compiler fully supports the try-finally construct
302 // then the macro should be
303 //
304 // #define try_return(S) { return(S); }
305 //
306 // If the compiler does not support the try-finally construct then the macro
307 // should be
308 //
309 // #define try_return(S) { S; goto try_exit; }
310 //
311 
312 #define try_return(S) { S; goto try_exit; }
