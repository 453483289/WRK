25 NTSTATUS
26 NtLockFile(
27  __in HANDLE FileHandle,
28  __in_opt HANDLE Event,
29  __in_opt PIO_APC_ROUTINE ApcRoutine,
30  __in_opt PVOID ApcContext,
31  __out PIO_STATUS_BLOCK IoStatusBlock,
32  __in PLARGE_INTEGER ByteOffset,
33  __in PLARGE_INTEGER Length,
34  __in ULONG Key,
35  __in BOOLEAN FailImmediately,

39 /*++
40 
41 Routine Description:
42 
43  This service locks a specified range of bytes on the file specified by
44  the FileHandle parameter. The lock may either be an exclusive lock or
45  a shared lock. Furthermore, the caller has the option of specifying
46  whether or not the service should return immediately if the lock cannot
47  be acquired without waiting.
48 
49 Arguments:
50 
51  FileHandle - Supplies a handle to an open file.
52 
53  Event - Supplies an optional event to be set to the Signaled state when
54  the operation is complete.
55 
56  ApcRoutine - Supplies an optional APC routine to be executed when the
57  operation is complete.
58 
59  ApcContext - Supplies a context parameter to be passed to the ApcRoutine,
60  if an ApcRoutine was specified.
61 
62  IoStatusBlock - Address of the caller's I/O status block.
63 
64  ByteOffset - Specifies the starting byte offset of the range to lock.
65 
66  Length - Specifies the length of the byte range to be locked.
67 
68  Key - Specifies the key to be associated with the lock.
69 
70  FailImmediately - Specifies that if the lock cannot immediately be
71  acquired that the service should return to the caller.
72 
73  ExclusiveLock - Specifies, if TRUE, that the lock should be an exclusive
74  lock; otherwise the lock is a shared lock.
75 
76 Return Value:
77 
78  The status returned is success if the operation was properly queued to
79  the I/O system. Once the operation completes, the status can be
80  determined by examining the Status field of the I/O status block.
81 
82 --*/

85  PIRP irp;
86  NTSTATUS status;
87  PFILE_OBJECT fileObject;
88  PDEVICE_OBJECT deviceObject;
89  PFAST_IO_DISPATCH fastIoDispatch;
90  PKEVENT eventObject = (PKEVENT) NULL;
91  KPROCESSOR_MODE requestorMode;
92  PIO_STACK_LOCATION irpSp;
93  LARGE_INTEGER fileOffset;
94  LARGE_INTEGER length;
95  ACCESS_MASK grantedAccess;
96  OBJECT_HANDLE_INFORMATION handleInformation;
97  BOOLEAN synchronousIo;
98  PETHREAD CurrentThread;
99 
100  PAGED_CODE();
101 
102  //
103  // Get the previous mode; i.e., the mode of the caller.
104  //
105 
106  CurrentThread = PsGetCurrentThread ();
107  requestorMode = KeGetPreviousModeByThread(&CurrentThread->Tcb);
108 
109  //
110  // Reference the file object so the target device can be found and the
111  // access rights mask can be used in the following checks for callers
112  // in user mode. Note that if the handle does not refer to a file
113  // object, then it will fail.
114  //
115 
116  status = ObReferenceObjectByHandle( FileHandle,
117  0L,
118  IoFileObjectType,
119  requestorMode,
120  (PVOID *) &fileObject,
121  &handleInformation);
122  if (!NT_SUCCESS( status )) {
123  return status;
124  }
125 
126  grantedAccess = handleInformation.GrantedAccess;
127 
128  if (requestorMode != KernelMode) {
129 
130  //
131  // The caller's access mode is not kernel so probe each of the arguments
132  // and capture them as necessary. If any failures occur, the condition
133  // handler will be invoked to handle them. It will simply cleanup and
134  // return an access violation status code back to the system service
135  // dispatcher.
136  //
137 
138  //
139  // Check to ensure that the caller has either READ or WRITE access to
140  // the file. If not, cleanup and return an error.
141  //
142 
143  if (!SeComputeGrantedAccesses( grantedAccess, FILE_READ_DATA | FILE_WRITE_DATA )) {
144  ObDereferenceObject( fileObject );
145  return STATUS_ACCESS_DENIED;
146  }
147 
148  try {
149 
150  //
151  // The IoStatusBlock parameter must be writeable by the caller.
152  //
153 
154  ProbeForWriteIoStatus(IoStatusBlock);
155 
156  //
157  // If this is a 32-bit asynchronous IO, then mark the Iosb being sent as so.
158  // Note: IopMarkApcRoutineIfAsyncronousIo32 must be called after probing
159  // the IoStatusBlock structure for write.
160  //
161 
162  IopMarkApcRoutineIfAsyncronousIo32(IoStatusBlock,ApcRoutine,(fileObject->Flags & FO_SYNCHRONOUS_IO));
163 
164  //
165  // The ByteOffset parameter must be readable by the caller. Probe
166  // and capture it.
167  //
168 
169  ProbeForReadSmallStructure( ByteOffset,
170  sizeof( LARGE_INTEGER ),
171  sizeof( ULONG ) );
172  fileOffset = *ByteOffset;
173 
174  //
175  // Likewise, the Length parameter must also be readable by the
176  // caller. Probe and capture it as well.
177  //
178 
179  ProbeForReadSmallStructure( Length,
180  sizeof( LARGE_INTEGER ),
181  sizeof( ULONG ) );
182  length = *Length;
183 
184  //
185  // If this file has an I/O completion port associated w/it, then
186  // ensure that the caller did not supply an APC routine, as the
187  // two are mutually exclusive methods for I/O completion
188  // notification.
189  //
190 
191  if (fileObject->CompletionContext && IopApcRoutinePresent( ApcRoutine )) {
192  ObDereferenceObject( fileObject );
193  return STATUS_INVALID_PARAMETER;
194  }
195 
196  } except(EXCEPTION_EXECUTE_HANDLER) {
197 
198  //
199  // An exception was incurred attempting to probe the caller's
200  // parameters. Dereference the file object and return an
201  // appropriate error status code.
202  //
203 
204  ObDereferenceObject( fileObject );
205  return GetExceptionCode();
206  }
207 
208  } else {
209 
210  //
211  // The caller's mode was kernel. Get the ByteOffset and Length
212  // parameter 's to the expected locations.
213  //
214 
215  fileOffset = *ByteOffset;
216  length = *Length;
217  }
218 
219  //
220  // Get the address of the event object and set the event to the Not-
221  // Signaled state, if an event was specified. Note here, too, that if
222  // the handle does not refer to an event, or if the event cannot be
223  // written, then the reference will fail. Since certain legacy
224  // applications rely on an old bug in Win32's LockFileEx, we must
225  // tolerate bad event handles.
226  //
227 
228  if (ARGUMENT_PRESENT( Event )) {
229  status = ObReferenceObjectByHandle( Event,
230  EVENT_MODIFY_STATE,
231  ExEventObjectType,
232  requestorMode,
233  (PVOID *) &eventObject,
234  NULL );
235  if (!NT_SUCCESS( status )) {
236  ASSERT( !eventObject );
237  } else {
238  KeClearEvent( eventObject );
239  }
240  }
241 
242  //
243  // Get the address of the target device object and the fast Io dispatch
244  // structure.
245  //
246 
247  deviceObject = IoGetRelatedDeviceObject( fileObject );
248  fastIoDispatch = deviceObject->DriverObject->FastIoDispatch;
249 
250  //
251  // Turbo lock support. If the fast Io Dispatch specifies a fast lock
252  // routine then we'll first try and calling it with the specified lock
253  // parameters.
254  //
255 
256  if (fastIoDispatch && fastIoDispatch->FastIoLock) {
257 
258  IO_STATUS_BLOCK localIoStatus;
259 
260  if (fastIoDispatch->FastIoLock( fileObject,
261  &fileOffset,
262  &length,
263  PsGetCurrentProcessByThread(CurrentThread),
264  Key,
265  FailImmediately,
266  ExclusiveLock,
267  &localIoStatus,
268  deviceObject )) {
269 
270  //
271  // Carefully return the I/O status.
272  //
273 
274  try {
275 #if defined(_WIN64)
276  //
277  // If this is a32-bit thread, and the IO request is
278  // asynchronous, then the IOSB is 32-bit. Wow64 always sends
279  // the 32-bit IOSB when the I/O is asynchronous.
280  //
281  if (IopIsIosb32(ApcRoutine)) {
282  PIO_STATUS_BLOCK32 UserIosb32 = (PIO_STATUS_BLOCK32)IoStatusBlock;
283 
284  UserIosb32->Information = (ULONG)localIoStatus.Information;
285  UserIosb32->Status = (NTSTATUS)localIoStatus.Status;
286  } else {
287  *IoStatusBlock = localIoStatus;
288  }
289 #else
290  *IoStatusBlock = localIoStatus;
291 #endif
292  } except( EXCEPTION_EXECUTE_HANDLER ) {
293  localIoStatus.Status = GetExceptionCode();
294  localIoStatus.Information = 0;
295  }
296 
297  //
298  // If a valid event was specified, set it.
299  //
300 
301  if (eventObject) {
302  KeSetEvent( eventObject, 0, FALSE );
303  ObDereferenceObject( eventObject );
304  }
305 
306  //
307  // Note that the file object event need not be set to the
308  // Signaled state, as it is already set.
309  //
310 
311  //
312  // If this file object has a completion port associated with it
313  // and this request has a non-NULL APC context then a completion
314  // message needs to be queued.
315  //
316 
317  if (fileObject->CompletionContext && ARGUMENT_PRESENT( ApcContext )) {
318  if (!NT_SUCCESS(IoSetIoCompletion( fileObject->CompletionContext->Port,
319  fileObject->CompletionContext->Key,
320  ApcContext,
321  localIoStatus.Status,
322  localIoStatus.Information,
323  TRUE ))) {
324  localIoStatus.Status = STATUS_INSUFFICIENT_RESOURCES;
325  }
326  }
327 
328  //
329  // Cleanup and return.
330  //
331 
332  fileObject->LockOperation = TRUE;
333  ObDereferenceObject( fileObject );
334  return localIoStatus.Status;
335  }
336  }
337 
338  //
339  // Make a special check here to determine whether this is a synchronous
340  // I/O operation. If it is, then wait here until the file is owned by
341  // the current thread.
342  //
343 
344  if (fileObject->Flags & FO_SYNCHRONOUS_IO) {
345 
346  BOOLEAN interrupted;
347 
348  if (!IopAcquireFastLock( fileObject )) {
349  status = IopAcquireFileObjectLock( fileObject,
350  requestorMode,
351  (BOOLEAN) ((fileObject->Flags & FO_ALERTABLE_IO) != 0),
352  &interrupted );
353  if (interrupted) {
354  if (eventObject) {
355  ObDereferenceObject( eventObject );
356  }
357  ObDereferenceObject( fileObject );
358  return status;
359  }
360  }
361  synchronousIo = TRUE;
362  } else {
363  synchronousIo = FALSE;
364  }
365 
366  //
367  // Set the file object to the Not-Signaled state and mark it as having had
368  // a lock operation performed on it.
369  //
370 
371  KeClearEvent( &fileObject->Event );
372  fileObject->LockOperation = TRUE;
373 
374  //
375  // Allocate and initialize the I/O Request Packet (IRP) for this operation.
376  // The allocation is performed with an exception handler in case the
377  // caller does not have enough quota to allocate the packet.
378 
379  irp = IoAllocateIrp( deviceObject->StackSize, !synchronousIo );
380  if (!irp) {
381 
382  //
383  // An IRP could not be allocated. Cleanup and return an appropriate
384  // error status code.
385  //
386 
387  IopAllocateIrpCleanup( fileObject, eventObject );
388 
389  return STATUS_INSUFFICIENT_RESOURCES;
390  }
391  irp->Tail.Overlay.OriginalFileObject = fileObject;
392  irp->Tail.Overlay.Thread = CurrentThread;
393  irp->RequestorMode = requestorMode;
394 
395  //
396  // Fill in the service independent parameters in the IRP.
397  //
398 
399  irp->UserEvent = eventObject;
400  irp->UserIosb = IoStatusBlock;
401  irp->Overlay.AsynchronousParameters.UserApcRoutine = ApcRoutine;
402  irp->Overlay.AsynchronousParameters.UserApcContext = ApcContext;
403 
404  //
405  // Get a pointer to the stack location for the first driver. This will be
406  // used to pass the original function codes and parameters.
407  //
408 
409  irpSp = IoGetNextIrpStackLocation( irp );
410  irpSp->MajorFunction = IRP_MJ_LOCK_CONTROL;
411  irpSp->MinorFunction = IRP_MN_LOCK;
412  irpSp->FileObject = fileObject;
413 
414  //
415  // Copy the caller's parameters to the service-specific portion of the
416  // IRP.
417  //
418 
419  irpSp->Flags = 0;
420  if (FailImmediately) {
421  irpSp->Flags = SL_FAIL_IMMEDIATELY;
422  }
423  if (ExclusiveLock) {
424  irpSp->Flags |= SL_EXCLUSIVE_LOCK;
425  }
426  irpSp->Parameters.LockControl.Key = Key;
427  irpSp->Parameters.LockControl.ByteOffset = fileOffset;
428 
429  try {
430  PLARGE_INTEGER lengthBuffer;
431 
432  //
433  // Attempt to allocate an intermediary buffer to hold the length of
434  // this lock operation. If it fails, either because there is no
435  // more quota, or because there are no more resources, then the
436  // exception handler will be invoked to cleanup and exit.
437  //
438 
439  lengthBuffer = ExAllocatePoolWithQuota( NonPagedPool,
440  sizeof( LARGE_INTEGER ) );
441 
442  *lengthBuffer = length;
443  irp->Tail.Overlay.AuxiliaryBuffer = (PCHAR) lengthBuffer;
444  irpSp->Parameters.LockControl.Length = lengthBuffer;
445  } except(EXCEPTION_EXECUTE_HANDLER) {
446 
447  //
448  // An exception was incurred. Simply clean everything up and
449  // return an appropriate error status code.
450  //
451 
452  IopExceptionCleanup( fileObject,
453  irp,
454  eventObject,
455  (PKEVENT) NULL );
456 
457  return GetExceptionCode();
458  }
459 
460  //
461  // Queue the packet, call the driver, and synchronize appropriately with
462  // I/O completion.
463  //
464 
465  return IopSynchronousServiceTail( deviceObject,
466  irp,
467  fileObject,
468  FALSE,
469  requestorMode,
470  synchronousIo,
471  OtherTransfer );
472 }

474 NTSTATUS
475 NtUnlockFile (
476  __in HANDLE FileHandle,
477  __out PIO_STATUS_BLOCK IoStatusBlock,
478  __in PLARGE_INTEGER ByteOffset,
479  __in PLARGE_INTEGER Length,
480  __in ULONG Key
481  )
482 
483 /*++
484 
485 Routine Description:
486 
487  This service releases the lock associated with the specified byte range
488  for the file specified by the FileHandle parameter.
489 
490 Arguments:
491 
492  FileHandle - Supplies a handle to an open file.
493 
494  IoStatusBlock - Address of the caller's I/O status block.
495 
496  ByteOffset - Specifies the byte offset of the range to unlock.
497 
498  Length - Specifies the length of the byte range to unlock.
499 
500  Key - Specifies the key associated with the locked range.
501 
502 Return Value:
503 
504  The status returned is the final completion status of the operation.
505 
506 --*/

514  PKEVENT event;
515  KPROCESSOR_MODE requestorMode;
516  PIO_STACK_LOCATION irpSp;
517  IO_STATUS_BLOCK localIoStatus;
518  LARGE_INTEGER fileOffset;
519  LARGE_INTEGER length;
520  ACCESS_MASK grantedAccess;
521  OBJECT_HANDLE_INFORMATION handleInformation;
522  BOOLEAN synchronousIo;
523  PETHREAD CurrentThread;

528  // Get the previous mode; i.e., the mode of the caller.
529  //
530

532  requestorMode = KeGetPreviousModeByThread(&CurrentThread->Tcb);
533

536  // access rights mask can be used in the following checks for callers

538  // object, then it will fail.
539  //
540

542  0L,
543  IoFileObjectType,
544  requestorMode,
545  (PVOID *) &fileObject,
546  &handleInformation);
547  if (!NT_SUCCESS( status )) {
548  return status;
549  }
550

552 
553  //
554  // Check to see if the requestor mode was user. If so, perform a bunch
555  // of extra checks.
556  //
557 
558  if (requestorMode != KernelMode) {
559 
560  //
561  // The caller's access mode is not kernel so probe each of the arguments
562  // and capture them as necessary. If any failures occur, the condition
563  // handler will be invoked to handle them. It will simply cleanup and
564  // return an access violation status code back to the system service
565  // dispatcher.
566  //
567 
568  //
569  // Check to ensure that the caller has either READ or WRITE access
570  // to the file. If not, cleanup and return an error.
571  //
572 
573  if (!SeComputeGrantedAccesses( grantedAccess, FILE_READ_DATA | FILE_WRITE_DATA )) {
574  ObDereferenceObject( fileObject );
575  return STATUS_ACCESS_DENIED;
576  }
577

579 
580  //
581  // The IoStatusBlock parameter must be writeable by the caller.
582  //
583 
584  ProbeForWriteIoStatus( IoStatusBlock );
585 
586  //

597  // Likewise, the Length parameter must also be readable by the

599  //
600 
601  ProbeForReadSmallStructure( Length,
602  sizeof( LARGE_INTEGER ),
603  sizeof( ULONG ) );
604  length = *Length;
605 
606  } except(EXCEPTION_EXECUTE_HANDLER) {
607 
608  //
609  // An exception was incurred while attempting to probe the
610  // caller's parameters. Dereference the file object and return
611  // an appropriate error status code.
612  //
613 
614  ObDereferenceObject( fileObject );
615  return GetExceptionCode();
616 
617  }
618 
619  } else {
620 
621  //
622  // The caller's mode was kernel. Get the ByteOffset and Length
623  // parameter 's to the expected locations.
624  //
625 
626  fileOffset = *ByteOffset;
627  length = *Length;
628  }
629 
630  //
631  // Get the address of the target device object. If this file represents
632  // a device that was opened directly, then simply use the device or its
633  // attached device(s) directly. Also get the fast I/O dispatch address.
634  //
635 
636  if (!(fileObject->Flags & FO_DIRECT_DEVICE_OPEN)) {
637  deviceObject = IoGetRelatedDeviceObject( fileObject );
638  } else {
639  deviceObject = IoGetAttachedDevice( fileObject->DeviceObject );
640  }
641  fastIoDispatch = deviceObject->DriverObject->FastIoDispatch;
642 
643  //
644  // Turbo lock support. If the fast Io Dispatch specifies a fast lock
645  // routine then we'll first try and calling it with the specified lock
646  // parameters.
647  //
648 
649  if (fastIoDispatch && fastIoDispatch->FastIoUnlockSingle) {
650 
651  IO_STATUS_BLOCK localIoStatus;
652 
653  if (fastIoDispatch->FastIoUnlockSingle( fileObject,
654  &fileOffset,
655  &length,
656  PsGetCurrentProcessByThread(CurrentThread),
657  Key,
658  &localIoStatus,
659  deviceObject )) {
660 
661  //
662  // Carefully return the I/O status.
663  //
664 
665  try {
666  *IoStatusBlock = localIoStatus;
667  } except( EXCEPTION_EXECUTE_HANDLER ) {
668  localIoStatus.Status = GetExceptionCode();
669  localIoStatus.Information = 0;
670  }
671 
672  //
673  // Cleanup and return.
674  //
675 
676  ObDereferenceObject( fileObject );
677  return localIoStatus.Status;
678  }
679  }
680 
681  //
682  // Make a special check here to determine whether this is a synchronous
683  // I/O operation. If it is, then wait here until the file is owned by
684  // the current thread. If this is not a (serialized) synchronous I/O
685  // operation, then allocate and initialize the local event.
686  //
687 
688  if (fileObject->Flags & FO_SYNCHRONOUS_IO) {
689 
690  BOOLEAN interrupted;
691 
692  if (!IopAcquireFastLock( fileObject )) {
693  status = IopAcquireFileObjectLock( fileObject,
694  requestorMode,
695  (BOOLEAN) ((fileObject->Flags & FO_ALERTABLE_IO) != 0),
696  &interrupted );
697  if (interrupted) {
698  ObDereferenceObject( fileObject );
699  return status;
700  }
701  }
702  synchronousIo = TRUE;
703  event = NULL;
704  } else {
705 
706  //
707  // This is a synchronous API being invoked for a file that is opened
708  // for asynchronous I/O. This means that this system service is
709  // to synchronize the completion of the operation before returning
710  // to the caller. A local event is used to do this.
711  //
712 
713  event = ExAllocatePool( NonPagedPool, sizeof( KEVENT ) );
714  if (event == NULL) {
715  ObDereferenceObject( fileObject );
716  return STATUS_INSUFFICIENT_RESOURCES;
717  }
718  KeInitializeEvent( event, SynchronizationEvent, FALSE );
719  synchronousIo = FALSE;
720  }
721 
722  //
723  // Set the file object to the Not-Signaled state.
724  //
725 
726  KeClearEvent( &fileObject->Event );
727 
728  //
729  // Allocate and initialize the I/O Request Packet (IRP) for this operation.
730  // The allocation is performed with an exception handler in case the
731  // caller does not have enough quota to allocate the packet.
732  //
733 
734  irp = IoAllocateIrp( deviceObject->StackSize, FALSE );
735  if (!irp) {
736 
737  //
738  // An IRP could not be allocated. Cleanup and return an appropriate
739  // error status code.
740  //
741 
742  if (event) {
743  ExFreePool( event );
744  }
745 
746  IopAllocateIrpCleanup( fileObject, (PKEVENT) NULL );
747 
748  return STATUS_INSUFFICIENT_RESOURCES;
749  }
750  irp->Tail.Overlay.OriginalFileObject = fileObject;
751  irp->Tail.Overlay.Thread = CurrentThread;
752  irp->RequestorMode = requestorMode;
753 
754  //
755  // Fill in the service independent parameters in the IRP.
756  //
757 
758  if (synchronousIo) {
759  irp->UserEvent = (PKEVENT) NULL;
760  irp->UserIosb = IoStatusBlock;
761  } else {
762  irp->UserEvent = event;
763  irp->UserIosb = &localIoStatus;
764  irp->Flags = IRP_SYNCHRONOUS_API;
765  }
766  irp->Overlay.AsynchronousParameters.UserApcRoutine = (PIO_APC_ROUTINE) NULL;
767 
768  //
769  // Get a pointer to the stack location for the first driver. This will
770  // be used to pass the original function codes and parameters.
771  //
772 
773  irpSp = IoGetNextIrpStackLocation( irp );
774  irpSp->MajorFunction = IRP_MJ_LOCK_CONTROL;
775  irpSp->MinorFunction = IRP_MN_UNLOCK_SINGLE;
776  irpSp->FileObject = fileObject;
777 
778  try {
779  PLARGE_INTEGER lengthBuffer;
780 
781  //
782  // Attempt to allocate an intermediary buffer to hold the length of
783  // this lock operation. If it fails, either because there is no
784  // more quota, or because there are no more resources, then the
785  // exception handler will be invoked to cleanup and exit.
786  //
787 
788  lengthBuffer = ExAllocatePoolWithQuota( NonPagedPool,
789  sizeof( LARGE_INTEGER ) );
790 
791  *lengthBuffer = length;
792  irp->Tail.Overlay.AuxiliaryBuffer = (PCHAR) lengthBuffer;
793  irpSp->Parameters.LockControl.Length = lengthBuffer;
794  } except(EXCEPTION_EXECUTE_HANDLER) {
795 
796  //
797  // An exception was incurred. Simply clean everything up and
798  // return an appropriate error status code.
