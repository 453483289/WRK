192 NTSTATUS
193 WmipDriverEntry(
194  IN PDRIVER_OBJECT DriverObject,
195  IN PUNICODE_STRING RegistryPath
196  )
197 
198 /*++
199 
200 Routine Description:
201 
202  This is the callback function when we call IoCreateDriver to create a
203  WMI Driver Object. In this function, we need to remember the
204  DriverObject, create a device object and then create a Win32 visible
205  symbolic link name so that the WMI user mode component can access us.
206 
207 Arguments:
208 
209  DriverObject - Pointer to the driver object created by the system.
210 
211  RegistryPath - is NULL.
212 
213 Return Value:
214 
215  STATUS_SUCCESS
216 
217 --*/
218 
219 {
220  NTSTATUS Status;
221  UNICODE_STRING DeviceName;
222  UNICODE_STRING ServiceSymbolicLinkName;
223  UNICODE_STRING AdminSymbolicLinkName;
224  PSECURITY_DESCRIPTOR AdminDeviceSd;
225  PFAST_IO_DISPATCH fastIoDispatch;
226  ANSI_STRING AnsiString;
227 
228  PAGED_CODE();
229  UNREFERENCED_PARAMETER(RegistryPath);
230 
231  //
232  // First thing to do is make sure our critical section has been initialized
233  //
234  KeInitializeMutex(&WmipSMMutex, 0);
235  KeInitializeMutex(&WmipTLMutex, 0);
236 
237  //
238  // Initialize internal WMI data structures
239  //
240  WmipInitializeRegistration(0);
241  WmipInitializeNotifications();
242  Status = WmipInitializeDataStructs();
243  if (! NT_SUCCESS(Status))
244  {
245  return(Status);
246  }
247 
248  //
249  // Since Io does not pass a registry path for this device we need to make
250  // up one
251  RtlInitAnsiString(&AnsiString,
252  "\\Registry\\Machine\\System\\CurrentControlSet\\Services\\WMI");
253  Status = RtlAnsiStringToUnicodeString(&WmipRegistryPath,
254  &AnsiString,
255  TRUE);
256  Status = WmipInitializeSecurity();
257  if (!NT_SUCCESS(Status))
258  {
259  return(Status);
260  }
261 
262  //
263  // We allocate a security descriptor to be placed on the Admin device.
264  // It will allow only Administrators access to the device and
265  // no access to anyone else.
266  //
267  Status = WmipCreateAdminSD(&AdminDeviceSd);
268  if (! NT_SUCCESS(Status))
269  {
270  return(Status);
271  }
272 
273  //
274  // Create the service device object and symbolic link
275  //
276  RtlInitUnicodeString( &DeviceName, WMIServiceDeviceObjectName );
277  Status = IoCreateDevice(
278  DriverObject,
279  0,
280  &DeviceName,
281  FILE_DEVICE_UNKNOWN,
282  FILE_DEVICE_SECURE_OPEN, // No standard device characteristics
283  FALSE, // This isn't an exclusive device
284  &WmipServiceDeviceObject
285  );
286 
287  if (! NT_SUCCESS(Status))
288  {
289  ExFreePool(AdminDeviceSd);
290  return(Status);
291  }
292 
293  RtlInitUnicodeString( &ServiceSymbolicLinkName,
294  WMIServiceSymbolicLinkName );
295  Status = IoCreateSymbolicLink( &ServiceSymbolicLinkName,
296  &DeviceName );
297  if (! NT_SUCCESS(Status))
298  {
299  IoDeleteDevice( WmipServiceDeviceObject );
300  ExFreePool(AdminDeviceSd);
301  return(Status);
302  }
303 
304 
305  //
306  // Now create an admin-only device object and symbolic link
307  //
308  RtlInitUnicodeString( &DeviceName, WMIAdminDeviceObjectName );
309  Status = IoCreateDevice(
310  DriverObject,
311  0,
312  &DeviceName,
313  FILE_DEVICE_UNKNOWN,
314  FILE_DEVICE_SECURE_OPEN, // No standard device characteristics
315  FALSE, // This isn't an exclusive device
316  &WmipAdminDeviceObject
317  );
318 
319  if (! NT_SUCCESS(Status))
320  {
321  IoDeleteDevice( WmipServiceDeviceObject );
322  IoDeleteSymbolicLink(&ServiceSymbolicLinkName);
323  ExFreePool(AdminDeviceSd);
324  return(Status);
325  }
326 
327 
328  Status = ObSetSecurityObjectByPointer(WmipAdminDeviceObject,
329  DACL_SECURITY_INFORMATION |
330  OWNER_SECURITY_INFORMATION,
331  AdminDeviceSd);
332 
333  ExFreePool(AdminDeviceSd);
334  AdminDeviceSd = NULL;
335 
336  if (! NT_SUCCESS(Status))
337  {
338  IoDeleteDevice( WmipServiceDeviceObject );
339  IoDeleteDevice( WmipAdminDeviceObject );
340  IoDeleteSymbolicLink(&ServiceSymbolicLinkName);
341  return(Status);
342  }
343 
344  RtlInitUnicodeString( &AdminSymbolicLinkName,
345  WMIAdminSymbolicLinkName );
346  Status = IoCreateSymbolicLink( &AdminSymbolicLinkName,
347  &DeviceName );
348  if (! NT_SUCCESS(Status))
349  {
350  IoDeleteSymbolicLink( &ServiceSymbolicLinkName );
351  IoDeleteDevice( WmipServiceDeviceObject );
352  IoDeleteDevice( WmipAdminDeviceObject );
353  return(Status);
354  }
355 
356  //
357  // Establish an initial irp stack size
358  WmipServiceDeviceObject->StackSize = WmiDeviceStackSize;
359  WmipAdminDeviceObject->StackSize = WmiDeviceStackSize;
360 
361  //
362  // Create dispatch entrypoints
363  //
364  DriverObject->MajorFunction[IRP_MJ_CREATE] = WmipOpenCloseCleanup;
365  DriverObject->MajorFunction[IRP_MJ_CLOSE] = WmipOpenCloseCleanup;
366  DriverObject->MajorFunction[IRP_MJ_DEVICE_CONTROL] = WmipIoControl;
367  DriverObject->MajorFunction[IRP_MJ_CLEANUP] = WmipOpenCloseCleanup;
368  DriverObject->MajorFunction[IRP_MJ_SYSTEM_CONTROL] = WmipSystemControl;
369  DriverObject->MajorFunction[IRP_MJ_SHUTDOWN] = WmipShutdown;
370 
371  //
372  // Register for notification of docking events
373 #if defined(_AMD64_) || defined(i386)
374  IoRegisterPlugPlayNotification(
375  EventCategoryHardwareProfileChange,
376  0,
377  NULL,
378  DriverObject,
379  WmipDockUndockEventCallback,
380  NULL,
381  &WmipDockUndockNotificationEntry);
382 #endif
383  //
384  // We reset this flag to let the IO manager know that the device
385  // is ready to receive requests. We only do this for the kernel
386  // dll since the IO manager does it when WMI loads as a normal
387  // device.
388  WmipServiceDeviceObject->Flags &= ~DO_DEVICE_INITIALIZING;
389  WmipAdminDeviceObject->Flags &= ~DO_DEVICE_INITIALIZING;
390 
391  IoWMIRegistrationControl(WmipServiceDeviceObject,
392  WMIREG_ACTION_REGISTER);
393 
394  fastIoDispatch = &WmipFastIoDispatch;
395  RtlZeroMemory(fastIoDispatch, sizeof(FAST_IO_DISPATCH));
396  fastIoDispatch->SizeOfFastIoDispatch = sizeof(FAST_IO_DISPATCH);
397  fastIoDispatch->FastIoDeviceControl = WmipFastIoDeviceControl;
398  DriverObject->FastIoDispatch = fastIoDispatch;
399  RtlZeroMemory(&WmipRefCount[0], MAXLOGGERS*sizeof(ULONG));
400  RtlZeroMemory(&WmipLoggerContext[0], MAXLOGGERS*sizeof(PWMI_LOGGER_CONTEXT));
401  WmipStartGlobalLogger(); // Try and see if we need to start this
402  IoRegisterShutdownNotification(WmipServiceDeviceObject);
403 
404  SharedUserData->TraceLogging = 0; //Initialize the Heap and Crisec Coll tracing status off
405 
406  return(Status);
407 }

426 void WmipUpdateDeviceStackSize(
427  CCHAR NewStackSize
428  )
429 /*++
430 
431 Routine Description:
432 
433  This routine will update the stack size that is specified in the WMI
434  device's device object. This needs to be protected since it can be updated
435  when a device registers and whenever an irp is forwarded to a device.
436  WMI needs to maintain a stack size one greater than the stack size of the
437  largest device stack to which it forwards irps to. Consider a bottom
438  driver that registers with WMI and has a stack size of 1. If 2 device
439  attach on top of it then WMI will forward to the topmost in the stack
440  which would need a stack size of 3, so the original WMI irp (ie the one
441  created by the IOCTL to the WMI device) would need a stack size of 4.
442 
443 Arguments:
444 
445  NewStackSize is the new stack size needed
446 
447 Return Value:
448 
449  NT status ccode
450 
451 --*/
452 {
453  PAGED_CODE();
454 
455  WmipEnterSMCritSection();
456  if (WmipServiceDeviceObject->StackSize < NewStackSize)
457  {
458  WmipServiceDeviceObject->StackSize = NewStackSize;
459  }
460  WmipLeaveSMCritSection();
461 }

472  ULONG InBufferLen = irpStack->Parameters.DeviceIoControl.InputBufferLength;
473  ULONG OutBufferLen = irpStack->Parameters.DeviceIoControl.OutputBufferLength;
474  PVOID Buffer = Irp->AssociatedIrp.SystemBuffer;
475  PWNODE_HEADER Wnode = (PWNODE_HEADER)Buffer;
476  ULONG Ioctl;
477 
478  PAGED_CODE();
479 
480  Ioctl = irpStack->Parameters.DeviceIoControl.IoControlCode;
481 
482  switch (Ioctl)
483  {
484  case IOCTL_WMI_OPEN_GUID:
485  case IOCTL_WMI_OPEN_GUID_FOR_QUERYSET:
486  case IOCTL_WMI_OPEN_GUID_FOR_EVENTS:
487  {
488  OBJECT_ATTRIBUTES CapturedObjectAttributes;
489  UNICODE_STRING CapturedGuidString;
490  WCHAR CapturedGuidBuffer[WmiGuidObjectNameLength + 1];
491  PWMIOPENGUIDBLOCK InGuidBlock;
492  HANDLE Handle;
493  ULONG DesiredAccess;
494 
495  InGuidBlock = (PWMIOPENGUIDBLOCK)Buffer;
496 
497  Status = WmipProbeWmiOpenGuidBlock(&CapturedObjectAttributes,
498  &CapturedGuidString,
499  CapturedGuidBuffer,
500  &DesiredAccess,
501  InGuidBlock,
502  InBufferLen,
503  OutBufferLen);
504 
505  if (NT_SUCCESS(Status))
506  {
507  Status = WmipOpenBlock(Ioctl,
508  UserMode,
509  &CapturedObjectAttributes,
510  DesiredAccess,
511  &Handle);
512  if (NT_SUCCESS(Status))
513  {
514 #if defined(_WIN64)
515  if (IoIs32bitProcess(NULL))
516  {
517  ((PWMIOPENGUIDBLOCK32)InGuidBlock)->Handle.Handle32 = PtrToUlong(Handle);
518  }
519  else
520 #endif
521  {
522  InGuidBlock->Handle.Handle = Handle;
523  }
524  }
525  }
526  break;
527  }
528 
529  case IOCTL_WMI_QUERY_ALL_DATA:
530  {
531  if (OutBufferLen < sizeof(WNODE_ALL_DATA))
532  {
533  //
534  // WMI will not send any request whose output buffer is not
535  // at least the size of a WNODE_ALL_DATA.
536  Status = STATUS_BUFFER_TOO_SMALL;
537  break;
538  }
539 
540  Status = WmipProbeWnodeAllData((PWNODE_ALL_DATA)Wnode,
541  InBufferLen,
542  OutBufferLen);
543 
544  if (NT_SUCCESS(Status))
545  {
546  Status = WmipQueryAllData(NULL,
547  Irp,
548  UserMode,
549  (PWNODE_ALL_DATA)Wnode,
550  OutBufferLen,
551  &OutBufferLen);
552 
553  } else {
554  WmipDebugPrintEx((DPFLTR_WMICORE_ID, DPFLTR_INFO_LEVEL,"WMI: Invalid IOCTL_WMI_QUERY_ALL_DATA Wnode\n"));
555  }
556  break;
557  }
558 
559  case IOCTL_WMI_QAD_MULTIPLE:
560  {
561  PWMIQADMULTIPLE QadMultiple;
562 
563  QadMultiple = (PWMIQADMULTIPLE)Buffer;
564 
565  //
566  // Check that the input/output sizes make sense and that we have a
567  // valid HandleCount.
568  //
569 
570  if ((OutBufferLen >= sizeof(WNODE_TOO_SMALL)) &&
571  RTL_CONTAINS_FIELD(QadMultiple, InBufferLen, HandleCount) &&
572  (QadMultiple->HandleCount > 0) &&
573  (QadMultiple->HandleCount < QUERYMULIPLEHANDLELIMIT) &&
574  RTL_CONTAINS_FIELD(QadMultiple, InBufferLen, Handles[QadMultiple->HandleCount - 1])) {
575 
576  Status = WmipQueryAllDataMultiple(0,
577  NULL,
578  Irp,
579  UserMode,
580  Buffer,
581  OutBufferLen,
582  QadMultiple,
583  &OutBufferLen);

586  }
587

589  }

596  //
