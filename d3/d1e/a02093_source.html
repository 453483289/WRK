299 /*++
300 
301 Routine Description:
302 
303  Marks the relevant parts of a hive dirty, so that they will
304  be flushed to backing store.
305 
306  If Hive->Cluster is not 1, then adjacent all logical sectors
307  in the given cluster will be forced dirty (and log space
308  allocated for them.) This must be done here rather than in
309  HvSyncHive so that we can know how much to grow the log.
310 
311  This is a noop for Volatile address range.
312 
313  NOTE: Range will not be marked dirty if operation fails.
314 
315  ATTENTION: This routine assumes that no more than a bin is marked
316  dirty at the time.
317 
318 Arguments:
319 
320  Hive - supplies a pointer to the hive control structure for the
321  hive of interest
322 
323  Start - supplies a hive virtual address (i.e., an HCELL_INDEX or
324  like form address) of the start of the area to mark dirty.
325 
326  Length - inclusive length in bytes of area to mark dirty.
327 
328  DirtyAndPin - indicates whether we should also pin the bin marked as dirty in memory
329 
330 
331 Return Value:
332 
333  TRUE - it worked
334 
335  FALSE - could not allocate log space, failure!
336 
337 --*/
338 {
339  ULONG Type;
340  PRTL_BITMAP BitMap;
341  ULONG First;
342  ULONG Last;
343  ULONG EndOfFile;
344  ULONG i;
345  ULONG Cluster;
346  ULONG OriginalDirtyCount;
347  ULONG DirtySectors;
348  BOOLEAN Result = TRUE;
349  PHMAP_ENTRY Map;
350  ULONG AdjustedFirst;
351  ULONG AdjustedLast;
352 
353  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"HvMarkDirty:\n\t"));
354  CmKdPrintEx((DPFLTR_CONFIG_ID,CML_IO,"Hive:%p Start:%08lx Length:%08lx\n", Hive, Start, Length));
355 
356 
357  ASSERT(Hive->Signature == HHIVE_SIGNATURE);
358  ASSERT(Hive->ReadOnly == FALSE);
359  ASSERT(Hive->DirtyCount == RtlNumberOfSetBits(&Hive->DirtyVector));
360 
361  Type = HvGetCellType(Start);
362 
363  if ( (Hive->HiveFlags & HIVE_VOLATILE) ||
364  (Type == Volatile) )
365  {
366  return TRUE;
367  }
368 
369  ASSERT_HIVE_WRITER_LOCK_OWNED((PCMHIVE)Hive);
370  ASSERT_HIVE_FLUSHER_LOCKED((PCMHIVE)Hive);
371 
372  BitMap = &(Hive->DirtyVector);
373  OriginalDirtyCount = Hive->DirtyCount;
374 
375  if( (DirtyAndPin == TRUE) && (((PCMHIVE)Hive)->FileObject != NULL) ) {
376  Map = HvpGetCellMap(Hive, Start);
377  VALIDATE_CELL_MAP(__LINE__,Map,Hive,Start);
378 
379  CmLockHiveViews(Hive);
380  if( (Map->BinAddress & (HMAP_INPAGEDPOOL|HMAP_INVIEW)) == 0){
381  PCM_VIEW_OF_FILE CmView;
382  //
383  // bin is neither in paged pool, nor in a mapped view
384  //
385  if( !NT_SUCCESS (CmpMapCmView((PCMHIVE)Hive,Start,&CmView,TRUE) ) ) {
386  CmUnlockHiveViews(Hive);
387  return FALSE;
388  }
389 
390 #if DBG
391  if(CmView != Map->CmView) {
392  CmKdPrintEx((DPFLTR_CONFIG_ID,DPFLTR_TRACE_LEVEL,"CmView = %p Map->CmView = %p\n",CmView,Map->CmView));
393  }
394 #endif
395  ASSERT( CmView == Map->CmView );
396 
397  }
398 
399  if( Map->BinAddress & HMAP_INVIEW ) {
400  //
401  // bin is mapped. Pin the view into memory
402  //
403  ASSERT( Map->CmView != NULL );
404 
405  if( IsListEmpty(&(Map->CmView->PinViewList)) == TRUE ) {
406  //
407  // the view is not already pinned. pin it
408  //
409  ASSERT_VIEW_MAPPED( Map->CmView );
410  if( !NT_SUCCESS(CmpPinCmView ((PCMHIVE)Hive,Map->CmView)) ) {
411  //
412  // couldn't pin view- some obscure error down in CcPinMappedData;
413  // this will be treated as STATUS_NO_LOG_SPACE
414  //
415  CmUnlockHiveViews(Hive);
416  return FALSE;
417  }
418  } else {
419  //
420  // view is already pinned; do nothing
421  //
422  ASSERT_VIEW_PINNED( Map->CmView );
423  }
424  }
425  CmUnlockHiveViews(Hive);
426  }
427 
428  AdjustedFirst = First = Start / HSECTOR_SIZE;
429  AdjustedLast = Last = (Start + Length - 1) / HSECTOR_SIZE;
430 
431  Cluster = Hive->Cluster;
432  if (Cluster > 1) {
433 
434  //
435  // Force Start down to base of cluster
436  // Force End up to top of cluster
437  //
438  AdjustedFirst = AdjustedFirst & ~(Cluster - 1);
439  AdjustedLast = ROUND_UP(AdjustedLast+1, Cluster) - 1;
440  }
441 
442  //
443  // we need to mark all page(s) dirty, so we don't conflict with cache manager
444  //
445  ASSERT( PAGE_SIZE >= HSECTOR_SIZE );
446  ASSERT( (PAGE_SIZE % HSECTOR_SIZE) == 0 );
447 
448  //
449  // adjust the range to fit an entire page
450  // make sure we account for the first HBLOCK at the beginning of the hive
451  //
452  AdjustedFirst = (AdjustedFirst + HSECTOR_COUNT) & ~(HSECTOR_PER_PAGE_COUNT - 1);
453  AdjustedLast = ROUND_UP(AdjustedLast + HSECTOR_COUNT + 1, HSECTOR_PER_PAGE_COUNT) - 1;
454 
455  AdjustedLast -= HSECTOR_COUNT;
456  if( AdjustedFirst ) {
457  AdjustedFirst -= HSECTOR_COUNT;
458  }
459  //
460  // when the PAGE_SIZE > HBLOCK_SIZE and the length of the hive does not round at PAGE_SIZE boundary
461  //
462  EndOfFile = Hive->Storage[Stable].Length / HSECTOR_SIZE;
463  if (AdjustedLast >= EndOfFile) {
464  AdjustedLast = EndOfFile-1;
465  }
466 
467  //
468  // make sure that between first and last all bins are valid (either pinned
469  // or allocated from paged pool).
470  //
471  ASSERT( First >= AdjustedFirst );
472  ASSERT( Last <= AdjustedLast );
473 
474  //
475  // adjust First and Last at HBLOCK_SIZE boundaries
476  //
477  First = First & ~(HSECTOR_COUNT - 1);
478  Last = ROUND_UP(Last+1, HSECTOR_COUNT) - 1;
479 
480  //
481  // sanity asserts; these prove we can skip HSECTOR_COUNT at one time below
482  //
483  ASSERT( First >= AdjustedFirst );
484  ASSERT( Last <= AdjustedLast );
485  ASSERT( (First % HSECTOR_COUNT) == 0 );
486  ASSERT( (AdjustedFirst % HSECTOR_COUNT) == 0 );
487  ASSERT( ((Last+1) % HSECTOR_COUNT) == 0 );
488  ASSERT( ((AdjustedLast +1) % HSECTOR_COUNT) == 0 );
489  ASSERT( ((First - AdjustedFirst) % HSECTOR_COUNT) == 0 );
490  ASSERT( ((AdjustedLast - Last) % HSECTOR_COUNT) == 0 );
491 
492  //
493  // when we exit this loop; First is always a valid bin/sector
494  //
495  while( First > AdjustedFirst ) {
496  //
497  // map-in this address, and if is valid, decrement First, else break out of the loop
498  //
499  First -= HSECTOR_COUNT;
500  Map = HvpGetCellMap(Hive, First*HSECTOR_SIZE);
501  if( BIN_MAP_ALLOCATION_TYPE(Map) == 0 ) {
502  //
503  // bin is not valid! bail out.
504  //
505  First += HSECTOR_COUNT;
506  break;
507  }
508  CmLockHiveViews(Hive);
509  if( Map->BinAddress & HMAP_INVIEW ) {
510  //
511  // previous bin mapped in view ==> view needs to be pinned
512  //
513  ASSERT( Map->CmView );
514  if( IsListEmpty(&(Map->CmView->PinViewList) ) == TRUE ) {
515  //
516  // bin not pinned. bail out.
517  //
518  First += HSECTOR_COUNT;
519  CmUnlockHiveViews(Hive);
520  break;
521  }
522  }
523  CmUnlockHiveViews(Hive);
524  }
525 
526  //
527  // when we exit this loop; Last is always a valid bin/sector
528  //
529  while( Last < AdjustedLast ) {
530  //
531  // map-in this address, and if is valid, increment Last, else break out of the loop
532  //
533  Last += HSECTOR_COUNT;
534  Map = HvpGetCellMap(Hive, Last*HSECTOR_SIZE);
535  if( BIN_MAP_ALLOCATION_TYPE(Map) == 0 ) {
536  //
537  // bin is not valid. bail out.
538  //
539  Last -= HSECTOR_COUNT;
540  break;
541  }
542  CmLockHiveViews(Hive);
543  if( Map->BinAddress & HMAP_INVIEW ) {
544  //
545  // previous bin mapped in view ==> view needs to be pinned
546  //
547  ASSERT( Map->CmView );
548  if( IsListEmpty(&(Map->CmView->PinViewList) ) == TRUE ) {
549  //
550  // bin not pinned. bail out.
551  //
552  Last -= HSECTOR_COUNT;
553  CmUnlockHiveViews(Hive);
554  break;
555  }
556  }
557  CmUnlockHiveViews(Hive);
558  }
559 
560  //
561  // Try and grow the log enough to accommodate all the dirty sectors.
562  //
563  DirtySectors = 0;
564  for (i = First; i <= Last; i++) {
565  if (RtlCheckBit(BitMap, i)==0) {
566  ++DirtySectors;
567  }
568  }
569  if (DirtySectors != 0) {
570  if (HvpGrowLog1(Hive, DirtySectors) == FALSE) {
571  return(FALSE);
572  }
573 
574  if ((OriginalDirtyCount == 0) && (First != 0)) {
575  Result = HvMarkDirty(Hive, 0, sizeof(HBIN),TRUE); // force header of 1st bin dirty
576  if (Result==FALSE) {
577  return(FALSE);
578  }
579  }
580 
581  //
582  // Log has been successfully grown, go ahead
583  // and set the dirty bits.
584  //
585  ASSERT(Hive->DirtyCount == RtlNumberOfSetBits(&Hive->DirtyVector));
586  ASSERT( First <= Last );
587  if( First <= Last ) {
588  Hive->DirtyCount += DirtySectors;
589  RtlSetBits(BitMap, First, Last-First+1);
590  }
591  }
592 
593  // mark this bin as writable
594  HvpMarkBinReadWrite(Hive,Start);
595 
596  if (!(Hive->HiveFlags & HIVE_NOLAZYFLUSH)) {
597  CmpLazyFlush();
598  }
599 
600  ASSERT(Hive->DirtyCount == RtlNumberOfSetBits(&Hive->DirtyVector));
601  return(TRUE);
602 }

685 HvRefreshHive(
686  PHHIVE Hive
687  )
688 /*++
689 
690 Routine Description:
691 
692  Undo the last sync.
693 
694  The story behind the scene:
695 
696  1. remove all discardable bins from FreeBins list. they'll be
697  enlisted afterwards with the right (accurate) values.
698  2. read the base block, and eventually free the tail of the hive
699  3. unpin and purge all pinned views; also clear the free cell
700  hint for mapped bins.
701  4. remap views purged at 3 and reenlist the bins inside. this
702  will fix free bins discarded at 1.
703  5. iterate through the map; read and reenlist all bins that are
704  in paged-pool (and dirty)
705 
706 
707  All I/O is done via HFILE_TYPE_PRIMARY.
708 
709 Arguments:
710 
711  Hive - supplies a pointer to the hive control structure for the
712  hive of interest.
713 
714 Return Value:
715 
716  NONE. Either works or BugChecks.
717 
718 Comments:
719 
720  In the new implementation, bins are not discarded anymore. Step 1.
721  above is not needed anymore.
722 
723  Discardable bins with FREE_HBIN_DISCARDABLE flag set fall into one
724  of the categories:
725  1. new bins (at the end of the hive) which didn't get a chance to
726  be saved yet. HvFreeHivePartial will take care of them.
727  2. bins inside the hive allocated from paged pool and discarded.
728  This can only happen for bins that are crossing the CM_VIEW_SIZE boundary.
729  We will take care of them at step 5
730 
731  Discardable bins with FREE_HBIN_DISCARDABLE flag NOT set are free bins
732  which came from mapped views. Step 3 will remove them from the FreeBins
733  list and step 4 will reenlist the ones that are still free after remapping
734 
735 --*/
736 {
737  HCELL_INDEX RootCell;
