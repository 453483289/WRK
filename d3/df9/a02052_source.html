173 VOID
174 CmpDelayCloseWorker(
175  IN PVOID Parameter
176  )
177 
178 /*++
179 
180 Routine Description:
181 
182  The fun part. We need to walk the cache and look for kcbs with refcount == 0
183  Get the oldest one and kick it out of cache.
184 
185 Arguments:
186 
187  Parameter - not used.
188 
189 Return Value:
190 
191  None.
192 
193 --*/
194 
195 {
196  PCM_DELAYED_CLOSE_ENTRY DelayedEntry;
197  ULONG ConvKey;
198  ULONG MaxIterations = MAX_DELAY_WORKER_ITERATIONS;
199 
200  CM_PAGED_CODE();
201 
202  UNREFERENCED_PARAMETER (Parameter);
203 
204  ASSERT(CmpDelayCloseWorkItemActive);
205 
206  BEGIN_LOCK_CHECKPOINT;
207  CmpLockRegistry();
208 
209  //
210  // process kick out every entry with RefCount == 0 && DelayCloseIndex == 0
211  // ignore the others; we only do this while there is excess of delay - close kcbs
212  //
213  LOCK_DELAY_CLOSE();
214  while( (CmpDelayedCloseElements > CmpDelayedCloseSize) && (MaxIterations--) ) {
215  ASSERT( !CmpIsListEmpty(&CmpDelayedLRUListHead) );
216  //
217  // We first need to get the hash entry and attempt to lock it.
218  //
219  DelayedEntry = (PCM_DELAYED_CLOSE_ENTRY)(CmpDelayedLRUListHead.Blink);
220  DelayedEntry = CONTAINING_RECORD( DelayedEntry,
221  CM_DELAYED_CLOSE_ENTRY,
222  DelayedLRUList);
223  ConvKey = DelayedEntry->KeyControlBlock->ConvKey;
224  UNLOCK_DELAY_CLOSE();
225  //
226  // now lock the hash then recheck if the entry is still first on the list
227  //
228  CmpLockHashEntryExclusive(ConvKey);
229  LOCK_DELAY_CLOSE();
230  if( CmpDelayedCloseElements <= CmpDelayedCloseSize ) {
231  //
232  // just bail out; no need to kick them out
233  //
234  CmpUnlockHashEntry(ConvKey);
235  break;
236  }
237  DelayedEntry = (PCM_DELAYED_CLOSE_ENTRY)(CmpDelayedLRUListHead.Blink);
238  DelayedEntry = CONTAINING_RECORD( DelayedEntry,
239  CM_DELAYED_CLOSE_ENTRY,
240  DelayedLRUList);
241  if( ConvKey == DelayedEntry->KeyControlBlock->ConvKey ) {
242  //
243  // same hash entry; proceed
244  // pull it out of the list
245  //
246  DelayedEntry = (PCM_DELAYED_CLOSE_ENTRY)RemoveTailList(&CmpDelayedLRUListHead);
247 
248  DelayedEntry = CONTAINING_RECORD( DelayedEntry,
249  CM_DELAYED_CLOSE_ENTRY,
250  DelayedLRUList);
251  CmpClearListEntry(&(DelayedEntry->DelayedLRUList));
252 
253  if( (DelayedEntry->KeyControlBlock->RefCount == 0) && (DelayedEntry->KeyControlBlock->DelayedCloseIndex == 0) ){
254  //
255  // free this kcb and the entry
256  //
257  UNLOCK_DELAY_CLOSE();
258  DelayedEntry->KeyControlBlock->DelayCloseEntry = NULL;
259  CmpCleanUpKcbCacheWithLock(DelayedEntry->KeyControlBlock,FALSE);
260  CmpDelayCloseFreeEntry(DelayedEntry);
261  InterlockedDecrement((PLONG)&CmpDelayedCloseElements);
262  } else {
263  //
264  // put it back at the top
265  //
266  InsertHeadList( &CmpDelayedLRUListHead,
267  &(DelayedEntry->DelayedLRUList)
268  );
269  UNLOCK_DELAY_CLOSE();
270  }
271  } else {
272  UNLOCK_DELAY_CLOSE();
273  }
274  CmpUnlockHashEntry(ConvKey);
275 
276  LOCK_DELAY_CLOSE();
277  }
278  if( CmpDelayedCloseElements > CmpDelayedCloseSize ) {
279  //
280  // iteration run was too short, there are more elements to process, queue ourselves for later
281  //
282  CmpArmDelayedCloseTimer();
283  } else {
284  //
285  // signal that we have finished our work.
286  //
287  CmpDelayCloseWorkItemActive = FALSE;
288  }
289  UNLOCK_DELAY_CLOSE();
290 
291 
292  CmpUnlockRegistry();
293  END_LOCK_CHECKPOINT;
294 }

347 /*++
348 
349 Routine Description:
350 
351  Removes a KCB from the delayed close table;
352 
353 Arguments:
354 
355  kcb - the kcb in question
356 
357 Note:
358 
359  kcb lock/resource should be acquired exclusively when this function is called
360 
361 Return Value:
362 
363  NONE.
364 
365 --*/
366 {
367  PCM_DELAYED_CLOSE_ENTRY DelayedEntry;
368 
369  CM_PAGED_CODE();
370 
371 
372  ASSERT( (CmpIsKCBLockedExclusive(kcb) == TRUE) || // this kcb is owned exclusive
373  (CmpTestRegistryLockExclusive() == TRUE) ); // or the entire registry is locked exclusive
374 
375  // not on delay close; don't try to remove it.
376  if (kcb->DelayedCloseIndex == CmpDelayedCloseSize) { // see if we really need this
377  ASSERT( FALSE );
378  return;
379  }
380 
381  DelayedEntry = (PCM_DELAYED_CLOSE_ENTRY)kcb->DelayCloseEntry;
382  if( DelayedEntry != NULL ) {
383  LOCK_DELAY_CLOSE();
384  CmpRemoveEntryList(&(DelayedEntry->DelayedLRUList));
385  UNLOCK_DELAY_CLOSE();
386 
387  //
388  // give back the entry
389  //
390  CmpDelayCloseFreeEntry(DelayedEntry);
391  InterlockedDecrement((PLONG)&CmpDelayedCloseElements);
392 #if DBG
393  if( kcb->InDelayClose == 0 ) {
394  ASSERT( FALSE );
395  }
396  {
397  LONG OldRefCount;
398  LONG NewRefCount;
399 
400  OldRefCount = *(PLONG)&kcb->InDelayClose; //get entire dword
401  ASSERT( OldRefCount == 1 );
402  NewRefCount = 0;
403  if( InterlockedCompareExchange((PLONG)&kcb->InDelayClose,NewRefCount,OldRefCount)
404  != OldRefCount ) {
405 
406  ASSERT( FALSE );
407  }
408  }
409 #endif //DBG
410  }
411 
412  //
413  // easy enough huh ?
414  //
415  kcb->DelayedCloseIndex = CmpDelayedCloseSize; // see if we really need this
416  kcb->DelayCloseEntry = NULL;
417 }

422  IN PCM_KEY_CONTROL_BLOCK kcb,
423  IN BOOLEAN RegLockHeldEx
424  )
425 /*++
426 
427 Routine Description:
428 
429  Adds a kcb to the delayed close table
430 
431 Arguments:
432 
433  kcb - the kcb in question
434 
435 Note:
436 
437  kcb lock/resource should be acquired exclusively when this function is called
438 
439 Return Value:
440 
441  NONE.
442 
443 --*/
444 {
445  PCM_DELAYED_CLOSE_ENTRY DelayedEntry = NULL;
446 
447  CM_PAGED_CODE();
448 
449  ASSERT( (CmpIsKCBLockedExclusive(kcb) == TRUE) || // this kcb is owned exclusive
450  (CmpTestRegistryLockExclusive() == TRUE) ); // or the entire registry is locked exclusive
451 
452 
453  // Already on delayed close, don't try to put on again
454  if (kcb->DelayedCloseIndex != CmpDelayedCloseSize) { // see if we really need this
455  ASSERT( FALSE );
456  return;
457  }
458 
459  ASSERT(kcb->RefCount == 0);
460 
461  //
462  // now materialize a new entry for this kcb
463  //
464  ASSERT_KEYBODY_LIST_EMPTY(kcb);
465  DelayedEntry = CmpDelayCloseAllocNewEntry();
466  if( !DelayedEntry ) {
467  //
468  // this is bad luck; we need to free the kcb in place
469  //
470  CmpCleanUpKcbCacheWithLock(kcb,RegLockHeldEx);
471  return;
472  }
473 #if DBG
474  if( kcb->InDelayClose != 0 ) {
475  ASSERT( FALSE );
476  }
477  {
478  LONG OldRefCount;
479  LONG NewRefCount;
480 
481  OldRefCount = *(PLONG)&kcb->InDelayClose; //get entire dword
482  ASSERT( OldRefCount == 0 );
483  NewRefCount = 1;
484  if( InterlockedCompareExchange((PLONG)&kcb->InDelayClose,NewRefCount,OldRefCount)
485  != OldRefCount ) {
486 
487  ASSERT( FALSE );
488  }
489  }
490 #endif //DBG
491  //
492  // populate the entry and insert it into the LRU list (at the top).
493  //
494  kcb->DelayedCloseIndex = 0; // see if we really need this
495  kcb->DelayCloseEntry = (PVOID)DelayedEntry; // need this for removing it back from here
496  DelayedEntry->KeyControlBlock = kcb;
497  InterlockedIncrement((PLONG)&CmpDelayedCloseElements);
498 
499  LOCK_DELAY_CLOSE();
500  InsertHeadList(
501  &CmpDelayedLRUListHead,
502  &(DelayedEntry->DelayedLRUList)
503  );
504  //
505  // check if limit hit and arm timer if not already armed
506  //
507  if( (CmpDelayedCloseElements > CmpDelayedCloseSize) && (!CmpDelayCloseWorkItemActive) ) {
508  CmpArmDelayedCloseTimer();
509  }
510  UNLOCK_DELAY_CLOSE();
511  //
512  // we're done here
513  //
514 
515 }

598 VOID
599 CmpRunDownDelayDerefKCBEngine( PCM_KEY_CONTROL_BLOCK KeyControlBlock,
600  BOOLEAN RegLockHeldEx)
601 {
602  PCM_DELAY_DEREF_KCB_ITEM DelayItem;
603  ULONG NewIndex;
604  ULONG Index1 = CmpHashTableSize;
605  ULONG Index2 = CmpHashTableSize;
606  PCMHIVE CmHive = NULL;
607  BOOLEAN DoUnloadCheck = FALSE;
608 
609  CM_PAGED_CODE();
610 
611  ASSERT_CM_LOCK_OWNED();
612 
613  ASSERT( (KeyControlBlock && CmpIsKCBLockedExclusive(KeyControlBlock)) || (CmpTestRegistryLockExclusive() == TRUE) );
614 
615  if( KeyControlBlock ) {
616  Index1 = GET_HASH_INDEX(KeyControlBlock->ConvKey);
617  if( KeyControlBlock->ParentKcb ) {
618  Index2 = GET_HASH_INDEX(KeyControlBlock->ParentKcb->ConvKey);
619  }
620  }
621  LOCK_KCB_DELAY_DEREF_LIST();
622  while( !CmpIsListEmpty(&CmpDelayDerefKCBListHead) ) {
623  //
624  // pull it out of the list
625  //
626  DelayItem = (PCM_DELAY_DEREF_KCB_ITEM)RemoveHeadList(&CmpDelayDerefKCBListHead);
627  UNLOCK_KCB_DELAY_DEREF_LIST();
628 
629  DelayItem = CONTAINING_RECORD( DelayItem,
630  CM_DELAY_DEREF_KCB_ITEM,
631  ListEntry);
632  CmpClearListEntry(&(DelayItem->ListEntry));
633 
634  //
635  // take additional precaution in the case the hive has been unloaded and this is the root
636  //
637  DoUnloadCheck = FALSE;
638  if( !DelayItem->Kcb->Delete ) {
639  CmHive = (PCMHIVE)CONTAINING_RECORD(DelayItem->Kcb->KeyHive, CMHIVE, Hive);
640  if( IsHiveFrozen(CmHive) ) {
641  //
642  // unload is pending for this hive;
643  //
644  DoUnloadCheck = TRUE;
645 
646  }
647  }
648 
649  NewIndex = GET_HASH_INDEX(DelayItem->Kcb->ConvKey);
650  //
651  // now deref and free
652  //
653  if( (NewIndex == Index1) || (NewIndex == Index2) ) {
654  //
655  // we already hold the lock
656  //
657  ASSERT_KCB_LOCKED_EXCLUSIVE(DelayItem->Kcb);
658  CmpDereferenceKeyControlBlockWithLock(DelayItem->Kcb,RegLockHeldEx);
659  } else {
660  //
661  // we can't afford to force acquire; try to acquire and if it fails bail out
662  //
663  if( CmpKCBLockForceAcquireAllowed(Index1,Index2,NewIndex) ) {
664  CmpLockHashEntryByIndexExclusive(NewIndex);
665  } else if( CmpTryToLockHashEntryByIndexExclusive(NewIndex) == FALSE ) {
666  //
667  // add the item back to the list and bail out
668  //
669  ASSERT( CmpTestRegistryLockExclusive() == FALSE );
670  LOCK_KCB_DELAY_DEREF_LIST();
671  InsertHeadList(&CmpDelayDerefKCBListHead,&(DelayItem->ListEntry));
672  UNLOCK_KCB_DELAY_DEREF_LIST();
673  return;
674  }
675  ASSERT_KCB_LOCKED_EXCLUSIVE(DelayItem->Kcb);
676  CmpDereferenceKeyControlBlockWithLock(DelayItem->Kcb,RegLockHeldEx);
677  CmpUnlockHashEntryByIndex(NewIndex);
678  }
679  CmpDelayDerefKCBFreeEntry(DelayItem);
680 
681  //
682  // if we dropped a reference inside a frozen hive, we may need to unload the hive
683  //
684  if( DoUnloadCheck == TRUE ) {
685  CmpDoQueueLateUnloadWorker(CmHive);
686  }
687 
688  LOCK_KCB_DELAY_DEREF_LIST();
689  }
690  UNLOCK_KCB_DELAY_DEREF_LIST();
691 }
